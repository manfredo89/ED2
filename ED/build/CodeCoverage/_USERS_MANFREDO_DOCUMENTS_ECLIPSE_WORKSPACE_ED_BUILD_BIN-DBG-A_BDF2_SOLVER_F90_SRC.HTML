     1) !==============================================================================
     2) ! Driver and core for numerically integrating the canopy via BDF2
     3) ! BDF2 is a trapezoidal split step method (backward and forward)
     4) !
     5) ! To solve for the next step, we must solve a set of linear equations
     6) ! using data on the current and previous step.
     7) !
     8) ! U_{n+1} = [4U_{n}+U_{n-1}][3I-2dtA]^{-1}
     9) ! 
    10) ! The crux of this method is solving the for the instantaneous derivatives
    11) ! dU/dt = A*U_{n+1} + B          (where B is a forcing like SW radiation
    12) !                                 energy flux from the ABL, or energy flux
    13) !                                 from the ground.  The mass and energy
    14) !                                 flux from the ground is solved elsewhere
    15) !
    16) ! One difficulty in creating the A matrix, is that some of these processes
    17) ! are nearly binary...is there ponded water or not at the next step.
    18) ! Therefore, it may be necessary to to a sanity check on the existence of
    19) ! surface water and test to see if it was correct at the end of the step.
    20) !
    21) !==============================================================================
    22) 
    23) subroutine bdf2_solver(cpatch,yprev,ycurr,ynext,dydt,dtf,dtb)
    24) 
    25)   use grid_coms,only         : nzg
    26)   use rk4_coms,only          : effarea_heat,rk4site,rk4patchtype,rk4aux &
    27)                              , bdf2patchtype,checkbudget,ibranch_thermo
    28)   use ed_misc_coms,only    : fast_diagnostics
    29)   use ed_state_vars,only   : patchtype
    30)   use therm_lib8,only      : reducedpress8,idealdenssh8
    31)   use ed_therm_lib,only    : ed_grndvap8
    32)   use consts_coms,only     : cpdry8,p00i8,rdry8, &
    33)        rocv8,cpocv8,cliq8,cice8,rocp8,cpocv,epim1,   &
    34)        alli8,t3ple8,alvi38,wdns8,cph2o8,tsupercool_liq8,pi18
    35)   use therm_lib8,only      : uint2tl8,uextcm2tl8,tq2enthalpy8, tl2uint8, cmtl2uext8
    36)   use soil_coms, only      : soil8
    37)   !$ use omp_lib
    38)   implicit none
    39)   
    40)   ! define the previous,current and next patch states
    41)   integer                                    :: nstate
    42)   real(kind=8),pointer,dimension(:)          :: Y
    43)   real(kind=8),pointer,dimension(:)          :: Yf
    44)   real(kind=8),pointer,dimension(:,:)        :: A
    45)   real(kind=8),pointer,dimension(:)          :: B
    46)   type(patchtype),target                     :: cpatch
    47)   type(bdf2patchtype), target                :: yprev
    48)   type(rk4patchtype), target                 :: ycurr
    49)   type(rk4patchtype), target                 :: ynext
    50)   type(rk4patchtype), target                 :: dydt
    51)   real(kind=8),intent(in)                    :: dtf        ! dt for n -> n+1
    52)   real(kind=8),intent(in)                    :: dtb        ! dt for n -> n-1
    53) 
    54)   integer :: id_tcan,id_tveg,ico
    55)   integer :: k,ksn
    56)   integer :: i,j
    57)   real(kind=8)                  :: shctop
    58)   real(kind=8)                  :: rhoc  ! canopy air density [kg/m3]
    59)   real(kind=8)                  :: dc    ! canopy depth       [m]
    60)   real(kind=8)                  :: qc    ! canopy spec. hum   [kg/kg]
    61)   real(kind=8)                  :: Tg    ! soil temperature   [K]
    62)   real(kind=8)                  :: gg    ! ground-can conductivity [m/s]
    63)   real(kind=8)                  :: mgc   ! mass flux ground->can   [kg/s/m2]
    64)   real(kind=8)                  :: ga    ! atm-canopy conductivity [m/s]
    65)   real(kind=8)                  :: Ta    ! atm temperature         [K]
    66)   real(kind=8)                  :: mac   ! mass flux atm->can      [kg/s/m2]
    67)   real(kind=8)                  :: dqcdt ! time partial of qc      [kg/kg/s]
    68)   real(kind=8)                  :: xc    ! lumped term (canopy)    [-]
    69)   real(kind=8)                  :: href  ! reference spec. enth.   [J/kg]
    70)   real(kind=8),dimension(300)   :: gv_hfa   ! effective area of heat times g
    71)   real(kind=8),dimension(300)   :: mlc   ! mass flux leaf->can      [kg/m2/s]
    72)   real(kind=8),dimension(300)   :: mtr   ! mass flux transp veg->can [kg/m2/s]
    73)   real(kind=8),dimension(300)   :: fliq  ! liquid fraction leaf surf [kg/kg]
    74)   real(kind=8),dimension(300)   :: qv    ! water mass on vegetation  [kg/m2]
    75)   real(kind=8),dimension(300)   :: dqvdt ! time partial of qv  [kg/m2/s]
    76)   real(kind=8),dimension(300)   :: xv    ! lumped term (vegetation)
    77)   real(kind=8),dimension(300)   :: hflx  ! enth flux from bunch of stuff
    78)   logical,dimension(300)        :: resolve ! self-explanatory
    79) 
    80)   real(kind=8)  :: veg_temp
    81)   real(kind=8)  :: eflxac
    82)   real(kind=8)  :: qwflxac
    83)   real(kind=8)  :: hflxac
    84)   real(kind=8)  :: qwflxlc,qwflxlc_tot
    85)   real(kind=8)  :: qwflxwc,qwflxwc_tot
    86)   real(kind=8)  :: qtransp,qtransp_tot
    87)   real(kind=8)  :: hflxlc,hflxlc_tot
    88)   real(kind=8)  :: hflxwc,hflxwc_tot
    89)   integer       :: ibuff
    90) 
    91)   ibuff = 1
    92)   !$ ibuff = OMP_get_thread_num()+1
    93) 
    94)   !==========================================================================!
    95)   !                                                                          !
    96)   ! Part 1: Build the forecasted transition matrix A                         !
    97)   !                                                                          !
    98)   ! Assume that Y takes the following form                                   !
    99)   ! Y = [ T_can T_veg1 T_veg2 ... T_vegN ]                                   !
   100)   !                                                                          !
   101)   !==========================================================================!
   102) 
   103)   !==========================================================================!
   104)   ! Row and Column indices for the transition matrix operators               !
   105)   !==========================================================================!
   106)   id_tcan   = 1
   107) 
   108)   !==========================================================================!
   109)   ! Heat flux to the atmosphere... few ways to go about this                 !
   110)   ! 1) use the heat flux at step time n                                      !
   111)   ! 2) use atmospheric temp at step n, but incorporate the canopy temp at n+1!
   112)   !    to calculate flux, uses conductivity at n                             !
   113)   ! 3) use forward time step of atmospheric temp as well (a little overboard)!
   114)   ! 4) also would use the forward euler canopy and leaf temps to recalc      !
   115)   !    aerodynamic conductances                                              !
   116)   ! **** Using method 2 right now                                            !
   117)   !==========================================================================!
   118) 
   119)   !--------------------------------------------------------------------------!
   120)   ! Ground Temp is estimated via an explicit method, it represents           !
   121)   ! the effective ground surface temperature, be it standing water           !
   122)   ! or moist/dry soil                                                        !
   123)   !--------------------------------------------------------------------------!
   124)   
   125) !!  ksn=0
   126) !!  do k=1,ycurr%nlev_sfcwater
   127) !!     if(ycurr%sfcwater_mass(k)>1.d-5)ksn=k
   128) !!  end do
   129)   
   130)   shctop = soil8(rk4site%ntext_soil(nzg))%slcpd
   131)   call uextcm2tl8(ynext%soil_energy(nzg),ynext%soil_water(nzg)*wdns8,shctop &
   132)        ,ynext%soil_tempk(nzg),ynext%soil_fracliq(nzg))
   133) 
   134)   ksn=ycurr%nlev_sfcwater
   135)   if (ksn>0) then
   136)      call uint2tl8(ycurr%sfcwater_energy(ksn)/ycurr%sfcwater_mass(ksn),              &
   137)           ycurr%sfcwater_tempk(ksn),ycurr%sfcwater_fracliq(ksn))
   138)   endif
   139) 
   140) 
   141)   k=max(1,ksn)
   142) 
   143)   call ed_grndvap8(ksn,ynext%soil_water(nzg),ynext%soil_tempk(nzg)        &
   144)        ,ynext%soil_fracliq(nzg),ycurr%sfcwater_tempk(k)                   &
   145)        ,ycurr%snowfac,ycurr%can_prss,ynext%can_shv                        &
   146)        ,ynext%ground_shv,ynext%ground_ssh,ynext%ground_temp               &
   147)        ,ynext%ground_fliq,ynext%ggsoil)
   148)   
   149) 
   150)   !===========================================================================!
   151)   ! Determine the leaf-branch thermodynamics scheme to be used.
   152)   ! Set some temporary variables.
   153)   ! Calculate the number of elements.
   154)   !===========================================================================!
   155) 
   156) 
   157)   resolve(:)=.false.
   158)   id_tveg = 1
   159)   select case(ibranch_thermo)
   160)   case(0,2)
   161)      do ico=1,cpatch%ncohorts
   162)         if (ycurr%leaf_resolvable(ico)) then
   163)            gv_hfa(ico) = ycurr%leaf_gbh(ico)/(ycurr%can_rhos*cpdry8) &
   164)                 *effarea_heat*ycurr%lai(ico)
   165)            mlc(ico)  = ycurr%wflxlc(ico)
   166)            mtr(ico)  = ycurr%wflxtr(ico)
   167)            fliq(ico) = ycurr%leaf_fliq(ico)
   168)            qv(ico)   = ynext%leaf_water(ico)
   169)            dqvdt(ico)= dydt%leaf_water(ico)
   170)            xv(ico)   = ycurr%leaf_fliq(ico)*cliq8*qv(ico) + &
   171)                        (1.d0-fliq(ico))*cice8*qv(ico)     + &
   172)                        ycurr%leaf_hcap(ico)
   173)            hflx(ico) = ycurr%hflx_lrsti(ico)
   174)            resolve(ico) = .true.
   175)            id_tveg   = id_tveg + 1
   176)         end if
   177)      end do
   178)      
   179)      if(ibranch_thermo.eq.2)then
   180)         do ico=1,cpatch%ncohorts
   181)            if (ycurr%wood_resolvable(ico)) then
   182)               id_tveg   = id_tveg + 1
   183)            end if
   184)         end do
   185)      end if
   186) 
   187)   case(1)
   188)      do ico=1,cpatch%ncohorts
   189)         gv_hfa(ico) = 0.0
   190)         mlc(ico)    = 0.0
   191)         mtr(ico)    = 0.0
   192)         fliq(ico)   = -1.0
   193)         qv(ico)     = 0.0
   194)         dqvdt(ico)  = 0.0
   195)         hflx(ico)   = 0.0
   196)         if (ycurr%leaf_resolvable(ico) .or. ycurr%wood_resolvable(ico))then
   197)            id_tveg = id_tveg+1
   198)            resolve(ico) = .true.
   199)         end if
   200) 
   201)         if (ycurr%leaf_resolvable(ico)) then
   202)            gv_hfa(ico) = ycurr%leaf_gbh(ico)/(ycurr%can_rhos*cpdry8) &
   203)                 *effarea_heat*ycurr%lai(ico)
   204)            mlc(ico)  = ycurr%wflxlc(ico)
   205)            mtr(ico)  = ycurr%wflxtr(ico)
   206)            fliq(ico) = ycurr%leaf_fliq(ico)
   207)            qv(ico)   = ynext%leaf_water(ico)
   208)            dqvdt(ico)= dydt%leaf_water(ico)
   209)            hflx(ico) = ycurr%hflx_lrsti(ico)
   210)         end if
   211)         if (ycurr%wood_resolvable(ico)) then
   212)            gv_hfa(ico) = gv_hfa(ico)+(ycurr%wood_gbh(ico)/(ycurr%can_rhos*cpdry8) &
   213)                 *effarea_heat*ycurr%wai(ico))
   214)            mlc(ico)  = mlc(ico)+ycurr%wflxwc(ico)
   215)            if(fliq(ico)>-0.99)then
   216)               fliq(ico) = (fliq(ico)+ycurr%wood_fliq(ico))/2.0
   217)            else
   218)               fliq(ico) = ycurr%wood_fliq(ico)
   219)            end if
   220)            
   221)            qv(ico)   = qv(ico)+ynext%wood_water(ico)
   222)            dqvdt(ico)= dqvdt(ico)+dydt%wood_water(ico)
   223)            hflx(ico) = hflx(ico)+ycurr%hflx_wrsti(ico)
   224)         end if
   225) 
   226)         ! We use the wood and the leaf heat capacities regardless 
   227)         ! of their ability to be solved.
   228)         ! ---------------------------------------------------------
   229) 
   230)         xv(ico)   = ycurr%leaf_fliq(ico)*cliq8*ynext%leaf_water(ico)        + &
   231)                     (1.d0-ycurr%leaf_fliq(ico))*cice8*ynext%leaf_water(ico) + &
   232)                     ycurr%leaf_hcap(ico)                                    + &
   233)                     ycurr%wood_fliq(ico)*cliq8*ynext%wood_water(ico)        + &
   234)                     (1.d0-ycurr%wood_fliq(ico))*cice8*ynext%wood_water(ico) + &
   235)                     ycurr%wood_hcap(ico)
   236) 
   237)      end do
   238) 
   239)   end select
   240) 
   241)   nstate = id_tveg
   242) 
   243)   !------ Allocate the matrices used for the linear operations --------------!
   244)   
   245)   allocate(Y(nstate))
   246)   allocate(Yf(nstate))
   247)   allocate(A(nstate,nstate))
   248)   allocate(B(nstate))
   249) 
   250)   !----- Initialize the matrices used for the linear operations -------------!
   251) 
   252)   A = 0.d0
   253)   B = 0.d0
   254)   Y = 0.d0
   255)   Yf= 0.d0
   256) 
   257) 
   258)   !----- Set temporary canopy level variables used for readability ----------!
   259)   
   260)   rhoc  = ycurr%can_rhos
   261)   dc    = ycurr%can_depth
   262)   qc    = ynext%can_shv
   263)   Tg    = ycurr%ground_temp
   264)   gg    = ycurr%ggnet
   265)   mgc   = ycurr%wflxsc + ycurr%wflxgc
   266)   ga    = ycurr%ggbare
   267)   Ta    = rk4site%atm_theta*rk4site%atm_exner/cpdry8
   268)   mac   = ycurr%wflxac
   269)   dqcdt = dydt%can_shv
   270)   xc    = rhoc*dc*((1.d0-qc)*cpdry8+qc*cph2o8)
   271)   href  = t3ple8*cice8+alvi38-cph2o8*t3ple8
   272) 
   273)   ! USES NEW TG
   274) !!  B(id_tcan) = (1.d0/xc)*       &
   275) !!       (gg*rhoc*cpdry8*Tg       &
   276) !!       + mgc*(href+cph2o8*Tg)    &
   277) !!       + ga*rhoc*cpdry8*Ta      &
   278) !!       + mac*href               &
   279) !!       + mac*cph2o8*0.5d0*Ta     &
   280) !!       + sum(ycurr%wflxlc)*href &
   281) !!       + sum(ycurr%wflxtr)*href &
   282) !!       + sum(ycurr%wflxwc)*href &
   283) !!       - dc*rhoc*href*dqcdt)
   284) 
   285)   ! USES ycurr HFLXSC+HFLXGC
   286)   B(id_tcan) = (1.d0/xc)*       &
   287)        ( ycurr%hflxsc           & 
   288)        + ycurr%hflxgc           &
   289)        + ycurr%qwflxsc          &
   290)        + ycurr%qwflxgc          &
   291)   !    + mgc*(href+cph2o8*Tg)   &
   292)        + ga*rhoc*cpdry8*Ta      &
   293)        + mac*href               &
   294)        + mac*cph2o8*0.5d0*Ta     &
   295)        + sum(ycurr%wflxlc)*href &
   296)        + sum(ycurr%wflxtr)*href &
   297)        + sum(ycurr%wflxwc)*href &
   298)        - dc*rhoc*href*dqcdt)
   299) 
   300) 
   301) 
   302)   ! USES NEW TG
   303) !!  A(id_tcan,id_tcan) = (1.d0/xc)*           &
   304) !!       (-gg*rhoc*cpdry8                     &
   305) !!       -ga*rhoc*cpdry8                      &
   306) !!       +0.5d0*mac*cph2o8)                   &
   307) !!       -(dqcdt*dc*rhoc/(xc**2.d0))*           &
   308) !!       ((1.d0-qc)*cpdry8 + qc*cph2o8)*(cph2o8-cpdry8)
   309)   
   310)   ! USES ycurr HFLXGC
   311)   A(id_tcan,id_tcan) = (1.d0/xc)*           &
   312)        (-ga*rhoc*cpdry8                      &
   313)        +0.5d0*mac*cph2o8)                   &
   314)        -(dqcdt*dc*rhoc/(xc**2.d0))*           &
   315)        ((1.d0-qc)*cpdry8 + qc*cph2o8)*(cph2o8-cpdry8)
   316)   
   317) 
   318) 
   319) 
   320)   Y(id_tcan) = (3.d0+(dtf/dtb))*ycurr%can_temp -        &
   321)        (dtf/dtb)*yprev%can_temp + 2.d0*B(id_tcan)*dtf
   322) 
   323) 
   324) 
   325)   !===========================================================================!
   326)   ! Derivatives associated with vegetation <-> canopy-air fluxes
   327)   !===========================================================================!
   328)   
   329)   id_tveg = id_tcan
   330) 
   331)   do ico=1,cpatch%ncohorts
   332) 
   333)      if (resolve(ico))then
   334)         
   335)         id_tveg = id_tveg+1
   336) 
   337)         ! dTc/dt ~ Tc)
   338)         A(id_tcan,id_tcan) = A(id_tcan,id_tcan) - gv_hfa(ico)*rhoc*cpdry8/xc
   339)         
   340)         ! A(dTc/dt ~ Tv)
   341)         A(id_tcan,id_tveg) = A(id_tcan,id_tveg)  &
   342)              + (1.d0/xc)*(gv_hfa(ico)*rhoc*cpdry8 + mlc(ico)*cph2o8 + mtr(ico)*cph2o8)
   343)         
   344)         ! B(dTv/dt)
   345)         B(id_tveg) = (1.d0/xv(ico))*           &
   346)              ( hflx(ico)                       &
   347)              - href*mlc(ico)                   &
   348)              - href*mtr(ico)                   &
   349)              + (fliq(ico)*cliq8*t3ple8 - fliq(ico)*cice8*t3ple8 - fliq(ico)*alli8)*dqvdt(ico))
   350)         
   351) 
   352)         ! A(dTv/dt ~ Tc)
   353)         A(id_tveg,id_tcan) = A(id_tveg,id_tcan) + &
   354)              (gv_hfa(ico)*rhoc*cpdry8)/xv(ico)
   355) 
   356) 
   357)         ! A(dTv/dt ~ Tv)
   358)         A(id_tveg,id_tveg) = A(id_tveg,id_tveg) + &
   359)              (1.d0/xv(ico))*( (fliq(ico)*cice8 - fliq(ico)*cliq8 - cice8)*dqvdt(ico) &
   360)              - gv_hfa(ico)*rhoc*cpdry8 - (mlc(ico)+mtr(ico))*cph2o8)
   361) 
   362) 
   363)         Y(id_tveg) = (3.d0+dtf/dtb)*ycurr%leaf_temp(ico) - &
   364)              (dtf/dtb)*yprev%leaf_temp(ico) + &
   365)              2.d0*B(id_tveg)*dtf
   366) 
   367) 
   368)      end if
   369)         
   370)   end do
   371) 
   372)   ! ============================================================!
   373)   ! If this is a thermo case where wood is separate, do another
   374)   ! ============================================================!
   375)   if(ibranch_thermo.eq.2)then
   376) 
   377)      do ico=1,cpatch%ncohorts
   378)         if (ycurr%wood_resolvable(ico)) then
   379)            gv_hfa(ico) = ycurr%wood_gbh(ico)/(ycurr%can_rhos*cpdry8) &
   380)                 *effarea_heat*ycurr%wai(ico)
   381)            mlc(ico)  = ycurr%wflxwc(ico)
   382)            mtr(ico)  = 0.0
   383)            fliq(ico) = ycurr%wood_fliq(ico)
   384)            qv(ico)   = ynext%wood_water(ico)
   385)            dqvdt(ico)= dydt%wood_water(ico)
   386)            xv(ico)   = ycurr%wood_fliq(ico)*cliq8*qv(ico) + &
   387)                 (1.d0-fliq(ico))*cice8*qv(ico) +          &
   388)                 ycurr%wood_hcap(ico)
   389)            hflx(ico) = ycurr%hflx_wrsti(ico)
   390)         end if
   391)      end do
   392)      
   393)      do ico=1,cpatch%ncohorts
   394)         if (ycurr%wood_resolvable(ico)) then
   395) 
   396)            id_tveg = id_tveg+1
   397) 
   398)            ! dTc/dt ~ Tc)
   399)            A(id_tcan,id_tcan) = A(id_tcan,id_tcan) - gv_hfa(ico)*rhoc*cpdry8/xc
   400)         
   401)            ! A(dTc/dt ~ Tv)
   402)            A(id_tcan,id_tveg) = A(id_tcan,id_tveg)  &
   403)                 + (1.d0/xc)*(gv_hfa(ico)*rhoc*cpdry8 + mlc(ico)*cph2o8 + mtr(ico)*cph2o8)
   404)            
   405)            ! B(dTv/dt)
   406)            B(id_tveg) = (1.d0/xv(ico))*           &
   407)                 ( hflx(ico)                       &
   408)                 - href*mlc(ico)                   &
   409)                 - href*mtr(ico)                   &
   410)                 + (fliq(ico)*cliq8*t3ple8 - fliq(ico)*cice8*t3ple8 &
   411)                 - fliq(ico)*alli8)*dqvdt(ico))
   412)         
   413)            
   414)            ! A(dTv/dt ~ Tc)
   415)            A(id_tveg,id_tcan) = A(id_tveg,id_tcan) + &
   416)                 (gv_hfa(ico)*rhoc*cpdry8)/xv(ico)
   417)            
   418)            
   419)            ! A(dTv/dt ~ Tv)
   420)            A(id_tveg,id_tveg) = A(id_tveg,id_tveg) + &
   421)                 (1.d0/xv(ico))*( (fliq(ico)*cice8 - fliq(ico)*cliq8 - cice8)*dqvdt(ico) &
   422)                 - gv_hfa(ico)*rhoc*cpdry8 - (mlc(ico)+mtr(ico))*cph2o8)
   423) 
   424)            Y(id_tveg) = (3.d0+dtf/dtb)*ycurr%wood_temp(ico) - &
   425)                 (dtf/dtb)*yprev%wood_temp(ico) + &
   426)                 2.d0*B(id_tveg)*dtf
   427)            
   428)         end if
   429)      end do
   430)   end if
   431) 
   432) 
   433)   ! Create the matrix  [3I-2Adt]
   434)   ! And create the upper portion of the equation [4U_{n}+U_{n-1}+2Bdt]
   435) 
   436)   do i=1,nstate
   437)      do j=1,nstate
   438)         A(i,j)=-A(i,j)*2.d0*dtf
   439)      end do
   440)      A(i,i) = 3.d0+A(i,i)
   441)   end do
   442)   
   443) 
   444)   call selective_gaussian_2body(A,Yf,Y,nstate)
   445) 
   446)   Y=Yf
   447) 
   448)   ! ------------------------------------------------------------------------!
   449)   ! Set the leaf and canopy temepratures in the memory buffer (rk4type)     !
   450)   ! Update vars that otherwise would not have been diagnostic if we were    !
   451)   ! using a different scheme.  For instance, rk4 and forward euler use      !
   452)   ! can_enthalpy
   453)   !-------------------------------------------------------------------------!
   454)   
   455)   ! Send the canopy and leaf temperatures to the forward 
   456)   ! Be sure to update leaf_energy and canopy ln-theta
   457) 
   458) 
   459)   ynext%can_temp = Y(1)
   460)   ynext%can_enthalpy = (1.d0-qc)*cpdry8*Y(1) + qc*(href + cph2o8*Y(1))
   461) 
   462)   ! ------------------------------------------------------------------------!
   463)   ! Note: Significant assumption being made here.  The partial derivative   !
   464)   ! of the leaf and wood temperature assumed that phase was constant.       !
   465)   ! This is true unless of course the water starts oscillating around the   !
   466)   ! freezing point.  This solver will assume the veg system will heat and   !
   467)   ! cool without phase change, which of course is not true, but nothing is  !
   468)   ! perfect, so deal with it.  AFter ynext%leaf_Temp crosses t3ple8, it will!
   469)   ! change the liquid fraction to either 0 or 1 depending on the direction  !
   470)   ! of change.                                                              !
   471)   ! One possible alternative, is to determine the change in energy assuming !
   472)   ! no phase change, and then apply that energy to the phase change at the  !
   473)   ! triple point.  If the change in energy is less than phase change, set   !
   474)   ! the ynext temp to t3ple and linearly scale the liquid fraction.         !
   475)   ! ------------------------------------------------------------------------!
   476) 
   477)   qwflxlc_tot = 0.d0
   478)   qwflxwc_tot = 0.d0
   479)   qtransp_tot = 0.d0
   480)   hflxlc_tot  = 0.d0
   481)   hflxwc_tot  = 0.d0
   482)   
   483) 
   484)   id_tveg=1
   485)   
   486)   select case(ibranch_thermo)
   487)   case(0,2)
   488)      do ico=1,cpatch%ncohorts
   489)         if (resolve(ico)) then
   490)            id_tveg=id_tveg+1
   491)            ynext%leaf_temp(ico) = Y(id_tveg)
   492)            
   493)            if(ynext%leaf_temp(ico) < t3ple8) then
   494)               ynext%leaf_fliq(ico) = 0.d0
   495)               ynext%leaf_energy(ico) = &
   496)                    ynext%leaf_water(ico)*cice8*ynext%leaf_temp(ico) +&
   497)                    ynext%leaf_temp(ico)*ycurr%leaf_hcap(ico)
   498)            else
   499)               ynext%leaf_fliq(ico) = 1.d0
   500)               ynext%leaf_energy(ico) = ynext%leaf_temp(ico)*              &
   501)                    (ycurr%leaf_hcap(ico)+ynext%leaf_water(ico)*cliq8) - &
   502)                    ynext%leaf_water(ico)*cliq8*tsupercool_liq8
   503)            end if
   504)            
   505)            ! Back calculate the latent and sensible heat fluxes of leaves
   506)            ! ========================================================================
   507)            
   508)            ! First calculate the effective qflxlc
   509)            qwflxlc = ycurr%wflxlc(ico)*tq2enthalpy8(ycurr%leaf_temp(ico),1.d0,.true.)        
   510)            
   511)            ! Then effective transpiraiton
   512)            qtransp = ycurr%wflxtr(ico)*tq2enthalpy8(ycurr%leaf_temp(ico),1.d0,.true.)
   513)            
   514)            ! Use the resulting change in leaf energy to back-caculate what heat
   515)            ! flux would had been
   516)            hflxlc  = ycurr%hflx_lrsti(ico) - qwflxlc - qtransp - &
   517)                 (ynext%leaf_energy(ico)-ycurr%leaf_energy(ico))/dtf
   518)            
   519)            qwflxlc_tot = qwflxlc_tot + qwflxlc
   520)            qtransp_tot = qtransp_tot + qtransp
   521)            hflxlc_tot  = hflxlc_tot  + hflxlc
   522)            
   523)         end if
   524)      end do
   525)      
   526)      if(ibranch_thermo.eq.2) then
   527)         do ico=1,cpatch%ncohorts
   528)            if (ycurr%wood_resolvable(ico)) then
   529)               id_tveg=id_tveg+1
   530)               ynext%wood_temp(ico) = Y(id_tveg)
   531)               if(ynext%wood_temp(ico) < t3ple8) then
   532)                  ynext%wood_fliq(ico) = 0.d0
   533)                  ynext%wood_energy(ico) = &
   534)                       ynext%wood_water(ico)*cice8*ynext%wood_temp(ico) +&
   535)                       ynext%wood_temp(ico)*ycurr%wood_hcap(ico)
   536)               else
   537)                  ynext%wood_fliq(ico) = 1.d0
   538)                  
   539)                  ynext%wood_energy(ico) = ynext%wood_temp(ico)*              &
   540)                       (ycurr%wood_hcap(ico)+ynext%wood_water(ico)*cliq8) - &
   541)                       ynext%wood_water(ico)*cliq8*tsupercool_liq8
   542)               end if
   543)               
   544)               ! Back calculate the latent and sensible heat fluxes of wood
   545)               ! ========================================================================
   546)               qwflxwc = ycurr%wflxwc(ico)*tq2enthalpy8(ycurr%wood_temp(ico),1.d0,.true.)
   547)               
   548)               hflxwc  = ycurr%hflx_wrsti(ico) - qwflxwc - &
   549)                    (ynext%wood_energy(ico)-ycurr%wood_energy(ico))/dtf
   550)               
   551)               qwflxwc_tot = qwflxwc_tot + qwflxwc
   552)               hflxwc_tot  = hflxwc_tot  + hflxwc
   553)            end if  !(if(wood_resolvable))
   554)         end do
   555)      end if
   556)      
   557)   case(1)  !select(ibranch_thermo)
   558)      
   559) !     print*,""
   560) 
   561)      do ico=1,cpatch%ncohorts
   562)         if (resolve(ico)) then
   563)            
   564)            id_tveg=id_tveg+1
   565) 
   566)            ! Update with the new temperature
   567)            ynext%leaf_temp(ico) = Y(id_tveg)
   568)            ynext%wood_temp(ico) = Y(id_tveg)
   569)            
   570)            if(ynext%leaf_temp(ico) < t3ple8) then
   571)               ynext%leaf_fliq(ico) = 0.d0
   572)               ynext%wood_fliq(ico) = 0.d0
   573)            else
   574)               ynext%leaf_fliq(ico) = 1.d0
   575)               ynext%wood_fliq(ico) = 1.d0
   576)            end if
   577) 
   578)            ! Back calculate the latent and sensible heat fluxes of leaves
   579)            ! ========================================================================
   580) 
   581)            ynext%veg_energy(ico)  = 0.0
   582)            ynext%wood_energy(ico) = 0.0
   583)            ynext%leaf_energy(ico) = 0.0
   584) 
   585)            if (ycurr%leaf_resolvable(ico) .or. ycurr%wood_resolvable(ico)) then
   586) 
   587)               ynext%leaf_energy(ico) = cmtl2uext8(     &
   588)                 ycurr%leaf_hcap (ico),              &
   589)                 ynext%leaf_water(ico),              &
   590)                 ynext%leaf_temp (ico),              &
   591)                 ynext%leaf_fliq (ico) )
   592)               
   593)               ynext%veg_energy(ico)=ynext%leaf_energy(ico)
   594) 
   595)               qwflxlc = ycurr%wflxlc(ico) * &
   596)                    tq2enthalpy8(ycurr%leaf_temp(ico),1.d0,.true.)        
   597)               
   598)               hflxlc  = ycurr%hflx_lrsti(ico) - qwflxlc - &
   599)                    (ynext%leaf_energy(ico)-ycurr%leaf_energy(ico))/dtf
   600)               
   601)               qwflxlc_tot = qwflxlc_tot + qwflxlc
   602)               hflxlc_tot  = hflxlc_tot  + hflxlc
   603) 
   604)               ynext%wood_energy(ico) = cmtl2uext8(     &
   605)                    ycurr%wood_hcap (ico),              &
   606)                    ynext%wood_water(ico),              &
   607)                    ynext%wood_temp (ico),              &
   608)                    ynext%wood_fliq (ico) )   
   609) 
   610)               ynext%veg_energy(ico) = ynext%veg_energy(ico)+ &
   611)                    ynext%wood_energy(ico)
   612)               
   613)               qwflxwc = ycurr%wflxwc(ico) * &
   614)                    tq2enthalpy8(ycurr%wood_temp(ico),1.d0,.true.)        
   615)               
   616)               hflxwc  = ycurr%hflx_wrsti(ico) - qwflxwc - &
   617)                    (ynext%wood_energy(ico)-ycurr%wood_energy(ico))/dtf
   618)               
   619)               qwflxwc_tot = qwflxwc_tot + qwflxwc
   620)               hflxwc_tot  = hflxwc_tot  + hflxwc
   621)            end if
   622) 
   623)            ! How does the temperature of the vegetation derived from
   624)            ! its energy compare?
   625) 
   626) 
   627)            call uextcm2tl8(ynext%veg_energy(ico),ynext%veg_water(ico)  &
   628)                 ,ycurr%veg_hcap(ico),veg_temp,fliq(ico))
   629) 
   630)            if(abs(veg_temp-ynext%leaf_temp(ico))>1d-10)then
   631)               print*,"ISSUE IN ENERGY CONSERVATION,BDF2"
   632)               stop
   633)            end if
   634)               
   635) 
   636) 
   637)         end if
   638)      end do
   639)   end select
   640) 
   641) 
   642)   ! Update eulerian based budget fluxes
   643)   ! ============================================================
   644)   
   645) 
   646)   eflxac = rk4aux(ibuff)%hcapcan*(ynext%can_enthalpy-ycurr%can_enthalpy)/dtf  - &
   647)        (dydt%avg_sensible_gc + ycurr%qwflxsc + ycurr%qwflxgc    + &
   648)        hflxlc_tot + qwflxlc_tot + qtransp_tot + hflxwc_tot + qwflxwc_tot)
   649) 
   650) !  print*,hflxlc_tot,qwflxlc_tot,qtransp_tot,hflxwc_tot,qwflxwc_tot
   651)   
   652)   qwflxac = ycurr%wflxac * tq2enthalpy8(0.5*(ycurr%can_temp+Ta),1.d0,.true.)
   653)   
   654)   hflxac  = eflxac-qwflxac
   655) 
   656)   if(checkbudget)then 
   657)      
   658)      ! Remove the previous integration
   659)      ynext%ebudget_loss2atm   = ynext%ebudget_loss2atm            - &
   660)           dydt%ebudget_loss2atm*dtf
   661)      
   662)      ! Add the new integration
   663)      dydt%ebudget_loss2atm    = -eflxac
   664)      ynext%ebudget_loss2atm   = ynext%ebudget_loss2atm - eflxac*dtf
   665)      
   666)   end if
   667)   
   668)   if (fast_diagnostics .or. checkbudget ) then     
   669) 
   670)      ! Update the sensible heat flux diagnostic
   671)      ynext%avg_sensible_ac = ynext%avg_sensible_ac - dydt%avg_sensible_ac*dtf
   672)      ynext%avg_sensible_ac = ynext%avg_sensible_ac + (hflxac)*dtf
   673) 
   674)   end if
   675) 
   676) 
   677)   ! Make corrections to sensible heat flux and tstar
   678)   ! ======================================================
   679)   
   680)   ! Remove the previous increment
   681)   ynext%tpwp      = ynext%tpwp-dydt%tpwp*dtf
   682)   ynext%avg_tstar = ynext%avg_tstar-dydt%tstar*dtf
   683)   
   684) 
   685)   ! Make the current increment
   686)   ynext%avg_tstar = ynext%tstar + &
   687)        dtf*(hflxac/(rhoc*ycurr%ustar*ycurr%can_exner))
   688)    
   689) 
   690)   ynext%tpwp = ynext%tpwp -(hflxac/(rhoc*ycurr%can_exner))*dtf
   691)   
   692) 
   693)   ! Free memory
   694)   deallocate(Y,Yf,A,B)
   695) 
   696) 
   697)   return
   698) end subroutine bdf2_solver

   702) subroutine selective_gaussian_2body(ad,yd,xd,np)
   703) 
   704)   implicit none
   705)   
   706)   integer, intent(in)                       :: np
   707)   real(kind=8), dimension(np),intent(out)   :: yd
   708)   real(kind=8), dimension(np,np),intent(in) :: ad
   709)   real(kind=8), dimension(np),intent(in)    :: xd
   710)   real(kind=8)                              :: ydl
   711)   integer                                   :: i
   712) 
   713) 
   714)   
   715)   if (np>1) then
   716) 
   717)      yd(1) = xd(1)
   718)      ydl   = ad(1,1)
   719) 
   720)      do i=2,np
   721)         yd(1) = yd(1)-ad(1,i)*xd(i)/ad(i,i)
   722)         ydl = ydl-ad(1,i)*ad(i,1)/ad(i,i)
   723)      end do
   724) 
   725)      ! Solved the first term by gaussian substitution
   726)      yd(1) = yd(1)/ydl
   727) 
   728)      ! Now do the remaining terms
   729) 
   730)      do i=2,np
   731)         yd(i) = (xd(i)-ad(i,1)*yd(1))/ad(i,i)
   732)      end do
   733) 
   734)   else
   735) 
   736)      ! Trivial solution, no cohorts
   737) 
   738)      yd(1) = xd(1)/ad(1,1)
   739) 
   740)   end if
   741) 
   742) 
   743) 
   744)   return
   745) end subroutine selective_gaussian_2body

   750) subroutine ludcmp_dble(ad,n,np,indx,d)
   751) 
   752)   implicit none
   753)   
   754)   real(kind=8), parameter :: tiny_offset = 1.0d-32
   755)   integer, intent(in) :: n
   756)   integer, intent(in) :: np
   757)   real, intent(out) :: d
   758) !  real, (kind=8),dimension(np,np), intent(inout) :: a
   759)   real(kind=8), dimension(np,np),intent(inout) :: ad
   760)   integer, dimension(n), intent(out) :: indx
   761)   real(kind=8), dimension(np) :: vv
   762)   integer :: i
   763)   integer :: j
   764)   integer :: k
   765)   integer :: imax
   766)   real(kind=8) :: sum
   767)   real(kind=8) :: aamax
   768)   real(kind=8) :: dum
   769) 
   770) !  ad = dble(a)
   771) 
   772)   d = 1.0
   773) 
   774)   do i = 1, n
   775)      aamax = 0.0d0
   776)      do j = 1, n
   777)         if(abs(ad(i,j)) > aamax)aamax = abs(ad(i,j))
   778)      enddo
   779)      if(aamax == 0.0d0)then
   780)         print*,'singular matrix in ludcmp'
   781)         do j=1,n
   782)            print*,i,j,abs(ad(i,j)),aamax
   783)         enddo
   784)         stop
   785)      endif
   786)      vv(i) = 1.0d0 / aamax
   787)   enddo
   788) 
   789)   do j = 1, n
   790)      if(j.gt.1)then
   791)         do i = 1, j - 1
   792)            sum = ad(i,j)
   793)            if(i > 1)then
   794)               do k = 1, i - 1
   795)                  sum = sum - ad(i,k) * ad(k,j)
   796)               enddo
   797)               ad(i,j) = sum
   798)            endif
   799)         enddo
   800)      endif
   801)      aamax = 0.0d0
   802)      do i=j,n
   803)         sum = ad(i,j)
   804)         if (j > 1)then
   805)            do k = 1, j - 1
   806)               sum = sum - ad(i,k) * ad(k,j)
   807)            enddo
   808)            ad(i,j) = sum
   809)         endif
   810)         dum = vv(i) * abs(sum)
   811)         if(dum >= aamax)then
   812)            imax = i
   813)            aamax = dum
   814)         endif
   815)      enddo
   816)      if(j /= imax)then
   817)         do k = 1, n
   818)            dum = ad(imax,k)
   819)            ad(imax,k) = ad(j,k)
   820)            ad(j,k) = dum
   821)         enddo
   822)         d = -d
   823)         vv(imax) = vv(j)
   824)      endif
   825)      indx(j) = imax
   826)      if(j /= n)then
   827)         if(ad(j,j) == 0.0d0) ad(j,j) = tiny_offset
   828)         dum = 1.0d0 / ad(j,j)
   829)         do i = j + 1, n
   830)            ad(i,j) = ad(i,j) * dum
   831)         enddo
   832)      endif
   833)   enddo
   834)   if(ad(n,n) == 0.0d0)ad(n,n) = tiny_offset
   835) 
   836) !  a = real(ad)
   837) 
   838)   return
   839) end subroutine ludcmp_dble

   843) subroutine lubksb_dble(ad,n,np,indx,bd)
   844)   implicit none
   845)   integer, intent(in) :: n
   846)   integer, intent(in) :: np
   847)   integer, dimension(n), intent(in) :: indx
   848) !  real(, dimension(n), intent(inout) :: b
   849)   real(kind=8), dimension(n),intent(inout) :: bd
   850) !  real(kind=8), dimension(np,np), intent(in) :: a
   851)   real(kind=8), dimension(np,np),intent(in) :: ad
   852)   integer :: ii
   853)   integer :: i
   854)   integer :: ll
   855)   real(kind=8) :: sum
   856)   integer :: j
   857) 
   858) !  ad = dble(a)
   859) !  bd = dble(b)
   860) 
   861)   ii = 0
   862)   do i=1,n
   863)      ll = indx(i)
   864)      sum = bd(ll)
   865)      bd(ll) = bd(i)
   866)      if(ii /= 0)then
   867)         do j=ii,i-1
   868)            sum = sum - ad(i,j) * bd(j)
   869)         enddo
   870)      elseif(sum /= 0.0d0)then
   871)         ii = i
   872)      endif
   873)      bd(i) = sum
   874)   enddo
   875)   do i=n,1,-1
   876)      sum = bd(i)
   877)      if ( i < n )then
   878)         do j=i+1,n
   879)            sum = sum - ad(i,j) * bd(j)
   880)         enddo
   881)      endif
   882)      bd(i) = sum / ad(i,i)
   883)   enddo
   884) 
   885) !  b = real(bd)
   886) 
   887)   return
   888) end subroutine lubksb_dble

   908) SUBROUTINE MIGS (A,N,X,INDX)
   909) !
   910) ! Subroutine to invert matrix A(N,N) with the inverse stored
   911) ! in X(N,N) in the output.  Copyright (c) Tao Pang 2001.
   912) !
   913)   IMPLICIT NONE
   914)   INTEGER, INTENT (IN) :: N
   915)   INTEGER :: I,J,K
   916)   INTEGER, INTENT (OUT), DIMENSION (N) :: INDX
   917)   REAL, INTENT (INOUT), DIMENSION (N,N):: A
   918)   REAL, INTENT (OUT), DIMENSION (N,N):: X
   919)   REAL, DIMENSION (N,N) :: B
   920) !
   921)   DO I = 1, N
   922)     DO J = 1, N
   923)       B(I,J) = 0.0
   924)     END DO
   925)   END DO
   926)   DO I = 1, N
   927)     B(I,I) = 1.0
   928)   END DO
   929) !
   930)   CALL ELGS (A,N,INDX)
   931) !
   932)   DO I = 1, N-1
   933)     DO J = I+1, N
   934)       DO K = 1, N
   935)         B(INDX(J),K) = B(INDX(J),K)-A(INDX(J),I)*B(INDX(I),K)
   936)       END DO
   937)     END DO
   938)   END DO
   939) !
   940)   DO I = 1, N
   941)     X(N,I) = B(INDX(N),I)/A(INDX(N),N)
   942)     DO J = N-1, 1, -1
   943)       X(J,I) = B(INDX(J),I)
   944)       DO K = J+1, N
   945)         X(J,I) = X(J,I)-A(INDX(J),K)*X(K,I)
   946)       END DO
   947)       X(J,I) =  X(J,I)/A(INDX(J),J)
   948)     END DO
   949)   END DO
   950) END SUBROUTINE MIGS
   951) !
   952) SUBROUTINE ELGS (A,N,INDX)
   953) !
   954) ! Subroutine to perform the partial-pivoting Gaussian elimination.
   955) ! A(N,N) is the original matrix in the input and transformed matrix
   956) ! plus the pivoting element ratios below the diagonal in the output.
   957) ! INDX(N) records the pivoting order.  Copyright (c) Tao Pang 2001.
   958) !
   959)   IMPLICIT NONE
   960)   INTEGER, INTENT (IN) :: N
   961)   INTEGER :: I,J,K,ITMP
   962)   INTEGER, INTENT (OUT), DIMENSION (N) :: INDX
   963)   REAL :: C1,PI,PI1,PJ
   964)   REAL, INTENT (INOUT), DIMENSION (N,N) :: A
   965)   REAL, DIMENSION (N) :: C
   966) !
   967) ! Initialize the index
   968) !
   969)   DO I = 1, N
   970)     INDX(I) = I
   971)   END DO
   972) !
   973) ! Find the rescaling factors, one from each row
   974) !
   975)   DO I = 1, N
   976)     C1= 0.0
   977)     DO J = 1, N
   978)       C1 = AMAX1(C1,ABS(A(I,J)))
   979)     END DO
   980)     C(I) = C1
   981)   END DO
   982) !
   983) ! Search the pivoting (largest) element from each column
   984) !
   985)   DO J = 1, N-1
   986)     PI1 = 0.0
   987)     DO I = J, N
   988)       PI = ABS(A(INDX(I),J))/C(INDX(I))
   989)       IF (PI.GT.PI1) THEN
   990)         PI1 = PI
   991)         K   = I
   992)       ENDIF
   993)     END DO
   994) !
   995) ! Interchange the rows via INDX(N) to record pivoting order
   996) !
   997)     ITMP    = INDX(J)
