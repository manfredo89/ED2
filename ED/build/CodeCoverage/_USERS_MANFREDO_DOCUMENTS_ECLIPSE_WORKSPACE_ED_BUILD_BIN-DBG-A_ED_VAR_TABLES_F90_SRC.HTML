   155)    !   This sub-routine creates the pointers for output, when the variable is real.        !
   156)    !                                                                                       !
   157)    !     NPTS           - Size of this variable                                            !
   158)    !     VAR            - The pointer of the current state variable                        !
   159)    !     NV             - The variable type number                                         !
   160)    !     IGR            - The number of the current grid                                   !
   161)    !     INIT           - Initialize the vt_info?                                          !
   162)    !     GLOB_ID        - The global index of the data                                     !
   163)    !     VAR_LEN        - The length of the states current vector                          !
   164)    !     VAR_LEN_GLOBAL - The length of the entire dataset's vector                        !
   165)    !     MAX_PTRS       - The maximum possible number of pointers necessary for this       !
   166)    !                      variable                                                         !
   167)    !     TABSTR         - The string describing the variables usage                        !
   168)    !---------------------------------------------------------------------------------------!
   169)    recursive subroutine vtable_edio_r(npts,var,nv,igr,init,glob_id,var_len,var_len_global  &
   170)                                      ,max_ptrs,tabstr)
   171)       use ed_max_dims, only : str_len ! ! intent(in)
   172)       implicit none
   173)       !----- Arguments. -------------------------------------------------------------------!
   174)       integer                                , intent(in) :: npts
   175)       real(kind=4)          , dimension(npts), target     :: var
   176)       integer                                , intent(in) :: nv
   177)       integer                                , intent(in) :: igr
   178)       integer                                , intent(in) :: init
   179)       integer                                , intent(in) :: glob_id
   180)       integer                                , intent(in) :: var_len
   181)       integer                                , intent(in) :: var_len_global
   182)       integer                                , intent(in) :: max_ptrs
   183)       character (len=*)                      , intent(in) :: tabstr
   184)       !----- Local variables. -------------------------------------------------------------!
   185)       integer                                             :: iptr
   186)       character(len=str_len), dimension(10)               :: tokens
   187)       character(len=8)                                    :: ctab
   188)       integer                                             :: ntok
   189)       integer                                             :: nt
   190)       !----- Local constants. -------------------------------------------------------------!
   191)       character (len=1)                      , parameter  :: toksep=':'
   192)       !------------------------------------------------------------------------------------!
   193) 
   194) 
   195) 
   196)       !------------------------------------------------------------------------------------!
   197)       !      Determine if this is the first time we view this variable.  If so, then fill  !
   198)       ! some descriptors for the vtable and allocate some space for any pointers that may  !
   199)       ! follow.                                                                            !
   200)       !------------------------------------------------------------------------------------!
   201)       if (init == 0) then
   202)          !----- Make sure we have a clean start. ------------------------------------------!
   203)          call reset_vt_vector_pointers(vt_info(nv,igr))
   204)          !---------------------------------------------------------------------------------!
   205) 
   206)          !----- Count the number of variables. --------------------------------------------!
   207)          num_var(igr) = num_var(igr) + 1
   208)          call tokenize1(tabstr,tokens,ntok,toksep)
   209) 
   210)          vt_info(nv,igr)%name           = tokens(1)
   211)          vt_info(nv,igr)%dtype          = 'R'  ! This is a real variable (vector)
   212)          vt_info(nv,igr)%nptrs          = 0
   213)          vt_info(nv,igr)%var_len_global = var_len_global
   214) 
   215)          vt_info(nv,igr)%vector_allocated = .true.
   216)          allocate(vt_info(nv,igr)%vt_vector(max_ptrs))
   217)          read(tokens(2),fmt=*) vt_info(nv,igr)%idim_type
   218) 
   219)          vt_info(nv,igr)%ihist    = 0
   220)          vt_info(nv,igr)%ianal    = 0
   221)          vt_info(nv,igr)%imean    = 0
   222)          vt_info(nv,igr)%ilite    = 0
   223)          vt_info(nv,igr)%impti    = 0
   224)          vt_info(nv,igr)%impt1    = 0
   225)          vt_info(nv,igr)%impt2    = 0
   226)          vt_info(nv,igr)%impt3    = 0
   227)          vt_info(nv,igr)%irecycle = 0
   228)          vt_info(nv,igr)%imont    = 0
   229)          vt_info(nv,igr)%idail    = 0
   230)          vt_info(nv,igr)%iyear    = 0
   231)          vt_info(nv,igr)%idcyc    = 0
   232)          vt_info(nv,igr)%iopti    = 0
   233)          
   234)          do nt=3,ntok
   235)             ctab=tokens(nt)
   236) 
   237)             select case (trim(ctab))
   238)             case('hist') 
   239)                vt_info(nv,igr)%ihist    = 1
   240) 
   241)             case('anal') 
   242)                vt_info(nv,igr)%ianal    = 1
   243) 
   244)             case('lite') 
   245)                vt_info(nv,igr)%ilite    = 1
   246) 
   247)             case('mpti') 
   248)                vt_info(nv,igr)%impti    = 1
   249) 
   250)             case('mpt1') 
   251)                vt_info(nv,igr)%impt1    = 1
   252) 
   253)             case('mpt2') 
   254)                vt_info(nv,igr)%impt2    = 1
   255) 
   256)             case('mpt3') 
   257)                vt_info(nv,igr)%impt3    = 1
   258) 
   259)             case('recycle') 
   260)                vt_info(nv,igr)%irecycle = 1
   261) 
   262)             case('mont') 
   263)                vt_info(nv,igr)%imont    = 1
   264) 
   265)             case('dail') 
   266)                vt_info(nv,igr)%idail    = 1
   267) 
   268)             case('dcyc') 
   269)                vt_info(nv,igr)%idcyc    = 1
   270) 
   271)             case('year') 
   272)                vt_info(nv,igr)%iyear    = 1
   273) 
   274)             case('opti') 
   275)                vt_info(nv,igr)%iopti    = 1
   276) 
   277)             case default
   278)                print*, 'Illegal table specification for var:', tokens(1),ctab
   279)                call fatal_error('Bad var table','vtable_edio_r','ed_var_tables.f90')
   280)             end select
   281)          end do
   282)       else
   283)          !---------------------------------------------------------------------------------!
   284)          !     Make sure that vt_info is associated. If not, call the function with        !
   285)          ! init = 0 then do this part.  Since I think this should never happen, I will     !
   286)          ! also make a fuss to warn the user.                                              !
   287)          !---------------------------------------------------------------------------------!
   288)          if (.not. vt_info(nv,igr)%vector_allocated) then
   289)             write (unit=*,fmt='(a)') ' '
   290)             write (unit=*,fmt='(a)') '----------------------------------------------'
   291)             write (unit=*,fmt='(a)') ' WARNING! WARNING! WARNING! WARNING! WARNING! '
   292)             write (unit=*,fmt='(a)') ' WARNING! WARNING! WARNING! WARNING! WARNING! '
   293)             write (unit=*,fmt='(a)') ' WARNING! WARNING! WARNING! WARNING! WARNING! '
   294)             write (unit=*,fmt='(a)') '----------------------------------------------'
   295)             write (unit=*,fmt='(a)') ' - Subroutine vtable_edio_r (file ed_var_tables.f90)'
   296)             write (unit=*,fmt='(a,1x,i4,1x,a,1x,i2,1x,a)')                                 &
   297)                                      ' - Vt_vector for variable',nv,'of grid',igr          &
   298)                                      ,'is not allocated              !'
   299)             write (unit=*,fmt='(a)') ' - I will allocate it now.'
   300)             write (unit=*,fmt='(a,1x,i20,1x,a)') ' - MAX_PTRS=',max_ptrs,'...'
   301)             write (unit=*,fmt='(a,1x,a,1x,a)') ' - Tabstr=',tabstr,'...'
   302)             write (unit=*,fmt='(a)') '----------------------------------------------'
   303)             write (unit=*,fmt='(a)') ' '
   304)             call vtable_edio_r(npts,var,nv,igr,0,glob_id,var_len,var_len_global,max_ptrs   &
   305)                               ,tabstr)
   306)          end if
   307)          
   308)          vt_info(nv,igr)%nptrs                  =  vt_info(nv,igr)%nptrs + 1
   309)          iptr                                   =  vt_info(nv,igr)%nptrs
   310)          vt_info(nv,igr)%vt_vector(iptr)%globid =  glob_id
   311) 
   312)          vt_info(nv,igr)%vt_vector(iptr)%var_rp => var
   313)          vt_info(nv,igr)%vt_vector(iptr)%varlen =  var_len
   314)       end if
   315)       return
   316)    end subroutine vtable_edio_r

   861)    !     This subroutine unlinks all pointers then deallocate the vector, for a safe re-   !
   862)    ! allocation.                                                                           !
   863)    !---------------------------------------------------------------------------------------!
   864)    subroutine reset_vt_vector_pointers(vt)
   865)       implicit none
   866)       !----- Arguments. -------------------------------------------------------------------!
   867)       type(var_table), intent(inout) :: vt
   868)       !----- Local variables. -------------------------------------------------------------!
   869)       integer                        :: iptr
   870)       !------------------------------------------------------------------------------------!
   871) 
   872)       !------------------------------------------------------------------------------------!
   873)       !      I actually don't know why, but the typical "if (allocated(vt%vt_vector))"     !
   874)       ! doesn't work here, so instead we save a logical test...                            !
   875)       !------------------------------------------------------------------------------------!
   876)       if (vt%vector_allocated) then
   877)          !---------------------------------------------------------------------------------!
   878)          !     Go through all pointer elements and nullify those associated with vari-     !
   879)          ! ables.  We must do that before we deallocate so only the pointers, not the true !
   880)          ! arrays, are deallocated.                                                        !
   881)          !---------------------------------------------------------------------------------!
   882)          do iptr=1,vt%nptrs
   883)             if (associated(vt%vt_vector(iptr)%var_rp)) nullify(vt%vt_vector(iptr)%var_rp)
   884)             if (associated(vt%vt_vector(iptr)%var_ip)) nullify(vt%vt_vector(iptr)%var_ip)
   885)             if (associated(vt%vt_vector(iptr)%var_cp)) nullify(vt%vt_vector(iptr)%var_cp)
   886)             if (associated(vt%vt_vector(iptr)%var_dp)) nullify(vt%vt_vector(iptr)%var_dp)
   887)             if (associated(vt%vt_vector(iptr)%sca_rp)) nullify(vt%vt_vector(iptr)%sca_rp)
   888)             if (associated(vt%vt_vector(iptr)%sca_ip)) nullify(vt%vt_vector(iptr)%sca_ip)
   889)             if (associated(vt%vt_vector(iptr)%sca_cp)) nullify(vt%vt_vector(iptr)%sca_cp)
   890)             if (associated(vt%vt_vector(iptr)%sca_dp)) nullify(vt%vt_vector(iptr)%sca_dp)
   891)          end do
   892)          deallocate(vt%vt_vector)
   893)       end if
   894)       !------------------------------------------------------------------------------------!
   895) 
   896) 
   897)       !------------------------------------------------------------------------------------!
   898)       !      Now it's safe to nullify; we also update the logical flag...                  !
   899)       !------------------------------------------------------------------------------------!
   900)       nullify(vt%vt_vector)
   901)       vt%vector_allocated = .false.
   902)       !------------------------------------------------------------------------------------!
   903) 
   904)       return
   905)    end subroutine reset_vt_vector_pointers
