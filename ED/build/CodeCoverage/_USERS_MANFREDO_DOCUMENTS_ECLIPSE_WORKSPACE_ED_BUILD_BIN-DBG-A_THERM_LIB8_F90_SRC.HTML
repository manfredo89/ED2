   135)    !     This function calculates the liquid saturation vapour pressure as a function of   !
   136)    ! Kelvin temperature. This expression came from MK05, equation (10).                    !
   137)    !---------------------------------------------------------------------------------------!
   138)    real(kind=8) function eslf8(temp,l1funout,l2funout,ttfunout)
   139)       use consts_coms, only : t008 ! ! intent(in)
   140)       implicit none
   141)       !----- Required arguments. ----------------------------------------------------------!
   142)       real(kind=8), intent(in)            :: temp     ! Temperature                [     K]
   143)       !----- Optional arguments. ----------------------------------------------------------!
   144)       real(kind=8), intent(out), optional :: l1funout ! Function for high temperatures
   145)       real(kind=8), intent(out), optional :: ttfunout ! Interpolation function
   146)       real(kind=8), intent(out), optional :: l2funout ! Function for low temperatures
   147)       !----- Local variables. -------------------------------------------------------------!
   148)       real(kind=8)                        :: l1fun    !
   149)       real(kind=8)                        :: ttfun    !
   150)       real(kind=8)                        :: l2fun    !
   151)       real(kind=8)                        :: x        !
   152)       !------------------------------------------------------------------------------------!
   153) 
   154) 
   155) 
   156)       !------------------------------------------------------------------------------------!
   157)       !     Choose between the old and the new thermodynamics.                             !
   158)       !------------------------------------------------------------------------------------!
   159)       if (newthermo) then
   160)          !----- Updated method, using MK05 ------------------------------------------------!
   161)          l1fun = l01_108(0) + l01_108(1)/temp + l01_108(2)*log(temp) + l01_108(3) * temp
   162)          l2fun = l02_108(0) + l02_108(1)/temp + l02_108(2)*log(temp) + l02_108(3) * temp
   163)          ttfun = tanh(ttt_108(1) * (temp - ttt_108(2)))
   164)          eslf8 = exp(l1fun + ttfun*l2fun)
   165)          !---------------------------------------------------------------------------------!
   166) 
   167)          if (present(l1funout)) l1funout = l1fun
   168)          if (present(l2funout)) l2funout = l2fun
   169)          if (present(ttfunout)) ttfunout = ttfun
   170)       else
   171)          !----- Original method, using polynomial fit (FWC92) -----------------------------!
   172)          x     = max(-8.0d1,temp-t008)
   173)          eslf8 = cll8(0) + x * (cll8(1) + x * (cll8(2) + x * (cll8(3) + x * (cll8(4)       &
   174)                          + x * (cll8(5) + x * (cll8(6) + x * (cll8(7) + x * cll8(8)) ))))))
   175)          !---------------------------------------------------------------------------------!
   176) 
   177)          if (present(l1funout)) l1funout = eslf8
   178)          if (present(l2funout)) l2funout = eslf8
   179)          if (present(ttfunout)) ttfunout = eslf8
   180)       end if
   181) 
   182)       return
   183)    end function eslf8

   245)    !     This function calculates the saturation vapour pressure as a function of  Kelvin  !

   250)       use consts_coms, only : t3ple8 ! ! intent(in)
   251)       implicit none
   252)       !----- Required arguments. ----------------------------------------------------------!
   253)       real(kind=8), intent(in)           :: temp

   256)       !----- Local variables. -------------------------------------------------------------!
   257)       logical                            :: frozen
   258)       !------------------------------------------------------------------------------------!
   259) 
   260) 
   261)       !------------------------------------------------------------------------------------!
   262)       !     Decide which function to use (saturation for liquid water or ice).             !
   263)       !------------------------------------------------------------------------------------!
   264)       if (present(useice)) then
   265)          frozen = useice  .and. temp < t3ple8
   266)       else
   267)          frozen = bulk_on .and. temp < t3ple8
   268)       end if
   269)       !------------------------------------------------------------------------------------!
   270) 
   271) 
   272)

   278)          !----- Saturation vapour pressure for ice. ---------------------------------------!
   279)          eslif8 = esif8(temp)
   280)          !---------------------------------------------------------------------------------!

   282)          !----- Saturation vapour pressure for liquid. ------------------------------------!
   283)          eslif8 = eslf8(temp)

  1154)       implicit none
  1155)       !----- Arguments. -------------------------------------------------------------------!
  1156)       real(kind=8), intent(in) :: pvap      ! Saturation vapour pressure           [    Pa]
  1157)       !----- Local variables for iterative method. ----------------------------------------!
  1158)       real(kind=8)             :: deriv     ! Function derivative                  [    Pa]
  1159)       real(kind=8)             :: fun       ! Function for which we seek a root.   [    Pa]
  1160)       real(kind=8)             :: funa      ! Smallest  guess function             [    Pa]
  1161)       real(kind=8)             :: funz      ! Largest   guess function             [    Pa]
  1162)       real(kind=8)             :: tempa     ! Smallest guess (or previous guess)   [    Pa]
  1163)       real(kind=8)             :: tempz     ! Largest guess (new guess in Newton)  [    Pa]
  1164)       real(kind=8)             :: delta     ! Aux. var --- 2nd guess for bisection [      ]
  1165)       integer                  :: itn       ! Iteration counter                    [   ---]
  1166)       integer                  :: itb       ! Iteration counter                    [   ---]
  1167)       logical                  :: converged ! Convergence handle                   [   ---]
  1168)       logical                  :: zside     ! Flag to check for one-sided approach [   ---]
  1169)       !------------------------------------------------------------------------------------!
  1170) 
  1171)       !----- First Guess, use Bolton (1980) equation 11, giving es in Pa and T in K -------!
  1172)       tempa = (2.965d1 * log(pvap) - 5.01678d3)/(log(pvap)-2.40854d1)
  1173)       funa  = eslf8(tempa) - pvap
  1174)       deriv = eslfp8(tempa)
  1175)       !------------------------------------------------------------------------------------!
  1176) 
  1177) 
  1178)       !----- Copy just in case it fails at the first iteration ----------------------------!
  1179)       tempz = tempa
  1180)       fun   = funa
  1181)       !------------------------------------------------------------------------------------!
  1182) 
  1183) 
  1184)       !----- Enter Newton's method loop: --------------------------------------------------!
  1185)       converged = .false.
  1186)       newloop: do itn = 1,maxfpo/6
  1187)          if (abs(deriv) < toler8) exit newloop !----- Too dangerous, go with bisection ----!
  1188) 
  1189)          !----- Copy the previous guess. --------------------------------------------------!
  1190)          tempa = tempz
  1191)          funa  = fun
  1192)          !---------------------------------------------------------------------------------!
  1193) 
  1194) 
  1195)          !----- New guess, its function, and derivative evaluation. -----------------------!
  1196)          tempz = tempa - fun/deriv
  1197)          fun   = eslf8(tempz) - pvap
  1198)          deriv = eslfp8(tempz)
  1199)          !---------------------------------------------------------------------------------!
  1200) 
  1201) 
  1202)          !----- Check convergence. --------------------------------------------------------!
  1203)          converged = abs(tempa-tempz) < toler8 * tempz
  1204)          if (converged) then
  1205)             tslf8 = 5.d-1 * (tempa+tempz)
  1206)             return
  1207)          elseif (fun == 0.0d0) then
  1208)             !----- Converged by luck. -----------------------------------------------------!
  1209)             tslf8 = tempz
  1210)             return
  1211)          end if
  1212)          !---------------------------------------------------------------------------------!
  1213)       end do newloop
  1214)       !------------------------------------------------------------------------------------!
  1215) 
  1216) 
  1217) 
  1218)       !------------------------------------------------------------------------------------!
  1219)       !     If we have reached this point, then Newton's method has failed.  Use bisection !
  1220)       ! instead.  For bisection, we need two guesses whose function has opposite signs.    !
  1221)       !------------------------------------------------------------------------------------!
  1222)       if (funa * fun < 0.d0) then
  1223)          !----- We already have two guesses with opposite signs. --------------------------!
  1224)          funz  = fun
  1225)          zside = .true.
  1226)          !---------------------------------------------------------------------------------!
  1227)       else
  1228)          !----- Need to find the guesses with opposite signs. -----------------------------!
  1229)          if (abs(fun-funa) < 1.d2*toler8*tempa) then
  1230)             delta = 1.d2*toler8*tempa
  1231)          else
  1232)             delta = max(abs(funa * (tempz-tempa)/(fun-funa)),1.d2*toler8*tempa)
  1233)          end if
  1234)          !---------------------------------------------------------------------------------!
  1235) 
  1236) 
  1237)          !---------------------------------------------------------------------------------!
  1238)          !    Try guesses on both sides of the first guess, increasingly further away      !
  1239)          ! until we spot a good guess.                                                     !
  1240)          !---------------------------------------------------------------------------------!
  1241)          tempz = tempa + delta
  1242)          zside = .false.
  1243)          zgssloop: do itb=1,maxfpo
  1244)             tempz = tempa + dble((-1)**itb * (itb+3)/2) * delta
  1245)             funz  = eslf8(tempz) - pvap
  1246)             zside = funa*funz < 0.d0
  1247)             if (zside) exit zgssloop
  1248)          end do zgssloop
  1249)          if (.not. zside) then
  1250)             write (unit=*,fmt='(a)')           '------------------------------------------'
  1251)             write (unit=*,fmt='(a)')           ' Failed finding the second guess:'
  1252)             write (unit=*,fmt='(a)')           '------------------------------------------'
  1253)             write (unit=*,fmt='(a)')           ' '
  1254)             write (unit=*,fmt='(a)')           ' Input:   '
  1255)             write (unit=*,fmt='(a,1x,es14.7)') ' + pvap  =',pvap
  1256)             write (unit=*,fmt='(a)')           ' '
  1257)             write (unit=*,fmt='(a)')           ' Output:  '

  1266)          end if
  1267)       end if
  1268) 
  1269) 
  1270)       bisloop: do itb=itn,maxfpo
  1271)          tslf8 =  (funz*tempa-funa*tempz)/(funz-funa)
  1272) 
  1273)          !---------------------------------------------------------------------------------!
  1274)          !     Now that we updated the guess, check whether they are really close.  If so, !
  1275)          ! it converged, I can use this as my guess.                                       !
  1276)          !---------------------------------------------------------------------------------!
  1277)          converged = abs(tslf8-tempa) < toler8 * tslf8
  1278)          if (converged) exit bisloop
  1279) 
  1280)          !------ Find the new function evaluation. ----------------------------------------!
  1281)          fun       =  eslf8(tslf8) - pvap
  1282)          !---------------------------------------------------------------------------------!
  1283) 
  1284) 
  1285)          !------ Define the new interval based on the intermediate value theorem. ---------!
  1286)          if (fun*funa < 0.d0 ) then
  1287)             tempz = tslf8
  1288)             funz  = fun
  1289)             !----- If we are updating zside again, modify aside (Illinois method). --------!
  1290)             if (zside) funa = funa * 5.d-1
  1291)             !----- We have just updated zside, so we set zside to true. -------------------!
  1292)             zside = .true.
  1293)          else
  1294)             tempa = tslf8
  1295)             funa   = fun
  1296)             !----- If we are updating aside again, modify zside (Illinois method). --------!
  1297)             if (.not. zside) funz = funz * 5.d-1
  1298)             !----- We have just updated aside, so we set zside to false. ------------------!
  1299)             zside = .false.
  1300)          end if
  1301)       end do bisloop
  1302) 
  1303)       if (.not.converged) then
  1304)          write (unit=*,fmt='(a)')           '------------------------------------------'
  1305)          write (unit=*,fmt='(a)')           ' Failed finding the solution:'
  1306)          write (unit=*,fmt='(a)')           '------------------------------------------'
  1307)          write (unit=*,fmt='(a)')           ' '
  1308)          write (unit=*,fmt='(a)')           ' Input:   '
  1309)          write (unit=*,fmt='(a,1x,es14.7)') ' + pvap  =',pvap
  1310)          write (unit=*,fmt='(a)')           ' '
  1311)          write (unit=*,fmt='(a)')           ' Output:  '
  1312)          write (unit=*,fmt='(a,1x,es14.7)') ' + tempa =',tempa
  1313)          write (unit=*,fmt='(a,1x,es14.7)') ' + funa  =',funa
  1314)          write (unit=*,fmt='(a,1x,es14.7)') ' + tempz =',tempz

  1316)          write (unit=*,fmt='(a)')           ' '
  1317)          write (unit=*,fmt='(a)')           '------------------------------------------'

  2460)    !     This function computes reduces the pressure from the reference height to the      !

  2464)    real(kind=8) function reducedpress8(pres,thetaref,shvref,zref,thetacan,shvcan,zcan)
  2465)       use consts_coms, only : epim18    & ! intent(in)
  2466)                             , p00k8     & ! intent(in)
  2467)                             , rocp8     & ! intent(in)

  2469)                             , cpdry8    & ! intent(in)
  2470)                             , grav8     ! ! intent(in)

  2472)       !----- Arguments --------------------------------------------------------------------!

  2475)       real(kind=8), intent(in) :: shvref   ! Vapour specific mass                [   kg/kg]
  2476)       real(kind=8), intent(in) :: zref     ! Height at reference level           [       m]
  2477)       real(kind=8), intent(in) :: thetacan ! Potential temperature               [       K]
  2478)       real(kind=8), intent(in) :: shvcan   ! Vapour specific mass                [   kg/kg]

  2487)       ! top and the reference level.                                                       !
  2488)       !------------------------------------------------------------------------------------!
  2489)       thvbar = 5.d-1 * ( thetaref * (1.d0 + epim18 * shvref)                               &
  2490)                        + thetacan * (1.d0 + epim18 * shvcan) )

  2501)       reducedpress8 = (pres**rocp8 + pinc ) ** cpor8

  3174)       !write(unit=89,fmt='(60a1)') ('-',itn=1,60)
  3175)       !write(unit=89,fmt='(5(a,1x,f11.4,1x))')                                              &
  3176)       !   'Input values: Exner =',exner,'thil=',thil,'rliq=',1000.*rliq,'rice=',1000.*rice  &
  3177)       !           ,'t1stguess=',t1stguess
  3178)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3179)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3180) 
  3181)       !------------------------------------------------------------------------------------!
  3182)       !      If we have reached this point, we must use root-finding method.  For the      !
  3183)       ! Newton's 1st. guess, use t1stguess.                                                !
  3184)       !------------------------------------------------------------------------------------!
  3185)       tempz     = t1stguess
  3186)       fun       = theta_iceliq8(exner,tempz,rliq,rice)
  3187)       deriv     = dthetail_dt8(.true.,fun,pres,tempz,rliq,rice)
  3188)       fun       = fun - thil
  3189)       !------------------------------------------------------------------------------------!
  3190) 
  3191)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3192)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3193)       !write(unit=89,fmt='(a,1x,i5,1x,a,1x,l1,1x,a,1x,f11.4,2(1x,a,1x,es12.5))')            &
  3194)       !   'itn=',0,'bisection=',.false.,'tempz=',tempz-t00                                  &
  3195)       !           ,'fun=',fun,'deriv=',deriv
  3196)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3197)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3198)       if (fun == 0.d0) then
  3199)          thil2temp8 = tempz
  3200)          converged = .true.
  3201)          return
  3202)       else
  3203)          tempa = tempz
  3204)          funa  = fun
  3205)       end if
  3206) 
  3207)       !----- Enter loop: it will probably skip when the air is not saturated --------------!
  3208)       converged = .false.
  3209)       newloop: do itn=1,maxfpo/6
  3210)          if (abs(deriv) < toler8) exit newloop !----- Too dangerous, go to bisection ------!
  3211)          tempa = tempz
  3212)          funa  = fun
  3213)          tempz = tempa - fun/deriv
  3214)          fun   = theta_iceliq8(exner,tempz,rliq,rice)
  3215)          deriv = dthetail_dt8(.true.,fun,pres,tempz,rliq,rice)
  3216)          fun   = fun - thil
  3217)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3218)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3219)          !write(unit=89,fmt='(a,1x,i5,1x,a,1x,l1,1x,a,1x,f11.4,2(1x,a,1x,es12.5))')         &
  3220)          !   'itn=',itn,'bisection=',.false.,'tempz=',tempz-t00                             &
  3221)          !           ,'fun=',fun,'deriv=',deriv
  3222)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3223)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3224) 
  3225)          converged = abs(tempa-tempz) < toler8*tempz
  3226)          !----- Converged, happy with that, return the average b/w the 2 previous guesses -!
  3227)          if (fun == 0.d0) then
  3228)             thil2temp8 = tempz
  3229)             converged  = .true.
  3230)             return
  3231)          elseif(converged) then
  3232)             thil2temp8 = 5.d-1 * (tempa+tempz)
  3233)             return
  3234)          end if
  3235)       end do newloop
  3236)       !------------------------------------------------------------------------------------!
  3237) 
  3238) 
  3239)       !------------------------------------------------------------------------------------!
  3240)       !     If we have reached this point then Newton's method failed.  Use bisection      !
  3241)       ! instead.  For bisection, We need two guesses whose function evaluations have       !
  3242)       ! opposite sign.                                                                     !
  3243)       !------------------------------------------------------------------------------------!
  3244)       if (funa * fun < 0.d0) then
  3245)          !----- Guesses have opposite sign. -----------------------------------------------!
  3246)          funz  = fun
  3247)          zside = .true.
  3248)       else
  3249)          if (abs(fun-funa) < toler8 * tempa) then
  3250)             delta = 1.d2 * toler8 * tempa
  3251)          else
  3252)             delta = max(abs(funa * (tempz-tempa)/(fun-funa)),1.d2 * toler8 * tempa)
  3253)          end if

  3255)          zside = .false.
  3256)          zgssloop: do itb=1,maxfpo
  3257)             tempz = tempa + dble((-1)**itb * (itb+3)/2) * delta
  3258)             funz  = theta_iceliq8(exner,tempz,rliq,rice) - thil
  3259)             zside = funa*funz < 0.d0
  3260)             if (zside) exit zgssloop
  3261)          end do zgssloop
  3262)          if (.not. zside) then
  3263)             write (unit=*,fmt='(a)') ' No second guess for you...'
  3264)             write (unit=*,fmt='(2(a,1x,i14,1x))')    'itn  =',itn  ,'itb =',itb
  3265)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'thil =',thil ,'t1st=',t1stguess
  3266)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'exner=',exner,'pres=',pres
  3267)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'rliq =',rliq ,'rice=',rice
  3268)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'tempa=',tempa,'funa=',funa
  3269)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'tempz=',tempz,'funz=',funz
  3270)             write (unit=*,fmt='(1(a,1x,es14.7,1x))') 'delta=',delta
  3271)             call fatal_error('Failed finding the second guess for regula falsi'            &
  3272)                             ,'thil2temp8','therm_lib8.f90')
  3273)          end if
  3274)       end if
  3275) 
  3276) 
  3277)       bisloop: do itb=itn,maxfpo
  3278)          thil2temp8 =  (funz*tempa-funa*tempz)/(funz-funa)
  3279) 
  3280)          !---------------------------------------------------------------------------------!
  3281)          !     Now that we updated the guess, check whether they are really close. If so,  !
  3282)          ! it converged, I can use this as my guess.                                       !
  3283)          !---------------------------------------------------------------------------------!
  3284)          converged = abs(thil2temp8 - tempa) < toler8 * thil2temp8
  3285)          if (converged) exit bisloop
  3286) 
  3287)          !------ Finding the new function -------------------------------------------------!
  3288)          fun  = theta_iceliq8(exner,tempz,rliq,rice) - thil
  3289) 
  3290)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3291)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3292)          !write(unit=89,fmt='(a,1x,i5,1x,a,1x,l1,1x,6(1x,a,1x,f11.4))')         &
  3293)          !   'itn=',itb,'bisection=',.true.                                                 &
  3294)          !           ,'temp=',thil2temp-t00,'tempa=',tempa-t00,'tempz=',tempz-t00           &
  3295)          !           ,'fun=',fun,'funa=',funa,'funz=',funz
  3296)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3297)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3298) 
  3299)          !------ Defining my new interval based on the intermediate value theorem. --------!
  3300)          if (fun*funa < 0.d0 ) then
  3301)             tempz = thil2temp8
  3302)             funz  = fun
  3303)             !----- If we are updating zside again, modify aside (Illinois method) ---------!
  3304)             if (zside) funa = funa * 5.d-1
  3305)             !----- We just updated zside, setting zside to true. --------------------------!
  3306)             zside = .true.
  3307)          else
  3308)             tempa  = thil2temp8
  3309)             funa   = fun
  3310)             !----- If we are updating aside again, modify aside (Illinois method) ---------!
  3311)             if (.not. zside) funz = funz * 5.d-1
  3312)             !----- We just updated aside, setting aside to true. --------------------------!
  3313)             zside = .false.
  3314)          end if
  3315)       end do bisloop
  3316) 
  3317) 
  3318)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3319)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3320)       !write(unit=89,fmt='(60a1)') ('-',itn=1,60)
  3321)       !write(unit=89,fmt='(a)') ' '
  3322)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3323)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3324) 
  3325)       if (.not.converged) then
  3326)          write (unit=*,fmt='(a)') '-------------------------------------------------------'
  3327)          write (unit=*,fmt='(a)') ' Temperature finding didn''t converge!!!'
  3328)          write (unit=*,fmt='(a,1x,i5,1x,a)') ' I gave up, after',maxfpo,'iterations...'
  3329)          write (unit=*,fmt='(a)') ' '
  3330)          write (unit=*,fmt='(a)') ' Input values.'
  3331)          write (unit=*,fmt='(a)') ' '
  3332)          write (unit=*,fmt='(a,1x,f12.4)' ) 'theta_il        [     K] =',thil
  3333)          write (unit=*,fmt='(a,1x,f12.4)' ) 'Exner           [J/kg/K] =',exner
  3334)          write (unit=*,fmt='(a,1x,f12.4)' ) 'rliq            [  g/kg] =',1.d3*rliq
  3335)          write (unit=*,fmt='(a,1x,f12.4)' ) 'rice            [  g/kg] =',1.d3*rice
  3336)          write (unit=*,fmt='(a,1x,f12.4)' ) 't1stguess       [    �C] =',t1stguess-t008
  3337)          write (unit=*,fmt='(a)') ' '
  3338)          write (unit=*,fmt='(a)') ' Last iteration outcome (downdraft values).'
  3339)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tempa           [    �C] =',tempa-t008
  3340)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tempz           [    �C] =',tempz-t008

  3344)          write (unit=*,fmt='(a,1x,f12.4)' ) 'deriv           [  ----] =',deriv
  3345)          write (unit=*,fmt='(a,1x,es12.4)') 'toler8          [  ----] =',toler8
  3346)          write (unit=*,fmt='(a,1x,es12.4)') 'error           [  ----] ='                   &
  3347)                                            , abs(thil2temp8-tempa)/thil2temp8

  3349) 
  3350)          call fatal_error('Temperature didn''t converge, giving up!!!'                     &
  3351)                          ,'thil2temp8','therm_lib8.f90')
  3352)       end if
  3353) 
  3354)       return

  3366)    !     This function computes the partial derivative of temperature as a function of the !
  3367)    ! saturation mixing ratio [kg/kg], keeping pressure constant.  This is based on the     !
  3368)    ! ice-liquid potential temperature equation.                                            !
  3369)    !---------------------------------------------------------------------------------------!
  3370)    real(kind=8) function dtempdrs8(exner,thil,temp,rliq,rice,rconmin)
  3371)       use consts_coms, only : alvl38      & ! intent(in)
  3372)                             , alvi38      & ! intent(in)
  3373)                             , dcpvl8      & ! intent(in)
  3374)                             , dcpvi8      & ! intent(in)
  3375)                             , cpdry8      & ! intent(in)
  3376)                             , cpdryi8     & ! intent(in)
  3377)                             , ttripoli8   & ! intent(in)
  3378)                             , htripolii8  ! ! intent(in)
  3379) 
  3380)       implicit none
  3381)       !----- Arguments --------------------------------------------------------------------!
  3382)       real(kind=8), intent(in) :: exner   ! Exner function                        [ J/kg/K]
  3383)       real(kind=8), intent(in) :: thil    ! Ice-liquid potential temperature (*)  [      K]
  3384)       real(kind=8), intent(in) :: temp    ! Temperature                           [      K]
  3385)       real(kind=8), intent(in) :: rliq    ! Liquid mixing ratio                   [  kg/kg]
  3386)       real(kind=8), intent(in) :: rice    ! Ice mixing ratio                      [  kg/kg]
  3387)       real(kind=8), intent(in) :: rconmin ! Min. non-zero condensate mix. ratio   [  kg/kg]
  3388)       !------------------------------------------------------------------------------------!
  3389)       ! (*) Thil is not used in this formulation but it may be used should you opt by      !
  3390)       !     other ways to compute theta_il, so don't remove this argument.                 !
  3391)       !------------------------------------------------------------------------------------!
  3392)       !----- Local variables --------------------------------------------------------------!
  3393)       real(kind=8)             :: qq      ! Enthalpy -- latent heat               [   J/kg]
  3394)       real(kind=8)             :: qpt     ! d(qq)/dT * T                          [   J/kg]
  3395)       real(kind=8)             :: hh      ! Enthalpy -- sensible heat             [   J/kg]
  3396)       real(kind=8)             :: rcon    ! Condensate mixing ratio               [  kg/kg]
  3397)       real(kind=8)             :: til     ! Ice-liquid temperature                [      K]
  3398)       !------------------------------------------------------------------------------------!
  3399) 
  3400) 
  3401)       !----- Find the total hydrometeor mixing ratio. -------------------------------------!
  3402)       rcon  = rliq+rice
  3403)       !------------------------------------------------------------------------------------!
  3404) 
  3405) 
  3406)       !------------------------------------------------------------------------------------!
  3407)       !      If the amount of condensate is negligible, temperature does not depend on     !
  3408)       ! saturation mixing ratio.                                                           !
  3409)       !------------------------------------------------------------------------------------!
  3410)       if (rcon < rconmin) then
  3411)          dtempdrs8 = 0.d0
  3412)       else
  3413) 
  3414)          !---------------------------------------------------------------------------------!
  3415)          !     Find the enthalpy associated with latent heat and its derivative            !
  3416)          ! correction.  This is dependent on the thermodynamics used.                      !
  3417)          !---------------------------------------------------------------------------------!
  3418)          if (newthermo) then
  3419)             qq  = alvl8(temp) * rliq + alvi8(temp) * rice
  3420)             qpt = dcpvl8 * rliq + dcpvi8 * rice
  3421)          else
  3422)             qq  = alvl38 * rliq + alvi38 * rice
  3423)             qpt = 0.d0
  3424)          end if
  3425)          !---------------------------------------------------------------------------------!
  3426) 
  3427) 
  3428)          !----- Find the enthalpy associated with sensible heat. --------------------------!
  3429)          hh = cpdry8 * temp
  3430)          !---------------------------------------------------------------------------------!
  3431) 
  3432) 
  3433)          !---------------------------------------------------------------------------------!
  3434)          !    Decide how to compute, based on the thermodynamics method.                   !
  3435)          !---------------------------------------------------------------------------------!
  3436)          if (newthermo) then
  3437)             dtempdrs8 = - temp * qq / ( rcon * (hh + qq - qpt) )
  3438)          else
  3439)             til   = cpdryi8 * thil * exner
  3440)             !----- Decide how to compute, based on temperature. ---------------------------!
  3441)             if (temp > ttripoli8) then
  3442)                dtempdrs8 = - til * qq / ( rcon * cpdry8 * (2.*temp-til))
  3443)             else
  3444)                dtempdrs8 = - til * qq * htripolii8 / rcon
  3445)             end if
  3446)             !------------------------------------------------------------------------------!
  3447)          end if
  3448)          !---------------------------------------------------------------------------------!
  3449)       end if
  3450)       !------------------------------------------------------------------------------------!
  3451) 
  3452)       return
  3453)    end function dtempdrs8

  3559) 
  3560) 
  3561)       !------------------------------------------------------------------------------------!
  3562)       !     Saturation term.                                                               !
  3563)       !------------------------------------------------------------------------------------!
  3564)       esterm = rocp8 * tlcl * desdtlcl / eslcl
  3565)       !------------------------------------------------------------------------------------!
  3566) 
  3567) 
  3568)       !------------------------------------------------------------------------------------!
  3569)       !     Find the enthalpy terms.                                                       !
  3570)       !------------------------------------------------------------------------------------!
  3571)       hhlcl  = cpdry8 * tlcl
  3572)       qqlcl  = alvl8(tlcl) * rtot
  3573)       qptlcl = dcpvl8 * rtot * tlcl
  3574)       !------------------------------------------------------------------------------------!
  3575) 
  3576) 
  3577) 
  3578)       !------------------------------------------------------------------------------------!
  3579)       !      Derivative.                                                                   !
  3580)       !------------------------------------------------------------------------------------!
  3581)       dthetaeiv_dtlcl8 = theiv / tlcl * (1.d0 - esterm - (qqlcl - qptlcl) / hhlcl)
  3582)       !------------------------------------------------------------------------------------!
  3583) 
  3584)       return
  3585)    end function dthetaeiv_dtlcl8

  3596)    !    This function computes the saturation ice-vapour equivalent potential temperature  !
  3597)    ! from theta_il and the total mixing ratio (split into saturated vapour plus liquid and !
  3598)    ! ice. This is equivalent to the equivalent potential temperature considering also the  !
  3599)    ! effects of fusion/melting/sublimation, and it is done separatedly from the regular    !
  3600)    ! thetae_iv because it doesn't require iterations.                                      !
  3601)    !                                                                                       !
  3602)    !    References:                                                                        !
  3603)    !    Tripoli, J. T.; and Cotton, W.R., 1981: The use of ice-liquid water potential tem- !
  3604)    !        perature as a thermodynamic variable in deep atmospheric models. Mon. Wea.     !
  3605)    !        Rev., v. 109, 1094-1102. (TC81)                                                !
  3606)    !                                                                                       !
  3607)    !    Some algebra was needed to find this equation, essentially combining (TC81-26) and !
  3608)    ! (TC81-27), and the conservation of total water (TC81-16). It assumes that the divi-   !
  3609)    ! sion between the three phases is already taken care of.                               !
  3610)    !---------------------------------------------------------------------------------------!
  3611)    real(kind=8) function thetaeivs8(thil,temp,rsat,rliq,rice)
  3612)       use consts_coms, only : cpdry8    ! ! intent(in)
  3613)       implicit none
  3614)       !----- Arguments. -------------------------------------------------------------------!
  3615)       real(kind=8), intent(in) :: thil  ! Theta_il, ice-liquid water pot. temp.    [     K]
  3616)       real(kind=8), intent(in) :: temp  ! Temperature                              [     K]
  3617)       real(kind=8), intent(in) :: rsat  ! Saturation water vapour mixing ratio     [ kg/kg]
  3618)       real(kind=8), intent(in) :: rliq  ! Liquid water mixing ratio                [ kg/kg]
  3619)       real(kind=8), intent(in) :: rice  ! Ice mixing ratio                         [ kg/kg]
  3620)       !----- Local variables --------------------------------------------------------------!
  3621)       real(kind=8)             :: rtots ! Saturated mixing ratio                   [     K]
  3622)       !------------------------------------------------------------------------------------!
  3623) 
  3624) 
  3625)       !------ Find the total saturation mixing ratio. -------------------------------------!
  3626)       rtots = rsat + rliq + rice
  3627)       !------------------------------------------------------------------------------------!
  3628) 
  3629) 
  3630)       !------ Find the saturation equivalent potential temperature. -----------------------!
  3631)       thetaeivs8 = thil * exp ( alvl8(temp) * rtots / (cpdry8 * temp))
  3632)       !------------------------------------------------------------------------------------!
  3633) 
  3634)       return
  3635)    end function thetaeivs8

  3646)    !    This function computes the derivative of saturation ice-vapour equivalent          !
  3647)    ! potential temperature, based on the expression used to compute the saturation         !
  3648)    ! ice-vapour equivalent potential temperature (function thetaeivs).                     !
  3649)    !                                                                                       !
  3650)    !    IMPORTANT!!! This CANNOT BE USED to compute d(Thetae_iv)/d(T_LCL), because here    !
  3651)    !                 we assume that temperature and pressure are known and constants, and  !
  3652)    !                 that the mixing ratio is a function of temperature. In case you want  !
  3653)    !                 d(Thetae_iv)/d(T_LCL), use the dthetaeiv_dtlcl function instead.      !
  3654)    !---------------------------------------------------------------------------------------!
  3655)    real(kind=8) function dthetaeivs_dt8(theivs,temp,pres,rsat,useice)
  3656)       use consts_coms, only : cpdry8     & ! intent(in)
  3657)                             , dcpvl8     ! ! intent(in)
  3658)       implicit none
  3659)       !----- Required arguments. ----------------------------------------------------------!
  3660)       real(kind=8), intent(in)           :: theivs ! Sat. ice-vap. eq. pot. temp. [      K]
  3661)       real(kind=8), intent(in)           :: temp   ! Temperature                  [      K]
  3662)       real(kind=8), intent(in)           :: pres   ! Pressure                     [     Pa]
  3663)       real(kind=8), intent(in)           :: rsat   ! Saturation mixing ratio      [  kg/kg]
  3664)       !----- Optional arguments. ----------------------------------------------------------!
  3665)       logical     , intent(in), optional :: useice ! Flag for considering ice     [    T|F]
  3666)       !----- Local variables --------------------------------------------------------------!
  3667)       real(kind=8)                       :: drsdt  ! Sat. mixing ratio derivative [kg/kg/K]
  3668)       real(kind=8)                       :: hh     ! Enthalpy -- sensible         [   J/kg]
  3669)       real(kind=8)                       :: qqaux  ! Enthalpy -- sensible         [   J/kg]
  3670)       !------------------------------------------------------------------------------------!
  3671) 
  3672) 
  3673)       !----- Find the derivative of rs with temperature. ----------------------------------!
  3674)       if (present(useice)) then
  3675)          drsdt = rslifp8(pres,temp,useice)
  3676)       else
  3677)          drsdt = rslifp8(pres,temp)
  3678)       end if
  3679)       !------------------------------------------------------------------------------------!
  3680) 
  3681) 
  3682) 
  3683)       !------------------------------------------------------------------------------------!
  3684)       !      Find the enthalpy terms.                                                      !
  3685)       !------------------------------------------------------------------------------------!
  3686)       hh    = cpdry8 * temp
  3687)       qqaux = alvl8(temp) * (drsdt * temp - rsat) + dcpvl8 * rsat * temp
  3688)       !------------------------------------------------------------------------------------!
  3689) 
  3690) 
  3691)       !----- Find the derivative.  Depending on the temperature, use different eqn. -------!
  3692)       dthetaeivs_dt8 = theivs / temp * ( 1.d0 + qqaux / hh )
  3693)       !------------------------------------------------------------------------------------!
  3694) 
  3695)       return
  3696)    end function dthetaeivs_dt8

  3707)    !    This function finds the ice-liquid potential temperature from the ice-vapour equi- !
  3708)    ! valent potential temperature.                                                         !
  3709)    ! Important remarks:                                                                    !
  3710)    ! 1. If you don't want to use ice thermodynamics, simply force useice to be .false.     !
  3711)    !    Otherwise, the model will decide based on the LEVEL given by the user from their   !
  3712)    !    RAMSIN.                                                                            !
  3713)    ! 2. If rtot < rsat, then this will convert theta_e into theta, which can be thought as !
  3714)    !    a particular case.                                                                 !
  3715)    !---------------------------------------------------------------------------------------!
  3716)    real(kind=8) function thetaeiv2thil8(theiv,pres,rtot,useice)
  3717)       use consts_coms, only : ep8       & ! intent(in)
  3718)                             , cpdry8    & ! intent(in)
  3719)                             , p008      & ! intent(in)
  3720)                             , rocp8     & ! intent(in)
  3721)                             , t3ple8    & ! intent(in)
  3722)                             , t008      ! ! intent(in)
  3723)       implicit none
  3724)       !----- Required arguments. ----------------------------------------------------------!
  3725)       real(kind=8), intent(in)           :: theiv     ! Ice vap. equiv. pot. temp. [     K]
  3726)       real(kind=8), intent(in)           :: pres      ! Pressure                   [    Pa]
  3727)       real(kind=8), intent(in)           :: rtot      ! Total mixing ratio         [ kg/kg]
  3728)       !----- Optional arguments. ----------------------------------------------------------!
  3729)       logical     , intent(in), optional :: useice    ! May I use ice thermodyn.   [   T|F]
  3730)       !----- Local variables for iterative method -----------------------------------------!
  3731)       real(kind=8)                       :: pvap      ! Sat. vapour pressure
  3732)       real(kind=8)                       :: theta     ! Potential temperature
  3733)       real(kind=8)                       :: deriv     ! Function derivative
  3734)       real(kind=8)                       :: funnow    ! Function for which we seek a root.
  3735)       real(kind=8)                       :: funa      ! Smallest guess function
  3736)       real(kind=8)                       :: funz      ! Largest guess function
  3737)       real(kind=8)                       :: tlcla     ! Smallest guess (Newton: old guess)
  3738)       real(kind=8)                       :: tlclz     ! Largest guess (Newton: new guess)
  3739)       real(kind=8)                       :: tlcl      ! What will be the LCL temperature
  3740)       real(kind=8)                       :: es00      ! Defined as p00*rt/(epsilon + rt)
  3741)       real(kind=8)                       :: delta     ! Aux. variable (For 2nd guess).
  3742)       integer                            :: itn       ! Iteration counters
  3743)       integer                            :: itb       ! Iteration counters
  3744)       integer                            :: ii        ! Another counter
  3745)       logical                            :: converged ! Convergence handle
  3746)       logical                            :: zside     ! Side checker for Regula Falsi
  3747)       logical                            :: frozen    ! Will use ice thermodynamics
  3748)       !------------------------------------------------------------------------------------!
  3749) 
  3750) 
  3751) 
  3752)       !----- Fill the flag for ice thermodynamics so it will be present. ------------------!
  3753)       if (present(useice)) then
  3754)          frozen = useice
  3755)       else
  3756)          frozen = bulk_on
  3757)       end if
  3758)       !------------------------------------------------------------------------------------!
  3759) 
  3760) 
  3761) 
  3762)       !----- Find es00, which is a constant. ----------------------------------------------!
  3763)       es00 = p008 * rtot / (ep8 + rtot)
  3764)       !------------------------------------------------------------------------------------!
  3765) 
  3766) 
  3767)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3768)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3769)       !write (unit=36,fmt='(a)') '----------------------------------------------------------'
  3770)       !write (unit=36,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x))')                                  &
  3771)       !   'INPUT : it=',-1,'theiv=',theiv,'pres=',0.01*pres,'rtot=',rtot*1000.
  3772)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3773)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3774) 
  3775)       !------------------------------------------------------------------------------------!
  3776)       !     The 1st. guess, we assume we are lucky and right at the LCL.                   !
  3777)       !------------------------------------------------------------------------------------!
  3778)       pvap      = pres * rtot / (ep8 + rtot)
  3779)       tlclz     = tslif8(pvap,frozen)
  3780)       theta     = tlclz * (es00 / pvap) ** rocp8
  3781)       funnow    = thetaeivs8(theta,tlclz,rtot,0.d0,0.d0)
  3782)       deriv     = dthetaeiv_dtlcl8(funnow,tlclz,rtot,pvap,frozen)
  3783)       funnow    = funnow - theiv
  3784)       !------------------------------------------------------------------------------------!
  3785) 
  3786)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3787)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3788)       !write (unit=36,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')                &
  3789)       !   'NEWTON: it=',0,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funnow,'deriv=',deriv
  3790)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3791)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3792) 
  3793)       !----- Put something in tlcla in case we never loop through Newton's method. --------!
  3794)       tlcla     = tlclz
  3795)       funa      = funnow
  3796)       converged = .false.
  3797)       !------------------------------------------------------------------------------------!
  3798) 
  3799)       !----- Looping: Newton's iterative method -------------------------------------------!
  3800)       newloop: do itn=1,maxfpo/6
  3801)          if (abs(deriv) < toler8) exit newloop !----- Too dangerous, skip to bisection ----!
  3802)          !----- Update guesses. -----------------------------------------------------------!
  3803)          tlcla   = tlclz
  3804)          funa    = funnow
  3805)          tlclz   = tlcla - funnow/deriv
  3806)          !---------------------------------------------------------------------------------!
  3807) 
  3808) 
  3809)          !----- Update the function evaluation and its derivative. ------------------------!
  3810)          pvap    = eslif8(tlclz,frozen)
  3811)          theta   = tlclz * (es00/pvap)**rocp8
  3812)          funnow  = thetaeivs8(theta,tlclz,rtot,0.d0,0.d0)
  3813)          deriv   = dthetaeiv_dtlcl8(funnow,tlclz,rtot,pvap,frozen)
  3814)          funnow  = funnow - theiv
  3815)          !---------------------------------------------------------------------------------!
  3816) 
  3817)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3818)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3819)          !write (unit=36,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')             &
  3820)          !   'NEWTON: it=',itn,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funnow           &
  3821)          !          ,'deriv=',deriv
  3822)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3823)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3824) 
  3825)          converged = abs(tlcla-tlclz) < toler8 * tlclz
  3826)          if (funnow == 0.d0) then
  3827)             tlcl = tlclz
  3828)             funz = funnow
  3829)             converged = .true.
  3830)             exit newloop
  3831)          elseif (converged) then
  3832)             tlcl = 5.d-1*(tlcla+tlclz)
  3833)             funz = funnow
  3834)             exit newloop
  3835)          end if
  3836)       end do newloop
  3837) 
  3838)       !------------------------------------------------------------------------------------!
  3839)       !     If I reached this point then it's because Newton's method failed. Using bisec- !
  3840)       ! tion instead.                                                                      !
  3841)       !------------------------------------------------------------------------------------!
  3842)       if (.not. converged) then
  3843)          !----- Set funz, and check whether funa and funz already have opposite sign. -----!
  3844)          funz = funnow
  3845)          zside=.true.
  3846)          if (funa*funnow > 0.d0) then
  3847)             !----- We fix funa, and try a funz that will work as 2nd guess ----------------!
  3848)             if (abs(funz-funa) < toler8*tlcla) then
  3849)                delta = 1.d2*toler8*tlcla
  3850)             else
  3851)                delta = max(abs(funa*(tlclz-tlcla)/(funz-funa)),1.d2*toler8*tlcla)
  3852)             end if
  3853)             tlclz = tlcla + delta
  3854) 
  3855)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3856)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3857)             !write (unit=36,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x),3(a,1x,es11.4,1x))')          &
  3858)             !   '2NGGSS: tt=',0,'tlclz=',tlclz-t00,'tlcla=',tlcla-t00,'pvap=',0.01*pvap     &
  3859)             !           ,'funa=',funa,'funz=',funnow,'delta=',delta
  3860)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3861)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3862)             zside = .false.
  3863)             zgssloop: do itb=1,maxfpo
  3864)                !----- So the first time tlclz = tlcla - 2*delta ---------------------------!
  3865)                tlclz = tlcla + dble((-1)**itb * (itb+3)/2) * delta
  3866)                pvap  = eslif8(tlclz,frozen)
  3867)                theta = tlclz * (es00/pvap)**rocp8
  3868)                funz  = thetaeivs8(theta,tlclz,rtot,0.d0,0.d0) - theiv
  3869) 
  3870)                !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3871)                !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3872)                !write (unit=36,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')       &
  3873)                !   '2NGGSS: tt=',itb,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funz       &
  3874)                !           ,'delta=',delta
  3875)                !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3876)                !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3877) 
  3878)                zside = funa*funz < 0.d0
  3879)                if (zside) exit zgssloop
  3880)             end do zgssloop
  3881)             if (.not. zside) then
  3882)                write (unit=*,fmt='(a)') ' No second guess for you...'
  3883)                write (unit=*,fmt='(2(a,1x,i14,1x))')    'itn   =',itn   ,'itb   =',itb
  3884)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'theiv =',theiv ,'rtot  =',rtot
  3885)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'pres  =',pres  ,'pvap  =',pvap
  3886)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'theta =',theta ,'delta =',delta
  3887)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'tlcla =',tlcla ,'funa  =',funa
  3888)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'tlclz =',tlclz ,'funz  =',funz
  3889)                call fatal_error('Failed finding the second guess for regula falsi'         &
  3890)                                ,'thetaeiv2thil8','therm_lib8.f90')
  3891)             end if
  3892)          end if
  3893)          !---- Continue iterative method. -------------------------------------------------!
  3894)          fpoloop: do itb=itn+1,maxfpo
  3895) 
  3896)             !----- Update the guess. ------------------------------------------------------!
  3897)             tlcl   = (funz*tlcla-funa*tlclz)/(funz-funa)
  3898) 
  3899)             !----- Updating function evaluation -------------------------------------------!
  3900)             pvap   = eslif8(tlcl,frozen)
  3901)             theta  = tlcl * (es00/pvap)**rocp8
  3902)             funnow = thetaeivs8(theta,tlcl,rtot,0.d0,0.d0) - theiv
  3903) 
  3904)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3905)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3906)             !write (unit=36,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),1(a,1x,es11.4,1x))')          &
  3907)             !   'REGFAL: it=',itb,'tlcl =',tlcl -t00,'pvap=',0.01*pvap,'fun=',funnow
  3908)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3909)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3910) 
  3911)             !------------------------------------------------------------------------------!
  3912)             !    Check for convergence. If it did, return, we found the solution.          !
  3913)             ! Otherwise, constrain the guesses.                                            !
  3914)             !------------------------------------------------------------------------------!
  3915)             converged = abs(tlcl-tlcla) < toler8 * tlcl
  3916)             if (converged) then
  3917)                exit fpoloop
  3918)             elseif (funnow*funa < 0.d0) then
  3919)                tlclz = tlcl
  3920)                funz  = funnow
  3921)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  3922)                if (zside) funa=funa * 5.d-1
  3923)                !----- We just updated zside, setting zside to true. -----------------------!
  3924)                zside = .true.
  3925)             else
  3926)                tlcla = tlcl
  3927)                funa  = funnow
  3928)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  3929)                if (.not.zside) funz = funz * 5.d-1
  3930)                !----- We just updated aside, setting zside to false -----------------------!
  3931)                zside = .false.
  3932)             end if
  3933)          end do fpoloop
  3934)       end if
  3935) 
  3936)       if (converged) then
  3937)          thetaeiv2thil8  = theiv * exp (- alvl8(tlcl) * rtot / (cpdry8 * tlcl) )
  3938)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3939)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3940)          !write (unit=36,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')             &
  3941)          !   'ANSWER: itb=',itn,'tlcl=',tlcl-t00,'eslcl=',0.01*pvap                         &

  3987)    !    This subroutine converts saturated ice-vapour equivalent potential temperature     !
  3988)    ! into temperature, given pressure in Pa, and theta_es in Kelvin. It also returns the   !
  3989)    ! potential temperature in Kelvin, and saturation vapour mixing ratio in kg/kg. As      !
  3990)    ! usual,  we seek T using Newton's method as a starting point, and if it fails, we fall !
  3991)    ! back to the modified regula falsi (Illinois method).                                  !
  3992)    !                                                                                       !
  3993)    ! OBS: In case you want to ignore ice, send useice as false. The default is to consider !
  3994)    !      when level >= 3 and to ignore otherwise.                                         !
  3995)    !---------------------------------------------------------------------------------------!
  3996)    subroutine thetaeivs2temp8(theivs,pres,theta,temp,rsat,useice)
  3997)       use consts_coms, only : cpdry8     & ! intent(in)
  3998)                             , ep8        & ! intent(in)
  3999)                             , p008       & ! intent(in)
  4000)                             , rocp8      & ! intent(in)
  4001)                             , t008       ! ! intent(in)
  4002)       implicit none
  4003)       !----- Arguments --------------------------------------------------------------------!
  4004)       real(kind=8), intent(in)            :: theivs     ! Sat. thetae_iv          [      K]
  4005)       real(kind=8), intent(in)            :: pres       ! Pressure                [     Pa]
  4006)       real(kind=8), intent(out)           :: theta      ! Potential temperature   [      K]
  4007)       real(kind=8), intent(out)           :: temp       ! Temperature             [      K]
  4008)       real(kind=8), intent(out)           :: rsat       ! Saturation mixing ratio [  kg/kg]
  4009)       logical     , intent(in) , optional :: useice     ! May use ice thermodyn.  [    T|F]
  4010)       !----- Local variables, with other thermodynamic properties -------------------------!
  4011)       real(kind=8)                        :: exnernormi ! 1./ (Norm. Exner func.) [    ---]
  4012)       logical                             :: frozen     ! Will use ice thermodyn. [    T|F]
  4013)       !----- Local variables for iterative method -----------------------------------------!
  4014)       real(kind=8)                        :: deriv      ! Function derivative
  4015)       real(kind=8)                        :: funnow     ! Current function evaluation
  4016)       real(kind=8)                        :: funa       ! Smallest guess function
  4017)       real(kind=8)                        :: funz       ! Largest guess function
  4018)       real(kind=8)                        :: tempa      ! Smallest guess (Newton: previous)
  4019)       real(kind=8)                        :: tempz      ! Largest guess (Newton: new)
  4020)       real(kind=8)                        :: delta      ! Aux. variable for 2nd guess.
  4021)       integer                             :: itn        ! Iteration counter
  4022)       integer                             :: itb        ! Iteration counter
  4023)       logical                             :: converged  ! Convergence handle
  4024)       logical                             :: zside      ! Flag for side check.
  4025)       !------------------------------------------------------------------------------------!
  4026) 
  4027) 
  4028)       !----- Set up the ice check, in case useice is not present. -------------------------!
  4029)       if (present(useice)) then
  4030)          frozen = useice
  4031)       else
  4032)          frozen = bulk_on
  4033)       end if
  4034)       !------------------------------------------------------------------------------------!
  4035) 
  4036) 
  4037) 
  4038)       !----- Finding the inverse of normalised Exner, which is constant in this routine ---!
  4039)       exnernormi = (p008 /pres) ** rocp8
  4040)       !------------------------------------------------------------------------------------!
  4041) 
  4042) 
  4043) 
  4044)       !------------------------------------------------------------------------------------!
  4045)       !     The 1st. guess, no idea, guess 0�C.                                            !
  4046)       !------------------------------------------------------------------------------------!
  4047)       tempz     = t008
  4048)       theta     = tempz * exnernormi
  4049)       rsat      = rslif8(pres,tempz,frozen)
  4050)       funnow    = thetaeivs8(theta,tempz,rsat,0.d0,0.d0)
  4051)       deriv     = dthetaeivs_dt8(funnow,tempz,pres,rsat,frozen)
  4052)       funnow    = funnow - theivs
  4053)       !------------------------------------------------------------------------------------!
  4054) 
  4055) 
  4056)       !----- Copy here just in case Newton is aborted at the 1st guess. -------------------!
  4057)       tempa     = tempz
  4058)       funa      = funnow
  4059)       !------------------------------------------------------------------------------------!
  4060) 
  4061)       converged = .false.
  4062)       !----- Newton's method loop. --------------------------------------------------------!
  4063)       newloop: do itn=1,maxfpo/6
  4064)          if (abs(deriv) < toler8) exit newloop !----- Too dangerous, skip to bisection ----!
  4065)          !----- Updating guesses ----------------------------------------------------------!
  4066)          tempa   = tempz
  4067)          funa    = funnow
  4068) 
  4069)          tempz   = tempa - funnow/deriv
  4070)          theta   = tempz * exnernormi
  4071)          rsat    = rslif8(pres,tempz,frozen)
  4072)          funnow  = thetaeivs8(theta,tempz,rsat,0.d0,0.d0)
  4073)          deriv   = dthetaeivs_dt8(funnow,tempz,pres,rsat,frozen)
  4074)          funnow  = funnow - theivs
  4075) 
  4076)          converged = abs(tempa-tempz) < toler8*tempz
  4077)          if (funnow == 0.d0) then
  4078)             converged =.true.
  4079)             temp = tempz
  4080)             exit newloop
  4081)          elseif (converged) then
  4082)             temp = 5.d-1*(tempa+tempz)
  4083)             exit newloop
  4084)          end if
  4085)       end do newloop
  4086) 
  4087)       !------------------------------------------------------------------------------------!
  4088)       !     If we have reached this point then it's because Newton's method failed.  Use   !
  4089)       ! bisection instead.                                                                 !
  4090)       !------------------------------------------------------------------------------------!
  4091)       if (.not. converged) then
  4092)          !----- Set funz, and check whether funa and funz already have opposite sign. -----!
  4093)          funz  = funnow
  4094)          zside = .false.
  4095)          !---------------------------------------------------------------------------------!
  4096) 
  4097)          if (funa*funnow > 0.d0) then
  4098)             !----- We fix funa, and try a funz that will work as 2nd guess ----------------!
  4099)             if (abs(funz-funa) < toler8*tempa) then
  4100)                delta = 1.d2*toler8*tempa
  4101)             else
  4102)                delta = max(abs(funa*(tempz-tempa)/(funz-funa)),1.d2*toler8*tempa)
  4103)             end if
  4104)             !------------------------------------------------------------------------------!
  4105) 
  4106)             tempz = tempa + delta
  4107)             zgssloop: do itb=1,maxfpo
  4108)                !----- So this will be +1 -1 +2 -2 etc. ------------------------------------!
  4109)                tempz = tempz + dble((-1)**itb * (itb+3)/2) * delta
  4110)                theta = tempz * exnernormi
  4111)                rsat  = rslif8(pres,tempz,frozen)
  4112)                funz  = thetaeivs8(theta,tempz,rsat,0.d0,0.d0) - theivs
  4113)                zside = funa*funz < 0.d0
  4114)                if (zside) exit zgssloop
  4115)             end do zgssloop
  4116)             if (.not. zside) then
  4117)                call fatal_error('Failed finding the second guess for regula falsi'         &
  4118)                                ,'thetaes2temp8','therm_lib8.f90')
  4119)             end if
  4120)          end if
  4121)          !---- Continue iterative method --------------------------------------------------!
  4122)          fpoloop: do itb=itn+1,maxfpo
  4123)             if (abs(funz-funa) < toler8*tempa) then
  4124)                temp   = 5.d-1*(tempa+tempz)
  4125)             else
  4126)                temp   = (funz*tempa-funa*tempz)/(funz-funa)
  4127)             end if
  4128)             theta  = temp * exnernormi
  4129)             rsat   = rslif8(pres,temp,frozen)
  4130)             funnow = thetaeivs8(theta,temp,rsat,0.d0,0.d0) - theivs
  4131) 
  4132)             !------------------------------------------------------------------------------!
  4133)             !    Checking for convergence. If it did, return, we found the solution.       !
  4134)             ! Otherwise, constrain the guesses.                                            !
  4135)             !------------------------------------------------------------------------------!
  4136)             converged = abs(temp-tempa) < toler8*temp
  4137)             if (converged) then
  4138)                exit fpoloop
  4139)             elseif (funnow*funa < 0.d0) then
  4140)                tempz = temp
  4141)                funz  = funnow
  4142)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  4143)                if (zside) funa=funa * 5.d-1
  4144)                !----- We just updated zside, setting zside to true. -----------------------!
  4145)                zside = .true.
  4146)             else
  4147)                tempa = temp
  4148)                funa  = funnow
  4149)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  4150)                if (.not. zside) funz = funz * 5.d-1
  4151)                !----- We just updated aside, setting zside to false -----------------------!
  4152)                zside = .false.
  4153)             end if
  4154)          end do fpoloop
  4155)       end if
  4156) 
  4157)       if (converged) then
  4158)          !----- Compute theta and rsat with temp just for consistency ---------------------!
  4159)          theta = temp * exnernormi
  4160)          rsat  = rslif8(pres,temp,frozen)
  4161)       else
  4162)          call fatal_error('Temperature didn''t converge, I gave up!'                       &
  4163)                          ,'thetaes2temp8','therm_lib8.f90')
  4164)       end if
  4165) 
  4166)       return
  4167)    end subroutine thetaeivs2temp8

  4178)    !    This subroutine finds the lifting condensation level given the ice-liquid          !
  4179)    ! potential temperature in Kelvin, temperature in Kelvin, the pressure in Pascal, and   !
  4180)    ! the mixing ratio in kg/kg. The output will give the LCL temperature and pressure, and !
  4181)    ! the thickness of the layer between the initial point and the LCL.                     !
  4182)    !                                                                                       !
  4183)    !    References:                                                                        !
  4184)    !    Tripoli, J. T.; and Cotton, W.R., 1981: The use of ice-liquid water potential      !
  4185)    !        temperature as a thermodynamic variable in deep atmospheric models. Mon. Wea.  !
  4186)    !        Rev., v. 109, 1094-1102. (TC81)                                                !
  4187)    !    Bolton, D., 1980: The computation of the equivalent potential temperature. Mon.    !
  4188)    !        Wea. Rev., v. 108, 1046-1053. (BO80)                                           !
  4189)    !                                                                                       !
  4190)    !    Some algebra was needed to find this equation, essentially combining (TC81-26) and !
  4191)    ! (TC81-27), and the conservation of total water (TC81-16). It assumes that the divi-   !
  4192)    ! sion between the three phases is already taken care of.                               !
  4193)    !    Iterative procedure is needed, and here we iterate looking for T(LCL). Theta_il    !
  4194)    ! can be rewritten in terms of T(LCL) only, and once we know this thetae_iv becomes     !
  4195)    ! straightforward. T(LCL) will be found using Newton's method, and in the unlikely      !
  4196)    ! event it fails,we will fall back to the modified regula falsi (Illinois method).      !
  4197)    !                                                                                       !
  4198)    ! Important remarks:                                                                    !
  4199)    ! 1. TLCL and PLCL are the actual TLCL and PLCL, so in case condensation exists, they   !
  4200)    !    will be larger than the actual temperature and pressure (because one would go down !
  4201)    !    to reach the equilibrium);                                                         !
  4202)    ! 2. DZLCL WILL BE SET TO ZERO in case the LCL is beneath the starting level. So in     !
  4203)    !    case you want to force TLCL <= TEMP and PLCL <= PRES, you can use this variable    !
  4204)    !    to run the saturation check afterwards. DON'T CHANGE PLCL and TLCL here, they will !
  4205)    !    be used for conversions between theta_il and thetae_iv as they are defined here.   !
  4206)    ! 3. In case you don't want ice, simply pass useice=.false.. Otherwise let the model    !
  4207)    !    decide by itself based on the LEVEL variable.                                      !
  4208)    !---------------------------------------------------------------------------------------!
  4209)    subroutine lcl_il8(thil,pres,temp,rtot,rvap,tlcl,plcl,dzlcl,useice)
  4210)       use consts_coms, only : cpog8     & ! intent(in)
  4211)                             , ep8       & ! intent(in)
  4212)                             , p008      & ! intent(in)
  4213)                             , rocp8     & ! intent(in)
  4214)                             , t3ple8    & ! intent(in)
  4215)                             , t008      ! ! intent(in)
  4216)       implicit none
  4217)       !----- Required arguments. ----------------------------------------------------------!
  4218)       real(kind=8), intent(in)            :: thil      ! Ice liquid pot. temp. (*)[      K]
  4219)       real(kind=8), intent(in)            :: pres      ! Pressure                 [     Pa]
  4220)       real(kind=8), intent(in)            :: temp      ! Temperature              [      K]
  4221)       real(kind=8), intent(in)            :: rtot      ! Total mixing ratio       [  kg/kg]
  4222)       real(kind=8), intent(in)            :: rvap      ! Vapour mixing ratio      [  kg/kg]
  4223)       real(kind=8), intent(out)           :: tlcl      ! LCL temperature          [      K]
  4224)       real(kind=8), intent(out)           :: plcl      ! LCL pressure             [     Pa]
  4225)       real(kind=8), intent(out)           :: dzlcl     ! Sub-LCL layer thickness  [      m]
  4226)       !------------------------------------------------------------------------------------!
  4227)       ! (*) This is the most general variable. Thil is exactly theta for no condensation   !
  4228)       !     condition, and it is the liquid potential temperature if no ice is present.    !
  4229)       !------------------------------------------------------------------------------------!
  4230)       !----- Optional arguments. ----------------------------------------------------------!
  4231)       logical     , intent(in) , optional :: useice    ! May use ice thermodyn.?  [    T|F]
  4232)       !----- Local variables. -------------------------------------------------------------!
  4233)       real(kind=8)                        :: pvap      ! Sat. vapour pressure
  4234)       real(kind=8)                        :: deriv     ! Function derivative
  4235)       real(kind=8)                        :: funnow    ! Current function evaluation
  4236)       real(kind=8)                        :: funa      ! Smallest guess function
  4237)       real(kind=8)                        :: funz      ! Largest  guess function
  4238)       real(kind=8)                        :: tlcla     ! Smallest guess (Newton: previous)
  4239)       real(kind=8)                        :: tlclz     ! Largest guess (Newton: new)
  4240)       real(kind=8)                        :: es00      ! Defined as p00*rt/(epsilon + rt)
  4241)       real(kind=8)                        :: delta     ! Aux. variable for bisection
  4242)       integer                             :: itn       ! Iteration counter
  4243)       integer                             :: itb       ! Iteration counter
  4244)       logical                             :: converged ! Convergence flag
  4245)       logical                             :: zside     ! Flag to check sides
  4246)       logical                             :: frozen    ! Will use ice thermodyn.  [    T|F]
  4247)       !------------------------------------------------------------------------------------!
  4248) 
  4249) 
  4250)       !------------------------------------------------------------------------------------!
  4251)       !     Check whether ice thermodynamics is the way to go.                             !
  4252)       !------------------------------------------------------------------------------------!
  4253)       if (present(useice)) then
  4254)          frozen = useice
  4255)       else
  4256)          frozen = bulk_on
  4257)       end if
  4258)       !------------------------------------------------------------------------------------!
  4259) 
  4260)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4261)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4262)       !write (unit=21,fmt='(a)') '----------------------------------------------------------'
  4263)       !write (unit=21,fmt='(a,1x,i5,1x,5(a,1x,f11.4,1x))')                                  &
  4264)       !   'INPUT : it=',-1,'thil=',thil,'pres=',0.01*pres,'temp=',temp-t00                  &
  4265)       !        ,'rvap=',rvap*1000.,'rtot=',rtot*1000.
  4266)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4267)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4268) 
  4269) 
  4270)       !----- Find es00, which is a constant. ----------------------------------------------!
  4271)       es00 = p008 * rtot / (ep8 + rtot)
  4272)       !------------------------------------------------------------------------------------!
  4273) 
  4274) 
  4275) 
  4276)       !------------------------------------------------------------------------------------!
  4277)       !     The 1st. guess, use equation 21 from Bolton (1980). For this we'll need the    !
  4278)       ! vapour pressure.                                                                   !
  4279)       !------------------------------------------------------------------------------------!
  4280)       pvap      = pres * rvap / (ep8 + rvap)
  4281)       tlclz     = 5.5d1 + 2.840d3 / (3.5d0 * log(temp) - log(1.d-2*pvap) - 4.805d0)
  4282)       pvap      = eslif8(tlclz,frozen)
  4283)       funnow    = tlclz * (es00/pvap)**rocp8 - thil
  4284)       deriv     = (funnow+thil)*(1.d0/tlclz - rocp8*eslifp8(tlclz,frozen)/pvap)
  4285)       !------------------------------------------------------------------------------------!
  4286) 
  4287)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4288)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4289)       !write (unit=21,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')                &
  4290)       !   'NEWTON: it=',0,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funnow,'deriv=',deriv
  4291)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4292)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4293)       tlcla     = tlclz
  4294)       funa      = funnow
  4295) 
  4296) 
  4297)       !------------------------------------------------------------------------------------!
  4298)       !      First loop: Newton's method.                                                  !
  4299)       !------------------------------------------------------------------------------------!
  4300)       newloop: do itn=1,maxfpo/6
  4301)          !----- If derivative is too small, skip Newton's and try bisection instead. ------!
  4302)          if (abs(deriv) < toler8) exit newloop
  4303)          !---------------------------------------------------------------------------------!
  4304) 
  4305) 
  4306)          !----- Otherwise, update guesses. ------------------------------------------------!
  4307)          tlcla  = tlclz
  4308)          funa   = funnow
  4309)          tlclz  = tlcla - funnow/deriv
  4310)          pvap   = eslif8(tlclz,frozen)
  4311)          funnow = tlclz * (es00/pvap)**rocp8 - thil
  4312)          deriv  = (funnow+thil)*(1.d0/tlclz - rocp8*eslifp8(tlclz,frozen)/pvap)
  4313) 
  4314)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4315)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4316)          !write (unit=21,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')             &
  4317)          !   'NEWTON: it=',itn,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funnow           &
  4318)          !          ,'deriv=',deriv
  4319)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4320)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4321) 
  4322)          !---------------------------------------------------------------------------------!
  4323)          !      Check for convergence.                                                     !
  4324)          !---------------------------------------------------------------------------------!
  4325)          converged = abs(tlcla-tlclz) < toler8*tlclz
  4326)          if (converged) then
  4327)             !----- Guesses are almost identical, average them. ----------------------------!
  4328)             tlcl = 5.d-1*(tlcla+tlclz)
  4329)             funz = funnow
  4330)             exit newloop
  4331)             !------------------------------------------------------------------------------!
  4332)          elseif (funnow == 0.d0) then
  4333)             !----- We've hit the answer by luck, copy the answer. -------------------------!
  4334)             tlcl = tlclz
  4335)             funz = funnow
  4336)             converged = .true.
  4337)             exit newloop
  4338)             !------------------------------------------------------------------------------!
  4339)          end if
  4340)       end do newloop
  4341)       !------------------------------------------------------------------------------------!
  4342) 
  4343) 
  4344) 
  4345)       !------------------------------------------------------------------------------------!
  4346)       !      Check whether Newton's method has converged.                                  !
  4347)       !------------------------------------------------------------------------------------!
  4348)       if (.not. converged) then
  4349)          !---------------------------------------------------------------------------------!
  4350)          !     Newton's method has failed.  We use regula falsi instead.  First, we must   !
  4351)          ! find two guesses whose function evaluations have opposite signs.                !
  4352)          !---------------------------------------------------------------------------------!
  4353)          if (funa*funnow < 0.d0 ) then
  4354)             !----- We already have two good guesses. --------------------------------------!
  4355)             funz  = funnow
  4356)             zside = .true.
  4357)             !------------------------------------------------------------------------------!
  4358)          else
  4359)             !------------------------------------------------------------------------------!
  4360)             !     We need to find another guess with opposite sign.                        !
  4361)             !------------------------------------------------------------------------------!
  4362) 
  4363)             !----- We fix funa, and try a funz that will work as 2nd guess ----------------!
  4364)             if (abs(funnow-funa) < toler8*tlcla) then
  4365)                delta = 1.d2*toler8*tlcla
  4366)             else
  4367)                delta = max(abs(funa*(tlclz-tlcla)/(funnow-funa)),1.d2*toler8*tlcla)
  4368)             end if
  4369)             tlclz = tlcla + delta
  4370)             !------------------------------------------------------------------------------!
  4371) 
  4372)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4373)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4374)             !write (unit=21,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x),3(a,1x,es11.4,1x))')          &
  4375)             !   '2NGGSS: tt=',0,'tlclz=',tlclz-t00,'tlcla=',tlcla-t00,'pvap=',0.01*pvap     &
  4376)             !           ,'funa=',funa,'funz=',funnow,'delta=',delta
  4377)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4378)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4379)             zside = .false.
  4380)             zgssloop: do itb=1,maxfpo
  4381) 
  4382)                !----- So the first time tlclz = tlcla - 2*delta ---------------------------!
  4383)                tlclz = tlcla + dble((-1)**itb * (itb+3)/2) * delta
  4384)                pvap  = eslif8(tlclz,frozen)
  4385)                funz  = tlclz * (es00/pvap)**rocp8 - thil
  4386)                !---------------------------------------------------------------------------!
  4387) 
  4388) 
  4389)                !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4390)                !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4391)                !write (unit=21,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')       &
  4392)                !   '2NGGSS: tt=',itb,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funz       &
  4393)                !           ,'delta=',delta
  4394)                !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4395)                !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4396)                zside = funa*funz < 0.d0
  4397)                if (zside) exit zgssloop
  4398)             end do zgssloop
  4399)             if (.not. zside) then
  4400)                write (unit=*,fmt='(a)') ' ====== No second guess for you... ======'
  4401)                write (unit=*,fmt='(a)') ' + INPUT variables: '
  4402)                write (unit=*,fmt='(a,1x,es14.7)') 'THIL =',thil
  4403)                write (unit=*,fmt='(a,1x,es14.7)') 'TEMP =',temp
  4404)                write (unit=*,fmt='(a,1x,es14.7)') 'PRES =',pres
  4405)                write (unit=*,fmt='(a,1x,es14.7)') 'RTOT =',rtot
  4406)                write (unit=*,fmt='(a,1x,es14.7)') 'RVAP =',rvap
  4407)                write (unit=*,fmt='(a)') ' ============ Failed guess... ==========='
  4408)                write (unit=*,fmt='(2(a,1x,es14.7))') 'TLCLA =',tlcla,'FUNA =',funa
  4409)                write (unit=*,fmt='(2(a,1x,es14.7))') 'TLCLZ =',tlclz,'FUNC =',funz
  4410)                write (unit=*,fmt='(2(a,1x,es14.7))') 'DELTA =',delta,'FUNN =',funnow
  4411)                call fatal_error('Failed finding the second guess for regula falsi'         &
  4412)                                ,'lcl_il8','therm_lib8.f90')
  4413)             end if
  4414)          end if
  4415)          !---------------------------------------------------------------------------------!
  4416) 
  4417) 
  4418)          !---------------------------------------------------------------------------------!
  4419)          !      We have the guesses, solve the regula falsi method.                        !
  4420)          !---------------------------------------------------------------------------------!
  4421)          fpoloop: do itb=itn+1,maxfpo
  4422)             !----- Update guess and function evaluation. ----------------------------------!
  4423)             tlcl   = (funz*tlcla-funa*tlclz)/(funz-funa)
  4424)             pvap   = eslif8(tlcl,frozen)
  4425)             funnow = tlcl * (es00/pvap)**rocp8 - thil
  4426)             !------------------------------------------------------------------------------!
  4427) 
  4428)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4429)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4430)             !write (unit=21,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),1(a,1x,es11.4,1x))')          &
  4431)             !   'REGFAL: it=',itb,'tlcl =',tlcl -t00,'pvap=',0.01*pvap,'fun=',funnow
  4432)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4433)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4434) 
  4435)             !------------------------------------------------------------------------------!
  4436)             !    Check for convergence.  If it did, return, we found the solution.         !
  4437)             ! Otherwise, we update one of the guesses.                                     !
  4438)             !------------------------------------------------------------------------------!
  4439)             converged = abs(tlcl-tlcla) < toler8*tlcl .and.  abs(tlcl-tlclz) < toler8*tlcl
  4440)             if (funnow == 0.d0 .or. converged) then
  4441)                converged = .true.
  4442)                exit fpoloop
  4443)             elseif (funnow*funa < 0.d0) then
  4444)                tlclz = tlcl
  4445)                funz  = funnow
  4446)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  4447)                if (zside) funa=funa * 5.d-1
  4448)                !---------------------------------------------------------------------------!
  4449) 
  4450) 
  4451)                !----- We have just updated zside, sett zside to true. ---------------------!
  4452)                zside = .true.
  4453)                !---------------------------------------------------------------------------!
  4454)             else
  4455)                tlcla = tlcl
  4456)                funa  = funnow
  4457)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  4458)                if (.not.zside) funz = funz * 5.d-1
  4459)                !---------------------------------------------------------------------------!
  4460) 
  4461) 
  4462)                !----- We have just updated aside, set zside to false. ---------------------!
  4463)                zside = .false.
  4464)                !---------------------------------------------------------------------------!
  4465)             end if
  4466)          end do fpoloop
  4467)       end if
  4468)       !------------------------------------------------------------------------------------!
  4469) 
  4470) 
  4471) 
  4472)       !------------------------------------------------------------------------------------!
  4473)       !     Check whether we have succeeded or not.                                        !
  4474)       !------------------------------------------------------------------------------------!
  4475)       if (converged) then
  4476)          !----- We have found a solution, find the remaining LCL properties. --------------!
  4477)          pvap  = eslif8(tlcl,frozen)
  4478)          plcl  = (ep8 + rvap) * pvap / rvap
  4479)          dzlcl = max(cpog8*(temp-tlcl),0.d0)
  4480)          !---------------------------------------------------------------------------------!
  4481) 
  4482) 
  4483)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4484)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4485)          !write (unit=21,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x),3(a,1x,es11.4,1x))')             &
  4486)          !   'ANSWER: itb=',itn,'tlcl=',tlcl-t00,'eslcl=',0.01*pvap                         &
  4487)          !        ,'dzlcl=',dzlcl,'plcl=',plcl*0.01,'funa=',funa,'funz=',funz
  4488)          !write (unit=21,fmt='(a)') '-------------------------------------------------------'
  4489)          !write (unit=21,fmt='(a)') ' '
  4490)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4491)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4492)       else
  4493)          write (unit=*,fmt='(a)') '-------------------------------------------------------'
  4494)          write (unit=*,fmt='(a)') ' LCL Temperature didn''t converge!!!'
  4495)          write (unit=*,fmt='(a,1x,i5,1x,a)') ' I gave up, after',maxfpo,'iterations...'
  4496)          write (unit=*,fmt='(a)') ' '
  4497)          write (unit=*,fmt='(a)') ' Input values.'
  4498)          write (unit=*,fmt='(a)') ' '
  4499)          write (unit=*,fmt='(a,1x,f12.4)' ) 'theta_il        [     K] =',thil
  4500)          write (unit=*,fmt='(a,1x,f12.4)' ) 'Pressure        [   hPa] =',1.d-2*pres
  4501)          write (unit=*,fmt='(a,1x,f12.4)' ) 'Temperature     [    �C] =',temp-t008
  4502)          write (unit=*,fmt='(a,1x,f12.4)' ) 'rtot            [  g/kg] =',1.d3*rtot
  4503)          write (unit=*,fmt='(a,1x,f12.4)' ) 'rvap            [  g/kg] =',1.d3*rvap
  4504)          write (unit=*,fmt='(a)') ' '
  4505)          write (unit=*,fmt='(a)') ' Last iteration outcome.'
  4506)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tlcla           [    �C] =',tlcla-t008
  4507)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tlclz           [    �C] =',tlclz-t008
  4508)          write (unit=*,fmt='(a,1x,f12.4)' ) 'fun             [     K] =',funnow
  4509)          write (unit=*,fmt='(a,1x,f12.4)' ) 'funa            [     K] =',funa
  4510)          write (unit=*,fmt='(a,1x,f12.4)' ) 'funz            [     K] =',funz
  4511)          write (unit=*,fmt='(a,1x,f12.4)' ) 'deriv           [  ----] =',deriv
  4512)          write (unit=*,fmt='(a,1x,es12.4)') 'toler8          [  ----] =',toler8
  4513)          write (unit=*,fmt='(a,1x,es12.4)') 'error           [  ----] ='                   &
  4514)                                                             ,abs(tlclz-tlcla)/tlclz
  4515)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tlcl            [    �C] =',tlcl
  4516)          call fatal_error('TLCL didn''t converge, gave up!','lcl_il8','therm_lib8.f90')
  4517)       end if
  4518)       return

  4530)    !     This subroutine computes a consistent set of temperature and condensated phases   !
  4531)    ! mixing ratio for a given theta_il, Exner function, and total mixing ratio. This is    !
  4532)    ! very similar to the function thil2temp, except that now we don't know rliq and rice,  !
  4533)    ! and for this reason they also become functions of temperature, since they are defined !
  4534)    ! as rtot-rsat(T,p), remembering that rtot and p are known. If the air is not           !
  4535)    ! saturated, we rather use the fact that theta_il = theta and skip the hassle.          !
  4536)    ! Otherwise, we use iterative methods.  We will always try Newton's method, since it    !
  4537)    ! converges fast. The caveat is that Newton may fail, and it actually does fail very    !
  4538)    ! close to the triple point, because the saturation vapour pressure function has a      !
  4539)    ! "kink" at the triple point (continuous, but not differentiable).  If that's the case, !
  4540)    ! then we fall back to a modified regula falsi (Illinois) method, which is a mix of     !
  4541)    ! secant and bisection and will converge.                                               !
  4542)    !---------------------------------------------------------------------------------------!
  4543)    subroutine thil2tqall8(thil,exner,pres,rtot,rliq,rice,temp,rvap,rsat)
  4544)       use consts_coms, only : cpdry8    & ! intent(in)
  4545)                             , cpdryi8   & ! intent(in)
  4546)                             , t008      & ! intent(in)
  4547)                             , toodry8   & ! intent(in)
  4548)                             , t3ple8    & ! intent(in)
  4549)                             , ttripoli8 ! ! intent(in)
  4550) 
  4551)       implicit none
  4552)       !----- Arguments. -------------------------------------------------------------------!
  4553)       real(kind=8), intent(in)    :: thil      ! Ice-liquid water potential temp.  [     K]
  4554)       real(kind=8), intent(in)    :: exner     ! Exner function                    [J/kg/K]
  4555)       real(kind=8), intent(in)    :: pres      ! Pressure                          [    Pa]
  4556)       real(kind=8), intent(in)    :: rtot      ! Total mixing ratio                [ kg/kg]
  4557)       real(kind=8), intent(out)   :: rliq      ! Liquid water mixing ratio         [ kg/kg]
  4558)       real(kind=8), intent(out)   :: rice      ! Ice mixing ratio                  [ kg/kg]
  4559)       real(kind=8), intent(inout) :: temp      ! Temperature                       [     K]
  4560)       real(kind=8), intent(out)   :: rvap      ! Water vapour mixing ratio         [ kg/kg]
  4561)       real(kind=8), intent(out)   :: rsat      ! Sat. water vapour mixing ratio    [ kg/kg]
  4562)       !----- Local variables --------------------------------------------------------------!
  4563)       real(kind=8)                :: tempa     ! Lower bound for regula falsi iteration
  4564)       real(kind=8)                :: tempz     ! Upper bound for regula falsi iteration
  4565)       real(kind=8)                :: t1stguess ! Book keeping temperature 1st guess
  4566)       real(kind=8)                :: fun1st    ! Book keeping 1st guess function
  4567)       real(kind=8)                :: funa      ! Function evaluation at tempa
  4568)       real(kind=8)                :: funz      ! Function evaluation at tempz
  4569)       real(kind=8)                :: funnow    ! Function at this iteration.
  4570)       real(kind=8)                :: delta     ! Aux. var in case we need regula falsi.
  4571)       real(kind=8)                :: deriv     ! Derivative of this function.
  4572)       integer                     :: itn       ! Iteration counter
  4573)       integer                     :: itb       ! Iteration counter
  4574)       integer                     :: ii        ! Iteration counter
  4575)       logical                     :: converged ! Convergence handle
  4576)       logical                     :: zside     ! Aux. Flag, for two purposes:
  4577)                                                ! 1. Found a 2nd guess for regula falsi.
  4578)                                                ! 2. I retained the "zside" (T/F)
  4579)       !------------------------------------------------------------------------------------!
  4580) 
  4581)       t1stguess = temp
  4582) 
  4583)       !------------------------------------------------------------------------------------!
  4584)       !      First check: try to find temperature assuming sub-saturation and check if     !
  4585)       ! this is the case. If it is, then there is no need to go through the iterative      !
  4586)       ! loop.                                                                              !
  4587)       !------------------------------------------------------------------------------------!
  4588)       tempz  = cpdryi8 * thil * exner
  4589)       rsat   = max(toodry8,rslif8(pres,tempz))
  4590)       if (tempz >= t3ple8) then
  4591)          rliq = max(0.d0,rtot-rsat)
  4592)          rice = 0.d0
  4593)       else
  4594)          rice = max(0.d0,rtot-rsat)
  4595)          rliq = 0.d0
  4596)       end if
  4597)       rvap = rtot-rliq-rice
  4598)       !------------------------------------------------------------------------------------!
  4599) 
  4600)       !------------------------------------------------------------------------------------!
  4601)       !    If rtot < rsat, this is not saturated, we can leave the subroutine and bypass   !
  4602)       ! the iterative part.                                                                !
  4603)       !------------------------------------------------------------------------------------!
  4604)       if (rtot < rsat) then
  4605)          temp = tempz
  4606)          return
  4607)       end if
  4608) 
  4609)       !------------------------------------------------------------------------------------!
  4610)       !   If not, then use the temperature the user gave as first guess and solve          !
  4611)       ! iteratively.  We use the user instead of what we just found because if the air is  !
  4612)       ! saturated, then this can be too far off which may be bad for Newton's method.      !
  4613)       !------------------------------------------------------------------------------------!
  4614)       tempz = temp
  4615)       rsat   = max(toodry8,rslif8(pres,tempz))
  4616)       if (tempz >= t3ple8) then
  4617)          rliq = max(0.d0,rtot-rsat)
  4618)          rice = 0.d0
  4619)       else
  4620)          rice = max(0.d0,rtot-rsat)
  4621)          rliq = 0.d0
  4622)       end if
  4623)       rvap = rtot-rliq-rice
  4624) 
  4625) 
  4626)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4627)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4628)       !write (unit=46,fmt='(a)') '--------------------------------------------------------'
  4629)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4630)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4631) 
  4632) 
  4633)       !------------------------------------------------------------------------------------!
  4634)       !     Find the function. We are seeking a temperature which is associated with the   !
  4635)       ! theta_il we provided. Thus, the function is simply the difference between the      !
  4636)       ! theta_il associated with our guess and the actual theta_il.                        !
  4637)       !------------------------------------------------------------------------------------!
  4638)       funnow = theta_iceliq8(exner,tempz,rliq,rice)
  4639)       !----- Updating the derivative. -----------------------------------------------------!
  4640)       deriv  = dthetail_dt8(.false.,funnow,pres,tempz,rliq,rice)
  4641)       funnow = funnow - thil
  4642)       fun1st = funnow
  4643)       !------------------------------------------------------------------------------------!
  4644) 
  4645)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4646)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4647)       !write (unit=46,fmt='(a,1x,i5,1x,6(a,1x,f11.4,1x),a,1x,es11.4,1x)')                  &
  4648)       !   'NEWTON: it=',0,'temp=',tempz-t00,'rsat=',1000.*rsat,'rliq=',1000.*rliq          &
  4649)       !  ,'rice=',1000.*rice,'rvap=',1000.*rvap,'fun=',funnow,'deriv=',deriv
  4650)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4651)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4652) 
  4653)       !------------------------------------------------------------------------------------!
  4654)       !    Now we enter at the Newton's method iterative loop. We are always going to try  !
  4655)       ! this first, because it's fast, but if it turns out to be a dangerous choice or if  !
  4656)       ! it doesn't converge fast, we will fall back to regula falsi.                       !
  4657)       !    We start by initialising the flag and copying temp to tempz, the newest guess.  !
  4658)       !------------------------------------------------------------------------------------!
  4659)       converged=.false.
  4660)       newloop: do itn=1,maxfpo/6
  4661)          !---------------------------------------------------------------------------------!
  4662)          !     Saving previous guess. We also save the function is in case we give up on   !
  4663)          ! Newton's and switch to regula falsi.                                            !
  4664)          !---------------------------------------------------------------------------------!
  4665)          funa  = funnow
  4666)          tempa = tempz
  4667) 
  4668)          !----- Go to bisection if the derivative is too flat (too dangerous...) ----------!
  4669)          if (abs(deriv) < toler8) exit newloop
  4670) 
  4671)          tempz = tempa - funnow / deriv
  4672) 
  4673)          !----- Finding the mixing ratios associated with this guess ----------------------!
  4674)          rsat  = max(toodry8,rslif8(pres,tempz))
  4675)          if (tempz >= t3ple8) then
  4676)             rliq = max(0.d0,rtot-rsat)
  4677)             rice = 0.d0
  4678)          else
  4679)             rice = max(0.d0,rtot-rsat)
  4680)             rliq = 0.d0
  4681)          end if
  4682)          rvap = rtot-rliq-rice
  4683) 
  4684)          !----- Updating the function -----------------------------------------------------!
  4685)          funnow = theta_iceliq8(exner,tempz,rliq,rice)
  4686)          !----- Updating the derivative. --------------------------------------------------!
  4687)          deriv  = dthetail_dt8(.false.,funnow,pres,tempz,rliq,rice)
  4688)          funnow = funnow - thil
  4689) 
  4690)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4691)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4692)          !write (unit=46,fmt='(a,1x,i5,1x,6(a,1x,f11.4,1x),a,1x,es11.4,1x)')               &
  4693)          !   'NEWTON: it=',itn,'temp=',tempz-t00,'rsat=',1000.*rsat,'rliq=',1000.*rliq     &
  4694)          !  ,'rice=',1000.*rice,'rvap=',1000.*rvap,'fun=',funnow,'deriv=',deriv
  4695)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4696)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4697) 
  4698)          converged = abs(tempa-tempz) < toler8*tempz
  4699)          !---------------------------------------------------------------------------------!
  4700)          !   Convergence. The temperature will be the mid-point between tempa and tempz.   !
  4701)          ! Fix the mixing ratios and return. But first check for converged due to luck. If !
  4702)          ! the guess gives a root, then that's it. It looks unlikely, but it actually      !
  4703)          ! happens sometimes and if not checked it becomes a singularity.                  !
  4704)          !---------------------------------------------------------------------------------!
  4705)          if (funnow == 0.d0) then
  4706)             temp = tempz
  4707)             converged = .true.
  4708)             exit newloop
  4709)          elseif (converged) then
  4710)             temp = 5.d-1 * (tempa+tempz)
  4711)             rsat  = max(toodry8,rslif8(pres,temp))
  4712)             if (temp >= t3ple8) then
  4713)                rliq = max(0.d0,rtot-rsat)
  4714)                rice = 0.d0
  4715)             else
  4716)                rice = max(0.d0,rtot-rsat)
  4717)                rliq = 0.d0
  4718)             end if
  4719)             rvap = rtot-rliq-rice
  4720)             exit newloop
  4721)          end if
  4722) 
  4723)       end do newloop
  4724)       !------------------------------------------------------------------------------------!
  4725) 
  4726)       !----- For debugging only -----------------------------------------------------------!
  4727)       itb = itn+1
  4728) 
  4729)       if (.not. converged) then
  4730)          !---------------------------------------------------------------------------------!
  4731)          !    If I reach this point, then it means that Newton's method failed finding the !
  4732)          ! equilibrium, so we are going to use the regula falsi instead.  If Newton's      !
  4733)          ! method didn't converge, we use tempa as one guess and now we seek a tempz with  !
  4734)          ! opposite sign.                                                                  !
  4735)          !---------------------------------------------------------------------------------!
  4736)          !----- Check funa and funnow have opposite signs. If so, we are ready to go ------!
  4737)          if (funa*funnow < 0.d0) then
  4738)             funz  = funnow
  4739)             zside = .true.
  4740)          !----- Otherwise, checking whether the 1st guess had opposite sign. --------------!
  4741)          elseif (funa*fun1st < 0.d0 ) then
  4742)             funz  = fun1st
  4743)             zside = .true.
  4744)          !---------------------------------------------------------------------------------!
  4745)          !    Looking for a guess. Extrapolate funa linearly, trying to get the -funa.  We !
  4746)          ! don't need it to be funa, just with the opposite sign.  If that's not enough,   !
  4747)          ! we keep going further... Force the guesses to be at least 1K apart              !
  4748)          !---------------------------------------------------------------------------------!
  4749)          else
  4750)             if (abs(funnow-funa) < 1.d2 * toler8 * tempa) then
  4751)                delta = 1.d2 * toler8 * tempa
  4752)             else
  4753)                delta = max(abs(funa)*abs((tempz-tempa)/(funnow-funa)),1.d2*toler8*tempa)
  4754)             end if
  4755)             tempz = tempa + delta
  4756)             funz  = funa
  4757)             !----- Just to enter at least once. The 1st time tempz=tempa-2*delta ----------!
  4758)             zside = .false.
  4759)             zgssloop: do itb=1,maxfpo
  4760)                 tempz = tempa + dble((-1)**itb * (itb+3)/2) * delta
  4761)                 rsat   = max(toodry8,rslif8(pres,tempz))
  4762)                 if (tempz >= t3ple8) then
  4763)                    rliq = max(0.d0,rtot-rsat)
  4764)                    rice = 0.d0
  4765)                 else
  4766)                    rice = max(0.d0,rtot-rsat)
  4767)                    rliq = 0.d0
  4768)                 end if
  4769)                 rvap = rtot-rliq-rice
  4770)                 funz = theta_iceliq8(exner,tempz,rliq,rice) - thil
  4771)                 zside = funa*funz < 0.d0
  4772)                 if (zside) exit zgssloop
  4773)             end do zgssloop
  4774)             if (.not. zside) then
  4775)                write (unit=*,fmt='(60a1)')        ('-',ii=1,60)
  4776)                write (unit=*,fmt='(a)')           ' THIL2TQALL: NO SECOND GUESS FOR YOU!'
  4777)                write (unit=*,fmt='(a)')           ' '
  4778)                write (unit=*,fmt='(a)')           ' -> Input: '
  4779)                write (unit=*,fmt='(a,1x,f12.5)')  '    THETA_IL [     K]:',thil
  4780)                write (unit=*,fmt='(a,1x,f12.5)')  '    PRESS    [   hPa]:',1.d-2*pres
  4781)                write (unit=*,fmt='(a,1x,f12.5)')  '    EXNER    [J/kg/K]:',exner
  4782)                write (unit=*,fmt='(a,1x,f12.5)')  '    RTOT     [  g/kg]:',1.d3*rtot
  4783)                write (unit=*,fmt='(a,1x,f12.5)')  '    T1ST     [  degC]:',t1stguess-t008
  4784)                write (unit=*,fmt='(a,1x,f12.5)')  '    TEMPA    [  degC]:',tempa-t008
  4785)                write (unit=*,fmt='(a,1x,f12.5)')  '    TEMPZ    [  degC]:',tempz-t008
  4786)                write (unit=*,fmt='(a,1x,f12.5)')  '    FUNNOW   [     K]:',funnow
  4787)                write (unit=*,fmt='(a,1x,f12.5)')  '    FUNA     [     K]:',funa
  4788)                write (unit=*,fmt='(a,1x,f12.5)')  '    FUNZ     [     K]:',funz
  4789)                write (unit=*,fmt='(a,1x,f12.5)')  '    DELTA    [     K]:',delta
  4790)                write (unit=*,fmt='(60a1)')        ('-',ii=1,60)
  4791) 
  4792)                call fatal_error('Failed finding the second guess for regula falsi'         &
  4793)                                ,'thil2tqall8','therm_lib8.f90')
  4794)             end if
  4795)          end if
  4796)          !---------------------------------------------------------------------------------!
  4797) 
  4798)          !---------------------------------------------------------------------------------!
  4799)          !     Now we loop until convergence is achieved.  One important thing to notice   !
  4800)          ! is that Newton's method fail only when T is almost T3ple, which means that ice  !
  4801)          ! and liquid should be present, and we are trying to find the saturation point    !
  4802)          ! with all ice or all liquid. This will converge but the final answer will        !
  4803)          ! contain significant error. To reduce it we redistribute the condensates between !
  4804)          ! ice and liquid conserving the total condensed mixing ratio.                     !
  4805)          !---------------------------------------------------------------------------------!
  4806)          fpoloop: do itb=itn,maxfpo
  4807)             temp = (funz*tempa-funa*tempz)/(funz-funa)
  4808)             !----- Checking whether this guess will fall outside the range ----------------!
  4809)             if (abs(temp-tempa) > abs(tempz-tempa) .or.                                    &
  4810)                 abs(temp-tempz) > abs(tempz-tempa)) then
  4811)                temp = 5.d-1*(tempa+tempz)
  4812)             end if
  4813)             !----- Distributing vapour into the three phases ------------------------------!
  4814)             rsat   = max(toodry8,rslif8(pres,temp))
  4815)             rvap   = min(rtot,rsat)
  4816)             if (temp >= t3ple8) then
  4817)                rliq = max(0.d0,rtot-rsat)
  4818)                rice = 0.d0
  4819)             else
  4820)                rliq = 0.d0
  4821)                rice = max(0.d0,rtot-rsat)
  4822)             end if
  4823)             !----- Updating function ------------------------------------------------------!
  4824)             funnow = theta_iceliq8(exner,temp,rliq,rice) - thil
  4825) 
  4826)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4827)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4828)             !write (unit=46,fmt='(a,1x,i5,1x,10(a,1x,f11.4,1x))')                          &
  4829)             !   'REGFAL: it=',itb,'temp=',temp-t00,'tempa=',tempa-t00,'tempz=',tempz-t00   &
  4830)             !  ,'rsat=',1000.*rsat,'rliq=',1000.*rliq,'rice=',1000.*rice                   &
  4831)             !  ,'rvap=',1000.*rvap,'fun=',funnow,'funa=',funa,'funz=',funz
  4832)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4833)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4834) 
  4835)             !------------------------------------------------------------------------------!
  4836)             !    Checking for convergence or lucky guess. If it did, return, we found the  !
  4837)             ! solution. Otherwise, constrain the guesses.                                  !
  4838)             !------------------------------------------------------------------------------!
  4839)             converged = abs(temp-tempa) < toler8*temp .and. abs(temp-tempz) < toler8*temp
  4840)             if (funnow == 0.d0 .or. converged) then
  4841)                converged = .true.
  4842)                exit fpoloop
  4843)             elseif (funnow*funa < 0.d0) then
  4844)                tempz = temp
  4845)                funz  = funnow
  4846)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  4847)                if (zside) funa = funa * 5.d-1
  4848)                !----- We just updated zside, setting zside to true. -----------------------!
  4849)                zside = .true.
  4850)             elseif (funnow*funz < 0.d0) then
  4851)                tempa = temp
  4852)                funa  = funnow
  4853)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  4854)                if (.not.zside) funz = funz * 5.d-1
  4855)                !----- We just updated aside, setting zside to false -----------------------!
  4856)                zside = .false.
  4857)             end if
  4858)             !------------------------------------------------------------------------------!
  4859)          end do fpoloop
  4860)          !---------------------------------------------------------------------------------!
  4861) 
  4862) 
  4863)          !---------------------------------------------------------------------------------!
  4864)          !    Almost done... Usually when the method goes through regula falsi, it means   !
  4865)          ! that the temperature is too close to the triple point, and often all three      !
  4866)          ! phases will coexist.  The problem with the method is that it converges for      !
  4867)          ! temperature, but whenever regula falsi is called the function evaluation is     !
  4868)          ! usually far from zero.  This can be improved by finding a better partition      !
  4869)          ! between ice and liquid given the temperature and saturation mixing ratio we     !
  4870)          ! just found. So just to round these edges, we will invert the ice-liquid         !
  4871)          ! potential temperature using the set of temperature and rsat, and fiding the     !
  4872)          ! liquid mixing ratio.                                                            !
  4873)          !---------------------------------------------------------------------------------!
  4874)          if (abs(temp-t3ple8) < toler8*temp) then
  4875)             !----- Find rliq. -------------------------------------------------------------!
  4876)             rliq   = ( alvi8(temp) * (rtot - rsat)                                         &
  4877)                      + cpdry8 * temp * log ( cpdryi8 * exner * thil / temp ) )             &
  4878)                    / ( alvi8(temp) - alvl8(temp) )
  4879)             !------------------------------------------------------------------------------!
  4880) 
  4881)             !----- Correct rliq so it is bounded, and then find rice as the remainder. ----!
  4882)             rliq   = max( 0.d0, min( rtot - rsat,  rliq ) )
  4883)             rice   = max( 0.d0, rtot-rsat-rliq)
  4884)             !------------------------------------------------------------------------------!
  4885) 
  4886) 
  4887)             !----- Function evaluation. ---------------------------------------------------!
  4888)             funnow = theta_iceliq8(exner,temp,rliq,rice) - thil
  4889)             !------------------------------------------------------------------------------!
  4890)          end if
  4891)          !---------------------------------------------------------------------------------!
  4892)          itb=itb+1
  4893)       end if
  4894)       !------------------------------------------------------------------------------------!
  4895) 
  4896)       if (.not. converged) then
  4897)          write (unit=*,fmt='(60a1)')        ('-',ii=1,60)
  4898)          write (unit=*,fmt='(a)')           ' THIL2TQALL8 failed!'
  4899)          write (unit=*,fmt='(a)')           ' '
  4900)          write (unit=*,fmt='(a)')           ' -> Input: '
  4901)          write (unit=*,fmt='(a,1x,f12.5)')  '    THETA_IL [     K]:',thil
  4902)          write (unit=*,fmt='(a,1x,f12.5)')  '    EXNER    [J/kg/K]:',exner
  4903)          write (unit=*,fmt='(a,1x,f12.5)')  '    RTOT     [  g/kg]:',1.d3*rtot
  4904)          write (unit=*,fmt='(a)')           ' '
  4905)          write (unit=*,fmt='(a)')           ' -> Output: '
  4906)          write (unit=*,fmt='(a,1x,i12)')    '    ITERATIONS       :',itb
  4907)          write (unit=*,fmt='(a,1x,f12.5)')  '    TEMP     [    �C]:',temp-t008
  4908)          write (unit=*,fmt='(a,1x,f12.5)')  '    RVAP     [  g/kg]:',1.d3*rvap
  4909)          write (unit=*,fmt='(a,1x,f12.5)')  '    RLIQ     [  g/kg]:',1.d3*rliq
  4910)          write (unit=*,fmt='(a,1x,f12.5)')  '    RICE     [  g/kg]:',1.d3*rice
  4911)          write (unit=*,fmt='(a,1x,f12.5)')  '    TEMPA    [    �C]:',tempa-t008
  4912)          write (unit=*,fmt='(a,1x,f12.5)')  '    TEMPZ    [    �C]:',tempz-t008
  4913)          write (unit=*,fmt='(a,1x,es12.5)') '    FUNA     [     K]:',funnow
  4914)          write (unit=*,fmt='(a,1x,es12.5)') '    FUNZ     [     K]:',funnow
  4915)          write (unit=*,fmt='(a,1x,es12.5)') '    DERIV    [   ---]:',deriv
  4916)          write (unit=*,fmt='(a,1x,es12.5)') '    ERR_A    [   ---]:',abs(temp-tempa)/temp
  4917)          write (unit=*,fmt='(a,1x,es12.5)') '    ERR_Z    [   ---]:',abs(temp-tempz)/temp
  4918)          write (unit=*,fmt='(a)')           ' '
  4919)          write (unit=*,fmt='(60a1)')        ('-',ii=1,60)
  4920)          call fatal_error('Failed finding equilibrium, I gave up!','thil2tqall8'           &
  4921)                          ,'therm_lib8.f90')
  4922)       end if
  4923)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4924)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4925)       !write (unit=46,fmt='(a,1x,i5,1x,6(a,1x,f11.4,1x))')                                 &
  4926)       !   'ANSWER: it=',itb,'funf=',funnow,'temp=',temp-t00                                &
  4927)       !  ,'rsat=',1000.*rsat,'rliq=',1000.*rliq,'rice=',1000.*rice,'rvap=',1000.*rvap
  4928)       !write (unit=46,fmt='(a)') '----------------------------------------------------------'
  4929)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4930)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4931)       return
  4932)    end subroutine thil2tqall8

  4943)    !     This subroutine computes a consistent set of temperature and condensated phases   !
  4944)    ! mixing ratio for a given theta_il, Exner function, and total mixing ratio.  This is   !
  4945)    ! very similar to the function thil2temp, except that now we don't know rliq.  Rliq     !
  4946)    ! becomes a function of temperature, since it is defined as rtot-rsat(T,p), remembering !
  4947)    ! that rtot and p are known. If the air is not saturated, we use the fact that theta_il !
  4948)    ! is theta and skip the hassle.  Otherwise, we use iterative methods.  We will always   !
  4949)    ! try Newton's method, since it converges fast.  Not always will Newton converge, and   !
  4950)    ! if that's the case we use a modified regula falsi (Illinois) method.  This method is  !
  4951)    ! a mix of secant and bisection and will always converge.                               !
  4952)    !---------------------------------------------------------------------------------------!
  4953)    subroutine thil2tqliq8(thil,exner,pres,rtot,rliq,temp,rvap,rsat)
  4954)       use consts_coms, only : cpdryi8   & ! intent(in)
  4955)                             , toodry8   ! ! intent(in)
  4956)       implicit none
  4957)       !----- Arguments. -------------------------------------------------------------------!
  4958)       real(kind=8), intent(in)    :: thil      ! Ice-liquid water potential temp.  [     K]
  4959)       real(kind=8), intent(in)    :: exner     ! Exner function                    [J/kg/K]
  4960)       real(kind=8), intent(in)    :: pres      ! Pressure                          [    Pa]
  4961)       real(kind=8), intent(in)    :: rtot      ! Total mixing ratio                [ kg/kg]
  4962)       real(kind=8), intent(out)   :: rliq      ! Liquid water mixing ratio         [ kg/kg]
  4963)       real(kind=8), intent(inout) :: temp      ! Temperature                       [     K]
  4964)       real(kind=8), intent(out)   :: rvap      ! Water vapour mixing ratio         [ kg/kg]
  4965)       real(kind=8), intent(out)   :: rsat      ! Sat. water vapour mixing ratio    [ kg/kg]
  4966)       !----- Local variables --------------------------------------------------------------!
  4967)       real(kind=8)                :: tempa     ! Lower bound for regula falsi iteration
  4968)       real(kind=8)                :: tempz     ! Upper bound for regula falsi iteration
  4969)       real(kind=8)                :: funa      ! Function evaluation at tempa
  4970)       real(kind=8)                :: funz      ! Function evaluation at tempz
  4971)       real(kind=8)                :: funnow    ! Function at this iteration.
  4972)       real(kind=8)                :: delta     ! Aux. var in case we need regula falsi.
  4973)       real(kind=8)                :: deriv     ! Derivative of this function.
  4974)       integer                     :: itn       ! Iteration counter
  4975)       integer                     :: itb       ! Iteration counter
  4976)       logical                     :: converged ! Convergence handle
  4977)       logical                     :: zside     ! Aux. Flag, for two purposes:
  4978)                                                ! 1. Found a 2nd guess for regula falsi.
  4979)                                                ! 2. I retained the "zside" (T/F)
  4980)       !------------------------------------------------------------------------------------!
  4981) 
  4982) 
  4983) 
  4984) 
  4985)       !------------------------------------------------------------------------------------!
  4986)       !      First check: try to find temperature assuming sub-saturation and check if     !
  4987)       ! this is the case. If it is, then there is no need to go through the iterative      !
  4988)       ! loop.                                                                              !
  4989)       !------------------------------------------------------------------------------------!
  4990)       tempz = cpdryi8 * thil * exner
  4991)       rsat  = max(toodry8,rslf8(pres,tempz))
  4992)       rliq  = max(0.d0,rtot-rsat)
  4993)       rvap  = rtot-rliq
  4994)       !------------------------------------------------------------------------------------!
  4995) 
  4996) 
  4997) 
  4998)       !------------------------------------------------------------------------------------!
  4999)       !    If rtot < rsat, this is not saturated, we can leave the subroutine and bypass   !
  5000)       ! the iterative part.                                                                !
  5001)       !------------------------------------------------------------------------------------!
  5002)       if (rtot < rsat) then
  5003)          temp = tempz
  5004)          return
  5005)       end if
  5006)       !------------------------------------------------------------------------------------!
  5007) 
  5008) 
  5009) 
  5010)       !------------------------------------------------------------------------------------!
  5011)       !   If not, then use the temperature the user gave as first guess and solve          !
  5012)       ! iteratively.  We use the user instead of what we just found because if the air is  !
  5013)       ! saturated, then this can be too far off which may be bad for Newton's method.      !
  5014)       !------------------------------------------------------------------------------------!
  5015)       tempz = temp
  5016)       rsat  = max(toodry8,rslf8(pres,tempz))
  5017)       rliq  = max(0.d0,rtot-rsat)
  5018)       rvap  = rtot-rliq
  5019)       !------------------------------------------------------------------------------------!
  5020) 
  5021) 
  5022)       !------------------------------------------------------------------------------------!
  5023)       !     Finding the function. We are seeking a temperature which is associated with    !
  5024)       ! the theta_il we provided. Thus, the function is simply the difference between the  !
  5025)       ! theta_il associated with our guess and the actual theta_il.                        !
  5026)       !------------------------------------------------------------------------------------!
  5027)       funnow = theta_iceliq8(exner,tempz,rliq,0.d0) ! Finding thil from our guess
  5028)       deriv  = dthetail_dt8(.false.,funnow,pres,tempz,rliq)
  5029)       funnow = funnow - thil ! Computing the function
  5030)       !------------------------------------------------------------------------------------!
  5031) 
  5032) 
  5033) 
  5034)       !------------------------------------------------------------------------------------!
  5035)       !    Now we enter at the Newton's method iterative loop. We are always going to try  !
  5036)       ! this first, because it's fast, but if it turns out to be a dangerous choice or if  !
  5037)       ! it doesn't converge fast, we will fall back to regula falsi.                       !
  5038)       !    We start by initialising the flag and copying temp to tempz, the newest guess.  !
  5039)       !------------------------------------------------------------------------------------!
  5040)       converged=.false.
  5041)       newloop: do itn=1,maxfpo/6
  5042)          !---------------------------------------------------------------------------------!
  5043)          !     Saving previous guess. We also save the function is in case we give up on   !
  5044)          ! Newton's and switch to regula falsi.                                            !
  5045)          !---------------------------------------------------------------------------------!
  5046)          funa  = funnow
  5047)          tempa = tempz
  5048) 
  5049)          !----- Go to bisection if the derivative is too flat (too dangerous...) ----------!
  5050)          if (abs(deriv) < toler8) exit newloop
  5051) 
  5052)          tempz = tempa - funnow / deriv
  5053) 
  5054)          !----- Finding the mixing ratios associated with this guess ----------------------!
  5055)          rsat  = max(toodry8,rslf8(pres,tempz))
  5056)          rliq = max(0.d0,rtot-rsat)
  5057)          rvap = rtot-rliq
  5058) 
  5059)          !----- Updating the function -----------------------------------------------------!
  5060)          funnow = theta_iceliq8(exner,tempz,rliq,0.d0)
  5061)          !----- Updating the derivative. --------------------------------------------------!
  5062)          deriv = dthetail_dt8(.false.,funnow,pres,tempz,rliq)
  5063)          funnow = funnow - thil
  5064) 
  5065)          converged = abs(tempa-tempz) < toler8*tempz
  5066)          !---------------------------------------------------------------------------------!
