     3) !      This sub-routine allows the user to change the solar radiation splitting that comes !
     4) ! from the meteorological forcing.  This is useful for when we don't really know how the   !
     5) ! radiation terms were found, or if we want to play around with the numbers.               !
     6) !------------------------------------------------------------------------------------------!
     7) subroutine solar_radiation_breakdown(cgrid,ipy)
     8)    use ed_state_vars        , only : edtype        ! ! intent(in)
     9)    use met_driver_coms      , only : imetrad       ! ! intent(in)
    10)    use canopy_radiation_coms, only : fvis_beam_def & ! intent(in)
    11)                                    , fvis_diff_def & ! intent(in)
    12)                                    , fnir_beam_def & ! intent(in)
    13)                                    , fnir_diff_def & ! intent(in)
    14)                                    , cosz_min      ! ! intent(in)
    15) 
    16) 
    17)    implicit none
    18)    !----- Arguments. ----------------------------------------------------------------------!
    19)    type(edtype), target     :: cgrid
    20)    integer     , intent(in) :: ipy
    21)    !----- Local variables. ----------------------------------------------------------------!
    22)    real                     :: rshort_beam
    23)    !---------------------------------------------------------------------------------------!
    24) 
    25) 
    26) 
    27) 
    28)    !---------------------------------------------------------------------------------------!
    29)    !     Add the radiation components, regardless of the method.                           !
    30)    !---------------------------------------------------------------------------------------!
    31)    cgrid%met(ipy)%rshort_diffuse = cgrid%met(ipy)%par_diffuse + cgrid%met(ipy)%nir_diffuse
    32)    cgrid%met(ipy)%rshort         = cgrid%met(ipy)%rshort_diffuse                           &
    33)                                  + cgrid%met(ipy)%par_beam + cgrid%met(ipy)%nir_beam
    34)    !---------------------------------------------------------------------------------------!
    35) 
    36) 
    37) 
    38)    !---------------------------------------------------------------------------------------!
    39)    !     Choose what to do with the radiation depending on the user choice.                !
    40)    !---------------------------------------------------------------------------------------!
    41)    select case (imetrad)
    42)    case (0)
    43)       !------------------------------------------------------------------------------------!
    44)       !    We just make sure there is no beam radiation in case the Sun is too low, other- !
    45)       ! wise, we leave radiation as is.                                                    !
    46)       !------------------------------------------------------------------------------------!
    47)       if (cgrid%cosz(ipy) <= cosz_min) then
    48)          cgrid%met(ipy)%rshort_diffuse = cgrid%met(ipy)%rshort
    49)          cgrid%met(ipy)%nir_diffuse    = cgrid%met(ipy)%nir_diffuse                        &
    50)                                        + cgrid%met(ipy)%nir_beam
    51)          cgrid%met(ipy)%par_diffuse    = cgrid%met(ipy)%par_diffuse                        &
    52)                                        + cgrid%met(ipy)%par_beam
    53)          cgrid%met(ipy)%nir_beam       = 0.
    54)          cgrid%met(ipy)%par_beam       = 0.
    55)       end if
    56)       !------------------------------------------------------------------------------------!
    57) 
    58)    case (1)
    59)       !------------------------------------------------------------------------------------!
    60)       !     Re-calculate the diffuse shortwave radiation using the SiB method.             !
    61)       !------------------------------------------------------------------------------------!
    62)       call short2diff_sib(cgrid%met(ipy)%rshort,cgrid%cosz(ipy)                            &
    63)                          ,cgrid%met(ipy)%rshort_diffuse)
    64)       rshort_beam = cgrid%met(ipy)%rshort - cgrid%met(ipy)%rshort_diffuse
    65)       cgrid%met(ipy)%nir_diffuse = fnir_diff_def * cgrid%met(ipy)%rshort_diffuse
    66)       cgrid%met(ipy)%par_diffuse = fvis_diff_def * cgrid%met(ipy)%rshort_diffuse
    67)       cgrid%met(ipy)%nir_beam    = fnir_beam_def * rshort_beam
    68)       cgrid%met(ipy)%par_beam    = fvis_beam_def * rshort_beam
    69)       !------------------------------------------------------------------------------------!
    70) 
    71)    case (2)
    72)       !------------------------------------------------------------------------------------!
    73)       !     Re-calculate all components using the Weiss and Norman (1985) method.          !
    74)       !------------------------------------------------------------------------------------!
    75)       call short_bdown_weissnorman(cgrid%met(ipy)%rshort                                   &
    76)                                   ,cgrid%met(ipy)%prss                                     &
    77)                                   ,cgrid%cosz(ipy)                                         &
    78)                                   ,cgrid%met(ipy)%par_beam                                 &
    79)                                   ,cgrid%met(ipy)%par_diffuse                              &
    80)                                   ,cgrid%met(ipy)%nir_beam                                 &
    81)                                   ,cgrid%met(ipy)%nir_diffuse                              &
    82)                                   ,cgrid%met(ipy)%rshort_diffuse )
    83)       !----- Make sure the total radiation is preserved. ----------------------------------!
    84)       cgrid%met(ipy)%rshort = cgrid%met(ipy)%par_beam + cgrid%met(ipy)%par_diffuse         &
    85)                             + cgrid%met(ipy)%nir_beam + cgrid%met(ipy)%nir_diffuse
    86)       !------------------------------------------------------------------------------------!
    87) 
    88)    case (3)
    89)       !------------------------------------------------------------------------------------!
    90)       !      Make all radiation diffuse.                                                   !
    91)       !------------------------------------------------------------------------------------!
    92)       cgrid%met(ipy)%rshort_diffuse = cgrid%met(ipy)%rshort
    93)       cgrid%met(ipy)%nir_diffuse    = cgrid%met(ipy)%nir_diffuse + cgrid%met(ipy)%nir_beam
    94)       cgrid%met(ipy)%par_diffuse    = cgrid%met(ipy)%par_diffuse + cgrid%met(ipy)%par_beam
    95)       cgrid%met(ipy)%nir_beam       = 0.
    96)       cgrid%met(ipy)%par_beam       = 0.
    97)       !------------------------------------------------------------------------------------!
    98) 
    99)    case (4)
   100)       !------------------------------------------------------------------------------------!
   101)       !     Make all radiation direct, except when the sun is too low, in which case we    !
   102)       ! must set to diffuse to avoid singularities.                                        !
   103)       !------------------------------------------------------------------------------------!
   104)       if (cgrid%cosz(ipy) > cosz_min) then
   105)          cgrid%met(ipy)%rshort_diffuse = 0.
   106)          cgrid%met(ipy)%nir_beam       = cgrid%met(ipy)%nir_diffuse                        &
   107)                                        + cgrid%met(ipy)%nir_beam
   108)          cgrid%met(ipy)%par_beam       = cgrid%met(ipy)%par_diffuse                        &
   109)                                        + cgrid%met(ipy)%par_beam
   110)          cgrid%met(ipy)%nir_diffuse    = 0.
   111)          cgrid%met(ipy)%par_diffuse    = 0.
   112)       else
   113)          cgrid%met(ipy)%rshort_diffuse = cgrid%met(ipy)%rshort
   114)          cgrid%met(ipy)%nir_diffuse    = cgrid%met(ipy)%nir_diffuse                        &
   115)                                        + cgrid%met(ipy)%nir_beam
   116)          cgrid%met(ipy)%par_diffuse    = cgrid%met(ipy)%par_diffuse                        &
   117)                                        + cgrid%met(ipy)%par_beam
   118)          cgrid%met(ipy)%nir_beam       = 0.
   119)          cgrid%met(ipy)%par_beam       = 0.
   120)       end if
   121)       !------------------------------------------------------------------------------------!
   122)    end select
   123)    !---------------------------------------------------------------------------------------!
   124) 
   125)    return
   126) end subroutine solar_radiation_breakdown

   137) !     This subroutine is an adaptation from the SiB2 model (Sellers et al. 1986), and it   !
   138) ! splits the radiation into diffuse and direct radiation.  This is currently used to       !
   139) ! convert the radiation from BRAMS.  In the future, we should use the values that come     !
   140) ! from the radiation schemes.                                                              !
   141) !------------------------------------------------------------------------------------------!
   142) subroutine short2diff_sib(rshort_tot,cosz,rshort_diff)
   143)    use canopy_radiation_coms, only : cosz_min  ! ! intent(in)
   144)    implicit none
   145)    !----- Arguments. ----------------------------------------------------------------------!
   146)    real, intent(in)    :: rshort_tot  ! Surface incident shortwave radiation
   147)    real, intent(in)    :: cosz        ! Cosine of the zenith distance
   148)    real, intent(out)   :: rshort_diff ! Surface incident diffuse shortwave radiation.
   149)    !----- Local variables. ----------------------------------------------------------------!
   150)    real                :: cloud
   151)    real                :: difrat
   152)    real                :: vnrat
   153)    !----- Local constants. ----------------------------------------------------------------!
   154)    real, parameter     :: c1 = 580.
   155)    real, parameter     :: c2 = 464.
   156)    real, parameter     :: c3 = 499.
   157)    real, parameter     :: c4 = 963.
   158)    real, parameter     :: c5 = 1160.
   159)    !---------------------------------------------------------------------------------------!
   160) 
   161)    if (cosz > cosz_min) then
   162) 
   163)       cloud       = min(1.,max(0.,(c5 * cosz - rshort_tot) / (c4 * cosz)))
   164)       difrat      = min(1.,max(0.,0.0604 / ( cosz -0.0223 ) + 0.0683))
   165) 
   166)       difrat      = difrat + ( 1. - difrat ) * cloud
   167)       vnrat       = ( c1 - cloud*c2 ) / ( ( c1 - cloud*c3 ) + ( c1 - cloud*c2 ))
   168) 
   169)       rshort_diff = difrat*vnrat*rshort_tot
   170)    else
   171)       rshort_diff = rshort_tot
   172)    end if
   173) 
   174)    return
   175) end subroutine short2diff_sib

   186) !      This subroutine computes the split between direct and diffuse radiation, and        !
   187) ! between visible and near-infrared radiation using the method suggested by:               !
   188) !                                                                                          !
   189) ! Weiss, A., J. M. Norman, 1985: Partitioning solar radiation into direct and diffuse,     !
   190) !     visible and near-infrared components.  Agric. For. Meteorol., 34, 205-213. (WN85)    !
   191) !------------------------------------------------------------------------------------------!
   192) subroutine short_bdown_weissnorman(rshort_full,atm_prss,cosz,par_beam,par_diff,nir_beam    &
   193)                                   ,nir_diff,rshort_diff)
   194)    use consts_coms          , only : solar         & ! intent(in)
   195)                                    , prefsea       & ! intent(in)
   196)                                    , twothirds     ! ! intent(in)
   197)    use canopy_radiation_coms, only : cosz_min      & ! intent(in)
   198)                                    , fvis_beam_def & ! intent(in)
   199)                                    , fvis_diff_def & ! intent(in)
   200)                                    , fnir_beam_def & ! intent(in)
   201)                                    , fnir_diff_def ! ! intent(in)
   202)    implicit none
   203)    !----- Arguments. ----------------------------------------------------------------------!
   204)    real, intent(in)    :: rshort_full          ! Incident SW radiation   (total)   [  W/m�]
   205)    real, intent(in)    :: atm_prss             ! Atmospheric pressure              [    Pa]
   206)    real, intent(in)    :: cosz                 ! cos(zenith distance)              [   ---]
   207)    real, intent(out)   :: par_beam             ! Incident PAR            (direct ) [  W/m�]
   208)    real, intent(out)   :: par_diff             ! Incident PAR            (diffuse) [  W/m�]
   209)    real, intent(out)   :: nir_beam             ! Incident near-infrared  (direct ) [  W/m�]
   210)    real, intent(out)   :: nir_diff             ! Incident near-infrared  (diffuse) [  W/m�]
   211)    real, intent(out)   :: rshort_diff          ! Incident SW radiation   (diffuse) [  W/m�]
   212)    !----- Local variables. ----------------------------------------------------------------!
   213)    real                :: par_beam_top         ! PAR at the top of atmosphere      [  W/m�]
   214)    real                :: nir_beam_top         ! NIR at the top of atmosphere      [  W/m�]
   215)    real                :: par_beam_pot         ! Potential incident PAR  (direct)  [  W/m�]
   216)    real                :: par_diff_pot         ! Potential incident PAR  (diffuse) [  W/m�]
   217)    real                :: par_full_pot         ! Potential incident PAR  (total)   [  W/m�]
   218)    real                :: nir_beam_pot         ! Potential  PAR          (direct)  [  W/m�]
   219)    real                :: nir_diff_pot         ! Potential  PAR          (diffuse) [  W/m�]
   220)    real                :: nir_full_pot         ! Potential  PAR          (total)   [  W/m�]
   221)    real                :: par_full             ! Actual incident PAR     (total)   [  W/m�]
   222)    real                :: nir_full             ! Actual incident NIR     (total)   [  W/m�]
   223)    real                :: fvis_beam_act        ! Actual beam fraction - PAR        [   ---]
   224)    real                :: fnir_beam_act        ! Actual beam fraction - NIR        [   ---]
   225)    real                :: fvis_diff_act        ! Actual diffuse fraction - PAR     [   ---]
   226)    real                :: fnir_diff_act        ! Actual diffuse fraction - NIR     [   ---]
   227)    real                :: ratio                ! Ratio between obs. and expected   [   ---]
   228)    real                :: aux_par              ! Auxiliary variable                [   ---]
   229)    real                :: aux_nir              ! Auxiliary variable                [   ---]
   230)    real                :: secz                 ! sec(zenith distance)              [   ---]
   231)    real                :: log10secz            ! log10[sec(zenith distance)]       [   ---]
   232)    real                :: w10                  ! Minimum Water absorption          [  W/m�]
   233)    !---------------------------------------------------------------------------------------!
   234)    !    Local constants.                                                                   !
   235)    !---------------------------------------------------------------------------------------!
   236)    !----- Extinction coefficient. (equations 1 and 4 of WN85) -----------------------------!
   237)    real              , parameter :: par_beam_expext  = -0.185
   238)    real              , parameter :: nir_beam_expext  = -0.060
   239)    !----- This is the typical conversion of diffuse radiation in sunny days. --------------!
   240)    real              , parameter :: par2diff_sun = 0.400
   241)    real              , parameter :: nir2diff_sun = 0.600
   242)    !----- Coefficients for various equations in WN85. -------------------------------------!
   243)    real, dimension(3), parameter :: wn85_06 = (/ -1.1950, 0.4459, -0.0345 /)
   244)    real, dimension(2), parameter :: wn85_11 = (/  0.90, 0.70 /)
   245)    real, dimension(2), parameter :: wn85_12 = (/  0.88, 0.68 /)
   246)    !---------------------------------------------------------------------------------------!
   247) 
   248) 
   249) 
   250)    !---------------------------------------------------------------------------------------!
   251)    !     First thing to check is whether this is daytime or "night-time".  If the zenith   !
   252)    ! angle is too close to horizon, we assume it's dawn/dusk and all radiation goes to     !
   253)    ! diffuse.                                                                              !
   254)    !---------------------------------------------------------------------------------------!
   255)    if (cosz <= cosz_min) then
   256)       rshort_diff  = rshort_full
   257)       par_beam     = 0.0
   258)       nir_beam     = 0.0
   259)       par_diff     = fvis_diff_def * rshort_diff
   260)       nir_diff     = fnir_diff_def * rshort_diff
   261)    else
   262)       !----- Save 1/cos(zen), which is the secant.  We will use this several times. -------!
   263)       secz      = 1. / cosz
   264)       log10secz = log10(secz)
   265)       !------------------------------------------------------------------------------------!
   266) 
   267) 
   268)       !----- Total radiation at the top [  W/m�], using ED defaults. ----------------------!
   269)       par_beam_top = fvis_beam_def * solar
   270)       nir_beam_top = fnir_beam_def * solar
   271)       !------------------------------------------------------------------------------------!
   272) 
   273)       !------------------------------------------------------------------------------------!
   274)       !    Find the potential PAR components (beam, diffuse, total), using equations 1, 3, !
   275)       ! and 9 of WN85.                                                                     !
   276)       !------------------------------------------------------------------------------------!
   277)       par_beam_pot = par_beam_top                                                          &
   278)                    * exp ( par_beam_expext * (atm_prss / prefsea) * secz) * cosz
   279)       par_diff_pot = par2diff_sun * (par_beam_top - par_beam_pot) * cosz
   280)       par_full_pot = par_beam_pot + par_diff_pot
   281)       !------------------------------------------------------------------------------------!
   282) 
   283) 
   284) 
   285)       !------------------------------------------------------------------------------------!
   286)       !     Find the NIR absorption of 10 mm of precipitable water, using WN85 equation 6. !
   287)       !------------------------------------------------------------------------------------!
   288)       w10 = solar * 10 ** (wn85_06(1) + log10secz * (wn85_06(2) + wn85_06(3) * log10secz))
   289)       !------------------------------------------------------------------------------------!
   290) 
   291) 
   292) 
   293)       !------------------------------------------------------------------------------------!
   294)       !     Find the potential direct and diffuse near-infrared radiation, using equations !
   295)       ! 4, 5, and 10 of WN85.                                                              !
   296)       !------------------------------------------------------------------------------------!
   297)       nir_beam_pot = ( nir_beam_top                                                        &
   298)                      * exp ( nir_beam_expext * (atm_prss / prefsea) * secz) - w10 ) * cosz
   299)       nir_diff_pot = nir2diff_sun * ( nir_beam_top - nir_beam_pot - w10 ) * cosz
   300)       nir_full_pot = nir_beam_pot + nir_diff_pot
   301)       !------------------------------------------------------------------------------------!
   302) 
   303) 
   304) 
   305)       !------------------------------------------------------------------------------------!
   306)       !     Find the actual total for PAR and NIR, using equations 7 and 8.                !
   307)       !------------------------------------------------------------------------------------!
   308)       ratio    = rshort_full / (par_full_pot + nir_full_pot)
   309)       par_full = ratio * par_full_pot
   310)       nir_full = ratio * nir_full_pot
   311)       !------------------------------------------------------------------------------------!
   312) 
   313) 
   314) 
   315)       !------------------------------------------------------------------------------------!
   316)       !     Find the fraction of PAR and NIR that stays as beam, using equations 11 and 12 !
   317)       ! of WN85.                                                                           !
   318)       !------------------------------------------------------------------------------------!
   319)       !----- Make sure that the ratio is bounded. -----------------------------------------!
   320)       aux_par       = min(wn85_11(1), max(0., ratio))
   321)       aux_nir       = min(wn85_12(1), max(0., ratio))
   322)       fvis_beam_act = min(1., max(0., par_beam_pot                                         &
   323)                                  * (1. - ((wn85_11(1) - aux_par)/wn85_11(2)) ** twothirds) &
   324)                                  / par_full_pot ) )
   325)       fnir_beam_act = min(1., max(0., nir_beam_pot                                         &
   326)                                  * (1. - ((wn85_12(1) - aux_nir)/wn85_12(2)) ** twothirds) &
   327)                                  / nir_full_pot ) )
   328)       fvis_diff_act = 1. - fvis_beam_act
   329)       fnir_diff_act = 1. - fnir_beam_act
   330)       !------------------------------------------------------------------------------------!
   331) 
   332) 
   333) 
   334)       !------------------------------------------------------------------------------------!
   335)       !     Find the radiation components.                                                 !
   336)       !------------------------------------------------------------------------------------!
   337)       par_beam    = fvis_beam_act * par_full
   338)       par_diff    = fvis_diff_act * par_full
   339)       nir_beam    = fnir_beam_act * nir_full
   340)       nir_diff    = fnir_diff_act * nir_full
   341)       rshort_diff = par_diff + nir_diff
   342)       !------------------------------------------------------------------------------------!
   343)    end if
   344) 
   345)    return
   346) end subroutine short_bdown_weissnorman

   357) !    This subroutine updates the 10-day running average of radiation, which is used for    !
   358) ! phenology.                                                                               !
   359) !------------------------------------------------------------------------------------------!
   360) subroutine update_rad_avg(cgrid)
   361)    use ed_state_vars , only : edtype      & ! structure
   362)                             , polygontype & ! structure
   363)                             , sitetype    ! ! structure
   364)    use ed_misc_coms     , only : radfrq      ! ! intent(in)
   365)    use consts_coms   , only : day_sec     ! ! intent(in)
   366)    implicit none
   367)    !----- Arguments. ----------------------------------------------------------------------!
   368)    type(edtype)     , target    :: cgrid
   369)    !----- Local variables. ----------------------------------------------------------------!
   370)    type(polygontype), pointer   :: cpoly
   371)    integer                      :: ipy
   372)    integer                      :: isi
   373)    real                         :: tfact
   374)    !----- Local constants. ----------------------------------------------------------------!
   375)    real             , parameter :: tendays_sec = 10.*day_sec
   376)    !---------------------------------------------------------------------------------------!
   377) 
   378)    tfact = radfrq/tendays_sec
   379) 
   380) 
   381)          
   382)    polyloop: do ipy = 1,cgrid%npolygons
   383)       cpoly => cgrid%polygon(ipy)
   384)       siteloop: do isi = 1,cpoly%nsites
   385)          cpoly%rad_avg(isi) = cpoly%rad_avg(isi) * (1.0 - tfact)                           &
   386)                             + cpoly%met(isi)%rshort * tfact
   387)       end do siteloop
   388)    end do polyloop
   389) 
   390)    return
   391) end subroutine update_rad_avg

   449)    !----- Locally saved variables, to delete and re-create scratch files. -----------------!
