     4) !==========================================================================================!
     5) !==========================================================================================!
     6) !    This subroutine controls the changes in leaf biomass due to phenology.                !
     7) !------------------------------------------------------------------------------------------!
     8) subroutine phenology_driver(cgrid, doy, month, tfact)
     9)    use ed_state_vars  , only : edtype                & ! structure
    10)                              , polygontype           & ! structure
    11)                              , sitetype              ! ! structure
    12)    use phenology_coms , only : iphen_scheme          ! ! intent(in)
    13)    use ed_misc_coms   , only : current_time          ! ! intent(in)
    14)    use phenology_aux  , only : prescribed_leaf_state & ! subroutine
    15)                              , update_thermal_sums   & ! subroutine
    16)                              , update_turnover       ! ! subroutine
    17)    implicit none
    18)    !----- Arguments -----------------------------------------------------------------------!
    19)    type(edtype)      , target      :: cgrid
    20)    integer           , intent(in)  :: doy
    21)    integer           , intent(in)  :: month
    22)    real              , intent(in)  :: tfact
    23)    !----- Local variables -----------------------------------------------------------------!
    24)    type(polygontype) , pointer     :: cpoly
    25)    type(sitetype)    , pointer     :: csite
    26)    integer                         :: ipy
    27)    integer                         :: isi
    28)    integer                         :: ipa
    29)    !---------------------------------------------------------------------------------------!
    30) 
    31)    do ipy = 1,cgrid%npolygons
    32)       cpoly => cgrid%polygon(ipy)
    33) 
    34)       do isi = 1,cpoly%nsites
    35)          csite => cpoly%site(isi)
    36) 
    37)          !---------------------------------------------------------------------------------!
    38)          !     Get the patch-level average daily temperature, which is needed for mortal-  !
    39)          ! ity, recruitment and some phenology schemes.                                    !
    40)          !---------------------------------------------------------------------------------!
    41)          do ipa = 1,csite%npatches
    42)             csite%avg_daily_temp(ipa) = csite%avg_daily_temp(ipa) * tfact
    43)          end do
    44)          
    45)          select case (iphen_scheme)
    46)          case (-1,0,2)
    47)             !------------------------------------------------------------------------------!
    48)             !     Default predictive scheme (Botta et al.) or the modified drought         !
    49)             ! deciduous phenology for broadleaf PFTs.                                      !
    50)             !------------------------------------------------------------------------------!
    51)             call update_thermal_sums(month, cpoly, isi, cgrid%lat(ipy))
    52)             call update_phenology(doy,cpoly,isi,cgrid%lat(ipy))
    53)             
    54)          case (1)
    55)             !----- Use prescribed phenology. ----------------------------------------------!
    56)             call prescribed_leaf_state(cgrid%lat(ipy), current_time%month                  &
    57)                                       ,current_time%year, doy                              &
    58)                                       ,cpoly%green_leaf_factor(:,isi)                      &
    59)                                       ,cpoly%leaf_aging_factor(:,isi),cpoly%phen_pars(isi)) 
    60)             call update_phenology(doy,cpoly,isi,cgrid%lat(ipy))
    61) 
    62) 
    63)          case (3)
    64)             !----- KIM light-controlled predictive phenology scheme. ----------------------!
    65)             call update_thermal_sums(month, cpoly, isi, cgrid%lat(ipy))
    66)             call update_turnover(cpoly,isi)
    67)             call update_phenology(doy,cpoly,isi,cgrid%lat(ipy))
    68)          end select
    69)       end do
    70)    end do
    71) 
    72)    return
    73) end subroutine phenology_driver

    86) !------------------------------------------------------------------------------------------!
    87) subroutine phenology_driver_eq_0(cgrid, doy, month, tfact)
    88)    use ed_state_vars  , only : edtype                & ! structure
    89)                              , polygontype           & ! structure
    90)                              , sitetype              ! ! structure
    91)    use phenology_coms , only : iphen_scheme          ! ! intent(in)
    92)    use ed_misc_coms   , only : current_time          ! ! intent(in)
    93)    use phenology_aux  , only : prescribed_leaf_state & ! subroutine
    94)                              , update_thermal_sums   & ! subroutine
    95)                              , update_turnover       ! ! subroutine
    96)    implicit none
    97)    !----- Arguments -----------------------------------------------------------------------!
    98)    type(edtype)      , target      :: cgrid
    99)    integer           , intent(in)  :: doy
   100)    integer           , intent(in)  :: month
   101)    real              , intent(in)  :: tfact
   102)    !----- Local variables -----------------------------------------------------------------!
   103)    type(polygontype) , pointer     :: cpoly
   104)    type(sitetype)    , pointer     :: csite
   105)    integer                         :: ipy
   106)    integer                         :: isi
   107)    integer                         :: ipa
   108)    !---------------------------------------------------------------------------------------!
   109) 
   110)    do ipy = 1,cgrid%npolygons
   111)       cpoly => cgrid%polygon(ipy)
   112) 
   113)       do isi = 1,cpoly%nsites
   114)          csite => cpoly%site(isi)
   115) 
   116)          !---------------------------------------------------------------------------------!
   117)          !     Get the patch-level average daily temperature, which is needed for mortal-  !
   118)          ! ity, recruitment and some phenology schemes.                                    !
   119)          !---------------------------------------------------------------------------------!
   120)          do ipa = 1,csite%npatches
   121)             csite%avg_daily_temp(ipa) = csite%avg_daily_temp(ipa) * tfact
   122)          end do
   123)          
   124)          select case (iphen_scheme)
   125)          case (-1,0,2)
   126)             !------------------------------------------------------------------------------!
   127)             !     Default predictive scheme (Botta et al.) or the modified drought         !
   128)             ! deciduous phenology for broadleaf PFTs.                                      !
   129)             !------------------------------------------------------------------------------!
   130)             call update_thermal_sums(month, cpoly, isi, cgrid%lat(ipy))
   131)             call update_phenology_eq_0(doy,cpoly,isi,cgrid%lat(ipy))
   132)             
   133)          case (1)
   134)             !----- Use prescribed phenology. ----------------------------------------------!
   135)             call prescribed_leaf_state(cgrid%lat(ipy), current_time%month                  &
   136)                                       ,current_time%year, doy                              &
   137)                                       ,cpoly%green_leaf_factor(:,isi)                      &
   138)                                       ,cpoly%leaf_aging_factor(:,isi),cpoly%phen_pars(isi)) 
   139)             call update_phenology_eq_0(doy,cpoly,isi,cgrid%lat(ipy))
   140) 
   141)          case (3)
   142)             !----- KIM light-controlled predictive phenology scheme. ----------------------!
   143)             call update_thermal_sums(month, cpoly, isi, cgrid%lat(ipy))
   144)             call update_turnover(cpoly,isi)
   145)             call update_phenology_eq_0(doy,cpoly,isi,cgrid%lat(ipy))
   146)          end select
   147)       end do
   148)    end do
   149) 
   150)    return
   151) end subroutine phenology_driver_eq_0

   162) subroutine update_phenology(doy, cpoly, isi, lat)
   163)    use stable_cohorts
   164)    use ed_state_vars  , only : polygontype              & ! structure
   165)                              , sitetype                 & ! structure
   166)                              , patchtype                ! ! structure
   167)    use grid_coms      , only : nzg                      ! ! intent(in)
   168)    use pft_coms       , only : phenology                & ! intent(in)
   169)                              , c2n_leaf                 & ! intent(in)
   170)                              , q                        & ! intent(in)
   171)                              , qsw                      & ! intent(in)
   172)                              , l2n_stem                 & ! intent(in)
   173)                              , c2n_stem                 & ! intent(in)
   174)                              , c2n_storage              ! ! intent(in)
   175)    use decomp_coms    , only : f_labile                 ! ! intent(in)
   176)    use phenology_coms , only : retained_carbon_fraction & ! intent(in)
   177)                              , iphen_scheme             & ! intent(in)
   178)                              , elongf_min               & ! intent(in)
   179)                              , elongf_flush             ! ! intent(in)
   180)    use consts_coms    , only : t3ple                    & ! intent(in)
   181)                              , cice                     & ! intent(in)
   182)                              , cliq                     & ! intent(in)
   183)                              , alli                     ! ! intent(in)
   184)    use ed_therm_lib   , only : calc_veg_hcap            & ! function
   185)                              , update_veg_energy_cweh   ! ! subroutine
   186)    use ed_max_dims    , only : n_pft                    ! ! intent(in)
   187)    use ed_misc_coms   , only : current_time             ! ! intent(in)
   188)    use allometry      , only : area_indices             & ! subroutine
   189)                              , ed_biomass               & ! function
   190)                              , size2bl                  ! ! function
   191)    use phenology_aux  , only : daylength                ! ! function
   192)    implicit none
   193)    !----- Arguments -----------------------------------------------------------------------!
   194)    type(polygontype)        , target     :: cpoly
   195)    real                     , intent(in) :: lat
   196)    integer                  , intent(in) :: doy    ! Day of year (1=Jan 1, 365/366=Dec 31)
   197)    integer                  , intent(in) :: isi
   198)    !----- Local variables -----------------------------------------------------------------!
   199)    type(sitetype)           , pointer    :: csite
   200)    type(patchtype)          , pointer    :: cpatch
   201)    integer                               :: ipa
   202)    integer                               :: ico
   203)    integer                               :: isoil_lev
   204)    integer                               :: kroot
   205)    integer                               :: ipft
   206)    logical                               :: leaf_out_cold
   207)    logical                               :: drop_cold
   208)    real                                  :: daylight
   209)    real                                  :: delta_bleaf
   210)    real                                  :: bl_max
   211)    real                                  :: old_leaf_hcap
   212)    real                                  :: old_wood_hcap
   213)    real                                  :: salloci
   214)    real                                  :: elongf_try
   215)    real                                  :: elongf_grow
   216)    !----- Variables used only for debugging purposes. -------------------------------------!
   217)    logical                  , parameter  :: printphen=.false.
   218)    logical, dimension(n_pft), save       :: first_time=.true.
   219)    !---------------------------------------------------------------------------------------!
   220) 
   221) 
   222)    !----- Level to evaluate the soil temperature. -----------------------------------------!
   223)    isoil_lev = nzg 
   224) 
   225)    !----- Calculate daylength for this gridcell. ------------------------------------------!
   226)    daylight = daylength(lat, doy) 
   227) 
   228)    !----- Loop over patches. --------------------------------------------------------------!
   229)    csite => cpoly%site(isi)
   230) 
   231)    patchloop: do ipa = 1,csite%npatches
   232)       cpatch => csite%patch(ipa)
   233) 
   234)       !----- Re-initialize litter inputs. -------------------------------------------------!
   235)       csite%fsc_in(ipa) = 0.0
   236)       csite%fsn_in(ipa) = 0.0
   237)       csite%ssc_in(ipa) = 0.0
   238)       csite%ssl_in(ipa) = 0.0
   239) 
   240)       !----- Determine what phenology thresholds have been crossed. -----------------------!
   241)       call phenology_thresholds(daylight,csite%soil_tempk(isoil_lev,ipa),csite%sum_chd(ipa)&
   242)                                ,csite%sum_dgd(ipa),drop_cold,leaf_out_cold)
   243) 
   244)       cohortloop: do ico = 1,cpatch%ncohorts
   245)          ipft    = cpatch%pft(ico)
   246)          kroot   = cpatch%krdepth(ico)
   247)          salloci = 1. / (1.0 + qsw(ipft) * cpatch%hite(ico) + q(ipft))
   248)          
   249)          !----- Initially, we assume all leaves stay. -------------------------------------!
   250)          cpatch%leaf_drop(ico) = 0.0
   251) 
   252)          !----- Find cohort-specific thresholds. ------------------------------------------!
   253)          select case (iphen_scheme)
   254)          case (1)
   255)             !----- Get cohort-specific thresholds for prescribed phenology. ---------------!
   256)             call assign_prescribed_phen(cpoly%green_leaf_factor(ipft,isi)                  &
   257)                                        ,cpoly%leaf_aging_factor(ipft,isi),cpatch%dbh(ico)  &
   258)                                        ,cpatch%hite(ico),ipft,drop_cold,leaf_out_cold      &
   259)                                        ,bl_max)
   260)          case default
   261)             !----- Drop_cold is computed in phenology_thresholds for Botta scheme. --------!
   262)             if (drop_cold) bl_max = 0.0
   263)          end select
   264) 
   265)          !---------------------------------------------------------------------------------!
   266)          !---------------------------------------------------------------------------------!
   267)          !     Here we decide what to do depending on the phenology habit. There are five  !
   268)          ! different types:                                                                !
   269)          ! 0. Evergreen         - neither cold nor drought makes these plants to drop      !
   270)          !                        their leaves;                                            !
   271)          ! 1. Drought deciduous - these plants will drop all leaves when drought           !
   272)          !                        conditions happen. By drought conditions we mean a time  !
   273)          !                        when the available water drops below a threshold;        !
   274)          ! 2. Cold deciduous    - these plants will drop their leaves when cold conditions !
   275)          !                        happen.                                                  !
   276)          ! 3. Light phenology   - these plants will control their leaf structure with      !
   277)          !                        the light history (last 10 days).  On top of that, they  !
   278)          !                        are also drought-deciduous, similar to phenology 4;      !
   279)          ! 4. Drought deciduous - similar to one, but the threshold is compared against    !
   280)          !                        a 10-day running average rather than the instantaneous   !
   281)          !                        value.                                                   !
   282)          !---------------------------------------------------------------------------------!
   283)          select case (phenology(ipft))
   284)          case (0)
   285)             !------------------------------------------------------------------------------!
   286)             !    Evergreen, there is nothing to be done here except to assign maximum      !
   287)             ! elongation factor.                                                           !
   288)             !------------------------------------------------------------------------------!
   289)             cpatch%elongf(ico)    = 1.0
   290)             !------------------------------------------------------------------------------!
   291) 
   292)          case (1)
   293)             !------------------------------------------------------------------------------!
   294)             !     Drought deciduous.  Now we must check whether the plants still have      !
   295)             ! enough water or it is too dry, or if there is some drought relief so leaves  !
   296)             ! can start to grow again.                                                     ! 
   297)             !------------------------------------------------------------------------------!
   298)             !----- This is the first guess for the new elongation factor. -----------------!
   299)             elongf_try  = max(0.0, min (1.0, cpatch%paw_avg(ico)))
   300)             !------------------------------------------------------------------------------!
   301) 
   302) 
   303) 
   304)             if (elongf_try < 1.0 .and. cpatch%phenology_status(ico) /= -2) then
   305)                !----- It is time to drop leaves.  Drop all leaves. ------------------------!
   306)                cpatch%leaf_drop(ico) = (1.0 - retained_carbon_fraction) * cpatch%bleaf(ico)
   307)                !---------------------------------------------------------------------------!
   308) 
   309) 
   310) 
   311)                !----- Update plant carbon pools. ------------------------------------------!
   312)                cpatch%balive  (ico) = cpatch%balive(ico) - cpatch%bleaf(ico)
   313)                cpatch%bstorage(ico) = cpatch%bstorage(ico)                                 &
   314)                                     + cpatch%bleaf(ico) * retained_carbon_fraction
   315)                cpatch%bleaf   (ico) = 0.0
   316)                cpatch%elongf  (ico) = 0.0
   317)                cpatch%phenology_status(ico) = -2
   318)                !---------------------------------------------------------------------------!
   319) 
   320) 
   321) 
   322)                !---------------------------------------------------------------------------!
   323)                !     Send the lost leaves to soil carbon and nitrogen pools.               !
   324)                !---------------------------------------------------------------------------!
   325)                csite%fsc_in(ipa) = csite%fsc_in(ipa)                                       &
   326)                                  + cpatch%nplant(ico) * cpatch%leaf_drop(ico)              &
   327)                                  * f_labile(ipft)
   328)                csite%fsn_in(ipa) = csite%fsn_in(ipa)                                       &
   329)                                  + cpatch%nplant(ico) * cpatch%leaf_drop(ico)              &
   330)                                  * f_labile(ipft) / c2n_leaf(ipft)
   331)                csite%ssc_in(ipa) = csite%ssc_in(ipa)                                       &
   332)                                  + cpatch%nplant(ico) * cpatch%leaf_drop(ico)              &
   333)                                  * (1.0 - f_labile(ipft))
   334)                csite%ssl_in(ipa) = csite%ssl_in(ipa)                                       &
   335)                                  + cpatch%nplant(ico) * cpatch%leaf_drop(ico)              &
   336)                                  * (1.0 - f_labile(ipft)) * l2n_stem / c2n_stem(ipft)
   337)                !---------------------------------------------------------------------------!
   338) 
   339) 
   340) 
   341)                !---------------------------------------------------------------------------!
   342)                !     Contribution due to the fact that c2n_leaf and c2n_storage may be     !
   343)                ! different.                                                                !
   344)                !---------------------------------------------------------------------------!
   345)                csite%fsn_in(ipa)            = csite%fsn_in(ipa)                            &
   346)                                             + cpatch%leaf_drop(ico) * cpatch%nplant(ico)   &
   347)                                             * (1.0 / c2n_leaf(ipft) - 1.0 / c2n_storage)
   348)                !---------------------------------------------------------------------------!
   349) 
   350) 
   351)                !---------------------------------------------------------------------------!
   352)                !      Deduct the leaf drop from the carbon balance.                        !
   353)                !---------------------------------------------------------------------------!
   354)                cpatch%cb          (13,ico)  = cpatch%cb          (13,ico)                  &
   355)                                             - cpatch%leaf_drop      (ico)
   356)                cpatch%cb_lightmax (13,ico)  = cpatch%cb_lightmax (13,ico)                  &
   357)                                             - cpatch%leaf_drop      (ico)
   358)                cpatch%cb_moistmax (13,ico)  = cpatch%cb_moistmax (13,ico)                  &
   359)                                             - cpatch%leaf_drop      (ico)
   360)                cpatch%cb_mlmax (13,ico)     = cpatch%cb_mlmax (13,ico)                     &
   361)                                             - cpatch%leaf_drop (ico)
   362)                
   363) 
   364) 
   365)                !---------------------------------------------------------------------------!
   366) 
   367)             elseif(elongf_try > 1.0 .and. cpatch%phenology_status(ico) == -2) then
   368)                !---------------------------------------------------------------------------!
   369)                !      It is time to flush.  Change phenology_status will update carbon     !
   370)                ! pools in growth_balive.                                                   !
   371)                !---------------------------------------------------------------------------!
   372)                cpatch%phenology_status(ico) = 1
   373)                cpatch%elongf          (ico) = 1.0
   374)                !---------------------------------------------------------------------------!
   375)             end if  ! critical moisture
   376) 
   377)          case (2)
   378)             !------------------------------------------------------------------------------!
   379)             !    Cold deciduous.  Here we must check two possibilities:                    !
   380)             !                                                                              !
   381)             ! 1. It is cold, and the plants have leaves, so we flag them with              !
   382)             !    phenology_status=0 (leaves not growing) and the plants will start losing  !
   383)             !    their leaves;                                                             !
   384)             ! 2. The plant has no leaves, but the temperature and light conditions are     !
   385)             !    okay again, and leaves can start growing.                                 !
   386)             !------------------------------------------------------------------------------!
   387)             if (cpatch%phenology_status(ico) /= -2 .and. drop_cold) then
   388)                if (cpoly%green_leaf_factor(ipft,isi) < elongf_min) then
   389)                    bl_max = 0.0
   390)                end if
   391)                delta_bleaf = cpatch%bleaf(ico) - bl_max
   392) 
   393)                if (delta_bleaf > 0.0) then
   394)                   !------------------------------------------------------------------------!
   395)                   !    Phenology_status = 0 means that the plant has leaves, but they are  !
   396)                   ! not growing (not necessarily because the leaves are fully flushed).    !
   397)                   !------------------------------------------------------------------------!
   398)                   cpatch%phenology_status(ico) = -1
   399)                   cpatch%leaf_drop(ico) = (1.0 - retained_carbon_fraction) * delta_bleaf
   400)                   csite%fsc_in(ipa) = csite%fsc_in(ipa)                                    &
   401)                                     + cpatch%nplant(ico) * cpatch%leaf_drop(ico)           &
   402)                                     * f_labile(ipft)
   403)                   csite%fsn_in(ipa) = csite%fsn_in(ipa)                                    &
   404)                                     + cpatch%nplant(ico) * cpatch%leaf_drop(ico)           &
   405)                                     * f_labile(ipft) / c2n_leaf(ipft)
   406)                   csite%ssc_in(ipa) = csite%ssc_in(ipa)                                    &
   407)                                     + cpatch%nplant(ico) * cpatch%leaf_drop(ico)           &
   408)                                     * (1.0 - f_labile(ipft))
   409)                   csite%ssl_in(ipa) = csite%ssl_in(ipa)                                    &
   410)                                     + cpatch%nplant(ico) * cpatch%leaf_drop(ico)           &
   411)                                     * (1.0 - f_labile(ipft)) * l2n_stem / c2n_stem(ipft)
   412)                   
   413)                   !----- Adjust plant carbon pools. ---------------------------------------!
   414)                   cpatch%balive(ico)   = cpatch%balive(ico) - delta_bleaf
   415)                   cpatch%bstorage(ico) = cpatch%bstorage(ico)                              &
   416)                                        + retained_carbon_fraction * delta_bleaf
   417) 
   418)                   !------------------------------------------------------------------------!
   419)                   !     Contribution due to the fact that c2n_leaf and c2n_storage may be  !
   420)                   ! different.                                                             !
   421)                   !------------------------------------------------------------------------!
   422)                   csite%fsn_in(ipa)     = csite%fsn_in(ipa)                                &
   423)                                         + delta_bleaf * cpatch%nplant(ico)                 &
   424)                                         * retained_carbon_fraction                         &
   425)                                         * (1.0 / c2n_leaf(ipft) - 1.0/c2n_storage)
   426)                   cpatch%bleaf(ico)     = bl_max
   427) 
   428)                   !------------------------------------------------------------------------!
   429)                   !      Deduct the leaf drop from the carbon balance.                     !
   430)                   !------------------------------------------------------------------------!
   431)                   cpatch%cb          (13,ico) = cpatch%cb         (13,ico)                 &
   432)                                               - cpatch%leaf_drop     (ico)
   433)                   cpatch%cb_lightmax (13,ico) = cpatch%cb_lightmax(13,ico)                 &
   434)                                               - cpatch%leaf_drop     (ico)
   435)                   cpatch%cb_moistmax (13,ico) = cpatch%cb_moistmax(13,ico)                 &
   436)                                               - cpatch%leaf_drop     (ico)
   437)                   cpatch%cb_mlmax (13,ico)    = cpatch%cb_mlmax (13,ico) &
   438)                                               - cpatch%leaf_drop (ico)
   439) 
   440) 
   441)                   !------------------------------------------------------------------------!
   442)                end if
   443) 
   444)                !----- Set status flag. ----------------------------------------------------!
   445)                if (bl_max == 0.0) then
   446)                   cpatch%phenology_status(ico) = -2
   447)                   cpatch%elongf(ico) = 0.
   448)                else
   449)                   cpatch%elongf(ico) = 1.0 ! It should become green_leaf_factor...
   450)                end if
   451)                
   452)             elseif (.not. drop_cold .and. cpatch%phenology_status(ico) == -2               &
   453)                     .and. leaf_out_cold) then
   454)                !---------------------------------------------------------------------------!
   455)                !      Update the phenology status (1 means that leaves are growing),       !
   456)                !---------------------------------------------------------------------------!
   457)                cpatch%phenology_status(ico) = 1
   458)                cpatch%elongf          (ico) = 1.0 ! It should become green_leaf_factor...
   459)                !---------------------------------------------------------------------------!
   460)             end if
   461) 
   462) 
   463)          case (3,4) 
   464)             !------------------------------------------------------------------------------!
   465)             !    Drought deciduous.  Here we must check two possibilities:                 !
   466)             !                                                                              !
   467)             ! 1. The soil has been dry recently, and the plants have leaves, so we flag    !
   468)             !    them with phenology_status=0 (leaves not growing) and the plants will     !
   469)             !    start losing their leaves;                                                !
   470)             ! 2. The plant has no leaves, but the soil has started to come back to more    !
   471)             !    moist conditions. Given this situation, leaves can start growing again.   !
   472)             !------------------------------------------------------------------------------!
   473)             !----- This is the first guess for the new elongation factor. -----------------!
   474)             elongf_try  = max(0.0, min (1.0, cpatch%paw_avg(ico)))
   475)             !----- If extremely dry, force the cohort to shed all leaves... ---------------!
   476)             if (elongf_try < elongf_min) elongf_try = 0.0
   477)             !------------------------------------------------------------------------------!
   478) 
   479) 
   480) 
   481)             !----- Find the maximum allowed leaf biomass. ---------------------------------!
   482)             bl_max = elongf_try * size2bl(cpatch%dbh(ico),cpatch%hite(ico),ipft)
   483)             !------------------------------------------------------------------------------!
   484) 
   485) 
   486) 
   487)             !------------------------------------------------------------------------------!
   488)             !     Delta_bleaf is the difference between the current leaf biomass and the   !
   489)             ! maximum permitted given the soil moisture conditions.  If delta_bleaf is     !
   490)             ! positive, it means that the plant has more leaves than it should.            !
   491)             !------------------------------------------------------------------------------!
   492)             delta_bleaf = cpatch%bleaf(ico) - bl_max
   493)             !------------------------------------------------------------------------------!
   494) 
   495) 
   496) 
   497)             !------------------------------------------------------------------------------!
   498)             !     Check whether drought is becoming more or less severe.                   !
   499)             !------------------------------------------------------------------------------!
   500)             if (delta_bleaf > 0.0) then
   501)                !---------------------------------------------------------------------------!
   502)                !    Drought conditions are becoming more severe, drop leaves.              !
   503)                !---------------------------------------------------------------------------!
   504)                if (elongf_try >= elongf_min) then
   505)                   cpatch%phenology_status(ico) = -1
   506)                else
   507)                   cpatch%phenology_status(ico) = -2
   508)                end if
   509)                cpatch%leaf_drop (ico) = (1.0 - retained_carbon_fraction) * delta_bleaf
   510)                cpatch%elongf    (ico) = elongf_try
   511)                !----- Adjust plant carbon pools. ------------------------------------------!
   512)                cpatch%bleaf     (ico) = bl_max
   513)                cpatch%balive    (ico) = cpatch%balive(ico)   - delta_bleaf
   514)                cpatch%bstorage  (ico) = cpatch%bstorage(ico)                               &
   515)                                       + retained_carbon_fraction * delta_bleaf
   516)                !---------------------------------------------------------------------------!
   517) 
   518) 
   519) 
   520)                !---------------------------------------------------------------------------!
   521)                !     Send the lost leaves to soil carbon and nitrogen pools.               !
   522)                !---------------------------------------------------------------------------!
   523)                csite%fsc_in           (ipa) = csite%fsc_in(ipa)                            &    
   524)                                             + cpatch%nplant(ico) * cpatch%leaf_drop(ico)   &    
   525)                                             * f_labile(ipft)                                
   526)                csite%fsn_in           (ipa) = csite%fsn_in(ipa)                            &    
   527)                                             + cpatch%nplant(ico) * cpatch%leaf_drop(ico)   &    
   528)                                             * f_labile(ipft) / c2n_leaf(ipft)               
   529)                csite%ssc_in           (ipa) = csite%ssc_in(ipa)                            &    
   530)                                             + cpatch%nplant(ico) * cpatch%leaf_drop(ico)   &    
   531)                                             * (1.0-f_labile(ipft))                          
   532)                csite%ssl_in           (ipa) = csite%ssl_in(ipa)                            &    
   533)                                             + cpatch%nplant(ico) * cpatch%leaf_drop(ico)   &    
   534)                                             * (1.0 - f_labile(ipft)) * l2n_stem            &
   535)                                             / c2n_stem(ipft)
   536)                !---------------------------------------------------------------------------!
   537) 
   538) 
   539)                !---------------------------------------------------------------------------!
   540)                !     Contribution due to the fact that c2n_leaf and c2n_storage may be     !
   541)                ! different.                                                                !
   542)                !---------------------------------------------------------------------------!
   543)                csite%fsn_in(ipa)     = csite%fsn_in(ipa) + delta_bleaf*cpatch%nplant(ico)  &
   544)                                      * retained_carbon_fraction                            &
   545)                                      * (1.0 / c2n_leaf(ipft) - 1.0/c2n_storage)
   546)                !---------------------------------------------------------------------------!
   547) 
   548)                !---------------------------------------------------------------------------!
   549)                !      Deduct the leaf drop from the carbon balance.                        !
   550)                !---------------------------------------------------------------------------!
   551)                cpatch%cb          (13,ico) = cpatch%cb          (13,ico)                   &
   552)                                            - cpatch%leaf_drop      (ico)
   553)                cpatch%cb_lightmax (13,ico) = cpatch%cb_lightmax (13,ico)                   &
   554)                                            - cpatch%leaf_drop      (ico)
   555)                cpatch%cb_moistmax (13,ico) = cpatch%cb_moistmax (13,ico)                   &
   556)                                            - cpatch%leaf_drop      (ico)
   557)                cpatch%cb_mlmax (13,ico)    = cpatch%cb_mlmax    (13,ico)                   &
   558)                                            - cpatch%leaf_drop      (ico)
   559) 
   560)                !---------------------------------------------------------------------------!
   561)             elseif (cpatch%phenology_status(ico) /= 0) then
   562)                !---------------------------------------------------------------------------!
   563)                !       Elongation factor could increase, but we first check whether it is  !
   564)                ! safe to do so based on the phenology status.                              !
   565)                !---------------------------------------------------------------------------!
   566)                select case(cpatch%phenology_status(ico))
   567)                case (1)
   568)                   !----- Leaves were already growing, keep growing. -----------------------!
   569)                   cpatch%elongf          (ico) = elongf_try
   570)                   !------------------------------------------------------------------------!
   571)                case (-1,-2)
   572)                   !------------------------------------------------------------------------!
   573)                   !     Leaves were dropping or gone, we first check that conditions are   !
   574)                   ! really improving before we turn on leaf production.                    !
   575)                   !------------------------------------------------------------------------!
   576)                   elongf_grow = min(1.0,max(elongf_flush,cpatch%elongf(ico)+0.02))
   577)                   if (elongf_try >= elongf_grow) then
   578)                      cpatch%elongf          (ico) = elongf_try
   579)                      cpatch%phenology_status(ico) = 1
   580)                   end if
   581)                   !------------------------------------------------------------------------!
   582)                end select
   583)                !---------------------------------------------------------------------------!
   584)             end if
   585)             !------------------------------------------------------------------------------!
   586)          end select
   587)          !---------------------------------------------------------------------------------!
   588) 
   589) 
   590) 
   591) 
   592)          !----- Update LAI, WAI, and CAI accordingly. -------------------------------------!
   593)          call area_indices(cpatch, ico)
   594)          !---------------------------------------------------------------------------------!
   595) 
   596) 
   597) 
   598) 
   599)          !----- Update above-ground biomass. ----------------------------------------------!
   600)          cpatch%agb(ico) = ed_biomass(cpatch, ico)
   601) 
   602)          !---------------------------------------------------------------------------------!
   603)          !    The leaf biomass of the cohort has changed, update the vegetation energy -   !
   604)          ! using a constant temperature assumption.                                        !
   605)          !---------------------------------------------------------------------------------!
   606)          old_leaf_hcap       = cpatch%leaf_hcap(ico)
   607)          old_wood_hcap       = cpatch%wood_hcap(ico)
   608)          call calc_veg_hcap(cpatch%bleaf(ico),cpatch%bdead(ico),cpatch%bsapwooda(ico)      &
   609)                            ,cpatch%nplant(ico),cpatch%pft(ico)                             &
   610)                            ,cpatch%leaf_hcap(ico),cpatch%wood_hcap(ico) )
   611)          call update_veg_energy_cweh(csite,ipa,ico,old_leaf_hcap,old_wood_hcap)
   612)          call is_resolvable(csite,ipa,ico)
   613)          !---------------------------------------------------------------------------------!
   614) 
   615)          !----- Printing some debugging stuff if the code is set for it. ------------------!
   616)          if (printphen) then
   617)             ipft=cpatch%pft(ico)
   618)             if (first_time(ipft)) then
   619)                first_time(ipft) = .false.
   620)                write (unit=40+ipft,fmt='(a10,6(1x,a12))')                                  &
   621)                   &'      TIME','       PATCH','      COHORT','      NPLANT'               &
   622)                   &            ,'   LEAF_DROP','     PAW_AVG','      ELONGF'
   623)             end if
   624)          
   625)             write (unit=40+ipft,fmt='(2(i2.2,a1),i4.4,2(1x,i12),5(1x,es12.5))')            &
   626)                  current_time%month,'/',current_time%date,'/',current_time%year,ipa,ico    &
   627)                 ,cpatch%nplant(ico),cpatch%leaf_drop(ico),cpatch%paw_avg(ico)              &
   628)                 ,cpatch%elongf(ico)
   629)          end if
   630)       end do cohortloop
   631)    end do patchloop
   632)    return
   633) end subroutine update_phenology
   634) !==========================================================================================!
   635) !==========================================================================================!

   642) !==========================================================================================!
   643) !==========================================================================================!
   644) !     This subroutine will make a simpler version of the leaf phenology:  it will change   !
   645) ! bleaf according to cold or drought, but it won't check whether there is enough carbon    !
   646) ! to change it, it will simply change it to the target bleaf.                              !
   647) !------------------------------------------------------------------------------------------!
   648) subroutine update_phenology_eq_0(doy, cpoly, isi, lat)
   649)    use stable_cohorts
   650)    use ed_state_vars  , only : polygontype              & ! structure
   651)                              , sitetype                 & ! structure
   652)                              , patchtype                ! ! structure
   653)    use grid_coms      , only : nzg                      ! ! intent(in)
   654)    use pft_coms       , only : phenology                & ! intent(in)
   655)                              , q                        & ! intent(in)
   656)                              , qsw                      ! ! intent(in)
   657)    use phenology_coms , only : retained_carbon_fraction & ! intent(in)
   658)                              , iphen_scheme             & ! intent(in)
   659)                              , elongf_min               & ! intent(in)
   660)                              , elongf_flush             ! ! intent(in)
   661)    use ed_therm_lib   , only : calc_veg_hcap            & ! function
   662)                              , update_veg_energy_cweh   ! ! subroutine
   663)    use ed_max_dims    , only : n_pft                    ! ! intent(in)
   664)    use allometry      , only : area_indices             & ! subroutine
   665)                              , ed_biomass               & ! function
   666)                              , size2bl                  ! ! function
   667)    use phenology_aux  , only : daylength                ! ! function
   668) 
   669)    implicit none
   670)    !----- Arguments -----------------------------------------------------------------------!
   671)    type(polygontype)        , target     :: cpoly
   672)    real                     , intent(in) :: lat
   673)    integer                  , intent(in) :: doy    ! Day of year (1=Jan 1, 365/366=Dec 31)
   674)    integer                  , intent(in) :: isi
   675)    !----- Local variables -----------------------------------------------------------------!
   676)    type(sitetype)           , pointer    :: csite
   677)    type(patchtype)          , pointer    :: cpatch
   678)    integer                               :: ipa
   679)    integer                               :: ico
   680)    integer                               :: isoil_lev
   681)    integer                               :: kroot
   682)    integer                               :: ipft
   683)    logical                               :: leaf_out_cold
   684)    logical                               :: drop_cold
   685)    real, dimension(nzg)                  :: theta
   686)    real                                  :: daylight
   687)    real                                  :: delta_bleaf
   688)    real                                  :: bleaf_new
   689)    real                                  :: old_leaf_hcap
   690)    real                                  :: old_wood_hcap
   691)    real                                  :: salloci
   692)    real                                  :: elongf_try
   693)    real                                  :: elongf_grow
   694)    !---------------------------------------------------------------------------------------!
   695) 
   696) 
   697)    !----- Level to evaluate the soil temperature. -----------------------------------------!
   698)    isoil_lev = nzg 
   699) 
   700)    !----- Calculate daylength for this gridcell. ------------------------------------------!
   701)    daylight = daylength(lat, doy) 
   702) 
   703)    !----- Loop over patches. --------------------------------------------------------------!
   704)    csite => cpoly%site(isi)
   705) 
   706)    patchloop: do ipa = 1,csite%npatches
   707)       cpatch => csite%patch(ipa)
   708) 
   709)       !----- Determine what phenology thresholds have been crossed. -----------------------!
   710)       call phenology_thresholds(daylight,csite%soil_tempk(isoil_lev,ipa)                   &
   711)                                ,csite%sum_chd(ipa),csite%sum_dgd(ipa),drop_cold            &
   712)                                ,leaf_out_cold)
   713) 
   714)       cohortloop: do ico = 1,cpatch%ncohorts
   715)          ipft    = cpatch%pft(ico)
   716)          kroot   = cpatch%krdepth(ico)
   717)          salloci = 1. / (1.0 + qsw(ipft) * cpatch%hite(ico) + q(ipft))
   718)          
   719)          !----- Initially, we assume all leaves stay. -------------------------------------!
   720)          cpatch%leaf_drop(ico) = 0.0
   721) 
   722)          !----- Find cohort-specific thresholds. ------------------------------------------!
   723)          select case (iphen_scheme)
   724)          case (1)
   725)             !----- Get cohort-specific thresholds for prescribed phenology. ---------------!
   726)             call assign_prescribed_phen(cpoly%green_leaf_factor(ipft,isi)                  &
   727)                                        ,cpoly%leaf_aging_factor(ipft,isi),cpatch%dbh(ico)  &
   728)                                        ,cpatch%hite(ico),ipft,drop_cold,leaf_out_cold      &
   729)                                        ,bleaf_new)
   730)          case default
   731)             !----- Drop_cold is computed in phenology_thresholds for Botta scheme. --------!
   732)             if (drop_cold) bleaf_new = 0.0
   733)          end select
   734) 
   735)          !---------------------------------------------------------------------------------!
   736)          !---------------------------------------------------------------------------------!
   737)          !     Here we decide what to do depending on the phenology habit. There are five  !
   738)          ! different types:                                                                !
   739)          ! 0. Evergreen         - neither cold nor drought makes these plants to drop      !
   740)          !                        their leaves;                                            !
   741)          ! 1. Drought deciduous - these plants will drop all leaves when drought           !
   742)          !                        conditions happen. By drought conditions we mean a time  !
   743)          !                        when the available water drops below a threshold;        !
   744)          ! 2. Cold deciduous    - these plants will drop their leaves when cold conditions !
   745)          !                        happen.                                                  !
   746)          ! 3. Light phenology   - these plants will control their leaf structure with      !
   747)          !                        the light history (last 10 days).  On top of that, they  !
   748)          !                        are also drought-deciduous, similar to phenology 4;      !
   749)          ! 4. Drought deciduous - similar to one, but the threshold is compared against    !
   750)          !                        a 10-day running average rather than the instantaneous   !
   751)          !                        value.                                                   !
   752)          !---------------------------------------------------------------------------------!
   753)          select case (phenology(ipft))
   754)          case (0)
   755)             !------------------------------------------------------------------------------!
   756)             !    Evergreen, there is nothing to be done here, assign elongation factor to  !
   757)             ! maximum and move on.                                                         !
   758)             !------------------------------------------------------------------------------!
   759)             cpatch%elongf(ico)    = 1.0
   760) 
   761)          case (1)
   762)             !------------------------------------------------------------------------------!
   763)             !     Drought deciduous.  Now we must check whether the plants still have      !
   764)             ! enough water or it is too dry, or if there is some drought relief so leaves  !
   765)             ! can start to grow again.                                                     ! 
   766)             !------------------------------------------------------------------------------!
   767)             if (theta(kroot) < 1.0) then
   768) 
   769)                !----- It is time to drop leaves. ------------------------------------------!
   770)                if (cpatch%phenology_status(ico) /= -2) then
   771)                   !----- Find the leaf drop. ----------------------------------------------!
   772)                   cpatch%leaf_drop(ico) = (1.0 - retained_carbon_fraction)                 &
   773)                                         * cpatch%lai(ico) / cpatch%sla(ico)                &
   774)                                         / cpatch%nplant(ico)
   775)                   !------------------------------------------------------------------------!
   776) 
   777)                   !----- Update plant carbon pools. ---------------------------------------!
   778)                   cpatch%balive              (ico) = cpatch%balive(ico)                    &
   779)                                                    - cpatch%bleaf(ico)
   780)                   cpatch%bleaf               (ico) = 0.0
   781)                   cpatch%phenology_status    (ico) = -2
   782)                   cpatch%elongf              (ico) = 0.
   783)                   cpatch%cb               (13,ico) = cpatch%cb           (13,ico)          &
   784)                                                    - cpatch%leaf_drop       (ico)
   785)                   cpatch%cb_lightmax      (13,ico) = cpatch%cb_lightmax  (13,ico)          &
   786)                                                    - cpatch%leaf_drop       (ico)
   787)                   cpatch%cb_moistmax      (13,ico) = cpatch%cb_moistmax  (13,ico)          &
   788)                                                    - cpatch%leaf_drop       (ico)
   789)                   cpatch%cb_mlmax         (13,ico) = cpatch%cb_mlmax     (13,ico)          &
   790)                                                    - cpatch%leaf_drop       (ico)
   791)                   !------------------------------------------------------------------------!
   792)                end if
   793)                
   794)             elseif(theta(kroot) > 1.0 .and. cpatch%phenology_status(ico) == -2) then
   795)                
   796)                !---------------------------------------------------------------------------! 
   797)                !    It is time to flush.  Change phenology_status here, and make all       !
   798)                ! leaves appear all of a sudden.                                            !
   799)                !---------------------------------------------------------------------------! 
   800)                cpatch%phenology_status(ico) = 1
   801)                cpatch%bleaf(ico)            = size2bl(cpatch%dbh(ico),cpatch%hite(ico),ipft)
   802)                cpatch%balive(ico)           = cpatch%balive(ico) + cpatch%bleaf(ico)
   803)                cpatch%elongf(ico)           = 1.0
   804)             end if  ! critical moisture
   805) 
   806)          case (2)
   807)             !------------------------------------------------------------------------------!
   808)             !    Cold deciduous.  Here we must check two possibilities:                    !
   809)             !                                                                              !
   810)             ! 1. It is cold, and the plants have leaves, so we flag them with              !
   811)             !    phenology_status=0 (leaves not growing) and the plants will start losing  !
   812)             !    their leaves;                                                             !
   813)             ! 2. The plant has no leaves, but the temperature and light conditions are     !
   814)             !    okay again, and leaves can start growing.                                 !
   815)             !------------------------------------------------------------------------------!
   816)             if (cpatch%phenology_status(ico) /= -2 .and. drop_cold) then
   817)             
   818)                if (cpoly%green_leaf_factor(ipft,isi) < elongf_min) then
   819)                   bleaf_new = 0.0
   820)                else
   821)                   bleaf_new = cpoly%green_leaf_factor(ipft,isi)                            &
   822)                             * size2bl(cpatch%dbh(ico),cpatch%hite(ico),ipft)
   823)                end if
   824)                
   825)                delta_bleaf = cpatch%bleaf(ico) - bleaf_new
   826) 
   827)                if (delta_bleaf > 0.0) then
   828)                   !------------------------------------------------------------------------!
   829)                   !    Phenology_status = 0 means that the plant has leaves, but they are  !
   830)                   ! not growing (not necessarily because the leaves are fully flushed).    !
   831)                   !------------------------------------------------------------------------!
   832)                   cpatch%phenology_status(ico) = -1
   833)                   cpatch%leaf_drop(ico) = (1.0 - retained_carbon_fraction) * delta_bleaf
   834)                   !------------------------------------------------------------------------!
   835) 
   836) 
   837) 
   838)                   !----- Adjust plant carbon pools. ---------------------------------------!
   839)                   cpatch%balive(ico) = cpatch%balive(ico) - delta_bleaf
   840)                   !------------------------------------------------------------------------!
   841) 
   842) 
   843) 
   844)                   !------------------------------------------------------------------------!
   845)                   !     Contribution due to the fact that c2n_leaf and c2n_storage may be  !
   846)                   ! different.                                                             !
   847)                   !------------------------------------------------------------------------!
   848)                   cpatch%bleaf          (ico) = bleaf_new
   849)                   cpatch%cb          (13,ico) = cpatch%cb          (13,ico)                &
   850)                                               - cpatch%leaf_drop      (ico)
   851)                   cpatch%cb_lightmax (13,ico) = cpatch%cb_lightmax (13,ico)                &
   852)                                               - cpatch%leaf_drop      (ico)
   853)                   cpatch%cb_moistmax (13,ico) = cpatch%cb_moistmax (13,ico)                &
   854)                                               - cpatch%leaf_drop      (ico)
   855)                   cpatch%cb_mlmax    (13,ico) = cpatch%cb_mlmax    (13,ico)                &
   856)                                               - cpatch%leaf_drop      (ico)
   857)                   !------------------------------------------------------------------------!
   858)                end if
   859) 
   860)                !----- Set status flag. ----------------------------------------------------!
   861)                if (bleaf_new == 0.0) then
   862)                   cpatch%phenology_status(ico) = -2
   863)                   cpatch%elongf          (ico) = 0.
   864)                else
   865)                   cpatch%elongf          (ico) = 1.
   866)                end if
   867)                !---------------------------------------------------------------------------!
   868) 
   869) 
   870)             elseif (.not. drop_cold .and. cpatch%phenology_status(ico) == -2               &
   871)                     .and. leaf_out_cold) then
   872)                !---------------------------------------------------------------------------!
   873)                !      Update the phenology status (1 means that leaves are growing),       !
   874)                !---------------------------------------------------------------------------!
   875)                cpatch%phenology_status(ico) = 1
   876)                cpatch%elongf          (ico) = 1.
   877)                cpatch%bleaf(ico)            = cpoly%green_leaf_factor(ipft,isi)            &
   878)                                             * size2bl(cpatch%dbh(ico),cpatch%hite(ico),ipft)
   879)                cpatch%balive(ico)           = cpatch%balive(ico) + cpatch%bleaf(ico)
   880)                !---------------------------------------------------------------------------!
   881)             end if
   882) 
   883) 
   884)          case (3,4) 
   885)             !------------------------------------------------------------------------------!
   886)             !    Drought deciduous.  Here we must check two possibilities:                 !
   887)             !                                                                              !
   888)             ! 1. The soil has been dry recently, and the plants have leaves, so we flag    !
   889)             !    them with phenology_status=0 (leaves not growing) and the plants will     !
   890)             !    start losing their leaves;                                                !
   891)             ! 2. The plant has no leaves, but the soil has started to come back to more    !
   892)             !    moist conditions. Given this situation, leaves can start growing again.   !
   893)             !------------------------------------------------------------------------------!
   894)             elongf_try = max(0.0, min (1.0, cpatch%paw_avg(ico)))
   895) 
   896)             !----- In case it is too dry, drop all the leaves. ----------------------------!
   897)             if (elongf_try < elongf_min) elongf_try = 0.
   898)             !----- Find the new leaf biomass. ---------------------------------------------!
   899)             bleaf_new = elongf_try * size2bl(cpatch%dbh(ico),cpatch%hite(ico),ipft)
   900) 
   901)             !------------------------------------------------------------------------------!
   902)             !     Delta_bleaf is the difference between the current leaf biomass and the   !
   903)             ! maximum permitted given the soil moisture conditions.  If delta_bleaf is     !
   904)             ! positive, it means that the plant has more leaves than it should.            !
   905)             !------------------------------------------------------------------------------!
   906)             delta_bleaf = cpatch%bleaf(ico) - bleaf_new
   907)             !------------------------------------------------------------------------------!
   908) 
   909) 
   910) 
   911)             !------------------------------------------------------------------------------!
   912)             !     Check whether plants have any leaves.                                    !
   913)             !------------------------------------------------------------------------------!
   914)             if (delta_bleaf > 0.0) then
   915)                !---------------------------------------------------------------------------!
   916)                !    Drought conditions are becoming more severe, drop leaves.              !
   917)                !---------------------------------------------------------------------------!
   918)                if (elongf_try > elongf_min) then
   919)                   cpatch%phenology_status(ico) = -1
   920)                else
   921)                   cpatch%phenology_status(ico) = -2
   922)                end if
   923)                cpatch%leaf_drop(ico) = (1.0 - retained_carbon_fraction) * delta_bleaf
   924)                cpatch%elongf   (ico) = elongf_try
   925)                !----- Adjust plant carbon pools. ------------------------------------------!
   926)                cpatch%balive   (ico) = cpatch%balive(ico) - delta_bleaf
   927)                cpatch%bleaf    (ico) = bleaf_new
   928)                cpatch%bstorage (ico) = cpatch%bstorage(ico)                                &
   929)                                      + retained_carbon_fraction * delta_bleaf
   930)                !---------------------------------------------------------------------------!
   931) 
   932)                !---------------------------------------------------------------------------!
   933)                !      Deduct the leaf drop from the carbon balance.                        !
   934)                !---------------------------------------------------------------------------!
   935)                cpatch%cb          (13,ico) = cpatch%cb          (13,ico)                   &
   936)                                            - cpatch%leaf_drop      (ico)
   937)                cpatch%cb_lightmax (13,ico) = cpatch%cb_lightmax (13,ico)                   &
   938)                                            - cpatch%leaf_drop      (ico)
   939)                cpatch%cb_moistmax (13,ico) = cpatch%cb_moistmax (13,ico)                   &
   940)                                            - cpatch%leaf_drop      (ico)
   941)                cpatch%cb_mlmax    (13,ico) = cpatch%cb_mlmax    (13,ico)                   &
   942)                                            - cpatch%leaf_drop      (ico)
   943)                !---------------------------------------------------------------------------!
   944)             elseif (cpatch%phenology_status(ico) /= 0) then
   945)                select case(cpatch%phenology_status(ico))
   946)                case (1)
   947)                   !----- Leaves were already growing, keep growing. -----------------------!
   948)                   cpatch%elongf          (ico) = elongf_try
   949)                   !----- Check whether leaves are fully flushed. --------------------------!
   950)                   if (elongf_try == 1.0) then
   951)                      cpatch%phenology_status(ico) = 0
   952)                   else
   953)                      cpatch%phenology_status(ico) = 1
   954)                   end if
   955)                   !------------------------------------------------------------------------!
   956)                case (-1,-2)
   957)                   !------------------------------------------------------------------------!
   958)                   !     Leaves were dropping or gone, we first check that conditions are   !
   959)                   ! really improving before we turn on leaf production.                    !
   960)                   !------------------------------------------------------------------------!
   961)                   elongf_grow = min(1.0,max(elongf_flush,cpatch%elongf(ico)+0.02))
   962)                   if (elongf_try >= elongf_grow) then

   967)                      else

   969)                      end if
   970)                      !---------------------------------------------------------------------!
   971)                   end if
   972)                   !------------------------------------------------------------------------!
   973)                end select
   974)                !---------------------------------------------------------------------------!
   975) 
   976) 
   977)                !---------------------------------------------------------------------------!
   978)                !     Conditions are slightly more humid.  Let them grow.                   !
   979)                !---------------------------------------------------------------------------!
   980)                cpatch%bleaf           (ico) = bleaf_new
   981)                cpatch%balive          (ico) = cpatch%balive(ico) - delta_bleaf
   982)                !---------------------------------------------------------------------------!
   983)             end if
   984)             !------------------------------------------------------------------------------!
   985)          end select

   993)

   995)          cpatch%agb(ico) = ed_biomass(cpatch, ico)
   996)

   999)          ! using a constant temperature assumption.                                        !

  1001)          old_leaf_hcap       = cpatch%leaf_hcap(ico)
  1002)          old_wood_hcap       = cpatch%wood_hcap(ico)
  1003)          call calc_veg_hcap(cpatch%bleaf(ico),cpatch%bdead(ico),cpatch%bsapwooda(ico)      &
  1004)                            ,cpatch%nplant(ico),cpatch%pft(ico)                             &
  1005)                            ,cpatch%leaf_hcap(ico),cpatch%wood_hcap(ico) )
  1006)          call update_veg_energy_cweh(csite,ipa,ico,old_leaf_hcap,old_wood_hcap)
  1007)          call is_resolvable(csite,ipa,ico)
  1008)       end do cohortloop
  1009)    end do patchloop
