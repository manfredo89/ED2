    21)    !  SUBROUTINE: DBALIVE_DT
    22)    !
    23)    !> \brief   Updates living biomass.
    24)    !> \details Calls a variety of subroutines controlling plant carbon balances, C and N
    25)    !>          xfers among plant pools, updating of storage and growth respiration,
    26)    !>          leaf maintenance,and update mortality rates.
    27)    !> \author  Translated from ED1 by Ryan Knox and Marcos Longo
    28)    !> \author  31 Aug 2015 - Big refactoring in commit b8fb585, Daniel Scott
    29)    !> \warning The order of the operations here affect the C/N budgets, so don't
    30)    !>          change it unless you really know what you are doing.
    31)    !---------------------------------------------------------------------------------------!
    32)    subroutine dbalive_dt(cgrid, tfact)
    33)       use stable_cohorts
    34)       use update_derived_props_module
    35)       use ed_state_vars   , only : edtype                 & ! structure
    36)                                  , polygontype            & ! structure
    37)                                  , sitetype               & ! structure
    38)                                  , patchtype              ! ! structure
    39)       use pft_coms        , only : q                      & ! intent(in)
    40)                                  , qsw                    & ! intent(in)
    41)                                  , plant_N_supply_scale   & ! intent(in)
    42)                                  , c2n_storage            & ! intent(in)
    43)                                  , growth_resp_factor     & ! intent(in)
    44)                                  , storage_turnover_rate  & ! intent(in)
    45)                                  , is_grass               ! ! intent(in)
    46)       use physiology_coms , only : N_plant_lim            ! ! intent(in)
    47)       use ed_therm_lib    , only : calc_veg_hcap          & ! function
    48)                                  , update_veg_energy_cweh ! ! function
    49)       use allometry       , only : area_indices           & ! subroutine
    50)                                  , ed_biomass             ! ! function
    51)       use mortality       , only : mortality_rates        ! ! subroutine
    52)       use fuse_fiss_utils , only : sort_cohorts           ! ! subroutine
    53)       use ed_misc_coms    , only : igrass                 & ! intent(in)
    54)                                  , growth_resp_scheme     & ! intent(in)
    55)                                  , storage_resp_scheme    ! ! intent(in)
    56)       use budget_utils    , only : update_budget          ! ! sub-routine
    57)       use consts_coms   , only : tiny_num     ! ! intent(in)
    58) 
    59)       implicit none
    60)       !----- Arguments. -------------------------------------------------------------------!
    61)       type(edtype)     , target     :: cgrid !< the ed grid
    62)       real             , intent(in) :: tfact !< "time factor" i.e. call frequency
    63)       !----- Local variables. -------------------------------------------------------------!
    64)       type(polygontype), pointer    :: cpoly
    65)       type(sitetype)   , pointer    :: csite
    66)       type(patchtype)  , pointer    :: cpatch
    67)       integer                       :: ipy
    68)       integer                       :: isi
    69)       integer                       :: ipa
    70)       integer                       :: ico
    71)       integer                       :: ipft
    72)       real                          :: salloc
    73)       real                          :: salloci
    74)       real                          :: daily_C_gain
    75)       real                          :: carbon_balance
    76)       real                          :: carbon_balance_pot
    77)       real                          :: carbon_balance_lightmax
    78)       real                          :: carbon_balance_moistmax
    79)       real                          :: carbon_balance_mlmax
    80)       real                          :: balive_in
    81)       real                          :: nitrogen_supply
    82)       real                          :: dndt
    83)       real                          :: dlnndt
    84)       real                          :: old_leaf_hcap
    85)       real                          :: old_wood_hcap
    86)       real                          :: nitrogen_uptake
    87)       real                          :: N_uptake_pot
    88)       real                          :: temp_dep
    89)       real                          :: growth_resp_int          ! Growth resp / balive
    90)       real                          :: storage_resp_int         ! Growth resp / balive
    91)       real                          :: tr_bleaf
    92)       real                          :: tr_broot
    93)       real                          :: tr_bsapwooda
    94)       real                          :: tr_bsapwoodb
    95)       real                          :: tr_bstorage
    96)       real                          :: cb_decrement
    97)       real                          :: carbon_debt
    98)       real                          :: balive_aim
    99)       logical                       :: flushing
   100)       logical                       :: on_allometry
   101)       !------------------------------------------------------------------------------------!
   102) 
   103) 
   104)       polyloop: do ipy = 1,cgrid%npolygons
   105)          cpoly => cgrid%polygon(ipy)
   106) 
   107)          siteloop: do isi = 1,cpoly%nsites
   108)             csite => cpoly%site(isi)
   109) 
   110)             patchloop: do ipa = 1,csite%npatches
   111)                cpatch => csite%patch(ipa)
   112) 
   113)                !----- Reset averaged variables. -------------------------------------------!
   114)                csite%total_plant_nitrogen_uptake(ipa) = 0.0
   115) 
   116)                !----- Loop over cohorts. --------------------------------------------------!
   117)                cohortloop: do ico = 1,cpatch%ncohorts
   118) 
   119)                   !----- Alias for current PFT. -------------------------------------------!
   120)                   ipft = cpatch%pft(ico)
   121) 
   122)                   !----- Initialize cohort nitrogen uptake. -------------------------------!
   123)                   nitrogen_uptake = 0.0
   124)                   N_uptake_pot    = 0.0
   125) 
   126)                   !----- Set allocation factors. ------------------------------------------!
   127)                   salloc  = 1.0 + qsw(ipft) * cpatch%hite(ico) + q(ipft)
   128)                   salloci = 1.0 / salloc
   129) 
   130)                   !------------------------------------------------------------------------!
   131)                   !     Compute and apply maintenance costs and get daily C gain.          !
   132)                   !------------------------------------------------------------------------!
   133)                   call get_maintenance(cpatch,ico,tfact,csite%avg_daily_temp(ipa))
   134)                   call get_daily_C_gain(cpatch,ico,daily_C_gain)
   135)                   call apply_maintenance(cpatch,ico,cb_decrement)
   136) 
   137)                   call update_cb(cpatch,ico,cb_decrement)
   138) 
   139)                   !------------------------------------------------------------------------!
   140)                   !    Storage respiration/turnover_rate.                                  !
   141)                   !    Calculate in same way as leaf and root turnover in kgC/plant/year.  !
   142)                   !------------------------------------------------------------------------!
   143) 
   144) 
   145)                   !------------------------------------------------------------------------!
   146)                   !     The commented line is an experimental and arbitrary test, borrowed !
   147)                   ! from maintainence temperature dependency. [[MCD]]                      !
   148)                   !------------------------------------------------------------------------!
   149)                   ! temp_dep = 1.0                                                         &
   150)                   !          / ( 1.0  + exp( 0.4 * (278.15 - csite%avg_daily_temp(ipa))))
   151)                   temp_dep = 1.0
   152)                   !------------------------------------------------------------------------!
   153)                   select case(storage_resp_scheme)
   154)                   case(0)
   155)                      cpatch%leaf_storage_resp(ico) = 0.0
   156)                      cpatch%root_storage_resp(ico) = 0.0
   157)                      cpatch%sapa_storage_resp(ico) = cpatch%bstorage(ico)                  &
   158)                                                    * storage_turnover_rate(ipft)           &
   159)                                                    * tfact * temp_dep
   160)                      cpatch%sapb_storage_resp(ico) = 0.0
   161) 
   162)                      cpatch%bstorage(ico) = cpatch%bstorage(ico)                           &
   163)                                             - cpatch%sapa_storage_resp(ico)
   164)                   case(1)
   165)                      storage_resp_int = cpatch%bstorage(ico) / cpatch%balive(ico)          &
   166)                                       * storage_turnover_rate(ipft) * tfact * temp_dep
   167) 
   168)                      cpatch%leaf_storage_resp(ico) = storage_resp_int *cpatch%bleaf(ico)
   169)                      cpatch%root_storage_resp(ico) = storage_resp_int *cpatch%broot(ico)
   170)                      cpatch%sapa_storage_resp(ico) = storage_resp_int *cpatch%bsapwooda(ico)
   171)                      cpatch%sapb_storage_resp(ico) = storage_resp_int *cpatch%bsapwoodb(ico)
   172) 
   173)                      cpatch%bstorage(ico) = cpatch%bstorage(ico)                           &
   174)                                           - cpatch%leaf_storage_resp(ico)                  &
   175)                                           - cpatch%root_storage_resp(ico)                  &
   176)                                           - cpatch%sapa_storage_resp(ico)                  &
   177)                                           - cpatch%sapb_storage_resp(ico)
   178)                   end select
   179) 
   180)                   !------------------------------------------------------------------------!
   181)                   !     When storage carbon is lost, allow the associated nitrogen to go   !
   182)                   ! to litter in order to maintain prescribed C2N ratio.                   !
   183)                   !------------------------------------------------------------------------!
   184)                   csite%fsn_in(ipa) = csite%fsn_in(ipa) + (cpatch%leaf_storage_resp(ico)   &
   185)                                                         +  cpatch%root_storage_resp(ico)   &
   186)                                                         +  cpatch%sapa_storage_resp(ico)   &
   187)                                                         +  cpatch%sapb_storage_resp(ico))  &
   188)                                                         / c2n_storage * cpatch%nplant(ico)
   189) 
   190)                   !------------------------------------------------------------------------!
   191)                   !      Calculate actual, potential and maximum carbon balances.          !
   192)                   !------------------------------------------------------------------------!
   193)                   call plant_carbon_balances(cpatch,ipa,ico,daily_C_gain,carbon_balance    &
   194)                                             ,carbon_balance_pot,carbon_balance_lightmax    &
   195)                                             ,carbon_balance_moistmax,carbon_balance_mlmax)
   196)                   !------------------------------------------------------------------------!
   197) 
   198) 
   199) 
   200)                   !------------------------------------------------------------------------!
   201)                   !      Compute respiration rates for coming day [kgC/plant/day].         !
   202)                   !------------------------------------------------------------------------!
   203)                   select case(growth_resp_scheme)
   204)                   case(0)
   205)                      cpatch%sapa_growth_resp(ico) = max(0.0, daily_C_gain                  &
   206)                                                              * growth_resp_factor(ipft))
   207)                      cpatch%leaf_growth_resp(ico) = 0.0
   208)                      cpatch%root_growth_resp(ico) = 0.0
   209)                      cpatch%sapb_growth_resp(ico) = 0.0
   210)                   case(1)
   211)                      growth_resp_int = max(0.0, daily_C_gain * growth_resp_factor(ipft)    &
   212)                                                              / cpatch%balive(ico))
   213) 
   214)                      cpatch%leaf_growth_resp(ico) = growth_resp_int * cpatch%bleaf(ico)
   215)                      cpatch%root_growth_resp(ico) = growth_resp_int * cpatch%broot(ico)
   216)                      cpatch%sapa_growth_resp(ico) = growth_resp_int * cpatch%bsapwooda(ico)
   217)                      cpatch%sapb_growth_resp(ico) = growth_resp_int * cpatch%bsapwoodb(ico)
   218)                   end select
   219)                   !------------------------------------------------------------------------!
   220) 
   221) 
   222) 
   223)                   !------------------------------------------------------------------------!
   224)                   !      Allocate plant carbon balance to balive and bstorage.             !
   225)                   !------------------------------------------------------------------------!
   226)                   balive_in = cpatch%balive(ico)
   227) 
   228)                   if(is_grass(ipft).and. igrass==1) then
   229)                       call alloc_plant_c_balance_grass(csite,ipa,ico,salloci               &
   230)                                                 ,carbon_balance,nitrogen_uptake            &
   231)                                                 ,cpoly%green_leaf_factor(ipft,isi))
   232)                       call sort_cohorts(cpatch)
   233)                   else
   234)                      call get_c_xfers(csite,ipa,ico,carbon_balance                         &
   235)                                      ,cpoly%green_leaf_factor(ipft,isi),tr_bleaf,tr_broot  &
   236)                                      ,tr_bsapwooda,tr_bsapwoodb,tr_bstorage,carbon_debt    &
   237)                                      ,flushing,balive_aim)
   238) 
   239)                      call apply_c_xfers(cpatch,ico,carbon_balance,tr_bleaf,tr_broot        &
   240)                                        ,tr_bsapwooda,tr_bsapwoodb,tr_bstorage)
   241) 
   242)                      call update_today_npp_vars(cpatch,ico,tr_bleaf,tr_broot,tr_bsapwooda  &
   243)                                                 ,tr_bsapwoodb,carbon_balance)
   244) 
   245)                      call update_nitrogen(flushing,ipft,carbon_balance,cpatch%nplant(ico)  &
   246)                                           ,tr_bleaf,tr_broot,tr_bstorage,nitrogen_uptake   &
   247)                                           ,csite%fsn_in(ipa))
   248) 
   249)                      !---------------------------------------------------------------------!
   250)                      !                  Update the phenology status.                       !
   251)                      !---------------------------------------------------------------------!
   252)                      on_allometry = balive_aim - cpatch%balive(ico) <= 0.000001*balive_aim
   253)                      if (flushing .and. cpatch%elongf(ico) == 1.0 .and. on_allometry) then
   254)                         cpatch%phenology_status(ico) = 0
   255)                      elseif(cpatch%bleaf(ico) < tiny_num .and.                             &
   256)                             (cpatch%phenology_status(ico) == 0 .or.                        &
   257)                              cpatch%phenology_status(ico) == 1)) then
   258)                         cpatch%phenology_status(ico) = 1
   259)                      end if
   260)                   end if
   261)                   !------------------------------------------------------------------------!
   262) 
   263) 
   264) 
   265)                   !------------------------------------------------------------------------!
   266)                   !     Do a shadow calculation to see what would have happened if stomata !
   267)                   ! were open.  This is used to calculate potential nitrogen uptake,       !
   268)                   ! N_uptake_pot.                                                          !
   269)                   !------------------------------------------------------------------------!
   270)                   if (N_plant_lim == 1) then
   271)                      call potential_N_uptake(cpatch,ico,carbon_balance_pot,N_uptake_pot    &
   272)                                             ,cpoly%green_leaf_factor(ipft,isi))
   273)                   end if
   274)                   !------------------------------------------------------------------------!
   275) 
   276) 
   277) 
   278)                   !------------------------------------------------------------------------!
   279)                   !  Increment the [kgN/m2] taken up during previous day.                  !
   280)                   !------------------------------------------------------------------------!
   281)                   csite%total_plant_nitrogen_uptake(ipa) =                                 &
   282)                                        csite%total_plant_nitrogen_uptake(ipa)              &
   283)                                      + nitrogen_uptake * cpatch%nplant(ico)
   284)                   !------------------------------------------------------------------------!
   285) 
   286) 
   287) 
   288)                   !----- Calculate plant N limitation factor. -----------------------------!
   289)                   if (n_plant_lim == 0 .or. N_uptake_pot <= 0.0) then
   290)                      cpatch%fsn(ico) = 1.0
   291)                   else
   292)                      nitrogen_supply = plant_N_supply_scale * cpatch%broot(ico)            &
   293)                                      * csite%mineralized_soil_N(ipa)
   294)                      cpatch%fsn(ico) = nitrogen_supply                                     &
   295)                                      / (nitrogen_supply + N_uptake_pot)
   296)                   end if
   297) 
   298)                   !------------------------------------------------------------------------!
   299)                   !      Update mortality rates.  Notice that the only mortality rate that !
   300)                   ! changes daily is the frost mortality, and the disturbance mortality is !
   301)                   ! not updated here (it is updated in the main disturbance procedure).    !
   302)                   !                                                                        !
   303)                   !      We no longer include mortality rates due to disturbance in the    !
   304)                   ! big-leaf simulations, this is now done at disturbance.f90.             !
   305)                   !------------------------------------------------------------------------!
   306)                   call mortality_rates(cpatch,ico,csite%avg_daily_temp(ipa),csite%age(ipa))
   307)                   dlnndt   = - sum(cpatch%mort_rate(1:4,ico))
   308)                   dndt     = dlnndt * cpatch%nplant(ico)
   309)                   !------------------------------------------------------------------------!
   310) 
   311)                   !----- Update monthly mortality rates [plants/m2/month and 1/month]. ----!
   312)                   cpatch%monthly_dndt  (ico) = cpatch%monthly_dndt  (ico) + dndt   * tfact
   313)                   cpatch%monthly_dlnndt(ico) = cpatch%monthly_dlnndt(ico) + dlnndt * tfact
   314)                   !------------------------------------------------------------------------!
   315) 
   316) 
   317)                   !----- Updating LAI, WAI, and CAI. --------------------------------------!
   318)                   call area_indices(cpatch, ico)
   319)                   !------------------------------------------------------------------------!
   320) 
   321) 
   322) 
   323)                   !----- Update above-ground biomass. -------------------------------------!
   324)                   cpatch%agb(ico) = ed_biomass(cpatch, ico)
   325)                   !------------------------------------------------------------------------!
   326) 
   327) 
   328) 
   329)                   !------------------------------------------------------------------------!
   330)                   !     It is likely that biomass has changed, therefore, update           !
   331)                   ! vegetation energy and heat capacity.                                   !
   332)                   !------------------------------------------------------------------------!
   333)                   old_leaf_hcap         = cpatch%leaf_hcap(ico)
   334)                   old_wood_hcap         = cpatch%wood_hcap(ico)
   335)                   call calc_veg_hcap(cpatch%bleaf(ico) ,cpatch%bdead(ico)                  &
   336)                                     ,cpatch%bsapwooda(ico),cpatch%nplant(ico)               &
   337)                                     ,cpatch%pft(ico)                                       &
   338)                                     ,cpatch%leaf_hcap(ico),cpatch%wood_hcap(ico))
   339)                   call update_veg_energy_cweh(csite,ipa,ico,old_leaf_hcap,old_wood_hcap)
   340)                   !----- Update the stability status. -------------------------------------!
   341)                   call is_resolvable(csite,ipa,ico)
   342)                   !------------------------------------------------------------------------!
   343)                end do cohortloop
   344)                !---------------------------------------------------------------------------!
   345) 
   346)                !----- Update litter. ------------------------------------------------------!
   347)                call litter(csite,ipa)
   348)                !---------------------------------------------------------------------------!
   349) 
   350)                !----- Update patch LAI, WAI, height, roughness... -------------------------!
   351)                call update_patch_derived_props(csite,ipa)
   352)                !---------------------------------------------------------------------------!
   353) 
   354)                !----- Recalculate storage terms (for budget assessment). ------------------!
   355)                call update_budget(csite,cpoly%lsl(isi),ipa)
   356)                !---------------------------------------------------------------------------!
   357) 
   358)                !----- It's a new day, reset average daily temperature. --------------------!
   359)                csite%avg_daily_temp(ipa) = 0.0
   360)                !---------------------------------------------------------------------------!
   361)             end do patchloop
   362)             !------------------------------------------------------------------------------!
   363)          end do siteloop
   364)          !---------------------------------------------------------------------------------!
   365)       end do polyloop
   366)       !------------------------------------------------------------------------------------!
   367) 
   368)       return
   369)    end subroutine dbalive_dt

   379)    !     This subroutine will compute the terms relative to growth of the living tissues,  !
   380)    ! without actually updating it.                                                         !
   381)    ! IMPORTANT: The order of the operations here affect the C/N budgets, so don't change   !
   382)    !            the order of the operations unless you really know what you are doing.     !
   383)    !---------------------------------------------------------------------------------------!
   384)    subroutine dbalive_dt_eq_0(cgrid, tfact)
   385)       use ed_state_vars   , only : edtype                 & ! structure
   386)                                  , polygontype            & ! structure
   387)                                  , sitetype               & ! structure
   388)                                  , patchtype              ! ! structure
   389)       use pft_coms        , only : q                      & ! intent(in)
   390)                                  , qsw                    & ! intent(in)
   391)                                  , plant_N_supply_scale   & ! intent(in)
   392)                                  , growth_resp_factor     & ! intent(in)
   393)                                  , storage_turnover_rate  ! ! intent(in)
   394)       use physiology_coms , only : N_plant_lim            ! ! intent(in)
   395)       use ed_therm_lib    , only : calc_veg_hcap          & ! function
   396)                                  , update_veg_energy_cweh ! ! function
   397)       use ed_misc_coms    , only : ibigleaf               & ! intent(in)
   398)                                  , growth_resp_scheme     & ! intent(in)
   399)                                  , storage_resp_scheme    ! ! intent(in)
   400)       use mortality       , only : mortality_rates        ! ! subroutine
   401)       implicit none
   402)       !----- Arguments. -------------------------------------------------------------------!
   403)       type(edtype)     , target     :: cgrid
   404)       real             , intent(in) :: tfact
   405)       !----- Local variables. -------------------------------------------------------------!
   406)       type(polygontype), pointer    :: cpoly
   407)       type(sitetype)   , pointer    :: csite
   408)       type(patchtype)  , pointer    :: cpatch
   409)       integer                       :: ipy
   410)       integer                       :: isi
   411)       integer                       :: ipa
   412)       integer                       :: ico
   413)       integer                       :: ipft
   414)       real                          :: salloc
   415)       real                          :: salloci
   416)       real                          :: daily_C_gain
   417)       real                          :: carbon_balance
   418)       real                          :: carbon_balance_pot
   419)       real                          :: carbon_balance_lightmax
   420)       real                          :: carbon_balance_moistmax
   421)       real                          :: carbon_balance_mlmax
   422)       real                          :: balive_in
   423)       real                          :: nitrogen_supply
   424)       real                          :: dndt
   425)       real                          :: dlnndt
   426)       real                          :: nitrogen_uptake
   427)       real                          :: N_uptake_pot
   428)       real                          :: temp_dep
   429)       real                          :: growth_resp_int         ! Growth resp / balive
   430)       real                          :: storage_resp_int        ! Growth resp / balive
   431)       !------------------------------------------------------------------------------------!
   432) 
   433) 
   434)       polyloop: do ipy = 1,cgrid%npolygons
   435)          cpoly => cgrid%polygon(ipy)
   436) 
   437)          siteloop: do isi = 1,cpoly%nsites
   438)             csite => cpoly%site(isi)
   439) 
   440)             patchloop: do ipa = 1,csite%npatches
   441)                cpatch => csite%patch(ipa)
   442) 
   443)                !----- Reset averaged variables. -------------------------------------------!
   444)                csite%total_plant_nitrogen_uptake(ipa) = 0.0
   445) 
   446)                !----- Loop over cohorts. --------------------------------------------------!
   447)                cohortloop: do ico = 1,cpatch%ncohorts
   448) 
   449)                   !----- Alias for current PFT. -------------------------------------------!
   450)                   ipft = cpatch%pft(ico)
   451) 
   452) 
   453)                   !----- Initialize cohort nitrogen uptake. -------------------------------!
   454)                   nitrogen_uptake = 0.0
   455)                   N_uptake_pot    = 0.0
   456) 
   457)                   !----- Set allocation factors. ------------------------------------------!
   458)                   salloc  = 1.0 + qsw(ipft) * cpatch%hite(ico) + q(ipft)
   459)                   salloci = 1.0 / salloc
   460) 
   461)                   !------------------------------------------------------------------------!
   462)                   !     Compute maintenance costs using actual pools.                      !
   463)                   !------------------------------------------------------------------------!
   464)                   call plant_maintenance(cpatch,ico,cpatch%broot(ico),cpatch%bleaf(ico)    &
   465)                                         ,tfact,daily_C_gain,csite%avg_daily_temp(ipa))
   466) 
   467)                   !------------------------------------------------------------------------!
   468)                   !    For the no vegetation dynamics case, we update the carbon balance   !
   469)                   ! but we do NOT update the living tissues.                               !
   470)                   !------------------------------------------------------------------------!
   471)                   cpatch%cb         (13,ico) = cpatch%cb                  (13,ico)         &
   472)                                              - cpatch%leaf_maintenance       (ico)         &
   473)                                              - cpatch%root_maintenance       (ico)
   474)                   cpatch%cb_lightmax(13,ico) = cpatch%cb_lightmax         (13,ico)         &
   475)                                              - cpatch%leaf_maintenance       (ico)         &
   476)                                              - cpatch%root_maintenance       (ico)
   477)                   cpatch%cb_moistmax(13,ico) = cpatch%cb_moistmax         (13,ico)         &
   478)                                              - cpatch%leaf_maintenance       (ico)         &
   479)                                              - cpatch%root_maintenance       (ico)
   480)                   cpatch%cb_mlmax(13,ico)    = cpatch%cb_mlmax            (13,ico)         &
   481)                                              - cpatch%leaf_maintenance       (ico)         &
   482)                                              - cpatch%root_maintenance       (ico)
   483)                   !------------------------------------------------------------------------!
   484) 
   485)                   !------------------------------------------------------------------------!
   486)                   !    Storage respriation/turnover_rate.                                  !
   487)                   !    Calculate in same way as leaf and root turnover in kgC/plant/year.  !
   488)                   !------------------------------------------------------------------------!
   489) 
   490) 
   491)                   !------------------------------------------------------------------------!
   492)                   !     The commented line is an experimental and arbitrary test, borrowed !
   493)                   ! from maintainence temperature dependency. [[MCD]]                      !
   494)                   !------------------------------------------------------------------------!
   495)                   ! temp_dep = 1.0                                                         &
   496)                   !          / ( 1.0  + exp( 0.4 * (278.15 - csite%avg_daily_temp(ipa))))
   497)                   temp_dep = 1.0
   498)                   !------------------------------------------------------------------------!
   499)                   select case(storage_resp_scheme)
   500)                   case(0)
   501)                      cpatch%sapa_storage_resp(ico) = cpatch%bstorage(ico)                  &
   502)                                                    * storage_turnover_rate(ipft)           &
   503)                                                    * tfact * temp_dep
   504)                      cpatch%leaf_storage_resp(ico) = 0.0
   505)                      cpatch%root_storage_resp(ico) = 0.0
   506)                      cpatch%sapb_storage_resp(ico) = 0.0
   507) 
   508)                      cpatch%bstorage(ico) = cpatch%bstorage(ico)                           &
   509)                                             - cpatch%sapa_storage_resp(ico)
   510)                   case(1)
   511)                      storage_resp_int = cpatch%bstorage(ico) * storage_turnover_rate(ipft) &
   512)                                       * tfact * temp_dep
   513) 
   514)                      cpatch%leaf_storage_resp(ico) = storage_resp_int *cpatch%bleaf(ico)
   515)                      cpatch%root_storage_resp(ico) = storage_resp_int *cpatch%broot(ico)
   516)                      cpatch%sapa_storage_resp(ico) = storage_resp_int *cpatch%bsapwooda(ico)
   517)                      cpatch%sapb_storage_resp(ico) = storage_resp_int *cpatch%bsapwoodb(ico)
   518) 
   519)                      cpatch%bstorage(ico) = cpatch%bstorage(ico)                           &
   520)                                             - cpatch%leaf_storage_resp(ico)                &
   521)                                             - cpatch%root_storage_resp(ico)                &
   522)                                             - cpatch%sapa_storage_resp(ico)                &
   523)                                             - cpatch%sapb_storage_resp(ico)
   524)                   end select
   525) 
   526)                   !------------------------------------------------------------------------!
   527)                   !      Calculate actual, potential and maximum carbon balances.          !
   528)                   !------------------------------------------------------------------------!
   529)                   call plant_carbon_balances(cpatch,ipa,ico,daily_C_gain,carbon_balance    &
   530)                                             ,carbon_balance_pot,carbon_balance_lightmax    &
   531)                                             ,carbon_balance_moistmax,carbon_balance_mlmax)
   532)                   !------------------------------------------------------------------------!
   533) 
   534) 
   535) 
   536)                   !------------------------------------------------------------------------!
   537)                   !      Compute respiration rates for coming day [kgC/plant/day].         !
   538)                   !------------------------------------------------------------------------!
   539)                   select case(growth_resp_scheme)
   540)                   case(0)
   541)                      cpatch%sapa_growth_resp(ico) = max(0.0, daily_C_gain                  &
   542)                                                              * growth_resp_factor(ipft))
   543)                      cpatch%leaf_growth_resp(ico) = 0.0
   544)                      cpatch%root_growth_resp(ico) = 0.0
   545)                      cpatch%sapb_growth_resp(ico) = 0.0
   546)                   case(1)
   547)                      growth_resp_int = max(0.0, daily_C_gain * growth_resp_factor(ipft)    &
   548)                                                              / cpatch%balive(ico))
   549) 
   550)                      cpatch%leaf_growth_resp(ico) = growth_resp_int * cpatch%bleaf(ico)
   551)                      cpatch%root_growth_resp(ico) = growth_resp_int * cpatch%broot(ico)
   552)                      cpatch%sapa_growth_resp(ico) = growth_resp_int * cpatch%bsapwooda(ico)
   553)                      cpatch%sapb_growth_resp(ico) = growth_resp_int * cpatch%bsapwoodb(ico)
   554)                   end select
   555)                   !------------------------------------------------------------------------!
   556) 
   557) 
   558) 
   559)                   !------------------------------------------------------------------------!
   560)                   !      Find the potential allocation to the living tissues, but don't    !
   561)                   ! update them.                                                           !
   562)                   !------------------------------------------------------------------------!
   563)                   balive_in = cpatch%balive(ico)
   564)                   call alloc_plant_c_balance_eq_0(csite,ipa,ico                            &
   565)                                                  ,carbon_balance,nitrogen_uptake           &
   566)                                                  ,cpoly%green_leaf_factor(ipft,isi))
   567)                   !------------------------------------------------------------------------!
   568) 
   569) 
   570) 
   571)                   !------------------------------------------------------------------------!
   572)                   !     Do a shadow calculation to see what would have happened if stomata !
   573)                   ! were open.  This is used to calculate potential nitrogen uptake,       !
   574)                   ! N_uptake_pot.                                                          !
   575)                   !------------------------------------------------------------------------!
   576)                   if (N_plant_lim == 1) then
   577)                      call potential_N_uptake(cpatch,ico,carbon_balance_pot,N_uptake_pot    &
   578)                                             ,cpoly%green_leaf_factor(ipft,isi))
   579)                   end if
   580)                   !------------------------------------------------------------------------!
   581) 
   582) 
   583) 
   584)                   !------------------------------------------------------------------------!
   585)                   !  Increment the [kgN/m2] taken up during previous day.                  !
   586)                   !------------------------------------------------------------------------!
   587)                   csite%total_plant_nitrogen_uptake(ipa) =                                 &
   588)                                        csite%total_plant_nitrogen_uptake(ipa)              &
   589)                                      + nitrogen_uptake * cpatch%nplant(ico)
   590)                   !------------------------------------------------------------------------!
   591) 
   592) 
   593) 
   594)                   !----- Calculate plant N limitation factor. -----------------------------!
   595)                   if (n_plant_lim == 0 .or. N_uptake_pot <= 0.0) then
   596)                      cpatch%fsn(ico) = 1.0
   597)                   else
   598)                      nitrogen_supply = plant_N_supply_scale * cpatch%broot(ico)            &
   599)                                      * csite%mineralized_soil_N(ipa)
   600)                      cpatch%fsn(ico) = nitrogen_supply                                     &
   601)                                      / (nitrogen_supply + N_uptake_pot)
   602)                   end if
   603) 
   604)                   !------------------------------------------------------------------------!
   605)                   !      Do mortality --- note that only frost mortality changes daily.    !
   606)                   !------------------------------------------------------------------------!
   607)                   call mortality_rates(cpatch,ico,csite%avg_daily_temp(ipa),csite%age(ipa))
   608)                   select case (ibigleaf)
   609)                   case (0)
   610)                      dlnndt   = - sum(cpatch%mort_rate(1:4,ico))
   611)                      dndt     = dlnndt * cpatch%nplant(ico)
   612)                   case (1)
   613)                      dlnndt   = - sum(cpatch%mort_rate(1:5,ico))
   614)                      dndt     = dlnndt * cpatch%nplant(ico)
   615)                   end select
   616)                   !------------------------------------------------------------------------!
   617) 
   618) 
   619)                   !----- Update monthly mortality rates [plants/m2/month and 1/month]. ----!
   620)                   cpatch%monthly_dndt  (ico) = cpatch%monthly_dndt  (ico) + dndt   * tfact
   621)                   cpatch%monthly_dlnndt(ico) = cpatch%monthly_dlnndt(ico) + dlnndt * tfact
   622)                   !------------------------------------------------------------------------!
   623) 
   624)                end do cohortloop
   625)                !---------------------------------------------------------------------------!
   626) 
   627)                !----- It's a new day, reset average daily temperature. --------------------!
   628)                csite%avg_daily_temp(ipa) = 0.0
   629)                !---------------------------------------------------------------------------!
   630)             end do patchloop
   631)             !------------------------------------------------------------------------------!
   632)          end do siteloop
   633)          !---------------------------------------------------------------------------------!
   634)       end do polyloop
   635)       !------------------------------------------------------------------------------------!
   636) 
   637)       return
   638)    end subroutine dbalive_dt_eq_0

   644)    subroutine get_maintenance(cpatch,ico,tfact,tempk)
   645)       use ed_state_vars, only : patchtype             ! ! structure
   646)       use pft_coms     , only : phenology             & ! intent(in)
   647)                               , root_turnover_rate    & ! intent(in)
   648)                               , leaf_turnover_rate    ! ! intent(in)
   649)       implicit none
   650)       !----- Arguments. -------------------------------------------------------------------!
   651)       type(patchtype), target       :: cpatch
   652)       integer        , intent(in)   :: ico
   653)       real           , intent(in)   :: tfact
   654)       real           , intent(in)   :: tempk
   655)       !----- Local variables. -------------------------------------------------------------!
   656)       integer                       :: ipft
   657)       real                          :: maintenance_temp_dep
   658)       !------------------------------------------------------------------------------------!
   659) 
   660)       !------ Alias for plant functional type. --------------------------------------------!
   661)       ipft = cpatch%pft(ico)
   662) 
   663)       !------------------------------------------------------------------------------------!
   664)       !      Find the maintenance costs.  This will depend on the type of phenology that   !
   665)       ! the PFT has.   The tfact term applied converts the maintenance rates to            !
   666)       ! [kgC/plant/day].                                                                   !
   667)       !------------------------------------------------------------------------------------!
   668)       cpatch%root_maintenance(ico) = root_turnover_rate(ipft) *cpatch%broot(ico) *tfact
   669)       cpatch%leaf_maintenance(ico) = leaf_turnover_rate(ipft) *cpatch%bleaf(ico) *tfact
   670) 
   671)       select case (phenology(ipft))
   672)       case (0)
   673)          !---------------------------------------------------------------------------------!
   674)          !     Evergreens, like pines.  The turnover rates will be adjusted by a function  !
   675)          ! of temperature, which approaches 0 as the temperature goes down.                !
   676)          !---------------------------------------------------------------------------------!
   677)          !------ Find a temperature dependence adjustment. --------------------------------!
   678)          maintenance_temp_dep = 1.0 / (1.0 + exp(0.4 * (278.15 - tempk)))
   679)          !----- Scale maintenance by biomass and apply the temperature correction. --------!
   680)          cpatch%leaf_maintenance(ico) = cpatch%leaf_maintenance(ico)* maintenance_temp_dep
   681)          cpatch%root_maintenance(ico) = cpatch%root_maintenance(ico)* maintenance_temp_dep
   682)          !---------------------------------------------------------------------------------!
   683) 
   684)       case (3)
   685)          !---------------------------------------------------------------------------------!
   686)          !      Light phenology.  Leaf turnover rate will be adjusted according to the     !
   687)          ! amplitude that comes from the dependence on the radiation (running average).    !
   688)          ! Roots are the same as the other plants that don't depend on temperature.        !
   689)          !---------------------------------------------------------------------------------!
   690)          cpatch%leaf_maintenance(ico) = cpatch%leaf_maintenance(ico)                       &
   691)                                       * cpatch%turnover_amp(ico)
   692)          !---------------------------------------------------------------------------------!
   693)       end select
   694)       !------------------------------------------------------------------------------------!
   695) 
   696)       return
   697)    end subroutine get_maintenance

   828)    subroutine update_nitrogen(flushing,ipft,carbon_balance,nplant,tr_bleaf,tr_broot        &
   829)                               ,tr_bstorage,nitrogen_uptake,fsn_in)
   830)       use consts_coms   , only : tiny_num                 ! ! intent(in)
   831)       use decomp_coms   , only : f_labile                 ! ! intent(in)
   832)       use pft_coms      , only : c2n_storage              & ! intent(in)
   833)                                , c2n_leaf                 & ! intent(in)
   834)                                , c2n_stem                 ! ! intent(in)
   835)       implicit none
   836)       !----- Arguments. -------------------------------------------------------------------!
   837)       logical        , intent(in)    :: flushing
   838)       integer        , intent(in)    :: ipft
   839)       real           , intent(in)    :: carbon_balance
   840)       real           , intent(in)    :: nplant
   841)       real           , intent(in)    :: tr_bleaf
   842)       real           , intent(in)    :: tr_broot
   843)       real           , intent(in)    :: tr_bstorage
   844)       real           , intent(inout) :: nitrogen_uptake
   845)       real           , intent(inout) :: fsn_in
   846)       !----- Local variables. -------------------------------------------------------------!
   847)       real                           :: n2c_labile
   848)       !------------------------------------------------------------------------------------!
   849) 
   850)       n2c_labile = (f_labile(ipft)/c2n_leaf(ipft) + (1.0-f_labile(ipft))/c2n_stem(ipft))
   851) 
   852)       if (flushing) then
   853)          !---------------------------------------------------------------------------------!
   854)          !     Check whether there we're adding to or taking from storage.                 !
   855)          !---------------------------------------------------------------------------------!
   856)          if (tr_bstorage <= 0.0)  then
   857)             !------------------------------------------------------------------------------!
   858)             ! We are using all of daily C gain and some of bstorage.                       !
   859)             ! Calculate N demand from using daily C gain.                                  !
   860)             !------------------------------------------------------------------------------!
   861)             if (carbon_balance < 0.0) then
   862)                nitrogen_uptake = nitrogen_uptake + carbon_balance / c2n_storage
   863)                nitrogen_uptake = nitrogen_uptake                                           &
   864)                                + (carbon_balance - tr_bstorage)                            &
   865)                                * ( f_labile(ipft) / c2n_leaf(ipft)                         &
   866)                                  + (1.0 - f_labile(ipft)) / c2n_stem(ipft)                 &
   867)                                  -  1.0 / c2n_storage)
   868) 
   869)             else
   870)                nitrogen_uptake = nitrogen_uptake + carbon_balance                          &
   871)                               * ( f_labile(ipft) / c2n_leaf(ipft)                          &
   872)                                 + (1.0 - f_labile(ipft)) / c2n_stem(ipft) )
   873) 
   874)                !---------------------------------------------------------------------------!
   875)                ! Calculate additional N uptake from transfer of C from storage to balive.  !
   876)                !---------------------------------------------------------------------------!
   877)                nitrogen_uptake  = nitrogen_uptake +  ( - 1.0 * tr_bstorage )               &
   878)                                 * ( f_labile(ipft)  / c2n_leaf(ipft)                       &
   879)                                   + (1.0 - f_labile(ipft)) / c2n_stem(ipft)                &
   880)                                   -  1.0 / c2n_storage)
   881)             end if
   882) 
   883)          else
   884)             !------------------------------------------------------------------------------!
   885)             !     N uptake for fraction of daily C gain going to balive.                   !
   886)             !------------------------------------------------------------------------------!
   887)             nitrogen_uptake = nitrogen_uptake + (carbon_balance - tr_bstorage)             &
   888)                             * ( f_labile(ipft) / c2n_leaf(ipft)                            &
   889)                               + (1.0 - f_labile(ipft)) / c2n_stem(ipft))
   890)             !----------- N uptake for fraction of daily C gain going to bstorage. ---------!
   891)             nitrogen_uptake = nitrogen_uptake + tr_bstorage / c2n_storage
   892)          end if
   893)          !---------------------------------------------------------------------------------!
   894)       else
   895)          if (tr_bstorage > 0.0) then
   896)             nitrogen_uptake = nitrogen_uptake + tr_bstorage / c2n_storage
   897)          else
   898)             fsn_in = fsn_in                                                                &
   899)                    - tr_bstorage           / c2n_storage * nplant                          &
   900)                    - (tr_bleaf + tr_broot) * n2c_labile  * nplant
   901)          end if
   902)       end if
   903) 
   904) 
   905)    end subroutine update_nitrogen

   984)       !        ,'        YEAR','       MONTH','         DAY','         PFT','   PHENOLOGY'  &
   985)       !        ,'PHEN_STAT_IN','PHN_STAT_OUT','  FLUSH_TIME',' AVAILABLE_C','      ELONGF'  &
   986)       !        ,'  GREEN_LEAF','    ON_ALLOM',' DELTA_BLEAF',' DELTA_BROOT','   DELTA_BSA'  &
   987)       !        ,'   DELTA_BSB','    TR_BLEAF','    TR_BROOT','      TR_BSA','      TR_BSB'
   988)       !      close (unit=66,status='keep')
   989)       !   end if
   990)       !   first_time = .false.
   991)       !end if
   992)       !------------------------------------------------------------------------------------!
   993) 
   994)       tr_bleaf     = 0.0
   995)       tr_broot     = 0.0
   996)       tr_bsapwooda = 0.0
   997)       tr_bsapwoodb = 0.0
   998)       tr_bstorage  = 0.0
   999) 
  1000)       cpatch => csite%patch(ipa)
  1001) 
  1002)       ipft = cpatch%pft(ico)
  1003)       phen_stat_in = cpatch%phenology_status(ico)
  1004)       !------------------------------------------------------------------------------------!
  1005)       !      When plants transit from dormancy to leaf flushing, it is possible that       !
  1006)       ! carbon_balance is negative, but the sum of carbon_balance and bstorage is          !
  1007)       ! positive. Under this circumstance, we have to allow plants to grow leaves.         !
  1008)       !------------------------------------------------------------------------------------!
  1009)       available_carbon = cpatch%bstorage(ico) + carbon_balance
  1010)       time_to_flush    = carbon_balance > 0.0 .or.                                         &
  1011)                          ( available_carbon > 0.0 .and. cpatch%phenology_status(ico) == 1 )
  1012)       !------------------------------------------------------------------------------------!
  1013) 
  1014)       !------------------------------------------------------------------------------------!
  1015)       !     Maximum bleaf that the allometric relationship would allow.  If the plant is   !
  1016)       ! drought stressed (elongf < 1), we don't allow it to get back to full allometry.    !
  1017)       !------------------------------------------------------------------------------------!
  1018)       bleaf_max      = size2bl(cpatch%dbh(ico),cpatch%hite(ico),ipft)
  1019)       bleaf_aim      = bleaf_max * green_leaf_factor * cpatch%elongf(ico)
  1020)       broot_aim      = bleaf_aim * q(ipft)
  1021)       bsapwooda_aim  = bleaf_aim * qsw(ipft) * cpatch%hite(ico) * agf_bs(ipft)
  1022)       bsapwoodb_aim  = bleaf_aim * qsw(ipft) * cpatch%hite(ico) * (1. - agf_bs(ipft))
  1023)       balive_aim     = bleaf_aim + broot_aim + bsapwooda_aim + bsapwoodb_aim
  1024)       !------------------------------------------------------------------------------------!
  1025) 
  1026) 
  1027)       !------------------------------------------------------------------------------------!
  1028)       !      Check whether to increase living tissue biomass or not.                       !
  1029)       !------------------------------------------------------------------------------------!
  1030)       flushing = .false.
  1031)       if (time_to_flush) then
  1032)          select case (cpatch%phenology_status(ico))
  1033)          case (0,1)
  1034)             flushing = .true.
  1035)             !------------------------------------------------------------------------------!
  1036)             !     There are leaves, we are not actively dropping leaves and we're off      !
  1037)             ! allometry.  Here we will compute the maximum amount that can go to balive    !
  1038)             ! pools, and put any excess in storage.                                        !
  1039)             !------------------------------------------------------------------------------!
  1040) 
  1041)             !---- Amount that bleaf, broot, and bsapwood are off allometry. ---------------!
  1042)             delta_bleaf     = max (0.0, bleaf_aim     - cpatch%bleaf    (ico))
  1043)             delta_broot     = max (0.0, broot_aim     - cpatch%broot    (ico))
  1044)             delta_bsapwooda = max (0.0, bsapwooda_aim - cpatch%bsapwooda(ico))
  1045)             delta_bsapwoodb = max (0.0, bsapwoodb_aim - cpatch%bsapwoodb(ico))
  1046)             !------------------------------------------------------------------------------!
  1047) 
  1048)             if(cpatch%elongf(ico) < tiny_num) then
  1049) 
  1050)                write(*,'(a)')' ============================================'
  1051)                write(*,'(a)')' LINE 990 growth_balive.f90'
  1052)                write(*,'(a)')' subroutine alloc_plant_c_balance'
  1053)                write(*,'(a)')' '
  1054)                write(*,'(a)')' An elongation factor of effectively zero'
  1055)                write(*,'(a)')' has been detected during the transfer'
  1056)                write(*,'(a)')' of storage carbon back to active leaf pool.'
  1057)                write(*,'(a)')' This routine is expecting a non-zero '
  1058)                write(*,'(a)')' elongation as status leaves exist.'
  1059)                write(*,'(a)')' This is a minor bug that appears to trigger'
  1060)                write(*,'(a)')' in rare cases when veg dynamics are off and'
  1061)                write(*,'(a)')' drought stress is high.'
  1062)                write(*,'(a)')' '
  1063)                write(*,'(a)')' Continuing with 0 storage transfer.'
  1064)                write(*,'(a)')' ============================================'
  1065) 
  1066)                f_total=0.0
  1067)             else
  1068) 
  1069)                !------------------------------------------------------------------------------!
  1070)                !     If the available carbon is less than what we need to get back to         !
  1071)                ! allometry.  Grow pools in proportion to demand.  If we have enough carbon,   !
  1072)                ! we'll put the extra into bstorage.                                           !
  1073)                !------------------------------------------------------------------------------!
  1074)                f_bleaf     = delta_bleaf     / bleaf_aim
  1075)                f_broot     = delta_broot     / broot_aim
  1076)                f_bsapwooda = delta_bsapwooda / bsapwooda_aim
  1077)                f_bsapwoodb = delta_bsapwoodb / bsapwoodb_aim
  1078)                f_total     = f_bleaf + f_broot + f_bsapwooda + f_bsapwoodb
  1079)                !------------------------------------------------------------------------------!
  1080)             end if
  1081) 
  1082)             !------------------------------------------------------------------------------!
  1083)             !     We only allow transfer from storage to living tissues if there is need   !
  1084)             ! to transfer.                                                                 !
  1085)             !------------------------------------------------------------------------------!
  1086)             if (f_total > 0.0) then
  1087)                tr_bleaf     = min(delta_bleaf    , f_bleaf     / f_total * available_carbon)
  1088)                tr_broot     = min(delta_broot    , f_broot     / f_total * available_carbon)
  1089)                tr_bsapwooda = min(delta_bsapwooda, f_bsapwooda / f_total * available_carbon)
  1090)                tr_bsapwoodb = min(delta_bsapwoodb, f_bsapwoodb / f_total * available_carbon)
  1091)             end if
  1092)             !------------------------------------------------------------------------------!
  1093) 
  1094)             tr_bstorage = carbon_balance - tr_bleaf - tr_broot - tr_bsapwooda - tr_bsapwoodb
  1095)          case default
  1096)             !------------------------------------------------------------------------------!
  1097)             !     Put carbon gain into storage.  If we're not actively dropping leaves or  !
  1098)             ! off-allometry, this will be used for structural growth at the end of the     !
  1099)             ! month.                                                                       !
  1100)             !------------------------------------------------------------------------------!
  1101)             tr_bstorage  = carbon_balance
  1102)             !------------------------------------------------------------------------------!
  1103)          end select
  1104)          !---------------------------------------------------------------------------------!
  1105) 
  1106)       else
  1107)          !---------------------------------------------------------------------------------!
  1108)          !   Carbon balance is negative, decide the source of carbon based on the          !
  1109)          ! phenology status.  If plants were already dropping leaves, then we don't take   !
  1110)          ! the carbon from storage unless there is no leaf or root biomass left.  If       !
  1111)          ! plants should be growing but they aren't, then we burn the storage first, and   !
  1112)          ! if the situation persists, then plants start destroying their living tissues.   !
  1113)          !---------------------------------------------------------------------------------!
  1114)          carbon_debt = -carbon_balance
  1115)          select case (cpatch%phenology_status(ico))
  1116)          case (0,1)
  1117)             !------------------------------------------------------------------------------!
  1118)             !    Plants should be growing or at their maximum, first we try to take all    !
  1119)             ! the carbon needed from storage.                                              !
  1120)             !------------------------------------------------------------------------------!
  1121)             if (cpatch%bstorage(ico) > carbon_debt) then
  1122)                tr_bstorage = carbon_balance
  1123)             else
  1124)                !---------------------------------------------------------------------------!
  1125)                !     Not enough carbon in storage.  Take everything then start destroying  !
  1126)                ! tissues.                                                                  !
  1127)                !---------------------------------------------------------------------------!
  1128)                carbon_debt = carbon_debt - cpatch%bstorage(ico)
  1129)                tr_bstorage = -1.0*cpatch%bstorage(ico)
  1130) 
  1131)                !---------------------------------------------------------------------------!
  1132)                !     Find total biomass that can be lost.  We take an amount proportional  !
  1133)                ! to the current biomass of each the pools.                                 !
  1134)                !---------------------------------------------------------------------------!
  1135)                bloss_max   = cpatch%bleaf(ico) + cpatch%broot(ico)
  1136)                f_bleaf     = cpatch%bleaf    (ico) / bloss_max
  1137)                f_broot     = cpatch%broot    (ico) / bloss_max
  1138) 
  1139)                if (bloss_max > carbon_debt) then
  1140)                   !----- Remove biomass accordingly. --------------------------------------!
  1141)                   tr_bleaf = f_bleaf * -1.0 * carbon_debt
  1142)                   tr_broot = f_broot * -1.0 * carbon_debt
  1143)                   !------------------------------------------------------------------------!
  1144)                else
  1145)                   !------------------------------------------------------------------------!
  1146)                   !     This cohort didn't know how to save carbon during its life, and    !
  1147)                   ! has spent everything it had and now it is sunk in huge debt that it    !
  1148)                   ! can't afford.  It is with profound sadness that we announce that this  !
  1149)                   ! cohort is going to fertilizer business.                                !
  1150)                   !------------------------------------------------------------------------!
  1151)                   carbon_debt = bloss_max
  1152)                   tr_bleaf = -1.0*cpatch%bleaf(ico);
  1153)                   tr_broot = -1.0*cpatch%broot(ico);
  1154)                   !------------------------------------------------------------------------!
  1155)                end if
  1156)                !---------------------------------------------------------------------------!
  1157)             end if
  1158)             !------------------------------------------------------------------------------!
  1159)          case (-1,-2)
  1160)             !------------------------------------------------------------------------------!
  1161)             !      Plants were already shedding leaves.  We swap the order here and remove !
  1162)             ! living tissues first, and only if there is nothing left that we remove       !
  1163)             ! storage.                                                                     !
  1164)             !------------------------------------------------------------------------------!
  1165)             bloss_max   = cpatch%bleaf(ico) + cpatch%broot(ico)
  1166)             f_bleaf     = cpatch%bleaf    (ico) / bloss_max
  1167)             f_broot     = cpatch%broot    (ico) / bloss_max
  1168) 
  1169)             if (bloss_max > carbon_debt) then
  1170)                !----- Remove biomass accordingly. -----------------------------------------!
  1171)                tr_bleaf = f_bleaf * -1.0 * carbon_debt
  1172)                tr_broot = f_broot * -1.0 * carbon_debt
  1173)                !---------------------------------------------------------------------------!
  1174)             else
  1175)                !---------------------------------------------------------------------------!
  1176)                !     Not enough biomass, remove everything.                                !
  1177)                !---------------------------------------------------------------------------!
  1178)                carbon_debt = carbon_debt - bloss_max
  1179)                tr_bleaf = -1.0*cpatch%bleaf(ico);
  1180)                tr_broot = -1.0*cpatch%broot(ico);
  1181)                !---------------------------------------------------------------------------!
  1182) 
  1183)                !---------------------------------------------------------------------------!
  1184)                !     The living tissues weren't enough to meet the demand, remove what is  !
  1185)                ! still needed from the storage.                                            !
  1186)                !---------------------------------------------------------------------------!
  1187)                if (cpatch%bstorage(ico) > carbon_debt) then
  1188)                   !----- Enough carbon in storage, take all carbon needed from there. -----!
  1189)                   tr_bstorage = -1.0 * carbon_debt
  1190)                   !cpatch%bstorage(ico) = cpatch%bstorage(ico) - carbon_debt
  1191)                   !------------------------------------------------------------------------!
  1192)                else
  1193)                   !------------------------------------------------------------------------!
  1194)                   !     This cohort didn't know how to save carbon during its life, and    !
  1195)                   ! has spent everything it had and now it is sunk in huge debt that it    !
  1196)                   ! can't afford.  It is with profound sadness that we announce that this  !
  1197)                   ! cohort is going to fertilizer business.                                !
  1198)                   !------------------------------------------------------------------------!
  1199)                   tr_bstorage = -1.0*cpatch%bstorage(ico)
  1200)                   !------------------------------------------------------------------------!
  1201)                end if
  1202)             end if
  1203)             !------------------------------------------------------------------------------!
  1204)          end select
  1205)          !---------------------------------------------------------------------------------!
  1206)       end if
  1207) 
  1208)       !------------------------------------------------------------------------------------!
  1209) !      if (printout) then
  1210) !         open (unit=66,file=fracfile,status='old',position='append',action='write')
  1211) !         write(unit=66,fmt='(7(i12,1x),1(11x,l1,1x),3(f12.6,1x),1(11x,l1,1x),8(f12.8,1x))')&
  1212) !               current_time%year,current_time%month,current_time%date,ipft,phenology(ipft) &
  1213) !              ,phen_stat_in,cpatch%phenology_status(ico),time_to_flush,available_carbon    &
  1214) !              ,cpatch%elongf(ico),green_leaf_factor,on_allometry,delta_bleaf,delta_broot   &
  1215) !              ,delta_bsapwooda,delta_bsapwoodb,tr_bleaf,tr_broot,tr_bsapwooda,tr_bsapwoodb
  1216) !         close (unit=66,status='keep')
  1217) !      end if
  1218)       !------------------------------------------------------------------------------------!
  1219) 
  1220)       return
  1221)    end subroutine get_c_xfers

  1230)    subroutine apply_c_xfers(cpatch,ico,carbon_balance,tr_bleaf,tr_broot,tr_bsapwooda       &
  1231)                            ,tr_bsapwoodb,tr_bstorage)
  1232)       use ed_state_vars , only : patchtype  ! ! structure
  1233)       implicit none
  1234)       !----- Arguments. -------------------------------------------------------------------!
  1235)       type(patchtype), target        :: cpatch
  1236)       integer        , intent(in)    :: ico
  1237)       real           , intent(in)    :: carbon_balance
  1238)       real           , intent(in)    :: tr_bleaf
  1239)       real           , intent(in)    :: tr_broot
  1240)       real           , intent(in)    :: tr_bsapwooda
  1241)       real           , intent(in)    :: tr_bsapwoodb
  1242)       real           , intent(in)    :: tr_bstorage
  1243)       !------------------------------------------------------------------------------------!
  1244) 
  1245)       !------------------------------------------------------------------------------!
  1246)       !     Update the carbon pools of the living tissues.                           !
  1247)       !------------------------------------------------------------------------------!
  1248)       cpatch%bleaf    (ico) = cpatch%bleaf    (ico) + tr_bleaf
  1249)       cpatch%broot    (ico) = cpatch%broot    (ico) + tr_broot
  1250)       cpatch%bsapwooda(ico) = cpatch%bsapwooda(ico) + tr_bsapwooda
  1251)       cpatch%bsapwoodb(ico) = cpatch%bsapwoodb(ico) + tr_bsapwoodb
  1252) 
  1253)       cpatch%balive   (ico) = cpatch%bleaf(ico) + cpatch%broot(ico)                  &
  1254)                             + cpatch%bsapwooda(ico) + cpatch%bsapwoodb(ico)
  1255)       !------------------------------------------------------------------------------!
  1256) 
  1257) 
  1258)       !----- NPP allocation in diff pools in KgC/m2/day. ----------------------------!
  1259)       cpatch%today_nppleaf(ico)   = tr_bleaf       * cpatch%nplant(ico)
  1260)       cpatch%today_nppfroot(ico)  = tr_broot       * cpatch%nplant(ico)
  1261)       cpatch%today_nppsapwood(ico)= (tr_bsapwooda + tr_bsapwoodb)* cpatch%nplant(ico)
  1262)       cpatch%today_nppdaily(ico)  = carbon_balance * cpatch%nplant(ico)
  1263)       !------------------------------------------------------------------------------!
  1264) 
  1265) 
  1266)       !------------------------------------------------------------------------------!
  1267)       !    Find the amount of carbon used to recover the tissues that were off-      !
  1268)       ! -allometry, take that from the carbon balance first, then use some of the    !
  1269)       ! storage if needed be.                                                        !
  1270)       !------------------------------------------------------------------------------!
  1271)       cpatch%bstorage(ico) = max(0.0, cpatch%bstorage(ico) + tr_bstorage)
  1272)       !------------------------------------------------------------------------------!
  1273)       return

  1284)    subroutine plant_maintenance(cpatch,ico,br,bl,tfact,daily_C_gain,tempk)
  1285)       use ed_state_vars, only : patchtype          ! ! structure
  1286)       use pft_coms     , only : phenology          & ! intent(in)
  1287)                               , root_turnover_rate & ! intent(in)
  1288)                               , leaf_turnover_rate ! ! intent(in)
  1289)       use consts_coms  , only : umol_2_kgC         & ! intent(in)
  1290)                               , day_sec            & ! intent(in)
  1291)                               , tiny_num           ! ! intent(in)
  1292)       implicit none
  1293)       !----- Arguments. -------------------------------------------------------------------!
  1294)       type(patchtype), target        :: cpatch
  1295)       integer        , intent(in)    :: ico
  1296)       real           , intent(in)    :: br
  1297)       real           , intent(in)    :: bl
  1298)       real           , intent(in)    :: tfact
  1299)       real           , intent(in)    :: tempk
  1300)       real           , intent(out)   :: daily_C_gain
  1301)       !----- Local variables. -------------------------------------------------------------!
  1302)       integer                        :: ipft
  1303)       real                           :: maintenance_temp_dep
  1304)       !------------------------------------------------------------------------------------!
  1305) 
  1306)       !------ Alias for plant functional type. --------------------------------------------!
  1307)       ipft = cpatch%pft(ico)
  1308) 
  1309)       !------------------------------------------------------------------------------------!
  1310)       !      Find the maintenance costs.  This will depend on the type of phenology that   !
  1311)       ! the PFT has.   The tfact term applied converts the maintenance rates to            !
  1312)       ! [kgC/plant/day].                                                                   !
  1313)       !------------------------------------------------------------------------------------!
  1314)       select case (phenology(ipft))
  1315)       case (0)
  1316)          !---------------------------------------------------------------------------------!
  1317)          !     Evergreens, like pines.  The turnover rates will be adjusted by a function  !
  1318)          ! of temperature, which approaches 0 as the temperature goes down.                !
  1319)          !---------------------------------------------------------------------------------!
  1320)          !------ Find a temperature dependence adjustment. --------------------------------!
  1321)          maintenance_temp_dep = 1.0 / (1.0 + exp(0.4 * (278.15 - tempk)))
  1322)          !----- Scale maintenance by biomass and apply the temperature correction. --------!
  1323)          cpatch%leaf_maintenance(ico) = leaf_turnover_rate(ipft) * bl                      &
  1324)                                       * maintenance_temp_dep * tfact
  1325)          cpatch%root_maintenance(ico) = root_turnover_rate(ipft) * br                      &
  1326)                                       * maintenance_temp_dep * tfact
  1327)          !---------------------------------------------------------------------------------!
  1328) 
  1329)       case (3)
  1330)          !---------------------------------------------------------------------------------!
  1331)          !      Light phenology.  Leaf turnover rate will be adjusted according to the     !
  1332)          ! amplitude that comes from the dependence on the radiation (running average).    !
  1333)          ! Roots are the same as the other plants that don't depend on temperature.        !
  1334)          !---------------------------------------------------------------------------------!
  1335)          cpatch%root_maintenance(ico) = root_turnover_rate(ipft) * br * tfact
  1336)          cpatch%leaf_maintenance(ico) = leaf_turnover_rate(ipft) * bl                      &
  1337)                                       * cpatch%turnover_amp(ico) * tfact
  1338)          !---------------------------------------------------------------------------------!
  1339) 
  1340) 
  1341)       case default
  1342)          !---------------------------------------------------------------------------------!
  1343)          !      Ohter phenologies, use the standard turnover rates, scaled by biomass      !
  1344)          ! only.                                                                           !
  1345)          !---------------------------------------------------------------------------------!
  1346)          cpatch%root_maintenance(ico) = root_turnover_rate(ipft) * br * tfact
  1347)          cpatch%leaf_maintenance(ico) = leaf_turnover_rate(ipft) * bl * tfact
  1348)          !---------------------------------------------------------------------------------!
  1349) 
  1350)       end select
  1351)       !------------------------------------------------------------------------------------!
  1352) 
  1353) 
  1354) 
  1355)       !------------------------------------------------------------------------------------!
  1356)       !      In case maintenance is too small, flush values to zero.                       !
  1357)       !------------------------------------------------------------------------------------!
  1358)       if (cpatch%leaf_maintenance(ico) < tiny_num) cpatch%leaf_maintenance(ico) = 0.0
  1359)       if (cpatch%root_maintenance(ico) < tiny_num) cpatch%root_maintenance(ico) = 0.0
  1360)       !------------------------------------------------------------------------------------!
  1361) 
  1362) 
  1363) 
  1364)       !----- Compute daily C uptake [kgC/plant/day]. --------------------------------------!
  1365)       if(cpatch%nplant(ico) > tiny(1.0)) then
  1366)          daily_C_gain = umol_2_kgC * day_sec * ( cpatch%today_gpp(ico)                     &
  1367)                                                - cpatch%today_leaf_resp(ico)               &
  1368)                                                - cpatch%today_root_resp(ico))              &
  1369)                                              / cpatch%nplant(ico)
  1370)       else
  1371)          daily_C_gain = 0.0
  1372)       end if
  1373) 
  1374)       return
  1375)    end subroutine plant_maintenance

  1386)    subroutine plant_carbon_balances(cpatch,ipa,ico,daily_C_gain,carbon_balance             &
  1387)                                    ,carbon_balance_pot,carbon_balance_lightmax             &
  1388)                                    ,carbon_balance_moistmax,carbon_balance_mlmax)
  1389)       use ed_state_vars  , only : patchtype          ! ! structure
  1390)       use pft_coms       , only : growth_resp_factor ! ! intent(in)
  1391)       use consts_coms    , only : umol_2_kgC         & ! intent(in)
  1392)                                 , day_sec            ! ! intent(in)
  1393)       use ed_misc_coms   , only : current_time       ! ! intent(in)
  1394)       use ed_max_dims    , only : n_pft              ! ! intent(in)
  1395)       implicit none
  1396)       !----- Arguments. -------------------------------------------------------------------!
  1397)       type(patchtype)          , target      :: cpatch
  1398)       integer                  , intent(in)  :: ipa
  1399)       integer                  , intent(in)  :: ico
  1400)       real                     , intent(in)  :: daily_C_gain
  1401)       real                     , intent(out) :: carbon_balance
  1402)       real                     , intent(out) :: carbon_balance_pot
  1403)       real                     , intent(out) :: carbon_balance_lightmax
  1404)       real                     , intent(out) :: carbon_balance_moistmax
  1405)       real                     , intent(out) :: carbon_balance_mlmax
  1406)       !----- Local variables. -------------------------------------------------------------!
  1407)       real                                   :: daily_C_gain_pot
  1408)       real                                   :: daily_C_gain_lightmax
  1409)       real                                   :: daily_C_gain_moistmax
  1410)       real                                   :: daily_C_gain_mlmax
  1411)       real                                   :: growth_respiration_pot
  1412)       real                                   :: growth_respiration_lightmax
  1413)       real                                   :: growth_respiration_moistmax
  1414)       real                                   :: growth_respiration_mlmax

  1417)       logical                  , parameter   :: print_debug = .false.
  1418)       !----- Locally saved variables. -----------------------------------------------------!
  1419)       logical, dimension(n_pft), save        :: first_time  = .true.
  1420)       !------------------------------------------------------------------------------------!
  1421) 
  1422)       !----- Alias for PFT type. ----------------------------------------------------------!
  1423)       ipft = cpatch%pft(ico)
  1424)       !------------------------------------------------------------------------------------!
  1425) 
  1426) 
  1427) 
  1428)       !------------------------------------------------------------------------------------!
  1429)       !       Calculate actual daily carbon balance: kgC/plant/day.                        !
  1430)       !------------------------------------------------------------------------------------!
  1431)       carbon_balance = daily_C_gain - cpatch%leaf_growth_resp (ico)                        &
  1432)                                     - cpatch%root_growth_resp (ico)                        &
  1433)                                     - cpatch%sapa_growth_resp (ico)                        &
  1434)                                     - cpatch%sapb_growth_resp (ico)
