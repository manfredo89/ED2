     1) !==========================================================================================!
     2) !==========================================================================================!
     3) !     This sub-routine converts a conventional SAS initialisation into a big leaf.         !
     4) !------------------------------------------------------------------------------------------!
     5) subroutine ed_bigleaf_init(cgrid)
     6)    use ed_node_coms   , only : mynum                ! ! intent(in)
     7)    use ed_state_vars  , only : edtype               & ! structure
     8)                              , polygontype          & ! structure
     9)                              , sitetype             & ! structure
    10)                              , patchtype            & ! structure
    11)                              , allocate_sitetype    & ! subroutine
    12)                              , deallocate_sitetype  & ! subroutine
    13)                              , allocate_patchtype   & ! subroutine
    14)                              , deallocate_patchtype ! ! subroutine
    15)    use ed_max_dims    , only : n_pft                & ! intent(in)
    16)                              , n_dist_types         ! ! intent(in)
    17)    use allometry      , only : size2bl              & ! function
    18)                              , dbh2bd               & ! function
    19)                              , area_indices         & ! function
    20)                              , ed_biomass           ! ! function
    21)    use pft_coms       , only : hgt_max              & ! intent(in)
    22)                              , dbh_bigleaf          & ! intent(in)
    23)                              , sla                  & ! intent(in)
    24)                              , q                    & ! intent(in)
    25)                              , qsw                  & ! intent(in)
    26)                              , agf_bs               ! ! intent(in)
    27)    use fuse_fiss_utils, only : sort_cohorts         & ! subroutine
    28)                              , sort_patches         ! ! subroutine
    29)    use consts_coms    , only : pio4                 ! ! intent(in)
    30)    implicit none
    31) 
    32)    !----- Arguments. ----------------------------------------------------------------------!
    33)    type(edtype)     , target                        :: cgrid
    34)    !----- Local variables. ----------------------------------------------------------------!
    35)    type(polygontype), pointer                       :: cpoly
    36)    type(sitetype)   , pointer                       :: csite
    37)    type(patchtype)  , pointer                       :: cpatch
    38)    integer          , dimension(n_dist_types)       :: lu_npatch
    39)    integer                                          :: site_npatch
    40)    integer                                          :: ipy
    41)    integer                                          :: isi
    42)    integer                                          :: ipa
    43)    integer                                          :: ico
    44)    integer                                          :: ipft
    45)    integer                                          :: ilu
    46)    integer                                          :: ncohorts
    47)    integer                                          :: npatchco
    48)    integer                                          :: nsitepat
    49)    integer                                          :: k
    50)    logical          , dimension(n_dist_types)       :: lu_desert
    51)    real             , dimension(n_pft)              :: pft_area
    52)    real             , dimension(n_pft,n_dist_types) :: lai
    53)    real             , dimension(n_dist_types)       :: area
    54)    real             , dimension(n_dist_types)       :: fsc
    55)    real             , dimension(n_dist_types)       :: ssc
    56)    real             , dimension(n_dist_types)       :: stsc
    57)    real             , dimension(n_dist_types)       :: stsl
    58)    real             , dimension(n_dist_types)       :: msn
    59)    real             , dimension(n_dist_types)       :: fsn
    60)    real             , dimension(n_dist_types)       :: sum_dgd
    61)    real             , dimension(n_dist_types)       :: sum_chd
    62)    real                                             :: area_sum
    63)    real                                             :: area_tot
    64)    real                                             :: patch_lai
    65)    real                                             :: patch_wai
    66)    real                                             :: patch_agb
    67)    real                                             :: patch_bsa
    68)    real                                             :: patch_pop
    69)    real                                             :: site_lai
    70)    real                                             :: site_agb
    71)    real                                             :: site_bsa
    72)    real                                             :: site_pop
    73)    real                                             :: site_fsc
    74)    real                                             :: site_ssc
    75)    real                                             :: site_stsc
    76)    real                                             :: poly_lai
    77)    real                                             :: poly_agb
    78)    real                                             :: poly_bsa
    79)    real                                             :: poly_pop
    80)    real                                             :: poly_fsc
    81)    real                                             :: poly_ssc
    82)    real                                             :: poly_stsc
    83)    !---------------------------------------------------------------------------------------!
    84) 
    85) 
    86) 
    87)    !---------------------------------------------------------------------------------------!
    88)    !       Loop over all sites.                                                            !
    89)    !---------------------------------------------------------------------------------------!
    90)    write(unit=*,fmt='(141a)'    ) ('-',k=1,141)
    91)    write(unit=*,fmt='(a,1x,i5)' ) ' Mynum = ',mynum
    92)    write(unit=*,fmt='(141a)'    ) ('-',k=1,141)
    93)    write(unit=*,fmt='(11(a,1x))') '         IPY','      NSITES','    NPATCHES'             &
    94)                                  ,'    NCOHORTS','      NPLANT','         LAI'             &
    95)                                  ,'         AGB','  BASAL_AREA','     FAST_SC'             &
    96)                                  ,'     SLOW_SC','   STRUCT_SC'
    97)    write(unit=*,fmt='(141a)'   ) ('-',k=1,141)
    98)    polyloop: do ipy=1,cgrid%npolygons
    99) 
   100)       cpoly => cgrid%polygon(ipy)
   101)       firstsiteloop: do isi=1,cpoly%nsites
   102) 
   103)          !----- Reset all structures before loop over paches. -----------------------------!
   104)          lai     (:,:) = 0.
   105)          area      (:) = 0.
   106)          fsc       (:) = 0.
   107)          ssc       (:) = 0.
   108)          stsc      (:) = 0.
   109)          stsl      (:) = 0.
   110)          msn       (:) = 0.
   111)          fsn       (:) = 0.
   112)          sum_dgd   (:) = 0.
   113)          sum_chd   (:) = 0.
   114) 
   115)          !---------------------------------------------------------------------------------!
   116)          !     Determine the site-level properties.                                        !
   117)          !---------------------------------------------------------------------------------!
   118)          csite => cpoly%site(isi)
   119)          inpatchloop: do ipa = 1,csite%npatches
   120) 
   121)             !------------------------------------------------------------------------------!
   122)             !    Find the land use type for this patch.  We will aggregate all patches     !
   123)             ! with this disturbance type.                                                  !
   124)             !------------------------------------------------------------------------------!
   125)             ilu = csite%dist_type(ipa)
   126)             area   (ilu) = area   (ilu) + csite%area              (ipa)
   127)             fsc    (ilu) = fsc    (ilu) + csite%fast_soil_C       (ipa) * csite%area(ipa)
   128)             ssc    (ilu) = ssc    (ilu) + csite%slow_soil_C       (ipa) * csite%area(ipa)
   129)             stsc   (ilu) = stsc   (ilu) + csite%structural_soil_C (ipa) * csite%area(ipa)
   130)             stsl   (ilu) = stsl   (ilu) + csite%structural_soil_L (ipa) * csite%area(ipa)
   131)             msn    (ilu) = msn    (ilu) + csite%mineralized_soil_N(ipa) * csite%area(ipa)
   132)             fsn    (ilu) = fsn    (ilu) + csite%fast_soil_N       (ipa) * csite%area(ipa)
   133)             sum_dgd(ilu) = sum_dgd(ilu) + csite%sum_dgd           (ipa) * csite%area(ipa)
   134)             sum_chd(ilu) = sum_chd(ilu) + csite%sum_chd           (ipa) * csite%area(ipa)
   135)             !------------------------------------------------------------------------------!
   136) 
   137)             !------------------------------------------------------------------------------!
   138)             !    Integrate the cohort-level properties.                                    !
   139)             !------------------------------------------------------------------------------!
   140)             cpatch => csite%patch(ipa)
   141)             incohortloop: do ico=1,cpatch%ncohorts
   142)                !---------------------------------------------------------------------------!
   143)                !     Find the PFT for this cohort.  We aggregate all cohorts of this PFT.  !
   144)                ! The only variable we need is the leaf area index, which is preserved at   !
   145)                ! the site level.  Because this subroutine is called immediately after the  !
   146)                ! initialisation and before the first phenology check, LAI is always        !
   147)                ! positive as long as the cohort exists.                                    !
   148)                !---------------------------------------------------------------------------!
   149)                ipft = cpatch%pft(ico)
   150)                lai(ipft,ilu) = lai(ipft,ilu) + cpatch%lai(ico) * csite%area(ipa)
   151)                !---------------------------------------------------------------------------!
   152)             end do incohortloop
   153)             !------------------------------------------------------------------------------!
   154) 
   155) 
   156)             !----- Remove all cohorts. ----------------------------------------------------!
   157)             call deallocate_patchtype(cpatch)
   158)             cpatch%ncohorts = 0
   159)             !------------------------------------------------------------------------------!
   160)          end do inpatchloop
   161)          !---------------------------------------------------------------------------------!
   162) 
   163) 
   164)          !----- Remove all patches. -------------------------------------------------------!
   165)          call deallocate_sitetype(csite)
   166)          !---------------------------------------------------------------------------------!
   167) 
   168) 
   169) 
   170)          !---------------------------------------------------------------------------------!
   171)          !     The new number of patches is going to be the total number of PFTs that      !
   172)          ! exist amongst all land use types.                                               !
   173)          ! 1.  We add bdead and bleaf because the new grasses don't have structural        !
   174)          !     biomass.                                                                    !
   175)          ! 2.  In case the biomass is zero but the area of the type is not, we make one    !
   176)          !     empty patch.  This can happen when the previous simulation evolved to a     !
   177)          !     desert, for example.                                                        !
   178)          !---------------------------------------------------------------------------------!
   179)          do ilu=1,n_dist_types
   180)             lu_npatch(ilu) = count(lai(:,ilu) > 0.)
   181)             lu_desert(ilu) = lu_npatch(ilu) == 0 .and. area(ilu) > 0.
   182)             if (lu_desert(ilu)) lu_npatch(ilu) = 1
   183)          end do
   184)          site_npatch = sum(lu_npatch)
   185)          !---------------------------------------------------------------------------------!
   186) 
   187) 
   188) 
   189)          !---------------------------------------------------------------------------------!
   190)          !      Allocate the right number of patches.                                      !
   191)          !---------------------------------------------------------------------------------!
   192)          call allocate_sitetype(csite,site_npatch)
   193)          !---------------------------------------------------------------------------------!
   194) 
   195) 
   196)          !---------------------------------------------------------------------------------!
   197)          !      Populate th patches with either 0 or 1 cohort each.                        !
   198)          !---------------------------------------------------------------------------------!
   199)          ipa = 0
   200)          addluloop: do ilu=1,n_dist_types
   201)             if (lu_desert(ilu)) then
   202)                !---------------------------------------------------------------------------!
   203)                !    This land use type existed but became a desert.  Create an empty       !
   204)                ! patch.                                                                    !
   205)                !---------------------------------------------------------------------------!
   206)                ipa = ipa + 1
   207)                csite%dist_type         (ipa) = ilu
   208)                csite%area              (ipa) = area(ilu)
   209)                csite%age               (ipa) = 0.0
   210)                csite%fast_soil_C       (ipa) = fsc (ilu) / area(ilu)
   211)                csite%slow_soil_C       (ipa) = ssc (ilu) / area(ilu)
   212)                csite%structural_soil_C (ipa) = stsc(ilu) / area(ilu)
   213)                csite%structural_soil_L (ipa) = stsl(ilu) / area(ilu)
   214)                csite%mineralized_soil_N(ipa) = msn (ilu) / area(ilu)
   215)                csite%fast_soil_N       (ipa) = fsn (ilu) / area(ilu)
   216)                csite%sum_dgd           (ipa) = 0.0
   217)                csite%sum_chd           (ipa) = 0.0
   218)                csite%cohort_count      (ipa) = 0
   219)                !---------------------------------------------------------------------------!
   220) 
   221)             elseif (lu_npatch(ilu) /= 0) then
   222)                !---------------------------------------------------------------------------!
   223)                !    This land use type has cohorts.   The first step is to find the total  !
   224)                ! area that each patch/cohort/PFT is entitled.                              !
   225)                !---------------------------------------------------------------------------!
   226)                pft_area(:) = lai(:,ilu) / sum(lai(:,ilu))
   227) 
   228)                !----- Loop over all PFTs, and add those that have something. --------------!
   229)                addpftloop: do ipft = 1,n_pft
   230)                   if (lai(ipft,ilu) > 0.0) then
   231)                      ipa = ipa + 1
   232)                      !----- Initialise the patch-level properties. ------------------------!
   233)                      csite%dist_type         (ipa) = ilu
   234)                      csite%area              (ipa) = pft_area(ipft) * area(ilu)
   235)                      csite%age               (ipa) = 0.0
   236)                      csite%fast_soil_C       (ipa) = fsc (ilu) / area(ilu)
   237)                      csite%slow_soil_C       (ipa) = ssc (ilu) / area(ilu)
   238)                      csite%structural_soil_C (ipa) = stsc(ilu) / area(ilu)
   239)                      csite%structural_soil_L (ipa) = stsl(ilu) / area(ilu)
   240)                      csite%mineralized_soil_N(ipa) = msn (ilu) / area(ilu)
   241)                      csite%fast_soil_N       (ipa) = fsn (ilu) / area(ilu)
   242)                      csite%sum_dgd           (ipa) = sum_dgd(ilu)
   243)                      csite%sum_chd           (ipa) = sum_chd(ilu)
   244)                      csite%cohort_count      (ipa) = 1
   245) 
   246)                      !----- Big-leaf model, therefore only one cohort per patch. ----------!
   247)                      cpatch => csite%patch(ipa)
   248)                      call allocate_patchtype(cpatch,1)
   249)                      cpatch%pft          (1) = ipft
   250)                      cpatch%hite         (1) = hgt_max(ipft)
   251)                      cpatch%dbh          (1) = dbh_bigleaf(ipft)
   252)                      cpatch%bleaf        (1) = size2bl(cpatch%dbh(1),cpatch%hite(1),ipft)
   253)                      cpatch%bdead        (1) = dbh2bd (cpatch%dbh(1),ipft)
   254)                      cpatch%broot        (1) = cpatch%bleaf(1) * q(ipft)
   255)                      cpatch%bsapwooda    (1) = agf_bs(ipft) * cpatch%bleaf(1)              &
   256)                                              * qsw(ipft) * cpatch%hite(1)
   257)                      cpatch%bsapwoodb    (1) = (1.0 - agf_bs(ipft)) * cpatch%bleaf(1)      &
   258)                                              * qsw(ipft) * cpatch%hite(1)
   259)                      cpatch%balive       (1) = cpatch%bleaf    (1) + cpatch%broot    (1)   &
   260)                                              + cpatch%bsapwooda(1) + cpatch%bsapwoodb(1)
   261)                      cpatch%sla          (1) = sla     (ipft)
   262)                      cpatch%nplant       (1) = lai (ipft,ilu)                              &
   263)                                              / ( cpatch%sla(1) * cpatch%bleaf(1)           &
   264)                                                * csite%area(ipa) )
   265)                      !---------------------------------------------------------------------!
   266) 
   267) 
   268)                      !---------------------------------------------------------------------!
   269)                      !     Start plants with full phenology and in great carbon balance,   !
   270)                      ! we will take care of phenology after this sub-routine.              !
   271)                      !---------------------------------------------------------------------!
   272)                      cpatch%cb               (1:12,1) = 1.0
   273)                      cpatch%cb_lightmax      (1:12,1) = 1.0
   274)                      cpatch%cb_moistmax      (1:12,1) = 1.0
   275)                      cpatch%cb_mlmax         (1:12,1) = 1.0
   276)                      cpatch%cb                 (13,1) = 0.0
   277)                      cpatch%cb_lightmax        (13,1) = 0.0
   278)                      cpatch%cb_moistmax        (13,1) = 0.0
   279)                      cpatch%cb_mlmax           (13,1) = 0.0
   280)                      cpatch%phenology_status      (1) = 0
   281)                      cpatch%bstorage              (1) = 0.0
   282)                      !---------------------------------------------------------------------!
   283) 
   284) 
   285) 
   286)                      !----- Assign LAI, WAI, and CAI --------------------------------------!
   287)                      call area_indices(cpatch, 1)
   288) 
   289)                      !----- Above ground biomass, use the allometry. ----------------------!
   290)                      cpatch%agb(1)     = ed_biomass(cpatch, 1)
   291)                      cpatch%basarea(1) = pio4 * cpatch%dbh(1) * cpatch%dbh(1)
   292) 
   293)                      !----- Growth rates, start with zero. --------------------------------!
   294)                      cpatch%dagb_dt  (1)  = 0.
   295)                      cpatch%dlnagb_dt(1)  = 0.
   296)                      cpatch%dba_dt   (1)  = 0.
   297)                      cpatch%dlnba_dt (1)  = 0.
   298)                      cpatch%ddbh_dt  (1)  = 0.
   299)                      cpatch%dlndbh_dt(1)  = 0.
   300) 
   301)                      !---------------------------------------------------------------------!
   302)                      !      Initialise other cohort variables.  Some of them won't be      !
   303)                      ! updated unless the lai exceeds lai_min.                             !
   304)                      !---------------------------------------------------------------------!
   305)                      cpatch%fsw  (1) = 1.0
   306)                      cpatch%gpp  (1) = 0.0
   307)                      cpatch%par_l(1) = 0.0
   308)                      !---------------------------------------------------------------------!
   309) 
   310)                      !----- Update the patch level above-ground biomass. ------------------!
   311)                      csite%plant_ag_biomass(ipa) = csite%plant_ag_biomass(ipa)             &
   312)                                                  + cpatch%agb(1) * cpatch%nplant(1)
   313)                      !---------------------------------------------------------------------!
   314)                   end if
   315)                end do addpftloop
   316)                !---------------------------------------------------------------------------!
   317)             end if
   318)             !------------------------------------------------------------------------------!
   319)          end do addluloop
   320)          !---------------------------------------------------------------------------------!
   321)       end do firstsiteloop
   322)       !------------------------------------------------------------------------------------!
   323) 
   324) 
   325) 
   326)       !------------------------------------------------------------------------------------!
   327)       !      Initialise all the other site , patch , and cohort level variables.           !
   328)       !------------------------------------------------------------------------------------!
   329)       initsiteloop: do isi = 1,cpoly%nsites
   330)          area_sum = 0.0
   331)          ncohorts = 0
   332) 
   333) 
   334)          !----- Make sure that the total patch area is 1. ---------------------------------!
   335)          csite => cpoly%site(isi)
   336)          area_tot      = sum(csite%area(1:csite%npatches))
   337)          csite%area(:) = csite%area(:)/area_tot
   338) 
   339)          !----- Find the patch-level LAI, WAI, and CAI. -----------------------------------!
   340)          do ipa=1,csite%npatches
   341)             area_sum        = area_sum + csite%area(ipa)
   342)             patch_lai       = 0.0
   343)             patch_wai       = 0.0
   344)             cpatch => csite%patch(ipa)
   345)             do ico = 1,cpatch%ncohorts
   346)                patch_lai = patch_lai + cpatch%lai(ico)
   347)                patch_wai = patch_wai + cpatch%wai(ico)
   348)                ncohorts  = ncohorts + 1
   349)             end do
   350)          end do
   351) 
   352)          !----- Initialise the cohort variables, then sort them by size. ------------------!
   353)          do ipa = 1,csite%npatches
   354)             cpatch => csite%patch(ipa)
   355)             do ico = 1,cpatch%ncohorts
   356)                call init_ed_cohort_vars(cpatch,ico,cpoly%lsl(isi))
   357)             end do
   358) 
   359)             !----- Make sure that cohorts are organised from tallest to shortest. ---------!
   360)             call sort_cohorts(cpatch)
   361)          end do
   362) 
   363)          !----- Initialise the patch-level variables. -------------------------------------!
   364)          call init_ed_patch_vars(csite,1,csite%npatches,cpoly%lsl(isi))
   365) 
   366)          !----- Make sure that patches are organised from oldest to youngest. -------------!
   367)          call sort_patches(csite)
   368)       end do initsiteloop
   369)       !------------------------------------------------------------------------------------!
   370) 
   371) 
   372) 
   373)       !----- Initialise site-level variables. ---------------------------------------------!
   374)       call init_ed_site_vars(cpoly)
   375) 
   376) 
   377)       !----- Get a diagnostic on the polygon's vegetation. --------------------------------!
   378)       poly_pop  = 0.0
   379)       poly_lai  = 0.0
   380)       poly_agb  = 0.0
   381)       poly_bsa  = 0.0
   382)       poly_fsc  = 0.0
   383)       poly_ssc  = 0.0
   384)       poly_stsc = 0.0
   385)       ncohorts  = 0
   386) 
   387)       do isi = 1,cpoly%nsites
   388)          nsitepat = 0
   389)          csite => cpoly%site(isi)
   390) 
   391)          site_pop  = 0.0
   392)          site_lai  = 0.0
   393)          site_agb  = 0.0
   394)          site_bsa  = 0.0
   395)          site_fsc  = 0.0
   396)          site_ssc  = 0.0
   397)          site_stsc = 0.0
   398)          do ipa = 1,csite%npatches
   399)             patch_lai  = 0.0
   400)             patch_wai  = 0.0
   401)             npatchco        = 0
   402) 
   403)             cpatch => csite%patch(ipa)
   404)             patch_pop       = 0.0
   405)             patch_agb       = 0.0
   406)             patch_bsa       = 0.0
   407)             do ico = 1,cpatch%ncohorts
   408)                ncohorts   = ncohorts+1
   409)                npatchco   = npatchco+1
   410)                patch_lai  = patch_lai  + cpatch%lai(ico)
   411)                patch_wai  = patch_wai  + cpatch%wai(ico)
   412)                patch_pop  = patch_pop  + cpatch%nplant(ico)
   413)                patch_agb  = patch_agb  + cpatch%nplant(ico) * cpatch%agb    (ico)
   414)                patch_bsa  = patch_bsa  + cpatch%nplant(ico) * cpatch%basarea(ico)
   415)             end do
   416)             site_pop  = site_pop  + patch_pop                    * csite%area(ipa)
   417)             site_lai  = site_lai  + patch_lai                    * csite%area(ipa)
   418)             site_agb  = site_agb  + patch_agb                    * csite%area(ipa)
   419)             site_bsa  = site_bsa  + patch_bsa                    * csite%area(ipa)
   420)             site_fsc  = site_fsc  + csite%fast_soil_C      (ipa) * csite%area(ipa)
   421)             site_ssc  = site_ssc  + csite%slow_soil_C      (ipa) * csite%area(ipa)
   422)             site_stsc = site_stsc + csite%structural_soil_C(ipa) * csite%area(ipa)
   423) 
   424)             csite%cohort_count(ipa) = npatchco
   425)             nsitepat                = nsitepat + 1
   426)          end do
   427) 
   428)          poly_pop  = poly_pop  + site_pop  * cpoly%area(isi)
   429)          poly_lai  = poly_lai  + site_lai  * cpoly%area(isi)
   430)          poly_agb  = poly_agb  + site_agb  * cpoly%area(isi)
   431)          poly_bsa  = poly_bsa  + site_bsa  * cpoly%area(isi)
   432)          poly_fsc  = poly_fsc  + site_fsc  * cpoly%area(isi)
   433)          poly_ssc  = poly_ssc  + site_ssc  * cpoly%area(isi)
   434)          poly_stsc = poly_stsc + site_stsc * cpoly%area(isi)
   435) 
   436)          cpoly%patch_count(isi) = nsitepat
   437) 
   438) 
   439)          write (unit=*,fmt='(4(i12,1x),7(f12.3,1x))') ipy,cpoly%nsites,nsitepat,ncohorts   &
   440)                                                      ,poly_pop,poly_lai,poly_agb,poly_bsa  &
   441)                                                      ,poly_fsc,poly_ssc,poly_stsc
   442)       end do
   443) 
   444)       !----- Initialise the polygon-level variables. --------------------------------------!
   445)       call init_ed_poly_vars(cgrid)
   446)    end do polyloop
   447)    write(unit=*,fmt='(141a)') ('-',k=1,141)
   448)    write(unit=*,fmt='(a)') ' '
   449)    write(unit=*,fmt='(a)') ' '
   450)    !---------------------------------------------------------------------------------------!
   451)    return
   452) end subroutine ed_bigleaf_init
