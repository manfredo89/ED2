    10)                             , polygontype & ! structure
    11)                             , sitetype    ! ! structure
    12)    implicit none
    13)    !----- Arguments -----------------------------------------------------------------------!

    15)    !----- Local variables -----------------------------------------------------------------!
    16)    type(polygontype) , pointer :: cpoly
    17)    type(sitetype)    , pointer :: csite
    18)    integer                     :: ipy
    19)    integer                     :: isi
    20)    integer                     :: ipa
    21)    !---------------------------------------------------------------------------------------!
    22)    
    23)    do ipy = 1,cgrid%npolygons

    51) !      This subroutine will take care of derived patch-level structural quantities.  These !
    52) ! depend on the results from reproduction, which in turn depends on structural growth      !
    53) ! results from all patches.                                                                !
    54) !------------------------------------------------------------------------------------------!
    55) subroutine update_patch_derived_props(csite,ipa)
    56)   
    57)    use ed_state_vars       , only : sitetype                   & ! structure
    58)                                   , patchtype                  ! ! structure
    59)    use allometry           , only : ed_biomass                 ! ! function
    60)    use patch_pft_size_profile_mod     ! ! subroutine
    61)    use canopy_air_coms     , only : veg_height_min             & ! intent(in)
    62)                                   , minimum_canopy_depth       & ! intent(in)
    63)                                   , vh2vr                      & ! intent(in)
    64)                                   , vh2dh                      ! ! intent(in)
    65)    use soil_coms           , only : soil_rough                 & ! intent(in)
    66)                                   , ny07_eq04_a                & ! intent(in)
    67)                                   , ny07_eq04_m                & ! intent(in)
    68)                                   , tiny_sfcwater_mass         ! ! intent(in)
    69)    use consts_coms         , only : wdns                       & ! intent(in)
    70)                                   , fsdns                      & ! intent(in)
    71)                                   , fsdnsi                     ! ! intent(in)
    72)    
    73)    implicit none
    74) 
    75)    !----- Arguments -----------------------------------------------------------------------!
    76)    type(sitetype)  , target     :: csite
    77)    integer         , intent(in) :: ipa
    78)    !----- Local variables -----------------------------------------------------------------!
    79)    type(patchtype) , pointer    :: cpatch
    80)    real                         :: weight
    81)    real                         :: weight_sum
    82)    real                         :: total_sfcw_mass
    83)    real                         :: bulk_sfcw_dens
    84)    integer                      :: ico
    85)    integer                      :: k
    86)    integer                      :: ksn
    87)    integer                      :: ipft
    88)    !---------------------------------------------------------------------------------------!
    89) 
    90) 
    91)    !----- Find the total snow depth. ------------------------------------------------------!
    92)    ksn = csite%nlev_sfcwater(ipa)
    93)    csite%total_sfcw_depth(ipa) = 0.
    94)    total_sfcw_mass             = 0.
    95)    do k=1,ksn
    96)       csite%total_sfcw_depth(ipa) = csite%total_sfcw_depth(ipa)                            &
    97)                                   + csite%sfcwater_depth(k,ipa)
    98)       total_sfcw_mass             = total_sfcw_mass                                        &
    99)                                   + csite%sfcwater_mass(k,ipa)
   100)    end do
   101)    !---------------------------------------------------------------------------------------!
   102) 
   103)    !----- Reset properties. ---------------------------------------------------------------!
   104)    csite%veg_height      (ipa) = 0.0
   105)    weight_sum                  = 0.0
   106)    csite%opencan_frac    (ipa) = 1.0
   107)    !---------------------------------------------------------------------------------------!
   108) 
   109) 
   110)    cpatch => csite%patch(ipa)
   111) 
   112)    !----- Loop over cohorts and integrate the patch-level properties. ---------------------!
   113)    do ico = 1,cpatch%ncohorts
   114) 
   115)       ipft = cpatch%pft(ico)
   116) 
   117) 
   118)       !----- Compute the patch-level above-ground biomass
   119)       csite%plant_ag_biomass(ipa) = csite%plant_ag_biomass(ipa)                            &
   120)                                   + ed_biomass(cpatch, ico) * cpatch%nplant(ico)
   121)       !------------------------------------------------------------------------------------!
   122) 
   123) 
   124) 
   125)       !------------------------------------------------------------------------------------!
   126)       !     Compute average vegetation height, weighting using basal area.  We add the     !
   127)       ! cohorts only until when the canopy is closed, this way we will not bias the        !
   128)       ! vegetation height or the canopy depth towards the cohorts that live in the under-  !
   129)       ! storey.  Also, we must take into account the depth of the temporary surface water  !
   130)       ! or snow, because this will make the plants "shorter".                              !
   131)       !------------------------------------------------------------------------------------!
   132)       if (csite%opencan_frac(ipa) > 0.0) then
   133)          weight                  = cpatch%nplant(ico) * cpatch%basarea(ico)
   134)          weight_sum              = weight_sum + weight
   135)          csite%veg_height(ipa)   = csite%veg_height(ipa) + cpatch%hite(ico) * weight
   136)          csite%opencan_frac(ipa) = csite%opencan_frac(ipa) * (1.0 - cpatch%crown_area(ico))
   137)       end if
   138)       !------------------------------------------------------------------------------------!
   139) 
   140)    end do
   141)    !---------------------------------------------------------------------------------------!
   142) 
   143) 
   144) 
   145)    !----- Normalise the vegetation height, making sure that it is above the minimum. ------!
   146)    if (weight_sum > tiny(1.0)) then
   147)       csite%veg_height(ipa)  = max(veg_height_min,csite%veg_height(ipa) / weight_sum)
   148)    else
   149)       csite%veg_height(ipa)  = veg_height_min
   150)    end if
   151)    !---------------------------------------------------------------------------------------!
   152) 
   153) 
   154) 
   155)    !----- Find the patch roughness due to vegetation. -------------------------------------!
   156)    csite%veg_rough(ipa) = vh2vr * csite%veg_height(ipa)
   157)    !---------------------------------------------------------------------------------------!
   158) 
   159) 
   160) 
   161)    !----- Find the 0-plane displacement due to vegetation. --------------------------------!
   162)    csite%veg_displace(ipa) = vh2dh * csite%veg_height(ipa)
   163)    !---------------------------------------------------------------------------------------!
   164) 
   165) 
   166) 
   167)    !----- Update the canopy depth, and impose the minimum if needed be. -------------------!
   168)    csite%can_depth(ipa) = max(csite%veg_height(ipa), minimum_canopy_depth)
   169)    !---------------------------------------------------------------------------------------!
   170) 
   171) 
   172) 
   173)    !---------------------------------------------------------------------------------------!
   174)    !     Find the fraction of the canopy covered in snow.  I could not find any            !
   175)    ! reference for the original method (commented out), so I implemented the method used   !
   176)    ! in CLM-4, which is based on:                                                          !
   177)    !                                                                                       !
   178)    ! Niu, G.-Y., and Z.-L. Yang (2007), An observation-based formulation of snow cover     !
   179)    !    fraction and its evaluation over large North American river basins,                !
   180)    !    J. Geophys. Res., 112, D21101, doi:10.1029/2007JD008674                            !
   181)    !---------------------------------------------------------------------------------------!
   182)    ! csite%snowfac(ipa) = min(0.99, csite%total_sfcw_depth(ipa)/csite%veg_height(ipa))
   183)    if (total_sfcw_mass > tiny_sfcwater_mass) then
   184)       bulk_sfcw_dens     = max( fsdns, min( wdns                                           &
   185)                               , total_sfcw_mass / csite%total_sfcw_depth(ipa)))
   186)       csite%snowfac(ipa) = max( 0.0, min( 0.99                                             &
   187)                               , tanh( csite%total_sfcw_depth(ipa)                          &
   188)                                     / ( ny07_eq04_a * soil_rough                           &
   189)                                       * (bulk_sfcw_dens * fsdnsi) ** ny07_eq04_m ) ) ) )
   190)    else
   191)       csite%snowfac(ipa) = 0.0
   192)    end if
   193)    !---------------------------------------------------------------------------------------!
   194) 
   195) 
   196) 
   197)    !----- Find the PFT-dependent size distribution of this patch. -------------------------!
   198)    call patch_pft_size_profile(csite,ipa)
   199)    !---------------------------------------------------------------------------------------!
   200) 
   201) 
   202)    !----- Update the cohort count (may be redundant as well...) ---------------------------!
   203)    csite%cohort_count(ipa) = cpatch%ncohorts
   204)    !---------------------------------------------------------------------------------------!
   205) 
   206)    return
   207) end subroutine update_patch_derived_props

   218) !      This subroutine will take care of some diagnostic thermodynamic properties.         !
   219) !------------------------------------------------------------------------------------------!
   220) subroutine update_patch_thermo_props(csite,ipaa,ipaz,mzg,mzs,ntext_soil)
   221)   
   222)    use ed_state_vars, only : sitetype         ! ! structure
   223)    use therm_lib    , only : idealdenssh      & ! function
   224)                            , press2exner      & ! function
   225)                            , extheta2temp     & ! function
   226)                            , uextcm2tl        & ! function
   227)                            , uint2tl          ! ! function
   228)    use consts_coms  , only : t00              & ! intent(in)
   229)                            , wdns             ! ! intent(in)
   230)    use soil_coms    , only : soil             & ! intent(in)
   231)                            , matric_potential ! ! function
   232)    implicit none
   233) 
   234)    !----- Arguments -----------------------------------------------------------------------!
   235)    type(sitetype)                , target     :: csite
   236)    integer                       , intent(in) :: ipaa
   237)    integer                       , intent(in) :: ipaz
   238)    integer                       , intent(in) :: mzg
   239)    integer                       , intent(in) :: mzs
   240)    integer       , dimension(mzg), intent(in) :: ntext_soil
   241)    !----- Local variables. ----------------------------------------------------------------!
   242)    integer                                    :: ipa
   243)    integer                                    :: nsoil
   244)    integer                                    :: ksn
   245)    integer                                    :: k
   246)    real                                       :: soilhcap
   247)    real                                       :: can_exner
   248)    !---------------------------------------------------------------------------------------!
   249) 
   250) 
   251)    do ipa=ipaa,ipaz
   252) 
   253)       !----- Canopy air temperature and density. ------------------------------------------!
   254)       can_exner           = press2exner (csite%can_prss(ipa))
   255)       csite%can_temp(ipa) = extheta2temp(can_exner,csite%can_theta(ipa))
   256)       csite%can_rhos(ipa) = idealdenssh ( csite%can_prss  (ipa)                            &
   257)                                         , csite%can_temp  (ipa)                            &
   258)                                         , csite%can_shv   (ipa)                            )
   259)       !------------------------------------------------------------------------------------!
   260) 
   261) 
   262)       !----- Update soil temperature and liquid water fraction. ---------------------------!
   263)       do k = 1, mzg
   264)          nsoil    = ntext_soil(k)
   265)          soilhcap = soil(nsoil)%slcpd
   266)          call uextcm2tl(csite%soil_energy(k,ipa),csite%soil_water(k,ipa)*wdns,soilhcap     &
   267)                        ,csite%soil_tempk(k,ipa),csite%soil_fracliq(k,ipa))
   268)          csite%soil_mstpot(k,ipa) = matric_potential(nsoil,csite%soil_water(k,ipa))
   269)       end do
   270)       !------------------------------------------------------------------------------------!
   271) 
   272) 
   273) 
   274)       !----- Update temporary surface water temperature and liquid water fraction. --------!
   275)       ksn = csite%nlev_sfcwater(ipa)
   276)       csite%total_sfcw_depth(ipa) = 0.
   277)       do k = 1, ksn
   278)          call uint2tl(csite%sfcwater_energy(k,ipa),csite%sfcwater_tempk(k,ipa)             &
   279)                      ,csite%sfcwater_fracliq(k,ipa))
   280)          csite%total_sfcw_depth(ipa) =  csite%total_sfcw_depth(ipa)                        &
   281)                                      +  csite%sfcwater_depth(k,ipa)
   282)       end do
   283)       do k = ksn+1,mzs
   284)          if (k == 1) then
   285)             csite%sfcwater_tempk  (k,ipa) = csite%soil_tempk  (mzg,ipa)
   286)             csite%sfcwater_fracliq(k,ipa) = csite%soil_fracliq(mzg,ipa)
   287)          else
   288)             csite%sfcwater_tempk  (k,ipa) = csite%sfcwater_tempk  (k-1,ipa)
   289)             csite%sfcwater_fracliq(k,ipa) = csite%sfcwater_fracliq(k-1,ipa)
   290)          end if
   291)       end do
   292)       !------------------------------------------------------------------------------------!
   293)    end do
   294)    !---------------------------------------------------------------------------------------!
   295) 
   296)    return
   297) end subroutine update_patch_thermo_props

   308) !      This subroutine will update the fast mean properties, similarly to the routine      !
   309) ! above.                                                                                   !
   310) !------------------------------------------------------------------------------------------!
   311) subroutine update_patch_thermo_fmean(csite,ipaa,ipaz,mzg,ntext_soil)
   312)   
   313)    use ed_state_vars, only : sitetype           ! ! structure
   314)    use therm_lib    , only : idealdenssh        & ! function
   315)                            , press2exner        & ! function
   316)                            , extheta2temp       & ! function
   317)                            , uextcm2tl          & ! function
   318)                            , uint2tl            ! ! function
   319)    use consts_coms  , only : t00                & ! intent(in)
   320)                            , wdns               ! ! intent(in)
   321)    use soil_coms    , only : soil               & ! intent(in)
   322)                            , tiny_sfcwater_mass & ! intent(in)
   323)                            , matric_potential   ! ! function
   324)    implicit none
   325) 
   326)    !----- Arguments -----------------------------------------------------------------------!
   327)    type(sitetype)                , target     :: csite
   328)    integer                       , intent(in) :: ipaa
   329)    integer                       , intent(in) :: ipaz
   330)    integer                       , intent(in) :: mzg
   331)    integer       , dimension(mzg), intent(in) :: ntext_soil
   332)    !----- Local variables. ----------------------------------------------------------------!
   333)    integer                                    :: ipa
   334)    integer                                    :: nsoil
   335)    integer                                    :: k
   336)    real                                       :: soilhcap
   337)    real                                       :: can_exner
   338)    !---------------------------------------------------------------------------------------!
   339) 
   340) 
   341)    !---------------------------------------------------------------------------------------!
   342)    do ipa=ipaa,ipaz
   343) 
   344)       !----- Canopy air temperature and density. ------------------------------------------!
   345)       can_exner                 = press2exner (csite%fmean_can_prss(ipa))
   346)       csite%fmean_can_temp(ipa) = extheta2temp(can_exner,csite%fmean_can_theta(ipa))
   347)       csite%fmean_can_rhos(ipa) = idealdenssh ( csite%fmean_can_prss  (ipa)                &
   348)                                               , csite%fmean_can_temp  (ipa)                &
   349)                                               , csite%fmean_can_shv   (ipa)                )
   350)       !------------------------------------------------------------------------------------!
   351) 
   352) 
   353)       !----- Update soil temperature and liquid water fraction. ---------------------------!
   354)       do k = 1, mzg
   355)          nsoil    = ntext_soil(k)
   356)          soilhcap = soil(nsoil)%slcpd
   357)          call uextcm2tl( csite%fmean_soil_energy(k,ipa)                                    &
   358)                        , csite%fmean_soil_water (k,ipa) * wdns                             &
   359)                        , soilhcap                                                          &
   360)                        , csite%fmean_soil_temp  (k,ipa)                                    &
   361)                        , csite%fmean_soil_fliq  (k,ipa) )
   362)          csite%fmean_soil_mstpot(k,ipa) = matric_potential( nsoil                          &
   363)                                                           , csite%fmean_soil_water(k,ipa))
   364)       end do
   365)       !------------------------------------------------------------------------------------!
   366) 
   367) 
   368) 
   369) 
   370)       !------------------------------------------------------------------------------------!
   371)       !   If the patch had some temporary snow/pounding layer, convert the mean energy to  !
   372)       ! J/kg, then find the mean temperature and liquid fraction.  Otherwise, set them to  !
   373)       ! either zero or default values.                                                     !
   374)       !------------------------------------------------------------------------------------!
   375)       if (csite%fmean_sfcw_mass(ipa) > tiny_sfcwater_mass) then
   376)          csite%fmean_sfcw_energy(ipa) = csite%fmean_sfcw_energy(ipa)                       &
   377)                                       / csite%fmean_sfcw_mass(ipa)
   378)          call uint2tl(csite%fmean_sfcw_energy(ipa),csite%fmean_sfcw_temp(ipa)              &
   379)                      ,csite%fmean_sfcw_fliq(ipa))
   380)       else
   381)          csite%fmean_sfcw_mass  (ipa)  = 0.
   382)          csite%fmean_sfcw_depth (ipa)  = 0.
   383)          csite%fmean_sfcw_energy(ipa)  = 0.
   384)          csite%fmean_sfcw_temp  (ipa)  = csite%fmean_soil_temp(mzg,ipa)
   385)          csite%fmean_sfcw_fliq  (ipa)  = csite%fmean_soil_fliq(mzg,ipa)
   386)       end if
   387)       !-----------------------------------------------------------------------------------!
   388)    end do
   389)    return
   390) end subroutine update_patch_thermo_fmean

   401) !     This subroutine will update the derived properties at the site level.                !
   402) !------------------------------------------------------------------------------------------!
   403) subroutine update_site_derived_props(cpoly,census_flag,isi)
   404)   
   405)    use ed_state_vars , only : polygontype  & ! structure
   406)                             , sitetype     & ! structure
   407)                             , patchtype    ! ! structure
   408)    use consts_coms   , only : pio4         ! ! intent(in)
   409)    implicit none
   410)    !----- Arguments -----------------------------------------------------------------------!
   411)    type(polygontype) , target     :: cpoly
   412)    integer           , intent(in) :: census_flag
   413)    integer           , intent(in) :: isi
   414)    !----- Local variables -----------------------------------------------------------------!
   415)    type(sitetype)    , pointer    :: csite
   416)    type(patchtype)   , pointer    :: cpatch
   417)    integer                        :: bdbh
   418)    integer                        :: ipa
   419)    integer                        :: ico
   420)    integer                        :: ipft
   421)    !---------------------------------------------------------------------------------------!
   422)    
   423)    !----- Initialise the variables before looping. ----------------------------------------!
   424)    cpoly%basal_area(:,:,isi) = 0.0
   425)    cpoly%agb       (:,:,isi) = 0.0
   426) 
   427)    csite => cpoly%site(isi)
   428) 
   429)    !----- Loop over patches. --------------------------------------------------------------!
   430)    do ipa = 1,csite%npatches
   431) 
   432)       cpatch => csite%patch(ipa)
   433) 
   434)       !----- Loop over cohorts. -----------------------------------------------------------!
   435)       do ico = 1,cpatch%ncohorts
   436)          ipft = cpatch%pft(ico)
   437) 
   438)          !----- Update basal area and above-ground biomass. -------------------------------!
   439)          if(census_flag == 0 .or. cpatch%first_census(ico) == 1)then
   440)             bdbh = max(0,min( int(cpatch%dbh(ico) * 0.1), 10)) + 1
   441) 
   442)             cpoly%basal_area(ipft,bdbh,isi) = cpoly%basal_area(ipft, bdbh,isi)             &
   443)                                             + cpatch%basarea(ico) * cpatch%nplant(ico)     &
   444)                                             * csite%area(ipa)   
   445)             cpoly%agb(ipft,bdbh,isi)        = cpoly%agb(ipft, bdbh,isi)                    &
   446)                                             + cpatch%agb(ico)     * cpatch%nplant(ico)     &
   447)                                             * csite%area(ipa)
   448)          end if
   449)       end do
   450)    end do
   451)    
   452)    return
   453) end subroutine update_site_derived_props

   464) !     The following subroutine finds the polygon averages from site-, patch-, and cohort-  !
   465) ! -level properties whose time step is longer than DTLSM (days, months, years).  Fluxes,   !
   466) ! meteorological input, thermodynamic properties, and radiation are aggregated in  sub-    !
   467) ! -routine aggregate_polygon_fmean.                                                        !
   468) !------------------------------------------------------------------------------------------!
   469) subroutine update_polygon_derived_props(cgrid)
   470)    use ed_state_vars         , only : edtype             & ! structure
   471)                                     , polygontype        & ! structure
   472)                                     , sitetype           & ! structure
   473)                                     , patchtype          ! ! structure
   474)    use soil_coms             , only : dslz               ! ! intent(in)
   475)    use grid_coms             , only : nzg                ! ! intent(in)
   476)    use ed_max_dims           , only : n_dbh              ! ! intent(in)
   477)    use ed_misc_coms          , only : ddbhi              ! ! intent(in)
   478)    use pft_coms              , only : c2n_leaf           & ! intent(in)
   479)                                     , c2n_stem           & ! intent(in)
   480)                                     , c2n_storage        & ! intent(in)
   481)                                     , c2n_recruit        & ! intent(in)
   482)                                     , c2n_slow           & ! intent(in)
   483)                                     , c2n_structural     ! ! intent(in)
   484)    use decomp_coms           , only : cwd_frac           ! ! intent(in)
   485) 
   486)    implicit none
   487)    !----- Arguments.      -----------------------------------------------------------------!
   488)    type(edtype)         , target  :: cgrid
   489)    !----- Local variables. ----------------------------------------------------------------!
   490)    type(polygontype)    , pointer :: cpoly
   491)    type(sitetype)       , pointer :: csite
   492)    type(patchtype)      , pointer :: cpatch
   493)    integer                        :: ipy
   494)    integer                        :: isi
   495)    integer                        :: ipa
   496)    integer                        :: ico
   497)    integer                        :: p
   498)    integer                        :: d
   499)    integer                        :: k
   500)    real                           :: poly_area_i
   501)    real                           :: site_area_i
   502)    real                           :: site_wgt
   503)    real                           :: patch_wgt
   504)    real                           :: rdepth
   505)    !---------------------------------------------------------------------------------------!
   506)

   510)    !---------------------------------------------------------------------------------------!
   511)    !    WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! WARNING!   !
   512)    !---------------------------------------------------------------------------------------!
   513)    !     Please, don't initialise polygon-level (cgrid) variables outside polyloop.  This  !
   514)    ! works in off-line runs, but it causes memory leaks (and crashes) in the coupled runs  !
   515)    ! over the ocean, where cgrid%npolygons can be 0 if one of the sub-domains falls        !
   516)    ! entirely over the ocean.  Thanks!                                                     !
   517)    !---------------------------------------------------------------------------------------!
   518)    ! cgrid%blah = 0. !<<--- This is a bad way of doing, look inside the loop for the
   519)    !                 !      safe way of initialising the variable.
   520)    !---------------------------------------------------------------------------------------!
   521)    polyloop: do ipy=1,cgrid%npolygons
   522)       cpoly => cgrid%polygon(ipy)
   523) 
   524)       !------------------------------------------------------------------------------------!
   525)       !                                                                                    !
   526)       !     Initialise properties.                                                         !
   527)       !                                                                                    !
   528)       !     This is the right and safe place to initialise polygon-level (cgrid) vari-     !
   529)       ! ables, so in case npolygons is zero this will not cause memory leaks.  I know,     !
   530)       ! this never happens in off-line runs, but it is quite common in coupled runs...     !
   531)       ! Whenever one of the nodes receives a sub-domain where all the points are over the  !
   532)       ! ocean, ED will not assign any polygon in that sub-domain, which means that that    !
   533)       ! node will have 0 polygons, and the variables cannot be allocated.  If you try to   !
   534)       ! access the polygon level variable outside the loop, then the model crashes due to  !
   535)       ! segmentation violation (a bad thing), whereas by putting the variables here both   !
   536)       ! the off-line model and the coupled runs will work, because this loop will be       !
   537)       ! skipped when there is no polygon.                                                  !
   538)       !------------------------------------------------------------------------------------!
   539)       ! cgrid%blah(ipy) = 0.0 ! <<- This way works for all cases. 
   540)       !------------------------------------------------------------------------------------!
   541)       cgrid%nplant              (:,:,ipy) = 0.0
   542)       cgrid%agb                 (:,:,ipy) = 0.0
   543)       cgrid%lai                 (:,:,ipy) = 0.0
   544)       cgrid%wai                 (:,:,ipy) = 0.0
   545)       cgrid%basal_area          (:,:,ipy) = 0.0
   546)       cgrid%bdead               (:,:,ipy) = 0.0

   551)       cgrid%bsapwoodb           (:,:,ipy) = 0.0

   557)       cgrid%broot_n             (:,:,ipy) = 0.0
   558)       cgrid%bsapwooda_n         (:,:,ipy) = 0.0
   559)       cgrid%bsapwoodb_n         (:,:,ipy) = 0.0
   560)       cgrid%bseeds_n            (:,:,ipy) = 0.0
   561)       cgrid%bstorage_n          (:,:,ipy) = 0.0
   562)       cgrid%leaf_maintenance    (:,:,ipy) = 0.0
   563)       cgrid%root_maintenance    (:,:,ipy) = 0.0
   564)       cgrid%leaf_drop           (:,:,ipy) = 0.0

   566)       cgrid%slow_soil_c             (ipy) = 0.0
   567)       cgrid%struct_soil_c           (ipy) = 0.0
   568)       cgrid%struct_soil_l           (ipy) = 0.0
   569)       cgrid%cwd_c                   (ipy) = 0.0
   570)       cgrid%fast_soil_n             (ipy) = 0.0
   571)       cgrid%mineral_soil_n          (ipy) = 0.0
   572)       cgrid%cwd_n                   (ipy) = 0.0
   573)       !------------------------------------------------------------------------------------!
   574)       !     Some of these variables are redundant with variables above.  Perhaps we should !
   575)       ! find a single way to report them.                                                  !
   576)       !------------------------------------------------------------------------------------!
   577)       cgrid%Nbiomass_uptake         (ipy) = 0.0
   578)       cgrid%Cleaf_litter_flux       (ipy) = 0.0
   579)       cgrid%Croot_litter_flux       (ipy) = 0.0
   580)       cgrid%Nleaf_litter_flux       (ipy) = 0.0
   581)       cgrid%Nroot_litter_flux       (ipy) = 0.0

   583)       cgrid%Nnet_min                (ipy) = 0.0
   584)       !------------------------------------------------------------------------------------!
   585) 
   586) 
   587) 
   588) 
   589)       !----- Inverse of this polygon area (it should be always 1.) ------------------------!
   590)       poly_area_i = 1./sum(cpoly%area)
   591)       !------------------------------------------------------------------------------------!
   592) 
   593) 
   594)       !------------------------------------------------------------------------------------!
   595)       !     Loop over sites.                                                               !
   596)       !------------------------------------------------------------------------------------!
   597)       siteloop: do isi=1,cpoly%nsites
   598)          csite => cpoly%site(isi)
   599)

   601)          site_area_i=1./sum(csite%area)

   603) 
   604) 
   605)          !----- Site weight. --------------------------------------------------------------!
   606)          site_wgt = cpoly%area(isi) * poly_area_i

   609)

   613)          patchloop: do ipa=1,csite%npatches
   614)             cpatch => csite%patch(ipa)
   615) 
   616) 
   617)             !----- Site weight. -----------------------------------------------------------!
   618)             patch_wgt = csite%area(ipa) * site_area_i * site_wgt
   619)             !------------------------------------------------------------------------------!
   620) 
   621) 
   622)             !----- Integrate soil properties. ---------------------------------------------!
   623)             cgrid%fast_soil_c   (ipy) = cgrid%fast_soil_c        (ipy)                     &
   624)                                       + csite%fast_soil_c        (ipa)                     &
   625)                                       * patch_wgt

   628)                                       * patch_wgt

   656) 
   657) 
   658)             !----- Zero the root fraction (patch-level diagnostic). -----------------------!
   659)             csite%rootdense(:,ipa) = 0.

   661) 
   662)

   669)                d = max(1,min(n_dbh,ceiling(cpatch%dbh(ico)*ddbhi)))
   670)                !---------------------------------------------------------------------------!
   671) 
   672) 
   673) 
   674)                !---------------------------------------------------------------------------!

   676)                !    broot*nplant/rooting_depth   [kg/plant]*[plant/m2]/[m]                 !
   677)                !---------------------------------------------------------------------------!
   678)                rdepth = sum(dslz(cpatch%krdepth(ico):nzg))
   679)                do k=cpatch%krdepth(ico),nzg
   680)                   csite%rootdense(k,ipa) = csite%rootdense(k,ipa)                          &
   681)                                          + cpatch%broot(ico)*cpatch%nplant(ico) / rdepth
   682)                end do
   683)                !---------------------------------------------------------------------------!
   684) 
   685) 
   686) 
   687)                !---------------------------------------------------------------------------!
   688)                !    Integrate cohort-based properties.  Make sure that the polygon-        !
   689)                ! -level gets the right units (i.e., no /plant, but /m2).                   !
   690)                !---------------------------------------------------------------------------!
   691)                cgrid%nplant          (p,d,ipy) = cgrid%nplant          (p,d,ipy)           &
   692)                                                + cpatch%nplant             (ico)           &
   693)                                                * patch_wgt
   694)                cgrid%lai             (p,d,ipy) = cgrid%lai             (p,d,ipy)           &
   695)                                                + cpatch%lai                (ico)           &
   696)                                                * patch_wgt
   697)                cgrid%wai             (p,d,ipy) = cgrid%wai             (p,d,ipy)           &
   698)                                                + cpatch%wai                (ico)           &
   699)                                                * patch_wgt
   700)                cgrid%agb             (p,d,ipy) = cgrid%agb             (p,d,ipy)           &
   701)                                                + cpatch%agb                (ico)           &
   702)                                                * cpatch%nplant             (ico)           &
   703)                                                * patch_wgt
   704)                cgrid%basal_area      (p,d,ipy) = cgrid%basal_area      (p,d,ipy)           &
   705)                                                + cpatch%basarea            (ico)           &
   706)                                                * cpatch%nplant             (ico)           &
   707)                                                * patch_wgt
   708)                cgrid%bdead           (p,d,ipy) = cgrid%bdead           (p,d,ipy)           &
   709)                                                + cpatch%bdead              (ico)           &
   710)                                                * cpatch%nplant             (ico)           &
   711)                                                * patch_wgt
   712)                cgrid%balive          (p,d,ipy) = cgrid%balive          (p,d,ipy)           &
   713)                                                + cpatch%balive             (ico)           &
   714)                                                * cpatch%nplant             (ico)           &
   715)                                                * patch_wgt
   716)                cgrid%bleaf           (p,d,ipy) = cgrid%bleaf           (p,d,ipy)           &
   717)                                                + cpatch%bleaf              (ico)           &
   718)                                                * cpatch%nplant             (ico)           &
   719)                                                * patch_wgt
   720)                cgrid%broot           (p,d,ipy) = cgrid%broot           (p,d,ipy)           &
   721)                                                + cpatch%broot              (ico)           &
   722)                                                * cpatch%nplant             (ico)           &
   723)                                                * patch_wgt
   724)                cgrid%bsapwooda       (p,d,ipy) = cgrid%bsapwooda       (p,d,ipy)           &
   725)                                                + cpatch%bsapwooda          (ico)           &
   726)                                                * cpatch%nplant             (ico)           &
   727)                                                * patch_wgt
   728)                cgrid%bsapwoodb       (p,d,ipy) = cgrid%bsapwoodb       (p,d,ipy)           &
   729)                                                + cpatch%bsapwoodb          (ico)           &
   730)                                                * cpatch%nplant             (ico)           &
   731)                                                * patch_wgt
   732)                cgrid%bseeds          (p,d,ipy) = cgrid%bseeds          (p,d,ipy)           &
   733)                                                + cpatch%bseeds             (ico)           &
   734)                                                * cpatch%nplant             (ico)           &
   735)                                                * patch_wgt
   736)                cgrid%bstorage        (p,d,ipy) = cgrid%bstorage        (p,d,ipy)           &
   737)                                                + cpatch%bstorage           (ico)           &
   738)                                                * cpatch%nplant             (ico)           &
   739)                                                * patch_wgt
   740)                cgrid%bdead_n         (p,d,ipy) = cgrid%bdead_n         (p,d,ipy)           &
   741)                                                + cpatch%bdead              (ico)           &
   742)                                                / c2n_stem                    (p)           &
   743)                                                * cpatch%nplant             (ico)           &
   744)                                                * patch_wgt
   745)                cgrid%balive_n        (p,d,ipy) = cgrid%balive_n        (p,d,ipy)           &
   746)                                                + ( ( cpatch%bleaf          (ico)           &
   747)                                                    + cpatch%broot          (ico) )         &
   748)                                                    / c2n_leaf                (p)           &
   749)                                                  + ( cpatch%bsapwooda      (ico)           &
   750)                                                    + cpatch%bsapwoodb      (ico) )         &
   751)                                                    / c2n_stem                (p)   )       &
   752)                                                * cpatch%nplant             (ico)           &
   753)                                                * patch_wgt
   754)                cgrid%bleaf_n         (p,d,ipy) = cgrid%bleaf_n         (p,d,ipy)           &
   755)                                                + cpatch%bleaf              (ico)           &
   756)                                                / c2n_leaf                    (p)           &
   757)                                                * cpatch%nplant             (ico)           &
   758)                                                * patch_wgt
   759)                cgrid%broot_n         (p,d,ipy) = cgrid%broot_n         (p,d,ipy)           &
   760)                                                + cpatch%broot              (ico)           &
   761)                                                / c2n_leaf                    (p)           &
   762)                                                * cpatch%nplant             (ico)           &
   763)                                                * patch_wgt
   764)                cgrid%bsapwooda_n     (p,d,ipy) = cgrid%bsapwooda_n     (p,d,ipy)           &
   765)                                                + cpatch%bsapwooda          (ico)           &
   766)                                                / c2n_stem                    (p)           &
   767)                                                * cpatch%nplant             (ico)           &
   768)                                                * patch_wgt
   769)                cgrid%bsapwoodb_n     (p,d,ipy) = cgrid%bsapwoodb_n     (p,d,ipy)           &
   770)                                                + cpatch%bsapwoodb          (ico)           &
   771)                                                / c2n_stem                    (p)           &
   772)                                                * cpatch%nplant             (ico)           &
   773)                                                * patch_wgt
   774)                cgrid%bseeds_n        (p,d,ipy) = cgrid%bseeds_n        (p,d,ipy)           &
   775)                                                + cpatch%bseeds             (ico)           &
   776)                                                / c2n_recruit                 (p)           &
   777)                                                * cpatch%nplant             (ico)           &
   778)                                                * patch_wgt
   779)                cgrid%bstorage_n      (p,d,ipy) = cgrid%bstorage_n      (p,d,ipy)           &
   780)                                                + cpatch%bstorage           (ico)           &
   781)                                                / c2n_storage                               &
   782)                                                * cpatch%nplant             (ico)           &
   783)                                                * patch_wgt
   784)                cgrid%leaf_maintenance(p,d,ipy) = cgrid%leaf_maintenance(p,d,ipy)           &
   785)                                                + cpatch%leaf_maintenance   (ico)           &
   786)                                                * cpatch%nplant             (ico)           &
   787)                                                * patch_wgt

   790)                                                * cpatch%nplant             (ico)           &
   791)                                                * patch_wgt
   792)                cgrid%leaf_drop       (p,d,ipy) = cgrid%leaf_drop       (p,d,ipy)           &
   793)                                                + cpatch%leaf_drop          (ico)           &
   794)                                                * cpatch%nplant             (ico)           &
   795)                                                * patch_wgt

   797)

   800)                !   CHECK! CHECK! CHECK! CHECK! CHECK! CHECK! CHECK! CHECK! CHECK! CHECK!   !
   801)                !---------------------------------------------------------------------------!

   803)                ! moved them to here just to be consistent (they are not "dmean"            !
   804)                ! variables).  However, they are slightly different than the original,      !
   805)                ! which were missing nplant (so the results were in not in units of         !
   806)                ! kg/m2/yr, but kg/plant/yr).                                               !
   807)                !---------------------------------------------------------------------------!
   808)                cgrid%Cleaf_litter_flux(ipy) = cgrid%Cleaf_litter_flux(ipy)                 &
   809)                                             + cpatch%leaf_maintenance(ico)                 &
   810)                                             * cpatch%nplant          (ico)                 &
   811)                                             * patch_wgt
   812)                cgrid%Croot_litter_flux(ipy) = cgrid%Croot_litter_flux(ipy)                 &
   813)                                             + cpatch%root_maintenance(ico)                 &
   814)                                             * cpatch%nplant          (ico)                 &
   815)                                             * patch_wgt
   816)                cgrid%Nleaf_litter_flux(ipy) = cgrid%Cleaf_litter_flux(ipy)                 &
   817)                                             + cpatch%leaf_maintenance(ico) / c2n_leaf(p)   &
   818)                                             * cpatch%nplant          (ico)                 &
   819)                                             * patch_wgt
   820)                cgrid%Nroot_litter_flux(ipy) = cgrid%Croot_litter_flux(ipy)                 &
   821)                                             + cpatch%root_maintenance(ico) / c2n_leaf(p)   &
   822)                                             * cpatch%nplant          (ico)                 &
   823)                                             * patch_wgt
   824)                !---------------------------------------------------------------------------!
   825)             end do cohortloop
   826)             !------------------------------------------------------------------------------!
   827) 
   828) 
   829) 
   830)             !------------------------------------------------------------------------------!
   831)             !    Update the patch-related N budget variables.                              !
   832)             !------------------------------------------------------------------------------!
   833)             cgrid%Ngross_min     (ipy) = cgrid%Ngross_min(ipy)                             &
   834)                                        + csite%mineralized_N_input   (ipa)   * patch_wgt
   835)             cgrid%Ngross_min     (ipy) = cgrid%Ngross_min(ipy)                             &
   836)                                        + ( csite%mineralized_N_input (ipa)                 &
   837)                                          - csite%mineralized_N_loss  (ipa) ) * patch_wgt
   838)             cgrid%Nbiomass_uptake(ipy) = cgrid%Ngross_min(ipy)                             &
   839)                                        + csite%total_plant_nitrogen_uptake(ipa)* patch_wgt
   840)             !------------------------------------------------------------------------------!
   841) 
   842)          end do patchloop
   843)          !---------------------------------------------------------------------------------!
   844)       end do siteloop
   845)       !------------------------------------------------------------------------------------!
   846)    end do polyloop
   847)    !---------------------------------------------------------------------------------------!
   848)    return

   861) !------------------------------------------------------------------------------------------!

   863)

   865)                             , polygontype  & ! structure

   874)    use grid_coms     , only : nzg          & ! intent(in)
   875)                             , nzs          ! ! intent(in)

   882)    type(sitetype)    , pointer    :: csite          ! Alias for current site

   885)    integer                        :: ilat           !
   886)    integer                        :: ilon           !
   887)    integer                        :: ilatf          !
   888)    integer                        :: ilonf          !
   889)    integer                        :: nls            !
   890)    integer                        :: nlsw1          !
   891)    integer                        :: k              !
   892)    integer                        :: ipy            !
   893)    integer                        :: isi            !
   894)    integer                        :: ipa            !
   895)    logical                        :: l1             !
   896)    real                           :: glat           !
   897)    real                           :: glon           !
   898)    real                           :: tmp1           !
   899)    real                           :: tmp2           !
   900)    real                           :: soilw1         !
   901)    real                           :: soilw2         !
   902)    !----- Local constants.  ---------------------------------------------------------------!
   903)    logical           , parameter  :: harvard_override = .false.
   904)    integer           , parameter  :: nlon = 144
   905)    integer           , parameter  :: nlat = 73
   906)    real              , parameter  :: dlon = 2.5
   907)    real              , parameter  :: dlat = 2.5
   908)    !---------------------------------------------------------------------------------------!
   909) 
   910)    !----- First thing, check whether the dataset exists and crash the run if it doesn�t. --!
   911)    inquire(file=trim(soilstate_db),exist=l1)
   912)    if (.not.l1) then
   913)       write (unit=*,fmt='(a)') ' Your namelist has ISOILSTATEINIT set to read the initial'

   916)       call fatal_error('Soil database '//trim(soilstate_db)//' not found!'                 &
   917)                      &,'read_soil_moist_temp','update_derived_props.f90')

   921)

   923)    latloop: do ilatf = 1,nlat
   924)

   926)       lonloop: do ilonf = 1,nlon 
   927)  
   928)          !---------------------------------------------------------------------------------!
   929)          !     Read in reanalysis: two temperatures and moistures, corresponding to        !
   930)          ! different depths.                                                               !

   937)          if (tmp1 > 0.0 .and. tmp2 > 0.0 .and. soilw1 > 0.0 .and. soilw2 > 0.0) then
   938) 
   939)             !----- Loop over land points. -------------------------------------------------!
   940)             polyloop: do ipy=1,cgrid%npolygons
   941)                cpoly => cgrid%polygon(ipy)
   942) 
   943)                !----- Land point lat, lon. ------------------------------------------------!
   944)                glat = cgrid%lat(ipy)

  1092) !     This sub-routine updates the model time.  It was moved to here to reduce the number  !

  1094) !------------------------------------------------------------------------------------------!
  1095) subroutine update_model_time_dm(ctime,dtlong)
  1096) 
  1097)    use ed_misc_coms, only : simtime ! ! variable type
  1098)    use consts_coms , only : day_sec & ! intent(in)
  1099)                           , hr_sec  & ! intent(in)
  1100)                           , min_sec ! ! intent(in)
  1101)    implicit none
  1102)    !----- Arguments. ----------------------------------------------------------------------!
  1103)    type(simtime)         , intent(inout) :: ctime  ! Current time
  1104)    real                  , intent(in)    :: dtlong ! Model time step
  1105)    !----- Local variables. ----------------------------------------------------------------!
  1106)    integer               , dimension(12) :: daymax
  1107)    !----- External functions. -------------------------------------------------------------!
  1108)    logical               , external      :: isleap ! This function check for leap years
  1109)    !---------------------------------------------------------------------------------------!
  1110) 
  1111) 
  1112) 
  1113)    !----- Assume that the year is not leap. -----------------------------------------------!
  1114)    daymax=(/31,28,31,30,31,30,31,31,30,31,30,31/)
  1115)    !---------------------------------------------------------------------------------------!
  1116) 
  1117) 
  1118)    !----- Update the time. ----------------------------------------------------------------!
  1119)    ctime%time = ctime%time + dtlong
  1120)    !---------------------------------------------------------------------------------------!
  1121) 
  1122)    if (ctime%time >= day_sec)then
  1123) 
  1124)       !----- If time is greater than one day, update the day. -----------------------------!
  1125)       ctime%time = ctime%time - day_sec
  1126)       ctime%date = ctime%date + 1
  1127) 
  1128)       !----- If the year is leap, correct February's number of days. ----------------------!
  1129)       if (isleap(ctime%year)) daymax(2) = 29
  1130) 
  1131)       !----- If we have reached the end of the month, update the month. -------------------!
  1132)       if (ctime%date > daymax(ctime%month)) then
  1133)          ctime%date  = 1
  1134)          ctime%month = ctime%month + 1
  1135) 
  1136)          !------ If we have reached the end of the year, update the year. -----------------!
  1137)          if (ctime%month == 13) then
  1138)             ctime%month = 1
  1139)             ctime%year = ctime%year + 1
  1140)          end if
  1141)       end if
  1142) 
  1143)    elseif (ctime%time < 0.0) then
  1144)       !----- Time is negative, go back one day. -------------------------------------------!
  1145)       ctime%time = ctime%time + day_sec
  1146)       ctime%date = ctime%date - 1
  1147) 
  1148)       !----- Day is zero, which means we must go to the previous month. -------------------!
  1149)       if (ctime%date == 0) then
  1150)          ctime%month = ctime%month - 1
  1151) 
  1152)          !----- Month is zero, which means that we must go to the previous year. ----------!
  1153)          if (ctime%month == 0) then
  1154)             ctime%month = 12
  1155)             ctime%year = ctime%year - 1
  1156)          end if
  1157) 
  1158)          !---------------------------------------------------------------------------------!
  1159)          !     Fix the month in case it it a leap year, and make the day the last of the   !
  1160)          ! previous month.                                                                 !
  1161)          !---------------------------------------------------------------------------------!
  1162)          if (isleap(ctime%year)) daymax(2) = 29
  1163)          ctime%date = daymax(ctime%month)
  1164)       end if
  1165)    end if
  1166)    !---------------------------------------------------------------------------------------!
  1167) 
  1168) 
  1169) 
  1170)    !----- Update the hours, minutes, and seconds. -----------------------------------------!
  1171)    ctime%hour = floor(ctime%time / hr_sec)
  1172)    ctime%min  = floor((ctime%time - real(ctime%hour)*hr_sec)/min_sec)
  1173)    ctime%sec  = floor(ctime%time - real(ctime%hour)*hr_sec - real(ctime%min)*min_sec)
  1174)    !---------------------------------------------------------------------------------------!
  1175) 
  1176) 
  1177)    return
  1178) end subroutine update_model_time_dm
