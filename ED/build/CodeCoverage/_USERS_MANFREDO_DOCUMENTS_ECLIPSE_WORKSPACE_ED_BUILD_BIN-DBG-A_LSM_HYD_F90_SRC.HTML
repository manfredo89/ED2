    37) ! Initializes Lateral Hydrology
    38) ! defines initial state of hydrology parameters
    39) ! and redistributes soil moisture initial condition among sites
    40) ! to be in "equilibrium" and avoid long spinup
    41) ! inputs: polygon_list - list of ED2 polygons for all grid resolutions
    42) ! NOTE: skip if restarting from binary history dump
    43) !==========================================================================================!

    45) subroutine initHydrology()
    46) 
    47)   use hydrology_constants
    48)   use grid_coms,only: ngrids
    49)   use ed_state_vars, only : edgrid_g
    50)   use ed_node_coms, only: mynum
    51)   implicit none
    52)   
    53)   integer :: igr,ipy
    54)   integer :: mpolys,msites
    55)   
    56)   !! Initialize Subsurface lateral flow
    57)   !! by doing a bulk redistribution of polygon water
    58)   !! to it's initial "equilibrium" state across sites
    59)   call initHydroSubsurface()
    60)   mpolys=0
    61)   msites=0
    62)   do igr=1,ngrids
    63)      mpolys=max(mpolys,edgrid_g(igr)%npolygons)
    64)      do ipy=1,edgrid_g(igr)%npolygons
    65)         msites=max(msites,edgrid_g(igr)%polygon(ipy)%nsites)
    66)      end do
    67)   end do
    68)   write(unit=*,fmt='(4(a,1x,i9,1x))')  &
    69)      'initHydrology | mynum=',mynum,'ngrids=',ngrids,'mpolys=',mpolys,'msites=',msites
    70)   allocate(qw4out(ngrids,mpolys,msites))  !!just for debugging
    71)   allocate(qh4out(ngrids,mpolys,msites))  !!just for debugging
    72)   qw4out = 0.0
    73)   qh4out = 0.0
    74)   write(unit=*,fmt='(4(a,1x,i9,1x))')  &
    75)      'Allocated | mynum=',mynum,'ngrids=',ngrids,'mpolys=',mpolys,'msites=',msites
    76) 
    77)   !!initialize runoff parameters
    78)   do igr=1,ngrids
    79)      call updateHydroParms (edgrid_g(igr))
    80)   end do
    81)   write(unit=*,fmt='(4(a,1x,i9,1x))')  &
    82)      'Updated | mynum=',mynum,'ngrids=',ngrids,'mpolys=',mpolys,'msites=',msites
    83)   deallocate(qw4out,qh4out)
    84)   write(unit=*,fmt='(4(a,1x,i9,1x))')  &
    85)      'Deallocated | mynum=',mynum,'ngrids=',ngrids,'mpolys=',mpolys,'msites=',msites
    86)   return
    87) end subroutine initHydrology

    98) ! code for initializing site hydrology by shifting the soil water among sites 
    99) ! to initial equilibrium state
   100) ! inputs: polygon_list - list of ED2 polygons for all grid resolutions
   101) ! note: ** initialization routine does not conserve water or energy **
   102) !==========================================================================================!
   103) !==========================================================================================!
   104) subroutine initHydroSubsurface()
   105)   use hydrology_constants
   106)   use hydrology_coms, only: useTOPMODEL, MoistRateTuning
   107)   use soil_coms, only: soil,slz,dslz,dslzi
   108)   use grid_coms, only: ngrids,nzg
   109)   use ed_misc_coms, only: dtlsm
   110)   use ed_state_vars, only: edgrid_g,edtype,polygontype,sitetype
   111)   use consts_coms, only: wdns,t3ple
   112)   use therm_lib, only: uextcm2tl, cmtl2uext
   113)   implicit none
   114) 
   115)   type(edtype)      , pointer :: cgrid
   116)   type(polygontype) , pointer :: cpoly
   117)   type(sitetype)    , pointer :: csite
   118)   integer :: igr, ipy, isi, ipa, k,  nsoil,slsl
   119)   real :: zbar_site,area_poly,zmin,fliq
   120)   !! SEE calcHydroSubsurface for variable definitions
   121) 
   122)   if(useTOPMODEL == 0) return
   123) 
   124)   !! Calculates TOPMODEL water table adjustment
   125) 
   126)   !! loop over grids and polygons
   127)   do igr = 1,ngrids
   128)      cgrid => edgrid_g(igr)
   129)      !initialize variables
   130) 
   131)      do ipy=1,cgrid%npolygons
   132)         cpoly => cgrid%polygon(ipy)
   133)         cgrid%zbar(ipy) = 0.0
   134)         !! FIRST calculate mean watertable (zbar)
   135)         area_poly = 0.0        !proportion
   136) 
   137)         do isi=1,cpoly%nsites
   138)            csite => cpoly%site(isi)
   139)            zbar_site = 0.0  !reset site vars
   140) 
   141)            do ipa=1,csite%npatches
   142)               !!first calculate water table depth for each patch 
   143)               call calcWatertable(cpoly,isi,ipa)
   144)               !!update site mean variables
   145)               zbar_site   = zbar_site   + csite%watertable(ipa)*csite%area(ipa)
   146)            enddo
   147)            !update polygon mean variables
   148)            area_poly       = area_poly       + cpoly%area(isi)
   149)            cgrid%zbar(ipy) = cgrid%zbar(ipy) + zbar_site*cpoly%area(isi)
   150)         end do
   151)         !normalize by non-water site areas
   152)         cgrid%zbar(ipy)      = cgrid%zbar(ipy)/area_poly 
   153) 
   154)         !!NEXT set new watertable depth
   155)         do isi=1,cpoly%nsites
   156) 
   157)            csite => cpoly%site(isi)
   158)            slsl=cpoly%lsl(isi)
   159)            nsoil=cpoly%ntext_soil(slsl,isi)
   160) 
   161)            cpoly%moist_zi(isi) = cgrid%zbar(ipy) + (cpoly%moist_W(isi) - cgrid%wbar(ipy))/cpoly%moist_f(isi)/MoistRateTuning !TOPMODEL equilibrium water depth
   162) 
   163)            zmin = slz(slsl) + soil(nsoil)%soilcp/soil(nsoil)%slmsts*dslz(slsl) 
   164)            !zmin = minimum height of water table assuming lowest soil layer is at minimal moisture level
   165) 
   166)            cpoly%moist_zi(isi) = max(cpoly%moist_zi(isi),zmin)
   167)            
   168)            do ipa=1,csite%npatches
   169)               !!set soil moisture at specified height
   170)               !!loop from bottom up 
   171)               do k = cpoly%lsl(isi),nzg
   172)                  nsoil = cpoly%ntext_soil(k,isi) !! mcd [9/30/08]
   173)                  if(cpoly%moist_zi(isi) < slz(k+1)) then
   174)                     csite%soil_water(k,ipa) = max(soil(nsoil)%soilcp, &
   175)                          soil(nsoil)%slmsts*(cpoly%moist_zi(isi)-slz(k))*dslzi(k))
   176)                     exit 
   177)                  else
   178)                     !! set soil to saturated
   179)                     csite%soil_water(k,ipa) = soil(nsoil)%slmsts
   180)                  endif
   181)               end do
   182)               !! reset energy
   183)               do k = cpoly%lsl(isi),nzg
   184)                  if (csite%soil_tempk(k,ipa) > t3ple) then
   185)                     fliq = 1.0
   186)                  elseif (csite%soil_tempk(k,ipa) < t3ple) then
   187)                     fliq = 0.0
   188)                  else
   189)                     fliq = 0.5
   190)                  end if
   191)                  
   192)                  csite%soil_energy(k,ipa)  = cmtl2uext( soil(nsoil)%slcpd                  &
   193)                                                       , csite%soil_water(k,ipa)  * wdns    &
   194)                                                       , csite%soil_tempk(k,ipa)            &
   195)                                                       , fliq                               )
   196)               end do
   197)               call calcWatertable(cpoly,isi,ipa)
   198)            enddo
   199)            cpoly%moist_tau(isi) = 100000.0*dtlsm !!set to some small rate initially
   200)         enddo
   201)      enddo
   202)   enddo
   203) end subroutine initHydroSubsurface

   214) subroutine calcHydroSubsurface()
   215) !! Main function for calculation subsurface lateral hydrologic flow
   216) !! assumed to be called on the fast timestep dtlsm
   217)   use ed_state_vars, only : edtype,polygontype,sitetype,edgrid_g
   218)   use hydrology_constants
   219)   use hydrology_coms, only: useTOPMODEL, FracLiqRunoff,MoistRateTuning
   220)   use soil_coms, only: soil,slz,dslz
   221)   use grid_coms, only: ngrids,nzg
   222)   use ed_misc_coms, only: dtlsm
   223)   use therm_lib, only: uint2tl,uextcm2tl
   224)   use consts_coms, only: wdns
   225)   implicit none
   226) 
   227)   type(edtype)      , pointer :: cgrid                ! Alias for current ED grid-type
   228)   type(polygontype) , pointer :: cpoly                ! Alias for current polygon
   229)   type(sitetype)    , pointer :: csite                ! Alias for current site
   230)   integer                     :: igr,ipy,isi,ipa      ! Grid,Polygon,Site and Patch Counters
   231)   real                        :: zbar_site            ! site area-weighted mean water table depth (m)
   232)   real                        :: area_land            ! polygon area that is land (proportion) 
   233)   real                        :: area_subtract        ! polygon area that is has water removed (proportion) 
   234)   real                        :: dz                   ! change in water table depth over time step (m)
   235)   real                        :: dzadd                ! sum of net additions to the water table (m)
   236)   real                        :: dw                   ! change in water content (m3/m2)
   237)   real                        :: tempk                ! water temperature (K)
   238)   real                        :: fracliq,fracliqtotal ! fraction of moisture that is liquid (proportion)
   239)   real                        :: sheat                ! soil heat (units??)
   240)   real                        :: swater               ! soil water (units??)
   241)   real                        :: zmin                 ! minimum water table height (m)
   242)   real                        :: soil_sat_energy      ! polygon-level mean energy in saturated zone (J/m2)
   243)   real                        :: soil_sat_water       ! polygon-level mean water volume in saturated zone  (m3/m2)
   244)   real                        :: soil_sat_heat        ! polygon-level mean soil heat capacity in saturated zone (J/m2)
   245)   real                        :: energy_site          ! site-level mean energy in saturated zone (J/m2)
   246)   real                        :: water_site           ! site-level mean water in saturated zone  (m/m2)
   247)   real                        :: heat_site            ! site-level mean soil heat capacity in saturated zone (J/m2)
   248)   real                        :: bf_site, bf_patch
   249)   integer                     :: slsl,nsoil
   250) 
   251)   !!******************************************************************************!!
   252)   !! If not using TOPMODEL, just do water table calculation then return           !!
   253)   !!******************************************************************************!!
   254)   if(useTOPMODEL == 0)then
   255)      do igr = 1,ngrids
   256)         cgrid => edgrid_g(igr)
   257)         do ipy=1,cgrid%npolygons
   258)            cpoly => cgrid%polygon(ipy)
   259)            cgrid%zbar(ipy)     = 0.0
   260)            cgrid%baseflow(ipy) = 0.0
   261)            area_land = 0.0        !proportion
   262)            do isi=1,cpoly%nsites
   263)               csite => cpoly%site(isi)
   264)               zbar_site   = 0.0  !reset site vars
   265)               do ipa=1,csite%npatches
   266)                  !!first calculate water table depth for each patch 
   267)                  call calcWatertable(cpoly,isi,ipa)
   268)                  !!update site mean variables
   269)                  zbar_site = zbar_site + csite%watertable(ipa)*csite%area(ipa)
   270)               end do
   271)               !update polygon mean variables
   272)               area_land       = area_land + cpoly%area(isi)
   273)               cgrid%zbar(ipy) = cgrid%zbar(ipy)+ zbar_site * cpoly%area(isi)
   274)            enddo
   275)            !normalize variables by non-water site areas
   276)            cgrid%zbar(ipy) = cgrid%zbar(ipy)/area_land 
   277)         enddo
   278)      enddo
   279)      return
   280)   endif
   281) 
   282)   ! Calculates TOPMODEL water table adjustment
   283) 
   284) 
   285)   do igr = 1,ngrids
   286)      cgrid => edgrid_g(igr)
   287)      do ipy=1,cgrid%npolygons
   288)         cpoly => cgrid%polygon(ipy)
   289) 
   290)         !!*************************************************************************!!
   291)         !! Initialize variables                                                    !!
   292)         !!*************************************************************************!!
   293)         cgrid%zbar(ipy)     = 0.0 !! Polygon-average water table depth (m)
   294)         cgrid%baseflow(ipy) = 0.0 !! Polygon total baseflow
   295)         soil_sat_energy = 0.0  !mean energy in saturated zone (J/m2)
   296)         soil_sat_water = 0.0   !mean water volume in saturated zone  (m3/m2)
   297)         soil_sat_heat = 0.0    !mean soil heat capacity in saturated zone (J/m2)
   298)         area_land = 0.0        !proportion of polygon that is land
   299) 
   300) 
   301)         !!*************************************************************************!!
   302)         !! Calculate patch-level watertable depth, polygon-level mean watertable   !!
   303)         !! depth (ZBAR) and find out the amount of water, energy, and heat capacity!!
   304)         !! is in the saturated zone                                                !! 
   305)         !!*************************************************************************!!
   306)         do isi=1,cpoly%nsites
   307)            csite => cpoly%site(isi)
   308) 
   309)            zbar_site   = 0.0  !reset site vars
   310)            energy_site = 0.0
   311)            heat_site   = 0.0
   312)            water_site  = 0.0
   313) 
   314)            do ipa=1,csite%npatches
   315) 
   316)               !!first calculate water table depth for each patch 
   317)               call calcWatertable(cpoly,isi,ipa)
   318) 
   319)               !!update site-level mean variables
   320)               zbar_site   = zbar_site   + csite%watertable(ipa)      * csite%area(ipa)
   321)               energy_site = energy_site + csite%soil_sat_energy(ipa) * csite%area(ipa)
   322)               heat_site   = heat_site   + csite%soil_sat_heat(ipa)   * csite%area(ipa)
   323)               water_site  = water_site  + csite%soil_sat_water(ipa)  * csite%area(ipa)
   324) 
   325)               !!zero variables to be used soon
   326)               csite%moist_dz(ipa) = 0.0
   327)            end do
   328)            !update polygon mean variables
   329)            area_land       = area_land       +                  cpoly%area(isi)
   330)            cgrid%zbar(ipy) = cgrid%zbar(ipy) + zbar_site      * cpoly%area(isi)
   331)            soil_sat_energy = soil_sat_energy + energy_site    * cpoly%area(isi)
   332)            soil_sat_heat   = soil_sat_heat   + heat_site      * cpoly%area(isi)
   333)            soil_sat_water  = soil_sat_water  + water_site     * cpoly%area(isi)
   334) 
   335)         end do
   336) 
   337)         !!*************************************************************************!!
   338)         !! normalize variables by non-water site areas                             !!
   339)         !!*************************************************************************!!
   340)         cgrid%zbar(ipy) = cgrid%zbar(ipy)/area_land 
   341)         soil_sat_energy = soil_sat_energy/area_land
   342)         soil_sat_heat   = soil_sat_heat/area_land
   343)         soil_sat_water  = soil_sat_water/area_land
   344) 
   345)         !!*************************************************************************!!
   346)         !! Compute mean temperature and liquid fraction of soil water              !!
   347)         !! soil water converted from m3 -> kg                                      !!
   348)         !!*************************************************************************!!
   349)         call uextcm2tl(soil_sat_energy,soil_sat_water*wdns,soil_sat_heat,tempk,fracliqtotal)
   350)        
   351) 
   352)         !!*************************************************************************!!
   353)         !!  If there is no saturated water or if saturated soil is more than       !!
   354)         !! threshold frozen, skip over soil hydrology.                             !!
   355)         !!*************************************************************************!!
   356)         if(soil_sat_water > 1.e-6 .and. fracliqtotal > FracLiqRunoff) then 
   357)            cgrid%swliq(ipy) = 1.0
   358) 
   359) 
   360)            !!**********************************************************************!!
   361)            !! Zero variables that track the amount of water removed from patches   !!
   362)            !!**********************************************************************!!
   363)            dzadd = 0.0            !total height to add to patches gaining water
   364)            soil_sat_energy = 0.0  !total energy in removed water
   365)            soil_sat_water = 0.0   !total removed water
   366)            area_subtract = 0.0    !total area of sites where water is removed
   367) 
   368)            !!**********************************************************************!!
   369)            !! First we will remove water from patches                              !!
   370)            !!**********************************************************************!!
   371)            do isi=1,cpoly%nsites
   372) 
   373)               csite => cpoly%site(isi)
   374)               slsl=cpoly%lsl(isi)
   375)               nsoil=cpoly%ntext_soil(slsl,isi)
   376) 
   377)               !!*******************************************************************!!
   378)               !! Calculate new site-level equilibrium watertable depth (MOIST_ZI)  !!
   379)               !!*******************************************************************!!
   380)               cpoly%moist_zi(isi) = cgrid%zbar(ipy) + (cpoly%moist_W(isi) - &
   381)                    cgrid%wbar(ipy))/cpoly%moist_f(isi)/MoistRateTuning !TOPMODEL water depth
   382)               zmin = slz(slsl) + soil(nsoil)%soilcp/soil(nsoil)%slmsts*dslz(slsl)
   383)               cpoly%moist_zi(isi)  = max(cpoly%moist_zi(isi),zmin)
   384) 
   385)               !!*******************************************************************!!
   386)               !! Calculate rate constant for lateral redistribution time scale     !!
   387)               !!*******************************************************************!!
   388)               cpoly%moist_tau(isi) = soil(cpoly%ntext_soil(nzg-1,isi))%slmsts / & 
   389)                    (MoistRateTuning*cpoly%moist_f(isi)*cgrid%Te(ipy)* &
   390)                    exp(MoistRateTuning*cpoly%moist_f(isi)*min(0.0,cgrid%zbar(ipy))) & !characteristic redistribution timescale
   391)                    *exp(-cgrid%wbar(ipy))*fracliqtotal) !!added a linear liquid fraction adjustment
   392)               !! added a min(0,zbar) to get sensible behaviour when watertable perched -> flux is overland, not subsurface
   393)               
   394)               !!*******************************************************************!!
   395)               !! Calculate rate constant for baseflow out of the bottom of the soil!!
   396)               !!*******************************************************************!!
   397)               cpoly%baseflow(isi) = soil(cpoly%ntext_soil(nzg-1,isi))%slmsts/ &
   398)                    (MoistRateTuning*cpoly%moist_f(isi)*cpoly%moist_tau(isi))*1000.0 !kg/m2/s
   399)               bf_site = 0.0
   400)               
   401) 
   402)               do ipa=1,csite%npatches
   403) 
   404)                  !!****************************************************************!!
   405)                  !! remove baseflow water and heat                                 !!
   406)                  !!****************************************************************!!
   407)                  bf_patch=cpoly%baseflow(isi)
   408)                  call updateWatertableBaseflow(cpoly,isi,ipa,bf_patch)
   409)                  bf_site = bf_site + bf_patch*csite%area(ipa)
   410) 
   411)                  
   412)                  !!****************************************************************!!
   413)                  !! calculate change in water table depth over time step           !!
   414)                  !!****************************************************************!!
   415)                  dz = (cpoly%moist_zi(isi)-csite%watertable(ipa))*dtlsm/cpoly%moist_tau(isi)
   416)                  csite%moist_dz(ipa) = dz !! assign dz to patch
   417) 
   418)                  !!****************************************************************!!
   419)                  !! if water flows out of the patch, attempt to reduce watertable  !!
   420)                  !! by DZ (meters) and keep track of water and energy removed      !!
   421)                  !!****************************************************************!!
   422)                  if(dz < 0.0) then  
   423)                     call updateWatertableSubtract(cpoly,isi,ipa,dz,sheat,swater)
   424)                     soil_sat_water  = soil_sat_water  + swater  * csite%area(ipa) * cpoly%area(isi)
   425)                     soil_sat_energy = soil_sat_energy + sheat   * csite%area(ipa) * cpoly%area(isi)
   426)                     area_subtract   = area_subtract   +           csite%area(ipa) * cpoly%area(isi)
   427)                  else                    
   428)                     dzadd           = dzadd           + dz      * csite%area(ipa) * cpoly%area(isi)
   429)                  end if
   430) 
   431)               end do
   432) 
   433)               !! Save the site-level mean baseflow
   434)               cpoly%baseflow(isi) = bf_site
   435) 
   436)            end do
   437) 
   438)            !!**********************************************************************!!
   439)            !! calculate mean heat content of water added to patches                !!
   440)            !! (J/m3 water i.e. per volumetic water fraction)                       !!
   441)            !!  assumes redistributed water is well mixed                           !!
   442)            !!**********************************************************************!!
   443)            sheat = soil_sat_energy / max(1.e-15,soil_sat_water)
   444)            cgrid%sheat(ipy) = sheat
   445)            if(soil_sat_water .lt. 1.e-15) soil_sat_water = 0.0
   446) 
   447)            !!**********************************************************************!!
   448)            !! temperature and liquid fraction of water in flux                     !!
   449)            !!**********************************************************************!!
   450)            call uint2tl(sheat/1000.0,tempk,fracliq)
   451)            
   452)            !!**********************************************************************!!
   453)            !! Next, add water to other patches                                     !!
   454)            !!**********************************************************************!!
   455)            do isi=1,cpoly%nsites
   456)               csite => cpoly%site(isi)
   457)               do ipa=1,csite%npatches
   458)                  !change in water table depth
   459)                  if(csite%moist_dz(ipa) > 0.0 .and. dzadd > tiny(1.0)) then
   460) 
   461)                     !!*************************************************************!!
   462)                     !! calculate water (m3/m2) to add to each patch as a proportion!!
   463)                     !! of the water that was removed from the upslope patches to   !!
   464)                     !! ensure conservation of mass                                 !!
   465)                     !!*************************************************************!!
   466)                     dw = csite%moist_dz(ipa)/dzadd*soil_sat_water
   467)                     if(dw /= dw) then
   468)                        call fatal_error('NaN in water table','calcHydroSubsurface','lsm_hyd.f90')
   469)                     endif
   470)                     csite%moist_dz(ipa) = dw/0.435 !! approximation for output
   471)                     
   472)                     !!*************************************************************!!
   473)                     !! Add water and energy to patch                               !!
   474)                     !!*************************************************************!!
   475)                     call updateWatertableAdd(cpoly,isi,ipa,dw,sheat)
   476) 
   477)                  end if
   478)               end do
   479) 
   480)               !!*************************************************************!!
   481)               !! increment polygon-level baseflow                            !!
   482)               !!*************************************************************!! 
   483)               cgrid%baseflow(ipy) = cgrid%baseflow(ipy) + cpoly%area(isi)*cpoly%baseflow(isi)
   484) 
   485)            end do
   486)         else
   487)            !!*************************************************************!!
   488)            !! if soil water too frozen, set flag                          !!
   489)            !!*************************************************************!!
   490)            cgrid%swliq(ipy) = 0.0

   492) 
   493)         !!*************************************************************!!

   495)         !!*************************************************************!!
   496)         cgrid%baseflow(ipy) = cgrid%baseflow(ipy)/area_land
   497) 
   498)      end do  !! loop over polygons

   585) subroutine calcWatertable(cpoly,isi,ipa)
   586)   ! Estimates  water table depth (meters)
   587)   ! For stability, constrains the rate at water table estimates can change over time
   588)   ! based on the variable Moist_dWT
   589)   ! inputs: cpoly   - current polygon
   590)   !         isi     - current site ID
   591)   !         ipa     - current patch ID
   592)   ! outputs: sets patch variables: 
   593)   !         watertable      - depth of saturated zone (m)
   594)   !         soil_sat_energy - energy in saturated zone 
   595)   !         soil_sat_water  - water volume in saturated zone
   596)   !         soil_sat_heat   - heat capacity of saturated zone
   597)   use ed_state_vars, only: polygontype,sitetype
   598)   use hydrology_constants
   599)   use hydrology_coms, only: MoistSatThresh
   600)   use soil_coms, only: soil,slz,dslz
   601)   use grid_coms, only: nzg
   602)   implicit none
   603)   type(sitetype)   , pointer :: csite
   604)   type(polygontype), target  :: cpoly
   605)   integer, intent(in) :: isi,ipa
   606)   
   607)   integer :: k            ! soil layer counter
   608)   integer :: nsoil        ! soil class
   609)   real :: fracw           ! fraction of saturation (m/m)
   610) 
   611)   !!******************************************************************************!!
   612)   !! Compute water table depth starting at the bottom of the lowest soil layer and!!
   613)   !! summing over all saturated levels, defined here as levels that have more     !!
   614)   !! than 95% of the full moisture capacity), but accounting for any deficit      !!
   615)   !! in those levels.  Sum the total soil energy ( soil_energy* dz), water        !!
   616)   !! content (soil_water * dz), and soil heat capacity (slcpd * dz ) in order     !!
   617)   !! to compute a mean temperature of transported water.                          !!
   618)   !!******************************************************************************!!
   619) 
   620)   csite => cpoly%site(isi)
   621) 
   622)   !WTold = csite%watertable(ipa)
   623) 
   624)   csite%watertable(ipa) = slz(cpoly%lsl(isi))   !initialize water table depth to deepest soil layer
   625)   csite%soil_sat_energy(ipa) = 0.0
   626)   csite%soil_sat_water(ipa)  = 0.0
   627)   csite%soil_sat_heat(ipa)   = 0.0
   628) 
   629)   layerloop: do k = cpoly%lsl(isi),nzg
   630)      nsoil = cpoly%ntext_soil(k,isi)  !look up soil type (switched to using SITE level soils [mcd 9/30/08]
   631)      fracw = csite%soil_water(k,ipa) / soil(nsoil)%slmsts !calculate fraction of moisture capacity
   632)      csite%watertable(ipa)      = csite%watertable(ipa)      + fracw *dslz(k)
   633)      csite%soil_sat_energy(ipa) = csite%soil_sat_energy(ipa) + csite%soil_energy(k,ipa)*dslz(k)
   634)      csite%soil_sat_water(ipa)  = csite%soil_sat_water(ipa)  + csite%soil_water(k,ipa)*dslz(k)
   635)      csite%soil_sat_heat(ipa)   = csite%soil_sat_heat(ipa)   + soil(nsoil)%slcpd*dslz(k)
   636)      
   637)      if (fracw < MoistSatThresh) exit layerloop!change from original version to go one layer up
   638)   end do layerloop
   639)   !!store watertable depth integer
   640)   csite%ksat(ipa) = real(max(cpoly%lsl(isi),k-1))
   641) 
   642)   if(k > nzg) then
   643)      !!soil completely saturated, add surface water too
   644)      if(csite%nlev_sfcwater(ipa) >= 1) then
   645)         do k=1,csite%nlev_sfcwater(ipa)
   646)            csite%soil_sat_energy(ipa) = csite%soil_sat_energy(ipa) + csite%sfcwater_energy(k,ipa) * &
   647)                 csite%sfcwater_mass(k,ipa)
   648)         enddo
   649)         k = csite%nlev_sfcwater(ipa)
   650)         csite%soil_sat_water(ipa) = csite%soil_sat_water(ipa) + csite%sfcwater_depth(k,ipa)
   651)         csite%watertable(ipa) = csite%watertable(ipa) + csite%sfcwater_depth(k,ipa)
   652)      endif
   653)   endif
   654) 
   655)    !! check that rate of change in water table depth is reasonable
   656) 
   657)    !! check that not first time and not surface water (allowed to change fast)
   658)    !  if(WTold .lt. 0.0 .and. cpoly%moist_tau(isi) > 0.0) then 
   659)         !! compute allowed rate of change
   660)    !     dWT = abs(Moist_dWT*dtlsm/cpoly%moist_tau(isi))
   661)    !     if( abs(WTold-csite%watertable(ipa)) > dWT) then
   662)            !! need to limit rate of change to dz
   663)    !        if(csite%watertable(ipa) > WTold) then
   664)    !           csite%watertable(ipa) = WTold+dWT
   665)    !        else
   666)    !           csite%watertable(ipa) = WTold-dWT
   667)    !        endif
   668)    !     endif
   669)    !  endif
   670)    return
   671) end subroutine calcWatertable

   682) subroutine updateWatertableAdd(cpoly,isi,ipa,dw,sheat)
   683)   !! Add subsurface water to a patch
   684)   !! inputs: cpoly - polygon
   685)   !!         isi   - site ID
   686)   !!         ipa   - patch ID
   687)   !!         dw    - change in water content (m3/m2)
   688)   !!         sheat - heat content of added water (J/m3)
   689)   use ed_state_vars, only: sitetype,polygontype
   690)   use soil_coms, only: soil,dslz,dslzi
   691)   use grid_coms, only: nzg
   692)   use therm_lib, only: uextcm2tl
   693)   use consts_coms, only: wdns
   694)   implicit none
   695)   type(polygontype) , target        :: cpoly
   696)   integer           , intent(in)    :: isi,ipa
   697)   real              , intent(inout) :: dw
   698)   real              , intent(in)    :: sheat       !heat content of water to add (J/m3)
   699) 
   700)   type(sitetype)    , pointer       :: csite
   701)   real                              :: tempk       !water temperature (K)
   702)   real                              :: fracliq     !water fraction liquid (proportion)
   703)   integer                           :: i,k         !counters
   704)   integer                           :: nsoil       !soil type
   705)   real                              :: wcap        !available capacity for adding moisture (m)
   706)   real                              :: fracw       !proportion of saturation for a soil layer
   707)   real                              :: dw_layer    !volumetric water content added to a layer (m3/m3)
   708)   logical                           :: done
   709) 
   710)   !!******************************************************************************!!
   711)   !! Return immediately if added water will be lost in round-off                  !!
   712)   !!******************************************************************************!!
   713)   if(dw .lt. tiny(1.0)) return
   714) 
   715)   done = .false.
   716)   csite => cpoly%site(isi)
   717) 
   718)   !!******************************************************************************!!
   719)   !! loop from lowest soil layer to the surface, adding moisture and energy       !!
   720)   !!******************************************************************************!!
   721)    layerloop: do k = cpoly%lsl(isi),nzg
   722)       nsoil = cpoly%ntext_soil(k,isi)  !look up soil type
   723)       fracw = csite%soil_water(k,ipa) / soil(nsoil)%slmsts  !calculate fraction of moisture capacity
   724)       if(fracw < 1.0) then
   725) 
   726)          wcap = soil(nsoil)%slmsts-csite%soil_water(k,ipa)*dslz(k) !!m3/m2
   727)          if(dw > wcap) then
   728) 
   729)             !!********************************************************************!!
   730)             !! layer can be saturated, add as much as we can and subtract from dw !!                   
   731)             !!********************************************************************!!
   732)             dw_layer = wcap
   733)             csite%soil_water(k,ipa) = soil(nsoil)%slmsts
   734)             dw = dw - wcap
   735) 
   736)          else
   737) 
   738)             !!********************************************************************!!
   739)             !! layer can accept all of the dw, add everything and exit the loop   !!                   
   740)             !!********************************************************************!!
   741)             dw_layer = dw
   742)             csite%soil_water(k,ipa) = csite%soil_water(k,ipa)+dw*dslzi(k)
   743)             dw = 0.0
   744)             done = .true.
   745) 
   746)          endif
   747) 
   748)          !!************************************************************************!!
   749)          !! update layer's soil heat based on the amount of water added            !!
   750)          !!************************************************************************!!
   751)          csite%soil_energy(k,ipa)  = csite%soil_energy(k,ipa) + dw_layer*sheat
   752)          if(csite%soil_energy(k,ipa) /= csite%soil_energy(k,ipa)) then
   753)             print*,"dw_layer",dw_layer
   754)             print*,"sheat",sheat
   755)             print*,"k= ",k," ipa= ",ipa
   756)             print*,"wcap",wcap," dw", dw
   757)             call fatal_error('Failed soil_energy sanity check in lsm_hyd' &
   758)                  ,'updateWatertableAdd','lsm_hyd.f90')
   759)          end if
   760) 
   761)          !!***********************************************************************!!
   762)          !!  update soil temperature and liquid fraction                          !!
   763)          !!***********************************************************************!!
   764)          call uextcm2tl(csite%soil_energy(k,ipa),csite%soil_water(k,ipa)*wdns            &
   765)                    ,soil(nsoil)%slcpd,tempk,fracliq)
   766)          csite%soil_tempk(k,ipa) = tempk
   767)          csite%soil_fracliq(k,ipa) = fracliq
   768) 
   769)          if(done) exit layerloop
   770)       end if
   771)    end do layerloop
   772) 
   773)    if(.not. done) then
   774)       !!***************************************************************************!!
   775)       !! reached top layer and still more water add,  put in surface water         !!
   776)       !! note: sfcwater_mass units = kg/m2                                         !!
   777)       !!       sfcwater_energy = J/(kg H2O)                                        !!
   778)       !!***************************************************************************!!
   779) 
   780)       if(csite%nlev_sfcwater(ipa) == 0 .or. csite%sfcwater_mass(1,ipa) < tiny(1.0)) then
   781)          !!currently no surface water, create as liquid water
   782)          csite%sfcwater_depth(1,ipa) = dw
   783)          csite%sfcwater_mass(1,ipa) = dw*wdns
   784)          csite%sfcwater_energy(1,ipa) = 0.001 * sheat
   785)          csite%nlev_sfcwater(ipa) = 1
   786)       else
   787)          !!fill from bottom 
   788)          !!assumes surface water is liquid
   789)          !!need to work on case where may be snow or snow/liquid mix
   790) 
   791)          ! convert J/kg to J/m2
   792)          csite%sfcwater_energy(1,ipa) = csite%sfcwater_energy(1,ipa) * csite%sfcwater_mass(1,ipa)
   793)          ! still in J/m2
   794)          csite%sfcwater_energy(1,ipa) = csite%sfcwater_energy(1,ipa) + sheat*dw
   795)          ! get new mass
   796)          csite%sfcwater_mass(1,ipa) = csite%sfcwater_mass(1,ipa) + dw*1000.0
   797)          ! convert back to J/kg
   798)          if(csite%sfcwater_mass(1,ipa) > 1.0e-10)then
   799)             csite%sfcwater_energy(1,ipa) = csite%sfcwater_energy(1,ipa) / csite%sfcwater_mass(1,ipa)
   800)          else
   801)             csite%sfcwater_energy(1,ipa) = 0.0
   802)          end if
   803)          if(csite%sfcwater_energy(1,ipa) /= csite%sfcwater_energy(1,ipa))then
   804)             call fatal_error('Failed sfcwater_energy sanity check in lsm_hyd' &
   805)                  ,'updateWaterTableAdd','lsm_hyd.f90')
   806)          end if
   807)          do i= 1,csite%nlev_sfcwater(ipa)
   808)             csite%sfcwater_depth(i,ipa) = csite%sfcwater_depth(i,ipa) + dw
   809)          end do
   810)       end if
   811)    end if
   812) 
   813)    return
   814) end subroutine updateWatertableAdd

   825) subroutine updateWatertableSubtract(cpoly,isi,ipa,dz,sheat,swater)
   826) !! Remove subsurface water from a patch
   827) !! inputs: 
   828) !!         cpoly  - current polygon
   829) !!         isi    - current site ID
   830) !!         ipa    - current patch ID
   831) !!         dz     - change in water table height (meters, negative)
   832) !! outputs:
   833) !!         sheat  - heat content of removed water (J/m2?)
   834) !!         swater - removed water (m3/m2)
   835) 
   836)    !!need to subtract water from patch
   837)    use hydrology_constants
   838)    use hydrology_coms, only: MoistSatThresh
   839)    use ed_state_vars,  only: polygontype, sitetype 
   840)    use consts_coms, only : wdns
   841)    use soil_coms, only: soil,dslz,dslzi
   842)    use grid_coms, only: nzg
   843)    use therm_lib, only : uextcm2tl,tl2uint
   844) 
   845)    implicit none
   846)    type(polygontype), target      :: cpoly
   847)    integer          , intent(in) :: ipa,isi
   848)    real             , intent(in)   :: dz      ! change in water table height (always negative)
   849)    real             , intent(out) :: sheat
   850)    real             , intent(out) :: swater
   851) 
   852)    type(sitetype)   , pointer    :: csite
   853) 
   854)    integer :: k       ! counters
   855)    integer :: nsoil   ! soil type
   856)    real    :: dzl     ! local copy of dz, safe to change 
   857)    real    :: wcap    ! available capacity for adding moisture (m)
   858)    real    :: fracw   ! proportion of saturation for a soil layer
   859)    real    :: dh      ! change in heat (J/m3)
   860)    real    :: dw      ! change in water (m3/m3)
   861)    real    :: fracliq ! water fraction liquid (proportion)
   862)    real    :: tempk   ! water temperature (K)
   863)    real    :: wtmp    ! water content of layer before adjustment
   864)    logical :: done    ! status variable to end loop early
   865) 
   866)    if( dz > 0.0) then
   867)       write(unit=*,fmt=*) 'dz > 0 in updateWatertableSubtract',dz
   868)       call fatal_error('invalid dz!','updateWatertableSubtract','lsm_hyd.f90') 
   869)    end if
   870) 
   871)    csite => cpoly%site(isi)
   872) 
   873)    done = .false.
   874)    dzl = dz        !make local copy
   875)    sheat = 0.0
   876)    swater = 0.0
   877) 
   878)    !!start by finding top of saturated layer
   879)    k = int(csite%ksat(ipa))
   880) 
   881)    nsoil = cpoly%ntext_soil(k,isi)  !look up soil type
   882)    fracw = csite%soil_water(k,ipa) / soil(nsoil)%slmsts  !calculate fraction of moisture capacity
   883)    if(fracw > MoistSatThresh) then  !if layer above threshold, move up one
   884)       k = k+1
   885)       nsoil = cpoly%ntext_soil(k,isi)
   886)       fracw = csite%soil_water(k,ipa)
   887)    end if
   888) 
   889)    if(k > nzg) then
   890)       !! soil completely saturated, reset to top layer
   891)       k = nzg
   892)       nsoil = cpoly%ntext_soil(k,isi)  !look up soil type
   893)       fracw = csite%soil_water(k,ipa) / soil(nsoil)%slmsts  !calculate fraction of moisture capacity
   894)    end if
   895) 
   896)    !!work down from saturation layer, removing water & heat
   897)    do while (k >= cpoly%lsl(isi))
   898)       call uextcm2tl(csite%soil_energy(k,ipa),csite%soil_water(k,ipa)*1.e3,soil(nsoil)%slcpd,tempk,fracliq)
   899)       !capacity for layer to loose moisture (unit = meters)
   900)       wcap = (fracw - soil(nsoil)%soilcp/soil(nsoil)%slmsts) * dslz(k)*fracliq 
   901) 
   902)       if (wcap > tiny(1.0)) then 
   903)          wtmp = real(csite%soil_water(k,ipa))
   904)          !!      if(wcap > dzl) then 
   905)          if(-dzl > wcap) then 
   906)             !!empty soil layer completely 
   907)             dw = -1.0*(csite%soil_water(k,ipa) - soil(nsoil)%soilcp)
   908)             dzl = dzl + wcap
   909)          else
   910)             !!reduce water by dz
   911)             dw = dzl*soil(nsoil)%slmsts*dslzi(k)
   912)             dzl = 0.0
   913)             done = .true.
   914)          end if
   915)          
   916)          !!update soil water
   917)          csite%soil_water(k,ipa) = csite%soil_water(k,ipa) + dw
   918)          !! error message for positive dw or soil below capacity
   919)          if(dw .gt. 1.0E-9 .or. (csite%soil_water(k,ipa)+1.0e-8) < soil(nsoil)%soilcp) then 
   920)             print*," "
   921)             print*,"dw = ",dw," in updateWatertableSubtract"
   922)             print*,dzl,wcap,k,csite%watertable(ipa),done,fracw,nsoil
   923)             print*,wtmp,csite%soil_water(k,ipa),soil(nsoil)%soilcp,soil(nsoil)%slmsts
   924)             call fatal_error ('Bad dw!','updateWatertableSubtract','lsm_hyd.f90')
   925)          endif
   926)          
   927)          !!update soil heat
   928)          dh = dw * wdns * tl2uint(tempk,1.0)
   929)          csite%soil_energy(k,ipa) = csite%soil_energy(k,ipa) + dh
   930)          sheat = sheat - dh*dslz(k)   !cumulative sum as return value
   931)          swater = swater - dw*dslz(k)
   932)          if(csite%soil_energy(k,ipa) /= csite%soil_energy(k,ipa)) then
   933)             call fatal_error('Failed soil_energy sanity check in lsm_hyd' &
   934)                  ,'updateWatertableSubtract','lsm_hyd.f90')
   935)          end if
   936)          
   937)          
   938)          !!update soil temperature
   939)          call uextcm2tl(csite%soil_energy(k,ipa),csite%soil_water(k,ipa)*wdns               &
   940)               ,soil(nsoil)%slcpd,tempk,fracliq)
   941)          csite%soil_tempk(k,ipa) = tempk
   942)          csite%soil_fracliq(k,ipa) = fracliq
   943)       end if
   944) 
   945)       !!iterate
   946)       if(done .or. k == cpoly%lsl(isi)) then
   947)          k = 0
   948)       else
   949)          k = k-1
   950)          nsoil = cpoly%ntext_soil(k,isi)  !look up soil type
   951)          fracw = csite%soil_water(k,ipa) / soil(nsoil)%slmsts
   952)       endif
   953)    enddo

   966) subroutine updateWatertableBaseflow(cpoly,isi,ipa,baseflow)
   967)    use ed_state_vars, only: polygontype, sitetype
   968)    use soil_coms, only: soil,dslz,dslzi,slcons1
   969)    use ed_misc_coms, only: dtlsm
   970)    use consts_coms, only: wdns
   971)    use therm_lib, only : uextcm2tl,tl2uint
   972)    implicit none
   973) 
   974)    real, parameter :: freezeCoef = 7.0        !! should probably move to the com
   975)    type(polygontype) , target        :: cpoly
   976)    integer           , intent(in)    :: isi, ipa
   977)    real              , intent(inout) :: baseflow !read in as mm/s
   978)   
   979)    type(sitetype)    , pointer       :: csite
   980)    real                              :: bf   !baseflow in water content (meters/timestep)
   981)    real                              :: potn_fd,wflux_fd ! potential & flux for free-drainage
   982)    integer                           :: nsoil,slsl
   983)    real                              :: tempk, fracliq, freezeCor
   984) 
   985)    
   986)    csite => cpoly%site(isi)
   987)    slsl  = cpoly%lsl(isi)
   988)    nsoil = cpoly%ntext_soil(slsl,isi)
   989)    !! determine freeze
   990)    call uextcm2tl(csite%soil_energy(slsl,ipa),csite%soil_water(slsl,ipa)*1.e3,soil(nsoil)%slcpd,tempk,fracliq)
   991)    freezeCor = fracliq
   992)    if(freezeCor .lt. 1.0) freezeCor = 10.0**(-freezeCoef*(1.0-freezeCor))
   993)    
   994)    !! calc max free-drainage as cap to baseflow
   995)    !! assumes layer below is permenantly at minimal water capacity
   996)    slsl  = cpoly%lsl(isi)
   997) !   nsoil = cpoly%ntext_soil(slsl,isi)
   998)    potn_fd = -dslzi(slsl)+soil(nsoil)%slpots* &
   999)         ((soil(nsoil)%slmsts/soil(nsoil)%soilcp)**soil(nsoil)%slbs - &
  1000)         (soil(nsoil)%slmsts/csite%soil_water(slsl,ipa))**soil(nsoil)%slbs)
  1001)    wflux_fd = slcons1(slsl,nsoil)  &
  1002)                  * (csite%soil_water(1,ipa)/ soil(nsoil)%slmsts)**(2. * soil(nsoil)%slbs + 3.)  &
  1003)                  * potn_fd * freezeCor
  1004)    bf = min(-wflux_fd,baseflow)*dtlsm*0.001
  1005) 
  1006)    if(bf < 0.0) then
  1007)       write (unit=*,fmt=*) "bf in wrong direction",bf
  1008)       call fatal_error('bad bf!','updateWatertableBaseflow','lsm_hyd.f90')
  1009)    end if
  1010) 
  1011)    !! first, remove baseflow water and heat
  1012)    csite%soil_water(slsl,ipa) = csite%soil_water(slsl,ipa)-bf*dslzi(slsl)
  1013)    if(csite%soil_water(slsl,ipa) < soil(nsoil)%soilcp) then
  1014)       write(unit=*,fmt=*) 'Bottom dry:',csite%soil_water(1,ipa),bf,soil(nsoil)%soilcp
  1015)       bf = bf + (csite%soil_water(slsl,ipa)-soil(nsoil)%soilcp)*dslz(slsl)
  1016)       csite%soil_water(slsl,ipa) = soil(nsoil)%soilcp
  1017)    end if
  1018)    csite%soil_energy(slsl,ipa) = csite%soil_energy(slsl,ipa) &
  1019)                                - bf*wdns*tl2uint(csite%soil_tempk(slsl,ipa),1.0)
  1020) 
  1021)    if(csite%soil_energy(slsl,ipa) /= csite%soil_energy(slsl,ipa)) then
  1022)                     call fatal_error('Failed soil_energy sanity check in lsm_hyd' &
  1023)                                     ,'updateWatertableBaseflow','lsm_hyd.f90')
  1024)                  end if
  1025) 
  1026)    baseflow = bf*1000.0/dtlsm !! reassign for return (m/step->mm/sec)
  1027)    return

  1196) subroutine calcHydroSurface()
  1197)   !! Main function for calculating surface runoff
  1198)   !! Calculates overland flow (runoff rate kg/s/m2)
  1199)   !! Runoff direction is based on site adjacency matrix
  1200)   !! Runoff rate is based on Manning's equation and is a function of:
  1201)   !!    water depth, slope, surface roughness 
  1202)   !! surface roughness is calculated in updateHydroParms
  1203)   !! Inputs:
  1204)   !! Outputs:
  1205)   !!         sets cgrid%runoff, cgrid%runoff_heat, cpoly%runoff, cpoly%runoff_heat
  1206)   use ed_state_vars, only : edgrid_g,edtype,polygontype,sitetype
  1207)   use hydrology_constants
  1208)   use hydrology_coms, only: useRUNOFF,FracLiqRunoff,runoff_Vmax
  1209)   use grid_coms, only: ngrids, nzg
  1210)   use ed_misc_coms, only: dtlsm
  1211)   use grid_coms,only:ngrids
  1212)   use consts_coms, only : wdns,t3ple,uiicet3,uiliqt3
  1213)   use soil_coms, only: water_stab_thresh
  1214)   use therm_lib, only: uint2tl,tl2uint
  1215)   implicit none
  1216)   type(edtype)       , pointer :: cgrid
  1217)   type(polygontype)  , pointer :: cpoly
  1218)   type(sitetype)     , pointer :: csite
  1219)   
  1220)   integer :: igr,ipy,isi,ines,ipa  ! Grid, polygon, site, neighbouring site and patch counters
  1221)   integer :: top_surf_water        ! index of top surface water layer
  1222)   integer :: curr_site             ! site counter
  1223)   integer :: i
  1224)  ! integer :: nsites            ! number of sites in the polygon
  1225)   real, allocatable :: qwout(:) ! flux of water out of a site (kg/s/m2?)
  1226)   real, allocatable :: qhout(:) ! flux of heat out of a site ( units /s/m2? ******************)
  1227)   real :: tempk                 ! water temperature (K)
  1228)   real :: fracliq               ! water fraction liquid (proportion)
  1229)   real :: surf_water_depth      ! depth of liquid surface water (m)
  1230)   real :: surf_water_heat       ! energy of liquid surface water (J/m3 ??)
  1231)   real :: swd_i                 ! surface water depth in layer i (m)
  1232)   real :: flow_vel,flow_denom   ! overland flow velocity (m/s)
  1233)   real :: patch_water_out       ! patch water flux out (kg/m2/s)
  1234)   real :: patch_heat_out        ! patch heat flux out (J/m2/s??? ***************)
  1235)   real :: site_water_in         ! site water flux in (kg/m2/s)
  1236)   real :: site_heat_in          ! site heat flux in (J/m2/s)
  1237) 
  1238) 
  1239)   !!! Mike,what does this eleven stand for?
  1240)   real,dimension(11) :: hts     ! heights of surface water layers (m)
  1241) 
  1242) 
  1243) 
  1244) 
  1245)   if(useRUNOFF == 0) return
  1246) 
  1247)   !! Loop over grids and polygons
  1248)   do igr = 1,ngrids
  1249)      cgrid => edgrid_g(igr)
  1250)      do ipy=1,cgrid%npolygons
  1251)         cpoly => cgrid%polygon(ipy)
  1252)         !! FIRST PASS, CALCULATE GROSS RUNOFF
  1253) 
  1254)         !! initialize variables
  1255)         allocate(qwout(cpoly%nsites))
  1256)         allocate(qhout(cpoly%nsites))
  1257)         qwout = 0.0
  1258)         qhout = 0.0
  1259) 
  1260)         do isi=1,cpoly%nsites
  1261)            csite => cpoly%site(isi)
  1262) 
  1263)            !site specific parms
  1264)            do ipa=1,csite%npatches
  1265)               !first check that there IS surface water
  1266)               top_surf_water = csite%nlev_sfcwater(ipa)
  1267)               if(top_surf_water > 0) then
  1268)                  !second check that some is liquid
  1269)                  call uint2tl(csite%sfcwater_energy(top_surf_water,ipa),tempk,fracliq)
  1270)                  if(fracliq > FracLiqRunoff) then
  1271)                     !! calculate water depth
  1272)                     swd_i = csite%sfcwater_mass(top_surf_water,ipa)*0.001*fracliq !convert liquid fraction from kg to meters
  1273)                     surf_water_depth = swd_i
  1274)                     surf_water_heat = swd_i*wdns * tl2uint(tempk,1.0)
  1275)                     do i=(top_surf_water-1),1,-1
  1276)                        call uint2tl(csite%sfcwater_energy(i,ipa),tempk,fracliq)
  1277)                        swd_i = csite%sfcwater_mass(top_surf_water,ipa)*0.001*fracliq
  1278)                        surf_water_depth = surf_water_depth + swd_i
  1279)                        surf_water_heat = surf_water_heat + swd_i*wdns*tl2uint(tempk,1.0)
  1280)                     end do
  1281)                     !!sanity check
  1282)                     if(surf_water_depth > 1.0) then  !!if there's more than 1 m of standing water
  1283)                       write (unit=*,fmt=*) 'FLOOD WARNING !!!!!!!!!!'
  1284)                       write(unit=*,fmt='(4(a,1x))') 'Level','Sfc.Water','SfcWater.Mass','Sfc.Water.Energy'
  1285)                       do i=1,top_surf_water
  1286)                          write(unit=*,fmt='(2(i9,1x),2(es14.7,1x))') &
  1287)                             i,csite%nlev_sfcwater(ipa),csite%sfcwater_mass(i,ipa),csite%sfcwater_energy(i,ipa)
  1288)                       end do
  1289)                       call fatal_error('I''m drowning!!!','calcHydroSurface','lsm_hydro.f90')
  1290)                     endif
  1291) 
  1292)                     !! calculate flow velocity (m/s)
  1293)                     flow_denom = 1.0+csite%runoff_a(2,ipa)*surf_water_depth**(4/3) & 
  1294)                          - csite%runoff_a(3,ipa)*surf_water_depth**(7/3)
  1295)                     if(flow_denom > 1.0) then
  1296)                        flow_vel = surf_water_depth**(2/3)*csite%runoff_a(1,ipa)/sqrt(flow_denom)
  1297)                     else
  1298)                        flow_vel = surf_water_depth**(2/3)*csite%runoff_a(1,ipa) !! asymptotic vel w/o vegetation
  1299)                     endif
  1300)                     flow_vel = min(flow_vel,runoff_vmax) !! clamp runoff velocity to maximum sensible value
  1301) 
  1302)                     !!patch level flux out (kg/m2/s)
  1303)                     patch_water_out = flow_vel*surf_water_depth*1000.0
  1304)                     patch_heat_out  = flow_vel*surf_water_heat  !energy/m2/s
  1305) 
  1306)                     !!update vars
  1307)                     csite%runoff (ipa) = patch_water_out
  1308)                     csite%qrunoff(ipa) = patch_heat_out
  1309)                     qwout(isi) = qwout(isi) + patch_water_out*csite%area(ipa)
  1310)                     qhout(isi) = qhout(isi) + patch_heat_out*csite%area(ipa)
  1311)                  endif
  1312)               endif
  1313)            end do
  1314)            !! for output
  1315)            qw4out(igr,ipy,isi) = qwout(isi)
  1316)            qh4out(igr,ipy,isi) = qhout(isi)
  1317)         end do
  1318)         
  1319) 
  1320) 
  1321)         !! SECOND PASS: CALCULATE RUN-IN
  1322)         do isi=1,cpoly%nsites
  1323)            !!calc NET site-level run-in rate (adjusts patch net water, prev calc was gross patch runoff)
  1324)            site_water_in = 0.0
  1325)            site_heat_in = 0.0
  1326) 
  1327)            do ines=1,isi-1
  1328)               site_water_in = site_water_in + qwout(ines)*cgrid%site_adjacency(ines,isi,ipy)
  1329)               site_heat_in  = site_heat_in  + qhout(ines)*cgrid%site_adjacency(ines,isi,ipy)
  1330)            end do
  1331)            do ines=isi+1,cpoly%nsites
  1332)               site_water_in = site_water_in + qwout(ines)*cgrid%site_adjacency(ines,isi,ipy)
  1333)               site_heat_in  = site_heat_in  + qhout(ines)*cgrid%site_adjacency(ines,isi,ipy)
  1334)            end do
  1335) 
  1336)            !!calc NET site-level run-off rate (just for output)
  1337)            cpoly%runoff (isi) = 0.0
  1338)            cpoly%qrunoff(isi) = 0.0
  1339)            do ines=1,isi-1
  1340)                  cpoly%runoff (isi) = cpoly%runoff (isi) + qwout(isi)*cgrid%site_adjacency(isi,ines,ipy)
  1341)                  cpoly%qrunoff(isi) = cpoly%qrunoff(isi) + qhout(isi)*cgrid%site_adjacency(isi,ines,ipy)
  1342)            end do
  1343)            do ines=isi+1,cpoly%nsites+1
  1344)                  cpoly%runoff (isi) = cpoly%runoff (isi) + qwout(isi)*cgrid%site_adjacency(isi,ines,ipy)
  1345)                  cpoly%qrunoff(isi) = cpoly%qrunoff(isi) + qhout(isi)*cgrid%site_adjacency(isi,ines,ipy)
  1346)            end do
  1347) 
  1348)            !! calc patch layer run in
  1349)            do ipa=1,csite%npatches
  1350)               !! adjust runoff rates for run-on
  1351)               !! negative values indicate a net gain of water from overland flow
  1352)               csite%runoff (ipa) = (1.-cgrid%site_adjacency(isi,isi,ipy))*csite%runoff (ipa) - site_water_in 
  1353)               csite%qrunoff(ipa) = (1.-cgrid%site_adjacency(isi,isi,ipy))*csite%qrunoff(ipa) - site_heat_in
  1354)               ! changed runoff code to assume recycled runoff occurrs at a patch level, not a intra-site level
  1355)               ! less realistic, but testing whether can avoid putting intersite runoff in the integrator
  1356) 
  1357)               !!compute contribution of each surface water layer -> hts(i)/surf_water_depth
  1358)               surf_water_depth = 0.0
  1359)               top_surf_water = csite%nlev_sfcwater(ipa)
  1360)               do i=1,top_surf_water
  1361)                  call uint2tl(csite%sfcwater_energy(i,ipa),tempk,fracliq)
  1362)                  hts(i) = csite%sfcwater_mass(i,ipa)*0.001*fracliq
  1363)                  surf_water_depth = surf_water_depth + hts(i)
