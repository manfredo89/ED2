     1) module rk4_misc
     2)   contains
     3) 
     4) !==========================================================================================!
     5) !==========================================================================================!
     6) !    This subroutine copies that variables that are integrated by the Runge-Kutta solver   !
     7) ! to a buffer structure.                                                                   !
     8) !------------------------------------------------------------------------------------------!
     9) subroutine copy_patch_init(sourcesite,ipa,targetp,vels)
    10)    use ed_state_vars         , only : sitetype               & ! structure
    11)                                     , patchtype              ! ! structure
    12)    use grid_coms             , only : nzg                    & ! intent(in)
    13)                                     , nzs                    ! ! intent(in) 
    14)    use ed_misc_coms          , only : fast_diagnostics       ! ! intent(in)
    15)    use consts_coms           , only : ep8                    & ! intent(in)
    16)                                     , epim18                 & ! intent(in)
    17)                                     , rdry8                  & ! intent(in)
    18)                                     , rdryi8                 & ! intent(in)
    19)                                     , cpdry8                 & ! intent(in)
    20)                                     , cph2o8                 ! ! intent(in)
    21)    use rk4_coms              , only : rk4patchtype           & ! structure
    22)                                     , rk4site                & ! structure
    23)                                     , rk4water_stab_thresh   & ! intent(in)
    24)                                     , checkbudget            & ! intent(in)
    25)                                     , print_detailed         & ! intent(in)
    26)                                     , rk4aux                 & 
    27)                                     , find_derived_thbounds  & ! sub-routine
    28)                                     , reset_rk4_fluxes       ! ! sub-routine
    29)    use ed_max_dims           , only : n_pft                  ! ! intent(in)
    30)    use therm_lib8            , only : uextcm2tl8             & ! subroutine
    31)                                     , cmtl2uext8             & ! function
    32)                                     , thetaeiv8              & ! function
    33)                                     , idealdenssh8           & ! function
    34)                                     , rehuil8                & ! function
    35)                                     , qslif8                 & ! function
    36)                                     , reducedpress8          & ! function
    37)                                     , press2exner8           & ! function
    38)                                     , extheta2temp8          & ! function
    39)                                     , tq2enthalpy8           ! ! function
    40)    use ed_therm_lib          , only : ed_grndvap8            ! ! subroutine
    41)    use canopy_air_coms       , only : ubmin8
    42)    use canopy_struct_dynamics, only : canopy_turbulence8     ! ! subroutine
    43)    !$ use omp_lib
    44)    implicit none
    45) 
    46)    !----- Arguments -----------------------------------------------------------------------!
    47)    type(rk4patchtype)    , target     :: targetp
    48)    type(sitetype)        , target     :: sourcesite
    49)    integer               , intent(in) :: ipa
    50)    real                  , intent(in) :: vels
    51)    !----- Local variables -----------------------------------------------------------------!
    52)    type(patchtype)       , pointer    :: cpatch
    53)    real(kind=8)                       :: atm_tmp_zcan
    54)    integer                            :: ico
    55)    integer                            :: ipft
    56)    integer                            :: k
    57)    integer                            :: ksn
    58)    integer                            :: ibuff
    59)    !---------------------------------------------------------------------------------------!
    60) 
    61)    ibuff = 1
    62)    !$ ibuff = OMP_get_thread_num()+1
    63) 
    64)    !---- Alias for the current patch. -----------------------------------------------------!
    65)    cpatch => sourcesite%patch(ipa)
    66)    !---------------------------------------------------------------------------------------!
    67) 
    68) 
    69) 
    70)    !---------------------------------------------------------------------------------------!
    71)    !     Between time steps the pressure may change because of change in atmospheric       !
    72)    ! pressure, which means that temperature is not conserved.  Potential temperature and   !
    73)    ! equivalent potential temperature, on the other hand, are conserved because there is   !
    74)    ! no heat flux between time steps.  So we use these instead to start all other vari-    !
    75)    ! ables.                                                                                !
    76)    !---------------------------------------------------------------------------------------!
    77)    !----- Update thermo variables that are conserved between steps. -----------------------!
    78)    targetp%can_theta    = dble(sourcesite%can_theta(ipa))
    79)    targetp%can_shv      = dble(sourcesite%can_shv(ipa))
    80)    targetp%can_co2      = dble(sourcesite%can_co2(ipa))
    81)    targetp%can_depth    = dble(sourcesite%can_depth(ipa))
    82)    !---------------------------------------------------------------------------------------!
    83) 
    84)    !----- Update the vegetation properties used for roughness. ----------------------------!
    85)    targetp%veg_height   = dble(sourcesite%veg_height  (ipa))
    86)    targetp%veg_displace = dble(sourcesite%veg_displace(ipa))
    87)    targetp%veg_rough    = dble(sourcesite%veg_rough   (ipa))
    88)    !---------------------------------------------------------------------------------------!
    89) 
    90)    !---------------------------------------------------------------------------------------!
    91)    !      Update the canopy pressure and Exner function.                                   !
    92)    !---------------------------------------------------------------------------------------!
    93)    targetp%can_prss  = reducedpress8(rk4site%atm_prss,rk4site%atm_theta,rk4site%atm_shv    &
    94)                                     ,rk4site%geoht,targetp%can_theta,targetp%can_shv       &
    95)                                     ,targetp%can_depth)
    96)    targetp%can_exner = press2exner8 (targetp%can_prss)
    97)    !---------------------------------------------------------------------------------------!
    98) 
    99) 
   100)    !---------------------------------------------------------------------------------------!
   101)    !     Find the pressure and Exner functions at the canopy depth, find the temperature   !
   102)    ! of the air above canopy at the canopy depth, and the specific enthalpy at that level. !
   103)    !---------------------------------------------------------------------------------------!
   104)    
   105) 
   106)    atm_tmp_zcan         = extheta2temp8(targetp%can_exner,rk4site%atm_theta)
   107)    targetp%atm_enthalpy = tq2enthalpy8 (atm_tmp_zcan,rk4site%atm_shv,.true.)
   108) 
   109)    !----- Get velocity for aerodynamic resistance. ----------------------------------------!
   110)    targetp%vels  = max(ubmin8,dble(vels))
   111) 
   112)    !---------------------------------------------------------------------------------------!
   113)    !      Initialise canopy air temperature and enthalpy.  Enthalpy is the actual          !
   114)    ! prognostic variable within one time step.                                             !
   115)    !---------------------------------------------------------------------------------------!
   116)    targetp%can_temp     = extheta2temp8(targetp%can_exner,targetp%can_theta)
   117)    targetp%can_enthalpy = tq2enthalpy8(targetp%can_temp,targetp%can_shv,.true.)
   118)    !---------------------------------------------------------------------------------------!
   119) 
   120) 
   121)    !---------------------------------------------------------------------------------------!
   122)    !      Set up the canopy air space heat capacity at constant pressure.                  !
   123)    !---------------------------------------------------------------------------------------!
   124)    targetp%can_cp = (1.d0 - targetp%can_shv) * cpdry8 + targetp%can_shv * cph2o8
   125)    !---------------------------------------------------------------------------------------!
   126) 
   127) 
   128)    !---------------------------------------------------------------------------------------!
   129)    !     Update density, relative humidity, and the saturation specific humidity.          !
   130)    !---------------------------------------------------------------------------------------!
   131)    targetp%can_rhos     = idealdenssh8(targetp%can_prss,targetp%can_temp,targetp%can_shv)
   132)    targetp%can_rhv      = rehuil8(targetp%can_prss,targetp%can_temp,targetp%can_shv,.true.)
   133)    targetp%can_ssh      = qslif8(targetp%can_prss,targetp%can_temp)
   134)    !---------------------------------------------------------------------------------------!
   135) 
   136) 
   137) 
   138)    !----- Find the lower and upper bounds for the derived properties. ---------------------!
   139)    call find_derived_thbounds(targetp%can_theta,targetp%can_temp,targetp%can_shv,          &
   140)                               targetp%can_prss ,targetp%can_depth)
   141)    !---------------------------------------------------------------------------------------!
   142) 
   143) 
   144) 
   145)    !----- Impose a non-sense number for flag_wflxgc. --------------------------------------!
   146)    targetp%flag_wflxgc  = -1
   147)    !---------------------------------------------------------------------------------------!
   148) 
   149) 
   150) 
   151)    !---------------------------------------------------------------------------------------!
   152)    !     Soil properties.                                                                  !
   153)    !---------------------------------------------------------------------------------------!
   154)    do k = rk4site%lsl, nzg
   155)       !------------------------------------------------------------------------------------!
   156)       !     Soil water may be slightly off-bounds.  This may happen when the soil moisture !
   157)       ! is exactly at the bounds, but then it is copied to single precision and then back  !
   158)       ! to double precision.  Therefore at this time only we must ensure that we bound it, !
   159)       ! otherwise the model will crash due to the round-off error.                         !            
   160)       !------------------------------------------------------------------------------------!
   161)       targetp%soil_water  (k) = min( rk4aux(ibuff)%rk4max_soil_water(k)                                  &
   162)                                    , max( rk4aux(ibuff)%rk4min_soil_water(k)                             &
   163)                                    , dble(sourcesite%soil_water(k,ipa)) ) )
   164)       targetp%soil_energy (k) = dble(sourcesite%soil_energy (k,ipa))
   165)       targetp%soil_mstpot (k) = dble(sourcesite%soil_mstpot (k,ipa))
   166)       targetp%soil_tempk  (k) = dble(sourcesite%soil_tempk  (k,ipa))
   167)       targetp%soil_fracliq(k) = dble(sourcesite%soil_fracliq(k,ipa))
   168)    end do
   169)    !---------------------------------------------------------------------------------------!
   170) 
   171) 
   172)    !---------------------------------------------------------------------------------------!
   173)    !     Copy the surface water information.  The only non-trivial one is the energy,      !
   174)    ! which is saved as J/kg outside the integration, but must be converted to J/m� because !
   175)    ! this linearises the differential equations and make the solution more stable.         !
   176)    !---------------------------------------------------------------------------------------!
   177)    targetp%nlev_sfcwater    = sourcesite%nlev_sfcwater(ipa)
   178)    ksn                      = targetp%nlev_sfcwater
   179)    targetp%total_sfcw_mass  = 0.d0
   180)    do k = 1, nzs
   181)       targetp%sfcwater_mass(k)    = max(0.d0,dble(sourcesite%sfcwater_mass(k,ipa)))
   182)       targetp%sfcwater_depth(k)   = dble(sourcesite%sfcwater_depth(k,ipa))
   183)       targetp%sfcwater_energy(k)  = dble(sourcesite%sfcwater_energy(k,ipa))                &
   184)                                   * dble(sourcesite%sfcwater_mass(k,ipa))
   185)       targetp%sfcwater_tempk(k)   = dble(sourcesite%sfcwater_tempk(k,ipa))
   186)       targetp%sfcwater_fracliq(k) = dble(sourcesite%sfcwater_fracliq(k,ipa))
   187)       targetp%total_sfcw_mass     = targetp%total_sfcw_mass  + targetp%sfcwater_mass (k)
   188)    end do
   189)    !----- Define the temporary surface water flag. ----------------------------------------!
   190)    if (targetp%nlev_sfcwater == 0) then
   191)       !----- No layer. --------------------------------------------------------------------!
   192)       targetp%flag_sfcwater = 0
   193)    elseif (targetp%total_sfcw_mass < rk4water_stab_thresh) then
   194)       !----- There is water, but the amount is very small. --------------------------------!
   195)       targetp%flag_sfcwater = 1
   196)    else
   197)       !----- There is enough water. -------------------------------------------------------!
   198)       targetp%flag_sfcwater = 2
   199)    end if
   200)    !---------------------------------------------------------------------------------------!
   201) 
   202) 
   203)    !---------------------------------------------------------------------------------------!
   204)    !     Find the total snow depth and fraction of canopy buried in snow.                  !
   205)    !---------------------------------------------------------------------------------------!
   206)    targetp%snowfac          = dble(sourcesite%snowfac(ipa))
   207)    targetp%total_sfcw_depth = dble(sourcesite%total_sfcw_depth(ipa))
   208)    !---------------------------------------------------------------------------------------!
   209) 
   210) 
   211) 
   212)    !---------------------------------------------------------------------------------------!
   213)    !     Compute the ground temperature and specific humidity.                             !
   214)    !---------------------------------------------------------------------------------------!
   215)    k = max(1,ksn)
   216)    call ed_grndvap8(ksn,targetp%soil_water(nzg),targetp%soil_tempk(nzg)                    &
   217)                    ,targetp%soil_fracliq(nzg),targetp%sfcwater_tempk(k)                    &
   218)                    ,targetp%snowfac,targetp%can_prss                                       &
   219)                    ,targetp%can_shv,targetp%ground_shv,targetp%ground_ssh                  &
   220)                    ,targetp%ground_temp,targetp%ground_fliq,targetp%ggsoil)
   221)    !---------------------------------------------------------------------------------------!
   222) 
   223) 
   224) 
   225)    !----- Initialise some turbulence properties. ------------------------------------------!
   226)    targetp%upwp          = dble(sourcesite%upwp  (ipa))
   227)    targetp%wpwp          = dble(sourcesite%wpwp  (ipa))
   228)    targetp%tpwp          = dble(sourcesite%tpwp  (ipa))
   229)    targetp%qpwp          = dble(sourcesite%qpwp  (ipa))
   230)    targetp%cpwp          = dble(sourcesite%cpwp  (ipa))
   231)    !---------------------------------------------------------------------------------------!
   232)   
   233) 
   234)    !---------------------------------------------------------------------------------------!
   235)    !     The virtual pools should be always zero, they are temporary entities used to      !
   236)    ! store the water that falls as throughfall or shedding in the middle of a time step.   !
   237)    ! The temperature is liquid fraction is initialised as the soil temperature, or the     !
   238)    ! temporary surface water temperature of the top most layer, if it exists.              !
   239)    !---------------------------------------------------------------------------------------!
   240)    targetp%virtual_water  = 0.0d0
   241)    targetp%virtual_energy = 0.0d0
   242)    targetp%virtual_depth  = 0.0d0
   243)    if (ksn == 0) then
   244)       targetp%virtual_tempk   = targetp%soil_tempk  (nzg)
   245)       targetp%virtual_fracliq = targetp%soil_fracliq(nzg)
   246)    else
   247)       targetp%virtual_tempk   = targetp%sfcwater_tempk  (ksn)
   248)       targetp%virtual_fracliq = targetp%sfcwater_fracliq(ksn)
   249)    end if
   250)    !---------------------------------------------------------------------------------------!
   251) 
   252) 
   253) 
   254) 
   255)    !---------------------------------------------------------------------------------------!
   256)    !     Here we find the minimum patch-level leaf heat capacity.  If the total patch leaf !
   257)    ! heat capacity is less than this, we scale the cohorts heat capacity inside the        !
   258)    ! integrator, so it preserves the proportional heat capacity and prevents the pool to   !
   259)    ! be too small.                                                                         !
   260)    !---------------------------------------------------------------------------------------!
   261)    rk4aux(ibuff)%any_resolvable = .false.
   262)    do ico=1, cpatch%ncohorts
   263)       !----- Copying the flag that determines whether this cohort is numerically stable. --!
   264)       targetp%leaf_resolvable(ico) = cpatch%leaf_resolvable(ico)
   265)       targetp%wood_resolvable(ico) = cpatch%wood_resolvable(ico)
   266)       if (targetp%leaf_resolvable(ico) .or. targetp%wood_resolvable(ico)) then
   267)          rk4aux(ibuff)%any_resolvable = .true.
   268)       end if
   269)    end do
   270) 
   271) 
   272)    !---------------------------------------------------------------------------------------!
   273)    !     Loop over cohorts.  We also use this loop to compute the fraction of open canopy, !
   274)    ! which is initialised with 1. in case this is an empty patch.                          !
   275)    !---------------------------------------------------------------------------------------!
   276)    targetp%opencan_frac   = dble(sourcesite%opencan_frac(ipa))
   277)    do ico = 1,cpatch%ncohorts
   278)       ipft=cpatch%pft(ico)
   279) 
   280)       !----- Copy the plant density. ------------------------------------------------------!
   281)       targetp%nplant(ico)     = dble(cpatch%nplant(ico))
   282) 
   283)       !----- Copy the leaf area index and total (leaf+branch+twig) area index. ------------!
   284)       targetp%lai(ico)        = dble(cpatch%lai(ico))
   285)       targetp%wai(ico)        = dble(cpatch%wai(ico))
   286)       targetp%tai(ico)        = targetp%lai(ico) + targetp%wai(ico)
   287)       targetp%crown_area(ico) = dble(cpatch%crown_area(ico))
   288)       targetp%elongf(ico)     = dble(cpatch%elongf(ico)) * rk4site%green_leaf_factor(ipft)
   289)       !------------------------------------------------------------------------------------!
   290) 
   291) 
   292)       !------------------------------------------------------------------------------------!
   293)       !     Check whether the leaves of this cohort are considered "safe" or not.  In case !
   294)       ! it is, we copy the water, heat capacity, and energy then compute the temperature   !
   295)       ! and the fraction of leaf water.  Otherwise, just fill with some safe values, but   !
   296)       ! the leaves won't be really solved.                                                 !
   297)       !------------------------------------------------------------------------------------!
   298)       if (targetp%leaf_resolvable(ico)) then
   299)          targetp%leaf_energy(ico) = dble(cpatch%leaf_energy(ico))
   300)          targetp%leaf_water (ico) = max(0.d0,dble(cpatch%leaf_water (ico)))
   301)          targetp%leaf_hcap  (ico) = dble(cpatch%leaf_hcap  (ico))
   302) 
   303)          call uextcm2tl8(targetp%leaf_energy(ico),targetp%leaf_water(ico)                  &
   304)                         ,targetp%leaf_hcap(ico),targetp%leaf_temp(ico)                     &
   305)                         ,targetp%leaf_fliq(ico))
   306)       else
   307)          targetp%leaf_fliq  (ico) = dble(cpatch%leaf_fliq  (ico))
   308)          targetp%leaf_temp  (ico) = dble(cpatch%leaf_temp  (ico))
   309)          targetp%leaf_water (ico) = dble(cpatch%leaf_water (ico))
   310)          targetp%leaf_hcap  (ico) = dble(cpatch%leaf_hcap  (ico))
   311)          targetp%leaf_energy(ico) = cmtl2uext8( targetp%leaf_hcap (ico)                    &
   312)                                               , targetp%leaf_water(ico)                    &
   313)                                               , targetp%leaf_temp (ico)                    &
   314)                                               , targetp%leaf_fliq (ico) )
   315)       end if
   316)       !------------------------------------------------------------------------------------!
   317) 
   318) 
   319)       !------------------------------------------------------------------------------------!
   320)       !     Check whether the wood of this cohort is considered "safe" or not.  In case    !
   321)       ! it is, we copy the water, heat capacity, and energy then compute the temperature   !
   322)       ! and the fraction of leaf water.  Otherwise, just fill with some safe values, but   !
   323)       ! the wood won't be really solved.                                                   !
   324)       !------------------------------------------------------------------------------------!
   325)       if (targetp%wood_resolvable(ico)) then
   326)          targetp%wood_energy(ico) = dble(cpatch%wood_energy(ico))
   327)          targetp%wood_water (ico) = max(0.d0,dble(cpatch%wood_water (ico)))
   328)          targetp%wood_hcap  (ico) = dble(cpatch%wood_hcap  (ico))
   329) 
   330)          call uextcm2tl8(targetp%wood_energy(ico),targetp%wood_water(ico)                  &
   331)                         ,targetp%wood_hcap(ico),targetp%wood_temp(ico)                     &
   332)                         ,targetp%wood_fliq(ico))
   333)       else
   334)          targetp%wood_fliq  (ico) = dble(cpatch%wood_fliq  (ico))
   335)          targetp%wood_temp  (ico) = dble(cpatch%wood_temp  (ico))
   336)          targetp%wood_water (ico) = dble(cpatch%wood_water (ico))
   337)          targetp%wood_hcap  (ico) = dble(cpatch%wood_hcap  (ico))
   338)          targetp%wood_energy(ico) = cmtl2uext8( targetp%wood_hcap (ico)                    &
   339)                                               , targetp%wood_water(ico)                    &
   340)                                               , targetp%wood_temp (ico)                    &
   341)                                               , targetp%wood_fliq (ico) )
   342)       end if
   343)       !------------------------------------------------------------------------------------!
   344) 
   345) 
   346)       !------------------------------------------------------------------------------------!
   347)       !   Make the combined leaf and branchwood pool.  It will be really used only if      !
   348)       ! ibranch_thermo is set to 1.                                                        !
   349)       !------------------------------------------------------------------------------------!
   350)       targetp%veg_resolvable(ico) = targetp%leaf_resolvable(ico) .or.                      &
   351)                                     targetp%wood_resolvable(ico)
   352)       targetp%veg_energy(ico)     = targetp%leaf_energy(ico) + targetp%wood_energy(ico)
   353)       targetp%veg_water(ico)      = targetp%leaf_water(ico)  + targetp%wood_water(ico)
   354)       targetp%veg_hcap(ico)       = targetp%leaf_hcap(ico)   + targetp%wood_hcap(ico)
   355)       !------------------------------------------------------------------------------------!
   356) 
   357) 
   358) 
   359)       !------------------------------------------------------------------------------------!
   360)       !     Compute the leaf intercellular specific humidity, assumed to be at saturation. !
   361)       !------------------------------------------------------------------------------------!
   362)       targetp%lint_shv(ico) = qslif8(targetp%can_prss,targetp%leaf_temp(ico))
   363)       !------------------------------------------------------------------------------------!
   364) 
   365) 
   366) 
   367)       !------ Copy the stomatal conductances and the fraction of open stomata. ------------!
   368)       targetp%fs_open   (ico) = dble(cpatch%fs_open   (ico))
   369)       targetp%gsw_open  (ico) = dble(cpatch%gsw_open  (ico))
   370)       targetp%gsw_closed(ico) = dble(cpatch%gsw_closed(ico))
   371)       !------------------------------------------------------------------------------------!
   372) 
   373) 
   374) 
   375)       !------ Copy the net absorbed radiation (short wave and long wave). -----------------!
   376)       targetp%rshort_l   (ico) = dble(cpatch%rshort_l (ico))
   377)       targetp%rlong_l    (ico) = dble(cpatch%rlong_l  (ico))
   378)       targetp%rshort_w   (ico) = dble(cpatch%rshort_w (ico))
   379)       targetp%rlong_w    (ico) = dble(cpatch%rlong_w  (ico))
   380)       !------------------------------------------------------------------------------------!
   381) 
   382) 
   383) 
   384)       !------------------------------------------------------------------------------------!
   385)       !     Initialise psi_open and psi_closed with zeroes, they will be averaged over the !
   386)       ! course of one Runge-Kutta time step.                                               !
   387)       !------------------------------------------------------------------------------------!
   388)       targetp%psi_open(ico)   = 0.d0
   389)       targetp%psi_closed(ico) = 0.d0
   390)       !------------------------------------------------------------------------------------!
   391)       
   392)    end do
   393)    !---------------------------------------------------------------------------------------!
   394) 
   395) 
   396) 
   397)    !----- Initialise the characteristic properties, and the heat capacities. --------------!
   398)    call canopy_turbulence8(sourcesite,targetp,ipa)
   399)    !---------------------------------------------------------------------------------------!
   400) 
   401)    !----- Diagnostics variables -----------------------------------------------------------!
   402)    if(fast_diagnostics) then
   403)       !------------------------------------------------------------------------------------!
   404)       !     The "budget" variables are not copied here because they are integrated outside !
   405)       ! RK4.  Inside RK4 we only want the contribution of those variables during the span  !
   406)       ! of one time step.                                                                  !
   407)       !------------------------------------------------------------------------------------!
   408)       targetp%avg_ustar          = dble(sourcesite%fmean_ustar         (ipa))
   409)       targetp%avg_tstar          = dble(sourcesite%fmean_tstar         (ipa))
   410)       targetp%avg_qstar          = dble(sourcesite%fmean_qstar         (ipa))
   411)       targetp%avg_cstar          = dble(sourcesite%fmean_cstar         (ipa))
   412)       targetp%avg_carbon_ac      = dble(sourcesite%fmean_carbon_ac     (ipa))
   413)       targetp%avg_carbon_st      = dble(sourcesite%fmean_carbon_st     (ipa))
   414)       targetp%avg_vapor_gc       = dble(sourcesite%fmean_vapor_gc      (ipa))
   415)       targetp%avg_throughfall    = dble(sourcesite%fmean_throughfall   (ipa))
   416)       targetp%avg_vapor_ac       = dble(sourcesite%fmean_vapor_ac      (ipa))
   417)       targetp%avg_drainage       = dble(sourcesite%fmean_drainage      (ipa))
   418)       targetp%avg_qdrainage      = dble(sourcesite%fmean_qdrainage     (ipa))
   419)       targetp%avg_qthroughfall   = dble(sourcesite%fmean_qthroughfall  (ipa))
   420)       targetp%avg_sensible_gc    = dble(sourcesite%fmean_sensible_gc   (ipa))
   421)       targetp%avg_sensible_ac    = dble(sourcesite%fmean_sensible_ac   (ipa))
   422) 
   423)       do k = rk4site%lsl, nzg
   424)          targetp%avg_sensible_gg(k) = dble(sourcesite%fmean_sensible_gg(k,ipa))
   425)          targetp%avg_smoist_gg(k)   = dble(sourcesite%fmean_smoist_gg(k,ipa)  )
   426)          targetp%avg_transloss(k)   = dble(sourcesite%fmean_transloss(k,ipa)  )
   427)       end do
   428) 
   429)       do ico=1,cpatch%ncohorts
   430)          targetp%avg_sensible_lc    (ico) = dble(cpatch%fmean_sensible_lc   (ico))
   431)          targetp%avg_sensible_wc    (ico) = dble(cpatch%fmean_sensible_wc   (ico))

   436)          targetp%avg_intercepted_aw (ico) = dble(cpatch%fmean_intercepted_aw(ico))

   438)          targetp%avg_wshed_wg       (ico) = dble(cpatch%fmean_wshed_wg      (ico))
   439)       end do
   440) 
   441)    end if
   442)    if (checkbudget) then
   443)       targetp%co2budget_storage     = dble(sourcesite%co2budget_initialstorage(ipa))
   444)       targetp%ebudget_storage       = dble(sourcesite%ebudget_initialstorage(ipa))
   445)       targetp%wbudget_storage       = dble(sourcesite%wbudget_initialstorage(ipa))
   446)       targetp%co2budget_loss2atm    = 0.d0
   447)       targetp%ebudget_netrad        = 0.d0
   448)       targetp%ebudget_loss2atm      = 0.d0
   449)       targetp%ebudget_loss2drainage = 0.d0
   450)       targetp%ebudget_loss2runoff   = 0.d0
   451)       targetp%wbudget_loss2atm      = 0.d0
   452)       targetp%wbudget_loss2drainage = 0.d0
   453)       targetp%wbudget_loss2runoff   = 0.d0
   454)    end if
   455)    !---------------------------------------------------------------------------------------!
   456) 
   457)    !----- Water deficit, always start with zero. ------------------------------------------!
   458)    targetp%water_deficit = 0.d0
   459)    !---------------------------------------------------------------------------------------!
   460) 
   461) 
   462)    if (print_detailed) call reset_rk4_fluxes(targetp)
   463)    return

   545) !     This subroutine is called before the sanity check, and updates the diagnostic vari-  !
   546) ! ables, namely the temperature and liquid fraction of leaf water, soil layers and         !
   547) ! temporary snow/pond layers.                                                                      !
   548) !------------------------------------------------------------------------------------------!
   549) subroutine update_diagnostic_vars(initp, csite,ipa)
   550)    use rk4_coms              , only : rk4site               & ! intent(in)
   551)                                     , ibranch_thermo        & ! intent(in)
   552)                                     , rk4tiny_sfcw_mass     & ! intent(in)
   553)                                     , rk4min_sfcw_mass      & ! intent(in)
   554)                                     , rk4min_virt_water     & ! intent(in)
   555)                                     , rk4min_can_shv        & ! intent(in)
   556)                                     , rk4max_can_shv        & ! intent(in)
   557)                                     , rk4min_veg_lwater     & ! intent(in)
   558)                                     , rk4min_veg_temp       & ! intent(in)
   559)                                     , rk4max_veg_temp       & ! intent(in)
   560)                                     , rk4min_soil_temp      & ! intent(in)
   561)                                     , rk4max_soil_temp      & ! intent(in)
   562)                                     , rk4aux                & 
   563)                                     , rk4min_sfcw_temp      & ! intent(in)
   564)                                     , rk4max_sfcw_temp      & ! intent(in)
   565)                                     , tiny_offset           & ! intent(in)
   566)                                     , rk4patchtype          ! ! structure
   567)    use ed_state_vars         , only : sitetype              & ! structure
   568)                                     , patchtype             ! ! structure
   569)    use soil_coms             , only : soil8                 & ! intent(in)
   570)                                     , dslz8                 & ! intent(in)
   571)                                     , dslzi8                & ! intent(in)
   572)                                     , soil_rough8           & ! intent(in)
   573)                                     , ny07_eq04_a8          & ! intent(in)
   574)                                     , ny07_eq04_m8          & ! intent(in)
   575)                                     , matric_potential8     ! ! function
   576)    use grid_coms             , only : nzg                   & ! intent(in)
   577)                                     , nzs                   ! ! intent(in)
   578)    use therm_lib8            , only : uextcm2tl8            & ! subroutine
   579)                                     , uint2tl8              & ! subroutine
   580)                                     , tl2uint8              & ! function
   581)                                     , cmtl2uext8            & ! function
   582)                                     , thetaeiv8             & ! function
   583)                                     , rehuil8               & ! function
   584)                                     , qslif8                & ! function
   585)                                     , hq2temp8              & ! function
   586)                                     , extemp2theta8         & ! function
   587)                                     , thil2tqall8           ! ! function
   588)    use consts_coms           , only : t3ple8                & ! intent(in)
   589)                                     , cpdry8                & ! intent(in)
   590)                                     , cph2o8                & ! intent(in)
   591)                                     , wdns8                 & ! intent(in)
   592)                                     , fsdns8                & ! intent(in)
   593)                                     , fsdnsi8               & ! intent(in)
   594)                                     , rdryi8                & ! intent(in)
   595)                                     , rdry8                 & ! intent(in)
   596)                                     , epim18                & ! intent(in)
   597)                                     , toodry8               ! ! intent(in)
   598)    use canopy_struct_dynamics, only : canopy_turbulence8    ! ! subroutine
   599)    use ed_therm_lib          , only : ed_grndvap8           ! ! subroutine
   600)    !$ use omp_lib
   601)    implicit none
   602)    !----- Arguments -----------------------------------------------------------------------!
   603)    type(rk4patchtype) , target     :: initp
   604)    type(sitetype)     , target     :: csite
   605)    integer            , intent(in) :: ipa
   606)    !----- Local variables -----------------------------------------------------------------!
   607)    type(patchtype)        , pointer :: cpatch
   608)    integer                          :: ico
   609)    integer                          :: k
   610)    integer                          :: ksn
   611)    integer                          :: nsoil
   612)    logical                          :: ok_shv
   613)    logical                          :: ok_enthalpy
   614)    logical                          :: ok_theta
   615)    logical                          :: ok_ground
   616)    logical                          :: ok_sfcw
   617)    logical                          :: ok_leaf
   618)    logical                          :: ok_wood
   619)    logical                          :: ok_slwater
   620)    logical                          :: ok_sltemp
   621)    real(kind=8)                     :: int_sfcw_energy
   622)    real(kind=8)                     :: int_virt_energy
   623)    real(kind=8)                     :: energy_tot
   624)    real(kind=8)                     :: wmass_tot
   625)    real(kind=8)                     :: veg_temp
   626)    real(kind=8)                     :: veg_fliq
   627)    real(kind=8)                     :: hcapdry_tot
   628)    real(kind=8)                     :: rk4min_veg_water
   629)    real(kind=8)                     :: rk4min_leaf_water
   630)    real(kind=8)                     :: rk4min_wood_water
   631)    real(kind=8)                     :: wgt_leaf
   632)    real(kind=8)                     :: wgt_wood
   633)    real(kind=8)                     :: bulk_sfcw_dens
   634)    integer                          :: ibuff
   635)    !---------------------------------------------------------------------------------------!
   636) 
   637)    ibuff = 1
   638)    !$ ibuff = OMP_get_thread_num()+1
   639) 
   640)    !----- Then we define some logicals to make the code cleaner. --------------------------!
   641)    ok_shv      = initp%can_shv      >= rk4min_can_shv        .and.                         &
   642)                  initp%can_shv      <= rk4max_can_shv
   643)    ok_enthalpy = initp%can_enthalpy >= rk4aux(ibuff)%rk4min_can_enthalpy   .and.           &
   644)                  initp%can_enthalpy <= rk4aux(ibuff)%rk4max_can_enthalpy
   645) 
   646)    !---------------------------------------------------------------------------------------!
   647)    !     Here we convert theta into temperature, potential temperature, and density, and   !
   648)    ! ice-vapour equivalent potential temperature.  The latter variable (or its natural     !
   649)    ! log) should eventually become the prognostic variable for canopy air space entropy    !
   650)    ! when we add condensed/frozen water in the canopy air space.                           !
   651)    !---------------------------------------------------------------------------------------!
   652)    if (ok_shv .and. ok_enthalpy) then
   653) 
   654) 
   655)       !----- Update the canopy air space heat capacity at constant pressure. --------------!
   656)       initp%can_cp = (1.d0 - initp%can_shv) * cpdry8 + initp%can_shv * cph2o8
   657)       !------------------------------------------------------------------------------------!
   658) 
   659)       !----- Find the canopy air temperature. ---------------------------------------------!
   660)       initp%can_temp  = hq2temp8(initp%can_enthalpy,initp%can_shv,.true.)
   661)       !------------------------------------------------------------------------------------!
   662) 
   663) 
   664)       !----- Find the new potential temperature. ------------------------------------------!
   665)       initp%can_theta = extemp2theta8(initp%can_exner,initp%can_temp)
   666)       !------------------------------------------------------------------------------------!
   667) 
   668) 
   669)       !----- Check whether the potential temperature makes sense or not. ------------------!
   670)       ok_theta = initp%can_theta >= rk4aux(ibuff)%rk4min_can_theta .and.                   &
   671)                  initp%can_theta <= rk4aux(ibuff)%rk4max_can_theta
   672)       !------------------------------------------------------------------------------------!
   673) 
   674) 
   675) 
   676)       !------------------------------------------------------------------------------------!
   677)       !      Compute the other canopy air parameters only if the potential temperature     !
   678)       ! makes sense.  Sometimes enthalpy makes sense even though the temperature is bad,   !
   679)       ! because can_shv is way off in the opposite direction of temperature.               !
   680)       !------------------------------------------------------------------------------------!
   681)       if (ok_theta) then
   682) 
   683) 
   684)          !---------------------------------------------------------------------------------!
   685)          !     Find the derived humidity variables.                                        !
   686)          !---------------------------------------------------------------------------------!
   687)          initp%can_rhv   = rehuil8(initp%can_prss,initp%can_temp,initp%can_shv,.true.)
   688)          initp%can_ssh   = qslif8(initp%can_prss,initp%can_temp)
   689)          !---------------------------------------------------------------------------------!
   690)       end if
   691)    else
   692)       !----- Either enthalpy or specific humidity is screwed, reject theta too. -----------!
   693)       ok_theta = .false.
   694)    end if
   695)    !---------------------------------------------------------------------------------------!
   696) 
   697) 
   698) 
   699) 
   700) 
   701)    !----- Update soil temperature and liquid water fraction. ------------------------------!
   702)    ok_slwater = .true.
   703)    ok_sltemp  = .true.
   704)    do k = rk4site%lsl, nzg
   705)       nsoil = rk4site%ntext_soil(k)
   706)       !----- Check whether soil water is fine. --------------------------------------------!
   707)       ok_slwater = ok_slwater                                  .and.                       &
   708)                    initp%soil_water(k) >= rk4aux(ibuff)%rk4min_soil_water(k) .and.                       &
   709)                    initp%soil_water(k) <= rk4aux(ibuff)%rk4max_soil_water(k)
   710)       if (ok_slwater) then
   711)          initp%soil_mstpot(k) = matric_potential8(nsoil,initp%soil_water(k))
   712)       end if
   713)       !------------------------------------------------------------------------------------!
   714) 
   715) 
   716)       !------------------------------------------------------------------------------------!
   717)       !      We find the temperature in any case.  If screwy soil water leads to screwy    !
   718)       ! temperature, the sanity check will fail twice.                                     !
   719)       !------------------------------------------------------------------------------------!
   720)       call uextcm2tl8(initp%soil_energy(k),initp%soil_water(k)*wdns8,soil8(nsoil)%slcpd    &
   721)                      ,initp%soil_tempk(k),initp%soil_fracliq(k))
   722)       ok_sltemp = ok_sltemp                                   .and.                        &
   723)                   initp%soil_tempk(k) >= rk4min_soil_temp     .and.                        &
   724)                   initp%soil_tempk(k) <= rk4max_soil_temp
   725)    end do
   726)    !---------------------------------------------------------------------------------------!
   727) 
   728) 
   729) 
   730)    !---------------------------------------------------------------------------------------!
   731)    !    Update surface water temperature and liquid water fraction, remembering that in-   !
   732)    ! side the RK4 integration, surface water energy is in J/m�. The abs is necessary be-   !
   733)    ! cause surface mass may indeed become too negative during the integration process and  !
   734)    ! if it happens, we want the step to be rejected.                                       !
   735)    !---------------------------------------------------------------------------------------!
   736)    ok_sfcw = .true.
   737)    ksn = initp%nlev_sfcwater
   738)    initp%total_sfcw_depth = 0.d0
   739)    initp%total_sfcw_mass  = 0.d0
   740)    sfcwloop: do k=1,ksn
   741)       if (initp%sfcwater_mass(k) < rk4min_sfcw_mass) then 
   742)          !---------------------------------------------------------------------------------!
   743)          !     Surface water mass doesn't make sense.  Skip because the step is going to   !
   744)          ! be rejected and we may not be able to compute the temperature.                  !
   745)          !---------------------------------------------------------------------------------!
   746)          ok_sfcw = .false.
   747)          cycle sfcwloop
   748)       elseif (initp%flag_sfcwater == 1) then
   749)          !---------------------------------------------------------------------------------!
   750)          !     Water layer is too thin to be computationally stable, apply the quick heat  !
   751)          ! exchange between the soil top layer and the temporary surface water.  Find the  !
   752)          ! total internal energy of the combined pool (top soil layer plus the thin        !
   753)          ! temporary surface water).  The units of soil properties are J/m3 for the        !
   754)          ! internal energy, and m3/m3 for soil water, whilst the temporary surface water   !
   755)          ! has units of J/m2 for internal energy and kg/m2 for mass.  We use the standard  !
   756)          ! for the temporary surface water.                                                !
   757)          !---------------------------------------------------------------------------------!
   758)          energy_tot  = initp%sfcwater_energy(k) + initp%soil_energy(nzg) * dslz8(nzg)
   759)          wmass_tot   = initp%sfcwater_mass  (k)                                            &
   760)                      + initp%soil_water (nzg) * dslz8(nzg) * wdns8
   761)          hcapdry_tot = soil8(rk4site%ntext_soil(nzg))%slcpd * dslz8(nzg)
   762)          !---------------------------------------------------------------------------------!
   763) 
   764) 
   765) 
   766)          !---------------------------------------------------------------------------------!
   767)          !      Find the equilibrium temperature and liquid/ice partition.   Because we    !
   768)          ! are assuming thermal equilibrium, the temperature and liquid fraction of the    !
   769)          ! attempted layer is the same as the average temperature of the augmented pool.   !
   770)          !---------------------------------------------------------------------------------!
   771)          call uextcm2tl8(energy_tot,wmass_tot,hcapdry_tot                                  &
   772)                         ,initp%sfcwater_tempk(k),initp%sfcwater_fracliq(k))
   773)          initp%soil_tempk  (nzg) = initp%sfcwater_tempk  (k)
   774)          initp%soil_fracliq(nzg) = initp%sfcwater_fracliq(k) 
   775)          !---------------------------------------------------------------------------------!
   776) 
   777) 
   778)          !---------------------------------------------------------------------------------!
   779)          !    Re-compute the internal energy of the temporary layer, using the temperature !
   780)          ! and fraction of liquid water distribution we have just found, keeping the mass  !
   781)          ! constant.                                                                       !
   782)          !---------------------------------------------------------------------------------!
   783)          initp%sfcwater_energy(k) = initp%sfcwater_mass(k)                                 &
   784)                                   * tl2uint8( initp%sfcwater_tempk(k)                      &
   785)                                             , initp%sfcwater_fracliq(k) )
   786)          !---------------------------------------------------------------------------------!
   787) 
   788) 
   789) 
   790)          !---------------------------------------------------------------------------------!
   791)          !    Re-calculate the top soil internal energy, by removing the attempted surface !
   792)          ! water energy from the total energy, and converting it back to J/m3.  The total  !
   793)          ! amount of water does not need to be re-calculated at this time.                 !
   794)          !---------------------------------------------------------------------------------!
   795)          initp%soil_energy(nzg)  = (energy_tot - initp%sfcwater_energy(k)) * dslzi8(nzg)
   796)          !---------------------------------------------------------------------------------!
   797)       else
   798)          !---------------------------------------------------------------------------------!
   799)          !      Convert surface water energy from extensive quantity (J/m2) to intensive   !
   800)          ! quantity (J/kg), then update the temperature and liquid water fraction.  We     !
   801)          ! only do this when there is enough mass, because otherwise the amount of energy  !
   802)          ! is too small that the code would give some unreasonable results.  Also, this is !
   803)          ! the minimum amount of water needed for a layer to exist, anything less than     !
   804)          ! that would be entirely absorbed by the soil layer.                              !
   805)          !---------------------------------------------------------------------------------!
   806)          int_sfcw_energy = initp%sfcwater_energy(k)/initp%sfcwater_mass(k)
   807)          call uint2tl8(int_sfcw_energy,initp%sfcwater_tempk(k),initp%sfcwater_fracliq(k))
   808)       end if
   809)       initp%total_sfcw_depth = initp%total_sfcw_depth + initp%sfcwater_depth(k)
   810)       initp%total_sfcw_mass  = initp%total_sfcw_mass  + initp%sfcwater_mass (k)
   811)    end do sfcwloop
   812)    !---------------------------------------------------------------------------------------!
   813)    !    For non-existent layers of temporary surface water, we copy the temperature and    !
   814)    ! liquid fraction from the layer beneath.                                               !
   815)    !---------------------------------------------------------------------------------------!
   816)    aboveloop: do k=ksn+1,nzs
   817)       if (k == 1) then
   818)          initp%sfcwater_tempk  (k) = initp%soil_tempk  (nzg)
   819)          initp%sfcwater_fracliq(k) = initp%soil_fracliq(nzg)
   820)       else
   821)          initp%sfcwater_tempk  (k) = initp%sfcwater_tempk  (k-1)
   822)          initp%sfcwater_fracliq(k) = initp%sfcwater_fracliq(k-1)
   823)       end if
   824)    end do aboveloop
   825)    !---------------------------------------------------------------------------------------!
   826) 
   827) 
   828) 
   829)    !---------------------------------------------------------------------------------------!
   830)    !     Update the fraction of the canopy covered in snow.  I could not find any          !
   831)    ! reference for the original method (commented out), so I implemented the method used   !
   832)    ! in CLM-4, which is based on:                                                          !
   833)    !                                                                                       !
   834)    ! Niu, G.-Y., and Z.-L. Yang (2007), An observation-based formulation of snow cover     !
   835)    !    fraction and its evaluation over large North American river basins,                !
   836)    !    J. Geophys. Res., 112, D21101, doi:10.1029/2007JD008674                            !
   837)    !---------------------------------------------------------------------------------------!
   838)    ! initp%snowfac = min(9.9d-1,initp%total_sfcw_depth/initp%veg_height)
   839)    if (initp%total_sfcw_mass > rk4tiny_sfcw_mass) then
   840)       bulk_sfcw_dens = max( fsdns8                                                         &
   841)                           , min( wdns8, initp%total_sfcw_mass / initp%total_sfcw_depth ) )
   842)       initp%snowfac  = max( 0.d0, min( 9.9d-1                                              &
   843)                           , tanh( initp%total_sfcw_depth                                   &
   844)                                 / ( ny07_eq04_a8 * soil_rough8                             &
   845)                                   * (bulk_sfcw_dens * fsdnsi8) ** ny07_eq04_m8 ) ) ) )
   846)    else
   847)       initp%snowfac  = 0.d0
   848)    end if
   849)    !---------------------------------------------------------------------------------------!
   850) 
   851) 
   852)    !---------------------------------------------------------------------------------------!
   853)    !     Update the virtual pool temperature and liquid water fraction, only when the      !
   854)    ! total mass of the virtual pool makes sense and is not zero.  When the amount of water !
   855)    ! is too small, we impose the same temperature as the exposed surface (either the top   !
   856)    ! soil layer or the top temporary surface water layer), because the amount of energy is !
   857)    ! too small to get an acurate temperature anyway, and all the water and energy are      !
   858)    ! going to be absorbed by the top soil layer.                                           !
   859)    !---------------------------------------------------------------------------------------!
   860)    if (initp%virtual_water < rk4min_virt_water) then
   861)       continue
   862)    elseif (abs(initp%virtual_water) > rk4tiny_sfcw_mass) then
   863)       int_virt_energy = initp%virtual_energy / initp%virtual_water
   864)       call uint2tl8(int_virt_energy,initp%virtual_tempk,initp%virtual_fracliq)
   865)    elseif (ksn == 0) then
   866)       initp%virtual_tempk   = initp%soil_tempk(nzg)
   867)       initp%virtual_fracliq = initp%soil_fracliq(nzg)
   868)    else
   869)       initp%virtual_tempk   = initp%sfcwater_tempk(ksn)
   870)       initp%virtual_fracliq = initp%sfcwater_fracliq(ksn)
   871)    end if
   872)    !---------------------------------------------------------------------------------------!
   873) 
   874) 
   875)    !---------------------------------------------------------------------------------------!
   876)    !     Compute the ground temperature and specific humidity.                             !
   877)    !---------------------------------------------------------------------------------------!
   878)    ok_ground = initp%soil_tempk(nzg) >= rk4min_soil_temp       .and.                       &
   879)                initp%soil_tempk(nzg) <= rk4max_soil_temp       .and.                       &
   880)                initp%soil_water(nzg) >= rk4aux(ibuff)%rk4min_soil_water(nzg) .and.                       &
   881)                initp%soil_water(nzg) <= rk4aux(ibuff)%rk4max_soil_water(nzg)
   882)    if (ksn > 0) then
   883)       ok_ground = ok_ground                                     .and.                      &
   884)                   initp%sfcwater_tempk(ksn) >= rk4min_sfcw_temp .and.                      &
   885)                   initp%sfcwater_tempk(ksn) <= rk4max_sfcw_temp
   886)    end if
   887)    if (ok_ground) then
   888)       k = max(1,ksn)
   889)       call ed_grndvap8(ksn,initp%soil_water(nzg),initp%soil_tempk(nzg)                     &
   890)                       ,initp%soil_fracliq(nzg),initp%sfcwater_tempk(k)                     &
   891)                       ,initp%snowfac,initp%can_prss                                        &
   892)                       ,initp%can_shv,initp%ground_shv,initp%ground_ssh,initp%ground_temp   &
   893)                       ,initp%ground_fliq,initp%ggsoil)
   894)    end if
   895)    !---------------------------------------------------------------------------------------!
   896) 
   897) 
   898)    !---------------------------------------------------------------------------------------!
   899)    !    Initialise the leaf and wood checks here.                                          !
   900)    !---------------------------------------------------------------------------------------!
   901)    ok_leaf = .true.
   902)    ok_wood = .true.
   903)    cpatch => csite%patch(ipa)
   904)    !---------------------------------------------------------------------------------------!
   905) 
   906) 
   907) 
   908) 
   909) 
   910)    !---------------------------------------------------------------------------------------!
   911)    !     Now we update leaf and branch properties, based on which kind of branch thermo-   !
   912)    ! dynamics we're using.                                                                 !
   913)    !---------------------------------------------------------------------------------------!
   914)    select case(ibranch_thermo)
   915)    case (1)
   916) 
   917)       !------------------------------------------------------------------------------------!
   918)       !     The combined case.  Here we assume the leaves and wood are eternally in        !
   919)       ! thermal equilibrium, and that the total water is evenly spread over branches and   !
   920)       ! leaves.  We then proceed to reconstruct the leaf and wood internal energy.  This,  !
   921)       ! of course, if the step produced decent values, otherwise we bypass and reject the  !
   922)       !step.                                                                               !
   923)       !------------------------------------------------------------------------------------!
   924)       vegloop: do ico=1,cpatch%ncohorts
   925)          !----- Check whether the cohort is safe... ---------------------------------------!
   926)          if (initp%veg_resolvable(ico)) then
   927) 
   928)             !------------------------------------------------------------------------------!
   929)             !     Find the weighting factors for leaves and branches, so we put the right  !
   930)             ! amount of each on top of each surface.                                       !
   931)             !------------------------------------------------------------------------------!
   932)             if (initp%leaf_resolvable(ico) .and. initp%wood_resolvable(ico)) then
   933)                !----- Both leaves and branchwood are solved, split according to LAI/WAI. --!
   934)                wgt_leaf = initp%lai(ico) / initp%tai(ico)
   935)                wgt_wood = 1.d0 - wgt_leaf
   936)             elseif (initp%leaf_resolvable(ico)) then
   937)                wgt_leaf = 1.d0
   938)                wgt_wood = 0.d0
   939)             else
   940)                wgt_leaf = 0.d0
   941)                wgt_wood = 1.d0
   942)             end if
   943)             !------------------------------------------------------------------------------!
   944) 
   945) 
   946) 
   947)             !----- Find the minimum vegetation water for this cohort. ---------------------!
   948)             rk4min_veg_water = rk4min_veg_lwater * initp%tai(ico)
   949)             !------------------------------------------------------------------------------!
   950) 
   951) 
   952) 
   953)             !------------------------------------------------------------------------------!
   954)             !     Update leaf temperature and liquid fraction only if leaf water makes     !
   955)             ! sense.                                                                       !
   956)             !------------------------------------------------------------------------------!
   957)             if (initp%veg_water(ico) < rk4min_veg_water) then
   958)                !---------------------------------------------------------------------------!
   959)                !     Water is too negative, we must reject the step.  To be safe, we cheat !
   960)                ! and put the total water to both pools.                                    !
   961)                !---------------------------------------------------------------------------!
   962)                initp%leaf_water(ico) = initp%veg_water(ico)
   963)                initp%wood_water(ico) = initp%veg_water(ico)
   964)                !---------------------------------------------------------------------------!
   965) 
   966) 
   967) 
   968)                !----- Flag the step as bad so it doesn't find turbulence parameters. ------!
   969)                ok_leaf = .false.
   970)                ok_wood = .false.
   971)                !---------------------------------------------------------------------------!
   972) 
   973) 
   974) 
   975)                !----- Move on to the next cohort. -----------------------------------------!
   976)                cycle vegloop
   977)                !---------------------------------------------------------------------------!
   978)             else
   979)                !----- Find the temperature and liquid fraction. ---------------------------!
   980)                call uextcm2tl8(initp%veg_energy(ico),initp%veg_water(ico)                  &
   981)                               ,initp%veg_hcap(ico),veg_temp,veg_fliq)
   982)                !---------------------------------------------------------------------------!
   983) 
   984) 
   985) 
   986)                if (veg_temp < rk4min_veg_temp .or. veg_temp > rk4max_veg_temp) then
   987)                   !------------------------------------------------------------------------!
   988)                   !     Temperature is off, we must reject the step.  To be safe, we       !
   989)                   ! cheat and put the bad temperature in both pools.                       !
   990)                   !------------------------------------------------------------------------!
   991)                   initp%leaf_temp(ico) = veg_temp
   992)                   initp%wood_temp(ico) = veg_temp
   993)                   initp%leaf_fliq(ico) = veg_fliq
   994)                   initp%wood_fliq(ico) = veg_fliq
   995)                   !------------------------------------------------------------------------!
   996) 
   997)                   !----- Flag the step as bad so it doesn't find turbulence parameters. ---!
   998)                   ok_leaf = .false.
   999)                   ok_wood = .false.
  1000)                   !------------------------------------------------------------------------!
  1001) 
  1002) 
  1003) 
  1004)                   !----- Move on to the next cohort. --------------------------------------!
  1005)                   cycle vegloop
  1006)                   !------------------------------------------------------------------------!
  1007)                else
  1008)                   !----- Copy veg temperature and liquid fraction to leaves and wood. -----!
  1009)                   initp%leaf_temp(ico) = veg_temp
  1010)                   initp%wood_temp(ico) = veg_temp
  1011)                   initp%leaf_fliq(ico) = veg_fliq
  1012)                   initp%wood_fliq(ico) = veg_fliq
  1013)                   !------------------------------------------------------------------------!
  1014) 
  1015) 
  1016)                   !----- Split vegetation water according to LAI and WAI. -----------------!
  1017)                   initp%leaf_water(ico) = initp%veg_water(ico) * wgt_leaf
  1018)                   initp%wood_water(ico) = initp%veg_water(ico) * wgt_wood
  1019)                   !------------------------------------------------------------------------!
  1020) 
  1021) 
  1022)                   !----- Find lead and wood internal energy. ------------------------------!
  1023)                   initp%leaf_energy(ico) = cmtl2uext8( initp%leaf_hcap (ico)               &
  1024)                                                      , initp%leaf_water(ico)               &
  1025)                                                      , initp%leaf_temp (ico)               &
  1026)                                                      , initp%leaf_fliq (ico) )
  1027)                   initp%wood_energy(ico) = cmtl2uext8( initp%wood_hcap (ico)               &
  1028)                                                      , initp%wood_water(ico)               &
  1029)                                                      , initp%wood_temp (ico)               &
  1030)                                                      , initp%wood_fliq (ico) )
  1031) 
  1032)                   !------------------------------------------------------------------------!
  1033) 
  1034) 
  1035)                   !------------------------------------------------------------------------!
  1036)                   !     Compute the leaf intercellular specific humidity, assumed to be at !
  1037)                   ! saturation.                                                            !
  1038)                   !------------------------------------------------------------------------!
  1039)                   initp%lint_shv(ico) = qslif8(initp%can_prss,initp%leaf_temp(ico))
  1040)                   !------------------------------------------------------------------------!
  1041)                end if
  1042)             end if
  1043)             !------------------------------------------------------------------------------!
  1044)          else
  1045)             !------------------------------------------------------------------------------!
  1046)             !     For plants with minimal foliage or very sparse patches, fix the leaf     !
  1047)             ! and wood temperatures to the canopy air space and force leaf_water and       !
  1048)             ! wood_water to be zero.                                                       !
  1049)             !------------------------------------------------------------------------------!
  1050)             initp%leaf_temp(ico)   = initp%can_temp 
  1051)             initp%wood_temp(ico)   = initp%can_temp
  1052)             initp%leaf_water(ico)  = 0.d0
  1053)             initp%wood_water(ico)  = 0.d0
  1054)             initp%veg_water(ico)   = 0.d0
  1055)             if (initp%can_temp == t3ple8) then
  1056)                initp%leaf_fliq(ico) = 5.d-1
  1057)                initp%wood_fliq(ico) = 5.d-1
  1058)             elseif (initp%can_temp > t3ple8) then
  1059)                initp%leaf_fliq(ico) = 1.d0
  1060)                initp%wood_fliq(ico) = 1.d0
  1061)             else
  1062)                initp%leaf_fliq(ico) = 0.d0
  1063)                initp%wood_fliq(ico) = 0.d0
  1064)             end if
  1065)             initp%leaf_energy(ico) = cmtl2uext8( initp%leaf_hcap (ico)                     &
  1066)                                                , initp%leaf_water(ico)                     &
  1067)                                                , initp%leaf_temp (ico)                     &
  1068)                                                , initp%leaf_fliq (ico) )
  1069) 
  1070)             initp%wood_energy(ico) = cmtl2uext8( initp%wood_hcap (ico)                     &
  1071)                                                , initp%wood_water(ico)                     &
  1072)                                                , initp%wood_temp (ico)                     &
  1073)                                                , initp%wood_fliq (ico) )
  1074)             initp%veg_energy(ico)  = initp%leaf_energy(ico) + initp%wood_energy(ico)
  1075)             !------------------------------------------------------------------------------!
  1076) 
  1077) 
  1078) 
  1079) 
  1080) 
  1081)             !------------------------------------------------------------------------------!
  1082)             !     Compute the leaf intercellular specific humidity, assumed to be at       !
  1083)             ! saturation.                                                                  !
  1084)             !------------------------------------------------------------------------------!
  1085)             initp%lint_shv(ico) = qslif8(initp%can_prss,initp%leaf_temp(ico))
  1086)             !------------------------------------------------------------------------------!
  1087)          end if
  1088)       end do vegloop
  1089)    case default
  1090)       !------------------------------------------------------------------------------------!
  1091)       !     Loop over cohorts to update the leaf temperature and liquid fraction.  This is !
  1092)       ! done here only if leaves aren't being solved together with branches as a combined  !
  1093)       ! pool (either being the only vegetation pool or treated as a separate pool).        !
  1094)       !------------------------------------------------------------------------------------!
  1095)       leafloop: do ico=1,cpatch%ncohorts
  1096)          !----- Check whether the leaves of this cohort are safe... -----------------------!
  1097)          if (initp%leaf_resolvable(ico)) then
  1098) 
  1099)             !----- Find the minimum leaf water for this cohort. ---------------------------!
  1100)             rk4min_leaf_water = rk4min_veg_lwater * initp%lai(ico)
  1101) 
  1102)             !------------------------------------------------------------------------------!
  1103)             !     Update leaf temperature and liquid fraction only if leaf water makes     !
  1104)             ! sense.                                                                       !
  1105)             !------------------------------------------------------------------------------!
  1106)             if (initp%leaf_water(ico) < rk4min_leaf_water) then
  1107)                ok_leaf = .false.
  1108)                cycle leafloop
  1109)             else
  1110)                call uextcm2tl8(initp%leaf_energy(ico),initp%leaf_water(ico)                &
  1111)                               ,initp%leaf_hcap(ico),initp%leaf_temp(ico)                   &
  1112)                               ,initp%leaf_fliq(ico))
  1113)                if (initp%leaf_temp(ico) < rk4min_veg_temp .or.                             &
  1114)                    initp%leaf_temp(ico) > rk4max_veg_temp) then
  1115)                   ok_leaf = .false.
  1116)                   cycle leafloop
  1117)                else
  1118)                   !------------------------------------------------------------------------!
  1119)                   !     Compute the leaf intercellular specific humidity, assumed to be at !
  1120)                   ! saturation.                                                            !
  1121)                   !------------------------------------------------------------------------!
  1122)                   initp%lint_shv(ico) = qslif8(initp%can_prss,initp%leaf_temp(ico))
  1123)                   !------------------------------------------------------------------------!
  1124)                end if
  1125)             end if
  1126)             !------------------------------------------------------------------------------!
  1127)          else
  1128)             !------------------------------------------------------------------------------!
  1129)             !     For plants with minimal foliage or very sparse patches, fix the leaf     !
  1130)             ! temperature to the canopy air space and force leaf_water to be zero.         !
  1131)             !------------------------------------------------------------------------------!
  1132)             initp%leaf_temp(ico)   = initp%can_temp
  1133)             initp%leaf_water(ico)  = 0.d0
  1134)             initp%leaf_energy(ico) = cmtl2uext8( initp%leaf_hcap (ico)                     &
  1135)                                                , initp%leaf_water(ico)                     &
  1136)                                                , initp%leaf_temp (ico)                     &
  1137)                                                , initp%leaf_fliq (ico) )
  1138) 
  1139)             if (initp%leaf_temp(ico) == t3ple8) then
  1140)                initp%leaf_fliq(ico) = 5.d-1
  1141)             elseif (initp%leaf_temp(ico) > t3ple8) then
  1142)                initp%leaf_fliq(ico) = 1.d0
  1143)             else
  1144)                initp%leaf_fliq(ico) = 0.d0
  1145)             end if
  1146)             !------------------------------------------------------------------------------!
  1147) 
  1148) 
  1149) 
  1150) 
  1151) 
  1152)             !------------------------------------------------------------------------------!
  1153)             !     Compute the leaf intercellular specific humidity, assumed to be at       !
  1154)             ! saturation.                                                                  !
  1155)             !------------------------------------------------------------------------------!
  1156)             initp%lint_shv(ico) = qslif8(initp%can_prss,initp%leaf_temp(ico))
  1157)             !------------------------------------------------------------------------------!
  1158)          end if
  1159)       end do leafloop
  1160)       !------------------------------------------------------------------------------------!
  1161) 
  1162) 
  1163) 
  1164)       !------------------------------------------------------------------------------------!
  1165)       !     Loop over cohorts to update the wood temperature and liquid fraction.  This is !
  1166)       ! done only when branches are being solved by themselves.                            !
  1167)       !------------------------------------------------------------------------------------!
  1168)       woodloop: do ico=1,cpatch%ncohorts
  1169)          !----- Check whether the leaves of this cohort are safe... -----------------------!
  1170)          if (initp%wood_resolvable(ico)) then
  1171) 
  1172)             !----- Find the minimum leaf water for this cohort. ---------------------------!
  1173)             rk4min_wood_water = rk4min_veg_lwater * initp%wai(ico)
  1174)             !------------------------------------------------------------------------------!
  1175) 
  1176) 
  1177) 
  1178)             !------------------------------------------------------------------------------!
  1179)             !     Update wood temperature and liquid fraction only if wood water makes     !
  1180)             ! sense.                                                                       !
  1181)             !------------------------------------------------------------------------------!
  1182)             if (initp%wood_water(ico) < rk4min_wood_water) then
  1183)                ok_wood = .false.
  1184)                cycle woodloop
  1185)             else
  1186)                call uextcm2tl8(initp%wood_energy(ico),initp%wood_water(ico)                &
  1187)                               ,initp%wood_hcap(ico),initp%wood_temp(ico)                   &
  1188)                               ,initp%wood_fliq(ico))
  1189)                if (initp%wood_temp(ico) < rk4min_veg_temp .or.                             &
  1190)                    initp%wood_temp(ico) > rk4max_veg_temp) then
  1191)                   ok_wood = .false.
  1192)                   cycle woodloop
  1193)                end if
  1194)             end if
  1195)             !------------------------------------------------------------------------------!
  1196)          else
  1197)             !------------------------------------------------------------------------------!
  1198)             !     In case we are not solving branches, or for cohorts that are very        !
  1199)             ! sparse, fix the wood temperature to the canopy air space and force           !
  1200)             ! wood_water to be zero.                                                       !
  1201)             !------------------------------------------------------------------------------!
  1202)             initp%wood_temp(ico)   = initp%can_temp
  1203)             initp%wood_water(ico)  = 0.d0
  1204)             initp%wood_energy(ico) = cmtl2uext8( initp%wood_hcap (ico)                     &
  1205)                                                , initp%wood_water(ico)                     &
  1206)                                                , initp%wood_temp (ico)                     &
  1207)                                                , initp%wood_fliq (ico) )
  1208) 
  1209)             if (initp%wood_temp(ico) == t3ple8) then
  1210)                initp%wood_fliq(ico) = 5.d-1
  1211)             elseif (initp%wood_temp(ico) > t3ple8) then
  1212)                initp%wood_fliq(ico) = 1.d0
  1213)             else
  1214)                initp%wood_fliq(ico) = 0.d0
  1215)             end if
  1216)             !------------------------------------------------------------------------------!
  1217)          end if
  1218)       end do woodloop
  1219)       !------------------------------------------------------------------------------------!
  1220) 
  1221) 
  1222) 
  1223)       !------------------------------------------------------------------------------------!
  1224)       !     Update vegetation properties, so everything is neat and consistent.  This can  !
  1225)       ! be done for non-resolvable and unrealistic cohorts, it's just a sum and the time   !
  1226)       ! step is about to be rejected anyway.                                               ! 
  1227)       !------------------------------------------------------------------------------------!
  1228)       do ico=1,cpatch%ncohorts
  1229)          initp%veg_energy(ico) = initp%leaf_energy(ico) + initp%wood_energy(ico)
  1230)          initp%veg_water(ico)  = initp%leaf_water(ico)  + initp%wood_water(ico)
  1231)       end do
  1232)       !------------------------------------------------------------------------------------!
  1233)    end select
  1234)    !---------------------------------------------------------------------------------------!
  1235) 
  1236) 
  1237) 
  1238)    !----- Compute canopy turbulence properties. -------------------------------------------!
  1239)    if (ok_leaf .and. ok_wood .and. ok_shv .and. ok_theta .and. ok_ground) then
  1240)       call canopy_turbulence8(csite,initp,ipa)
  1241)    end if
  1242)    !---------------------------------------------------------------------------------------!
  1243) 
  1244) 
  1245)    return
  1246) end subroutine update_diagnostic_vars

  1342)    real(kind=8)                        :: depth_perc
  1343)    real(kind=8)                        :: i_energy_try
  1344)    real(kind=8)                        :: energy_try
  1345)    real(kind=8)                        :: wmass_try
  1346)    real(kind=8)                        :: depth_try
  1347)    real(kind=8)                        :: temp_try
  1348)    real(kind=8)                        :: fliq_try
  1349)    real(kind=8)                        :: energy_tot
  1350)    real(kind=8)                        :: wmass_tot
  1351)    real(kind=8)                        :: hcapdry_tot
  1352)    real(kind=8)                        :: wmass_room
  1353)    real(kind=8)                        :: energy_room
  1354)    real(kind=8)                        :: snden
  1355)    real(kind=8)                        :: sndenmin
  1356)    real(kind=8)                        :: sndenmax
  1357)    real(kind=8)                        :: Cr               ! snow waterholding capacity
  1358)    real(kind=8)                        :: gi               ! Partial density of ice
  1359)    integer                             :: nsoil
  1360)    !----- Variables used for the water and energy budget. ---------------------------------!
  1361)    real(kind=8)                        :: wmass_cas_beg
  1362)    real(kind=8)                        :: wmass_cas_end
  1363)    real(kind=8)                        :: enthalpy_cas_beg
  1364)    real(kind=8)                        :: enthalpy_cas_end
  1365)    real(kind=8)                        :: wmass_virtual_beg
  1366)    real(kind=8)                        :: energy_virtual_beg
  1367)    real(kind=8)                        :: wmass_sfcw_beg
  1368)    real(kind=8)                        :: energy_sfcw_beg
  1369)    real(kind=8)                        :: wmass_soil_beg
  1370)    real(kind=8)                        :: energy_soil_beg
  1371)    real(kind=8)                        :: wmass_total_beg
  1372)    real(kind=8)                        :: energy_total_beg
  1373)    real(kind=8)                        :: wmass_virtual_end
  1374)    real(kind=8)                        :: energy_virtual_end
  1375)    real(kind=8)                        :: wmass_sfcw_end
  1376)    real(kind=8)                        :: energy_sfcw_end
  1377)    real(kind=8)                        :: wmass_soil_end
  1378)    real(kind=8)                        :: energy_soil_end
  1379)    real(kind=8)                        :: wmass_total_end
  1380)    real(kind=8)                        :: energy_total_end
  1381)    real(kind=8)                        :: wmass_total_rch
  1382)    real(kind=8)                        :: energy_total_rch
  1383)    !----- Constants -----------------------------------------------------------------------!
  1384)    logical                , parameter  :: debug   = .false.
  1385)    real(kind=8)           , parameter  :: Crmin   = 3.d-2
  1386)    real(kind=8)           , parameter  :: Crmax   = 1.d-1
  1387)    real(kind=8)           , parameter  :: ge      = 2.d2
  1388)    !---------------------------------------------------------------------------------------!
  1389) 
  1390)    ibuff = 1
  1391)    !$ ibuff = OMP_get_thread_num()+1
  1392) 
  1393)    !----- Find the inverse of the time step. ----------------------------------------------!
  1394)    hdidi      = 1.d0 / hdid
  1395)    !---------------------------------------------------------------------------------------!
  1396) 
  1397) 
  1398)    !----- Copy the original number of temporary surface water layers to ksn. --------------!
  1399)    ksn       = initp%nlev_sfcwater
  1400)    !---------------------------------------------------------------------------------------!
  1401) 
  1402) 
  1403) 
  1404)    !----- Copy the soil type at the topmost level to nsoil. -------------------------------!
  1405)    nsoil     = rk4site%ntext_soil(nzg)
  1406)    !---------------------------------------------------------------------------------------!
  1407)    
  1408) 
  1409) 
  1410)    !---------------------------------------------------------------------------------------!
  1411)    !      Determine the total amount of temporary surface water available as well as       !
  1412)    ! derived properties.                                                                   !
  1413)    !---------------------------------------------------------------------------------------!
  1414)    sum_sfcw_mass   = 0.d0
  1415)    sum_sfcw_energy = 0.d0
  1416)    sum_sfcw_depth  = 0.d0
  1417)    do k=1,ksn
  1418)       sum_sfcw_mass   = sum_sfcw_mass   + initp%sfcwater_mass  (k)
  1419)       sum_sfcw_energy = sum_sfcw_energy + initp%sfcwater_energy(k)
  1420)       sum_sfcw_depth  = sum_sfcw_depth  + initp%sfcwater_depth (k)
  1421)    end do
  1422)    !---------------------------------------------------------------------------------------!
  1423) 
  1424) 
  1425) 
  1426)    !---------------------------------------------------------------------------------------!
  1427)    !      Initialise the budget variables.                                                 !
  1428)    !---------------------------------------------------------------------------------------!
  1429)    wmass_cas_beg      = initp%can_shv * rk4aux(ibuff)%wcapcan
  1430)    enthalpy_cas_beg   = initp%can_enthalpy * rk4aux(ibuff)%hcapcan
  1431)    wmass_virtual_beg  = initp%virtual_water
  1432)    energy_virtual_beg = initp%virtual_energy
  1433)    wmass_sfcw_beg     = sum_sfcw_mass
  1434)    energy_sfcw_beg    = sum_sfcw_energy
  1435)    wmass_soil_beg     = initp%soil_water(nzg)  * dslz8(nzg) * wdns8
  1436)    energy_soil_beg    = initp%soil_energy(nzg) * dslz8(nzg)
  1437)    wmass_total_beg    = wmass_virtual_beg  + wmass_sfcw_beg  + wmass_soil_beg              &
  1438)                       + wmass_cas_beg
  1439)    energy_total_beg   = energy_virtual_beg + energy_sfcw_beg + energy_soil_beg             &
  1440)                       + enthalpy_cas_beg
  1441)    !---------------------------------------------------------------------------------------!
  1442) 
  1443) 
  1444) 
  1445)    !---------------------------------------------------------------------------------------!
  1446)    !     Check the total amount of water that has just fallen plus the amount that is al-  !
  1447)    ! ready sitting over the top soil layer.  We must do this as the first step because we  !
  1448)    ! may want to eliminate this water by binding it to the top soil layer in case there is !
  1449)    ! too little water.                                                                     !
  1450)    !---------------------------------------------------------------------------------------!
  1451)    if (initp%virtual_water < rk4min_virt_water .or. sum_sfcw_mass < rk4min_sfcw_mass ) then
  1452)       !------------------------------------------------------------------------------------!
  1453)       !     Either the virtual layer or the temporary surface water has too negative mass, !
  1454)       ! so this step doesn't make sense.  We quit the sub-routine here so the sanity check !
  1455)       ! can reject this step.                                                              !
  1456)       !------------------------------------------------------------------------------------!
  1457)       return
  1458)    elseif ((initp%virtual_water + sum_sfcw_mass) < 0.d0) then
  1459)       !------------------------------------------------------------------------------------!
  1460)       !     The mass of the potential new temporary surface water is within bounds but it  !
  1461)       ! is negative.  This can only happen if the layer evaporated more than what it       !
  1462)       ! should, so we condense some of the water back from the canopy air space.  If it is !
  1463)       ! going to deplete the canopy air space specific humidity too much, then we leave    !
  1464)       ! the remainder to be stolen from the top soil, but this is dangerous because the    !
  1465)       ! soil may be too dry too.                                                           !
  1466)       !------------------------------------------------------------------------------------!
  1467)       wmass_needed  = - (initp%virtual_water  + sum_sfcw_mass  )
  1468)       energy_needed = - (initp%virtual_energy + sum_sfcw_energy)
  1469)       depth_needed  = - (initp%virtual_depth  + sum_sfcw_depth )
  1470)       !------------------------------------------------------------------------------------!
  1471) 
  1472) 
  1473) 
  1474)       !----- Find the amount available at the canopy air space. ---------------------------!
  1475)       wmass_available  = rk4aux(ibuff)%wcapcan * (initp%can_shv - 5.d0 * rk4min_can_shv)
  1476)       !------------------------------------------------------------------------------------!
  1477) 
  1478)       if ( wmass_available >= wmass_needed) then
  1479) 
  1480)          !---------------------------------------------------------------------------------!
  1481)          !     Find the latent heat associated with the phase change.                      !
  1482)          !---------------------------------------------------------------------------------!
  1483)          call uint2tl8(energy_needed/wmass_needed,tempk_needed,fracliq_needed)
  1484)          !----- Remove the energy. --------------------------------------------------------!
  1485)          energy_latent = wmass_needed * ( (1.d0 - fracliq_needed) * alvi8(tempk_needed)    &
  1486)                                         + fracliq_needed * alvl8(tempk_needed) )
  1487)          !---------------------------------------------------------------------------------!
  1488) 
  1489) 
  1490)          !---------------------------------------------------------------------------------!
  1491)          !    There is enough water vapour. The transfer will require phase change, so the !
  1492)          ! energy transfer will be a latent heat flux.  Remove the water from the canopy   !
  1493)          ! air space.  The energy lost by the canopy air space to pad the missing water at !
  1494)          ! the virtual+temporary surface water layer must go somewhere, so we add it to    !
  1495)          ! the soil because it is the closest to the pounding layer.                       !
  1496)          !---------------------------------------------------------------------------------!
  1497)          initp%can_shv          = initp%can_shv - wmass_needed * rk4aux(ibuff)%wcapcani
  1498)          initp%can_enthalpy     = initp%can_enthalpy                                       &
  1499)                                 - (energy_needed + energy_latent) * rk4aux(ibuff)%hcapcani
  1500)          initp%avg_vapor_gc     = initp%avg_vapor_gc - wmass_needed * hdidi
  1501)          initp%soil_energy(nzg) = initp%soil_energy(nzg)  + energy_latent * dslzi8(nzg)
  1502)          !---------------------------------------------------------------------------------!
  1503) 
  1504) 
  1505)          wmass_free  = 0.d0
  1506)          energy_free = 0.d0
  1507)          depth_free  = 0.d0
  1508) 
  1509)       elseif (wmass_available > 0.d0) then
  1510) 
  1511)          !---------------------------------------------------------------------------------!
  1512)          !     Find the latent heat associated with the phase change.                      !
  1513)          !---------------------------------------------------------------------------------!
  1514)          call uint2tl8(energy_needed/wmass_needed,tempk_needed,fracliq_needed)
  1515)          !----- Remove the energy. --------------------------------------------------------!
  1516)          energy_available = wmass_available * energy_needed / wmass_needed
  1517)          energy_latent    = wmass_available * ( (1.d0 - fracliq_needed)                    &
  1518)                                               * alvi8(tempk_needed)                        &
  1519)                                               + fracliq_needed * alvl8(tempk_needed) )
  1520)          ! Using frost density for the frozon component
  1521)          depth_available  = wmass_available *                                              &
  1522)               ( fracliq_needed * wdnsi8 + (1.d0-fracliq_needed * fdnsi8))
  1523)          !---------------------------------------------------------------------------------!
  1524) 
  1525) 
  1526)          !---------------------------------------------------------------------------------!
  1527)          !    There is not enough water vapour. Dry down to the minimum, and correct       !
  1528)          ! energy.  Since there is so little negative mass needed, we include the latent   !
  1529)          ! heat associated with this condensation to the soil, because otherwise we could  !
  1530)          ! end up with energy and water mass with opposite signs.                          !
  1531)          !---------------------------------------------------------------------------------!
  1532)          initp%can_shv          = initp%can_shv - wmass_available  * rk4aux(ibuff)%wcapcani
  1533)          initp%can_enthalpy     = initp%can_enthalpy                                       &
  1534)                                 - ( energy_available + energy_latent ) * rk4aux(ibuff)%hcapcani
  1535)          initp%avg_vapor_gc     = initp%avg_vapor_gc  - wmass_available * hdidi
  1536)          initp%soil_energy(nzg) = initp%soil_energy(nzg) + energy_latent * dslzi8(nzg)
  1537)          !---------------------------------------------------------------------------------!
  1538) 
  1539)          wmass_free         = wmass_available  - wmass_needed 
  1540)          energy_free        = energy_available - energy_needed
  1541)          depth_free         = depth_available  - depth_needed
  1542)       else
  1543)          !---------------------------------------------------------------------------------!
  1544)          !    There is not any water vapour.  Hope for the best.                           !
  1545)          !---------------------------------------------------------------------------------!
  1546)          wmass_free         = wmass_needed
  1547)          energy_free        = energy_needed
  1548)          depth_free         = depth_needed
  1549)          !---------------------------------------------------------------------------------!
  1550)       end if
  1551) 
  1552)       !----- Reset both the temporary surface water and the virtual layer. ----------------!
  1553)       initp%virtual_water      = 0.d0
  1554)       initp%virtual_energy     = 0.d0
  1555)       initp%virtual_depth      = 0.d0
  1556)       initp%sfcwater_mass  (:) = 0.d0
  1557)       initp%sfcwater_energy(:) = 0.d0
  1558)       initp%sfcwater_depth (:) = 0.d0
  1559)       !----- Set ksnnew to zero to force all free water to go to the soil. ----------------!
  1560)       ksnnew                   = 0
  1561)    elseif ((initp%virtual_water + sum_sfcw_mass) < rk4tiny_sfcw_mass) then
  1562)       !------------------------------------------------------------------------------------!
  1563)       !     The mass of the potential new temporary surface water is within bounds but it  !
  1564)       ! is too small to be maintained.  We add both the virtual mass and the total surface !
  1565)       ! water and dump in the free water, but set ksnnew to zero so all the water is       !
  1566)       ! infiltrated in the top soil layer.                                                 !
  1567)       !------------------------------------------------------------------------------------!
  1568)       wmass_free               = initp%virtual_water  + sum_sfcw_mass
  1569)       energy_free              = initp%virtual_energy + sum_sfcw_energy
  1570)       depth_free               = initp%virtual_depth  + sum_sfcw_depth
  1571)       !----- Reset both the temporary surface water and the virtual layer. ----------------!
  1572)       initp%virtual_water      = 0.d0
  1573)       initp%virtual_energy     = 0.d0
  1574)       initp%virtual_depth      = 0.d0
  1575)       initp%sfcwater_mass  (:) = 0.d0
  1576)       initp%sfcwater_energy(:) = 0.d0
  1577)       initp%sfcwater_depth (:) = 0.d0
  1578)       !----- Set ksnnew to zero to force all free water to go to the soil. ----------------!
  1579)       ksnnew                   = 0
  1580)    else
  1581)       !------------------------------------------------------------------------------------!
  1582)       !     The mass of the potential new temporary surface water is within bounds and     !
  1583)       ! could create at least one layer.  If there is already a temporary surface water or !
  1584)       ! snow layer, the new amount is initially put there, otherwise, we attempt to create !
  1585)       ! the first layer.                                                                   !
  1586)       !------------------------------------------------------------------------------------!
  1587)       wmass_free               = initp%virtual_water
  1588)       energy_free              = initp%virtual_energy
  1589)       depth_free               = initp%virtual_depth
  1590)       initp%virtual_water      = 0.d0
  1591)       initp%virtual_energy     = 0.d0
  1592)       initp%virtual_depth      = 0.d0
  1593)       ksnnew                   = max(ksn,1)
  1594)    end if
  1595)    !---------------------------------------------------------------------------------------!
  1596) 
  1597) 
  1598) 
  1599)    !---------------------------------------------------------------------------------------!
  1600)    !    Update the prognostic and diagnostic variables by adding the free standing water.  !
  1601)    ! Then we check the size of the temporary surface water layers, and update the          !
  1602)    ! temperature in a way that ensure the layer stability.  During this process, we        !
  1603)    ! update the total temporary surface water mass, energy, and depth, which will be used  !
  1604)    ! later in the sub-routine.                                                             !
  1605)    !---------------------------------------------------------------------------------------!
  1606)    sum_sfcw_mass   = 0.d0
  1607)    sum_sfcw_energy = 0.d0
  1608)    sum_sfcw_depth  = 0.d0
  1609)    do k = ksnnew,1,-1
  1610)       !------------------------------------------------------------------------------------!
  1611)       !    Find the potential mass, energy, and depth of the temporary layer if all the    !
  1612)       ! free water became part of this layer.                                              !
  1613)       !------------------------------------------------------------------------------------!
  1614)       energy_try = initp%sfcwater_energy(k) + energy_free
  1615)       wmass_try  = initp%sfcwater_mass(k)   + wmass_free
  1616)       depth_try  = initp%sfcwater_depth(k)  + depth_free
  1617)       !------------------------------------------------------------------------------------!
  1618) 
  1619) 
  1620) 
  1621)       !------------------------------------------------------------------------------------!
  1622)       !    In case this is a single layer, and a very thin one, we may have a hard time    !
  1623)       ! achieving numerical stability.  We can treat this case like the leaf case, in      !
  1624)       ! the sense that the water sitting on the top of the surface is in thermal           !
  1625)       ! equilibrium with the surface.                                                      !
  1626)       !------------------------------------------------------------------------------------!
  1627)       if (ksnnew == 1 .and. wmass_try < rk4water_stab_thresh) then
  1628)          !---------------------------------------------------------------------------------!
  1629)          !     Find the total internal energy of the combined pool (top soil layer plus    !
  1630)          ! the thin temporary surface water).  The units of soil properties are J/m3 for   !
  1631)          ! the internal energy, and m3/m3 for soil water, whilst the temporary surface     !
  1632)          ! water has units of J/m2 for internal energy and kg/m2 for mass.  We use the     !
  1633)          ! standard for the temporary surface water.                                       !
  1634)          !---------------------------------------------------------------------------------!
  1635)          energy_tot  = energy_try + initp%soil_energy(nzg) * dslz8(nzg)
  1636)          wmass_tot   = wmass_try  + initp%soil_water(nzg)  * dslz8(nzg) * wdns8
  1637)          hcapdry_tot = soil8(nsoil)%slcpd * dslz8(nzg)
  1638)          !---------------------------------------------------------------------------------!
  1639) 
  1640) 
  1641) 
  1642)          !---------------------------------------------------------------------------------!
  1643)          !      Find the equilibrium temperature and liquid/ice partition.   Because we    !
  1644)          ! are assuming thermal equilibrium, the temperature and liquid fraction of the    !
  1645)          ! attempted layer is the same as the average temperature of the augmented pool.   !
  1646)          !---------------------------------------------------------------------------------!
  1647)          call uextcm2tl8(energy_tot,wmass_tot,hcapdry_tot,temp_try,fliq_try)
  1648)          !---------------------------------------------------------------------------------!
  1649) 
  1650) 
  1651)          !---------------------------------------------------------------------------------!
  1652)          !    Re-compute the internal energy of the temporary layer, using the temperature !
  1653)          ! and fraction of liquid water distribution we have just found, keeping the mass  !
  1654)          ! constant.                                                                       !
  1655)          !---------------------------------------------------------------------------------!
  1656)          energy_try = wmass_try * tl2uint8(temp_try,fliq_try)
  1657)          !---------------------------------------------------------------------------------!
  1658) 
  1659) 
  1660) 
  1661)          !---------------------------------------------------------------------------------!
  1662)          !    Re-calculate the top soil internal energy, by removing the attempted surface !
  1663)          ! water energy from the total energy, and converting it back to J/m3.  The total  !
  1664)          ! amount of water does not need to be re-calculated at this time.                 !
  1665)          !---------------------------------------------------------------------------------!
  1666)          initp%soil_energy(nzg)  = (energy_tot - energy_try) * dslzi8(nzg)
  1667)          !---------------------------------------------------------------------------------!
  1668)       else
  1669)          !---------------------------------------------------------------------------------!
  1670)          !      Layer is computationally stable, find temperature and liquid fraction of   !
  1671)          ! the attempted layer.                                                            !
  1672)          !---------------------------------------------------------------------------------!
  1673)          i_energy_try = energy_try / wmass_try
  1674)          call uint2tl8(i_energy_try,temp_try,fliq_try)
  1675)         !---------------------------------------------------------------------------------!
  1676)       end if
  1677)       !------------------------------------------------------------------------------------!
  1678) 
  1679) 
  1680) 
  1681)       !------------------------------------------------------------------------------------!
  1682)       !     Determine a first guess for the amount of mass that can be lost from this      !
  1683)       ! layer through percolation (wmass_perc).                                            !
  1684)       !------------------------------------------------------------------------------------!
  1685)       select case (ipercol)
  1686)       case (0)
  1687)          !---------------------------------------------------------------------------------!
  1688)          !     Original method, from LEAF-3.  Shed liquid in excess of a 1:9               !
  1689)          ! liquid-to-ice ratio through percolation.                                        !
  1690)          !---------------------------------------------------------------------------------!
  1691)          wmass_perc  = max(0.d0, wmass_try * (fliq_try - 1.d-1) / 9.d-1)
  1692)          !---------------------------------------------------------------------------------!
  1693)       case (1,2)
  1694)          !---------------------------------------------------------------------------------!
  1695)          !    Alternative "free" water calculation.                                        !
  1696)          !    Anderson (1976), NOAA Tech Report NWS 19.                                    !
  1697)          !---------------------------------------------------------------------------------!
  1698)          gi          = wmass_try/max(rk4tiny_sfcw_depth,depth_try) * (1.d0 - fliq_try)
  1699)          Cr          = max(Crmin, Crmin + (Crmax - Crmin) * (ge - gi) / ge)
  1700)          wmass_perc  = max(0.d0,wmass_try * (fliq_try - Cr / (1.d0 + Cr)))
  1701)          !---------------------------------------------------------------------------------!
  1702)       end select
  1703)       !------------------------------------------------------------------------------------!
  1704) 
  1705) 
  1706) 
  1707)       !------------------------------------------------------------------------------------!
  1708)       !     Determinte whether the layer beneath the current one is another temporary      !
  1709)       ! surface water/snow layer, or the top soil layer.  In case it is the latter, we     !
  1710)       ! must check whether there is enough room for the percolate water to infiltrate      !
  1711)       ! (i.e., the soil will not become super-saturated), in which case we must reduce the !
  1712)       ! total amount of percolation.                                                       !
  1713)       !------------------------------------------------------------------------------------!
  1714)       if (k == 1) then
  1715)          !---------------------------------------------------------------------------------!
  1716)          !     Compute the available "room" for water at the top soil layer.  We must      !
  1717)          ! multiply by density and depth to make sure that the units match.                !
  1718)          !---------------------------------------------------------------------------------!
  1719)          wmass_room = max(0.d0, soil8(nsoil)%slmsts - initp%soil_water(nzg))               &
  1720)                     * wdns8 * dslz8(nzg) 
  1721)          wmass_perc = min(wmass_perc,wmass_room)
  1722)          !---------------------------------------------------------------------------------!
  1723)       end if
  1724)       !------------------------------------------------------------------------------------!
  1725) 
  1726) 
  1727) 
  1728) 
  1729) 
  1730)       !------------------------------------------------------------------------------------!
  1731)       !     Re-calculate the total water mass and energy of this temporary surface water.  !
  1732)       ! Here we must check whether the soil layer would be with too little mass, and if    !
  1733)       ! that is the case, we will eliminate the layer by forcing the tiny left-over to go  !
  1734)       ! to the layer beneath.                                                              !
  1735)       !------------------------------------------------------------------------------------!
  1736)       if (wmass_try - wmass_perc > rk4tiny_sfcw_mass) then
  1737)          !---------------------------------------------------------------------------------!
  1738)          !      Enough mass to keep this layer.                                            !
  1739)          !---------------------------------------------------------------------------------!
  1740)          !----- Compute the internal energy and depth associated with percolated water. ---!
  1741)          energy_perc = wmass_perc * tl2uint8(temp_try,1.d0)
  1742)          depth_perc  = wmass_perc * wdnsi8
  1743)          !----- Find the new water mass and energy for this layer. ------------------------!
  1744)          initp%sfcwater_mass  (k) = wmass_try  - wmass_perc
  1745)          initp%sfcwater_energy(k) = energy_try - energy_perc
  1746) 
  1747)          !---------------------------------------------------------------------------------!
  1748)          !      Calculate density and depth of snow.  Start with the difference of depths, !
  1749)          ! but then we adjust it because the loss through percolation changes the ratio    !
  1750)          ! between ice and liquid in this layer                                            !
  1751)          !---------------------------------------------------------------------------------!
  1752)          initp%sfcwater_depth (k) = depth_try  - depth_perc
  1753)          snden    = initp%sfcwater_mass(k)                                                 &
  1754)                   / max(rk4tiny_sfcw_depth,initp%sfcwater_depth(k))
  1755)          sndenmax = wdns8
  1756)          sndenmin = max(3.d1, 2.d2 * (wmass_free + wmass_perc) / initp%sfcwater_mass(k) )
  1757)          snden    = min(sndenmax, max(sndenmin,snden))
  1758)          initp%sfcwater_depth (k) = initp%sfcwater_mass(k) / snden
  1759)       else
  1760)          !---------------------------------------------------------------------------------!
  1761)          !      The layer would be too small, eliminate mass from this layer and send all  !
  1762)          ! mass to the layer beneath as percolated water.                                  !
  1763)          !---------------------------------------------------------------------------------!
  1764)          initp%sfcwater_mass  (k) = 0.d0
  1765)          initp%sfcwater_energy(k) = 0.d0
  1766)          initp%sfcwater_depth (k) = 0.d0
  1767)          wmass_perc               = wmass_try
  1768)          energy_perc              = energy_try
  1769)          depth_perc               = depth_try
  1770)       end if
  1771)       !------------------------------------------------------------------------------------!
  1772) 
  1773) 
  1774) 
  1775)       !------------------------------------------------------------------------------------!
  1776)       !     Integrate the total temporary surface water properties.                        !
  1777)       !------------------------------------------------------------------------------------!
  1778)       sum_sfcw_mass   = sum_sfcw_mass   + initp%sfcwater_mass  (k)
  1779)       sum_sfcw_energy = sum_sfcw_energy + initp%sfcwater_energy(k)
  1780)       sum_sfcw_depth  = sum_sfcw_depth  + initp%sfcwater_depth (k)
  1781)       !------------------------------------------------------------------------------------!
  1782) 
  1783) 
  1784) 
  1785)       !------------------------------------------------------------------------------------!
  1786)       !     The water available for the layer beneath is going to be the total percolated  !
  1787)       ! water.                                                                             !
  1788)       !------------------------------------------------------------------------------------!
  1789)       wmass_free  = wmass_perc
  1790)       energy_free = energy_perc
  1791)       depth_free  = depth_perc
  1792)       !------------------------------------------------------------------------------------!
  1793)    end do
  1794)    !---------------------------------------------------------------------------------------!
  1795) 
  1796) 
  1797) 
  1798)    !---------------------------------------------------------------------------------------!
  1799)    !     There may be a tiny amount of free standing water left.  We dump what we can in   !
  1800)    ! the soil, and if there is still some water to be removed we  evaporate what is left.  !
  1801)    !---------------------------------------------------------------------------------------!
  1802)    if (wmass_free > 0.d0) then
  1803)       wmass_room  = max(0.d0, soil8(nsoil)%slmsts - initp%soil_water(nzg))                 &
  1804)                   * wdns8 * dslz8(nzg)
  1805)       energy_room = energy_free * wmass_room / wmass_free
  1806) 
  1807)       if (wmass_room >= wmass_free) then
  1808)          !---------------------------------------------------------------------------------!
  1809)          !     There is enough space in the top soil layer for the remaining water, put    !
  1810)          ! all the free water there.                                                       !
  1811)          !---------------------------------------------------------------------------------!
  1812)          initp%soil_water(nzg)  = initp%soil_water(nzg)  + wmass_free  * dslzi8(nzg)       &
  1813)                                 * wdnsi8
  1814)          initp%soil_energy(nzg) = initp%soil_energy(nzg) + energy_free * dslzi8(nzg)
  1815) 
  1816)          wmass_free  = 0.d0
  1817)          energy_free = 0.d0
  1818)          depth_free  = 0.d0
  1819)       else
  1820)          !---------------------------------------------------------------------------------!
  1821)          !     There is not enough space in the top soil layer for the remaining water,    !
  1822)          ! put what we can there, and boil the remaining.                                  !
  1823)          !---------------------------------------------------------------------------------!
  1824) 
  1825) 
  1826)          !----- Remove the water that can go to the soil. ---------------------------------!
  1827)          wmass_free  = wmass_free  - wmass_room
  1828)          energy_free = energy_free - energy_room
  1829)          !---------------------------------------------------------------------------------!
  1830) 
  1831) 
  1832)          !----- Find the amount of latent heat associated with boiling. -------------------!
  1833)          call uint2tl8(energy_free/wmass_free,tempk_free,fracliq_free)
  1834)          energy_latent = wmass_free * ( (1.d0 - fracliq_free) * alvi8(tempk_free)          &
  1835)                                       + fracliq_free * alvl8(tempk_free) )
  1836)          !---------------------------------------------------------------------------------!
  1837) 
  1838) 
  1839) 
  1840)          !---------------------------------------------------------------------------------!
  1841)          !     Dump what we can dump on the top soil layer.  Since no energy will be left  !
  1842)          ! in the free layer, we must get the energy for latent heat from somewhere, and   !
  1843)          ! we take it from the top most soil layer.                                        !
  1844)          !---------------------------------------------------------------------------------!
  1845)          initp%soil_water(nzg)  = initp%soil_water(nzg)  + wmass_room  * dslzi8(nzg)       &
  1846)                                 * wdnsi8
  1847)          initp%soil_energy(nzg) = initp%soil_energy(nzg)                                   &
  1848)                                 + ( energy_room - energy_latent ) * dslzi8(nzg)
  1849)          !---------------------------------------------------------------------------------!
  1850) 
  1851) 
  1852)          !---------------------------------------------------------------------------------!
  1853)          !     Boil the remaining.                                                         !
  1854)          !---------------------------------------------------------------------------------!
  1855)          !------ Update the canopy air space properties. ----------------------------------!
  1856)          initp%can_shv      = initp%can_shv       + wmass_free   * rk4aux(ibuff)%wcapcani
  1857)          initp%can_enthalpy = initp%can_enthalpy                                           &
  1858)                             + ( energy_free + energy_latent )    * rk4aux(ibuff)%hcapcani
  1859)          !---------------------------------------------------------------------------------!
  1860) 
  1861) 
  1862)          !------ Update the fluxes. -------------------------------------------------------!
  1863)          initp%avg_vapor_gc = initp%avg_vapor_gc  + wmass_free   * hdidi
  1864)          !---------------------------------------------------------------------------------!
  1865) 
  1866)          wmass_free  = 0.d0
  1867)          energy_free = 0.d0
  1868)          depth_free  = 0.d0
  1869)       end if
  1870)    elseif (wmass_free < 0.d0) then
  1871)       wmass_needed     = - wmass_free
  1872)       energy_needed    = - energy_free
  1873)       depth_needed     = - depth_free
  1874) 
  1875)       !------ Find the amount of water that the soil can provide. -------------------------!
  1876)       wmass_available  = max(0.d0,initp%soil_water(nzg) - soil8(nsoil)%soilcp)             &
  1877)                        * wdns8 * dslz8(nzg)
  1878)       energy_available = energy_free * wmass_available / wmass_free
  1879) 
  1880)       if (wmass_available >= wmass_needed) then
  1881)          !---------------------------------------------------------------------------------!
  1882)          !     There is enough space in the top soil layer to correct remaining negative   !
  1883)          ! water, get all the water needed there.                                          !
  1884)          !---------------------------------------------------------------------------------!
  1885)          initp%soil_water(nzg)  = initp%soil_water(nzg)  - wmass_needed  * dslzi8(nzg)     &
  1886)                                 * wdnsi8
  1887)          initp%soil_energy(nzg) = initp%soil_energy(nzg) - energy_needed * dslzi8(nzg)
  1888)          wmass_needed     = 0.d0
  1889)          energy_needed    = 0.d0
  1890)          depth_needed     = 0.d0
  1891)       else
  1892)          !---------------------------------------------------------------------------------!
  1893)          !     There is not enough space in the top soil layer to correct remaining        !
  1894)          ! negative water, get all the water we can from the top soil and condense the     !
  1895)          ! remaining.                                                                      !
  1896)          !---------------------------------------------------------------------------------!
  1897) 
  1898) 
  1899)          !----- Add the water that can come from the soil. --------------------------------!
  1900)          wmass_needed  = wmass_needed  - wmass_available
  1901)          energy_needed = energy_needed - energy_available
  1902)          !---------------------------------------------------------------------------------!
  1903) 
  1904) 
  1905)          !----- Find the amount of latent heat associated with condensation. --------------!
  1906)          call uint2tl8(energy_needed/wmass_needed,tempk_needed,fracliq_needed)
  1907)          energy_latent = wmass_needed * ( (1.d0 - fracliq_needed) * alvi8(tempk_needed)    &
  1908)                                         + fracliq_needed * alvl8(tempk_needed) )
  1909)          !---------------------------------------------------------------------------------!
  1910) 
  1911) 
  1912)          !----- Dump what we can dump on the top soil layer. ------------------------------!
  1913)          initp%soil_water(nzg)  = initp%soil_water(nzg)  - wmass_available  * dslzi8(nzg)  &
  1914)                                 * wdnsi8
  1915)          initp%soil_energy(nzg) = initp%soil_energy(nzg)                                   &
  1916)                                 - ( energy_available - energy_latent) * dslzi8(nzg)
  1917)          !---------------------------------------------------------------------------------!
  1918) 
  1919) 
  1920)          !---------------------------------------------------------------------------------!
  1921)          !     Condense the remaining, and hope for the best.                              !
  1922)          !---------------------------------------------------------------------------------!
  1923)          !----- Update the canopy air space properties. -----------------------------------!
  1924)          initp%can_shv      = initp%can_shv       - wmass_needed   * rk4aux(ibuff)%wcapcani
  1925)          initp%can_enthalpy = initp%can_enthalpy                                           &
  1926)                             - ( energy_needed + energy_latent ) * rk4aux(ibuff)%hcapcani
  1927)          !----- Update the fluxes. --------------------------------------------------------!
  1928)          initp%avg_vapor_gc = initp%avg_vapor_gc  - wmass_needed   * hdidi
  1929)          !---------------------------------------------------------------------------------!
  1930) 
  1931)          wmass_free  = 0.d0
  1932)          energy_free = 0.d0
  1933)          depth_free  = 0.d0
  1934)          !---------------------------------------------------------------------------------!
  1935) 
  1936)       end if
  1937)       !------------------------------------------------------------------------------------!
  1938)    end if
  1939)    !---------------------------------------------------------------------------------------!
  1940) 
  1941) 
  1942)    !---------------------------------------------------------------------------------------!
  1943)    !     Check the total amount of mass in the temporary surface water/snow, and adjust    !
  1944)    ! the number of layer accordingly.                                                      !
  1945)    !---------------------------------------------------------------------------------------!
  1946)    if (sum_sfcw_mass <= rk4tiny_sfcw_mass) then
  1947)       !----- Not enough water in the temporary surface water, eliminate all layers. -------!
  1948)       initp%nlev_sfcwater = 0
  1949)       !------------------------------------------------------------------------------------!
  1950) 
  1951) 
  1952)       !----- Update the flag for temporary surface water. ---------------------------------!
  1953)       initp%flag_sfcwater = 0
  1954)       !------------------------------------------------------------------------------------!
  1955) 
  1956) 
  1957) 
  1958)       !------------------------------------------------------------------------------------!
  1959)       !      The total mass should be either zero or greater than rk4tiny_sfcw_mass,       !
  1960)       ! but, just in case, we add any remaining energy to the top soil layer.              !
  1961)       !------------------------------------------------------------------------------------!
  1962)       initp%soil_water(nzg)  = initp%soil_water(nzg)  + sum_sfcw_mass   * dslzi8(nzg)      &
  1963)                                                       * wdnsi8
  1964)       initp%soil_energy(nzg) = initp%soil_energy(nzg) + sum_sfcw_energy * dslzi8(nzg)
  1965)       !------------------------------------------------------------------------------------!
  1966) 
  1967)       !----- Loop all layers and re-set all extensive variables to zero. ------------------!
  1968)       do k = 1, nzs
  1969)          initp%sfcwater_mass(k)    = 0.d0
  1970)          initp%sfcwater_energy(k)  = 0.d0
  1971)          initp%sfcwater_depth(k)   = 0.d0
  1972)       end do
  1973)       !------------------------------------------------------------------------------------!
  1974)    elseif (sum_sfcw_mass < rk4water_stab_thresh) then
  1975) 
  1976) 
  1977)       !----- Not much water in the temporary surface water, impose a single layer. --------!
  1978)       initp%nlev_sfcwater = 1
  1979)       !------------------------------------------------------------------------------------!
  1980) 
  1981) 
  1982)       !----- Update the flag for temporary surface water. ---------------------------------!
  1983)       initp%flag_sfcwater = 1
  1984)       !------------------------------------------------------------------------------------!
  1985) 
  1986) 
  1987)       !------------------------------------------------------------------------------------!
  1988)       !    If the total amount of temporary surface water is not enough to make it stable, !
  1989)       ! we impose it to have a single layer with all the ponding/snow in there.            !
  1990)       !------------------------------------------------------------------------------------!
  1991)       initp%sfcwater_mass  (1) = sum_sfcw_mass
  1992)       initp%sfcwater_energy(1) = sum_sfcw_energy
  1993)       initp%sfcwater_depth (1) = sum_sfcw_depth
  1994)       do k=2,nzs
  1995)          initp%sfcwater_mass   (k) = 0.d0
  1996)          initp%sfcwater_energy (k) = 0.d0
  1997)          initp%sfcwater_depth  (k) = 0.d0
  1998)       end do
  1999)       !------------------------------------------------------------------------------------!
  2000) 
  2001) 
  2002)    else
  2003)       !----- Update the flag for temporary surface water. ---------------------------------!
  2004)       initp%flag_sfcwater = 2
  2005)       !------------------------------------------------------------------------------------!
  2006) 
  2007) 
  2008)       !---- Check whether there is enough snow for a new layer. ---------------------------!
  2009)       nlayers   = ksnnew
  2010)       newlayers = 1
  2011)       do k = 1,ksnnew
  2012)          !---------------------------------------------------------------------------------!
  2013)          !     Check whether the layer as is meet the minimum requirements to stand as a   !
  2014)          ! new layer by itself.                                                            !
  2015)          !---------------------------------------------------------------------------------!
  2016) !         if ( initp%sfcwater_mass(k)   >  rk4tiny_sfcw_mass              .and.             &
  2017) !              rk4snowmin * thicknet(k) <= sum_sfcw_mass                  .and.             &
  2018) !              initp%sfcwater_energy(k) <  initp%sfcwater_mass(k)*uiliqt38      ) then
  2019)          if ( initp%sfcwater_mass(k)   >=  rk4snowmin		             .and.             &
  2020)               rk4snowmin * fsdnsi8 <= initp%sfcwater_depth(k)            .and.             &
  2021)               initp%sfcwater_energy(k) <  initp%sfcwater_mass(k)*uiliqt38      ) then
  2022)             newlayers = newlayers + 1
  2023)          end if
  2024)          !---------------------------------------------------------------------------------!
  2025)       end do
  2026) 
  2027)       !----- Newlayers is the new number of temporary surface water layers. ---------------!
  2028)       newlayers = min(newlayers, nzs, nlayers + 1)
  2029)       
  2030)       if (newlayers == 1) then
  2031)          newsfcw_mass  (1) = sum_sfcw_mass
  2032)          newsfcw_energy(1) = sum_sfcw_energy
  2033)          newsfcw_depth (1) = sum_sfcw_depth
  2034)       else
  2035)          kold  = 1
  2036)          wtnew = 1.d0
  2037)          wtold = 1.d0
  2038)          do k = 1,newlayers
  2039)             newsfcw_mass(k)   = sum_sfcw_mass * thick(k,newlayers)
  2040)             newsfcw_energy(k) = 0.d0
  2041)             newsfcw_depth(k)  = 0.d0
  2042)             !----- Find the properties of this new layer. ---------------------------------!
  2043)             find_layer: do
  2044) 
  2045)                !----- Difference between old and new snow ---------------------------------!
  2046)                wdiff = wtnew * newsfcw_mass(k) - wtold * initp%sfcwater_mass(kold)  
  2047) 
  2048)                if (wdiff > 0.d0) then
  2049)                   newsfcw_energy(k) = newsfcw_energy(k)                                    &
  2050)                                     + wtold * initp%sfcwater_energy(kold)
  2051)                   newsfcw_depth(k)  = newsfcw_depth(k)                                     &
  2052)                                     + wtold * initp%sfcwater_depth(kold)
  2053)                   wtnew  = wtnew - wtold * initp%sfcwater_mass(kold) / newsfcw_mass(k)
  2054)                   kold   = kold + 1
  2055)                   wtold  = 1.0
  2056)                   if (kold > nlayers) exit find_layer
  2057)                else
  2058)                   newsfcw_energy(k) = newsfcw_energy(k) + wtnew * newsfcw_mass(k)             &
  2059)                                     * initp%sfcwater_energy(kold)                             &
  2060)                                     / max(rk4tiny_sfcw_mass,initp%sfcwater_mass(kold))
  2061)                   newsfcw_depth(k)  = newsfcw_depth(k)  + wtnew * newsfcw_mass(k)             &
  2062)                                     * initp%sfcwater_depth(kold)                              &
  2063)                                     / max(rk4tiny_sfcw_mass,initp%sfcwater_mass(kold))
  2064)                   wtold = wtold - wtnew * newsfcw_mass(k)                                     &
  2065)                                 / max(rk4tiny_sfcw_mass,initp%sfcwater_mass(kold))
  2066)                   wtnew = 1.
  2067)                   exit find_layer
  2068)                end if
  2069)             end do find_layer
  2070)          end do
  2071)       end if
  2072) 
  2073)       !----- Update the water/snow layer prognostic properties. ---------------------------!
  2074)       initp%nlev_sfcwater = newlayers
  2075)       do k = 1,newlayers
  2076)          initp%sfcwater_mass(k)   = newsfcw_mass(k)
  2077)          initp%sfcwater_energy(k) = newsfcw_energy(k)
  2078)          initp%sfcwater_depth(k)  = newsfcw_depth(k)
  2079)       end do
  2080)       do k = newlayers + 1, nzs
  2081)          initp%sfcwater_mass   (k) = 0.d0
  2082)          initp%sfcwater_energy (k) = 0.d0
  2083)          initp%sfcwater_depth  (k) = 0.d0
  2084)       end do
  2085)    end if
  2086)    !---------------------------------------------------------------------------------------!
  2087) 
  2088) 
  2089)    !---------------------------------------------------------------------------------------!
  2090)    !      Compute the budget variables after the adjustments.                              !
  2091)    !---------------------------------------------------------------------------------------!
  2092)    wmass_cas_end      = initp%can_shv      * rk4aux(ibuff)%wcapcan
  2093)    enthalpy_cas_end   = initp%can_enthalpy * rk4aux(ibuff)%hcapcan
  2094)    wmass_virtual_end  = initp%virtual_water
  2095)    energy_virtual_end = initp%virtual_energy
  2096)    wmass_sfcw_end     = sum_sfcw_mass
  2097)    energy_sfcw_end    = sum_sfcw_energy
  2098)    wmass_soil_end     = initp%soil_water(nzg)  * dslz8(nzg) * wdns8
  2099)    energy_soil_end    = initp%soil_energy(nzg) * dslz8(nzg)
  2100)    wmass_total_end    = wmass_virtual_end  + wmass_sfcw_end  + wmass_soil_end              &
  2101)                       + wmass_cas_end
  2102)    energy_total_end   = energy_virtual_end + energy_sfcw_end + energy_soil_end             &
  2103)                       + enthalpy_cas_end
  2104)    !---------------------------------------------------------------------------------------!
  2105) 
  2106) 
  2107) 
  2108)    !---------------------------------------------------------------------------------------!
  2109)    !      Check whether energy and mass are conserved.                                     !
  2110)    !---------------------------------------------------------------------------------------!
  2111)    wmass_total_rch  = 2.d0 * abs(wmass_total_end  - wmass_total_beg)                       &
  2112)                     / (abs(wmass_total_end ) + abs(wmass_total_beg ))
  2113)    energy_total_rch = 2.d0 * abs(energy_total_end - energy_total_beg)                      &
  2114)                     / (abs(energy_total_end) + abs(energy_total_beg))
  2115)    if (wmass_total_rch > 1.d-6 .or. energy_total_rch > 1.d-6) then
  2116)       write (unit=*,fmt='(a)')           '------------------------------------------------'
  2117)       write (unit=*,fmt='(a)')           ' Water or energy conservation was violated!!!   '
  2118)       write (unit=*,fmt='(a)')           '------------------------------------------------'
  2119)       write (unit=*,fmt='(a)')           ' '
  2120)       write (unit=*,fmt='(a)')           ' - Initial conditions: '
  2121)       write (unit=*,fmt='(a,1x,es14.7)') '   + Total water mass    = ',wmass_total_beg
  2122)       write (unit=*,fmt='(a,1x,es14.7)') '   + CAS mass            = ',wmass_cas_beg
  2123)       write (unit=*,fmt='(a,1x,es14.7)') '   + Virtual mass        = ',wmass_virtual_beg
  2124)       write (unit=*,fmt='(a,1x,es14.7)') '   + Ponding/snow mass   = ',wmass_sfcw_beg
  2125)       write (unit=*,fmt='(a,1x,es14.7)') '   + Soil mass           = ',wmass_soil_beg
  2126)       write (unit=*,fmt='(a,1x,es14.7)') '   + Total energy        = ',energy_total_beg
  2127)       write (unit=*,fmt='(a,1x,es14.7)') '   + CAS enthalpy        = ',enthalpy_cas_beg
  2128)       write (unit=*,fmt='(a,1x,es14.7)') '   + Virtual energy      = ',energy_virtual_beg
  2129)       write (unit=*,fmt='(a,1x,es14.7)') '   + Ponding/snow energy = ',energy_sfcw_beg
  2130)       write (unit=*,fmt='(a,1x,es14.7)') '   + Soil energy         = ',energy_soil_beg
  2131)       write (unit=*,fmt='(a)')           ' '
  2132)       write (unit=*,fmt='(a)')           ' - Final conditions: '
  2133)       write (unit=*,fmt='(a,1x,es14.7)') '   + Total water mass    = ',wmass_total_end
  2134)       write (unit=*,fmt='(a,1x,es14.7)') '   + CAS mass            = ',wmass_cas_end
  2135)       write (unit=*,fmt='(a,1x,es14.7)') '   + Virtual mass        = ',wmass_virtual_end
  2136)       write (unit=*,fmt='(a,1x,es14.7)') '   + Ponding/snow mass   = ',wmass_sfcw_end
  2137)       write (unit=*,fmt='(a,1x,es14.7)') '   + Soil mass           = ',wmass_soil_end
  2138)       write (unit=*,fmt='(a,1x,es14.7)') '   + Total energy        = ',energy_total_end
  2139)       write (unit=*,fmt='(a,1x,es14.7)') '   + CAS enthalpy        = ',enthalpy_cas_end
  2140)       write (unit=*,fmt='(a,1x,es14.7)') '   + Virtual energy      = ',energy_virtual_end
  2141)       write (unit=*,fmt='(a,1x,es14.7)') '   + Ponding/snow energy = ',energy_sfcw_end
  2142)       write (unit=*,fmt='(a,1x,es14.7)') '   + Soil energy         = ',energy_soil_end
  2143)       write (unit=*,fmt='(a)')           ' '
  2144)       write (unit=*,fmt='(a)')           ' - Relative error: '
  2145)       write (unit=*,fmt='(a,1x,es14.7)') '   + Total water mass    = ',wmass_total_rch
  2146)       write (unit=*,fmt='(a,1x,es14.7)') '   + Total energy        = ',energy_total_rch
  2147)       write (unit=*,fmt='(a)')           ' '
  2148)       write (unit=*,fmt='(a)')           '------------------------------------------------'
  2149)       call fatal_error('Energy or water is not being conserved!!!'                         &
  2150)                       ,'adjust_sfcw_properties','rk4_misc.f90')
  2151)    end if
  2152) 
  2153)    return

  2165) !     This subroutine will ensure that top soil properties are within the allowed range.   !
  2166) ! We currently test this only for the top soil layer because this is the most likely to    !
  2167) ! cause problems, but if it does happen in other layers, we could easily extend this for   !

  2174) !                                                                                          !

  2356)       elseif (water_available > 0.d0) then
  2357)          !---------------------------------------------------------------------------------!
  2358)          !     This water will not be enough, but we use it and seek for other sources to  !

  2368)          initp%virtual_energy  = 0.d0
  2369)          initp%virtual_depth   = 0.d0

  2373)       !    If we hit this point the temporary layers were not enough.  We now look for     !

  2379)       nsbeneath       = rk4site%ntext_soil(kb)

  2385) 
  2386) 
  2387)

  2393)          energy_needed = water_needed                                                      &

  2396) 
  2397)

  2404)

  2421)          water_needed = water_needed - water_available

  2423) 
  2424) 
  2425)

  2562)          initp%virtual_depth      = initp%virtual_depth     + depth_excess

  2564) 
  2565)          return
  2566)       end if

  2577)       !------------------------------------------------------------------------------------!
  2578)       if (initp%virtual_water > rk4eps * rk4eps * rk4tiny_sfcw_mass) then

  2584)          !---------------------------------------------------------------------------------!
  2585)          call uint2tl8(initp%virtual_energy/initp%virtual_water                            &
  2586)                       ,virtual_tempk,virtual_fracliq)
  2587)          virtual_latent = initp%virtual_water                                              &
  2588)                         * ( (1.d0 - virtual_fracliq) * alvi8(virtual_tempk)                &
  2589)                           + virtual_fracliq * alvl8(virtual_tempk) )
  2590)          !---------------------------------------------------------------------------------!
  2591) 
  2592)

  2594)          initp%can_shv          = initp%can_shv      + initp%virtual_water  * rk4aux(ibuff)%wcapcani
  2595)          initp%can_enthalpy     = initp%can_enthalpy                                       &
  2596)                                 + ( initp%virtual_energy + virtual_latent ) * rk4aux(ibuff)%hcapcani
  2597)          initp%soil_energy(nzg) = initp%soil_energy(nzg) - virtual_latent * dslz8(nzg)
  2598)          initp%avg_vapor_gc     = initp%avg_vapor_gc + initp%virtual_water  * hdidi
  2599)          !---------------------------------------------------------------------------------!
  2600) 
  2601)          
  2602)          !----- Say goodbye to the virtual layer... ---------------------------------------!
  2603)          initp%virtual_energy = 0.d0
  2604)          initp%virtual_water  = 0.d0
  2605)          initp%virtual_depth  = 0.d0
  2606)          !---------------------------------------------------------------------------------!
  2607)       elseif (initp%virtual_water > 0.d0) then
  2608)          !---------------------------------------------------------------------------------!
  2609)          !    The amount of water is so small that round-off errors are bound to become    !
  2610)          ! more important than the error of not conserving energy and water.  Simply       !
  2611)          ! extinguish the virtual layer.                                                   !
  2612)          !---------------------------------------------------------------------------------!
  2613)          initp%virtual_energy = 0.d0
  2614)          initp%virtual_water  = 0.d0
  2615)          initp%virtual_depth  = 0.d0
  2616)       endif 
  2617) 
  2618)       !------------------------------------------------------------------------------------!
  2619)       !     Back to the top soil layer, we still need to decide where we should send the   !
  2620)       ! excess...  First we try to send to the layer beneath.  This transfer of water      !
  2621)       ! implies that some energy is also transferred. Since soil layers are not pure       !
  2622)       !  water, we must find the actual amount of energy associated with the water         !
  2623)       ! transfer.  So first we compute the temperature and liquid fraction of the layer    !
  2624)       ! beneath the top.                                                                   !
  2625)       !------------------------------------------------------------------------------------!
  2626)       kb         = nzg-1
  2627)       nsbeneath  = rk4site%ntext_soil(kb)
  2628)       shcbeneath = soil8(nsbeneath)%slcpd
  2629)       call uextcm2tl8(initp%soil_energy(kb),initp%soil_water(kb)*wdns8,shcbeneath          &
  2630)                      ,initp%soil_tempk(kb),initp%soil_fracliq(kb))
  2631)       water_room = (soil8(nsbeneath)%slmsts-initp%soil_water(kb)) * wdns8 * dslz8(kb)
  2632) 
  2633)       if (water_room > water_excess) then
  2634)          !---------------------------------------------------------------------------------!
  2635)          !    The layer beneath still has some room for this water excess, send the water  !
  2636)          ! down one level.                                                                 !
  2637)          !---------------------------------------------------------------------------------!
  2638)          initp%soil_water(kt)  = initp%soil_water(kt)  - water_excess  * dslzi8(kt)*wdnsi8
  2639)          initp%soil_energy(kt) = initp%soil_energy(kt) - energy_excess * dslzi8(kt)
  2640)          initp%soil_water(kb)  = initp%soil_water(kb)  + water_excess  * dslzi8(kb)*wdnsi8
  2641)          initp%soil_energy(kb) = initp%soil_energy(kb) + energy_excess * dslzi8(kb)
  2642)          !----- Update the fluxes too... --------------------------------------------------!
  2643)          initp%avg_smoist_gg(kt) = initp%avg_smoist_gg(kt) - water_excess * hdidi
  2644)          initp%avg_smoist_gg(kb) = initp%avg_smoist_gg(kb) + water_excess * hdidi
  2645)          !---------------------------------------------------------------------------------!
  2646) 
  2647)          return
  2648)       elseif (water_room > 0.d0) then
  2649)          !---------------------------------------------------------------------------------!
  2650)          !   The layer beneath the top layer can't take all water, but we send all that we !
  2651)          ! can to that layer so we reduce the amount that will be "boiled".  The remaining !
  2652)          ! will go to the canopy air space.  Even if some supersaturation happens, the     !
  2653)          ! excess won't be too much to cause the run to crash.                             !
  2654)          !---------------------------------------------------------------------------------!
  2655)          water_excess = water_excess - water_room
  2656)          energy_room  = water_room * tl2uint8(initp%soil_tempk(kt),initp%soil_fracliq(kt))
  2657)          !---------------------------------------------------------------------------------!
  2658) 
  2659) 
  2660) 
  2661)          !---------------------------------------------------------------------------------!
  2662)          !    The layer beneath still has some room for this water excess, send the water  !
  2663)          ! down one level.                                                                 !
  2664)          !---------------------------------------------------------------------------------!
  2665)          initp%soil_water (kt) = initp%soil_water (kt) - water_room  * dslzi8(kt) * wdnsi8
  2666)          initp%soil_energy(kt) = initp%soil_energy(kt) - energy_room * dslzi8(kt)
  2667)          initp%soil_water (kb) = initp%soil_water (kb) + water_room  * dslzi8(kb) * wdnsi8
  2668)          initp%soil_energy(kb) = initp%soil_energy(kb) + energy_room * dslzi8(kb)
  2669)          !----- Update the fluxes too... --------------------------------------------------!
  2670)          initp%avg_smoist_gg(kt) = initp%avg_smoist_gg(kt) - water_room * hdidi
  2671)          initp%avg_smoist_gg(kb) = initp%avg_smoist_gg(kb) + water_room * hdidi
  2672)          !---------------------------------------------------------------------------------!
  2673) 
  2674) 
  2675) 
  2676)          !---------------------------------------------------------------------------------!
  2677)          !     The water that is about to leave will do it through "boiling".  Find the    !
  2678)          ! latent heat associated with this phase change.                                  !
  2679)          !---------------------------------------------------------------------------------!
  2680)          energy_excess = water_excess * tq2enthalpy8(initp%soil_tempk(kt),1.d0,.true.)
  2681)          !---------------------------------------------------------------------------------!
  2682) 
  2683) 
  2684) 
  2685)          !----- Send the water and energy to the canopy. ----------------------------------!
  2686)          initp%soil_water(kt)  = initp%soil_water(kt)  - water_excess  * dslzi8(kt)*wdnsi8
  2687)          initp%soil_energy(kt) = initp%soil_energy(kt) - energy_excess * dslzi8(kt)
  2688)          initp%can_shv         = initp%can_shv         + water_excess  * rk4aux(ibuff)%wcapcani
  2689)          initp%can_enthalpy    = initp%can_enthalpy    + energy_excess * rk4aux(ibuff)%hcapcani
  2690)          initp%avg_vapor_gc    = initp%avg_vapor_gc    + water_excess  * hdidi
  2691)          !---------------------------------------------------------------------------------!
  2692)       end if
  2693)       !------------------------------------------------------------------------------------!
  2694)    end if
  2695)    !---------------------------------------------------------------------------------------!
  2696) 
  2697)    return

  3000)             !------------------------------------------------------------------------------!

  3035) 
  3036)    !---------------------------------------------------------------------------------------!
  3037)    !    The water that fell from the leaves and branches must go somewhere...  Here we     !
  3038)    ! decide which place is the most suitable.  In case there is already a temporary        !
  3039)    ! surface water layer we can add the water there, otherwise we dump it into the virtual !
  3040)    ! layer, which may or may not become a temporary surface water layer.                   !
  3041)    !---------------------------------------------------------------------------------------!
  3042)    ksn = initp%nlev_sfcwater

  3057)                                  + leaf_wshed_tot  + wood_wshed_tot

  3059)                                  + leaf_qwshed_tot + wood_qwshed_tot

  3061)                                  + leaf_dwshed_tot + wood_dwshed_tot
  3062)       !------------------------------------------------------------------------------------!
  3063) 
  3064)    end select

  3066) 
  3067) 
  3068) 
  3069)    !----- Update the canopy air specific humidity and enthalpy. ---------------------------!
  3070)    initp%can_shv      = initp%can_shv                                                      &
  3071)                       + (leaf_boil_tot  + wood_boil_tot  - leaf_dew_tot  - wood_dew_tot)   &
  3072)                       * rk4aux(ibuff)%wcapcani
  3073)    initp%can_enthalpy = initp%can_enthalpy                                                 &

  3075)                       * rk4aux(ibuff)%hcapcani
  3076)    !---------------------------------------------------------------------------------------!
  3077)
