     1) !==========================================================================================!
     2) !==========================================================================================!
     3) !     This subroutine reads ED-1.0 and ED-2.0 history files (aka restart), for a polygon   !
     4) ! level run with previous ecological state, but with standard initial thermodynamic        !
     5) ! conditions.                                                                              !
     6) !------------------------------------------------------------------------------------------!
     7) subroutine read_ed10_ed20_history_file
     8) 
     9) 
    10)    use ed_max_dims    , only : n_pft               & ! intent(in)
    11)                              , huge_patch          & ! intent(in)
    12)                              , huge_cohort         & ! intent(in)
    13)                              , max_water           & ! intent(in)
    14)                              , str_len             & ! intent(in)
    15)                              , maxfiles            & ! intent(in)
    16)                              , maxlist             ! ! intent(in)
    17)    use pft_coms       , only : q                   & ! intent(in)
    18)                              , qsw                 & ! intent(in)
    19)                              , min_dbh             & ! intent(in)
    20)                              , min_bdead           & ! intent(in)
    21)                              , is_grass            & ! intent(in)
    22)                              , include_pft         & ! intent(in)
    23)                              , include_pft_ag      & ! intent(in)
    24)                              , pft_1st_check       & ! intent(in)
    25)                              , agf_bs              & ! intent(in)
    26)                              , include_these_pft   ! ! intent(in)
    27)    use ed_misc_coms   , only : sfilin              & ! intent(in)
    28)                              , ied_init_mode       ! ! intent(in)
    29)    use consts_coms    , only : pio180              & ! intent(in)
    30)                              , pio4                ! ! intent(in)
    31)    use ed_misc_coms   , only : use_target_year     & ! intent(in)
    32)                              , restart_target_year ! ! intent(in)
    33)    use ed_state_vars  , only : polygontype         & ! variable type
    34)                              , sitetype            & ! variable type
    35)                              , patchtype           & ! variable type
    36)                              , edtype              & ! variable type
    37)                              , edgrid_g            & ! variable type
    38)                              , allocate_sitetype   & ! subroutine
    39)                              , allocate_patchtype  ! ! subroutine
    40)    use grid_coms      , only : ngrids              ! ! intent(in)
    41)    use allometry      , only : bd2dbh              & ! function
    42)                              , dbh2h               & ! function
    43)                              , dbh2bd              & ! function
    44)                              , size2bl             & ! function
    45)                              , ed_biomass          & ! function
    46)                              , area_indices        ! ! subroutine
    47)    use fuse_fiss_utils, only : sort_cohorts        & ! subroutine
    48)                              , sort_patches        ! ! subroutine
    49)    use disturb_coms   , only : ianth_disturb       ! ! intent(in)
    50)    implicit none
    51) 
    52)    !----- Local constants. ----------------------------------------------------------------!
    53)    real(kind=8), parameter :: min_area = 1.d-7           ! Minimum acceptable area.
    54)    real(kind=8), parameter :: min_ok   = 1.d-20          ! Minimum acceptable value for
    55)                                                          !    any restart variable.
    56)    logical     , parameter :: harvard_override = .false. ! Overwrite some initial values
    57)                                                          !    for a specific Harvard run?
    58)                                                          !    (this should never be used
    59)                                                          !     unless you are 110% sure of
    60)                                                          !     what you are doing.)
    61)    !----- Local variables. ----------------------------------------------------------------!
    62)    type(edtype)          , pointer                        :: cgrid
    63)    type(polygontype)     , pointer                        :: cpoly
    64)    type(sitetype)        , pointer                        :: csite
    65)    type(patchtype)       , pointer                        :: cpatch
    66)    character(len=str_len), dimension(maxlist)             :: full_list
    67)    character(len=str_len), dimension(maxfiles)            :: site_list
    68)    character(len=str_len), dimension(maxfiles)            :: pss_list
    69)    character(len=str_len), dimension(maxfiles)            :: css_list
    70)    character(len=str_len), dimension(huge_patch)          :: pname
    71)    character(len=str_len)                                 :: pss_name
    72)    character(len=str_len)                                 :: css_name
    73)    character(len=str_len)                                 :: cdum
    74)    character(len=str_len), dimension(huge_cohort)         :: cname
    75)    character(len=str_len), dimension(huge_cohort)         :: cpname
    76)    integer               , dimension(huge_patch)          :: trk
    77)    integer               , dimension(huge_patch)          :: sitenum
    78)    integer               , dimension(huge_cohort)         :: leaves_on
    79)    integer               , dimension(huge_cohort)         :: ipft
    80)    integer                                                :: year
    81)    integer                                                :: igr
    82)    integer                                                :: ipy
    83)    integer                                                :: isi
    84)    integer                                                :: ipa
    85)    integer                                                :: ico
    86)    integer                                                :: ip
    87)    integer                                                :: ip2
    88)    integer                                                :: ic
    89)    integer                                                :: ic2
    90)    integer                                                :: nwater
    91)    integer                                                :: ierr
    92)    integer                                                :: nf
    93)    integer                                                :: nflist
    94)    integer                                                :: nflsite
    95)    integer                                                :: nflpss
    96)    integer                                                :: nflcss
    97)    integer                                                :: nclosest
    98)    integer                                                :: ncohorts
    99)    integer                                                :: npatchco
   100)    integer                                                :: npatches
   101)    integer                                                :: nsitepat
   102)    integer                                                :: npatch2
   103)    integer                                                :: nw
   104)    logical              , dimension(huge_cohort)          :: add_this_cohort
   105)    logical                                                :: renumber_pfts
   106)    logical                                                :: site_match
   107)    real                 , dimension(max_water)            :: depth
   108)    real                 , dimension(huge_patch)           :: time
   109)    real                 , dimension(huge_patch)           :: age
   110)    real                 , dimension(huge_patch)           :: area
   111)    real                 , dimension(huge_patch)           :: fsc
   112)    real                 , dimension(huge_patch)           :: stsc
   113)    real                 , dimension(huge_patch)           :: stsl
   114)    real                 , dimension(huge_patch)           :: ssc
   115)    real                 , dimension(huge_patch)           :: psc
   116)    real                 , dimension(huge_patch)           :: msn
   117)    real                 , dimension(huge_patch)           :: fsn
   118)    real                 , dimension(max_water,huge_patch) :: water
   119)    real                 , dimension(12,huge_cohort)       :: cb
   120)    real                 , dimension(12,huge_cohort)       :: cb_max
   121)    real                 , dimension(huge_cohort)          :: balive
   122)    real                 , dimension(huge_cohort)          :: avgRg
   123)    real                 , dimension(huge_cohort)          :: bdead
   124)    real                 , dimension(huge_cohort)          :: nplant
   125)    real                 , dimension(huge_cohort)          :: hite
   126)    real                 , dimension(huge_cohort)          :: dbh
   127)    real                 , dimension(huge_cohort)          :: ctime
   128)    real                 , dimension(maxfiles)             :: slon_list,slat_list
   129)    real                 , dimension(maxfiles)             :: plon_list,plat_list
   130)    real                 , dimension(maxfiles)             :: clon_list,clat_list
   131)    real                 , dimension(maxfiles)             :: file_pdist,file_cdist
   132)    real                                                   :: dummy
   133)    real                                                   :: area_tot
   134)    real                                                   :: area_sum
   135)    real(kind=8)         , dimension(max_water)            :: dwater
   136)    real(kind=8)                                           :: dage
   137)    real(kind=8)                                           :: darea
   138)    real(kind=8)                                           :: dfsc
   139)    real(kind=8)                                           :: dstsc
   140)    real(kind=8)                                           :: dstsl
   141)    real(kind=8)                                           :: dssc
   142)    real(kind=8)                                           :: dpsc
   143)    real(kind=8)                                           :: dmsn
   144)    real(kind=8)                                           :: dfsn
   145)    !----- External function. --------------------------------------------------------------!
   146)    real                 , external                        :: sngloff
   147)    real                 , external                        :: dist_gc
   148)    !---------------------------------------------------------------------------------------!
   149) 
   150) 
   151) 
   152) 
   153) 
   154) 
   155)    !---------------------------------------------------------------------------------------!
   156)    !      Now we loop over all all grids and polygons, and fill them with patches and      !
   157)    ! cohorts from the closest polygon.                                                     !
   158)    !---------------------------------------------------------------------------------------!
   159)    gridloop: do igr = 1,ngrids
   160) 
   161)       !----- Retrieve all files with the specified prefix. --------------------------------!
   162)       call ed_filelist(full_list,sfilin(igr),nflist)
   163) 
   164)       !----- Retrieve LON/LAT information for sites ---------------------------------------!
   165)       select case (ied_init_mode)
   166)       case (3)
   167)          renumber_pfts = .false.
   168)          call ed1_fileinfo('.site',nflist,full_list,nflsite,site_list,slon_list,slat_list)
   169)       case (6)
   170)          renumber_pfts = .false.
   171)       case default
   172)          renumber_pfts = .true.
   173)       end select
   174) 
   175)       !----- Retrieve LON/LAT information for patches and cohorts -------------------------!
   176)       call ed1_fileinfo('.pss',nflist,full_list,nflpss,pss_list,plon_list,plat_list)
   177)       call ed1_fileinfo('.css',nflist,full_list,nflcss,css_list,clon_list,clat_list)
   178) 
   179)       cgrid => edgrid_g(igr)
   180) 
   181)       polyloop: do ipy = 1,cgrid%npolygons
   182) 
   183)          cpoly => cgrid%polygon(ipy)
   184) 
   185)          !---------------------------------------------------------------------------------!
   186)          !     Especial set up of soil types for Harvard forest when running some specific !
   187)          ! test.                                                                           !
   188)          !---------------------------------------------------------------------------------!
   189)          if (harvard_override) then
   190)             cpoly%lsl(1)            = 1
   191)             cpoly%ntext_soil(1,isi) = 2
   192)             cpoly%ntext_soil(2,isi) = 2
   193)             cpoly%ntext_soil(3,isi) = 3
   194)             cpoly%ntext_soil(4,isi) = 3
   195)             cpoly%ncol_soil(isi)    = 10
   196)          end if
   197) 
   198) 
   199)          !---------------------------------------------------------------------------------!
   200)          !    Initialise the distances as very large numbers, so if we don't fill all the  !
   201)          ! patches and cohorts, we will not going to take non-sense as a valid polygon.    !
   202)          !---------------------------------------------------------------------------------!
   203)          file_pdist(:) = 1.e20
   204)          file_cdist(:) = 1.e20
   205) 
   206)          !---------------------------------------------------------------------------------!
   207)          !    Compute the distances between every polygon in the restart files and the     !
   208)          ! current polygon.                                                                !
   209)          !---------------------------------------------------------------------------------!
   210)          do nf=1,nflpss
   211)             file_pdist(nf) = dist_gc(cgrid%lon(ipy),plon_list(nf)                          &
   212)                                     ,cgrid%lat(ipy),plat_list(nf) )
   213)          end do
   214)          do nf=1,nflcss
   215)             file_cdist(nf) = dist_gc(cgrid%lon(ipy),clon_list(nf)                          &
   216)                                     ,cgrid%lat(ipy),clat_list(nf) )
   217)          end do
   218)          !---------------------------------------------------------------------------------!
   219) 
   220) 
   221)          find_nonwater: do nf=1,nflpss
   222)             !------------------------------------------------------------------------------!
   223)             !     Find the file that is the closest to the current polygon, based on the   !
   224)             ! distance vector.                                                             !
   225)             !------------------------------------------------------------------------------!
   226)             nclosest = minloc(file_pdist,dim=1)
   227)             pss_name = trim(pss_list(nclosest))
   228)             write (unit=*,fmt='(2a)') 'Using patch file: ',trim(pss_name)
   229) 
   230)             !------------------------------------------------------------------------------!
   231)             !    Open the patch file and read in all patches.                              !
   232)             !------------------------------------------------------------------------------!
   233)             open(unit=12,file=trim(pss_name),form='formatted',status='old',action='read')
   234)             read(unit=12,fmt='(a4)')  cdum
   235) 
   236)             !----- Read the other information from the header (if there is any...). -------!
   237)             nwater = 1
   238)             if (ied_init_mode == 1 ) then
   239)                read (unit=12,fmt=*) cdum,nwater
   240)                read (unit=12,fmt=*) cdum,depth(1:nwater)
   241)                read (unit=12,fmt=*)
   242)             end if
   243) 
   244)             !------------------------------------------------------------------------------!
   245)             !     Now we loop over all patches and decide whether they should be included  !
   246)             ! or not.                                                                      !
   247)             !------------------------------------------------------------------------------!
   248)             ip      = 1
   249)             sitenum = 0
   250)             count_patches: do
   251) 
   252)                !---------------------------------------------------------------------------!
   253)                !     We must check whether we are not exceeding the maximum number of      !
   254)                ! patches that we can read.                                                 !
   255)                !---------------------------------------------------------------------------!
   256)                if (ip > huge_patch) then
   257)                   write (unit=*,fmt='(a,1x,a)')  ' In file:',trim(pss_name)
   258)                   write (unit=*,fmt='(a)')       ' Number of patches is > HUGE_PATCH...'
   259)                   write (unit=*,fmt='(a,1x,i7)') ' HUGE_PATCH:',huge_patch
   260)                   write (unit=*,fmt='(a)')       ' Increase HUGE_PATCH to read this...'
   261)                   call fatal_error('Too many patches to be read...'                        &
   262)                                   ,'read_ed10_ed20_history_file','ed_history_io.f90')
   263)                end if
   264) 
   265)                !---------------------------------------------------------------------------!
   266)                !    If we can still add new patches, read the next one, according to the   !
   267)                ! input format type.                                                        !
   268)                !---------------------------------------------------------------------------!
   269)                select case (ied_init_mode)
   270)                case (1)
   271)                   !----- ED-1.0 file. -----------------------------------------------------!
   272)                   read(unit=12,fmt=*,iostat=ierr) time(ip),pname(ip),trk(ip),dage,darea    &
   273)                                                  ,dfsc,dstsc,dstsl,dssc,dpsc,dmsn,dfsn     &
   274)                                                  ,dwater(1:nwater)
   275) 
   276)                   !------------------------------------------------------------------------!
   277)                   !     Check whether the file has hit the end, and if so, leave the loop. !
   278)                   !------------------------------------------------------------------------!
   279)                   if (ierr /= 0) exit count_patches
   280) 
   281)                   !----- Copy the double-precision scratch variables to the arrays. -------!
   282)                   area  (ip) = sngloff(darea      ,min_area)
   283)                   age   (ip) = sngloff(dage       ,min_ok  )
   284)                   fsc   (ip) = sngloff(dfsc       ,min_ok  )
   285)                   stsc  (ip) = sngloff(dstsc      ,min_ok  )
   286)                   stsl  (ip) = sngloff(dstsl      ,min_ok  )
   287)                   ssc   (ip) = sngloff(dssc       ,min_ok  )
   288)                   psc   (ip) = sngloff(dpsc       ,min_ok  )
   289)                   msn   (ip) = sngloff(dmsn       ,min_ok  )
   290)                   fsn   (ip) = sngloff(dfsn       ,min_ok  )
   291)                   do nw=1,nwater
   292)                      water(nw,ip) = sngloff(dwater(nw) ,min_ok  )
   293)                   end do
   294) 
   295)                case (2,6)
   296)                   !----- Standard ED-2.0 file. --------------------------------------------!
   297)                   read(unit=12,fmt=*,iostat=ierr) time(ip),pname(ip),trk(ip),dage,darea    &
   298)                                                  ,dwater(1),dfsc,dstsc,dstsl,dssc,dummy    &
   299)                                                  ,dmsn,dfsn
   300) 
   301)                   !------------------------------------------------------------------------!
   302)                   !     Check whether the file has hit the end, and if so, leave the loop. !
   303)                   !------------------------------------------------------------------------!
   304)                   if(ierr /= 0)exit count_patches
   305) 
   306)                   !----- Copy the double-precision scratch variables to the arrays. -------!
   307)                   area   (ip) = sngloff(darea    ,min_area)
   308)                   age    (ip) = sngloff(dage     ,min_ok  )
   309)                   fsc    (ip) = sngloff(dfsc     ,min_ok  )
   310)                   stsc   (ip) = sngloff(dstsc    ,min_ok  )
   311)                   stsl   (ip) = sngloff(dstsl    ,min_ok  )
   312)                   ssc    (ip) = sngloff(dssc     ,min_ok  )
   313)                   msn    (ip) = sngloff(dmsn     ,min_ok  )
   314)                   fsn    (ip) = sngloff(dfsn     ,min_ok  )
   315)                   water(1,ip) = sngloff(dwater(1),min_ok  )
   316) 
   317)                case (3)
   318)                   !----- ED-2.0 file, with site information. ------------------------------!
   319)                   read(unit=12,fmt=*,iostat=ierr) sitenum(ip),time(ip),pname(ip),trk(ip)   &
   320)                                                  ,age(ip),darea,water(1,ip),fsc(ip)        &
   321)                                                  ,stsc(ip),stsl(ip),ssc(ip),psc(ip)        &
   322)                                                  ,msn(ip),fsn(ip)
   323) 
   324)                   !------------------------------------------------------------------------!
   325)                   !     Check whether the file has hit the end, and if so, leave the loop. !
   326)                   !------------------------------------------------------------------------!
   327)                   if (ierr /= 0) exit count_patches
   328) 
   329) 
   330)                   area(ip)=sngloff(darea, min_area)
   331) 
   332)                   site_match = .false.
   333)                   do isi = 1,cpoly%nsites
   334)                      if (sitenum(ip).eq.cpoly%sitenum(isi)) then
   335)                         site_match = .true.
   336)                      end if
   337)                   end do
   338)                   if (.not.site_match) then
   339)                      write (unit=*,fmt='(a,1x,i6)') ' ISI           = ',isi
   340)                      write (unit=*,fmt='(a,1x,i6)') ' CPOLY%NSITES  = ',cpoly%nsites
   341)                      write (unit=*,fmt='(a,1x,i6)') ' IP            = ',ip
   342)                      write (unit=*,fmt='(a,1x,i6)') ' CPOLY%SITENUM = ',cpoly%sitenum
   343)                      write (unit=*,fmt='(a,1x,a)')  ' Patch file    = ',trim(pss_name)
   344)                      write (unit=*,fmt='(a,1x,i5,1x,a)')                                   &
   345)                                                     ' Site number', sitenum,'not found!'
   346)                      call fatal_error('Error reading from patch file'                      &
   347)                                      ,'read_ed10_ed20_history_file','ed_history_io.F90')
   348)                   end if
   349)                case default !Nearly bare ground
   350)                   exit count_patches
   351)                end select
   352)                !---------------------------------------------------------------------------!
   353) 
   354) 
   355) 
   356)                !---------------------------------------------------------------------------!
   357)                !      By adding ip only if the area is above minimum, we avoid including   !
   358)                ! patches that are tiny since they will be overwritten by the next patch.   !
   359)                !---------------------------------------------------------------------------!
   360)                if (area(ip) > min_area) ip = ip + 1
   361)                !---------------------------------------------------------------------------!
   362) 
   363)             end do count_patches
   364)             !------------------------------------------------------------------------------!
   365) 
   366) 
   367) 
   368)             !------------------------------------------------------------------------------!
   369)             !      Here we determine the number of patches.  Unless there are no patches,  !
   370)             ! (this may happen if the file used was all water.  In that case, then we      !
   371)             ! should use the next closest file.                                            !
   372)             !------------------------------------------------------------------------------!
   373)             npatches = max(ip-1,0)
   374) 
   375)             close(unit=12,status='keep')
   376) 
   377)             if (npatches > 0) then
   378) 
   379)                !---------------------------------------------------------------------------!
   380)                !      We have found a suitable file, we will leave the loop, after we find !
   381)                ! the name of the corresponding cohort file.                                !
   382)                !---------------------------------------------------------------------------!
   383)                nclosest = minloc(abs(file_pdist(nclosest)-file_cdist),dim=1)
   384)                css_name = trim(css_list(nclosest))
   385)                write (unit=*,fmt='(2a)') 'Using cohort file: ',trim(css_name)
   386)                exit find_nonwater
   387)             else
   388)                !----- The closest file was no good, so we make it far away for now. -------!
   389)                file_pdist(nclosest) = 1.e20
   390)             end if
   391)          end do find_nonwater
   392)          !---------------------------------------------------------------------------------!
   393) 
   394) 
   395)          !---------------------------------------------------------------------------------!
   396)          !     Allocate the patches.                                                       !
   397)          !---------------------------------------------------------------------------------!
   398)          select case (ied_init_mode)
   399)          case (3) !---- Multiple-site run, find the number of patches in each site. -------!
   400) 
   401)             !----- Loop over all sites. ---------------------------------------------------!
   402)             do isi = 1,cpoly%nsites
   403) 
   404)                npatch2 = 0
   405) 
   406)                do ip=1,npatches
   407)                   if(sitenum(ip) == cpoly%sitenum(isi)) npatch2 = npatch2 + 1
   408)                end do
   409) 
   410)                csite => cpoly%site(isi)
   411) 
   412)                !----- Allocate the patches in this site. ----------------------------------!
   413)                call allocate_sitetype(csite,npatch2)
   414)                !---------------------------------------------------------------------------!
   415) 
   416)                !---------------------------------------------------------------------------!
   417)                !     Go through the patch list and fill the sites' patches with data.      !
   418)                !---------------------------------------------------------------------------!
   419)                ip2 = 0
   420)                do ip=1,npatches
   421) 
   422)                   if (sitenum(ip) == cpoly%sitenum(isi)) then
   423)                      ip2 = ip2 + 1
   424) 
   425)                      !----- Translate the land use type. ----------------------------------!
   426)                      select case (trk(ip))
   427)                      case (0)
   428)                         !----- Agriculture. -----------------------------------------------!
   429)                         csite%dist_type (ip2) = 1
   430)                         !------------------------------------------------------------------!
   431)                      case (1)
   432)                         !------------------------------------------------------------------!
   433)                         !     Secondary forest.  This was ambiguous in previous versions,  !
   434)                         ! so we decide which type to point this patch based on whether     !
   435)                         ! anthropogenic disturbance is turned on or not.  Using flags 5 or !
   436)                         ! 6 is unambiguous.                                                !
   437)                         !------------------------------------------------------------------!
   438)                         select case (ianth_disturb)
   439)                         case (0)
   440)                            !----- No anthropogenic, assume abandoned lands. ---------------!
   441)                            csite%dist_type (ip2) = 5
   442)                            !---------------------------------------------------------------!
   443)                         case (1)
   444)                            !----- Anthropogenic, assume logging. --------------------------!
   445)                            csite%dist_type (ip2) = 6
   446)                            !---------------------------------------------------------------!
   447)                         end select
   448)                         !------------------------------------------------------------------!
   449)                      case (2)
   450)                         !----- Primary forest. --------------------------------------------!
   451)                         csite%dist_type (ip2) = 3
   452)                         !------------------------------------------------------------------!
   453)                      case (3)
   454)                         !----- Forest plantation. -----------------------------------------!
   455)                         csite%dist_type (ip2) = 2
   456)                         !------------------------------------------------------------------!
   457)                      case (4)
   458)                         !----- Burnt patch. -----------------------------------------------!
   459)                         csite%dist_type (ip2) = 4
   460)                         !------------------------------------------------------------------!
   461)                      case (5)
   462)                         !----- Abandoned land (secondary growth). -------------------------!
   463)                         csite%dist_type (ip2) = 5
   464)                         !------------------------------------------------------------------!
   465)                      case (6)
   466)                         !----- Logged forest. ---------------------------------------------!
   467)                         csite%dist_type (ip2) = 6
   468)                         !------------------------------------------------------------------!
   469)                      end select
   470)                      !---------------------------------------------------------------------!
   471) 
   472)                      csite%age               (ip2) = age (ip)
   473)                      csite%area              (ip2) = area(ip)
   474)                      csite%fast_soil_C       (ip2) = fsc (ip)
   475)                      csite%slow_soil_C       (ip2) = ssc (ip)
   476)                      csite%structural_soil_C (ip2) = stsc(ip)
   477)                      csite%structural_soil_L (ip2) = stsl(ip)
   478)                      csite%mineralized_soil_N(ip2) = msn (ip)
   479)                      csite%fast_soil_N       (ip2) = fsn (ip)
   480)                      csite%pname             (ip2) = trim(pname(ip))
   481)                      csite%sum_dgd           (ip2) = 0.0
   482)                      csite%sum_chd           (ip2) = 0.0
   483)                      csite%cohort_count      (ip2) = 0
   484)                   end if
   485)                end do
   486)                !---- Initialize the cohort counts per patch. ------------------------------!
   487)                csite%cohort_count(:) = 0
   488)             end do
   489) 
   490)          case default
   491)             !------------------------------------------------------------------------------!
   492)             !      We don't have information on sites, assume all sites to be the same.    !
   493)             !------------------------------------------------------------------------------!
   494)             do isi = 1, cpoly%nsites
   495)                csite => cpoly%site(isi)
   496) 
   497)                !----- Allocate the patches in this site. ----------------------------------!
   498)                call allocate_sitetype(csite,npatches)
   499) 
   500)                do ip=1,npatches
   501) 
   502)                   !----- Translate the land use type. -------------------------------------!
   503)                   select case (trk(ip))
   504)                   case (0)
   505)                      !----- Agriculture. --------------------------------------------------!
   506)                      csite%dist_type (ip) = 1
   507)                      !---------------------------------------------------------------------!
   508)                   case (1)
   509)                      !---------------------------------------------------------------------!
   510)                      !     Secondary forest.  This was ambiguous in previous versions,     !
   511)                      ! so we decide which type to point this patch based on whether        !
   512)                      ! anthropogenic disturbance is turned on or not.  Using flags 5 or    !
   513)                      ! 6 is unambiguous.                                                   !
   514)                      !---------------------------------------------------------------------!
   515)                      select case (ianth_disturb)
   516)                      case (0)
   517)                         !----- No anthropogenic, assume abandoned lands. ------------------!
   518)                         csite%dist_type (ip) = 5
   519)                         !------------------------------------------------------------------!
   520)                      case (1)
   521)                         !----- Anthropogenic, assume logging. -----------------------------!
   522)                         csite%dist_type (ip) = 6
   523)                         !------------------------------------------------------------------!
   524)                      end select
   525)                      !---------------------------------------------------------------------!
   526)                   case (2)
   527)                      !----- Primary forest. -----------------------------------------------!
   528)                      csite%dist_type (ip) = 3
   529)                      !---------------------------------------------------------------------!
   530)                   case (3)
   531)                      !----- Forest plantation. --------------------------------------------!
   532)                      csite%dist_type (ip) = 2
   533)                      !---------------------------------------------------------------------!
   534)                   case (4)
   535)                      !----- Burnt patch. --------------------------------------------------!
   536)                      csite%dist_type (ip) = 4
   537)                      !---------------------------------------------------------------------!
   538)                   case (5)
   539)                      !----- Abandoned land (secondary growth). ----------------------------!
   540)                      csite%dist_type (ip) = 5
   541)                      !---------------------------------------------------------------------!
   542)                   case (6)
   543)                      !----- Logged forest. ------------------------------------------------!
   544)                      csite%dist_type (ip) = 6
   545)                      !---------------------------------------------------------------------!
   546)                   end select
   547)                   !------------------------------------------------------------------------!
   548) 
   549)                   csite%age               (ip) = age (ip)
   550)                   csite%area              (ip) = area(ip)
   551)                   csite%fast_soil_C       (ip) = fsc (ip)
   552)                   csite%slow_soil_C       (ip) = ssc (ip)
   553)                   csite%structural_soil_C (ip) = stsc(ip)
   554)                   csite%structural_soil_L (ip) = stsl(ip)
   555)                   csite%mineralized_soil_N(ip) = msn (ip)
   556)                   csite%fast_soil_N       (ip) = fsn (ip)
   557)                   csite%pname             (ip) = trim(pname(ip))
   558)                   csite%sum_dgd           (ip) = 0.0
   559)                   csite%sum_chd           (ip) = 0.0
   560)                   csite%cohort_count      (ip) = 0
   561)                end do
   562)                !---------------------------------------------------------------------------!
   563) 
   564)                !---- Initialize the cohort counts per patch. ------------------------------!
   565)                csite%cohort_count(:) = 0
   566)                !---------------------------------------------------------------------------!
   567)             end do
   568)             !------------------------------------------------------------------------------!
   569)          end select
   570)          !---------------------------------------------------------------------------------!
   571) 
   572)          close(unit=12,status='keep')
   573) 
   574)          !---------------------------------------------------------------------------------!
   575)          !    Open the cohort file and read in all cohorts.                                !
   576)          !---------------------------------------------------------------------------------!
   577)          open(unit=12,file=trim(css_name),form='formatted',status='old')
   578)          read(unit=12,fmt='(a4)')  cdum
   579) 
   580)          if (ied_init_mode == 1) then
   581)             read(unit=12,fmt=*) !---- Skip second line. -----------------------------------!
   582)          end if
   583)          ic = 0
   584) 
   585)          !---------------------------------------------------------------------------------!
   586)          !     Now we loop over all patches and decide whether they should be included     !
   587)          ! or not.                                                                         !
   588)          !---------------------------------------------------------------------------------!
   589)          read_cohorts: do
   590) 
   591)             ic = ic + 1
   592)             add_this_cohort(ic) = .true.
   593) 
   594)             !------------------------------------------------------------------------------!
   595)             !     We must check whether we are not exceeding the maximum number of patches !
   596)             ! that we can read.                                                            !
   597)             !------------------------------------------------------------------------------!
   598)             if (ic > huge_cohort) then
   599)                write (unit=*,fmt='(a,1x,a)')  ' In file:',trim(css_name)
   600)                write (unit=*,fmt='(a)')       ' Number of cohorts is > HUGE_COHORT...'
   601)                write (unit=*,fmt='(a,1x,i7)') ' HUGE_COHORT:',huge_cohort
   602)                write (unit=*,fmt='(a)')       ' Increase HUGE_COHORT to read this...'
   603)                call fatal_error('Too many cohorts to be read...'                           &
   604)                                ,'read_ed10_ed20_history_file','ed_history_io.f90')
   605)             end if
   606) 
   607)             !------------------------------------------------------------------------------!
   608)             !    If we can still add new cohorts, read the next one, according to the      !
   609)             ! input format type.                                                           !
   610)             !------------------------------------------------------------------------------!
   611)             select case (ied_init_mode)
   612)             case (1)
   613)                !----- ED-1.0 file. --------------------------------------------------------!
   614)                read(unit=12,fmt=*,iostat=ierr)  ctime(ic),cpname(ic),cname(ic),dbh(ic)     &
   615)                                                ,hite(ic),ipft(ic),nplant(ic),bdead(ic)     &
   616)                                                ,balive(ic),avgRg(ic),leaves_on(ic)         &
   617)                                                ,cb(1:12,ic),cb_max(1:12,ic)
   618) 
   619)                !---------------------------------------------------------------------------!
   620)                !     Check whether the file has hit the end, and if so, leave the loop.    !
   621)                !---------------------------------------------------------------------------!
   622)                if(ierr /= 0) exit read_cohorts
   623) 
   624)             case (2,3,6)
   625)                !----- ED-2.0 file. --------------------------------------------------------!
   626)                read(unit=12,fmt=*,iostat=ierr) ctime(ic),cpname(ic),cname(ic),dbh(ic)      &
   627)                                               ,hite(ic),ipft(ic),nplant(ic),bdead(ic)      &
   628)                                               ,balive(ic),avgRg(ic)
   629)                !---------------------------------------------------------------------------!
   630)                !     Check whether the file has hit the end, and if so, leave the loop.    !
   631)                !---------------------------------------------------------------------------!
   632)                if(ierr /= 0)exit read_cohorts
   633) 
   634) 
   635)                !----- No carbon balance information.  Assign 1. ---------------------------!
   636)                cb(1:12,ic)     = 1.0
   637)                cb_max(1:12,ic) = 1.0
   638)             end select
   639) 
   640)             !------------------------------------------------------------------------------!
   641)             !     The PFT classes has changed between different ED versions, here we       !
   642)             ! standardise this.                                                            !
   643)             !------------------------------------------------------------------------------!
   644)             if(renumber_pfts) then
   645)                if (ipft(ic) < 100) then
   646)                   ipft(ic) = ipft(ic) + 1
   647)                   if(ipft(ic) >= 5) ipft(ic) = ipft(ic) - 3
   648)                else
   649)                   ipft(ic) = ipft(ic) - 100
   650)                end if
   651)             end if
   652) 
   653)             !----- Check if the year matches.  If not, we will ignore this cohort. --------!
   654)             year = int(ctime(ic))
   655)             if(use_target_year == 1 .and. year /= restart_target_year) then
   656)                add_this_cohort(ic) = .false.
   657)             end if
   658) 
   659)             !----- Remove cohort in case nplant > 0. --------------------------------------!
   660)             if(nplant(ic) < tiny(1.0)) add_this_cohort(ic) = .false.
   661) 
   662) 
   663)             !------------------------------------------------------------------------------!
   664)             !     Find site and patch to which this cohort belong, and start counting how  !
   665)             ! many we need to allocate.                                                    !
   666)             !------------------------------------------------------------------------------!
   667)             put_cohort:do isi=1,cpoly%nsites
   668)                csite => cpoly%site(isi)
   669)                do ipa=1,csite%npatches
   670) 
   671)                   !------------------------------------------------------------------------!
   672)                   !    Here we test whether the PFT of this cohort is expected to be       !
   673)                   ! included.                                                              !
   674)                   !------------------------------------------------------------------------!
   675)                   if (.not. include_pft(ipft(ic))) then
   676)                      !----- This PFT wasn't expected... -----------------------------------!
   677)                      select case (pft_1st_check)
   678)                      case (0)
   679)                         !----- Stop the run. ----------------------------------------------!
   680)                         write (unit=*,fmt='(a,1x,i5,1x,a)')                                &
   681)                              'I found a cohort with PFT=',ipft(ic)                         &
   682)                             ,' and it is not in your include_these_pft...'
   683)                         call fatal_error('Invalid PFT in history file'                     &
   684)                                         ,'read_ed10_ed20_history_file','ed_history_io.f90')
   685) 
   686)                      case (1)
   687)                         !----- Add the unexpected PFT to the list of possible PFTs. -------!
   688)                         write (unit=*,fmt='(a,1x,i5,1x,a)')                                &
   689)                              'I found a cohort with PFT=',ipft(ic)                         &
   690)                             ,'... Including this PFT in your include_these_pft...'
   691)                         include_pft(ipft(ic))                 = .true.
   692)                         include_these_pft(count(include_pft)) = ipft(ic)
   693)                         call sort_up(include_these_pft,n_pft)
   694)                         if (is_grass(ipft(ic))) include_pft_ag(ipft(ic)) = .true.
   695) 
   696)                      case (2)
   697)                         !----- Ignore the cohort. -----------------------------------------!
   698)                         write (unit=*,fmt='(a,1x,i5,1x,a)')                                &
   699)                              'I found a cohort with PFT=',ipft(ic),'... Ignoring it...'
   700)                         add_this_cohort(ic) = .false.
   701)                      end select
   702)                   end if
   703) 
   704)                   if (trim(csite%pname(ipa)) == trim(cpname(ic) ) .and.                    &
   705)                       add_this_cohort(ic)                              ) then
   706)                      csite%cohort_count(ipa) = csite%cohort_count(ipa) + 1
   707)                      exit put_cohort
   708)                   end if
   709)                end do
   710)             end do put_cohort
   711)          end do read_cohorts
   712) 
   713)          !----- Find the total number of cohorts. -----------------------------------------!
   714)          ncohorts = max(ic-1,0)
   715) 
   716) 
   717)          close (unit=12,status='keep')
   718) 
   719)          loop_sites: do isi=1,cpoly%nsites
   720)             csite => cpoly%site(isi)
   721) 
   722)             loop_patches: do ipa=1,csite%npatches
   723)                cpatch => csite%patch(ipa)
   724) 
   725)                if (csite%cohort_count(ipa) /= 0) then
   726) 
   727)                   call allocate_patchtype(cpatch,csite%cohort_count(ipa))
   728)                   csite%plant_ag_biomass(ipa) = 0.
   729)                   ic2 = 0
   730)                   do ic = 1,ncohorts
   731) 
   732)                      if (trim(csite%pname(ipa)) == trim(cpname(ic) ) .and.                 &
   733)                          add_this_cohort(ic)                              ) then
   734) 
   735)                         ic2 = ic2 + 1
   736)                         cpatch%pft(ic2) = ipft(ic)
   737) 
   738) 
   739)                         !------------------------------------------------------------------!
   740)                         !      There were some differences on how a few variables were     !
   741)                         ! defined between ED-1.0 and ED-2.0.  Here we will put everything  !
   742)                         ! in the ED-2.1 standard.                                          !
   743)                         !------------------------------------------------------------------!
   744)                         !----- Plant density, it should be in [plants/m2]. ----------------!
   745)                         if(ied_init_mode == 1) then
   746)                            cpatch%nplant(ic2) = nplant(ic) / (csite%area(ipa) )
   747)                         else
   748)                            cpatch%nplant(ic2) = nplant(ic)
   749)                         end if
   750) 
   751) 
   752)                         !------------------------------------------------------------------!
   753)                         !     Select which history file we are using.  We use DBH only     !
   754)                         ! when ied_init_mode is 6 (inventory initialisation, when DBH is   !
   755)                         ! most likely to be the actual measured variable), otherwise we    !
   756)                         ! use BDEAD instead.                                               !
   757)                         !------------------------------------------------------------------!
   758)                         select case(ied_init_mode)
   759)                         case (6)
   760)                            !----- Inventory.  Read DBH and find the other stuff. ----------!
   761)                            cpatch%dbh(ic2)   = max(dbh(ic),min_dbh(ipft(ic)))
   762)                            cpatch%hite(ic2)  = dbh2h(ipft(ic),dbh(ic))
   763)                            cpatch%bdead(ic2) = dbh2bd(dbh(ic),ipft(ic))
   764) 
   765)                         case default
   766)                            !---------------------------------------------------------------!
   767)                            !    Old ED files.  Check whether bdead is valid.  If it is, we !
   768)                            ! initialise DBH and height from BDEAD, othewise we use DBH     !
   769)                            ! instead.  This is because allometry may be different, and ED  !
   770)                            ! biomass varies less than DBH under different allometric       !
   771)                            ! equations.                                                    !
   772)                            !---------------------------------------------------------------!
   773)                            if (bdead(ic) > 0.0) then
   774)                               cpatch%bdead(ic2) = max(bdead(ic),min_bdead(ipft(ic)))
   775)                               cpatch%dbh(ic2)   = bd2dbh(ipft(ic),bdead(ic))
   776)                               cpatch%hite(ic2)  = dbh2h(ipft(ic),dbh(ic))
   777)                            else
   778)                               cpatch%dbh(ic2)   = dbh(ic)
   779)                               cpatch%hite(ic2)  = dbh2h(ipft(ic),dbh(ic))
   780)                               cpatch%bdead(ic2) = dbh2bd(dbh(ic),ipft(ic))
   781)                            end if
   782)                         end select
   783)                         !------------------------------------------------------------------!
   784) 
   785) 
   786) 
   787)                         !------------------------------------------------------------------!
   788)                         !     Use allometry to define leaf and the other live biomass      !
   789)                         ! pools.                                                           !
   790)                         !------------------------------------------------------------------!
   791)                         cpatch%bleaf(ic2)     = size2bl(dbh(ic), hite(ic),ipft(ic))
   792)                         cpatch%balive(ic2)    = cpatch%bleaf(ic2) * (1.0 + q(ipft(ic))     &
   793)                                               + qsw(ipft(ic)) * cpatch%hite(ic2))
   794)                         cpatch%broot(ic2)     = cpatch%balive(ic2) * q(ipft(ic))           &
   795)                                               / ( 1.0 + q(ipft(ic)) + qsw(ipft(ic))        &
   796)                                                * cpatch%hite(ic2))
   797)                         cpatch%bsapwooda(ic2) = agf_bs(ipft(ic)) * cpatch%balive(ic2)                &
   798)                                              * qsw(ipft(ic))* cpatch%hite(ic2)             &
   799)                                              / ( 1.0 + q(ipft(ic)) + qsw(ipft(ic))         &
   800)                                                * cpatch%hite(ic2))
   801)                         cpatch%bsapwoodb(ic2) = (1.-agf_bs(ipft(ic))) * cpatch%balive(ic2)           &
   802)                                              * qsw(ipft(ic))* cpatch%hite(ic2)             &
   803)                                              / ( 1.0 + q(ipft(ic)) + qsw(ipft(ic))         &
   804)                                                * cpatch%hite(ic2))
   805) 
   806) 
   807)                         !------------------------------------------------------------------!
   808)                         !     Start plants with full phenology, we will take care of       !
   809)                         ! phenology after this sub-routine.                                !
   810)                         !------------------------------------------------------------------!
   811)                         cpatch%phenology_status(ic2) = 0
   812)                         cpatch%bstorage        (ic2) = 0.0
   813)                         !------------------------------------------------------------------!
   814) 
   815) 
   816) 
   817)                         !----- Assign LAI, WAI, and CAI -----------------------------------!
   818)                         call area_indices(cpatch, ic2)
   819) 
   820)                         !------------------------------------------------------------------!
   821)                         !     Initialise the carbon balance.  We ignore the carbon balance !
   822)                         ! even for ED-1.0, the models are so different that there is no    !
   823)                         ! reason to use the stored value.                                  !
   824)                         !------------------------------------------------------------------!
   825)                         cpatch%cb         (1:12,ic2) = 1.0
   826)                         cpatch%cb_lightmax(1:12,ic2) = 1.0
   827)                         cpatch%cb_moistmax(1:12,ic2) = 1.0
   828)                         cpatch%cb_mlmax   (1:12,ic2) = 1.0
   829)                         cpatch%cb         (  13,ic2) = 0.0
   830)                         cpatch%cb_lightmax(  13,ic2) = 0.0
   831)                         cpatch%cb_moistmax(  13,ic2) = 0.0
   832)                         cpatch%cb_mlmax   (  13,ic2) = 0.0
   833)                         !------------------------------------------------------------------!
   834) 
   835)                         !----- Above ground biomass, use the allometry. -------------------!
   836)                         cpatch%agb(ic2) = ed_biomass(cpatch, ic2)
   837)                         cpatch%basarea(ic2)  = pio4 * cpatch%dbh(ic2) * cpatch%dbh(ic2)
   838) 
   839)                         !----- Growth rates, start with zero. -----------------------------!
   840)                         cpatch%dagb_dt  (ic2)  = 0.
   841)                         cpatch%dlnagb_dt(ic2)  = 0.
   842)                         cpatch%dba_dt   (ic2)  = 0.
   843)                         cpatch%dlnba_dt (ic2)  = 0.
   844)                         cpatch%ddbh_dt  (ic2)  = 0.
   845)                         cpatch%dlndbh_dt(ic2)  = 0.
   846) 
   847)                         !------------------------------------------------------------------!
   848)                         !      Initialise other cohort variables.  Some of them won't be   !
   849)                         ! updated unless the lai exceeds lai_min.                          !
   850)                         !------------------------------------------------------------------!
   851)                         cpatch%fsw(ic2)   = 1.0
   852)                         cpatch%gpp(ic2)   = 0.0
   853)                         cpatch%par_l(ic2) = 0.0
   854) 
   855)                         !----- Update the patch level above-ground biomass. ---------------!
   856)                         csite%plant_ag_biomass(ipa) = csite%plant_ag_biomass(ipa)          &
   857)                                                     + cpatch%agb(ic2) * cpatch%nplant(ic2)
   858)                      end if
   859)                   end do
   860)                end if
   861)             end do loop_patches
   862)          end do loop_sites
   863) 
   864)          close (unit=12,status='keep')
   865) 
   866)          !----- Initialise all the other site-, patch-, and cohort-level variables. -------!
   867)          do isi = 1,cpoly%nsites
   868) 
   869)             area_sum = 0.0
   870)             ncohorts = 0
   871) 
   872) 
   873)             !----- Make sure that the total patch area is 1. ------------------------------!
   874)             csite => cpoly%site(isi)
   875)             area_tot      = sum(csite%area(1:csite%npatches))
   876)             csite%area(:) = csite%area(:)/area_tot
   877) 
   878)             !----- Find the patch-level LAI, WAI, and CAI. --------------------------------!
   879)             do ipa=1,csite%npatches
   880)                area_sum        = area_sum + csite%area(ipa)
   881) 
   882)                cpatch => csite%patch(ipa)
   883)                do ico = 1,cpatch%ncohorts
   884)                   ncohorts        = ncohorts + 1
   885)                end do
   886)             end do
   887) 
   888)             !----- Initialise the cohort variables, then sort them by size. ---------------!
   889)             do ipa = 1,csite%npatches
   890)                cpatch => csite%patch(ipa)
   891)                do ico = 1,cpatch%ncohorts
   892)                   call init_ed_cohort_vars(cpatch,ico,cpoly%lsl(isi))
   893)                end do
   894) 
   895)                !----- Make sure that cohorts are organised from tallest to shortest. ------!
   896)                call sort_cohorts(cpatch)
   897)             end do
   898) 
   899)             !----- Initialise the patch-level variables. ----------------------------------!
   900)             call init_ed_patch_vars(csite,1,csite%npatches,cpoly%lsl(isi))
   901) 
   902)             !----- Make sure that patches are organised from oldest to youngest. ----------!
   903)             call sort_patches(csite)
   904)          end do
   905) 
   906)          !----- Initialise site-level variables. ------------------------------------------!
   907)          call init_ed_site_vars(cpoly,cgrid%lat(ipy))
   908) 
   909) 
   910)          !----- Get a diagnostic on the polygon's vegetation. -----------------------------!
   911)          ncohorts = 0
   912) 
   913)          do isi = 1,cpoly%nsites
   914)             nsitepat = 0
   915)             csite => cpoly%site(isi)
   916) 
   917)             do ipa = 1,csite%npatches
   918)                npatchco        = 0
   919) 
   920)                cpatch => csite%patch(ipa)
   921)                do ico = 1,cpatch%ncohorts
   922)                   ncohorts        = ncohorts+1
   923)                   npatchco        = npatchco+1
   924)                end do
   925)                csite%cohort_count(ipa) = npatchco
   926)                nsitepat                = nsitepat + 1
   927)             end do
   928) 
   929)             cpoly%patch_count(isi) = nsitepat
   930)          end do
   931)       end do polyloop
   932) 
   933)       !----- Initialise the polygon-level variables. --------------------------------------!
   934)       call init_ed_poly_vars(cgrid)
   935)    end do gridloop
   936) 
   937)    return
   938) end subroutine read_ed10_ed20_history_file

   949) !     This subroutine will create the ED1.0/ED2.0 file name based on the longitude and     !
   950) ! latitude.                                                                                !
   951) !------------------------------------------------------------------------------------------!
   952) subroutine create_ed10_ed20_fname(lat,ed_res,lon,sfilin,pss_name,css_name,site_name)
   953)    use ed_max_dims, only : str_len
   954)    implicit none
   955)    !----- Arguments. ----------------------------------------------------------------------!
   956)    real                  , intent(in)  :: lat
   957)    real                  , intent(in)  :: lon
   958)    real                  , intent(in)  :: ed_res
   959)    character(len=*)      , intent(in)  :: sfilin
   960)    character(len=str_len), intent(out) :: pss_name
   961)    character(len=str_len), intent(out) :: css_name
   962)    character(len=str_len), intent(out) :: site_name
   963)    !----- Local variables. ----------------------------------------------------------------!
   964)    real                                :: flon
   965)    real                                :: flat
   966)    character(len=str_len)              :: ed_fname
   967)    !---------------------------------------------------------------------------------------!
   968) 
   969) 
   970) 
   971)    !----- Find the latitude and longitude corresponding to the dataset. -------------------!
   972)    if (lat >= 0.0) then
   973)       flat =   ed_res * real(int( lat / ed_res)) + 0.5 * ed_res
   974)    else
   975)       flat = - ed_res * real(int(-lat / ed_res)) - 0.5 * ed_res
   976)    end if
   977)    if (lon >= 0.0) then
   978)       flon =   ed_res * real(int( lon / ed_res)) + 0.5 * ed_res
   979)    else
   980)       flon = - ed_res * real(int(-lon / ed_res)) - 0.5 * ed_res
   981)    endif
   982) 
   983)    if (ed_res > 0.999 .and. ed_res < 1.001) then
   984) 
   985)       if (flat <= -10.0) then
   986)          write(ed_fname,'(a,f5.1)')trim(sfilin)//'lat',flat
   987)       elseif(flat < 0.0 .or. flat >= 10.0)then
   988)          write(ed_fname,'(a,f4.1)')trim(sfilin)//'lat',flat
   989)       else
