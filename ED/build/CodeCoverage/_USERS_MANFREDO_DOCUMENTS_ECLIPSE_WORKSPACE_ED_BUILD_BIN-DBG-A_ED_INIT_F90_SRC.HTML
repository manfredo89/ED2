     6) !    This subroutine will assign the longitude, latitude, and soil class for all non-empty !
     7) ! polygons.                                                                                !
     8) !------------------------------------------------------------------------------------------!
     9) subroutine set_polygon_coordinates()
    10)    use grid_coms     , only : ngrids    ! ! intent(in)
    11)    use ed_work_vars  , only : work_v    ! ! structure
    12)    use ed_node_coms  , only : mynum     ! ! intent(in)
    13)    use ed_state_vars , only : edgrid_g  & ! structure
    14)                             , gdpy      & ! intent(in)
    15)                             , edtype    ! ! intent(in)
    16)    implicit none 
    17)    !----- Local variables -----------------------------------------------------------------!
    18)    integer               :: ifm
    19)    integer               :: ipy
    20)    integer               :: npoly
    21)    type(edtype), pointer :: cgrid
    22)    !---------------------------------------------------------------------------------------!
    23) 
    24)    !---------------------------------------------------------------------------------------!
    25)    !  Ifort-11.0.083 complained about using edgrid_g(ifm) directly, replaced by cgrid.     !
    26)    !---------------------------------------------------------------------------------------!
    27)    gridloop: do ifm=1,ngrids
    28)       cgrid => edgrid_g(ifm)
    29) 
    30)       npoly=gdpy(mynum,ifm)
    31)       !----- Go through every polygon. ----------------------------------------------------!
    32)       polyloop: do ipy=1,npoly
    33)          cgrid%lon (ipy) = work_v(ifm)%glon   (ipy)
    34)          cgrid%lat (ipy) = work_v(ifm)%glat   (ipy)
    35)          cgrid%xatm(ipy) = work_v(ifm)%xid    (ipy)
    36)          cgrid%yatm(ipy) = work_v(ifm)%yid    (ipy)
    37)       end do polyloop
    38)       !------------------------------------------------------------------------------------!
    39)    end do gridloop
    40)    !---------------------------------------------------------------------------------------!
    41) 
    42) 
    43)    return
    44) end subroutine set_polygon_coordinates

    55) !    This subroutine assigns the site areas and the soil types when not running with       !
    56) ! ied_init_node 3 or 4.                                                                    !
    57) !------------------------------------------------------------------------------------------!
    58) subroutine set_site_defprops()
    59)    use grid_coms     , only : ngrids               & ! intent(in)
    60)                             , nzg                  ! ! intent(in)
    61)    use ed_work_vars  , only : work_v               ! ! structure
    62)    use ed_state_vars , only : edgrid_g             & ! structure
    63)                             , edtype               & ! structure
    64)                             , polygontype          & ! structure
    65)                             , allocate_polygontype ! ! subroutine
    66)    use soil_coms     , only : soil                 & ! intent(in)
    67)                             , slz                  ! ! intent(in)
    68)    use mem_polygons  , only : maxsite              ! ! intent(in)
    69)    use ed_misc_coms  , only : min_site_area        ! ! intent(in)
    70)    implicit none 
    71)    !----- Local variables -----------------------------------------------------------------!
    72)    type(edtype)     , pointer :: cgrid
    73)    type(polygontype), pointer :: cpoly
    74)    integer                    :: ifm
    75)    integer                    :: ipy
    76)    integer                    :: isi
    77)    integer                    :: itext
    78)    integer                    :: nsite
    79)    integer                    :: k
    80)    integer                    :: sc
    81)    real                       :: zmin
    82)    real                       :: fa
    83)    real                       :: fb
    84)    real                       :: te
    85)    real                       :: t0
    86)    real                       :: k0
    87)    real                       :: text_area
    88)    real                       :: text_area_i
    89)    !---------------------------------------------------------------------------------------!
    90) 
    91) 
    92) 
    93)    gridloop: do ifm=1,ngrids
    94)       cgrid => edgrid_g(ifm)
    95) 
    96)       polyloop: do ipy=1,cgrid%npolygons
    97)       
    98)          !----- Initialise load adjacency with dummy value. -------------------------------!
    99)          cgrid%load_adjacency(ipy) = 0
   100) 
   101)          !----- Alias to current polygon. -------------------------------------------------!
   102)          cpoly => cgrid%polygon(ipy)
   103) 
   104)          !---------------------------------------------------------------------------------!
   105)          !    Find the total usable area and the number of sites that will be allocated.   !
   106)          !---------------------------------------------------------------------------------!
   107)          nsite     = min(maxsite, count(work_v(ifm)%soilfrac(:,ipy) > min_site_area))
   108)          text_area = sum(work_v(ifm)%soilfrac(1:nsite,ipy))
   109)          !----- Sanity check. -------------------------------------------------------------!
   110)          if (nsite == 0 .or. text_area <= 0.) then
   111)             write (unit=*,fmt='(a)'          ) '------------------------------------------'
   112)             write (unit=*,fmt='(a,1x,i12)'   ) ' + NSITE     = ',nsite
   113)             write (unit=*,fmt='(a,1x,es12.5)') ' + TEXT_AREA = ',text_area
   114)             write (unit=*,fmt='(a,1x,es12.5)') ' + MIN_VALID = ',min_site_area
   115)             write (unit=*,fmt='(a,1x,es12.5)') ' + MIN_AREA  = '                           &
   116)                                               ,minval(work_v(ifm)%soilfrac(:,ipy))
   117)             write (unit=*,fmt='(a,1x,es12.5)') ' + MAX_AREA  = '                           &
   118)                                               ,maxval(work_v(ifm)%soilfrac(:,ipy))
   119)             write (unit=*,fmt='(a)'          ) '------------------------------------------'
   120)             call fatal_error('Invalid number of sites!','set_site_defprops'                &
   121)                             ,'ed_init.f90')
   122)          else
   123)             text_area_i = 1. / text_area
   124)          end if
   125) 
   126) 
   127)          !------ Allocate the number of sites that have enough area. ----------------------!
   128)          call allocate_polygontype(cpoly,nsite)
   129)          call soil_default_fill(cgrid,ifm,ipy)
   130) 
   131)          !---------------------------------------------------------------------------------!
   132)          !     Populate the sites.                                                         !
   133)          !---------------------------------------------------------------------------------!
   134)          isi = 0
   135)          siteloop: do itext=1,maxsite
   136)             if (work_v(ifm)%soilfrac(itext,ipy) > min_site_area) then
   137)                !----- Update counter. -----------------------------------------------------!
   138)                isi = isi +1
   139) 
   140)                !---------------------------------------------------------------------------!
   141)                !     Initialise area, using the soil texture area scaled by the total be-  !
   142)                ! ing used.                                                                 !
   143)                !---------------------------------------------------------------------------!
   144)                cpoly%area(isi) = work_v(ifm)%soilfrac(itext,ipy) * text_area_i
   145)                !---------------------------------------------------------------------------!
   146) 
   147) 
   148)                !---------------------------------------------------------------------------!
   149)                !     Use the soil type and populate the site-level soil texture.           !
   150)                !---------------------------------------------------------------------------!
   151)                do k=1,nzg
   152)                   cpoly%ntext_soil(k,isi) = work_v(ifm)%ntext(itext,ipy)
   153)                end do
   154)                !---------------------------------------------------------------------------!
   155) 
   156) 
   157)                !---------------------------------------------------------------------------!
   158)                !       Set soil moisture decay function, based on second layer's K value.  !
   159)                ! We use the second layer instead of the top in case top is organic/peat.   !
   160)                !---------------------------------------------------------------------------!
   161)                sc = cpoly%ntext_soil(nzg-1,isi)
   162)                cpoly%moist_f(isi) = -log(soil(sc)%slcons / soil(sc)%slcons0) / 2.0
   163)                !---------------------------------------------------------------------------!
   164) 
   165) 
   166) 
   167) 
   168)                !----- Derive adjustments to f. --------------------------------------------!
   169)                zmin = slz(cpoly%lsl(isi))
   170)                fa   = -1.0/zmin !! should be 1/(depth to bedrock)
   171)                if(cpoly%moist_f(isi)*zmin < 0.0) then
   172)                   fb = cpoly%moist_f(isi)/(1.0-exp(cpoly%moist_f(isi)*zmin))
   173)                   cpoly%moist_f(isi) = max(fa,fb)
   174)                else
   175)                   cpoly%moist_f(isi) = fa
   176)                endif
   177) 
   178)                cpoly%sitenum(isi)   = 1
   179)                cpoly%elevation(isi) = 0.0
   180)                cpoly%slope(isi)     = 0.0
   181)                cpoly%aspect(isi)    = 0.0
   182)                cpoly%TCI(isi)       = 0.0
   183)             end if
   184)          end do siteloop
   185) 
   186)          !----- This should not happen in this sub-routine, but in case things change... --!
   187)          if (cgrid%load_adjacency(ipy) /= 0) then
   188)             call calc_flow_routing(cgrid,ipy)
   189)          end if
   190)          !---------------------------------------------------------------------------------!
   191) 
   192)          !---------------------------------------------------------------------------------!
   193)          !     Not sure what these things do, just copying from hydrology...               !
   194)          !---------------------------------------------------------------------------------!
   195)          !----- Part 1. -------------------------------------------------------------------!
   196)          Te = 0.0 
   197)          do isi = 1,cpoly%nsites
   198)             sc = cpoly%ntext_soil(nzg-1,isi)
   199)             K0 = soil(sc)%slcons0
   200)             T0 = K0 / cpoly%moist_f(isi)
   201)             Te = Te + T0*cpoly%area(isi)
   202)          end do
   203)          cgrid%Te(ipy) = Te
   204)          !----- Part 2. -------------------------------------------------------------------!
   205)          cgrid%wbar(ipy) = 0.0
   206)          do isi = 1,cpoly%nsites
   207)             sc = cpoly%ntext_soil(nzg-1,isi)
   208)             K0 = soil(sc)%slcons0
   209)             T0 = K0 / cpoly%moist_f(isi)
   210)             cpoly%moist_W(isi) = cpoly%TCI(isi) + log(Te) - log(T0)
   211)             cgrid%wbar(ipy)    = cgrid%wbar(ipy) + cpoly%moist_W(isi) * cpoly%area(isi)
   212)          end do
   213)          !---------------------------------------------------------------------------------!
   214)       end do polyloop
   215)       !------------------------------------------------------------------------------------!
   216) 
   217)    end do gridloop
   218)    !---------------------------------------------------------------------------------------!
   219) 
   220) 
   221)    return
   222) end subroutine set_site_defprops

   233) !     This subroutine fills the lsl, soil colour, and soil texture based on the defaults.  !
   234) ! In case isoildepthflg was zero, then the layer_index matrix was filled with ones, so we  !
   235) ! do not need to worry about this here.                                                    !
   236) !------------------------------------------------------------------------------------------!
   237) subroutine soil_default_fill(cgrid,ifm,ipy)
   238)    
   239)    use soil_coms     , only : layer_index ! ! intent(in)
   240)    use ed_state_vars , only : edtype      & ! structure
   241)                             , polygontype ! ! structure
   242)    use ed_work_vars  , only : work_v      ! ! structure
   243)    use grid_coms     , only : nzg         ! ! intent(in)
   244)    implicit none
   245)    !----- Arguments -----------------------------------------------------------------------!
   246)    type(edtype) , target       :: cgrid
   247)    integer      , intent(in)   :: ifm
   248)    integer      , intent(in)   :: ipy
   249)    !----- Local variables -----------------------------------------------------------------!
   250)    type(polygontype) , pointer :: cpoly
   251)    integer                     :: ilat_bin
   252)    integer                     :: ilon_bin
   253)    integer                     :: isi
   254)    integer                     :: k
   255)    !---------------------------------------------------------------------------------------!
   256) 
   257) 
   258)    ilat_bin = min(180,int(90.0 - cgrid%lat(ipy)) + 1)
   259)    ilon_bin = int(180.0 + cgrid%lon(ipy)) + 1
   260) 
   261)    cpoly => cgrid%polygon(ipy)
   262)    do isi=1,cpoly%nsites
   263)       !------------------------------------------------------------------------------------!
   264)       !    Require at least 2 layers.  This requirement was taken in consideration when    !
   265)       ! layer_index was filled at the first initialization, so it is safe to just copy.    !
   266)       !------------------------------------------------------------------------------------!
   267)       cpoly%lsl(isi) =layer_index(ilat_bin,ilon_bin) 
   268)       !------------------------------------------------------------------------------------!
   269) 
   270) 
   271)       !------------------------------------------------------------------------------------!
   272)       !     Use the commonest soil type and populate the site-level soil texture.          !
   273)       !------------------------------------------------------------------------------------!
   274)       do k=1,nzg
   275)          cpoly%ntext_soil(k,isi) = work_v(ifm)%ntext(1,ipy)
   276)       end do
   277)       !------------------------------------------------------------------------------------!
   278) 
   279) 
   280)       !------------------------------------------------------------------------------------!
   281)       !     Use the polygon-level soil colour to populate the site-level.                  !
   282)       !------------------------------------------------------------------------------------!
   283)       cpoly%ncol_soil(isi) = work_v(ifm)%nscol(ipy)
   284)       !------------------------------------------------------------------------------------!
   285)    end do
   286)    !---------------------------------------------------------------------------------------!
   287) 
   288)    return
   289) end subroutine soil_default_fill

   300) !    Several Procedures requiring ASCII reads follow.  If this is a parallel run, then the !
   301) ! nodes must queue.  Since we will access sequential format files, each node needs to wait !
   302) ! its turn to access it... MPI_File commands won't work with ASCII files, so that's a      !
   303) ! bottleneck here. If the run is serial mynum=nnodetot, so I don't need to wait.           !
   304) !------------------------------------------------------------------------------------------!
   305) subroutine load_ecosystem_state()
   306)    use landuse_init_module
   307)    use ed_nbg_init
   308)    use ed_misc_coms      , only : ied_init_mode   & ! intent(in)
   309)                                 , ibigleaf        ! ! intent(in)
   310)    use phenology_startup , only : phenology_init  ! ! intent(in)
   311)    use ed_node_coms      , only : mynum           ! ! intent(in)
   312)    use grid_coms         , only : ngrids          ! ! intent(in)
   313)    use ed_state_vars     , only : edgrid_g        ! ! structure
   314) 
   315)    implicit none
   316) #if defined(RAMS_MPI)
   317)    include 'mpif.h'
   318) #endif
   319)    !----- Local variables -----------------------------------------------------------------!
   320)    integer                :: igr
   321)    integer                :: ping 
   322)    !---------------------------------------------------------------------------------------!
   323) 
   324)    ping = 741776
   325) 
   326) 
   327)    if (mynum == 1) write(unit=*,fmt='(a)') ' + Doing sequential initialization over nodes.'
   328) 
   329) 
   330)    !---------------------------------------------------------------------------------------!
   331)    ! STEP 1: Read in Site files and initialize hydrologic adjacencies.                     !
   332)    !---------------------------------------------------------------------------------------!
   333) #if defined(RAMS_MPI)
   334)    if (mynum /= 1) &
   335)       call MPI_Recv(ping,1,MPI_INTEGER,recvnum,100,MPI_COMM_WORLD,MPI_STATUS_IGNORE,ierr)
   336) #endif
   337) 
   338)    select case (ied_init_mode)
   339)    case (3)
   340)       !----- Hydrology run.  Use specific scheme. -----------------------------------------!
   341)       do igr = 1,ngrids
   342)          call read_site_file(edgrid_g(igr),igr)
   343)       end do
   344)    case (4,7)
   345)       continue
   346)    case default
   347)       call set_site_defprops()
   348)    end select
   349) 
   350) #if defined(RAMS_MPI)
   351)    if (mynum < nnodetot) call MPI_Send(ping,1,MPI_INTEGER,sendnum,100,MPI_COMM_WORLD,ierr)
   352) #endif
   353)    !---------------------------------------------------------------------------------------!
   354) 
   355) 
   356) 
   357) 
   358) 
   359)    !---------------------------------------------------------------------------------------!
   360)    ! STEP 3: Do ASCII type restart initialization of site patch and cohort biophysical     !
   361)    !         states.                                                                       !
   362)    !---------------------------------------------------------------------------------------!
   363) #if defined(RAMS_MPI)
   364)    if (mynum /= 1) &
   365)       call MPI_RECV(ping,1,MPI_INTEGER,recvnum,101,MPI_COMM_WORLD,MPI_STATUS_IGNORE,ierr)
   366) #endif
   367) 
   368)    select case (ied_init_mode)
   369)    case (-8,-1,0)
   370)    
   371)       select case (ibigleaf)
   372)       case (0)
   373)          !----- Initialize everything with near-bare ground -------------------------------!
   374)          if (mynum /= 1) then
   375)             write (unit=*,fmt='(a)') ' + Doing near bare ground initialization...'
   376)          end if
   377)          do igr=1,ngrids
   378)               call near_bare_ground_init(edgrid_g(igr))
   379)          end do
   380) 
   381)       case (1)
   382)          !----- Initialize everything with near-bare ground -------------------------------!
   383)          if (mynum /= 1) then
   384)             write(unit=*,fmt='(a)') ' + Doing near-bare-ground big-leaf initialization...'
   385)          end if
   386)          do igr=1,ngrids
   387)             call near_bare_ground_big_leaf_init(edgrid_g(igr))
   388)          end do
   389)       end select
   390) 
   391)    case (1,2,3,6)
   392)       !----- Initialize with ED1-type restart information. --------------------------------!
   393)       write(unit=*,fmt='(a,i3.3)') ' + Initializing from ED restart file. Node: ',mynum
   394)       call read_ed10_ed20_history_file
   395)       select case (ibigleaf)
   396)       case (1)
   397)          do igr=1,ngrids
   398)             call ed_bigleaf_init(edgrid_g(igr))
   399)          end do
   400)       end select
   401) 
   402)    case (4)   
   403)       write(unit=*,fmt='(a,i3.3)') ' + Initializing from ED2.1 state file. Node: ',mynum
   404)       call read_ed21_history_file
   405)       select case (ibigleaf)
   406)       case (1)
   407)          do igr=1,ngrids
   408)             call ed_bigleaf_init(edgrid_g(igr))
   409)          end do
   410)       end select
   411) 
   412)    case (5,99)
   413)       write(unit=*,fmt='(a,i3.3)')                                                         &
   414)           ' + Initializing from a collection of ED2.1 state files. Node: ',mynum
   415)       call read_ed21_history_unstruct
   416)       select case (ibigleaf)
   417)       case (1)
   418)          do igr=1,ngrids
   419)             call ed_bigleaf_init(edgrid_g(igr))
   420)          end do
   421)       end select
   422) 
   423)    case (7)
   424)       write(unit=*,fmt='(a,i3.3)')                                                         &
   425)            ' + Initializing Soils+Veg from nearest neighbor ED2.1 files. Node: ',mynum
   426)       call read_ed21_polyclone
   427)       select case (ibigleaf)
   428)       case (1)
   429)          do igr=1,ngrids
   430)             call ed_bigleaf_init(edgrid_g(igr))
   431)          end do
   432)       end select
   433)       
   434)    end select
   435) 
   436) 
   437) 
   438) #if defined(RAMS_MPI)
   439)    if (mynum < nnodetot) call MPI_Send(ping,1,MPI_INTEGER,sendnum,101,MPI_COMM_WORLD,ierr)
   440) #endif
   441)    !---------------------------------------------------------------------------------------!
   442) 
   443) 
   444) 
   445) 
   446)    !---------------------------------------------------------------------------------------!
   447)    ! STEP 4: Initialize phenology parameters and thermal sums.                             !
   448)    !---------------------------------------------------------------------------------------!
   449) #if defined(RAMS_MPI)
   450)    if (mynum /= 1) &
   451)       call MPI_Recv(ping,1,MPI_INTEGER,recvnum,102,MPI_COMM_WORLD,MPI_STATUS_IGNORE,ierr)
   452) #endif
   453) 
   454)    write(unit=*,fmt='(a,i3.3)') ' + Initializing phenology. Node: ',mynum
   455)    call phenology_init()
   456) 
   457) #if defined(RAMS_MPI)
   458)    if (mynum < nnodetot ) call MPI_Send(ping,1,MPI_INTEGER,sendnum,102,MPI_COMM_WORLD,ierr)
   459) #endif
   460)    !---------------------------------------------------------------------------------------!
   461) 
   462) 
   463) 
   464)    !---------------------------------------------------------------------------------------!
   465)    ! STEP 5: Initialize anthropogenic disturbance.                                         !
   466)    !---------------------------------------------------------------------------------------!
   467) #if defined(RAMS_MPI)
   468)    if (mynum /= 1) &
   469)      call MPI_Recv(ping,1,MPI_INTEGER,recvnum,103,MPI_COMM_WORLD,MPI_STATUS_IGNORE,ierr)
   470) #endif
   471) 
   472)    write(unit=*,fmt='(a,i3.3)')                                                            &
   473)       ' + Initializing anthropogenic disturbance forcing. Node: ',mynum
   474) 
   475)    call landuse_init()
   476) 
   477) #if defined(RAMS_MPI)
   478)    if (mynum < nnodetot ) call MPI_Send(ping,1,MPI_INTEGER,sendnum,103,MPI_COMM_WORLD,ierr)
   479) #endif
   480)    !---------------------------------------------------------------------------------------!
   481) 
   482)    if (mynum == 1) then
   483)       do igr=1,ngrids
   484)          call ed_newgrid(igr)
   485)          call print_soil_info(edgrid_g(igr),igr)
   486)       end do
   487)    end if
   488) 
   489) 
   490)    return
   491) end subroutine load_ecosystem_state

   550)    real                   :: slz0
   551)    real                   :: ezg
   552)    !---------------------------------------------------------------------------------------!
   553) 
   554) 
   555)    !----- Soil vertical grid spacing arrays (some with timestep info). --------------------!
   556)    slz (nzg+1) = 0.
   557)    slz8(nzg+1) = 0.d0
   558) 
   559)    do k = 1,nzg
   560)       dslz    (k) = slz(k+1) - slz(k)
   561)       dslzo2  (k) = .5 * dslz(k)
   562)       dslzi   (k) = 1. / dslz(k)
   563)       dslzidt (k) = dslzi(k) * dtlsm
   564)       slzt    (k) = .5 * (slz(k) + slz(k+1))
   565)       slz8    (k) = dble(slz    (k))
   566)       dslz8   (k) = dble(dslz   (k))
   567)       dslzo28 (k) = dble(dslzo2 (k))
   568)       dslzi8  (k) = dble(dslzi  (k))
   569)       dslzidt8(k) = dble(dslzidt(k))
   570)       slzt8   (k) = dble(slzt   (k))
   571)    end do
   572) 
   573)    !----- Find the exponential increase factor to estimate the bottom boundary condition. -!
   574)    ezg  = log(slz(1)/slz(nzg)) / log(real(nzg))
   575)    slz0 = slz(1) * (real(nzg+1)/real(nzg))**ezg
   576)    !---------------------------------------------------------------------------------------!
   577) 
   578) 
   579)    !----- Find the thickness of the bottom boundary condition layer. ----------------------!
   580)    dslz    (0) = slz(1) - slz0
   581)    dslzo2  (0) = .5 * dslz(0)
   582)    dslzi   (0) = 1. / dslz(0)
   583)    dslzidt (0) = dslzi(0) * dtlsm
   584)    dslz8   (0) = dble(dslz   (0))
   585)    dslzo28 (0) = dble(dslzo2 (0))
   586)    dslzi8  (0) = dble(dslzi  (0))
   587)    dslzidt8(0) = dble(dslzidt(0))
   588)    !---------------------------------------------------------------------------------------!
   589) 
   590) 
   591)    !----- Find the height at the middle of the bottom boundary condition. -----------------!
   592)    slzt   (0) = .5 * (slz0 + slz(1))
   593)    slzt8  (0) = dble(slzt(0))
   594)    !---------------------------------------------------------------------------------------!
   595) 
   596)    do k = 1,nzg
   597)       dslzt    (k) = slzt(k) - slzt(k-1)
   598)       dslzti   (k) = 1. / dslzt(k)
   599)       dslztidt (k) = dslzti(k) * dtlsm
   600)       dslzt8   (k) = dble(dslzt   (k))
   601)       dslzti8  (k) = dble(dslzti  (k))
   602)       dslztidt8(k) = dble(dslztidt(k))
   603)    end do
   604) 
   605) 
   606)    !----- Soil constants. -----------------------------------------------------------------!
   607)    refdepth = -0.5
   608) 
   609)    do nnn = 1,ed_nstyp
   610)       if (nnn /= 13) then
   611)          fhydraul(nnn) = log (soil(nnn)%slcons / soil(nnn)%slcons0) / refdepth
   612)       else
   613)          fhydraul(nnn) = 0.
   614)       end if
   615)       do k = 0,nzg
   616)       
   617)          select case (ipercol)
   618)          case (0,1)
   619)             !----- Original form, constant with depth.  -----------------------------------!
   620)             slcons1(k,nnn) = soil(nnn)%slcons
   621)          case (2)
   622)             !------------------------------------------------------------------------------!
   623)             !    TOPMODEL form, similar to CLM.  Here we use the same definition of slcons !
   624)             ! from Cosby et al. (1984) because it has a stronger spread and it accounts    !
   625)             ! for sand and clay contents.                                                  !
   626)             !------------------------------------------------------------------------------!
   627)             slcons1(k,nnn) = soil(nnn)%slcons * exp ( - slzt(k) / refdepth)
   628)          end select
   629) 
   630)          !------ Find the double precision. -----------------------------------------------!
   631)          slcons18(k,nnn) = dble(slcons1(k,nnn))
   632)       end do
   633) 
   634)       slden    (nnn) =  soil(nnn)%slden    
   635)    end do
   636) 
   637)    !----- Defining some snow thickness variables ------------------------------------------!
   638)    stretch = 2.0
   639)    do kzs = 1,nzs
   640)       thik          = 1.0
   641)       thicknet(kzs) = 0.0
   642)       do k = 1,(kzs+1)/2
   643)          thick(k,kzs)       = thik
   644)          thick(kzs+1-k,kzs) = thik
   645)          thicknet(kzs)      = thicknet(kzs) + 2. * thik
   646)          thik               = thik * stretch
   647)       end do
   648)       if (mod(kzs, 2) .ne. 0) thicknet(kzs) = thicknet(kzs) - thik/stretch
   649)       do k = 1,kzs
   650)          thick(k,kzs) = thick(k,kzs) / thicknet(kzs)
   651)       end do
   652)    end do
   653) 
   654)    !----- Assigning some soil grid-dependent RK4 variables --------------------------------!
   655)    rk4min_sfcw_mass  = rk4min_sfcw_moist * wdns   * dslz(nzg)
   656)    rk4min_virt_water = rk4min_virt_moist * wdns   * dslz(nzg)
   657) 
   658)    return
   659) end subroutine sfcdata_ed

   748) subroutine read_site_file(cgrid,igr)
   749)    ! function for loading sites within polygons and initializing polygon parms
   750)    ! call prior to loading pss/css files but after basic polygon established
   751)    use soil_coms, only: soil,slz
   752)    use grid_coms, only: nzg
   753)    use ed_misc_coms, only: sfilin, vary_elev, vary_rad, vary_hyd
   754)    use mem_polygons, only: edres
   755)    use ed_state_vars, only: edtype,polygontype,sitetype,allocate_polygontype
   756)    use ed_max_dims, only: max_site,n_pft,str_len
   757) 
   758)    implicit none
   759)    integer :: igr
   760)    logical :: no_rad   = .false.  !! true turns effect OFF
   761)    logical :: no_lapse = .false.
   762)    logical :: no_hyd   = .false.
   763) 
   764)    character(len=str_len) :: site_name,pss_name,css_name
   765)   
   766)    type(edtype) :: cgrid
   767)    type(polygontype),pointer :: cpoly
   768) 
   769)    character(len=200) :: cdummy,cdummy2
   770)    integer :: i,nsc,nsites=1
   771)    real(kind=8) :: area_sum = 0.0d+0
   772)    integer :: sc             ! soil classa
   773)    integer :: fformat = 0    ! file format
   774)    logical :: fexist         ! file exists
   775)    real :: Te,T0,K0
   776)    real :: fa,fb,zmin=-2.0
   777)    integer :: sitenum,get_site_line,get_mat_line,get_header,found_mat_header=0,lcount=0,mcount=0
   778)    real :: area,TCI,elevation,slope,aspect
   779)    integer,allocatable :: soilclass(:)
   780)    integer :: ipy,isi
   781)    integer :: ierr
   782) 
   783)    if(vary_elev == 0)  no_lapse = .true.
   784)    if(vary_rad == 0)   no_rad = .true.
   785)    if(vary_hyd == 0) no_hyd = .true.
   786) 
   787) 
   788) 
   789) 
   790)    ! ASSUMING FOR NOW THAT THERE IS NO WATER SITE
   791) 
   792)    ! init values
   793)    if (associated(cgrid%load_adjacency)) cgrid%load_adjacency = 0
   794) 
   795)   
   796)    do ipy = 1,cgrid%npolygons
   797) 
   798)       cpoly => cgrid%polygon(ipy)
   799) 
   800)       call create_ed10_ed20_fname(cgrid%lat(ipy), edres, cgrid%lon(ipy) &
   801)                                  ,trim(sfilin(igr)),pss_name,css_name,site_name)
   802)       !! check if site file exists
   803)       inquire(file=trim(site_name),exist=fexist)
   804)       if(.not.fexist) then
   805)          print*,"error opening site file ",site_name
   806)          print*,"setting ied_init_mode to 2 and loading as single site"
   807)       end if
   808) 
   809)       
   810)       ! If there is no terrestrial site
   811)       if(.not. fexist) then  !! have no site file or ied_init_mode is not 3
   812) 
   813)          ! Allocate single site vector information to
   814)          ! the swap polygon
   815)          
   816)          call allocate_polygontype(cpoly,1)
   817)          call soil_default_fill(cgrid,igr,ipy)
   818) 
   819)          cpoly%area(1) = 1.0             ! Initialize the area to all
   820) 
   821)          ! Set soil moisture decay function, based on second layer's K value
   822)          ! use the second layer instead of the top in case top is organic/peat
   823)          sc = cpoly%ntext_soil(nzg-1,1)
   824)          cpoly%moist_f(1) = -log(soil(sc)%slcons / soil(sc)%slcons0) / 2.0
   825) 
   826)          !! derive adjustments to f
   827)          zmin = slz(cpoly%lsl(1))
   828)          fa = -1.0/zmin !! should be 1/(depth to bedrock)
   829)          if(cpoly%moist_f(1)*zmin < 0.0) then
   830)             fb = cpoly%moist_f(1)/(1.0-exp(cpoly%moist_f(1)*zmin))
   831)             cpoly%moist_f(1) = max(fa,fb)
   832)          else
   833)             cpoly%moist_f(1) = fa
   834)          endif
   835) 
   836)          cpoly%sitenum(1)   = 1
   837)          cpoly%elevation(1) = 0.0
   838)          cpoly%slope(1)     = 0.0
   839)          cpoly%aspect(1)    = 0.0
   840)          cpoly%TCI(1)       = 0.0
   841) 
   842)       else
   843) 
   844)          !! Read data from site file
   845)          open(unit=12,file=trim(site_name),form='formatted',status='old')
   846) 
   847)          !/* read file format line */
   848)          read(unit=12,fmt=*)cdummy,nsites,cdummy2,fformat
   849)          !/*line format: "nsites <nsites> format <fformat>" */   
   850)          
   851) !         print*,"reading",nsites,"sites using file format",fformat
   852) 
   853)          call allocate_polygontype(cpoly,nsites)
   854)          call soil_default_fill(cgrid,igr,ipy)
   855)      
   856)          if(fformat <=0 .or. fformat > 3) then
   857)             print*,""
   858)             print*,"ERROR :: unrecognized file format specifier in",site_name
   859)             stop
   860)          endif
   861)          if(fformat == 2) cgrid%load_adjacency(ipy) = 1
   862)          nsc = 1
   863)          if(fformat == 3) nsc = nzg
   864) 
   865)          !/* discard file header line */
   866)          read(unit=12,fmt=*)
   867)          
   868)          !/* read data*/
   869)          allocate(soilclass(nzg))
   870)          lcount = 0 
   871)          mcount = 0
   872)          found_mat_header = 0
   873)          isi = 0
   874)          area_sum = 0.0d+0
   875)          count_sites: do
   876) 
   877)             isi = isi + 1  ! The site counter
   878)             
   879)             !           read(12,*,iostat=ierr)time,pname,trk,age,area,fsc,stsc,  &
   880)             !                stsl,ssc,psc,msn,fsn,water(1:nwater)
   881)             !           if(ierr /= 0)exit count_patches
   882)             
   883)             lcount = lcount + 1 !! line counter
   884)             get_site_line=0
   885)             get_mat_line=0
   886)             get_header=0 !! line flags
   887)             
   888)             !/********* decide what type of line to read ***************/
   889)             select case (fformat)
   890)             case (1,3)
   891)                if(lcount <= nsites) then !we're reading data
   892)                   get_site_line=1
   893)                   get_mat_line=0
   894)                   get_header=0
   895)                else                      !we're past data, discard
   896)                   get_site_line=0
   897)                   get_mat_line=0
   898)                   get_header=1             
   899)                endif
   900)             case (2)
   901)                if(lcount <= nsites) then  !know we're in the site section
   902)                   get_site_line=1
   903)                   get_mat_line=0
   904) 
   905)                elseif(found_mat_header.eq.1)then !know we're in the matrix section
   906)                   if(mcount < nsites)then
   907)                      get_site_line = 0
   908)                      get_mat_line=1
   909)                      get_header = 0
   910)                   else  !//already got all the matrix lines, discard what's left
   911)                      get_site_line = 0
   912)                      get_mat_line = 0
   913)                      get_header = 1
   914)                   endif
   915)                else 
   916)                   
   917)                   !!assume line is header, remove and then assume header found
   918)                   
   919)                   get_site_line = 0
   920)                   get_mat_line = 0
   921)                   get_header = 1
   922)                   found_mat_header=1
   923)                   
   924)                endif
   925)                
   926)             end select
   927) 
   928) !            print*,"line indicators",get_site_line,get_mat_line,get_header
   929)     
   930)             if(get_site_line == 1)then   !/********** READ SITE LINE ***************/
   931)                
   932)                !/* line format: sitenum, area, TCI, elevation, slope, aspect,ntext_soil(s) */              
   933)                read(unit=12,fmt=*,iostat=ierr)sitenum,area,TCI,elevation,slope,aspect,soilclass(1:nsc)
   934)                if(ierr == 0) then
   935)                   !/*create data object for each new site */
   936) !                  print*,sitenum, area, TCI, elevation,slope,aspect,soilclass(1:nsc)
   937)                   cpoly%area(isi) = area            ! Initialize the area to all
   938) 
   939)                   area_sum = area_sum + dble(area)
   940)                   cpoly%sitenum(isi)      = sitenum
   941)                   cpoly%elevation(isi)    = elevation
   942)                   cpoly%slope(isi)        = slope
   943)                   cpoly%aspect(isi)       = aspect
   944)                   cpoly%TCI(isi)          = TCI+13.96962
   945) 
   946)                   !! flags to turn effects off
   947)                   if(no_lapse) cpoly%elevation(isi) = 0.0
   948)                   if(no_hyd)   cpoly%TCI(isi)       = 8.0
   949)                   if(no_rad) then
   950)                                cpoly%slope(isi)     = 0.0
   951)                                cpoly%aspect(isi)    = 0.0
   952)                   end if
   953) 
   954) !print*,"SITE",cpoly%elevation(isi),cpoly%TCI(isi),cpoly%slope(isi),cpoly%aspect(isi)
   955) 
   956)                   if(fformat == 3) then
   957)                      do i=1,nzg
   958)                         cpoly%ntext_soil(i,isi) = soilclass(i)
   959)                      end do
   960)                   else
   961)                      do i=1,nzg
   962)                         cpoly%ntext_soil(i,isi) = soilclass(1)
   963)                      end do
   964)                   end if
   965)                   
   966)                   !//Currently do nothing with setting site-level soils
   967) 
   968)                   sc = cpoly%ntext_soil(nzg-1,1)
   969)                   cpoly%moist_f(isi) = -log(soil(sc)%slcons / soil(sc)%slcons0) / 2.0
   970)                   !! derive adjustments to f
   971)                   zmin = slz(cpoly%lsl(isi))
   972)                   fa = -1.0/zmin !! should be 1/(depth to bedrock)
   973)                   if(cpoly%moist_f(isi)*zmin < 0.0) then
   974)                      fb = cpoly%moist_f(isi)/(1.0-exp(cpoly%moist_f(isi)*zmin))
   975)                      cpoly%moist_f(isi) = max(fa,fb)
   976)                   else
   977)                      cpoly%moist_f(isi) = fa
   978)                   endif
   979)                end if  ! end valid line
   980)             end if        !//********** END READ SITE LINE**************
   981) 
   982)             ! RGK
   983)             if(get_header == 1) read(unit=12,fmt=*,iostat=ierr) !/* discard file line */
   984)             ! RGK
   985)             if(get_mat_line == 1) then
   986) 
   987)                !/*read line into site adjacency matrix*/
   988)                
   989)                read(unit=12,fmt=*,iostat=ierr) cgrid%site_adjacency(mcount,1:(nsites+1),ipy)
   990) 
   991)                mcount = mcount+1
   992)                
   993)             endif
   994)             if(ierr /= 0) exit count_sites
   995)         
   996)          end do count_sites
   997)          deallocate(soilclass)
   998)       end if
   999) 
  1000)       !adjust areas
  1001)       !assume that if area_sum ~ 1 need to renormalize terrestrial
  1002)       if( area_sum > 0.995d+0) then
  1003)          cpoly%area(:) = real(dble(cpoly%area(:))/area_sum)
  1004)       end if
  1005) 
  1006) 
  1007)       if(cgrid%load_adjacency(ipy) /= 0) then  
  1008)          call calc_flow_routing(cgrid,ipy)
  1009)       endif
  1010) 
  1011)       ! calculate summary stats - pass 1: Te
  1012)       ! On terrestrial site still
  1013)       Te = 0.0 
  1014)       area_sum = 0.0d+0
  1015)       do isi = 1,cpoly%nsites
  1016)          sc = cpoly%ntext_soil(nzg-1,isi)
  1017)          K0 = soil(sc)%slcons0
  1018)          T0 = K0/cpoly%moist_f(isi)
  1019)          Te = Te + T0*cpoly%area(isi)
  1020)          area_sum = area_sum + dble(cpoly%area(isi))
  1021)       end do
  1022) 
  1023)       Te = Te/real(area_sum)
  1024)       cgrid%Te(ipy) = Te
  1025) 
  1026)       !pass 2: W, Wbar
  1027) 
  1028)       cgrid%wbar(ipy) = 0.0
  1029) 
  1030)       do isi = 1,cpoly%nsites
  1031)          sc = cpoly%ntext_soil(nzg-1,isi)
  1032)          K0 = soil(sc)%slcons0

  1079)       ihys=find_rank(i,nsites,hyrank)
  1080)       ines=find_rank(i+1,nsites,hyrank)
  1081)       cgrid%site_adjacency(ihys,ihys,ipy) = 2.0*side*(side-1.0)*cpoly%area(ihys) &

  1086)    ihys=find_rank(nsites,nsites,hyrank)
  1087)    ines=find_rank(1,nsites,hyrank)

  1093)    do ihys=1,nsites
