     1) !==========================================================================================!
     2) !==========================================================================================!
     3) !    This subroutine will solve the within canopy two-stream radiation for shortwave,      !
     4) ! considering the light absorption by leaves, and branches if the user wants so.  In both  !
     5) ! cases we consider the clumpiness effect (i.e., the fact that leaves and branches aren't  !
     6) ! randomly distributed).  The crown area will be also considered when the user wants it.   !
     7) !------------------------------------------------------------------------------------------!
     8) subroutine old_sw_two_stream (salbedo_par,salbedo_nir,scosaoi,ncoh,pft,lai,wai,canopy_area &
     9)                              ,radprof_flip,par_beam_flip,par_diff_flip,sw_abs_beam_flip    &
    10)                              ,sw_abs_diff_flip,dw_parlo_beam,dw_parlo_diff                 &
    11)                              ,uw_parhi_diff,dw_nirlo_beam,dw_nirlo_diff,uw_nirhi_diff      &
    12)                              ,par_level_beam,par_level_diffd,par_level_diffu               &
    13)                              ,light_level,light_beam_level                                 &
    14)                              ,light_diff_level)
    15) 
    16)    use ed_max_dims          , only : n_pft                   & ! intent(in)
    17)                                    , n_radprof               ! ! intent(in)
    18)    use rk4_coms             , only : tiny_offset             ! ! intent(in)
    19)    use canopy_radiation_coms, only : leaf_backscatter_nir    & ! intent(in)
    20)                                    , leaf_backscatter_vis    & ! intent(in)
    21)                                    , leaf_scatter_nir        & ! intent(in)
    22)                                    , leaf_scatter_vis        & ! intent(in)
    23)                                    , wood_backscatter_nir    & ! intent(in)
    24)                                    , wood_backscatter_vis    & ! intent(in)
    25)                                    , wood_scatter_nir        & ! intent(in)
    26)                                    , wood_scatter_vis        & ! intent(in)
    27)                                    , clumping_factor         & ! intent(in)
    28)                                    , phi1                    & ! intent(in)
    29)                                    , phi2                    & ! intent(in)
    30)                                    , mu_bar                  & ! intent(in)
    31)                                    , par_beam_norm           & ! intent(in)
    32)                                    , par_diff_norm           & ! intent(in)
    33)                                    , nir_beam_norm           & ! intent(in)
    34)                                    , nir_diff_norm           & ! intent(in)
    35)                                    , cosz_min8               ! ! intent(in)
    36)    implicit none
    37)    !----- Arguments -----------------------------------------------------------------------!
    38)    integer, dimension(ncoh)       , intent(in)    :: pft
    39)    integer                        , intent(in)    :: ncoh
    40)    real(kind=8), dimension(ncoh)  , intent(in)    :: lai
    41)    real(kind=8), dimension(ncoh)  , intent(in)    :: wai
    42)    real(kind=8), dimension(ncoh)  , intent(in)    :: canopy_area
    43)    real                           , intent(in)    :: salbedo_par
    44)    real                           , intent(in)    :: salbedo_nir
    45)    real                           , intent(in)    :: scosaoi
    46)    real, dimension(n_radprof,ncoh), intent(inout) :: radprof_flip
    47)    real, dimension(ncoh)          , intent(out)   :: par_beam_flip
    48)    real, dimension(ncoh)          , intent(out)   :: par_diff_flip
    49)    real, dimension(ncoh)          , intent(out)   :: sw_abs_beam_flip
    50)    real, dimension(ncoh)          , intent(out)   :: sw_abs_diff_flip
    51)    real                           , intent(out)   :: uw_parhi_diff
    52)    real                           , intent(out)   :: uw_nirhi_diff
    53)    real                           , intent(out)   :: dw_parlo_beam
    54)    real                           , intent(out)   :: dw_parlo_diff
    55)    real                           , intent(out)   :: dw_nirlo_beam
    56)    real                           , intent(out)   :: dw_nirlo_diff
    57)    real(kind=8), dimension(ncoh)  , intent(out)   :: par_level_beam
    58)    real(kind=8), dimension(ncoh)  , intent(out)   :: par_level_diffu
    59)    real(kind=8), dimension(ncoh)  , intent(out)   :: par_level_diffd
    60)    real(kind=8), dimension(ncoh)  , intent(out)   :: light_level
    61)    real(kind=8), dimension(ncoh)  , intent(out)   :: light_beam_level
    62)    real(kind=8), dimension(ncoh)  , intent(out)   :: light_diff_level
    63)    !----- Local variables -----------------------------------------------------------------!
    64)    integer     , dimension(2*ncoh)                :: indx
    65)    integer                                        :: il
    66)    integer                                        :: ipft
    67)    integer                                        :: ncoh2
    68)    integer                                        :: iband
    69)    integer                                        :: i
    70)    integer                                        :: j
    71)    integer                                        :: ind
    72)    real(kind=8), dimension(ncoh)                  :: expkl_top
    73)    real(kind=8), dimension(ncoh)                  :: expkl_bot
    74)    real(kind=8), dimension(ncoh)                  :: expamk_top
    75)    real(kind=8), dimension(ncoh)                  :: expamk_bot
    76)    real(kind=8), dimension(ncoh)                  :: expapk_top
    77)    real(kind=8), dimension(ncoh)                  :: expapk_bot
    78)    real(kind=8), dimension(ncoh)                  :: a_top
    79)    real(kind=8), dimension(ncoh)                  :: a_bot
    80)    real(kind=8), dimension(ncoh)                  :: b_top
    81)    real(kind=8), dimension(ncoh)                  :: b_bot
    82)    real(kind=8), dimension(ncoh)                  :: c_top
    83)    real(kind=8), dimension(ncoh)                  :: c_bot
    84)    real(kind=8), dimension(ncoh)                  :: f_top
    85)    real(kind=8), dimension(ncoh)                  :: f_bot
    86)    real(kind=8), dimension(ncoh)                  :: g_top
    87)    real(kind=8), dimension(ncoh)                  :: g_bot
    88)    real(kind=8), dimension(ncoh)                  :: h_top
    89)    real(kind=8), dimension(ncoh)                  :: h_bot
    90)    real(kind=8), dimension(ncoh)                  :: beam_bot
    91)    real(kind=8), dimension(ncoh)                  :: beam_bot_crown
    92)    real(kind=8), dimension(ncoh)                  :: tai
    93)    real(kind=8), dimension(ncoh)                  :: eff_tai
    94)    real(kind=8), dimension(ncoh)                  :: lambda
    95)    real(kind=8), dimension(ncoh)                  :: beam_backscatter
    96)    real(kind=8), dimension(ncoh)                  :: cohort_scatter_vis
    97)    real(kind=8), dimension(ncoh)                  :: cohort_backscatter_vis
    98)    real(kind=8), dimension(ncoh)                  :: cohort_scatter_nir
    99)    real(kind=8), dimension(ncoh)                  :: cohort_backscatter_nir
   100)    real(kind=8), dimension(ncoh)                  :: cohort_scatter
   101)    real(kind=8), dimension(ncoh)                  :: cohort_backscatter
   102)    real(kind=8), dimension(ncoh)                  :: cohort_clumping
   103)    real(kind=8), dimension(ncoh+1)                :: upward_par_beam
   104)    real(kind=8), dimension(ncoh+1)                :: upward_par_diff
   105)    real(kind=8), dimension(ncoh+1)                :: upward_nir_beam
   106)    real(kind=8), dimension(ncoh+1)                :: upward_nir_diff
   107)    real(kind=8), dimension(ncoh+1)                :: downward_nir_beam
   108)    real(kind=8), dimension(ncoh+1)                :: downward_nir_diff
   109)    real(kind=8), dimension(ncoh+1)                :: downward_par_beam
   110)    real(kind=8), dimension(ncoh+1)                :: downward_par_diff
   111)    real(kind=8), dimension(2*ncoh)                :: mastervec_beam
   112)    real(kind=8), dimension(2*ncoh)                :: masveccp_beam
   113)    real(kind=8), dimension(2*ncoh)                :: mastervec_diff
   114)    real(kind=8), dimension(2*ncoh)                :: masveccp_diff
   115)    real(kind=8), dimension(2*ncoh,2)              :: matal
   116)    real(kind=8), dimension(2*ncoh,5)              :: mastermat
   117)    real(kind=8), dimension(2*ncoh,2*ncoh)         :: masmatcp
   118)    real(kind=8)                                   :: albedo
   119)    real(kind=8)                                   :: cosaoi
   120)    real(kind=8)                                   :: eta
   121)    real(kind=8)                                   :: zeta
   122)    real(kind=8)                                   :: iota
   123)    real(kind=8)                                   :: exk
   124)    real(kind=8)                                   :: exki
   125)    real(kind=8)                                   :: zetai
   126)    real(kind=8)                                   :: d
   127)    real(kind=8)                                   :: rhoo
   128)    real(kind=8)                                   :: psi
   129)    real(kind=8)                                   :: sigma
   130)    real(kind=8)                                   :: source_bot
   131)    real(kind=8)                                   :: source_top
   132)    real(kind=8)                                   :: beam_top
   133)    real(kind=8)                                   :: diff_top
   134)    real(kind=8)                                   :: weight_leaf
   135)    real(kind=8)                                   :: weight_wood
   136)    real(kind=8)                                   :: proj_area
   137)    real(kind=8)                                   :: snglscat_alb
   138)    !----- External functions. -------------------------------------------------------------!
   139)    real(kind=4)                  , external       :: sngloff
   140)    !---------------------------------------------------------------------------------------!
   141) 
   142)    
   143)    !----- Convert input variable to double precision. -------------------------------------!
   144)    cosaoi = max(cosz_min8,dble(scosaoi))
   145) 
   146)    !----- Calculate factors common for NIR, PAR. ------------------------------------------!
   147)    ncoh2      = 2*ncoh
   148)    lambda     = 5.d-1/cosaoi
   149)    do il=1,ncoh
   150)       ipft = pft(il)
   151) 
   152)       !------------------------------------------------------------------------------------!
   153)       !     We find the optical depth of the direct beam (lambda), following CLM technical !
   154)       ! note (equation 3.3 and text after equation 3.2).                                   !
   155)       !------------------------------------------------------------------------------------!
   156)       proj_area      = phi1(ipft) + phi2(ipft) * cosaoi
   157)       lambda(il)     = proj_area / cosaoi
   158)       !------------------------------------------------------------------------------------!
   159) 
   160) 
   161) 
   162)       !------------------------------------------------------------------------------------!
   163)       !      Find the backscatter for direct beam radiation, following equations (3.14)    !
   164)       ! and (3.15) of the CLM technical note.  The forward scattering was omitted on both  !
   165)       ! equations because they are different for PAR and NIR, but in both cases they       !
   166)       ! cancel out.                                                                        !
   167)       !------------------------------------------------------------------------------------!
   168)       snglscat_alb         = 5.d-1 * proj_area / (phi2(ipft) * cosaoi + proj_area)         &
   169)                            * ( 1.d0 - phi1(ipft) * cosaoi                                  &
   170)                              / (phi2(ipft) * cosaoi + proj_area)                           &
   171)                              * log (1.d0  + (phi2(ipft) * cosaoi + proj_area)              &
   172)                                           / (phi1(ipft) * cosaoi)))
   173)       beam_backscatter(il) = ( 1.d0 + mu_bar(ipft) * lambda(il) ) * snglscat_alb           &
   174)                            / ( mu_bar(ipft) * lambda(il) )
   175)       !------------------------------------------------------------------------------------!
   176) 
   177) 
   178)       !------------------------------------------------------------------------------------!
   179)       !     Find the effective TAI for this layer, and the net clumping factor.            !
   180)       !------------------------------------------------------------------------------------!
   181)       tai                   (il) = lai(il) + wai(il)
   182)       eff_tai               (il) = clumping_factor(ipft) * lai(il) + wai(il)
   183)       cohort_clumping       (il) = eff_tai(il) / tai(il)
   184)       !------------------------------------------------------------------------------------!
   185) 
   186) 
   187) 
   188) 
   189)       !----- Find the scattering coefficients. --------------------------------------------!
   190)       weight_leaf                = clumping_factor(ipft) * lai(il) / eff_tai(il)
   191)       weight_wood                = 1.d0 - weight_leaf
   192)       cohort_scatter_vis    (il) = leaf_scatter_vis(ipft)     * weight_leaf                &
   193)                                  + wood_scatter_vis(ipft)     * weight_wood
   194)       cohort_scatter_nir    (il) = leaf_scatter_nir(ipft)     * weight_leaf                &
   195)                                  + wood_scatter_nir(ipft)     * weight_wood
   196)       cohort_backscatter_vis(il) = leaf_backscatter_vis(ipft) * weight_leaf                &
   197)                                  + wood_backscatter_vis(ipft) * weight_wood
   198)       cohort_backscatter_nir(il) = leaf_backscatter_nir(ipft) * weight_leaf                &
   199)                                  + wood_backscatter_nir(ipft) * weight_wood
   200)       
   201)    end do
   202) 
   203)    !----- Loop over bands (currently Visible and near infrared). --------------------------!
   204)    bandloop: do iband = 1,2
   205)       select case(iband)
   206)       case (1) !----- Visible (or PAR). ---------------------------------------------------!
   207)          do il = 1,ncoh
   208)             cohort_scatter    (il) = cohort_scatter_vis    (il)
   209)             cohort_backscatter(il) = cohort_backscatter_vis(il)
   210)          end do
   211)          beam_top = par_beam_norm
   212)          diff_top = par_diff_norm
   213)          albedo   = dble(salbedo_par)
   214)       case (2) !----- Near infrared (or NIR). ---------------------------------------------!
   215)          do il = 1,ncoh
   216)             cohort_scatter    (il) = cohort_scatter_nir    (il)
   217)             cohort_backscatter(il) = cohort_backscatter_nir(il)
   218)          end do
   219)          beam_top = nir_beam_norm
   220)          diff_top = nir_diff_norm
   221)          albedo   = dble(salbedo_nir)
   222)       end select
   223)       !------------------------------------------------------------------------------------!
   224) 
   225) 
   226) 
   227) 
   228) 
   229)       !------------------------------------------------------------------------------------!
   230)       !     Calculate more factors for this band.  We start by calculating the forcings,   !
   231)       ! using the effective tree area index, considering the leaf clumpiness and, if the   !
   232)       ! user asked for it, the branchwood area index (in case the user doesn't want,       !
   233)       ! WAI=0.).  We distinguish between the light underneath a crown (beam_bot_crown) and !
   234)       ! the total light passing through a layer (beam_bot).  Also, we consider that the    !
   235)       ! canopy may be partially open in case the user wants this distinction (otherwise    !
   236)       ! canopy_area is one.).                                                              !
   237)       !------------------------------------------------------------------------------------!
   238)       !----- Start with the tallest cohort, moving downwards. -----------------------------!
   239)       beam_bot_crown(ncoh)  = beam_top * exp( - lambda(ncoh) * eff_tai(ncoh)               &
   240)                                               / canopy_area(ncoh) )
   241) !!      par_level_beam(ncoh)  = beam_top * exp( - 5.d-1 * lambda(ncoh) * eff_tai(ncoh)       &
   242) !!                                              / canopy_area(ncoh) )
   243)       beam_bot(ncoh)        = beam_top * (1.d0 - canopy_area(ncoh))                        &
   244)                             + canopy_area(ncoh) * beam_bot_crown(ncoh)
   245)       do il=ncoh-1,1,-1
   246)          beam_bot_crown(il) = beam_bot(il+1) * exp( - lambda(il) * eff_tai(il)             &
   247)                                                     / canopy_area(il) )
   248)          beam_bot(il)       = beam_bot(il+1) * (1.d0 - canopy_area(il))                    & 
   249)                             + canopy_area(il) * beam_bot_crown(il)
   250) !!         par_beam_level(il) = beam_bot(il+1)                                               &
   251)  !!                           * exp(- 5.d-1 * lambda(il) * eff_tai(il) / canopy_area(il))    &
   252)  !!                           * canopy_area(il)                                              &
   253)  !!                           + (1.d0-canopy_area(il)) * beam_bot(il+1)
   254)       end do
   255) 
   256)       do il=1,ncoh
   257)          ipft  = pft(il)
   258)          eta   = cohort_clumping(il)                                                       &
   259)                * (1.0d0 - ( 1.0d0 - cohort_backscatter(il)) * cohort_scatter(il))          &
   260)                / mu_bar(ipft)
   261)          zeta  = cohort_scatter(il) * cohort_backscatter(il) * cohort_clumping(il)         &
   262)                / mu_bar(ipft)
   263)          psi   = cohort_clumping(il) * lambda(il)
   264)          iota  = psi * cohort_scatter(il) * beam_backscatter(il)
   265) 
   266) 
   267)          !----- Find derived properties. --------------------------------------------------!
   268)          exk   = sqrt(eta*eta - zeta*zeta)
   269)          exki  = 1.0d0/exk
   270)          zetai = 1.0d0/zeta
   271) 
   272)          sigma = psi * psi + zeta * zeta - eta * eta
   273) 
   274)          !----- Sources. ------------------------------------------------------------------!
   275)          source_bot = iota * beam_bot_crown(il)
   276)          source_top = source_bot * exp(lambda(il) * eff_tai(il))
   277) 
   278)          !----- Forcing coefficients. -----------------------------------------------------!
   279)          rhoo  = - (zeta + eta + psi) * source_bot
   280) 
   281)          !----- Calculate exponentials only once. -----------------------------------------!
   282)          expkl_bot (il) = 1.0d0
   283)          expkl_top (il) = exp(exk * tai(il))
   284)          expamk_bot(il) = 1.0d0
   285)          expamk_top(il) = exp((psi-exk) * tai(il))
   286)          expapk_bot(il) = 1.0d0
   287)          expapk_top(il) = exp((psi+exk) * tai(il))
   288)          a_bot(il)      = -source_bot*zetai
   289)          a_top(il)      = -source_top*zetai                                                &
   290)                         + 5.d-1*zetai*(eta*exki-1.0d0)*expkl_top(il)*rhoo/(psi-exk)        &
   291)                         * (expamk_top(il)-expamk_bot(il))                                  &
   292)                         - 5.d-1*zetai*(eta*exki+1.0d0) / expkl_top(il)*rhoo/(psi+exk)      &
   293)                         * (expapk_top(il)-expapk_bot(il))
   294)          b_bot(il)      = 5.d-1*zetai*(eta*exki-1.0d0)
   295)          b_top(il)      = 5.d-1*zetai*(eta*exki-1.0d0)*expkl_top(il)
   296)          c_bot(il)      = -5.d-1*zetai*(eta*exki+1.0d0)
   297)          c_top(il)      = -5.d-1*zetai*(eta*exki+1.0d0)/expkl_top(il)
   298)          f_bot(il)      = 0.0d0
   299)          f_top(il)      = 5.d-1*exki*expkl_top(il)*rhoo/(psi-exk)                          &
   300)                         * (expamk_top(il)-expamk_bot(il))                                  &
   301)                         - 5.d-1*exki/expkl_top(il)*rhoo/(psi+exk)                          &
   302)                         * (expapk_top(il)-expapk_bot(il))
   303)          g_bot(il)      = 5.d-1*exki
   304)          g_top(il)      = 5.d-1*exki*expkl_top(il)
   305)          h_bot(il)      = -5.d-1*exki
   306)          h_top(il)      = -5.d-1*exki/expkl_top(il)
   307)       end do
   308) 
   309)       
   310)       !----- Organize the matrix coefficients. --------------------------------------------!
   311)       do j=1,ncoh2
   312)          do i=1,ncoh2
   313)             masmatcp(i,j)=0.0d0
   314)          end do
   315)          mastervec_beam(j)    = 0.0d0
   316)          masveccp_beam(j)     = 0.0d0
   317)          mastervec_diff(j) = 0.0d0
   318)          masveccp_diff(j)  = 0.0d0
   319)       end do
   320) 
   321)       masmatcp(1,1)        = g_top(ncoh)
   322)       masmatcp(1,2)        = h_top(ncoh)
   323)       mastervec_beam(1)    = -f_top(ncoh)
   324)       mastervec_diff(1) = diff_top
   325)       masveccp_beam(1)     = mastervec_beam(1)
   326)       masveccp_diff(1)  = mastervec_diff(1)
   327) 
   328)       do i=2,ncoh2-2,2
   329)          masmatcp(i,i-1) = g_bot(nint(real(ncoh2-i+2)*0.5))
   330)          masmatcp(i,i)   = h_bot(nint(real(ncoh2-i+2)*0.5))
   331) 
   332)          masmatcp(i,i+1)      = -g_top(nint(real(ncoh2-i)*0.5))
   333)          masmatcp(i,i+2)      = -h_top(nint(real(ncoh2-i)*0.5))
   334)          mastervec_beam(i)    = -f_bot(nint(real(ncoh2-i+2)*0.5))                          &
   335)                               + f_top(nint(real(ncoh2-i)*0.5))
   336)          mastervec_diff(i) = 0.0d0
   337)          masveccp_beam(i)     = mastervec_beam(i)
   338)          masveccp_diff(i)  = mastervec_diff(i)
   339)       end do
   340) 
   341)       do i=3,ncoh2-1,2
   342)          masmatcp(i,i-2)      = b_bot(nint(real(ncoh2-i+3)*0.5))
   343)          masmatcp(i,i-1)      = c_bot(nint(real(ncoh2-i+3)*0.5))
   344)          masmatcp(i,i)        = -b_top(nint(real(ncoh2-i+1)*0.5))
   345)          masmatcp(i,i+1)      = -c_top(nint(real(ncoh2-i+1)*0.5))
   346)          mastervec_beam(i)    = -a_bot(nint(real(ncoh2-i+3)*0.5))                          &
   347)                               + a_top(nint(real(ncoh2-i+1)*0.5))
   348)          masveccp_beam(i)     = mastervec_beam(i)
   349)          mastervec_diff(i) = 0.0d0
   350)          masveccp_diff(i)  = mastervec_diff(i)
   351)       end do
   352)       masmatcp(ncoh2,ncoh2-1)  =  b_bot(1) - albedo * g_bot(1)
   353)       masmatcp(ncoh2,ncoh2)    =  c_bot(1) - albedo * h_bot(1)
   354)       mastervec_beam(ncoh2)    = -a_bot(1) + albedo * beam_bot(1)
   355)       masveccp_beam(ncoh2)     = mastervec_beam(ncoh2)
   356)       mastervec_diff(ncoh2) = 0.0d0
   357)       masveccp_diff(ncoh2)  = mastervec_diff(ncoh2)
   358)       
   359)       !----- Prepare for inversion. -------------------------------------------------------!
   360)       mastermat(1,1) = 0.d0
   361)       mastermat(1,2) = 0.d0
   362)       mastermat(1,3) = masmatcp(1,1)
   363)       mastermat(1,4) = masmatcp(1,2)
   364)       mastermat(1,5) = 0.d0
   365) 
   366)       do i=2,ncoh2-2,2
   367)          mastermat(i,1) = 0.d0
   368)          mastermat(i,2) = masmatcp(i,i-1)
   369)          mastermat(i,3) = masmatcp(i,i)
   370)          mastermat(i,4) = masmatcp(i,i+1)
   371)          mastermat(i,5) = masmatcp(i,i+2)
   372)       end do
   373) 
   374)       do i=3,ncoh2-1,2
   375)          mastermat(i,1) = masmatcp(i,i-2)
   376)          mastermat(i,2) = masmatcp(i,i-1)
   377)          mastermat(i,3) = masmatcp(i,i)
   378)          mastermat(i,4) = masmatcp(i,i+1)
   379)          mastermat(i,5) = 0.d0
   380)       end do
   381) 
   382)       mastermat(ncoh2,1) = 0.d0
   383)       mastermat(ncoh2,2) = masmatcp(ncoh2,ncoh2-1)
   384)       mastermat(ncoh2,3) = masmatcp(ncoh2,ncoh2)
   385)       mastermat(ncoh2,4) = 0.d0
   386)       mastermat(ncoh2,5) = 0.d0
   387)       
   388)       !----- Invert the matrix. -----------------------------------------------------------!
   389)       call bandec(mastermat,ncoh2,ncoh2,2,2,matal,indx,d)
   390) 
   391)       !----- Backsubstitute for beam and diffuse. -----------------------------------------!
   392)       call banbks(mastermat,ncoh2,ncoh2,2,2,matal,indx,mastervec_beam)
   393)       call banbks(mastermat,ncoh2,ncoh2,2,2,matal,indx,mastervec_diff)
   394)       
   395)       !----- Improve the solution. --------------------------------------------------------!
   396)       call mprove(masmatcp,mastermat,matal,ncoh2,ncoh2,5,2,indx                            &
   397)                  ,masveccp_beam,mastervec_beam)
   398)       call mprove(masmatcp,mastermat,matal,ncoh2,ncoh2,5,2,indx                            &
   399)                  ,masveccp_diff,mastervec_diff)
   400) 
   401)       select case (iband)
   402)       case (1) 
   403)          !---- Visible (or PAR) band. -----------------------------------------------------!
   404)          do i=3,ncoh2-1,2
   405)             ind                     = nint(real(ncoh2-i+1)*0.5)+1
   406)             upward_par_beam(ind)    = a_bot(ind) + b_bot(ind) * mastervec_beam(i-2)        &
   407)                                     + c_bot(ind) * mastervec_beam(i-1)
   408)             upward_par_diff(ind)    = b_bot(ind) * mastervec_diff(i-2)                     &
   409)                                     + c_bot(ind) * mastervec_diff(i-1)
   410)          end do
   411) 
   412)          do i=2,ncoh2-2,2
   413)             ind                       = nint(real(ncoh2-i)*0.5)+1
   414)             downward_par_beam(ind)    = beam_bot(ind) + f_bot(ind)                         &
   415)                                       + h_bot(ind) * mastervec_beam(i)                     &
   416)                                       + g_bot(ind) * mastervec_beam(i-1)
   417)             downward_par_diff(ind)    = h_bot(ind) * mastervec_diff(i)                     &
   418)                                       + g_bot(ind) * mastervec_diff(i-1)
   419)          end do
   420) 
   421)          upward_par_beam(ncoh+1)      = b_top(ncoh) * mastervec_beam(1)                    &
   422)                                       + c_top(ncoh) * mastervec_beam(2) + a_top(ncoh)
   423)          upward_par_diff(ncoh+1)      = b_top(ncoh) * mastervec_diff(1)                    &
   424)                                       + c_top(ncoh) * mastervec_diff(2)
   425)          downward_par_beam(ncoh+1)    = beam_top
   426)          downward_par_diff(ncoh+1) = diff_top
   427)          downward_par_beam(1)         = g_bot(1) * mastervec_beam(ncoh2-1)                 &
   428)                                       + h_bot(1) * mastervec_beam(ncoh2) + f_bot(1)        &
   429)                                       + beam_bot(1)
   430)          downward_par_diff(1)         = g_bot(1) * mastervec_diff(ncoh2-1)                 &
   431)                                       + h_bot(1) * mastervec_diff(ncoh2)
   432)          upward_par_beam(1)           = albedo * downward_par_beam(1)
   433)          upward_par_diff(1)           = albedo * downward_par_diff(1)
   434)       case (2)
   435) 
   436)          !---- Near infra-red. ------------------------------------------------------------!
   437)          do i=3,ncoh2-1,2
   438)             ind                     = nint(real(ncoh2-i+1)*0.5)+1
   439)             upward_nir_beam(ind)    = a_bot(ind) + b_bot(ind) * mastervec_beam(i-2)        &
   440)                                     + c_bot(ind) * mastervec_beam(i-1)
   441)             upward_nir_diff(ind)    = b_bot(ind) * mastervec_diff(i-2)                     &
   442)                                     + c_bot(ind) * mastervec_diff(i-1)
   443)          end do
   444) 
   445)          do i=2,ncoh2-2,2
   446)             ind                       = nint(real(ncoh2-i)*0.5)+1
   447)             downward_nir_beam(ind)    = beam_bot(ind) + f_bot(ind)                         &
   448)                                       + h_bot(ind) * mastervec_beam(i)                     &
   449)                                       + g_bot(ind) * mastervec_beam(i-1)
   450)             downward_nir_diff(ind)    = h_bot(ind) * mastervec_diff(i)                     &
   451)                                       + g_bot(ind) * mastervec_diff(i-1)
   452)          end do
   453) 
   454)          upward_nir_beam(ncoh+1)      = b_top(ncoh) * mastervec_beam(1)                    &
   455)                                       + c_top(ncoh) * mastervec_beam(2) + a_top(ncoh)
   456)          upward_nir_diff(ncoh+1)      = b_top(ncoh) * mastervec_diff(1)                    &
   457)                                       + c_top(ncoh) * mastervec_diff(2)
   458)          downward_nir_beam(ncoh+1)    = beam_top
   459)          downward_nir_diff(ncoh+1)    = diff_top
   460)          downward_nir_beam(1)         = g_bot(1) * mastervec_beam(ncoh2-1)                 &
   461)                                       + h_bot(1) * mastervec_beam(ncoh2) + f_bot(1)        &
   462)                                       + beam_bot(1)
   463)          downward_nir_diff(1)         = g_bot(1) * mastervec_diff(ncoh2-1)                 &
   464)                                       + h_bot(1) * mastervec_diff(ncoh2)
   465)          upward_nir_beam(1)           = albedo * downward_nir_beam(1)
   466)          upward_nir_diff(1)           = albedo * downward_nir_diff(1)
   467)       end select
   468)    end do bandloop
   469)    
   470)    do il=1,ncoh
   471)       !------------------------------------------------------------------------------------!
   472)       !     Find the light level of the visible band, which is the one that the plants     !
   473)       ! use, and the total light level, which includes the near-infrared.                  !
   474)       !------------------------------------------------------------------------------------!
   475)       par_level_diffd(il)          = 5.d-1 * ( downward_par_diff(il  )                     &
   476)                                  + downward_par_diff(il+1) ) / (par_diff_norm+par_beam_norm)
   477) 
   478)       par_level_diffu(il)          = 5.d-1 * ( upward_par_diff(il  )                     &
   479)                                  + upward_par_diff(il+1) ) / (par_diff_norm+par_beam_norm)
   480) 
   481)       par_level_beam(il)          = 5.d-1 * ( downward_par_beam   (il  )                   &
   482)                                         + downward_par_beam   (il+1) ) / (par_beam_norm+par_diff_norm)
   483) 
   484)       light_level(il)         = 5.d-1 * ( downward_par_diff(il  )                          &
   485)                                         + downward_par_beam   (il  )                       &
   486)                                         + downward_nir_diff(il  )                          &
   487)                                         + downward_nir_beam   (il  )                       &
   488)                                         + downward_par_diff(il+1)                          &
   489)                                         + downward_par_beam   (il+1)                       &
   490)                                         + downward_nir_diff(il+1)                          &
   491)                                         + downward_nir_beam   (il+1) )
   492) 
   493)       light_beam_level(il)    = 5.d-1 * ( downward_par_beam   (il  )                       &
   494)                                         + downward_nir_beam   (il  )                       &
   495)                                         + downward_par_beam   (il+1)                       &
   496)                                         + downward_nir_beam   (il+1) )                     &
   497)                               / ( par_beam_norm + nir_beam_norm )
   498) 
   499)       light_diff_level(il)    = 5.d-1 * ( downward_par_diff(il  )                          &
   500)                                         + downward_nir_diff(il  )                          &
   501)                                         + downward_par_diff(il+1)                          &
   502)                                         + downward_nir_diff(il+1) )                        &
   503)                               / ( par_diff_norm + nir_beam_norm )
   504)       !------------------------------------------------------------------------------------!
   505) 
   506)       par_beam_flip(il)    = sngloff(  downward_par_beam       (il+1)                      &
   507)                                      - downward_par_beam       (il  )                      &
   508)                                      + upward_par_beam         (il  )                      &
   509)                                      - upward_par_beam         (il+1)                      &
   510)                                      , tiny_offset )
   511)       par_diff_flip(il)    = sngloff(  downward_par_diff (il+1)                            &
   512)                                      - downward_par_diff (il  )                            &
   513)                                      + upward_par_diff   (il  )                            &
   514)                                      - upward_par_diff   (il+1)                            &
   515)                                      , tiny_offset )
   516) 
   517)       radprof_flip(1,il)   = sngloff(  downward_par_beam (il), tiny_offset)
   518)       radprof_flip(2,il)   = sngloff(  upward_par_beam   (il), tiny_offset)
   519)       radprof_flip(3,il)   = sngloff(  downward_par_diff (il), tiny_offset)
   520)       radprof_flip(4,il)   = sngloff(  upward_par_diff   (il), tiny_offset)
   521)       radprof_flip(5,il)   = sngloff(  downward_nir_beam (il), tiny_offset)
   522)       radprof_flip(6,il)   = sngloff(  upward_nir_beam   (il), tiny_offset)
   523)       radprof_flip(7,il)   = sngloff(  downward_nir_diff (il), tiny_offset)
   524)       radprof_flip(8,il)   = sngloff(  upward_nir_diff   (il), tiny_offset)
   525) 
   526) 
   527)       sw_abs_beam_flip(il) = par_beam_flip(il) + sngloff(  downward_nir_beam   (il+1)      &
   528)                                                          - downward_nir_beam   (il  )      &
   529)                                                          + upward_nir_beam     (il  )      &
   530)                                                          - upward_nir_beam     (il+1)      &
   531)                                                          , tiny_offset )
   532) 
   533)       sw_abs_diff_flip(il) = par_diff_flip(il) + sngloff(  downward_nir_diff(il+1)         &
   534)                                                          - downward_nir_diff(il  )         &
   535)                                                          + upward_nir_diff  (il  )         &
   536)                                                          - upward_nir_diff  (il+1)         &
   537)                                                          , tiny_offset )
   538) 
   539)       !----- Ensure that we don't get any negative radiation... ---------------------------!
   540)       par_beam_flip   (il)  = max(0.0,par_beam_flip   (il) )
   541)       par_diff_flip   (il)  = max(0.0,par_diff_flip   (il) )
   542)       sw_abs_beam_flip(il)  = max(0.0,sw_abs_beam_flip(il) )
   543)       sw_abs_diff_flip(il)  = max(0.0,sw_abs_diff_flip(il) )
   544)    end do
   545)    !---------------------------------------------------------------------------------------!
   546) 
   547) 
   548) 
   549) 
   550) 
   551)    !----- Copy to the output variables. ---------------------------------------------------!
   552)    dw_parlo_beam    = max(0.0,sngl(downward_par_beam        (1) ) )
   553)    dw_parlo_diff    = max(0.0,sngl(downward_par_diff        (1) ) )
   554)    uw_parhi_diff    = max(0.0,sngl(upward_par_beam(ncoh+1) + upward_par_diff(ncoh+1) ) )
   555)    dw_nirlo_beam    = max(0.0,sngl(downward_nir_beam        (1) ) )
   556)    dw_nirlo_diff    = max(0.0,sngl(downward_nir_diff        (1) ) )
   557)    uw_nirhi_diff    = max(0.0,sngl(upward_nir_beam(ncoh+1) + upward_nir_diff(ncoh+1) ) )
   558)    !---------------------------------------------------------------------------------------!
   559) 
   560)    return
   561) end subroutine old_sw_two_stream

   572) !   This subroutine finds the matrix pseudo-inverse.                                       !
   573) !------------------------------------------------------------------------------------------!
   574) subroutine bandec(a,ntot,nuse,m1,m2,al,indx,d)
   575)    use consts_coms, only : tiny_num8
   576)    implicit none
   577)    !----- Arguments -----------------------------------------------------------------------!
   578)    integer                              , intent(in)    :: ntot
   579)    integer                              , intent(in)    :: nuse
   580)    integer                              , intent(in)    :: m1
   581)    integer                              , intent(in)    :: m2
   582)    integer     , dimension(ntot)        , intent(inout) :: indx
   583)    real(kind=8), dimension(ntot,m1+m2+1), intent(inout) :: a
   584)    real(kind=8), dimension(ntot,m1)     , intent(inout) :: al
   585)    real(kind=8)                         , intent(out)   :: d
   586)    !----- Local variables -----------------------------------------------------------------!
   587)    integer                                              :: i
   588)    integer                                              :: j
   589)    integer                                              :: k
   590)    integer                                              :: l
   591)    integer                                              :: mm
   592)    real(kind=8)                                         :: tvar
   593)    real(kind=8)                                         :: dum
   594)    !---------------------------------------------------------------------------------------!
   595) 
   596) 
   597)    mm=m1+m2+1
   598)    l=m1
   599)    do i=1,m1
   600)       do j=m1+2-i,mm
   601)          a(i,j-l)=a(i,j)
   602)       end do
   603) 
   604)       l=l-1
   605) 
   606)       do j=mm-l,mm
   607)          a(i,j)=0.d0
   608)       end do
   609)    end do
   610) 
   611)    d=1.0d0
   612) 
   613)    l=m1
   614)    do k=1,nuse
   615)       dum=a(k,1)
   616)       i=k
   617)       if (l < nuse) l = l + 1
   618)       do j=k+1,l
   619)          if (abs(a(j,1)) > abs(dum)) then
   620)             dum = a(j,1)
   621)             i   = j
   622)          end if
   623)       end do
   624)       indx(k) = i
   625)       if (dum == 0.0d0) a(k,1) = tiny_num8
   626)       if (i /= k) then
   627)          d=-d
   628)          do j=1,mm
   629)             tvar   = a(k,j)
   630)             a(k,j) = a(i,j)
   631)             a(i,j) = tvar
   632)          end do
   633)       end if
   634) 
   635)       do i=k+1,l
   636)          dum       = a(i,1) / a(k,1)
   637)          al(k,i-k) = dum
   638)          do j=2,mm
   639)             a(i,j-1) = a(i,j) - dum * a(k,j)
   640)          end do
   641) 
   642)          a(i,mm) = 0.0d0
   643) 
   644)       end do
   645)    end do
   646)    return
   647) end subroutine bandec

   658) !    This routine will perform the back substitution of the linear system.                 !
   659) !------------------------------------------------------------------------------------------!
   660) subroutine banbks(a,ntot,nuse,m1,m2,al,indx,b)
   661)    implicit none
   662)    !----- Arguments -----------------------------------------------------------------------!
   663)    integer                              , intent(in)    :: ntot
   664)    integer                              , intent(in)    :: nuse
   665)    integer                              , intent(in)    :: m1
   666)    integer                              , intent(in)    :: m2
   667)    integer     , dimension(ntot)        , intent(in)    :: indx
   668)    real(kind=8), dimension(ntot,m1+m2+1), intent(in)    :: a
   669)    real(kind=8), dimension(ntot,m1)     , intent(in)    :: al
   670)    real(kind=8), dimension(ntot)        , intent(inout) :: b
   671)    !----- Local variables -----------------------------------------------------------------!
   672)    integer                                              :: i
   673)    integer                                              :: k
   674)    integer                                              :: l
   675)    integer                                              :: mm
   676)    real(kind=8)                                         :: dum
   677)    real(kind=8)                                         :: tvar
   678)    !---------------------------------------------------------------------------------------!
   679) 
   680)    mm = m1 + m2 + 1
   681)    l  = m1
   682) 
   683)    do k=1,nuse
   684)       i=indx(k)
   685) 
   686)       if (i /= k) then
   687)          tvar = b(k)
   688)          b(k) = b(i)
   689)          b(i) = tvar
   690)       end if
   691) 
   692)       if(l < nuse) l = l + 1
   693) 
   694)       do i=k+1,l
   695)          b(i) = b(i) - al(k,i-k) * b(k)
   696)       end do
   697)    end do
   698) 
   699)    l=1
   700)    do i=nuse,1,-1
   701)       dum = b(i)
   702)       do k=2,l
   703)          dum = dum - a(i,k) * b(k+i-1)
   704)       end do
   705)       b(i) = dum / a(i,1)
   706)       if (l < mm) l = l + 1
   707)    end do
   708) 
   709)    return
   710) end subroutine banbks

   721) !   This subroutine will perform add a higher order term to improve the solution.          !
   722) !------------------------------------------------------------------------------------------!
   723) subroutine mprove(a,alud,matal,ntot,nuse,np,npp,indx,b,x)
   724)    implicit none
   725)    !----- Local constants -----------------------------------------------------------------!
   726)    integer     , parameter :: nmax=100
   727)    !----- Arguments -----------------------------------------------------------------------!
   728)    integer                           , intent(in)    :: ntot
   729)    integer                           , intent(in)    :: nuse
   730)    integer                           , intent(in)    :: np
   731)    integer                           , intent(in)    :: npp
   732)    integer     , dimension(ntot)     , intent(in)    :: indx
   733)    real(kind=8), dimension(ntot,ntot), intent(in)    :: a
   734)    real(kind=8), dimension(ntot)     , intent(in)    :: b
   735)    real(kind=8), dimension(ntot,np)  , intent(in)    :: alud
   736)    real(kind=8), dimension(ntot,npp) , intent(in)    :: matal
   737)    real(kind=8), dimension(ntot)     , intent(inout) :: x
   738)    !----- Local variables -----------------------------------------------------------------!
   739)    integer                                           :: i
   740)    integer                                           :: j
   741)    real(kind=8), dimension(ntot)                     :: r
   742)    real(kind=8)                                      :: sdp
   743)    !---------------------------------------------------------------------------------------!
   744) 
   745) 
   746)    do i=1,nuse
   747)       sdp = -b(i)
   748)       do j=1,nuse
   749)          sdp = sdp + a(i,j) * x(j)
   750)       end do
   751)       r(i) = sdp
   752)    end do
   753) 
   754)    call banbks(alud,ntot,nuse,npp,npp,matal,indx,r)
   755) 
   756)    do i=1,nuse
   757)       x(i) = x(i) - r(i)
   758)    end do
   759) 
   760)    return
   761) end subroutine mprove
