     7) !------------------------------------------------------------------------------------------!

     9) #if USE_HDF5
    10)    use hdf5
    11) #endif
    12)    use ed_init_full_history
    13)    use ed_init        , only : soil_default_fill
    14)    use ed_max_dims    , only : n_pft                   & ! intent(in)
    15)                              , huge_polygon            & ! intent(in)
    16)                              , str_len                 ! ! intent(in)
    17)    use pft_coms       , only : q                       & ! intent(in)
    18)                              , qsw                     & ! intent(in)
    19)                              , min_dbh                 & ! intent(in)
    20)                              , min_bdead               & ! intent(in)
    21)                              , is_grass                & ! intent(in)
    22)                              , include_pft             & ! intent(in)
    23)                              , include_pft_ag          & ! intent(in)
    24)                              , pft_1st_check           & ! intent(in)
    25)                              , agf_bs                  & ! intent(in)
    26)                              , include_these_pft       ! ! intent(in)
    27)    use ed_misc_coms   , only : sfilin                  & ! intent(in)
    28)                              , imonthh                 & ! intent(in)
    29)                              , iyearh                  & ! intent(in)
    30)                              , idateh                  & ! intent(in)
    31)                              , igrass                  & ! intent(in)
    32)                              , itimeh                  ! ! intent(in)
    33)    use ed_state_vars  , only : polygontype             & ! variable type
    34)                              , sitetype                & ! variable type
    35)                              , patchtype               & ! variable type
    36)                              , edtype                  & ! variable type
    37)                              , edgrid_g                & ! variable type
    38)                              , allocate_polygontype    & ! subroutine
    39)                              , allocate_sitetype       & ! subroutine
    40)                              , allocate_patchtype      ! ! subroutine
    41)    use grid_coms      , only : ngrids                  & ! intent(in)
    42)                              , nzg                     ! ! intent(in)
    43)    use consts_coms    , only : pio4                    ! ! intent(in)
    44)    use hdf5_coms      , only : file_id                 & ! intent(in)
    45)                              , dset_id                 & ! intent(in)
    46)                              , dspace_id               & ! intent(in)
    47)                              , globdims                & ! intent(in)
    48)                              , chnkdims                & ! intent(in)
    49)                              , chnkoffs                & ! intent(in)
    50)                              , memdims                 & ! intent(in)
    51)                              , memoffs                 & ! intent(in)
    52)                              , memsize                 ! ! intent(in)
    53)    use allometry      , only : area_indices            & ! function
    54)                              , ed_biomass              & ! function
    55)                              , bd2dbh                  & ! function
    56)                              , size2bl                 & ! function
    57)                              , dbh2h                   & ! function
    58)                              , dbh2bd                  ! ! function
    59)    use fuse_fiss_utils, only : terminate_cohorts       ! ! subroutine
    60)    use disturb_coms   , only : ianth_disturb           ! ! intent(in)
    61) 
    62)    implicit none
    63) #if USE_HDF5
    64) 
    65)    !------ Local variables. ---------------------------------------------------------------!
    66)    type(edtype)          , pointer     :: cgrid
    67)    type(polygontype)     , pointer     :: cpoly
    68)    type(sitetype)        , pointer     :: csite
    69)    type(patchtype)       , pointer     :: cpatch
    70)    character(len=3)                    :: cgr
    71)    character(len=str_len)              :: hnamel
    72) 
    73)    integer, dimension(:) , allocatable :: pysi_n
    74)    integer, dimension(:) , allocatable :: pysi_id
    75)    integer, dimension(:) , allocatable :: sipa_n
    76)    integer, dimension(:) , allocatable :: sipa_id
    77)    integer, dimension(:) , allocatable :: paco_n
    78)    integer, dimension(:) , allocatable :: paco_id
    79)    integer, dimension(:) , allocatable :: islakesite
    80)    integer, dimension(:) , allocatable :: plantation
    81)    integer                             :: igr
    82)    integer                             :: ipy
    83)    integer                             :: isi
    84)    integer                             :: is
    85)    integer                             :: ipa
    86)    integer                             :: ico
    87)    integer                             :: k
    88)    integer                             :: dset_npolygons_global
    89)    integer                             :: dset_nsites_global
    90)    integer                             :: dset_npatches_global
    91)    integer                             :: dset_ncohorts_global
    92)    integer                             :: dset_nzg
    93)    integer                             :: hdferr
    94)    integer                             :: ifpy
    95)    integer                             :: ipft
    96)    integer                             :: py_index
    97)    integer                             :: si_index
    98)    integer                             :: pa_index
    99)    integer                             :: dsetrank
   100)    integer                             :: iparallel
   101)    integer                             :: ndry_sites
   102)    logical                             :: exists
   103)    real, dimension(:)    , allocatable :: file_lats
   104)    real, dimension(:)    , allocatable :: file_lons
   105)    real                                :: minrad
   106)    real                                :: currad
   107)    real                                :: elim_nplant
   108)    real                                :: elim_lai
   109)    real                                :: salloc
   110)    real                                :: salloci
   111)    logical                             :: foundvar
   112)    !----- Local constants. ----------------------------------------------------------------!
   113)    real                  , parameter   :: tiny_biomass = 1.e-20
   114)    !----- External function. --------------------------------------------------------------!
   115)    real                  , external    :: dist_gc
   116)    !---------------------------------------------------------------------------------------!
   117) 
   118) 
   119) 
   120)    !----- Open the HDF environment. -------------------------------------------------------!
   121)    call h5open_f(hdferr)
   122) 
   123)    !----- Turn off automatic error printing. ----------------------------------------------!
   124)    call h5eset_auto_f(0,hdferr)
   125) 
   126) 
   127)    !----- Initialize the dimensional control variables for the H5 slabs. ------------------!
   128)    globdims = 0_8
   129)    chnkdims = 0_8
   130)    chnkoffs = 0_8
   131)    memoffs  = 0_8
   132)    memdims  = 0_8
   133)    memsize  = 1_8
   134) 
   135) 
   136)    !---------------------------------------------------------------------------------------!
   137)    !     Walk the tree and pull data from the dataset.                                     !
   138)    !---------------------------------------------------------------------------------------!
   139)    gridloop: do igr = 1,ngrids
   140) 
   141)       cgrid => edgrid_g(igr)
   142) 
   143) 
   144)       !----- Write the grid number as a character. ----------------------------------------!
   145)       write(cgr,'(a1,i2.2)') 'g',igr
   146) 
   147)       !------------------------------------------------------------------------------------!
   148)       !     The file name should be the exact file that will be read in, except the part   !
   149)       ! that lists the grid number and extension.  Because this is a history restart, only !
   150)       ! the first SFILIN will be used so we ensure that we will continue from the same     !
   151)       ! simulation.                                                                        !
   152)       !------------------------------------------------------------------------------------!
   153)       hnamel = trim(sfilin(1))//"-"//trim(cgr)//".h5"
   154) 
   155) 
   156)       !------------------------------------------------------------------------------------!
   157)       !     Check whether the file we want to open actually exists.                        !
   158)       !------------------------------------------------------------------------------------!
   159)       inquire(file=trim(hnamel),exist=exists)
   160)       if (.not.exists) then
   161)          !----- It doesn't, stop the run. -------------------------------------------------!
   162)          write (unit=*,fmt='(a,1x,a)')    'SFILIN(1) = ',trim(sfilin(1))
   163)          write (unit=*,fmt='(a,1x,i4.4)') 'IYEARH    = ',iyearh
   164)          write (unit=*,fmt='(a,1x,i2.2)') 'IMONTHH   = ',imonthh
   165)          write (unit=*,fmt='(a,1x,i2.2)') 'IDATEH    = ',idateh
   166)          write (unit=*,fmt='(a,1x,i4.4)') 'ITIMEH    = ',itimeh
   167)          call fatal_error ('File '//trim(hnamel)//' not found.'                            &
   168)                           ,'read_ed21_history_file','ed_read_ed21_history.F90')
   169)       else
   170)          !----- File is there, read it. ---------------------------------------------------!
   171)          call h5fopen_f(hnamel, H5F_ACC_RDONLY_F, file_id, hdferr)
   172)          if (hdferr < 0) then
   173)             write(unit=*,fmt='(a,1x,i4)') ' - Error opening HDF5 file - error - ',hdferr
   174)             write(unit=*,fmt='(a,1x,a)') ' - File name: ',trim(hnamel)
   175)             call fatal_error('Error opening HDF5 file - error - '//trim(hnamel)            &
   176)                             ,'read_ed21_history_file','ed_read_ed21_history.F90')
   177)          end if
   178)       end if
   179)       !------------------------------------------------------------------------------------!
   180) 
   181) 
   182) 
   183) 
   184)       !------------------------------------------------------------------------------------!
   185)       !      Retrieve global vector sizes and mapping tree.                                !
   186)       !------------------------------------------------------------------------------------!
   187)       globdims    = 0_8
   188)       chnkdims    = 0_8
   189)       chnkoffs    = 0_8
   190)       globdims(1) = 1_8
   191) 
   192)       call h5dopen_f(file_id,'NZG', dset_id, hdferr)
   193)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   194)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,dset_nzg,globdims, hdferr)
   195)       call h5sclose_f(dspace_id, hdferr)
   196)       call h5dclose_f(dset_id, hdferr)
   197) 
   198)       call h5dopen_f(file_id,'NPOLYGONS_GLOBAL', dset_id, hdferr)
   199)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   200)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,dset_npolygons_global,globdims, hdferr)
   201)       call h5sclose_f(dspace_id, hdferr)
   202)       call h5dclose_f(dset_id, hdferr)
   203) 
   204)       call h5dopen_f(file_id,'NSITES_GLOBAL', dset_id, hdferr)
   205)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   206)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,dset_nsites_global,globdims, hdferr)
   207)       call h5sclose_f(dspace_id, hdferr)
   208)       call h5dclose_f(dset_id, hdferr)
   209) 
   210)       call h5dopen_f(file_id,'NPATCHES_GLOBAL', dset_id, hdferr)
   211)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   212)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,dset_npatches_global,globdims, hdferr)
   213)       call h5sclose_f(dspace_id, hdferr)
   214)       call h5dclose_f(dset_id, hdferr)
   215) 
   216)       call h5dopen_f(file_id,'NCOHORTS_GLOBAL', dset_id, hdferr)
   217)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   218)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,dset_ncohorts_global,globdims, hdferr)
   219)       call h5sclose_f(dspace_id, hdferr)
   220)       call h5dclose_f(dset_id, hdferr)
   221) 
   222)       globdims(1) = int(dset_npolygons_global,8)
   223) 
   224)       allocate(pysi_n(dset_npolygons_global))
   225)       allocate(pysi_id(dset_npolygons_global))
   226) 
   227)       call h5dopen_f(file_id,'PYSI_N', dset_id, hdferr)
   228)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   229)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,pysi_n,globdims, hdferr)
   230)       call h5sclose_f(dspace_id, hdferr)
   231)       call h5dclose_f(dset_id, hdferr)
   232) 
   233)       call h5dopen_f(file_id,'PYSI_ID', dset_id, hdferr)
   234)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   235)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,pysi_id,globdims, hdferr)
   236)       call h5sclose_f(dspace_id, hdferr)
   237)       call h5dclose_f(dset_id, hdferr)
   238) 
   239)       globdims(1) = int(dset_nsites_global,8)
   240) 
   241)       allocate(sipa_n(dset_nsites_global))
   242)       allocate(sipa_id(dset_nsites_global))
   243) 
   244)       call h5dopen_f(file_id,'SIPA_N', dset_id, hdferr)
   245)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   246)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,sipa_n,globdims, hdferr)
   247)       call h5sclose_f(dspace_id, hdferr)
   248)       call h5dclose_f(dset_id, hdferr)
   249) 
   250)       call h5dopen_f(file_id,'SIPA_ID', dset_id, hdferr)
   251)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   252)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,sipa_id,globdims, hdferr)
   253)       call h5sclose_f(dspace_id, hdferr)
   254)       call h5dclose_f(dset_id, hdferr)
   255) 
   256)       globdims(1) = int(dset_npatches_global,8)
   257)       allocate(paco_n(dset_npatches_global))
   258)       allocate(paco_id(dset_npatches_global))
   259) 
   260)       call h5dopen_f(file_id,'PACO_N', dset_id, hdferr)
   261)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   262)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,paco_n,globdims, hdferr)
   263)       call h5sclose_f(dspace_id, hdferr)
   264)       call h5dclose_f(dset_id, hdferr)
   265) 
   266)       call h5dopen_f(file_id,'PACO_ID', dset_id, hdferr)
   267)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   268)       call h5dread_f(dset_id, H5T_NATIVE_INTEGER,paco_id,globdims, hdferr)
   269)       call h5sclose_f(dspace_id, hdferr)
   270)       call h5dclose_f(dset_id, hdferr)
   271) 
   272)       !----- Retrieve the polygon coordinates data. ---------------------------------------!
   273)       globdims(1) = int(dset_npolygons_global,8)
   274)       allocate(file_lats(dset_npolygons_global))
   275)       allocate(file_lons(dset_npolygons_global))
   276) 
   277)       call h5dopen_f(file_id,'LATITUDE', dset_id, hdferr)
   278)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   279)       call h5dread_f(dset_id, H5T_NATIVE_REAL,file_lats,globdims, hdferr)
   280)       call h5sclose_f(dspace_id, hdferr)
   281)       call h5dclose_f(dset_id, hdferr)
   282) 
   283)       call h5dopen_f(file_id,'LONGITUDE', dset_id, hdferr)
   284)       call h5dget_space_f(dset_id, dspace_id, hdferr)
   285)       call h5dread_f(dset_id, H5T_NATIVE_REAL,file_lons,globdims, hdferr)
   286)       call h5sclose_f(dspace_id, hdferr)
   287)       call h5dclose_f(dset_id, hdferr)
   288) 
   289)       !------------------------------------------------------------------------------------!
   290)       !     Loop over all the input polygons.                                              !
   291)       !------------------------------------------------------------------------------------!
   292)       polyloop: do ipy = 1,cgrid%npolygons
   293)          cpoly => cgrid%polygon(ipy)
   294) 
   295)          !----- Initialise polygon map and start closes polygon with a huge distance. -----!
   296)          py_index = 0
   297)          minrad = 1.e20
   298) 
   299)          !---------------------------------------------------------------------------------!
   300)          !     Loop over all input polygons and find the one that is the closest to this   !
   301)          ! current polygon.                                                                !
   302)          !---------------------------------------------------------------------------------!
   303)          do ifpy = 1,dset_npolygons_global
   304)             currad = dist_gc(file_lons(ifpy),cgrid%lon(ipy),file_lats(ifpy),cgrid%lat(ipy))
   305)             if (currad <  minrad) then
   306)                py_index = ifpy
   307)                minrad   = currad
   308)             end if
   309)          end do
   310) 
   311)          !---------------------------------------------------------------------------------!
   312)          !      A suitably close polygon has been located in the datasets.  Use these      !
   313)          ! values, and its children values in sites, patchs and cohorts.                   !
   314)          !---------------------------------------------------------------------------------!
   315)          iparallel = 0
   316) 
   317)          !---------------------------------------------------------------------------------!
   318)          !      POLYGON level variables.                                                   !
   319)          !---------------------------------------------------------------------------------!
   320)          !----- Load 1D dataset. ----------------------------------------------------------!
   321)          dsetrank = 1
   322)          globdims(1) = int(dset_npolygons_global,8)
   323)          chnkdims(1) = 1_8
   324)          chnkoffs(1) = int(py_index - 1,8)
   325)          memdims(1)  = 1_8
   326)          memoffs(1)  = 0_8
   327)          memsize(1)  = 1_8
   328)          !---- The ipy:ipy notation is needed for ifort when checking interfaces. ---------!
   329)          call hdf_getslab_i(cgrid%load_adjacency(ipy:ipy),'LOAD_ADJACENCY '                &
   330)                            ,dsetrank,iparallel,.true.,foundvar)
   331)          call hdf_getslab_r(cgrid%wbar(ipy:ipy),'WBAR ',dsetrank,iparallel,.true.,foundvar)
   332) 
   333)          !----- Load the workload (2D). ---------------------------------------------------!
   334)          dsetrank    = 2
   335)          globdims(1) = int(13,8)
   336)          chnkdims(1) = int(13,8)
   337)          memdims(1)  = int(13,8)
   338)          memsize(1)  = int(13,8)
   339)          chnkoffs(1) = 0_8
   340)          memoffs(1)  = 0_8
   341)          globdims(2) = int(dset_npolygons_global,8)
   342)          chnkdims(2) = 1_8
   343)          chnkoffs(2) = int(py_index - 1,8)
   344)          memdims(2)  = 1_8
   345)          memsize(2)  = 1_8
   346)          memoffs(2)  = 0_8
   347)          call hdf_getslab_r(cgrid%workload(:,ipy),'WORKLOAD ',dsetrank,iparallel,.false.   &
   348)                            ,foundvar)
   349)          !---------------------------------------------------------------------------------!
   350) 
   351) 
   352) 
   353)          !----- Load the lakesite data-----------------------------------------------------!
   354)          allocate(islakesite(pysi_n(py_index)))
   355)          islakesite  = 0
   356)          dsetrank    = 1_8
   357)          globdims(1) = int(dset_nsites_global,8)
   358)          chnkdims(1) = int(pysi_n(py_index),8)
   359)          chnkoffs(1) = int(pysi_id(py_index) - 1,8)
   360)          memdims(1)  = int(pysi_n(py_index),8)
   361)          memsize(1)  = int(pysi_n(py_index),8)
   362)          memoffs(1)  = 0_8
   363)          call hdf_getslab_i(islakesite,'ISLAKESITE ',dsetrank,iparallel,.false.,foundvar)
   364) 
   365)          ndry_sites = int(pysi_n(py_index))-sum(islakesite)
   366)          !---------------------------------------------------------------------------------!
   367) 
   368) 
   369)          !----- Allocate the vector of sites in the polygon. ------------------------------!
   370)          call allocate_polygontype(cpoly,ndry_sites)
   371)          call soil_default_fill(cgrid,igr,ipy)
   372)          !---------------------------------------------------------------------------------!
   373) 
   374)          is = 0
   375)          siteloop: do isi=1,pysi_n(py_index)
   376) 
   377)             si_index = pysi_id(py_index) + isi - 1
   378) 
   379)             if (islakesite(isi) == 0) then
   380)                is=is+1
   381) 
   382)                !----- Reset the HDF5 auxiliary variables before moving to the next level. -!
   383)                globdims = 0_8
   384)                chnkdims = 0_8
   385)                chnkoffs = 0_8
   386)                memoffs  = 0_8
   387)                memdims  = 0_8
   388)                memsize  = 1_8
   389)                !---------------------------------------------------------------------------!
   390) 
   391) 
   392) 
   393)                !---------------------------------------------------------------------------!
   394)                !      SITE level variables.                                                !
   395)                !---------------------------------------------------------------------------!
   396)                !----- Load 1D dataset. ----------------------------------------------------!
   397)                dsetrank    = 1_8
   398)                globdims(1) = int(dset_nsites_global,8)
   399)                chnkdims(1) = int(1,8)
   400)                chnkoffs(1) = int(pysi_id(py_index) - 2 + isi,8)
   401)                memdims(1)  = int(1,8)
   402)                memsize(1)  = int(1,8)
   403)                memoffs(1)  = 0_8
   404) 
   405)                call hdf_getslab_r( cpoly%area       (is:is), 'AREA_SI '                    &
   406)                                  , dsetrank, iparallel, .true.,foundvar)
   407)                call hdf_getslab_r( cpoly%moist_f    (is:is), 'MOIST_F '                    &
   408)                                  , dsetrank, iparallel, .true.,foundvar)
   409)                call hdf_getslab_r( cpoly%moist_W    (is:is), 'MOIST_W '                    &
   410)                                  , dsetrank, iparallel, .true.,foundvar)
   411)                call hdf_getslab_r( cpoly%elevation  (is:is), 'ELEVATION '                  &
   412)                                  , dsetrank, iparallel, .true.,foundvar)
   413)                call hdf_getslab_r( cpoly%slope      (is:is), 'SLOPE '                      &
   414)                                  , dsetrank, iparallel, .true.,foundvar)
   415)                call hdf_getslab_r( cpoly%aspect     (is:is), 'ASPECT '                     &
   416)                                  , dsetrank, iparallel, .true.,foundvar)
   417)                call hdf_getslab_r( cpoly%TCI        (is:is), 'TCI '                        &
   418)                                  , dsetrank, iparallel, .true.,foundvar)
   419)                call hdf_getslab_i( cpoly%patch_count(is:is), 'PATCH_COUNT '                &
   420)                                  , dsetrank, iparallel, .true.,foundvar)
   421) 
   422)                !----- Load 2D dataset. ----------------------------------------------------!
   423)                dsetrank     = 2_8
   424)                globdims(1)  = int(dset_nzg,8)     ! How many layers in the dataset?
   425)                chnkdims(1)  = int(1,8)            ! We are only extracting one layer
   426)                memdims(1)   = int(1,8)            ! We only need memory for one layer
   427)                memsize(1)   = int(1,8)            ! On both sides
   428)                chnkoffs(1)  = int(dset_nzg - 1,8) ! Take the top layer, not the bottom
   429)                memoffs(1)   = 0_8
   430) 
   431)                globdims(2)  = int(dset_nsites_global,8)
   432)                chnkdims(2)  = int(1,8)
   433)                chnkoffs(2)  = int(pysi_id(py_index) - 2 + isi,8)
   434)                memdims(2)   = int(1,8)
   435)                memsize(2)   = int(1,8)
   436)                memoffs(2)   = 0_8
   437)                call hdf_getslab_i(cpoly%ntext_soil(nzg:nzg,is:is),'NTEXT_SOIL '            &
   438)                                  ,dsetrank,iparallel,.true.,foundvar)
   439) 
   440)                !----- Now fill the soil column based on the top layer data. ---------------!
   441)                do k=1,nzg-1
   442)                   cpoly%ntext_soil(k,is) = cpoly%ntext_soil(nzg,is)
   443)                end do
   444) 
   445)                csite => cpoly%site(isi)
   446) 
   447)                if (sipa_n(si_index) > 0) then
   448) 
   449)                   !----- Fill 1D polygon (site unique) level variables. -------------------!
   450)                   call allocate_sitetype(csite,sipa_n(si_index))
   451) 
   452)                   !----- Reset the HDF5 auxiliary variables before moving to the next level. -!
   453)                   globdims = 0_8
   454)                   chnkdims = 0_8
   455)                   chnkoffs = 0_8
   456)                   memoffs  = 0_8
   457)                   memdims  = 0_8
   458)                   memsize  = 1_8
   459)                   !------------------------------------------------------------------------!
   460) 
   461)                   iparallel = 0
   462) 
   463)                   dsetrank = 1
   464)                   globdims(1) = int(dset_npatches_global,8)
   465)                   chnkdims(1) = int(csite%npatches,8)
   466)                   chnkoffs(1) = int(sipa_id(si_index) - 1,8)
   467)                   memdims(1)  = int(csite%npatches,8)
   468)                   memsize(1)  = int(csite%npatches,8)
   469)                   memoffs(1)  = 0
   470) 
   471)                   call hdf_getslab_i(csite%dist_type ,'DIST_TYPE '                         &
   472)                                     ,dsetrank,iparallel,.true.,foundvar)
   473)                   call hdf_getslab_r(csite%age       ,'AGE '                               &
   474)                                     ,dsetrank,iparallel,.true.,foundvar)
   475)                   call hdf_getslab_r(csite%area      ,'AREA '                              &
   476)                                     ,dsetrank,iparallel,.true.,foundvar)
   477)                   call hdf_getslab_r(csite%sum_dgd   ,'SUM_DGD '                           &
   478)                                     ,dsetrank,iparallel,.true.,foundvar)
   479)                   call hdf_getslab_r(csite%sum_chd   ,'SUM_CHD '                           &
   480)                                     ,dsetrank,iparallel,.true.,foundvar)
   481) 
   482)                   call hdf_getslab_r(csite%fast_soil_C       ,'FAST_SOIL_C '               &
   483)                                     ,dsetrank,iparallel,.true.,foundvar)
   484)                   call hdf_getslab_r(csite%slow_soil_C       ,'SLOW_SOIL_C '               &
   485)                                     ,dsetrank,iparallel,.true.,foundvar)
   486)                   call hdf_getslab_r(csite%fast_soil_N       ,'FAST_SOIL_N '               &
   487)                                     ,dsetrank,iparallel,.true.,foundvar)
   488)                   call hdf_getslab_r(csite%structural_soil_C ,'STRUCTURAL_SOIL_C '         &
   489)                                     ,dsetrank,iparallel,.true.,foundvar)
   490)                   call hdf_getslab_r(csite%structural_soil_L ,'STRUCTURAL_SOIL_L '         &
   491)                                     ,dsetrank,iparallel,.true.,foundvar)
   492)                   call hdf_getslab_r(csite%mineralized_soil_N,'MINERALIZED_SOIL_N '        &
   493)                                     ,dsetrank,iparallel,.true.,foundvar)
   494)                   !------------------------------------------------------------------------!
   495) 
   496) 
   497)                   !------------------------------------------------------------------------!
   498)                   !     Check whether the history file is new or old.  We determine this   !
   499)                   ! by searching for variable plantation.  In case this variable isn't     !
   500)                   ! present, then it must be the new history.   Otherwise we correct the   !
   501)                   ! indices for secondary forests.                                         !
   502)                   !------------------------------------------------------------------------!
   503)                   allocate (plantation(csite%npatches))
   504)                   plantation(:) = 0
   505)                   call hdf_getslab_i(plantation ,'PLANTATION '                             &
   506)                                     ,dsetrank,iparallel,.false.,foundvar)
   507)                   if (foundvar) then
   508)                      do ipa=1,csite%npatches
   509)                         select case(csite%dist_type(ipa))
   510)                         case (2)
   511)                            !---------------------------------------------------------------!
   512)                            !     Secondary forests are now divided in three categories.    !
   513)                            !---------------------------------------------------------------!
   514)                            if (plantation(ipa) == 0 .and. ianth_disturb == 0) then
   515)                               csite%dist_type(ipa) = 5
   516)                            else if (plantation(ipa) == 0 .and. ianth_disturb == 1) then
   517)                               csite%dist_type(ipa) = 6
   518)                            else
   519)                               csite%dist_type(ipa) = 2
   520)                            end if
   521)                            !---------------------------------------------------------------!
   522)                         end select
   523)                         !------------------------------------------------------------------!
   524)                      end do
   525)                      !---------------------------------------------------------------------!
   526)                   end if
   527)                   deallocate(plantation)
   528)                   !------------------------------------------------------------------------!
   529) 
   530)                   !------------------------------------------------------------------------!
   531)                   !     Loop over all sites and fill the patch-level variables.            !
   532)                   !------------------------------------------------------------------------!
   533)                   patchloop: do ipa = 1,csite%npatches
   534) 
   535)                      !---------------------------------------------------------------------!
   536)                      !     Reset the HDF5 auxiliary variables before moving to the next    !
   537)                      ! level.                                                              !
   538)                      !---------------------------------------------------------------------!
   539)                      globdims = 0_8
   540)                      chnkdims = 0_8
   541)                      chnkoffs = 0_8
   542)                      memoffs  = 0_8
   543)                      memdims  = 0_8
   544)                      memsize  = 1_8
   545)                      !---------------------------------------------------------------------!
   546) 
   547)                      cpatch => csite%patch(ipa)
   548) 
   549)                      !---------------------------------------------------------------------!
   550)                      !    Initialise patch-level variables that depend on the cohort ones. !
   551)                      !---------------------------------------------------------------------!
   552)                      csite%plant_ag_biomass(ipa)  = 0.0
   553) 
   554)                      pa_index = sipa_id(si_index) + ipa - 1
   555)                      call allocate_patchtype(cpatch,paco_n(pa_index))
   556) 
   557)                      !---------------------------------------------------------------------!
   558)                      !     Empty patches may exist, so make sure that this part is called  !
   559)                      ! only when there are cohorts.                                        !
   560)                      !---------------------------------------------------------------------!
   561)                      if (cpatch%ncohorts > 0) then
   562)                         !----- First the 1-D variables. -----------------------------------!
   563)                         dsetrank = 1
   564)                         globdims(1) = int(dset_ncohorts_global,8)
   565)                         chnkdims(1) = int(cpatch%ncohorts,8)
   566)                         chnkoffs(1) = int(paco_id(pa_index) - 1,8)
   567)                         memdims(1)  = int(cpatch%ncohorts,8)
   568)                         memsize(1)  = int(cpatch%ncohorts,8)
   569)                         memoffs(1)  = 0_8
   570) 
   571)                         call hdf_getslab_r(cpatch%dbh   ,'DBH '                            &
   572)                                           ,dsetrank,iparallel,.true.,foundvar)
   573)                         call hdf_getslab_r(cpatch%bdead ,'BDEAD '                          &
   574)                                           ,dsetrank,iparallel,.true.,foundvar)
   575)                         call hdf_getslab_i(cpatch%pft   ,'PFT '                            &
   576)                                           ,dsetrank,iparallel,.true.,foundvar)
   577)                         call hdf_getslab_r(cpatch%nplant,'NPLANT '                         &
   578)                                           ,dsetrank,iparallel,.true.,foundvar)
   579) 
   580)                         !------------------------------------------------------------------!
   581)                         !    Find derived properties from Bdead.  In the unlikely case     !
   582)                         ! that bdead is zero, then we use DBH as the starting point.  In   !
   583)                         ! both cases we assume that plants are in allometry.               !
   584)                         !------------------------------------------------------------------!
   585)                         do ico=1,cpatch%ncohorts
   586)                            ipft = cpatch%pft(ico)
   587) 
   588)                            if (igrass == 1 .and. is_grass(ipft)                            &
   589)                                            .and. cpatch%bdead(ico)>0.0) then
   590)                               !-- if the initial file was running with igrass = 0, bdead   !
   591)                               ! should be nonzero.  If the new run has igrass = 1, bdead   !
   592)                               ! is set to zero and that biomass is discarded               !
   593)                               cpatch%dbh(ico)   = max(cpatch%dbh(ico),min_dbh(ipft))
   594)                               cpatch%hite(ico)  = dbh2h (ipft,cpatch%dbh  (ico))
   595)                               cpatch%bdead(ico) = 0.0
   596) 
   597)                            else if (cpatch%bdead(ico) > 0.0 .and. igrass == 0) then
   598)                               ! grasses have bdead in both input and current run (igrass=0)
   599)                               cpatch%bdead(ico) = max(cpatch%bdead(ico),min_bdead(ipft))
   600)                               cpatch%dbh(ico)   = bd2dbh(ipft,cpatch%bdead(ico))
   601)                               cpatch%hite(ico)  = dbh2h (ipft,cpatch%dbh  (ico))
   602)                            else
   603)                               ! it is either a new grass (igrass=1) in the initial file,   !
   604)                               ! or the value for bdead is missing from the files           !
   605)                               cpatch%dbh(ico)   = max(cpatch%dbh(ico),min_dbh(ipft))
   606)                               cpatch%hite(ico)  = dbh2h (ipft,cpatch%dbh  (ico))
   607)                               cpatch%bdead(ico) = dbh2bd(cpatch%dbh  (ico),ipft)
   608)                            end if
   609) 
   610) 
   611)                            cpatch%bleaf(ico)  = size2bl(cpatch%dbh(ico),cpatch%hite(ico)   &
   612)                                                         ,cpatch%pft(ico))
   613) 
   614)                            !----- Find the other pools. -----------------------------------!
   615)                            salloc  = (1.0 + q(ipft) + qsw(ipft) * cpatch%hite(ico))
   616)                            salloci = 1.0 / salloc
   617)                            cpatch%balive  (ico) = cpatch%bleaf(ico) * salloc
   618)                            cpatch%broot   (ico) = cpatch%balive(ico) * q(ipft) * salloci
   619)                            cpatch%bsapwooda(ico) = cpatch%balive(ico) * qsw(ipft)          &
   620)                                                  * cpatch%hite(ico) * salloci * agf_bs(ipft)
   621)                            cpatch%bsapwoodb(ico) = cpatch%balive(ico) * qsw(ipft)          &
   622)                                                  * cpatch%hite(ico) * salloci              &
   623)                                                  * (1.-agf_bs(ipft))
   624)                            cpatch%bstorage(ico) = 0.0
   625)                            cpatch%phenology_status(ico) = 0
   626) 
   627) 
   628)                         end do
   629) 
   630) 
   631)                         !------------------------------------------------------------------!
   632)                         !     Carbon balance variables.                                    !
   633)                         ! MLO.  I commented this out because a restart is likely to have   !
   634)                         !       different settings and different environment, so it's      !
   635)                         !       likely that the system will reach a different equilibrium. !
   636)                         !       For simplicity, we just use the typical initialisation in  !
   637)                         !       which we give plants one year to adjust to the new         !
   638)                         !       conditions.                                                !
   639)                         !------------------------------------------------------------------!
   640)                         ! dsetrank    = 2
   641)                         ! globdims(1) = 13_8
   642)                         ! chnkdims(1) = 13_8
   643)                         ! chnkoffs(1) = 0_8
   644)                         ! memdims(1)  = 13_8
   645)                         ! memsize(1)  = 13_8
   646)                         ! memoffs(2)  = 0_8
   647)                         ! globdims(2) = int(dset_ncohorts_global,8)
   648)                         ! chnkdims(2) = int(cpatch%ncohorts,8)
   649)                         ! chnkoffs(2) = int(paco_id(pa_index) - 1,8)
   650)                         ! memdims(2)  = int(cpatch%ncohorts,8)
   651)                         ! memsize(2)  = int(cpatch%ncohorts,8)
   652)                         ! memoffs(2)  = 0_8
   653) 
   654)                         ! call hdf_getslab_r(cpatch%cb    ,'CB '                           &
   655)                         !                   ,dsetrank,iparallel,.true.,foundvar)
   656)                         ! call hdf_getslab_r(cpatch%cb_lightmax,'CB_LIGHTMAX '             &
   657)                         !                   ,dsetrank,iparallel,.true.,foundvar)
   658)                         ! call hdf_getslab_r(cpatch%cb_moistmax,'CB_MOISTMAX '             &
   659)                         !                   ,dsetrank,iparallel,.true.,foundvar)
   660)                         do ico = 1, cpatch%ncohorts
   661)                            cpatch%cb          (1:12,ico) = 1.0
   662)                            cpatch%cb_lightmax (1:12,ico) = 1.0
   663)                            cpatch%cb_moistmax (1:12,ico) = 1.0
   664)                            cpatch%cb_mlmax    (1:12,ico) = 1.0
   665)                            cpatch%cb          (  13,ico) = 0.0
   666)                            cpatch%cb_lightmax (  13,ico) = 0.0
   667)                            cpatch%cb_moistmax (  13,ico) = 0.0
   668)                            cpatch%cb_mlmax    (  13,ico) = 0.0
   669)                         end do
   670)                         !------------------------------------------------------------------!
   671) 
   672) 
   673) 
   674)                         cohortloop: do ico=1,cpatch%ncohorts
   675)                            !---------------------------------------------------------------!
   676)                            !    We will now check the PFT of each cohort, so we determine  !
   677)                            ! if this is a valid PFT.  If not, then we must decide what we  !
   678)                            ! should do...                                                  !
   679)                            !---------------------------------------------------------------!
   680)                            if (.not. include_pft(cpatch%pft(ico))) then
   681)                               select case(pft_1st_check)
   682)                               case (0)
   683)                                  !----- Stop the run. -------------------------------------!
   684)                                  write (unit=*,fmt='(a,1x,i5,1x,a)')                       &
   685)                                        'I found a cohort with PFT=',cpatch%pft(ico)        &
   686)                                       ,' and it is not in your include_these_pft...'
   687)                                  call fatal_error('Invalid PFT in history file'            &
   688)                                                  ,'read_ed21_history_file'                 &
   689)                                                  ,'ed_read_ed21_history.F90')
   690) 
   691)                               case (1)
   692)                                  !----- Include the unexpected PFT in the list. -----------!
   693)                                  write (unit=*,fmt='(a,1x,i5,1x,a)')                       &
   694)                                        'I found a cohort with PFT=',cpatch%pft(ico)        &
   695)                                       ,'... Including this PFT in your include_these_pft...'
   696)                                  include_pft(cpatch%pft(ico))          = .true.
   697)                                  include_these_pft(count(include_pft)) = cpatch%pft(ico)
   698) 
   699)                                  call sort_up(include_these_pft,n_pft)
   700) 
   701)                                  if (is_grass(cpatch%pft(ico))) then
   702)                                     include_pft_ag(cpatch%pft(ico)) = .true.
   703)                                  end if
   704) 
   705)                               case (2)
   706)                                  !----- Ignore the unexpect PFT. --------------------------!
   707)                                  write (unit=*,fmt='(a,1x,i5,1x,a)')                       &
   708)                                        'I found a cohort with PFT=',cpatch%pft(ico)        &
   709)                                       ,'... Ignoring it...'
   710)                                  !---------------------------------------------------------!
   711)                                  !    The way we will ignore this cohort is by setting its !
   712)                                  ! nplant to zero, and calling the "terminate_cohorts"     !
   713)                                  ! subroutine right after this.                            !
   714)                                  !---------------------------------------------------------!
   715)                                  cpatch%nplant(ico) = 0.
   716)                               end select
   717)                            end if
   718) 
   719)                            !---------------------------------------------------------------!
   720)                            !     Make sure that the biomass won't lead to FPE.  This       !
   721)                            ! should never happen when using a stable ED-2.1 version, but   !
   722)                            ! older versions had "zombie" cohorts.  Here we ensure that     !
   723)                            ! the model initialises with stable numbers whilst ensuring     !
   724)                            ! that the cohorts will be eliminated.                          !
   725)                            !---------------------------------------------------------------!
   726)                            if (cpatch%balive(ico) > 0.            .and.                    &
   727)                                cpatch%balive(ico) < tiny_biomass) then
   728)                               cpatch%balive(ico) = tiny_biomass
   729)                            end if
   730)                            if (cpatch%bleaf(ico) > 0.                .and.                 &
   731)                                cpatch%bleaf(ico) < tiny_biomass)     then
   732)                               cpatch%bleaf(ico) = tiny_biomass
   733)                            end if
   734)                            if (cpatch%broot(ico) > 0.                .and.                 &
   735)                                cpatch%broot(ico) < tiny_biomass)     then
   736)                               cpatch%broot(ico) = tiny_biomass
   737)                            end if
   738)                            if (cpatch%bsapwooda(ico) > 0.            .and.                 &
   739)                                cpatch%bsapwooda(ico) < tiny_biomass) then
   740)                               cpatch%bsapwooda(ico) = tiny_biomass
   741)                            end if
   742)                            if (cpatch%bsapwoodb(ico) > 0.            .and.                 &
   743)                                cpatch%bsapwoodb(ico) < tiny_biomass) then
   744)                               cpatch%bsapwoodb(ico) = tiny_biomass
   745)                            end if
   746)                            if (cpatch%bdead(ico) > 0.                .and.                 &
   747)                                cpatch%bdead(ico) < tiny_biomass)     then
   748)                               cpatch%bdead(ico) = tiny_biomass
   749)                            end if
   750)                            if (cpatch%bstorage(ico) > 0.             .and.                 &
   751)                                cpatch%bstorage(ico) < tiny_biomass)  then
   752)                               cpatch%bstorage(ico) = tiny_biomass
   753)                            end if
   754)                            !---------------------------------------------------------------!
   755) 
   756) 
   757) 
   758) 
   759)                            !----- Compute the above-ground biomass. -----------------------!
   760)                            cpatch%agb(ico) = ed_biomass(cpatch, ico)
   761)                            cpatch%basarea(ico)  = pio4 * cpatch%dbh(ico) * cpatch%dbh(ico)
   762) 
   763) 
   764)                            !----- Assign LAI, WAI, and CAI --------------------------------!
   765)                            call area_indices(cpatch, ico)
   766) 
   767) 
   768)                            !----- Update the derived patch-level variables. ---------------!
   769)                            csite%plant_ag_biomass(ipa) = csite%plant_ag_biomass(ipa)       &
   770)                                                        + cpatch%agb(ico)*cpatch%nplant(ico)
   771) 
   772)                            !----- Initialise the other cohort level variables. ------------!
   773)                            call init_ed_cohort_vars(cpatch,ico,cpoly%lsl(isi))
   774)                         end do cohortloop
   775) 
   776)                         !------------------------------------------------------------------!
   777)                         !    Eliminate any "unwanted" cohort (i.e., those which nplant was !
   778)                         ! set to zero so it would be removed).                             !
   779)                         !------------------------------------------------------------------!
   780)                         call terminate_cohorts(csite,ipa,elim_nplant,elim_lai)
   781) 
   782)                      end if
   783)                   end do patchloop
   784)                else
   785)                   !----- This should never happen, but, just in case... -------------------!
   786)                   call fatal_error('A site with no patches was found...'                   &
   787)                                   ,'read_ed21_history_file','ed_read_ed21_history.F90')
   788)                end if
   789) 
   790)                !----- Initialise the other patch-level variables. -------------------------!
   791)                call init_ed_patch_vars(csite,1,csite%npatches,cpoly%lsl(isi))
   792)             end if
   793)             !------------------------------------------------------------------------------!
   794)          end do siteloop
   795)          !---------------------------------------------------------------------------------!
   796) 
   797)          deallocate(islakesite)
   798) 
   799)          !----- Initialise some site-level variables. -------------------------------------!
   800)          call init_ed_site_vars(cpoly)
   801)          !---------------------------------------------------------------------------------!
   802) 
   803)       end do polyloop
   804) 
   805) 
   806)       !----- Initialise the other polygon-level variables. --------------------------------!
   807)       call init_ed_poly_vars(cgrid)
   808) 
   809) 
   810)       !----- Close the dataset. -----------------------------------------------------------!
   811)       call h5fclose_f(file_id, hdferr)
   812)       if (hdferr /= 0) then
   813)          write (unit=*,fmt='(a,1x,a)') 'File: ',trim(hnamel)
   814)          write (unit=*,fmt='(a)'     ) 'Problem: Failed closing the HDF5 dataset.'
   815)          write (unit=*,fmt='(a,1x,i4)') 'HDFerr: ',hdferr
   816)          call fatal_error('Could not close the HDF file'                                   &
   817)                          ,'read_ed21_history_file','ed_read_ed21_history.F90')
   818)       end if
   819) 
   820)       !------ Deallocate the temporary vectors, so no memory leak happens. ----------------!
   821)       deallocate(file_lats)
   822)       deallocate(file_lons)
   823)       deallocate(paco_n   )
   824)       deallocate(paco_id  )
   825)       deallocate(sipa_n   )
   826)       deallocate(sipa_id  )
   827)       deallocate(pysi_n   )
   828)       deallocate(pysi_id  )
   829) 
   830)    end do gridloop
   831) 
   832)    !----- Reset auto error printing to "on". ----------------------------------------------!
   833)    call h5eset_auto_f(1,hdferr)
   834) 
   835)    !----- Close the file. -----------------------------------------------------------------!
   836)    call h5close_f(hdferr)
   837) 
   838) #else
   839)    call fatal_error ('You cannot restart with ED-2.1 without using HDF5...'                &
   840)                     ,'read_ed21_history_file','ed_read_ed21_history.F90')
   841) #endif
   842) 
   843)    return
   844) end subroutine read_ed21_history_file

   865) #endif

   880)                              , pft_1st_check           & ! intent(in)

   882)                              , agf_bs                  ! ! intent(in)
   883)    use ed_misc_coms   , only : sfilin                  & ! intent(in)
   884)                              , ied_init_mode           & ! intent(in)

  1742)                      chnkoffs = 0_8
  1743)                      memoffs  = 0_8
  1744)                      memdims  = 0_8
  1745)                      memsize  = 1_8
  1746)                      !---------------------------------------------------------------------!
  1747) 
  1748) 
  1749) 
  1750)                      !---------------------------------------------------------------------!
  1751)                      !     Initialise patch-level variables that depend on the cohort      !
  1752)                      ! ones.                                                               !
  1753)                      !---------------------------------------------------------------------!
  1754)                      csite%plant_ag_biomass(ipa)  = 0.0
  1755) 
  1756)                      pa_index = sipa_id(si_index) + ipa - 1
  1757)                      call allocate_patchtype(cpatch,paco_n(pa_index))
  1758) 
  1759)                      !---------------------------------------------------------------------!
  1760)                      !     Empty patches may exist, so make sure that this part is called  !
  1761)                      ! only when there are cohorts.                                        !
  1762)                      !---------------------------------------------------------------------!
  1763)                      if (cpatch%ncohorts > 0) then
  1764)                         !----- First the 1-D variables. -----------------------------------!
  1765)                         dsetrank = 1
  1766)                         globdims(1) = int(dset_ncohorts_global,8)
  1767)                         chnkdims(1) = int(cpatch%ncohorts,8)
  1768)                         chnkoffs(1) = int(paco_id(pa_index) - 1,8)
  1769)                         memdims(1)  = int(cpatch%ncohorts,8)
  1770)                         memsize(1)  = int(cpatch%ncohorts,8)
  1771)                         memoffs(1)  = 0_8
  1772) 
  1773)                         call hdf_getslab_r(cpatch%dbh             ,'DBH '                  &
  1774)                                           ,dsetrank,iparallel,.true.,foundvar)
  1775)                         call hdf_getslab_r(cpatch%bdead           ,'BDEAD '                &
  1776)                                           ,dsetrank,iparallel,.true.,foundvar)
  1777)                         call hdf_getslab_i(cpatch%pft             ,'PFT '                  &
  1778)                                           ,dsetrank,iparallel,.true.,foundvar)
  1779)                         call hdf_getslab_r(cpatch%nplant          ,'NPLANT '               &
  1780)                                           ,dsetrank,iparallel,.true.,foundvar)
  1781) 
  1782)                         !------------------------------------------------------------------!
  1783)                         !    Find derived properties from Bdead.  In the unlikely case     !
  1784)                         ! that bdead is zero, then we use DBH as the starting point.  In   !
  1785)                         ! both cases we assume that plants are in allometry.               !
  1786)                         !------------------------------------------------------------------!
  1787)                         do ico=1,cpatch%ncohorts
  1788)                            ipft = cpatch%pft(ico)
  1789) 
  1790)                            if (igrass == 1 .and. is_grass(ipft)                            &
  1791)                                            .and. cpatch%bdead(ico)>0.0) then
  1792)                               !-- if the initial file was running with igrass = 0, bdead   !
  1793)                               ! should be nonzero.  If the new run has igrass = 1, bdead   !
  1794)                               ! is set to zero and the mass is discarded                   !
  1795)                               cpatch%dbh(ico)   = max(cpatch%dbh(ico),min_dbh(ipft))
  1796)                               cpatch%hite(ico)  = dbh2h (ipft,cpatch%dbh  (ico))
  1797)                               cpatch%bdead(ico) = 0.0
  1798) 
  1799)                            else if (cpatch%bdead(ico) > 0.0 .and. igrass == 0) then
  1800)                               ! grasses have bdead in both input and current run (igrass=0)
  1801)                               cpatch%bdead(ico) = max(cpatch%bdead(ico),min_bdead(ipft))
  1802)                               cpatch%dbh(ico)   = bd2dbh(ipft,cpatch%bdead(ico))
  1803)                               cpatch%hite(ico)  = dbh2h (ipft,cpatch%dbh  (ico))
  1804)                            else
  1805)                               ! it is either a new grass (igrass=1) in the initial file,   !
  1806)                               ! or the value for bdead is missing from the files           !
  1807)                               cpatch%dbh(ico)   = max(cpatch%dbh(ico),min_dbh(ipft))
  1808)                               cpatch%hite(ico)  = dbh2h (ipft,cpatch%dbh  (ico))
  1809)                               cpatch%bdead(ico) = dbh2bd(cpatch%dbh  (ico),ipft)
  1810)                            end if
  1811) 
  1812)                            cpatch%bleaf(ico)  = size2bl( cpatch%dbh (ico)                  &
  1813)                                                        , cpatch%hite(ico)                  &
  1814)                                                        , ipft )
  1815) 
  1816)                            !----- Find the other pools. -----------------------------------!
  1817)                            salloc  = (1.0 + q(ipft) + qsw(ipft) * cpatch%hite(ico))
  1818)                            salloci = 1.0 / salloc
  1819)                            cpatch%balive  (ico)  = cpatch%bleaf(ico)  * salloc
  1820)                            cpatch%broot    (ico) = cpatch%balive(ico) * q(ipft) * salloci
  1821)                            cpatch%bsapwooda(ico) = cpatch%balive(ico) * qsw(ipft)          &
  1822)                                                  * cpatch%hite(ico) * salloci * agf_bs(ipft)
  1823)                            cpatch%bsapwoodb(ico) = cpatch%balive(ico) * qsw(ipft)          &
  1824)                                                  * cpatch%hite(ico) * salloci              &
  1825)                                                  * (1.-agf_bs(ipft))
  1826)                            cpatch%bstorage(ico)  = 0.0
  1827)                            cpatch%phenology_status(ico) = 0
  1828) 
  1829)                         end do
  1830) 
  1831)                         !------------------------------------------------------------------!
  1832)                         !     Carbon balance variables.                                    !
  1833)                         ! MLO.  I commented this out because a restart is likely to have   !
  1834)                         !       different settings and different environment, so it's      !
  1835)                         !       likely that the system will reach a different equilibrium. !
  1836)                         !       For simplicity, we just use the typical initialisation in  !
  1837)                         !       which we give plants one year to adjust to the new         !
  1838)                         !       conditions.                                                !
  1839)                         !------------------------------------------------------------------!
  1840)                         ! dsetrank    = 2
  1841)                         ! globdims(1) = 13_8
  1842)                         ! chnkdims(1) = 13_8
  1843)                         ! chnkoffs(1) = 0_8
  1844)                         ! memdims(1)  = 13_8
  1845)                         ! memsize(1)  = 13_8
  1846)                         ! memoffs(2)  = 0_8
  1847)                         ! globdims(2) = int(dset_ncohorts_global,8)
  1848)                         ! chnkdims(2) = int(cpatch%ncohorts,8)
  1849)                         ! chnkoffs(2) = int(paco_id(pa_index) - 1,8)
  1850)                         ! memdims(2)  = int(cpatch%ncohorts,8)
  1851)                         ! memsize(2)  = int(cpatch%ncohorts,8)
  1852)                         ! memoffs(2)  = 0_8
  1853) 
  1854)                         ! call hdf_getslab_r(cpatch%cb    ,'CB '                           &
  1855)                         !                   ,dsetrank,iparallel,.true.,foundvar)
  1856)                         ! call hdf_getslab_r(cpatch%cb_lightmax,'CB_LIGHTMAX '             &
  1857)                         !                   ,dsetrank,iparallel,.true.,foundvar)
  1858)                         ! call hdf_getslab_r(cpatch%cb_moistmax,'CB_MOISTMAX '             &
  1859)                         !                   ,dsetrank,iparallel,.true.,foundvar)
  1860)                         do ico = 1, cpatch%ncohorts
  1861)                            cpatch%cb          (1:12,ico) = 1.0
  1862)                            cpatch%cb_lightmax (1:12,ico) = 1.0
  1863)                            cpatch%cb_moistmax (1:12,ico) = 1.0
  1864)                            cpatch%cb_mlmax    (1:12,ico) = 1.0
  1865)                            cpatch%cb          (  13,ico) = 0.0
  1866)                            cpatch%cb_lightmax (  13,ico) = 0.0
  1867)                            cpatch%cb_moistmax (  13,ico) = 0.0
  1868)                            cpatch%cb_mlmax    (  13,ico) = 0.0
  1869)                         end do
  1870)                         !------------------------------------------------------------------!
  1871) 
  1872)                         cohortloop: do ico=1,cpatch%ncohorts
  1873)                            !---------------------------------------------------------------!
  1874)                            !    We will now check the PFT of each cohort, so we determine  !
  1875)                            ! if this is a valid PFT.  If not, then we must decide what we  !
  1876)                            ! should do...                                                  !
  1877)                            !---------------------------------------------------------------!
  1878)                            if (.not. include_pft(cpatch%pft(ico))) then
  1879)                               select case(pft_1st_check)
  1880)                               case (0)
  1881)                                  !----- Stop the run. -------------------------------------!
  1882)                                  write (unit=*,fmt='(a,1x,i5,1x,a)')                       &
  1883)                                        'I found a cohort with PFT=',cpatch%pft(ico)        &
  1884)                                       ,' and it is not in your include_these_pft...'
  1885)                                  call fatal_error('Invalid PFT in history file'            &
  1886)                                                  ,'read_ed21_history_file'                 &
  1887)                                                  ,'ed_read_ed21_history.F90')
  1888) 
  1889)                               case (1)
  1890)                                  !----- Include the unexpected PFT in the list. -----------!
  1891)                                  write (unit=*,fmt='(a,1x,i5,1x,a)')                       &
  1892)                                       'I found a cohort with PFT=',cpatch%pft(ico)         &
  1893)                                      ,'... Including this PFT in your include_these_pft...'
  1894)                                  include_pft(cpatch%pft(ico))          = .true.
  1895)                                  include_these_pft(count(include_pft)) = cpatch%pft(ico)
  1896) 
  1897)                                  call sort_up(include_these_pft,n_pft)
  1898) 
  1899)                                  if (is_grass(cpatch%pft(ico))) then
  1900)                                     include_pft_ag(cpatch%pft(ico)) = .true.
  1901)                                  end if
  1902) 
  1903)                               case (2)
  1904)                                  !----- Ignore the unexpect PFT. --------------------------!
  1905)                                  write (unit=*,fmt='(a,1x,i5,1x,a)')                       &
  1906)                                        'I found a cohort with PFT=',cpatch%pft(ico)        &
  1907)                                       ,'... Ignoring it...'
  1908)                                  !---------------------------------------------------------!
  1909)                                  !    The way we will ignore this cohort is by setting its !
  1910)                                  ! nplant to zero, and calling the "terminate_cohorts"     !
  1911)                                  ! subroutine right after this.                            !
  1912)                                  !---------------------------------------------------------!
  1913)                                  cpatch%nplant(ico) = 0.
  1914)                               end select
  1915)                            end if
  1916) 
  1917)                            !---------------------------------------------------------------!
  1918)                            !     Make sure that the biomass won't lead to FPE.  This       !
  1919)                            ! should never happen when using a stable ED-2.1 version, but   !
  1920)                            ! older versions had "zombie" cohorts.  Here we ensure that     !
  1921)                            ! the model initialises with stable numbers whilst ensuring     !
  1922)                            ! that the cohorts will be eliminated.                          !
  1923)                            !---------------------------------------------------------------!
  1924)                            if (cpatch%balive(ico) > 0.            .and.                    &
  1925)                                cpatch%balive(ico) < tiny_biomass) then
  1926)                               cpatch%balive(ico) = tiny_biomass
  1927)                            end if
  1928)                            if (cpatch%bleaf(ico) > 0.            .and.                     &
  1929)                                cpatch%bleaf(ico) < tiny_biomass) then
  1930)                               cpatch%bleaf(ico) = tiny_biomass
  1931)                            end if
  1932)                            if (cpatch%broot(ico) > 0.            .and.                     &
  1933)                                cpatch%broot(ico) < tiny_biomass) then
  1934)                               cpatch%broot(ico) = tiny_biomass
  1935)                            end if
  1936)                            if (cpatch%bsapwooda(ico) > 0.        .and.                     &
  1937)                                cpatch%bsapwooda(ico) < tiny_biomass) then
  1938)                               cpatch%bsapwooda(ico) = tiny_biomass
  1939)                            end if
  1940)                            if (cpatch%bsapwoodb(ico) > 0.        .and.                     &
  1941)                                cpatch%bsapwoodb(ico) < tiny_biomass) then
  1942)                               cpatch%bsapwoodb(ico) = tiny_biomass
  1943)                            end if
  1944)                            if (cpatch%bdead(ico) > 0.            .and.                     &
  1945)                                cpatch%bdead(ico) < tiny_biomass) then
  1946)                               cpatch%bdead(ico) = tiny_biomass
  1947)                            end if
  1948)                            if (cpatch%bstorage(ico) > 0.            .and.                  &
  1949)                                cpatch%bstorage(ico) < tiny_biomass) then
  1950)                               cpatch%bstorage(ico) = tiny_biomass
  1951)                            end if
  1952)                            !---------------------------------------------------------------!
  1953) 
  1954) 
  1955)                            !----- Compute the above-ground biomass. -----------------------!
  1956)                            cpatch%agb(ico) = ed_biomass(cpatch, ico)
  1957) 
  1958)                            cpatch%basarea(ico)  = pio4 * cpatch%dbh(ico) * cpatch%dbh(ico)
  1959) 
  1960)                            !----- Assign LAI, WAI, and CAI --------------------------------!
  1961)                            call area_indices(cpatch, ico)
  1962) 
  1963) 
  1964)                            !----- Update the derived patch-level variables. ---------------!
  1965)                            csite%plant_ag_biomass(ipa) = csite%plant_ag_biomass(ipa)       &
  1966)                                                        + cpatch%agb(ico)*cpatch%nplant(ico)
  1967) 
  1968)                            !----- Initialise the other cohort level variables. ------------!
  1969)                            call init_ed_cohort_vars(cpatch,ico,cpoly%lsl(isi))
  1970)                         end do cohortloop
  1971) 
  1972)                         !------------------------------------------------------------------!
  1973)                         !    Eliminate any "unwanted" cohort (i.e., those which nplant was !
  1974)                         ! set to zero so it would be removed).                             !
  1975)                         !------------------------------------------------------------------!
  1976)                         call terminate_cohorts(csite,ipa,elim_nplant,elim_lai)
  1977) 
  1978)                      end if
  1979)                   end do patchloop
  1980)                else
  1981)                   !----- This should never happen, but, just in case... -------------------!
  1982)                   call fatal_error('A site with no patches was found...'                   &
  1983)                                   ,'read_ed21_history_file','ed_read_ed21_history.F90')
  1984)                end if
  1985) 
  1986)                !----- Initialise the other patch-level variables. -------------------------!
  1987)                call init_ed_patch_vars(csite,1,csite%npatches,cpoly%lsl(isi))
  1988) 
  1989)             end do siteloop2
  1990) 
  1991) 
  1992)             !----- Initialise some site-level variables. ----------------------------------!
  1993)             call init_ed_site_vars(cpoly)
  1994)             !------------------------------------------------------------------------------!
  1995) 
  1996)             !----- Deallocate the temporary polygon and soil structure. -------------------!
  1997)             deallocate (this_ntext        )
  1998)             deallocate (tpoly_area        )
  1999)             deallocate (tpoly_moist_f     )
  2000)             deallocate (tpoly_moist_w     )
  2001)             deallocate (tpoly_elevation   )
  2002)             deallocate (tpoly_slope       )
  2003)             deallocate (tpoly_aspect      )
  2004)             deallocate (tpoly_TCI         )
  2005)             deallocate (tpoly_patch_count )
  2006)             deallocate (tpoly_lsl         )
  2007)             deallocate (tpoly_ntext_soil  )
  2008)             deallocate (islakesite        )
  2009)             !------------------------------------------------------------------------------!
  2010) 
  2011)          end do polyloop
  2012) 
  2013)          !----- Close the dataset. --------------------------------------------------------!
  2014)          call h5fclose_f(file_id, hdferr)
  2015)          if (hdferr /= 0) then
  2016)             write (unit=*,fmt='(a,1x,a)') 'File: ',trim(hnamel)
  2017)             write (unit=*,fmt='(a)'     ) 'Problem: Failed closing the HDF5 dataset.'
  2018)             write (unit=*,fmt='(a,1x,i4)') 'HDFerr: ',hdferr
  2019)             call fatal_error('Could not close the HDF file'                                &
  2020)                             ,'read_ed21_history_file','ed_read_ed21_history.F90')
  2021)          end if
  2022) 
  2023)          deallocate(paco_n    ,paco_id  )
  2024)          deallocate(sipa_n    ,sipa_id  )
  2025)          deallocate(pysi_n    ,pysi_id  )
  2026)       end do rstfileloop
  2027) 
  2028)       !----- Initialise the other polygon-level variables. --------------------------------!
  2029)       call init_ed_poly_vars(cgrid)
  2030) 
  2031)       !----- Deallocate the closest index vector. -----------------------------------------!
  2032)       deallocate(pclosest,psrcfile)
  2033)    end do gridloop
  2034) 
  2035)    !----- Turn off automatic error printing. ----------------------------------------------!
  2036)    call h5eset_auto_f(1,hdferr)
  2037) 
  2038)    !----- Close the HDF5 environment. -----------------------------------------------------!
  2039)    call h5close_f(hdferr)
  2040) 
  2041) #else
  2042)    call fatal_error ('You cannot restart with ED-2.1 without using HDF5...'                &
  2043)                     ,'read_ed21_history_unstruct','ed_read_ed21_history.F90')
  2044) #endif
  2045) 
  2046) 
  2047)    return
