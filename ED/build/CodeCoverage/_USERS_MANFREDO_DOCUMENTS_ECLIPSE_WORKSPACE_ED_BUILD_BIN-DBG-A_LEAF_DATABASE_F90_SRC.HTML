     1) !==========================================================================================!
     2) !==========================================================================================!
     3) subroutine leaf_database(ofn,nsite,nlandsea,iaction,lat,lon,classout,pctout)
     4) 
     5)    use hdf5_utils , only : shdf5_open_f  & ! subroutine
     6)                          , shdf5_close_f & ! subroutine
     7)                          , shdf5_irec_f  ! ! subroutine
     8)    use soil_coms  , only : nslcon        & ! intent(in)
     9)                          , isoilcol      & ! intent(in)
    10)                          , ed_nstyp      & ! intent(in)
    11)                          , ed_nvtyp      ! ! intent(in)
    12)    implicit none
    13)    !----- Arguments. ----------------------------------------------------------------------!
    14)    character(len=*)                          , intent(in)  :: ofn
    15)    character(len=*)                          , intent(in)  :: iaction
    16)    integer                                   , intent(in)  :: nsite
    17)    integer                                   , intent(in)  :: nlandsea
    18)    real           , dimension(    3,nlandsea), intent(in)  :: lat
    19)    real           , dimension(    3,nlandsea), intent(in)  :: lon
    20)    integer        , dimension(nsite,nlandsea), intent(out) :: classout
    21)    real           , dimension(nsite,nlandsea), intent(out) :: pctout
    22)    !----- Variables describing the files. -------------------------------------------------!
    23)    integer                                                 :: nio
    24)    integer                                                 :: njo
    25)    integer                                                 :: nperdeg
    26)    integer                                                 :: niosh
    27)    integer                                                 :: njosh
    28)    integer                                                 :: ifiles
    29)    integer                                                 :: jfiles
    30)    real                                                    :: offpix
    31)    integer                                                 :: west
    32)    integer                                                 :: south
    33)    !----- Variables for file handling. ----------------------------------------------------!
    34)    integer                                                 :: iwoc
    35)    integer                                                 :: isoc
    36)    integer                                                 :: isocpt
    37)    integer                                                 :: isocpo
    38)    integer                                                 :: iwocpt
    39)    integer                                                 :: iwocpo
    40)    integer                                                 :: iwocph
    41)    character(len=3)                                        :: title1
    42)    character(len=4)                                        :: title2
    43)    character(len=128)                                      :: title3
    44)    logical                                                 :: l1
    45)    logical                                                 :: l2
    46)    integer                                                 :: ndims
    47)    integer        , dimension(2)                           :: idims
    48)    integer                                                 :: ierr
    49)    !----- Variables for file indexing. ----------------------------------------------------!
    50)    integer        , dimension(:,:)           , allocatable :: nump
    51)    integer        , dimension(:,:,:)         , allocatable :: cnr_ind
    52)    integer(kind=4), dimension(:,:)           , allocatable :: idato
    53)    integer        , dimension(4*nlandsea)                  :: cnr_i1
    54)    integer        , dimension(4*nlandsea)                  :: cnr_i2
    55)    integer        , dimension(4*nlandsea)                  :: cnr_j1
    56)    integer        , dimension(4*nlandsea)                  :: cnr_j2
    57)    integer        , dimension(4*nlandsea)                  :: cnr_ipoly
    58)    integer        , dimension(:,:)           , allocatable :: class_count
    59)    integer        , dimension(:)             , allocatable :: rankpct
    60)    real           , dimension(:)             , allocatable :: fraction
    61)    real                                                    :: total_count
    62)    integer                                                 :: ifile
    63)    integer                                                 :: ifile2
    64)    integer                                                 :: jfile
    65)    integer                                                 :: jfile2
    66)    integer                                                 :: jo_full
    67)    integer                                                 :: io_full
    68)    integer                                                 :: ind
    69)    integer                                                 :: io_loc
    70)    integer                                                 :: jo_loc
    71)    integer                                                 :: max_per_file
    72)    integer                                                 :: cid
    73)    integer                                                 :: i
    74)    integer                                                 :: irank
    75)    integer                                                 :: itext
    76)    integer                                                 :: j
    77)    integer                                                 :: i1
    78)    integer                                                 :: i2
    79)    integer                                                 :: j1
    80)    integer                                                 :: j2
    81)    integer                                                 :: ic
    82)    integer                                                 :: jc
    83)    integer                                                 :: ilandsea
    84)    integer                                                 :: dq
    85)    integer                                                 :: ed_nctyp
    86)    !------ External functions. ------------------------------------------------------------!
    87)    integer, external                                       :: find_rank
    88)    !---------------------------------------------------------------------------------------!
    89) 
    90) 
    91) 
    92) 
    93)    !----- Allocate some auxiliary variables and initialise them. --------------------------!
    94)    ed_nctyp = max(ed_nvtyp,ed_nstyp)
    95)    allocate(class_count(0:ed_nctyp,nlandsea))
    96)    allocate(rankpct(ed_nstyp))
    97)    allocate(fraction(ed_nstyp))
    98)    class_count(:,:) = 0
    99)    rankpct    (:)   = 0
   100)    fraction   (:)   = 0.
   101)    !---------------------------------------------------------------------------------------!
   102) 
   103) 
   104) 
   105)    !---------------------------------------------------------------------------------------!
   106)    !     Read header file.                                                                 !
   107)    !---------------------------------------------------------------------------------------!
   108)    write(unit=*,fmt='(2a)') trim(ofn),'HEADER'
   109)    open (unit=29,file=trim(ofn)//'HEADER',form='formatted',status='old')
   110)    read (unit=29,fmt=*,iostat=ierr) nio, njo, nperdeg,west,south
   111)    !----- Check whether this was read sucessfully.  If not, rewind it and assume global. --!
   112)    if (ierr /= 0) then
   113)       rewind(unit=29)
   114)       read  (unit=29,fmt=*) nio, njo, nperdeg
   115)       west  = -180
   116)       south =  -90
   117)    end if
   118)    close(unit=29,status='keep')
   119)    !---------------------------------------------------------------------------------------!
   120) 
   121) 
   122) 
   123)    !---------------------------------------------------------------------------------------!
   124)    !     Compute number of pixels in a shift to adjacent file (niosh and njosh).  Compute  !
   125)    ! number of files in database that span all latitudes and longitudes on Earth.  [This   !
   126)    ! algorithm will change when multiple resolutions of the SRTM data become available.]   !
   127)    !---------------------------------------------------------------------------------------!
   128)    if (mod(nio,nperdeg) == 2) then
   129)       offpix = .5
   130)       niosh = nio - 2
   131)       njosh = njo - 2
   132)    else
   133)       offpix = 0.
   134)       niosh = nio - 1
   135)       njosh = njo - 1
   136)    end if
   137)    !---------------------------------------------------------------------------------------!
   138) 
   139)    !---------------------------------------------------------------------------------------!
   140)    !     Compute number of files in input dataset that span all latitudes and longitudes   !
   141)    ! on Earth.  This could be done more efficiently, but for the time being we will only   !
   142)    ! check the southernmost and westernmost points, and assume that the data set could go  !
   143)    ! to the North Pole and the International Date Line.                                    !
   144)    !---------------------------------------------------------------------------------------!
   145)    ifiles = (180 -  west) * nperdeg / niosh
   146)    jfiles = ( 90 - south) * nperdeg / njosh
   147)    !---------------------------------------------------------------------------------------!
   148) 
   149) 
   150) 
   151)    !----- Allocate nump and idato. --------------------------------------------------------!
   152)    allocate (nump    (ifiles,jfiles))
   153)    allocate (idato   (nio,njo))
   154) 
   155)    do jfile = 1,jfiles
   156)       do ifile = 1,ifiles
   157)          nump(ifile,jfile) = 0
   158)       end do
   159)    end do
   160) 
   161) 
   162)    !---------------------------------------------------------------------------------------!
   163)    !      Get file index (ifile,jfile) within full dataset and count the number of corners !
   164)    ! that appear in every file.                                                            !
   165)    !---------------------------------------------------------------------------------------!
   166)    max_per_file = 0
   167)    do ilandsea = 1, nlandsea
   168)       do ic=2,3
   169)          do jc=2,3
   170)             call get_file_indices(lat(jc,ilandsea),lon(ic,ilandsea),west,south,nperdeg     &
   171)                                  ,niosh,njosh,io_full,jo_full,io_loc,jo_loc,ifile,jfile)
   172) 
   173)             nump(ifile,jfile) = nump(ifile,jfile) + 1
   174)             if(nump(ifile,jfile)>max_per_file) max_per_file = nump(ifile,jfile)
   175)          end do
   176)       end do
   177)    end do
   178) 
   179)    !----- Allocate the mapping of file indices to global indices. -------------------------!
   180)    allocate (cnr_ind(ifiles,jfiles,max_per_file))
   181)    
   182)    !----- Re-zero nump. -------------------------------------------------------------------!
   183)    nump(:,:) = 0
   184)    ind       = 0
   185)    do ilandsea = 1, nlandsea
   186)       
   187)       !------ Add all four corners to the search process. ---------------------------------!
   188)       do ic=2,3
   189)          do jc=2,3
   190) 
   191)             ind = ind+1
   192)             i2  = 5-ic
   193)             j2  = 5-jc
   194) 
   195)             !----- Get the file indices and the local indices within the file. ------------!
   196)             call get_file_indices(lat(jc,ilandsea),lon(ic,ilandsea),west,south,nperdeg     &
   197)                                  ,niosh,njosh,io_full,jo_full,io_loc,jo_loc,ifile,jfile)
   198) 
   199)             nump(ifile,jfile) = nump(ifile,jfile) + 1
   200) 
   201)             cnr_ind(ifile,jfile,nump(ifile,jfile)) = ind
   202) 
   203)             cnr_ipoly(ind) = ilandsea
   204)             
   205)             !----- This is the local index where you find the starting point. -------------!
   206)             cnr_i1(ind) = io_loc
   207)             cnr_j1(ind) = jo_loc
   208) 
   209)             !----- Get the local indices of the other corner. -----------------------------!
   210)             call get_file_indices(lat(j2,ilandsea),lon(i2,ilandsea),west,south,nperdeg      &
   211)                                  ,niosh,njosh,io_full,jo_full,io_loc,jo_loc,ifile2,jfile2)
   212) 
   213)             !----- This is the local index where you find the second i point. -------------!
   214)             if (ifile2 == ifile) then
   215)                cnr_i2(ind) = io_loc
   216)             else if (ifile2 > ifile) then
   217)                cnr_i2(ind) = niosh+1
   218)             else
   219)                cnr_i2(ind) = 2
   220)             end if
   221) 
   222)             !----- This is the local index where you find the second j point. -------------!
   223)             if (jfile2 == jfile) then
   224)                cnr_j2(ind) = jo_loc
   225)             else if (jfile2 > jfile) then
   226)                cnr_j2(ind) = njosh+1
   227)             else
   228)                cnr_j2(ind) = 2
   229)             end if
   230)          end do
   231)       end do
   232)    end do
   233) 
   234) 
   235)    !----- Read files and extract data. ----------------------------------------------------!
   236)    do jfile = 1,jfiles
   237)       do ifile = 1,ifiles
   238)          
   239)          if  (nump(ifile,jfile) > 0) then
   240)             !----- SW longitude of current file. ------------------------------------------!
   241)             iwoc = (ifile - 1) * niosh/nperdeg + west
   242)             !----- SW latitude of current file. -------------------------------------------!
   243)             isoc = (jfile - 1) * njosh/nperdeg + south
   244)             
   245)             !----- Construct filename. ----------------------------------------------------!
   246)             isocpt = abs(isoc) / 10
   247)             isocpo = abs(isoc) - isocpt*10
   248)             iwocph = abs(iwoc) / 100
   249)             iwocpt = (abs(iwoc) - iwocph * 100) / 10
   250)             iwocpo = abs(iwoc) - iwocph * 100 - iwocpt * 10
   251)             
   252)             if (isoc >= 0) then
   253)                write(title1,'(2i1,a1)') isocpt,isocpo,'N'
   254)             else
   255)                write(title1,'(2i1,a1)') isocpt,isocpo,'S'
   256)             end if
   257)             
   258)             if (iwoc >= 0) then
   259)                write(title2,'(3i1,a1)') iwocph,iwocpt,iwocpo,'E'
   260)             else
   261)                write(title2,'(3i1,a1)') iwocph,iwocpt,iwocpo,'W'
   262)             end if
   263)             
   264)             title3 = trim(ofn)//title1//title2//'.h5'
   265)             inquire(file=trim(title3),exist=l1,opened=l2)
   266)             
   267)             !----- Read file. -------------------------------------------------------------!
   268)             if (l1) then
   269)                write (unit=*,fmt='(a,1x,a,a)') ' -> Getting file:',trim(title3),'...'
   270)                
   271)                call shdf5_open_f(trim(title3),'R')
   272)                
   273)                ndims = 2
   274)                idims(1) = nio
   275)                idims(2) = njo
   276)                
   277)                select case (trim(iaction))
   278)                case ('leaf_class')
   279)                   call shdf5_irec_f(ndims,idims,'oge2',ivara=idato)
   280)                case ('soil_text')
   281)                   call shdf5_irec_f(ndims,idims,'fao',ivara=idato)
   282)                case ('soil_col')
   283)                   call shdf5_irec_f(ndims,idims,'colour',ivara=idato)
   284)                case default
   285)                   call fatal_error('Incorrect action specified in leaf_database'           &
   286)                                   ,'leaf_database','leaf_database.f90')
   287)                end select
   288) 
   289)                call shdf5_close_f()
   290)             else
   291)                write (unit=*,fmt='(a,1x,a,a)') ' -> File:',trim(title3)                    &
   292)                                               ,'doesn''t exist.  Using default values...'
   293)                select case (trim(iaction))
   294)                case ('leaf_class')
   295)                   idato(:,:) = 0
   296)                case ('soil_text')
   297)                   idato(:,:) = nslcon
   298)                case ('soil_col')
   299)                   idato(:,:) = isoilcol
   300)                case default
   301)                   call fatal_error('Incorrect action specified in leaf_database'           &
   302)                                   ,'leaf_database','leaf_database.f90')
   303)                end select
   304)             end if
   305)             !------------------------------------------------------------------------------!
   306) 
   307) 
   308) 
   309) 
   310)             !------------------------------------------------------------------------------!
   311)             !     Loop through all of the corner points that are inside this file.         !
   312)             !------------------------------------------------------------------------------!
   313)             do cid = 1,nump(ifile,jfile)
   314) 
   315)                !------ Get the global index of this point. --------------------------------!
   316)                ind = cnr_ind(ifile,jfile,cid)
   317)                !---------------------------------------------------------------------------!
   318) 
   319) 
   320)                !------ Get the polygon index of this corner. ------------------------------!
   321)                ilandsea = cnr_ipoly(ind)
   322)                !---------------------------------------------------------------------------!
   323) 
   324) 
   325)                if (cnr_i1(ind) < cnr_i2(ind)) then
   326)                   i1 = cnr_i1(ind)
   327)                   i2 = cnr_i2(ind)
   328)                else
   329)                   i1 = cnr_i2(ind)
   330)                   i2 = cnr_i1(ind)
   331)                end if
   332)                
   333)                if (cnr_j1(ind) < cnr_j2(ind)) then
   334)                   j1 = cnr_j1(ind)
   335)                   j2 = cnr_j2(ind)
   336)                else
   337)                   j1 = cnr_j2(ind)
   338)                   j2 = cnr_j1(ind)
   339)                endif
   340) 
   341)                !----- Now walk through the point spaces for soil or vegetation class. -----!
   342)                select case (trim(iaction))
   343)                case ('leaf_class')
   344)                   do j=j1,j2
   345)                      do i=i1,i2
   346)                         call ed_datp_datq(idato(i,j),dq)
   347)                         class_count(dq,ilandsea) = class_count(dq,ilandsea) + 1
   348)                      end do
   349)                   end do
   350)                      
   351)                case ('soil_text','soil_col')
   352)                   do j=j1,j2
   353)                      do i=i1,i2
   354)                         dq = idato(i,j)
   355)                         class_count(dq,ilandsea) = class_count(dq,ilandsea) + 1
   356)                      end do
   357)                   end do
   358) 
   359)                end select
   360)                !---------------------------------------------------------------------------!
   361)             end do
   362)          end if
   363)       end do
   364)    end do
   365)    !---------------------------------------------------------------------------------------!
   366) 
   367) 
   368)    !---------------------------------------------------------------------------------------!
   369)    !      Decide what to do based on whether we are loading land classes or soil classes.  !
   370)    !---------------------------------------------------------------------------------------!
   371)    select case (trim(iaction))
   372)    case ('leaf_class')
   373)       !------------------------------------------------------------------------------------!
   374)       !     Leaf class.  We really don't care about the leaf class, we just want to know   !
   375)       ! the percentage that is not water.                                                  !
   376)       !------------------------------------------------------------------------------------!
   377)       do ilandsea = 1,nlandsea
   378) 
   379)          total_count          = real(sum(class_count(0:ed_nvtyp,ilandsea)))
   380)          pctout  (:,ilandsea) = 0.
   381)          pctout  (1,ilandsea) = real(sum(class_count(2:ed_nvtyp,ilandsea))) / total_count
   382)          !----- This won't be used, assign any number. ------------------------------------!
   383)          classout(:,ilandsea) = 6
   384)       end do
   385)       !------------------------------------------------------------------------------------!
   386) 
   387)    case ('soil_text')
   388)       !------------------------------------------------------------------------------------!
   389)       !     We must determine the abundance of each soil classes so that the output will   !
   390)       ! be have the commonest soil type as the first, and the rarest as the last.          !
   391)       !------------------------------------------------------------------------------------!
   392)       do ilandsea = 1,nlandsea
   393)          !---------------------------------------------------------------------------------!
   394)          !     We don't consider soil type 0 as water is determined by other means.  If    !
   395)          ! that causes the total_count variable to be zero, we assume that only one site   !
   396)          ! can exist, with the user-defined default.                                       !
   397)          !---------------------------------------------------------------------------------!
   398)          total_count         = real(sum(class_count(1:ed_nstyp,ilandsea)))
   399)          if (total_count == 0.) then
   400)             pctout(:,ilandsea) = 0.
   401)             pctout(1,ilandsea) = 1.
   402)             !------------------------------------------------------------------------------!
   403)             !    Assign the default class as this may be used if soil and land use maps    !
   404)             ! disagree about whether a place is land or water.                             !
   405)             !------------------------------------------------------------------------------!
   406)             classout(:,ilandsea) = nslcon
   407)             !------------------------------------------------------------------------------!
   408) 
   409)          else
   410)             !------ Normalise the areas. --------------------------------------------------!
   411)             fraction(1:ed_nstyp) = real(class_count(1:ed_nstyp,ilandsea)) / total_count
   412)             !------------------------------------------------------------------------------!
   413) 
   414)             !------ Rank the sites (1 = commonest, ed_nstyp = rarest). --------------------!
   415)             call rank_down(ed_nstyp,fraction,rankpct)
   416)             !------------------------------------------------------------------------------!
   417) 
   418) 
   419)             !------ Fill in sites with the right order. -----------------------------------!
   420)             do irank = 1,nsite
   421)                !----- Itext is the texture type that has rank irank. ----------------------!
   422)                itext = find_rank(irank,ed_nstyp,rankpct)
   423)                
   424)                classout(irank,ilandsea) = itext
   425)                pctout  (irank,ilandsea) = fraction(itext)
   426)             end do
   427)             !------------------------------------------------------------------------------!
   428)          end if
   429)       end do
   430)    end select
   431)    !---------------------------------------------------------------------------------------!
   432) 
   433) 
   434) 
   435)    !----- Free memory so it won't leak it. ------------------------------------------------!
   436)    deallocate(nump       )
   437)    deallocate(idato      )
   438)    deallocate(cnr_ind    )
   439)    deallocate(class_count)
   440)    deallocate(rankpct    )
   441)    deallocate(fraction   )
   442)    !---------------------------------------------------------------------------------------!
   443)   return
   444) end subroutine leaf_database

   455) !     This subroutine looks for the indices of a given longitude and latitude within an    !
   456) ! input file.                                                                              !
   457) !------------------------------------------------------------------------------------------!
   458) subroutine get_file_indices(lat,lon,west,south,nper,ni,nj,io_full,jo_full,io_loc,jo_loc    &
   459)                            ,ifile,jfile)
   460)    implicit none
   461)    !----- Arguments. ----------------------------------------------------------------------!
   462)    real   , intent(in)  :: lat
   463)    real   , intent(in)  :: lon
   464)    integer, intent(in)  :: west
   465)    integer, intent(in)  :: south
   466)    integer, intent(in)  :: nper
   467)    integer, intent(in)  :: ni
   468)    integer, intent(in)  :: nj
   469)    integer, intent(out) :: io_full
   470)    integer, intent(out) :: jo_full
   471)    integer, intent(out) :: io_loc
   472)    integer, intent(out) :: jo_loc
   473)    integer, intent(out) :: ifile
   474)    integer, intent(out) :: jfile
   475)    !----- Local variables. ----------------------------------------------------------------!
   476)    real                 :: wlat
   477)    real                 :: wlon
   478)    real                 :: rio_full
   479)    real                 :: rjo_full
   480)    !---------------------------------------------------------------------------------------!
   481) 
   482) 
   483) 
   484)    !----- Set the lat and lon for file lookup. --------------------------------------------!
   485)    wlat = max(-89.9999,min(89.9999,lat))
   486)    wlon = lon
   487)    if(wlon >= 180.) wlon = wlon - 360.
   488)    if(wlon < -180.) wlon = wlon + 360.
   489)    wlon = max(-179.9999,min(179.9999,wlon))
   490)    
   491)    rio_full = (wlon -  west) * real(nper) ! must ignore pixel offset here
   492)    rjo_full = (wlat - south) * real(nper) ! must ignore pixel offset here
   493)    
   494)    io_full = int(rio_full)
   495)    jo_full = int(rjo_full)
   496)    
   497)    ifile = io_full / ni + 1
   498)    jfile = jo_full / nj + 1
   499)    
   500)    io_loc = mod(io_full,ni) + 1
   501)    jo_loc = mod(jo_full,nj) + 1
   502)    
   503)    return
   504) end subroutine get_file_indices

   513) !==========================================================================================!
   514) !==========================================================================================!
   515) !      This subroutine maps the input datp classes to a smaller set datq which represents  !
   516) ! the full set of LEAF-2 or LEAF-3 classes for which LSP values are defined.  We don't use !
   517) ! the vegetation information for anything regarding ED plant communities: all we want to   !
   518) ! know is whether the place is water or land.                                              !
   519) !------------------------------------------------------------------------------------------!
   520) subroutine ed_datp_datq(datp,datq)
   521) 
   522) 
   523)    implicit none
   524)    !----- Arguments. ----------------------------------------------------------------------!
   525)    integer, intent(in)       :: datp      ! Input data, the Olson Global Ecosystems dataset
   526)    integer, intent(out)      :: datq      ! Output data, the LEAF-3 classification. 
   527)    !----- Local variable. -----------------------------------------------------------------!
   528)    integer, dimension(0:100) :: oge2leaf3 ! Conversion table
   529)    !---------------------------------------------------------------------------------------!
   530) 
   531) 
   532) 
   533)    !---------------------------------------------------------------------------------------!
   534)    !     Olson Global Ecosystems dataset OGE_2 (96 classes) mapped to LEAF-3 classes (see  !
   535)    ! leaf3_document). 97 and 98 are not used, 99 is Goode Homolosine empty data, and 100   !
   536)    ! is missing data. They are all mapped to ocean.                                        !
   537)    !---------------------------------------------------------------------------------------!
   538)    oge2leaf3 = (/                                      0, &  !   0 -   0
   539)                   19,  8,  4,  5,  6,  7,  9,  3, 11, 16, &  !   1 -  10
   540)                   10,  2, 20,  0,  0, 12, 13, 14, 18,  4, &  !  11 -  20
   541)                    4,  4, 14, 14,  6,  6,  4,  7,  7, 15, &  !  21 -  30
   542)                   15,  6,  7,  7, 15, 16, 16, 16, 16,  8, &  !  31 -  40
   543)                    8,  8, 18, 17, 17, 12, 12,  7, 10,  3, &  !  41 -  50

   663) !     This sub-routine reads in the soil depth files.                                      !
   664) !------------------------------------------------------------------------------------------!
   665) subroutine read_soil_depth()
   666)    use soil_coms, only : soildepth_db   & ! intent(in)
   667)                        , isoildepthflg  & ! intent(in)
   668)                        , nlon_lyr       & ! intent(in)
   669)                        , nlat_lyr       & ! intent(in)
   670)                        , layer_index    & ! intent(in)
   671)                        , slz            ! ! intent(in)
   672)    use grid_coms, only : nzg            ! ! intent(in)
   673)    implicit none
   674)    !----- Local variables. ----------------------------------------------------------------!
   675)    real, dimension(nlat_lyr,nlon_lyr) :: soil_depth
   676)    logical                            :: found
   677)    real                               :: rdum
   678)    integer                            :: ilat
   679)    integer                            :: k
   680)    !---------------------------------------------------------------------------------------!
   681) 
   682) 
   683)    !----- Allocate the scratch array containing the soil depth index layer. ---------------!
   684)    if (allocated(layer_index)) deallocate(layer_index)
   685)    allocate(layer_index(nlat_lyr,nlon_lyr))
   686)    
   687)    !---------------------------------------------------------------------------------------! 
   688)    !     If the user doesn't want to use the soil depth database, set all layer indices to !
   689)    ! 1 and quit.  This will make all soil layers active across all polygons.               !
   690)    !---------------------------------------------------------------------------------------! 
   691)    if (isoildepthflg /= 1) then 
   692)       layer_index(:,:) = 1
   693)       return
   694)    end if
   695)    !---------------------------------------------------------------------------------------! 
   696) 
   697) 
   698)    !---------------------------------------------------------------------------------------! 
   699)    !    If we reach this point, the user provided soil depth data.  Initialise the array   !
   700)    ! by assigning just the top 2 layers, which is the minimum required. 
   701)    !---------------------------------------------------------------------------------------! 
   702)    layer_index=nzg-1
   703)    !---------------------------------------------------------------------------------------! 
   704) 
   705) 
   706) 
   707)    !----- Open the file, provided it exists. If it doesn't, abort the run.-----------------!
   708)    inquire (file=trim(soildepth_db),exist=found)
   709)    if(.not.found) then
   710)       write(unit=*,fmt='(a)') 'Your ED2IN has isoildepthflg=1, however the file'
   711)       write(unit=*,fmt='(a)') trim(soildepth_db)//' doesn''t exist.'
   712)       call fatal_error('File specified in SOILDEPTH_DB not found!'                         &
   713)                       ,'read_soil_depth','leaf_database.f90')
   714)    end if
   715)    open (unit=12,file=trim(soildepth_db),form='formatted',action='read',status='old')
   716)    !---------------------------------------------------------------------------------------! 
   717) 
   718)    !---------------------------------------------------------------------------------------! 
   719)    !      Read in the database.  At this point we are assuming the grid resolution of the  !
   720)    ! input data.  We may need to re-visit this if we ever find another soil depth dataset. !
   721)    ! For the current data, the first latitude bin is 90N-89N, and the first longitude bin  !
   722)    ! is 180W-179W.                                                                         !
   723)    !---------------------------------------------------------------------------------------! 
   724)    do ilat=1,nlat_lyr
   725)       read(unit=12,fmt=*) rdum,soil_depth(ilat,1:nlon_lyr)
   726)    end do
   727)    !---------------------------------------------------------------------------------------! 
   728) 
   729) 
   730)    !------ Close the file, we got all that we need from it. -------------------------------!
   731)    close(unit=12,status='keep')
   732)    !---------------------------------------------------------------------------------------! 
   733) 
   734) 
   735) 
   736)    !----- Convert it to metres and use the same notation as ED (negative values) ----------!
   737)    soil_depth(:,:) = - soil_depth(:,:) * 0.01
   738)    !---------------------------------------------------------------------------------------! 
   739) 
   740) 
   741)    !---------------------------------------------------------------------------------------!
   742)    !     Do something in case the atmospheric land cell happens to fall in an FAO water    !
   743)    ! cell.                                                                                 !
   744)    !---------------------------------------------------------------------------------------!
   745)    where (soil_depth(:,:) == 0.0) soil_depth(:,:) = slz(1)
   746)    !---------------------------------------------------------------------------------------!
   747) 
   748) 
   749)    !----- Loop through the layers, and find points in which the soil is shallower. --------!
   750)    do k = nzg-2,1,-1
   751)       where (soil_depth(:,:) < slz(k+1)) layer_index(:,:) = k
   752)    end do
   753)    !---------------------------------------------------------------------------------------!
   754)    
   755)    return
   756) end subroutine read_soil_depth
