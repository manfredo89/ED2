     3) !      This subroutine prints patch, cohort, polygon or site level data, upscales that     !
     4) ! data to the site level and stores the data in its spatial array coordinate.  The data is !
     5) ! then printed to the screen, based on a specified window of data.  Note, this may be      !
     6) ! printing windows on various nodes, or this may be called from a master process.  Be      !
     7) ! conscious of this; as it will dictate what part of the domain you are accessing vari-    !
     8) ! ables from, and whether or not the variable of interest is stored in memory at that      !
     9) ! time.  For instance, many variables are stored only on the slave nodes, and need not be  !
    10) ! passed back to the master.  Likewise, many slave node data will accumulate after each    !
    11) ! lsm call, until they are passed back to the master, where they are normalized.  These    !
    12) ! variables will be immediately zeroed on the slaves after being sent to the master.       !
    13) ! Don't forget to adjust the number precision on the format string at the end.             !
    14) !------------------------------------------------------------------------------------------!
    15) subroutine print_fields(ifm,cgrid)
    16)  
    17)    use ed_node_coms , only : mynum         & ! intent(in)
    18)                            , nnodetot      ! ! intent(in)
    19)    use ed_state_vars, only : edtype        & ! structure
    20)                            , polygontype   ! ! structure
    21)    use ed_misc_coms , only : printvars     & ! intent(in)
    22)                            , ipmax         & ! intent(in)
    23)                            , ipmin         & ! intent(in)
    24)                            , pfmtstr       ! ! intent(in)
    25)    use ed_var_tables, only : vt_info       & ! intent(in)
    26)                            , num_var       ! ! intent(in)
    27)    use ed_max_dims  , only : str_len_short ! ! intent(in)
    28)    implicit none
    29)    !----- Standard common blocks. ---------------------------------------------------------!
    30) #if defined(RAMS_MPI)
    31)    include 'mpif.h'
    32) #endif
    33)    !----- Arguments. ----------------------------------------------------------------------!
    34)    integer                            , intent(in) :: ifm
    35)    type(edtype)                       , target     :: cgrid
    36)    !----- Local variables. ----------------------------------------------------------------!
    37) #if defined(RAMS_MPI)
    38)    integer, dimension(MPI_STATUS_SIZE)             :: status
    39) #endif
    40)    real   , dimension(:)              , pointer    :: pvar_l
    41)    real   , dimension(:)              , pointer    :: pvar_g
    42)    character(len=30)                               :: fmtstr
    43)    character(len=str_len_short)                    :: pvar_name
    44)    integer                                         :: nv
    45)    integer                                         :: np
    46)    integer                                         :: i
    47)    integer                                         :: ip
    48)    integer                                         :: ping
    49)    integer                                         :: npolys
    50)    integer                                         :: g_idmin
    51)    integer                                         :: g_idmax
    52)    integer                                         :: l_idmin
    53)    integer                                         :: l_idmax
    54)    integer                                         :: node_idmin
    55)    integer                                         :: node_idmax
    56)    integer                                         :: g_id
    57)    integer                                         :: g_ln
    58)    integer                                         :: ncols
    59)    integer                                         :: row
    60)    integer                                         :: maxrows
    61)    integer                                         :: col
    62)    logical                                         :: pvartrue
    63)    logical                                         :: ptr_send
    64)    !----- Local constants. ----------------------------------------------------------------!
    65)    integer                            , parameter  :: maxcols  = 10
    66)    real                               , parameter  :: undef    = -99.9
    67)    !---------------------------------------------------------------------------------------!
    68) 
    69)    
    70)    
    71)    
    72)    !----- Ping is just any flag to be used by MPI. ----------------------------------------!
    73)    ping   = 8675309
    74) 
    75)    !----- Find the number of polygons to be plotted. --------------------------------------!
    76)    npolys = ipmax - ipmin + 1
    77)    
    78) 
    79)    !----- Adjust the format string according to the chosen variables. ---------------------!
    80) 
    81)    !----- Check the window size. ----------------------------------------------------------!
    82)    if (ipmax > cgrid%npolygons_global) then
    83)       write(unit=*,fmt='(a)')       '====================================================='
    84)       write(unit=*,fmt='(a,1x,i5)') ' IPMAX                     = ',ipmax
    85)       write(unit=*,fmt='(a,1x,i5)') ' Global number of polygons = ',cgrid%npolygons_global
    86)       write(unit=*,fmt='(a)')       ' '
    87)       write(unit=*,fmt='(a)')       ' You have specified a print index greater than the'
    88)       write(unit=*,fmt='(a)')       ' total number of polygons.  Please reduce IPMAX...'
    89)       write(unit=*,fmt='(a)')       '====================================================='
    90)       call fatal_error('Too many polygons to be written on standard output.'               &
    91)                       ,'print_fields','ed_print.f90')
    92)    end if
    93) 
    94) 
    95)    !---------------------------------------------------------------------------------------!
    96)    !     Allocate the print (always) and scratch vector (if this is the "master" node or a ! 
    97)    ! serial run.                                                                           !
    98)    !---------------------------------------------------------------------------------------!
    99)    allocate(pvar_l(npolys))
   100)    if (mynum == nnodetot .or. nnodetot == 1) allocate(pvar_g(npolys))
   101)    !---------------------------------------------------------------------------------------!
   102) 
   103)    
   104)    !----- Loop through the printvar entries from the namelist. ----------------------------!
   105)    ip = 0
   106)    count_pvars: do
   107) 
   108)       ip = ip+1
   109)       pvar_name = printvars(ip)
   110)       if (len_trim(pvar_name) == str_len_short) then
   111)          exit count_pvars
   112)       endif
   113) 
   114) #if defined(RAMS_MPI)
   115)       if (nnodetot /= 1) call MPI_Barrier(MPI_COMM_WORLD,ierr)
   116) #endif
   117) 
   118)       pvartrue = .false.
   119)       do nv = 1,num_var(ifm)
   120) 
   121)          if (trim(vt_info(nv,ifm)%name) .eq. trim(pvar_name)) then
   122)             pvartrue = .true.
   123)             
   124)             !------------------------------------------------------------------------------!
   125)             !     If this is root, then collect the sends, keep reading to find out what   !
   126)             ! it is receiving.                                                             !
   127)             !------------------------------------------------------------------------------!
   128)             if (nnodetot > 1) then
   129)                
   130)                if (mynum == nnodetot) then
   131)                   
   132)                   pvar_g = undef
   133)                   !----- Loop through the variable table to match the print variable. -----!
   134)                   write(unit=*,fmt='( a)') ' '
   135)                   write(unit=*,fmt='(3a)') ' =========== ',trim(pvar_name),' =========== '
   136)                   write(unit=*,fmt='( a)') ' '
   137) 
   138) #if defined(RAMS_MPI)
   139)                   do nm = 1,nnodetot-1
   140) 
   141)                      call MPI_Recv(ptr_recv,1,MPI_LOGICAL,machs(nm),120,MPI_COMM_WORLD     &
   142)                                   ,status,ierr)
   143) 
   144)                      if (ptr_recv) then
   145)                         call MPI_Recv(mast_idmin,1,MPI_INTEGER,machs(nm),121               &
   146)                                      ,MPI_COMM_WORLD,status,ierr)
   147) 
   148)                         call MPI_Recv(mast_idmax,1,MPI_INTEGER,machs(nm),122               &
   149)                                      ,MPI_COMM_WORLD,status,ierr)
   150) 
   151)                         call MPI_Recv(pvar_g(mast_idmin:mast_idmax)                        &
   152)                                      ,mast_idmax-mast_idmin+1,MPI_REAL                     &
   153)                                      ,machs(nm),123,MPI_COMM_WORLD,status,ierr)
   154)                      end if
   155)                   end do
   156) #endif
   157)                end if
   158)             else
   159)                
   160)                !----- Loop through the variable table to match the print variable. --------!
   161)                write(unit=*,fmt='( a)') ' '
   162)                write(unit=*,fmt='(3a)') ' =========== ',trim(pvar_name),' =========== '
   163)                write(unit=*,fmt='( a)') ' '
   164)                pvar_g = undef
   165)             end if
   166)             !------------------------------------------------------------------------------!
   167) 
   168) 
   169) 
   170)             !------------------------------------------------------------------------------!
   171)             !     The namelist print entry has been matched with the var_table entry.  Now !
   172)             ! lets cycle through our machines and determine if those machines hold data    !
   173)             ! that should be printed.  If so, then send that data to the master (last      !
   174)             ! machine).                                                                    !
   175)             !------------------------------------------------------------------------------!
   176)             !----- This is scratch space that all machines will use. ----------------------!
   177)             pvar_l = undef
   178)             !----- Set the blocking recieve to allow ordering, start with machine 1. ------!
   179) #if defined(RAMS_MPI)
   180)             if (mynum /= 1) then
   181)                call MPI_Recv(ping,1,MPI_INTEGER,recvnum,93,MPI_COMM_WORLD,status,ierr)
   182)             end if
   183) #endif
   184) 
   185)             !------------------------------------------------------------------------------!
   186)             !     Cycle through this node's pointers for the current variable.  If the     !
   187)             ! index falls within the printable range. Save that pointer to a local array.  !
   188)             ! Once all the pointers have been cycled, send the local array to the master   !
   189)             ! to populate a global array and print.                                        !
   190)             !------------------------------------------------------------------------------!
   191)             ptr_send   = .false.
   192)             node_idmin = -1
   193)             node_idmax = -1
   194)             do np = 1,vt_info(nv,ifm)%nptrs
   195)                g_id = vt_info(nv,ifm)%vt_vector(np)%globid+1
   196)                g_ln = vt_info(nv,ifm)%vt_vector(np)%varlen
   197)                !---------------------------------------------------------------------------!
   198)                !      Determine if any of this segment falls within the range desired for  !
   199)                ! output, globid+1 is the global index.                                     !
   200)                !---------------------------------------------------------------------------!
   201)                if (g_id <= ipmax .and. g_id+g_ln-1 >= ipmin ) then
   202) 
   203)                   !---- OK, this segment is good, set the send flag. ----------------------!
   204)                   ptr_send = .true.
   205)                   !------------------------------------------------------------------------!
   206)                   !     These are the indices of the data in the current segment to use    !
   207)                   ! and the indices in the global array they will be sent to.              !
   208)                   !------------------------------------------------------------------------!
   209)                   if (g_id >= ipmin) then
   210)                      l_idmin = 1
   211)                      g_idmin = g_id - ipmin + 1
   212)                   else
   213)                      l_idmin = ipmin - g_id + 1
   214)                      g_idmin = 1
   215)                   end if
   216) 
   217)                   if (g_id+g_ln-1 < ipmax) then
   218)                      l_idmax = g_ln
   219)                      g_idmax = g_id + g_ln - ipmin
   220)                   else
   221)                      l_idmax = ipmax - g_id + 1
   222)                      g_idmax = ipmax - ipmin + 1
   223)                   end if
   224)                   
   225)                   !------------------------------------------------------------------------!
   226)                   !     These should be the same size, if not stop.                        !
   227)                   !------------------------------------------------------------------------!
   228)                   if (l_idmax - l_idmin /= g_idmax - g_idmin ) then
   229)                      write(unit=*,fmt='(a)'      ) '--------------------------------------'
   230)                      write(unit=*,fmt='(a,1x,i5)') ' L_IDMIN  = ',l_idmin
   231)                      write(unit=*,fmt='(a,1x,i5)') ' L_IDMAX  = ',l_idmax
   232)                      write(unit=*,fmt='(a,1x,i5)') ' G_IDMIN  = ',g_idmin
   233)                      write(unit=*,fmt='(a,1x,i5)') ' G_IDMAX  = ',g_idmax
   234)                      write(unit=*,fmt='(a,1x,i5)') ' L_IDSIZE = ',l_idmax - l_idmin + 1
   235)                      write(unit=*,fmt='(a,1x,i5)') ' G_IDSIZE = ',g_idmax - g_idmin + 1
   236)                      write(unit=*,fmt='(a)'      ) '--------------------------------------'
   237)                      call fatal_error('L_IDSIZE and G_IDSIZE don''t match!'                &
   238)                                      ,'print_fields','ed_print.F90')
   239)                   end if
   240)                   !------------------------------------------------------------------------!
   241) 
   242) 
   243) 
   244)                   !------------------------------------------------------------------------!
   245)                   !     Shift the global dataset so that it is applied to the first index. !
   246)                   ! Note that only real arrays can be used this time.                      !
   247)                   !------------------------------------------------------------------------!
   248)                   select case (vt_info(nv,ifm)%dtype)
   249)                   case ('R')
   250)                      call fillvar_l(pvar_l,vt_info(nv,ifm)%vt_vector(np)%var_rp,npolys     &
   251)                                    ,g_ln,g_idmin,l_idmin,l_idmax)
   252)                   case default
   253)                      write(unit=*,fmt='(a)')      '---------------------------------------'
   254)                      write(unit=*,fmt='(a,1x,a)') ' VNAME: ',vt_info(nv,ifm)%name
   255)                      write(unit=*,fmt='(a,1x,a)') ' DTYPE: ',vt_info(nv,ifm)%dtype
   256)                      write(unit=*,fmt='(a)')      '---------------------------------------'
   257)                      call fatal_error('Only variables of DTYPE=R can be printed...'        &
   258)                                      ,'print_fields','ed_print.f90')
   259)                   end select
   260)                   !------------------------------------------------------------------------!
   261) 
   262)                   !----- Determine the minimum and maximum indices that will be sent. -----!
   263)                   if (g_idmin < node_idmin .or. node_idmin == -1) node_idmin = g_idmin
   264)                   if (g_idmax > node_idmax .or. node_idmax == -1) node_idmax = g_idmax
   265)                   !------------------------------------------------------------------------!
   266)                end if
   267)             end do
   268) 
   269)             !------------------------------------------------------------------------------!
   270)             !     Decide what to do based on whether this is a serial or parallel run.     !
   271)             !------------------------------------------------------------------------------!
   272)             if (nnodetot > 1) then
   273) 
   274)                !------ Parallel run, decide whether this is a master or a slave node. -----!
   275)                if (mynum /= nnodetot) then
   276)                   !------------------------------------------------------------------------!
   277)                   !      Slave node.  The local array for this machine has been created.   !
   278)                   ! Send it off to the master.                                             !
   279)                   !------------------------------------------------------------------------!
   280) #if defined(RAMS_MPI)
   281) 
   282)                   call MPI_Send(ptr_send,1,MPI_LOGICAL,machs(nnodetot),120,MPI_COMM_WORLD  &
   283)                                ,ierr)
   284) 
   285)                   if (ptr_send) then
   286)                      call MPI_Send(node_idmin,1,MPI_INTEGER,machs(nnodetot),121            &
   287)                                   ,MPI_COMM_WORLD,ierr)
   288) 
   289)                      call MPI_Send(node_idmax,1,MPI_INTEGER,machs(nnodetot),122            &
   290)                                   ,MPI_COMM_WORLD,ierr)
   291) 
   292)                      call MPI_Send(pvar_l(node_idmin:node_idmax),node_idmax-node_idmin+1   &
   293)                                   ,MPI_REAL,machs(nnodetot),123,MPI_COMM_WORLD,ierr)
   294)                   end if
   295)                   
   296) 
   297)                   !------------------------------------------------------------------------!
   298)                   !     When this node is finished, send the blocking MPI_Send to the next !
   299)                   ! machine.                                                               !
   300)                   !------------------------------------------------------------------------!
   301)                   call MPI_Send(ping,1,MPI_INTEGER,sendnum,93,MPI_COMM_WORLD,ierr)
   302) #else
   303)                   continue
   304) #endif
   305) 
   306)                else
   307)                   !----- Master node, just copy the local array to the global. ------------!
   308)                   if (ptr_send) then
   309)                      pvar_g(node_idmin:node_idmax) = pvar_l(node_idmin:node_idmax)
   310)                   end if
   311)                end if
   312)             else
   313)                !----- Serial run, simply copy the entire local array to the global. -------!
   314)                pvar_g = pvar_l
   315)             end if
   316)             !------------------------------------------------------------------------------!
   317) 
   318) 
   319) 
   320)             !------------------------------------------------------------------------------!
   321)             !      The data over the sought range of indices have been collected.  If this !
   322)             ! is the only machine or the root machine, then print it to standard output.   !
   323)             !------------------------------------------------------------------------------!
   324)             if (mynum == nnodetot .or. nnodetot == 1) then
   325) 
   326)                !----- Print out a maximum of 10 variables per row... ----------------------!
   327)                maxrows = ceiling(real(npolys)/real(maxcols))
   328) 
   329)                do row = 1,maxrows
   330)                   ncols = min( maxcols,npolys-((row-1)*maxcols)   )
   331)                   col   = ( (row-1)*maxcols)+1
   332)                   
   333)                   write(fmtstr,'(i3)') ncols
   334)                   fmtstr = '('// trim(fmtstr)  // '(2x,' // trim(pfmtstr(ip)) // '))'
   335)                   write(unit=*,fmt=trim(fmtstr)) (pvar_g(i),i=col,col+ncols-1)
   336)                end do
   337)                write (unit=*,fmt=*) ''
   338)                write (unit=*,fmt=*) ''
   339)             end if
   340)          end if
   341)       end do
   342) 
   343)       !----- Check to see if we matched the variable. -------------------------------------!
   344)       if (.not.pvartrue) then
   345)          write (unit=*,fmt='(a)')      '--------------------------------------------------'
   346)          write (unit=*,fmt='(a,1x,a)') ' SOUGHT VARIABLE : ',trim(pvar_name)
   347)          write (unit=*,fmt='(a)')      '  '
   348)          write (unit=*,fmt='(a)')      ' The diagnostic variable doesn''t match any of the'
   349)          write (unit=*,fmt='(a)')      ' var_table variables.  Check your namelist '
   350)          write (unit=*,fmt='(a)')      ' entries, and the variable registry and/or remove'
   351)          write (unit=*,fmt='(a)')      ' this diagnostic variable.'
   352)          call fatal_error('Bad variable name.','print_fields','ed_print.f90')
   353)       end if
   354)    end do count_pvars
   355)    !---------------------------------------------------------------------------------------!
   356) 
   357) 
   358) 
   359)    !----- Don't proceed until everything is written out. ----------------------------------!
   360) #if defined(RAMS_MPI)
   361)    if (nnodetot /= 1) call MPI_Barrier(MPI_COMM_WORLD,ierr)
   362) #endif
   363)    !---------------------------------------------------------------------------------------!
   364) 
   365)    return
   366) end subroutine print_fields

   381)    implicit none
   382)    !----- Arguments. ----------------------------------------------------------------------!

   384)    real, dimension(npts_in) , intent(in)    :: vt_ptr
   385)    integer                  , intent(in)    :: npts_in
   386)    integer                  , intent(in)    :: npts_out
