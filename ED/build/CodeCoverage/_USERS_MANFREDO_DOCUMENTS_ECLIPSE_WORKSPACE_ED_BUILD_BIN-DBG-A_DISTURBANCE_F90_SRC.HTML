    31)    !     This is the main disturbance driver.  It will be called every New Year day, and   !
    32)    ! it will decide whether a new patch should be created.  Disturbances can be natural    !
    33)    ! and anthropogenic, and the new patch will be always assigned an identification flag   !
    34)    ! that will tell how that patch was created.  Three categories are currently possible:  !
    35)    ! 1 - agriculture: conversion to agriculture by land clearing;                          !
    36)    ! 2 - secondary forest: logging, land abandonment, and harvest create this patch;       !
    37)    ! 3 - primary forest: natural disturbances (treefall or fire).                          !
    38)    !---------------------------------------------------------------------------------------!
    39)    subroutine apply_disturbances(cgrid)
    40)       use forestry
    41)       use update_derived_props_module
    42)       use ed_state_vars, only : edtype                    & ! structure
    43)                               , polygontype               & ! structure
    44)                               , sitetype                  & ! structure
    45)                               , patchtype                 ! ! structure
    46)       use ed_misc_coms , only : current_time              & ! intent(in)
    47)                               , ibigleaf                  ! ! intent(in)
    48)       use disturb_coms , only : min_patch_area            & ! intent(in)
    49)                               , mature_harvest_age        & ! intent(in)
    50)                               , plantation_year           & ! intent(in)
    51)                               , plantation_rotation       & ! intent(in)
    52)                               , time2canopy               ! ! intent(in)
    53)       use ed_max_dims  , only : n_dist_types              & ! intent(in)
    54)                               , n_pft                     & ! intent(in)
    55)                               , n_dbh                     ! ! intent(in)
    56)       use mem_polygons , only : maxcohort                 ! ! intent(in)
    57)       use grid_coms    , only : nzg                       & ! intent(in)
    58)                               , nzs                       ! ! intent(in)
    59)       use pft_coms     , only : include_pft               ! ! intent(in)
    60)       use allometry    , only : area_indices              ! ! function
    61)       use mortality    , only : disturbance_mortality     ! ! subroutine
    62)       use consts_coms  , only : lnexp_max                 & ! intent(in)
    63)                               , tiny_num                  & ! intent(in)
    64)                               , huge_num                  ! ! intent(in)
    65)       use budget_utils , only : update_budget             ! ! sub-routine
    66)       implicit none
    67)       !----- Arguments. -------------------------------------------------------------------!
    68)       type(edtype)                    , target      :: cgrid
    69)       !----- Local variables. -------------------------------------------------------------!
    70)       type(polygontype)               , pointer     :: cpoly
    71)       type(sitetype)                  , pointer     :: csite
    72)       type(sitetype)                  , pointer     :: tsite
    73)       type(patchtype)                 , pointer     :: cpatch
    74)       type(patchtype)                 , pointer     :: qpatch
    75)       integer, dimension(n_pft)                     :: allind
    76)       integer                                       :: ipy
    77)       integer                                       :: isi
    78)       integer                                       :: ipa
    79)       integer                                       :: npa
    80)       integer                                       :: ico
    81)       integer                                       :: ipft
    82)       integer                                       :: i
    83)       integer                                       :: apa
    84)       integer                                       :: zpa
    85)       integer                                       :: mypfts
    86)       integer                                       :: onsp
    87)       integer                                       :: nnsp_ble
    88)       integer                                       :: old_lu
    89)       integer                                       :: new_lu
    90)       integer                                       :: dist_path
    91)       integer, dimension(:)           , allocatable :: pfts
    92)       logical, dimension(:)           , allocatable :: disturb_mask
    93)       real   , dimension(:)           , allocatable :: original_area
    94)       real   , dimension(:)           , allocatable :: original_lu
    95)       real   , dimension(:)           , allocatable :: harvestable_agb
    96)       real   , dimension(:)           , allocatable :: pot_area_harv
    97)       real   , dimension(:,:)         , allocatable :: pot_area_loss
    98)       real   , dimension(:,:)         , allocatable :: act_area_loss
    99)       real   , dimension(n_dist_types)              :: pot_area_gain
   100)       real   , dimension(n_dist_types)              :: act_area_gain
   101)       logical                                       :: biomass_harvest
   102)       logical                                       :: disturbed
   103)       logical                                       :: same_pft
   104)       logical                                       :: mature_plantation
   105)       logical                                       :: mature_primary
   106)       logical                                       :: mature_secondary
   107)       logical                                       :: mature_patch
   108)       real   , dimension(n_pft,n_dbh)               :: initial_agb
   109)       real   , dimension(n_pft,n_dbh)               :: initial_basal_area
   110)       real   , dimension(n_pft)                     :: mindbh_harvest
   111)       real                                          :: pot_area_remain
   112)       real                                          :: area_fac
   113)       real                                          :: orig_area
   114)       real                                          :: dist_area
   115)       real                                          :: dist_rate
   116)       real                                          :: elim_nplant
   117)       real                                          :: elim_lai
   118)       real                                          :: new_nplant
   119)       logical                                       :: is_plantation
   120)       !----- Debugging controls. ----------------------------------------------------------!
   121)       logical                         , parameter   :: print_debug = .false.
   122)       !------------------------------------------------------------------------------------!
   123) 
   124) 
   125) 
   126)       !------------------------------------------------------------------------------------!
   127)       !     Create a temporary vector with the allowed PFTs.  This will be later used      !
   128)       ! to define the new patches in case this is a big leaf simulation.                   !
   129)       !------------------------------------------------------------------------------------!
   130)       mypfts = count(include_pft)
   131)       allocate(pfts(mypfts))
   132)       allind = (/ (i, i=1, n_pft) /)
   133)       pfts   = pack(allind,include_pft)
   134)       !------------------------------------------------------------------------------------!
   135) 
   136) 
   137) 
   138)       !------------------------------------------------------------------------------------!
   139)       !      nnsp_ble stands for the new number of site patches for big leaf.  Each site   !
   140)       ! may have additional mypfts, except if land use is 1 (cropland/pasture) or 2        !
   141)       ! (forest plantation), in which case only one PFT is allowed.                        !
   142)       !------------------------------------------------------------------------------------!
   143)       nnsp_ble = 2 + (n_dist_types - 2) * mypfts
   144)       !------------------------------------------------------------------------------------!
   145) 
   146) 
   147) 
   148)       !----- Allocate the temporary site that will host the original patches. -------------!
   149)       nullify (tsite)
   150)       allocate(tsite)
   151)       !------------------------------------------------------------------------------------!
   152) 
   153) 
   154) 
   155)       !------------------------------------------------------------------------------------!
   156)       !     Loop over polygons and sites.                                                  !
   157)       !------------------------------------------------------------------------------------!
   158)       polyloop: do ipy = 1,cgrid%npolygons
   159)          cpoly => cgrid%polygon(ipy)
   160)          siteloop: do isi = 1,cpoly%nsites
   161) 
   162)             csite => cpoly%site(isi)
   163) 
   164)             !----- Save the Original Number (of) Site Patches, onsp... --------------------!
   165)             onsp = csite%npatches
   166)             !------------------------------------------------------------------------------!
   167) 
   168) 
   169) 
   170)             !----- Test whether the new explored patch will be plantation or logged. ------!
   171)             is_plantation = cpoly%plantation(isi) == 1               .and.                 &
   172)                             current_time%year     >  plantation_year
   173)             !------------------------------------------------------------------------------!
   174) 
   175) 
   176) 
   177)             !------------------------------------------------------------------------------!
   178)             !     Allocate the area losses and gains.  We define potential and actual      !
   179)             ! so we can correct patches that would become too small or new patches that    !
   180)             ! would be too small to be generated.                                          !
   181)             !------------------------------------------------------------------------------!
   182)             allocate (original_area  (onsp)             )
   183)             allocate (original_lu    (onsp)             )
   184)             allocate (harvestable_agb(onsp)             )
   185)             allocate (pot_area_harv  (onsp)             )
   186)             allocate (pot_area_loss  (onsp,n_dist_types))
   187)             allocate (act_area_loss  (onsp,n_dist_types))
   188)             original_area  (:  ) = 0.0
   189)             original_lu    (:  ) = 0.0
   190)             harvestable_agb(:  ) = 0.0
   191)             pot_area_harv  (:  ) = 0.0
   192)             pot_area_harv  (:  ) = 0.0
   193)             pot_area_loss  (:,:) = 0.0
   194)             act_area_loss  (:,:) = 0.0
   195)             pot_area_gain  (  :) = 0.0
   196)             act_area_gain  (  :) = 0.0
   197)             !------------------------------------------------------------------------------!
   198) 
   199) 
   200) 
   201) 
   202)             !------------------------------------------------------------------------------!
   203)             !     Save the original area, it will be used to correct the transition matrix !
   204)             ! at the end of the sub-routine.                                               !
   205)             !------------------------------------------------------------------------------!
   206)             save_area: do ipa=1,onsp
   207)                original_area(ipa) = csite%area     (ipa)
   208)                original_lu  (ipa) = csite%dist_type(ipa)
   209)             end do save_area
   210)             !------------------------------------------------------------------------------!
   211) 
   212) 
   213)             !----- Store AGB, basal area profiles in memory. ------------------------------!
   214)             call update_site_derived_props(cpoly, 1,isi)
   215)             initial_agb       (1:n_pft,1:n_dbh) = cpoly%agb       (1:n_pft,1:n_dbh,isi)
   216)             initial_basal_area(1:n_pft,1:n_dbh) = cpoly%basal_area(1:n_pft,1:n_dbh,isi)
   217)             !------------------------------------------------------------------------------!
   218) 
   219) 
   220) 
   221)             !------------------------------------------------------------------------------!
   222)             !      Find the area to be harvested when biomass targets have been            !
   223)             ! established.                                                                 !
   224)             !------------------------------------------------------------------------------!
   225)             call find_harvest_area(cpoly,isi,onsp,harvestable_agb,pot_area_harv)
   226)             !------------------------------------------------------------------------------!
   227) 
   228) 
   229) 
   230)             !------------------------------------------------------------------------------!
   231)             !     Prepare the current site to receive the newly disturbed patches.  This   !
   232)             ! step is done differently depending on whether this is a big leaf or a SAS    !
   233)             ! simulation.                                                                  !
   234)             !------------------------------------------------------------------------------!
   235)             select case (ibigleaf)
   236)             case (0)
   237)                !---------------------------------------------------------------------------!
   238)                !     Size-and-age structure.                                               !
   239)                !---------------------------------------------------------------------------!
   240) 
   241)                !---------------------------------------------------------------------------!
   242)                !     Transfer the original patch values to a temporary patch.  Then re-    !
   243)                ! -allocate csite with room for new patches, and transfer the original      !
   244)                ! patches back.                                                             !
   245)                !---------------------------------------------------------------------------!
   246)                call allocate_sitetype  (tsite,onsp)
   247)                call copy_sitetype      (csite,tsite,1,onsp,1,onsp)
   248)                call deallocate_sitetype(csite)
   249)                call allocate_sitetype  (csite,onsp + n_dist_types)
   250)                call copy_sitetype      (tsite,csite,1,onsp,1,onsp)
   251)                call deallocate_sitetype(tsite)
   252)                !---------------------------------------------------------------------------!
   253) 
   254) 
   255)                !----- Allocate and initialise a disturbance mask vector. ------------------!
   256)                allocate(disturb_mask(onsp + n_dist_types))
   257)                disturb_mask(:)      = .false.
   258)                disturb_mask(1:onsp) = .true.
   259)                !---------------------------------------------------------------------------!
   260) 
   261) 
   262)                !---------------------------------------------------------------------------!
   263)                !      Initialize all the potential as well as implemented disturbance      !
   264)                ! patches.  n_dist_types new patches will be created, each one containing a !
   265)                ! different patch type.  In case no conversion to that kind of patch has    !
   266)                ! happened, or if the newly created patch is tiny, it will be removed soon. !
   267)                !---------------------------------------------------------------------------!
   268)                init_dist_sas: do new_lu = onsp+1, onsp+n_dist_types
   269)                   call initialize_disturbed_patch(csite,cpoly%met(isi)%atm_tmp,new_lu      &
   270)                                                  ,cpoly%lsl(isi))
   271)                end do init_dist_sas
   272)                !---------------------------------------------------------------------------!
   273) 
   274)             case (1)
   275)                !---------------------------------------------------------------------------!
   276)                !     Big-leaf ED.                                                          !
   277)                !---------------------------------------------------------------------------!
   278) 
   279) 
   280)                !---------------------------------------------------------------------------!
   281)                !     Transfer the original patch values to a temporary patch.  Then re-    !
   282)                ! -allocate csite with room for new PFT/patches, and transfer the original  !
   283)                ! patches back.                                                             !
   284)                !---------------------------------------------------------------------------!
   285)                call allocate_sitetype  (tsite,onsp)
   286)                call copy_sitetype      (csite,tsite,1,onsp,1,onsp)
   287)                call deallocate_sitetype(csite)
   288)                call allocate_sitetype  (csite,onsp + nnsp_ble)
   289)                call copy_sitetype      (tsite,csite,1,onsp,1,onsp)
   290)                call deallocate_sitetype(tsite)
   291)                !---------------------------------------------------------------------------!
   292) 
   293) 
   294) 
   295)                !---------------------------------------------------------------------------!
   296)                !     Allocate the mask to decide which patches will remain by the end.     !
   297)                !---------------------------------------------------------------------------!
   298)                allocate(disturb_mask(onsp + nnsp_ble))
   299)                disturb_mask         = .false.
   300)                disturb_mask(1:onsp) = .true.
   301)                !---------------------------------------------------------------------------!
   302) 
   303) 
   304)                !---------------------------------------------------------------------------!
   305)                !      Initialize all the potential as well as implemented disturbance      !
   306)                ! patches.  n_dist_types new patches will be created, each one containing a !
   307)                ! different patch type.  In case no conversion to that kind of patch has    !
   308)                ! happened, or if the newly created patch is tiny, it will be removed soon. !
   309)                !---------------------------------------------------------------------------!
   310)                init_distpatch_ble: do ipa = onsp+1, onsp+nnsp_ble
   311)                   call initialize_disturbed_patch(csite,cpoly%met(isi)%atm_tmp,ipa         &
   312)                                                  ,cpoly%lsl(isi))
   313)                end do init_distpatch_ble
   314)                !---------------------------------------------------------------------------!
   315)             end select
   316)             !------------------------------------------------------------------------------!
   317) 
   318) 
   319) 
   320) 
   321)             !------------------------------------------------------------------------------!
   322)             !       Reset mortality rates due to disturbance.                              !
   323)             !------------------------------------------------------------------------------!
   324)             resetdist: do ipa=1,onsp
   325)                cpatch => csite%patch(ipa)
   326)                do ico=1,cpatch%ncohorts
   327)                   cpatch%mort_rate(5,ico) = 0.0
   328)                end do
   329)             end do resetdist
   330)             !------------------------------------------------------------------------------!
   331) 
   332) 
   333) 
   334)             !------------------------------------------------------------------------------!
   335)             !      Loop over new_lu, the new land use type, and find the contribution      !
   336)             ! of all old patches to these potential new patches.                           !
   337)             !------------------------------------------------------------------------------!
   338)             new_lu_l1st: do new_lu = 1, n_dist_types
   339) 
   340)                !---------------------------------------------------------------------------!
   341)                !     Loop over the old patches, find the potential area that could go      !
   342)                ! to the new patch.                                                         !
   343)                !---------------------------------------------------------------------------!
   344)                old_lu_l1st: do ipa=1,onsp
   345) 
   346)                   !----- Save the old land use in a shorter variable for convenience. -----!
   347)                   old_lu        = csite%dist_type(ipa)
   348)                   !------------------------------------------------------------------------!
   349) 
   350) 
   351) 
   352)                   !----- Check whether the patch is ready  be harvested. ------------------!
   353)                   mature_primary    = old_lu == 3                                 .and.    &
   354)                                       csite%age(ipa) > mature_harvest_age
   355)                   mature_plantation = old_lu == 2                                 .and.    &
   356)                                       csite%age(ipa) > plantation_rotation
   357)                   mature_secondary  = old_lu >= 4 .and. old_lu <= 6               .and.    &
   358)                                       csite%age(ipa) > mature_harvest_age
   359)                   mature_patch      = mature_primary .or. mature_secondary .or.            &
   360)                                       mature_plantation
   361)                   !------------------------------------------------------------------------!
   362) 
   363) 
   364) 
   365)                   !------------------------------------------------------------------------!
   366)                   !    Check whether to apply the disturbance transition from the old      !
   367)                   ! patch to the current land use.                                         !
   368)                   !                                                                        !
   369)                   ! * ploughed  - conversion from primary/secondary land to agriculture.   !
   370)                   ! * planted   - conversion to forest plantation.                         !
   371)                   ! * natural   - natural disturbance from primary/secondary land to       !
   372)                   !               primary land due to tree fall                            !
   373)                   ! * burnt     - conversion from primary/secondary land to secondary      !
   374)                   !               land due to fire                                         !
   375)                   ! * abandoned - conversion from agriculture to secondary land.           !
   376)                   ! * logged    - conversion from primary/secondary land to secondary      !
   377)                   !               land due to logging.                                     !
   378)                   !------------------------------------------------------------------------!
   379)                   !----- Check whether this patch can be disturbed. -----------------------!
   380)                   select case (new_lu)
   381)                   case (1)
   382)                      !----- Conversion to agriculture (cropland/pasture). -----------------!
   383)                      biomass_harvest = .false.
   384)                      disturbed       = old_lu /= 1
   385)                      !---------------------------------------------------------------------!
   386)                   case (2)
   387)                      !----- Establishment or rotation of forest plantation. ---------------!
   388)                      biomass_harvest = pot_area_harv(ipa) > 0. .and. is_plantation
   389)                      disturbed       = ( mature_patch .and. is_plantation ) .or.           &
   390)                                        ( old_lu == 2 .and.  csite%age(ipa) > time2canopy )
   391)                      disturbed       = disturbed .or. biomass_harvest
   392)                      !---------------------------------------------------------------------!
   393)                   case (3)
   394)                      !----- Tree fall. ----------------------------------------------------!
   395)                      biomass_harvest = .false.
   396)                      disturbed       = old_lu /= 1 .and. old_lu /= 2 .and.                 &
   397)                                        csite%age(ipa) > time2canopy
   398)                      !---------------------------------------------------------------------!
   399)                   case (4)
   400)                      !----- Fire disturbance. ---------------------------------------------!
   401)                      biomass_harvest = .false.
   402)                      disturbed       = old_lu /= 1 .and. old_lu /= 2
   403)                      !---------------------------------------------------------------------!
   404)                   case (5)
   405)                      !----- Abandonment. --------------------------------------------------!
   406)                      biomass_harvest = .false.
   407)                      disturbed       = old_lu == 1 .or.  old_lu == 2
   408)                      !---------------------------------------------------------------------!
   409)                   case (6)
   410)                      !----- Logging. ------------------------------------------------------!
   411)                      biomass_harvest = pot_area_harv(ipa) > 0.                  .and.      &
   412)                                        ( .not. is_plantation )
   413)                      disturbed       = ( mature_primary .or. mature_secondary ) .and.      &
   414)                                        ( .not. is_plantation )
   415)                      disturbed       = disturbed .or. biomass_harvest
   416)                      !---------------------------------------------------------------------!
   417)                   end select
   418)                   !------------------------------------------------------------------------!
   419) 
   420) 
   421) 
   422)                   !------------------------------------------------------------------------!
   423)                   !    Accumulate the lost area to the new patch.                          !
   424)                   !------------------------------------------------------------------------!
   425)                   if  (disturbed .and. biomass_harvest) then
   426)                      !---------------------------------------------------------------------!
   427)                      !     Patch has been harvest using biomass target.  Use the area from !
   428)                      ! pot_area_harv.                                                      !
   429)                      !---------------------------------------------------------------------!
   430)                      pot_area_loss(ipa,new_lu) = pot_area_harv(ipa)
   431)                      !---------------------------------------------------------------------!
   432)                   else if (disturbed) then
   433)                      !---------------------------------------------------------------------!
   434)                      !      Disturbance rate is the total for this time plus any value     !
   435)                      ! that was not previously applied.                                    !
   436)                      !---------------------------------------------------------------------!
   437)                      dist_rate = cpoly%disturbance_rates (new_lu,old_lu,isi)               &
   438)                                + cpoly%disturbance_memory(new_lu,old_lu,isi)
   439)                      dist_rate = min(lnexp_max,max(0.,dist_rate))
   440)                      !---------------------------------------------------------------------!
   441) 
   442)                      !----- Save the potentially lost area. -------------------------------!
   443)                      pot_area_loss(ipa,new_lu) = csite%area(ipa) * (1.0 - exp(-dist_rate))
   444)                      !---------------------------------------------------------------------!
   445)                   end if
   446)                   !------------------------------------------------------------------------!
   447)                end do old_lu_l1st
   448)                !---------------------------------------------------------------------------!
   449)             end do new_lu_l1st
   450)             !------------------------------------------------------------------------------!
   451) 
   452) 
   453) 
   454) 
   455)             !------------------------------------------------------------------------------!
   456)             !     Loop over the new LU patches, and check whether any of them would        !
   457)             ! have enough area to become a patch.                                          !
   458)             !------------------------------------------------------------------------------!
   459)             new_lu_l2nd: do new_lu=1,n_dist_types
   460)                pot_area_gain(new_lu) = sum(pot_area_loss(:,new_lu))
   461)             end do new_lu_l2nd
   462)             !------------------------------------------------------------------------------!
   463) 
   464) 
   465) 
   466) 
   467)             !------------------------------------------------------------------------------!
   468)             !     Loop over the old patches, and check whether any of them would become    !
   469)             ! too small.  In case so, then we must adjust the area.                        !
   470)             !------------------------------------------------------------------------------!
   471)             old_lu_l2nd: do ipa=1,onsp
   472)                !----- Only the LU transitions with enough area are actually created. ------!
   473)                where (pot_area_gain(:) >= min_patch_area)
   474)                   act_area_loss(ipa,:) = pot_area_loss(ipa,:)
   475)                end where
   476)                !---------------------------------------------------------------------------!
   477) 
   478) 
   479)                !---------------------------------------------------------------------------!
   480)                !    In case the area would become smaller than the minimum area, the patch !
   481)                ! must be eliminated.  The commented section below adjusts the disturbance  !
   482)                ! rate to completely eliminate the patch.                                   !
   483)                !---------------------------------------------------------------------------!
   484)                ! pot_area_remain = csite%area(ipa) - sum(act_area_loss(ipa,:))
   485)                ! if ( sum(act_area_loss(ipa,:)) > tiny_num       .and.                     &
   486)                !      pot_area_remain           < min_patch_area ) then
   487)                !    area_fac             = csite%area(ipa) / sum(act_area_loss(ipa,:))
   488)                !    act_area_loss(ipa,:) = act_area_loss(ipa,:) * area_fac
   489)                !
   490)                !    !----------------------------------------------------------------------!
   491)                !    !     Set disturb_mask to false, so this patch is purged from csite    !
   492)                !    ! later in this sub-routine.                                           !
   493)                !    !----------------------------------------------------------------------!
   494)                !    disturb_mask (ipa)   = .false.
   495)                !    !----------------------------------------------------------------------!
   496)                ! end if
   497)                !---------------------------------------------------------------------------!
   498) 
   499) 
   500)                !---------------------------------------------------------------------------!
   501)                !    In case the area would become smaller than the minimum area,  the      !
   502)                ! patch must be eliminated. Set disturb_mask to false, so this patch is     !
   503)                ! purged from csite later in this sub-routine.                              !
   504)                !---------------------------------------------------------------------------!
   505)                pot_area_remain = csite%area(ipa) - sum(act_area_loss(ipa,:))
   506)                if ( pot_area_remain < min_patch_area ) then
   507)                   disturb_mask (ipa)   = .false.
   508)                end if
   509)                !---------------------------------------------------------------------------!
   510)             end do old_lu_l2nd
   511)             !------------------------------------------------------------------------------!
   512) 
   513) 
   514)             !------------------------------------------------------------------------------!
   515)             !     Another loop over the new patches.  This time we define the patches      !
   516)             ! that will be generated.                                                      !
   517)             !------------------------------------------------------------------------------!
   518)             new_lu_l3rd: do new_lu=1,n_dist_types
   519)                !----- Retrieve the area gain for this new land use type. ------------------!
   520)                act_area_gain(new_lu) = sum(act_area_loss(:,new_lu))
   521)                !---------------------------------------------------------------------------!
   522) 
   523) 
   524)                !---------------------------------------------------------------------------!
   525)                !      Populate the new patch in case there is sufficient disturbed area.   !
   526)                !---------------------------------------------------------------------------!
   527)                if (act_area_gain(new_lu) >= min_patch_area) then
   528)                   write(unit=*,fmt='(a,1x,es12.5,1x,a,1x,i5)')                             &
   529)                       ' ---> Making new patch, with area=',act_area_gain(new_lu)           &
   530)                                              ,' for dist_type=',new_lu
   531) 
   532)                   !------------------------------------------------------------------------!
   533)                   !    Fix flags for new patches.  Here it matters whether this simulation !
   534)                   ! is big-leaf or SAS.                                                    !
   535)                   !------------------------------------------------------------------------!
   536)                   select case (ibigleaf)
   537)                   case (0)
   538)                      !---------------------------------------------------------------------!
   539)                      !     Size-and-age structure.                                         !
   540)                      !---------------------------------------------------------------------!
   541) 
   542) 
   543)                      !---------------------------------------------------------------------!
   544)                      !     Set the flag that this patch should be kept as a newly created  !
   545)                      ! transition patch.                                                   !
   546)                      !---------------------------------------------------------------------!
   547)                      disturb_mask    (onsp+new_lu) = .true.
   548)                      csite%dist_type (onsp+new_lu) = new_lu
   549)                      csite%area      (onsp+new_lu) = act_area_gain(new_lu)
   550)                      !---------------------------------------------------------------------!
   551) 
   552) 
   553)                      !---------------------------------------------------------------------!
   554)                      !     Initialize to zero the new trasitioned patches.                 !
   555)                      !---------------------------------------------------------------------!
   556)                      call initialize_disturbed_patch(csite,cpoly%met(isi)%atm_tmp          &
   557)                                                     ,onsp+new_lu,cpoly%lsl(isi))
   558)                      !---------------------------------------------------------------------!
   559) 
   560)                   case (1)
   561)                      !---------------------------------------------------------------------!
   562)                      !     Big leaf.                                                       !
   563)                      !---------------------------------------------------------------------!
   564)                      !---------------------------------------------------------------------!
   565)                      !     Set the flag that this patch should be kept as a newly created  !
   566)                      ! transition patch.                                                   !
   567)                      !---------------------------------------------------------------------!
   568)                      select case (new_lu)
   569)                      case (1,2)
   570)                         !------------------------------------------------------------------!
   571)                         !     Cropland, pasture, and forest plantation.  Only one PFT is   !
   572)                         ! allowed in such patches.                                         !
   573)                         !------------------------------------------------------------------!
   574) 
   575) 
   576)                         !------------------------------------------------------------------!
   577)                         !     Set the flag that this patch should be kept as a newly       !
   578)                         ! created transition patch.                                        !
   579)                         !------------------------------------------------------------------!
   580)                         disturb_mask    (onsp+new_lu)  = .true.
   581)                         csite%dist_type (onsp+new_lu)  = new_lu
   582)                         csite%area      (onsp+new_lu)  = act_area_gain(new_lu)
   583)                         !------------------------------------------------------------------!
   584) 
   585) 
   586) 
   587)                         !----- Initialize to zero the new trasitioned patches. ------------!
   588)                         call initialize_disturbed_patch(csite,cpoly%met(isi)%atm_tmp       &
   589)                                                        ,onsp+new_lu,cpoly%lsl(isi))
   590)                         !------------------------------------------------------------------!
   591) 
   592)                      case default
   593)                         !------------------------------------------------------------------!
   594)                         !     Non-cultivated lands.  Create one patch for each PFT that    !
   595)                         ! may be present.                                                  !
   596)                         !------------------------------------------------------------------!
   597)                         pft_add_loop: do ipft=1,mypfts
   598)                            !---------------------------------------------------------------!
   599)                            !     Set the flag that this patch should be kept as a newly    !
   600)                            ! created transition patch.                                     !
   601)                            !---------------------------------------------------------------!
   602)                            npa                   = onsp + 2 + (new_lu - 3) * mypfts + ipft
   603)                            disturb_mask    (npa) = .true.
   604)                            csite%dist_type (npa) = new_lu
   605)                            csite%area      (npa) = act_area_gain(new_lu) / real(mypfts)
   606)                            !---------------------------------------------------------------!
   607) 
   608) 
   609)                            !----- Initialize to zero the new trasitioned patches. ---------!
   610)                            call initialize_disturbed_patch(csite,cpoly%met(isi)%atm_tmp    &
   611)                                                           ,npa,cpoly%lsl(isi))
   612)                            !---------------------------------------------------------------!
   613)                         end do pft_add_loop
   614)                         !------------------------------------------------------------------!
   615)                      end select
   616)                      !---------------------------------------------------------------------!
   617)                   end select
   618)                   !------------------------------------------------------------------------!
   619) 
   620) 
   621) 
   622)                   !------------------------------------------------------------------------!
   623)                   !    Now go through patches, adding its contribution to the new          !
   624)                   ! patch.                                                                 !
   625)                   !------------------------------------------------------------------------!
   626)                   old_lu_l3rd: do ipa=1,onsp
   627)                      !----- Point to the current patch. -----------------------------------!
   628)                      cpatch => csite%patch(ipa)
   629)                      !---------------------------------------------------------------------!
   630) 
   631) 
   632)                      !---------------------------------------------------------------------!
   633)                      !     Save the old land use in a shorter variable for convenience.    !
   634)                      !---------------------------------------------------------------------!
   635)                      old_lu        = csite%dist_type(ipa)
   636)                      !---------------------------------------------------------------------!
   637) 
   638) 
   639) 
   640)                      !----- Check whether the patch is ready  be harvested. ---------------!
   641)                      mature_primary    = old_lu == 3                              .and.    &
   642)                                          csite%age(ipa) > mature_harvest_age
   643)                      mature_plantation = old_lu == 2                              .and.    &
   644)                                          csite%age(ipa) > plantation_rotation
   645)                      mature_secondary  = old_lu >= 4 .and. old_lu <= 6            .and.    &
   646)                                          csite%age(ipa) > mature_harvest_age
   647)                      biomass_harvest   = pot_area_harv(ipa) > 0.
   648)                      !---------------------------------------------------------------------!
   649) 
   650) 
   651) 
   652)                      !---------------------------------------------------------------------!
   653)                      !    Check whether to apply disturbance to this patch, and assign     !
   654)                      ! default values for disturbance pathway and minimum DBH for          !
   655)                      ! harvesting.                                                         !
   656)                      !---------------------------------------------------------------------!
   657)                      disturbed               = act_area_loss(ipa,new_lu) > tiny_num
   658)                      dist_path               = 0
   659)                      mindbh_harvest(1:n_pft) = huge(1.)
   660)                      !---------------------------------------------------------------------!
   661) 
   662) 
   663) 
   664) 
   665)                      !---------------------------------------------------------------------!
   666)                      !     Some disturbance types may contain multiple disturbance         !
   667)                      ! pathways.  Besides, logging may have different harvest              !
   668)                      ! requirements.                                                       !
   669)                      !---------------------------------------------------------------------!
   670)                      select case (new_lu)
   671)                      case (2)
   672)                         if (mature_plantation .or. biomass_harvest .or. old_lu /= 2) then
   673)                            !----- Time to harvest. ----------------------------------------!
   674)                            dist_path = 20
   675)                            !---------------------------------------------------------------!
   676)                         else
   677)                            !----- Losses due to tree fall. --------------------------------!
   678)                            dist_path = 21
   679)                            !---------------------------------------------------------------!
   680)                         end if
   681)                         !------------------------------------------------------------------!
   682) 
   683)                      case (5)
   684)                         !----- Abandonment. -----------------------------------------------!
   685)                         select case (old_lu)
   686)                         case (1)
   687)                            !----- Abandoned cropland/pasture. -----------------------------!
   688)                            dist_path = 50
   689)                            !---------------------------------------------------------------!
   690)                         case (2)
   691)                            !----- Abandoned plantation following fire. --------------------!
   692)                            dist_path = 51
   693)                            !---------------------------------------------------------------!
   694)                         end select
   695)                         !------------------------------------------------------------------!
   696)                      case (6)
   697)                         !----- Logging. ---------------------------------------------------!
   698)                         if (mature_primary) then
   699)                            mindbh_harvest(1:n_pft) = cpoly%mindbh_primary(1:n_pft,isi)
   700)                         else if (mature_secondary) then
   701)                            mindbh_harvest(1:n_pft) = cpoly%mindbh_secondary(1:n_pft,isi)
   702)                         end if
   703)                         !------------------------------------------------------------------!
   704)                      end select
   705)                      !---------------------------------------------------------------------!
   706) 
   707) 
   708) 
   709)                      !---------------------------------------------------------------------!
   710)                      !    If the patch is going to be disturbed, add the area and          !
   711)                      ! survivors to the new patch.  The disturbed area has been already    !
   712)                      ! deducted (check loop old_lu_l2nd), no need to change the old        !
   713)                      ! patch area here.                                                    !
   714)                      !---------------------------------------------------------------------!
   715)                      if (disturbed) then
   716)                         !------------------------------------------------------------------!
   717)                         !     Find the actual disturbance rate associated with this        !
   718)                         ! transition.                                                      !
   719)                         !------------------------------------------------------------------!
   720)                         if ( act_area_loss(ipa,new_lu) < csite%area(ipa) ) then
   721)                            dist_rate = log( csite%area(ipa)                                &
   722)                                           / (csite%area(ipa)-act_area_loss(ipa,new_lu)) )
   723)                         else
   724)                            dist_rate = lnexp_max
   725)                         end if
   726)                         !----- Find the mortality associated with disturbance. ------------!
   727)                         call disturbance_mortality(csite,ipa,dist_rate,new_lu              &
   728)                                                   ,dist_path,mindbh_harvest)
   729)                         !------------------------------------------------------------------!
   730) 
   731) 
   732) 
   733)                         !------------------------------------------------------------------!
   734)                         !     Add area and survivors to the new patches.  Here again we    !
   735)                         ! must check whether this is a conventional or a  big leaf         !
   736)                         ! simulation.                                                      !
   737)                         !------------------------------------------------------------------!
   738)                         select case (ibigleaf)
   739)                         case (0)
   740)                            !---------------------------------------------------------------!
   741)                            !     Size-and-age structure.                                   !
   742)                            !---------------------------------------------------------------!
   743)                            area_fac = act_area_loss(ipa,new_lu) / csite%area(onsp+new_lu)
   744)                            call increment_patch_vars(csite,onsp+new_lu,ipa,area_fac)
   745)                            call insert_survivors(csite,onsp+new_lu,ipa,new_lu,area_fac     &
   746)                                                 ,dist_path,mindbh_harvest)
   747)                            call accum_dist_litt(csite,onsp+new_lu,ipa,new_lu,area_fac      &
   748)                                                ,dist_path,mindbh_harvest)
   749)                            !---------------------------------------------------------------!
   750)                         case (1)
   751)                            !---------------------------------------------------------------!
   752)                            !     Big leaf ED.  Here we must also check the type of patch   !
   753)                            ! we are creating.                                              !
   754)                            !---------------------------------------------------------------!
   755)                            select case (new_lu)
   756)                            case (1,2)
   757)                               !------------------------------------------------------------!
   758)                               !     Cropland, pasture, or forest plantation.               !
   759)                               !------------------------------------------------------------!
   760)                               area_fac = act_area_loss(ipa,new_lu)/csite%area(onsp+new_lu)
   761)                               call increment_patch_vars(csite,onsp+new_lu,ipa,area_fac)
   762)                               call insert_survivors(csite,onsp+new_lu,ipa,new_lu,area_fac  &
   763)                                                    ,dist_path,mindbh_harvest)
   764)                               call accum_dist_litt(csite,onsp+new_lu,ipa,new_lu,area_fac   &
   765)                                                   ,dist_path,mindbh_harvest)
   766)                               !------------------------------------------------------------!
   767)                            case default
   768)                               !------------------------------------------------------------!
   769)                               !     Non-cultivated lands.  We must check that the donor    !
   770)                               ! patch and receptor patch have the same PFT, or the donor   !
   771)                               ! is empty.                                                  !
   772)                               !------------------------------------------------------------!
   773)                               do ipft=1,mypfts
   774)                                  npa = onsp + 2 + (new_lu - 3) * mypfts + ipft
   775) 
   776)                                  if (cpatch%ncohorts == 0) then
   777)                                     !------------------------------------------------------!
   778)                                     !    Donor cohort is empty, split disturbed area       !
   779)                                     ! evenly amongst receptor cohorts.                     !
   780)                                     !------------------------------------------------------!
   781)                                     same_pft = ipft == 1
   782)                                     area_fac = act_area_loss(ipa,new_lu)                   &
   783)                                              / ( real(mypfts) * csite%area(npa) )
   784)                                     !------------------------------------------------------!
   785)                                  else
   786)                                     if (ipft == cpatch%pft(1)) then
   787)                                        !---------------------------------------------------!
   788)                                        !    PFTs match, send all disturbed area to this    !
   789)                                        ! patch.                                            !
   790)                                        !---------------------------------------------------!
   791)                                        same_pft = .true.
   792)                                        area_fac = act_area_loss(ipa,new_lu)/csite%area(npa)
   793)                                        !---------------------------------------------------!
   794)                                     else
   795)                                        !---------------------------------------------------!
   796)                                        !    PFTs do not match, don't send anything to the  !
   797)                                        ! new patch.                                        !
   798)                                        !---------------------------------------------------!
   799)                                        same_pft = .false.
   800)                                        area_fac = 0.
   801)                                        !---------------------------------------------------!
   802)                                     end if
   803)                                     !------------------------------------------------------!
   804)                                  end if
   805)                                  !---------------------------------------------------------!
   806) 
   807) 
   808)                                  !---------------------------------------------------------!
   809)                                  !     Accumulate survivors to the new patch.              !
   810)                                  !---------------------------------------------------------!
   811)                                  if (same_pft) then
   812)                                     call increment_patch_vars(csite,npa,ipa,area_fac)
   813)                                     call insert_survivors(csite,npa,ipa,new_lu,area_fac    &
   814)                                                          ,dist_path,mindbh_harvest)
   815)                                     call accum_dist_litt(csite,npa,ipa,new_lu,area_fac     &
   816)                                                         ,dist_path,mindbh_harvest)
   817)                                  end if
   818)                                  !---------------------------------------------------------!
   819)                               end do
   820)                               !------------------------------------------------------------!
   821)                            end select
   822)                            !---------------------------------------------------------------!
   823)                         end select
   824)                         !------------------------------------------------------------------!
   825)                      end if
   826)                      !---------------------------------------------------------------------!
   827)                   end do old_lu_l3rd
   828)                   !------------------------------------------------------------------------!
   829) 
   830) 
   831) 
   832)                   !------------------------------------------------------------------------!
   833)                   !      Update temperature and density.  This must be done before         !
   834)                   ! planting, since the leaf temperature is initially assigned as the      !
   835)                   ! canopy air temperature.                                                !
   836)                   !------------------------------------------------------------------------!
   837)                   call update_patch_thermo_props(csite,onsp+new_lu,onsp+new_lu,nzg,nzs     &
   838)                                                 ,cpoly%ntext_soil(:,isi))
   839)                   call update_patch_thermo_fmean(csite,onsp+new_lu,onsp+new_lu,nzg         &
   840)                                                 ,cpoly%ntext_soil(:,isi))
   841)                   !------------------------------------------------------------------------!
   842) 
   843) 
   844) 
   845) 
   846)                   !------------------------------------------------------------------------!
   847)                   !     In case the new patch is agriculture or a forest plantation, add   !
   848)                   ! the new grasses/trees.                                                 !
   849)                   !------------------------------------------------------------------------!
   850)                   select case (new_lu)
   851)                   case (1)
   852)                      call plant_patch(csite,onsp+new_lu,nzg                                &
   853)                                      ,cpoly%agri_stocking_pft(isi)                         &
   854)                                      ,cpoly%agri_stocking_density(isi)                     &
   855)                                      ,cpoly%ntext_soil(:,isi), 1.0                         &
   856)                                      ,cpoly%lsl(isi))
   857)                   case (2)
   858)                      call plant_patch(csite,onsp+new_lu,nzg                                &
   859)                                      ,cpoly%plantation_stocking_pft(isi)                   &
   860)                                      ,cpoly%plantation_stocking_density(isi)               &
   861)                                      ,cpoly%ntext_soil(:,isi),2.0                          &
   862)                                      ,cpoly%lsl(isi))
   863)                   end select
   864)                   !------------------------------------------------------------------------!
   865) 
   866) 
   867) 
   868)                   !------------------------------------------------------------------------!
   869)                   !    Update the cohorts.  This step is done only in case this is a size- !
   870)                   ! -and-age structure.                                                    !
   871)                   !------------------------------------------------------------------------!
   872)                   qpatch => csite%patch(onsp+new_lu)
   873)                   if (ibigleaf == 0 .and. qpatch%ncohorts > 0 .and. maxcohort >= 0) then
   874)                      call fuse_cohorts(csite,onsp+new_lu                                   &
   875)                                       ,cpoly%lsl(isi),.false.)
   876)                      call terminate_cohorts(csite,onsp+new_lu,elim_nplant,elim_lai)
   877)                      call split_cohorts(qpatch, cpoly%green_leaf_factor(:,isi))
   878)                   end if
   879)                   !------------------------------------------------------------------------!
   880) 
   881) 
   882) 
   883) 
   884)                   !----- Store AGB, basal area profiles in memory. ------------------------!
   885)                   initial_agb(1:n_pft,1:n_dbh) = cpoly%agb(1:n_pft, 1:n_dbh,isi)
   886)                   initial_basal_area(1:n_pft,1:n_dbh) =                                    &
   887)                                                    cpoly%basal_area(1:n_pft,1:n_dbh,isi)
   888)                   !------------------------------------------------------------------------!
   889) 
   890) 
   891) 
   892) 
   893)                   !------------------------------------------------------------------------!
   894)                   !     Update the derived properties including veg_height, and patch-     !
   895)                   ! -level LAI, WAI.                                                       !
   896)                   !------------------------------------------------------------------------!
   897)                   call update_patch_derived_props(csite,onsp+new_lu)
   898)                   !----- Update soil temperature, liquid fraction, etc. -------------------!
   899)                   call new_patch_sfc_props(csite,onsp+new_lu,nzg,nzs                       &
   900)                                           ,cpoly%ntext_soil(:,isi))
   901)                   !----- Update budget properties. ----------------------------------------!
   902)                   call update_budget(csite,cpoly%lsl(isi),onsp+new_lu)
   903)                   !----- Update AGB, basal area. ------------------------------------------!
   904)                   call update_site_derived_props(cpoly,1,isi)
   905)                   !----- Update either cut or mortality. ----------------------------------!
   906)                   select case (new_lu)
   907)                   case (1,2,6)
   908)                      cpoly%agb_cut(1:n_pft,1:n_dbh,isi) =                                  &
   909)                             cpoly%agb_cut(1:n_pft, 1:n_dbh,isi)                            &
   910)                           + initial_agb(1:n_pft, 1:n_dbh)                                  &
   911)                           - cpoly%agb(1:n_pft, 1:n_dbh,isi)
   912)                      cpoly%basal_area_cut(1:n_pft, 1:n_dbh,isi) =                          &
   913)                             cpoly%basal_area_cut(1:n_pft, 1:n_dbh,isi)                     &
   914)                           + initial_basal_area(1:n_pft, 1:n_dbh)                           &
   915)                           - cpoly%basal_area(1:n_pft, 1:n_dbh,isi)
   916)                   case default
   917)                      cpoly%agb_mort(1:n_pft,1:n_dbh,isi) =                                 &
   918)                             cpoly%agb_mort(1:n_pft,1:n_dbh,isi)                            &
   919)                           + initial_agb(1:n_pft,1:n_dbh)                                   &
   920)                           - cpoly%agb(1:n_pft,1:n_dbh,isi)
   921)                      cpoly%basal_area_mort(1:n_pft, 1:n_dbh,isi) =                         &
   922)                             cpoly%basal_area_mort(1:n_pft, 1:n_dbh,isi)                    &
   923)                           + initial_basal_area(1:n_pft, 1:n_dbh)                           &
   924)                           - cpoly%basal_area(1:n_pft, 1:n_dbh,isi)
   925)                   end select
   926)                   !------------------------------------------------------------------------!
   927) 
   928) 
   929)                   !----- Clear the disturbance memory for this disturbance type. ----------!
   930)                   cpoly%disturbance_memory(new_lu,1:n_dist_types,isi) = 0.0
   931)                   !------------------------------------------------------------------------!
   932) 
   933)                elseif (pot_area_gain(new_lu) > 0.0)then
   934)                   !------------------------------------------------------------------------!
   935)                   !     The patch creation has been skipped because the area was too       !
   936)                   ! small.  Put the current disturbance rates in memory to be added at     !
   937)                   ! the next timestep.                                                     !
   938)                   !------------------------------------------------------------------------!
   939)                   cpoly%disturbance_memory(new_lu,1:n_dist_types,isi) =                    &
   940)                          cpoly%disturbance_memory(new_lu,1:n_dist_types,isi)               &
   941)                        + cpoly%disturbance_rates (new_lu,1:n_dist_types,isi)
   942)                   !------------------------------------------------------------------------!
   943)                end if
   944)                !---------------------------------------------------------------------------!
   945)             end do new_lu_l3rd
   946)             !------------------------------------------------------------------------------!
   947) 
   948) 
   949)             !------------------------------------------------------------------------------!
   950)             !     Print the entire disturbance transition matrix if debugging mode is on.  !
   951)             !------------------------------------------------------------------------------!
   952)             if (print_debug) then
   953)                write(unit=*,fmt='(a)')          ' '
   954)                write(unit=*,fmt='(85a)')        ('-',i=1,85)
   955)                write(unit=*,fmt='(2(a,1x,i5))') ' Summary for IPY =',ipy,'; ISI =',isi
   956)                write(unit=*,fmt='(a)')          ' '
   957)                write(unit=*,fmt='(10(1x,a))')   '  IPA','  ILU'                            &
   958)                                               ,'    AREA','  TO_AGR','  TO_FPL','  TO_TFL' &
   959)                                               ,'  TO_BRN','  TO_ABN','  TO_LOG','TOT_LOSS'
   960)                write(unit=*,fmt='(85a)') ('-',i=1,85)
   961)                do ipa=1,onsp
   962)                   write(unit=*,fmt='(2(1x,i5),8(1x,f8.5))')                                &
   963)                                                   ipa,csite%dist_type(ipa),csite%area(ipa) &
   964)                                                    ,(act_area_loss(ipa,new_lu),new_lu=1,6) &
   965)                                                    ,sum(act_area_loss(ipa,:))
   966)                end do
   967)                write(unit=*,fmt='(85a)') ('-',i=1,85)
   968)                write(unit=*,fmt='(1x,a,8(1x,f8.5))')                '            TOT_GAIN' &
   969)                                                        ,(act_area_gain(new_lu),new_lu=1,6) &
   970)                                                        ,sum(act_area_gain(:))
   971)                write(unit=*,fmt='(85a)') ('-',i=1,85)
   972)                write(unit=*,fmt='(a)')    ' '
   973)             end if
   974)             !------------------------------------------------------------------------------!
   975) 
   976) 
   977)             !------------------------------------------------------------------------------!
   978)             !      Deduct the distubed area from this patch.                               !
   979)             !------------------------------------------------------------------------------!
   980)             old_lu_l4th: do ipa=1,onsp
   981)                csite%area(ipa) = csite%area(ipa) - sum(act_area_loss(ipa,:))
   982)             end do old_lu_l4th
   983)             !------------------------------------------------------------------------------!
   984) 
   985)             if (include_pft(17)) then
   986)             prune_loop: do new_lu=1,n_dist_types
   987)             !----------------------- Prune the lianas -------------------------------------!
   988)                call prune_lianas(csite, onsp + new_lu, cpoly%lsl(isi))
   989)             !------------------------------------------------------------------------------!
   990)             end do prune_loop
   991)             end if
   992) 
   993) 
   994)             !------------------------------------------------------------------------------!
   995)             !     Big-leaf only.  In BigLeaf-ED, we do not track age since last            !
   996)             ! disturbance; instead, each land use type has one patch for each PFT or one   !
   997)             ! patch and one PFT only in case the patch is cultivated.  Here, we search for !
   998)             ! duplicated patches that have the same PFT and same LU, and fuse them.        !
   999)             !------------------------------------------------------------------------------!
  1000)             select case (ibigleaf)
  1001)             case (1)
  1002)                merge_lu_loop: do new_lu = 1, n_dist_types
  1003)                   !----- Define search limits for this LU. --------------------------------!
  1004)                   select case (new_lu)
  1005)                   case (1)
  1006)                      apa = onsp + 1
  1007)                      zpa = onsp + 1
  1008)                   case (2)
  1009)                      apa = onsp + 2
  1010)                      zpa = onsp + 2
  1011)                   case default
  1012)                      apa = onsp + 2 + ( new_lu - 3 ) * mypfts + 1
  1013)                      zpa = onsp + 2 + ( new_lu - 3 ) * mypfts + mypfts
  1014)                   end select
  1015)                   !------------------------------------------------------------------------!
  1016) 
  1017) 
  1018) 
  1019) 
  1020)                   !------------------------------------------------------------------------!
  1021)                   !     Loop over the old patches and seek land use matches.               !
  1022)                   !------------------------------------------------------------------------!
  1023)                   merge_patch_loop: do npa=apa,zpa
  1024) 
  1025)                      !----- Check whether the patch has been created. ---------------------!
  1026)                      if (disturb_mask(npa)) then
  1027) 
  1028)                         !------------------------------------------------------------------!
  1029)                         !     Loop over old patches.                                       !
  1030)                         !------------------------------------------------------------------!
  1031)                         old_lu_search: do ipa = 1,onsp
  1032)                            old_lu = csite%dist_type(ipa)
  1033) 
  1034)                            !---------------------------------------------------------------!
  1035)                            !    If this is a good receptor, merge the patches.             !
  1036)                            !                                                               !
  1037)                            !    Requirements for a good receptor:                          !
  1038)                            ! 1. Same land use                                              !
  1039)                            ! 2. Receptor is not slated to be purged.                       !
  1040)                            ! 3. Receptor has the same PFT as the donor (or both patches    !
  1041)                            !    are empty).                                                !
  1042)                            !---------------------------------------------------------------!
  1043)                            if     ( csite%patch(ipa)%ncohorts == 0 .and.                   &
  1044)                                     csite%patch(npa)%ncohorts == 0 ) then
  1045)                               same_pft = .true.
  1046)                            elseif ( csite%patch(ipa)%ncohorts == 0 .or.                    &
  1047)                                     csite%patch(npa)%ncohorts == 0 ) then
  1048)                               same_pft = .false.
  1049)                            else
  1050)                               same_pft = csite%patch(ipa)%pft(1) == csite%patch(npa)%pft(1)
  1051)                            end if
  1052) 
  1053)                            if ( old_lu == new_lu .and. disturb_mask(ipa) .and. same_pft)   &
  1054)                            then
  1055)                               !----- Fuse both patches. -----------------------------------!
  1056)                               call fuse_2_patches(csite,npa,ipa,nzg,nzs                    &
  1057)                                                  ,cpoly%lsl(isi)                           &
  1058)                                                  ,cpoly%ntext_soil(:,isi)                  &
  1059)                                                  ,cpoly%green_leaf_factor(:,isi)           &
  1060)                                                  ,.false.,elim_nplant,elim_lai)
  1061)                               !------------------------------------------------------------!
  1062) 
  1063) 
  1064) 
  1065)                               !------------------------------------------------------------!
  1066)                               !    Make sure that all cohorts are fused.                   !
  1067)                               !------------------------------------------------------------!
  1068)                               cpatch => csite%patch(ipa)
  1069)                               do ico=2,cpatch%ncohorts
  1070)                                  new_nplant = cpatch%nplant(ico) + cpatch%nplant(1)
  1071)                                  ipft       = cpatch%pft(1)
  1072)                                  call fuse_2_cohorts(cpatch,ico,1, csite%can_prss(ipa)     &
  1073)                                                     ,csite%can_shv (ipa),cpoly%lsl(isi)    &
  1074)                                                     ,.false.)
  1075) 
  1076)                                  !---------------------------------------------------------!
  1077)                                  !     Set nplant to a tiny number, we will delete this    !
  1078)                                  ! cohort soon.                                            !
  1079)                                  !---------------------------------------------------------!
  1080)                                  cpatch%nplant(ico) = tiny_num
  1081)                                  !---------------------------------------------------------!
  1082)                               end do
  1083)                               !------------------------------------------------------------!
  1084) 
  1085) 
  1086)                               !------ Remove emptied cohorts. -----------------------------!
  1087)                               call terminate_cohorts(csite,ipa,elim_nplant,elim_lai)
  1088)                               !------------------------------------------------------------!
  1089) 
  1090) 
  1091)                               !------ Remove the patch. -----------------------------------!
  1092)                               disturb_mask(npa) = .false.
  1093)                               !------------------------------------------------------------!
  1094) 
  1095) 
  1096)                               !------ Patch is gone, stop searching old patches. ----------!
  1097)                               exit old_lu_search
  1098)                               !------------------------------------------------------------!
  1099)                            end if
  1100)                            !---------------------------------------------------------------!
  1101)                         end do old_lu_search
  1102)                         !------------------------------------------------------------------!
  1103)                      end if
  1104)                      !---------------------------------------------------------------------!
  1105)                   end do merge_patch_loop
  1106)                   !------------------------------------------------------------------------!
  1107)                end do merge_lu_loop
  1108)                !---------------------------------------------------------------------------!
  1109)             end select
  1110)             !------------------------------------------------------------------------------!
  1111) 
  1112) 
  1113) 
  1114) 
  1115)             !------------------------------------------------------------------------------!
  1116)             !      Reallocate the current site to fit the original patches, except         !
  1117)             ! those that lost all area through disturbance, and the new disturbed          !
  1118)             ! patches that were actually generated.                                        !
  1119)             !------------------------------------------------------------------------------!
  1120)             !----- Temporary site, with size equal to the patches that will remain. -------!
  1121)             call allocate_sitetype(tsite,count(disturb_mask))
  1122)             call copy_sitetype_mask(csite,tsite,disturb_mask,size(disturb_mask)            &
  1123)                                    ,count(disturb_mask))
  1124)             call deallocate_sitetype(csite)
  1125)             !----- Re-allocate the tracked site and copy data from temporary site. --------!
  1126)             call allocate_sitetype(csite,tsite%npatches)
  1127)             call copy_sitetype(tsite,csite,1,tsite%npatches,1,tsite%npatches)
  1128)             call deallocate_sitetype(tsite)
  1129)             !------------------------------------------------------------------------------!
  1130) 
  1131) 
  1132)             !------------------------------------------------------------------------------!
  1133)             !      Update the disturbance rates to be consistent with the applied          !
  1134)             ! transitions.  This is necessary because some transitions are not calculated  !
  1135)             ! directly from disturbance_rates (for example, biomass-based harvesting), and !
  1136)             ! some exceptions may occur (for example, when previous memory has been        !
  1137)             ! applied, or when there wasn't enough area to create a new patch or when      !
  1138)             ! disturbance was extended to the entire patch because the remaining area      !
  1139)             ! would be too small otherwise.                                                !
  1140)             !------------------------------------------------------------------------------!
  1141)             !----- Print the area loss and disturbance transitions. -----------------------!
  1142)             if (print_debug) then
  1143)                write(unit=*,fmt='(a)')          ' '
  1144)                write(unit=*,fmt='(45a)')        ('-',i=1,45)
  1145)                write(unit=*,fmt='(2(a,1x,i5))') ' Transitions for IPY =',ipy,'; ISI =',isi
  1146)                write(unit=*,fmt='(a)')          ' '
  1147)                write(unit=*,fmt='(5(1x,a))')    'OLD_LU','NEW_LU'                          &
  1148)                                                ,'ORIG_AREA','AREA_LOSS','DIST_RATE'
  1149)                write(unit=*,fmt='(45a)')        ('-',i=1,45)
  1150)             end if
  1151)             !------------------------------------------------------------------------------!
  1152) 
  1153)             dist_from_loop: do old_lu=1,n_dist_types
  1154)                dist_to_loop: do new_lu=1,n_dist_types
  1155) 
  1156)                   !----- Reset areas for this transtion (old_lu -> new_lu). ---------------!
  1157)                   orig_area = 0.0
  1158)                   dist_area = 0.0
  1159)                   dist_orig_loop: do ipa = 1, onsp
  1160)                      if (original_lu(ipa) == old_lu .and. act_area_loss(ipa,new_lu) > 0.)  &
  1161)                      then
  1162)                         orig_area = orig_area + original_area(ipa)
  1163)                         dist_area = dist_area + act_area_loss(ipa,new_lu)
  1164)                      end if
  1165)                   end do dist_orig_loop
  1166)                   !------------------------------------------------------------------------!
  1167) 
  1168) 
  1169)                   !----- Update the disturbance rate. -------------------------------------!
  1170)                   if (orig_area == 0.0) then
  1171)                      cpoly%disturbance_rates(new_lu,old_lu,isi) = 0.
  1172)                   elseif (dist_area <  orig_area ) then
  1173)                      cpoly%disturbance_rates(new_lu,old_lu,isi) =                          &
  1174)                                                 log( orig_area / ( orig_area - dist_area ))
  1175)                   else
  1176)                      cpoly%disturbance_rates(new_lu,old_lu,isi) = lnexp_max
  1177)                   end if
  1178)                   !------------------------------------------------------------------------!
  1179) 
  1180) 
  1181) 
  1182)                   !----- Print the area loss and disturbance transitions. -----------------!
  1183)                   if (print_debug) then
  1184)                      write(unit=*,fmt='(2(1x,i6),3(1x,f9.6))')                             &
  1185)                                                  old_lu,new_lu,orig_area,dist_area         &
  1186)                                                 ,cpoly%disturbance_rates(new_lu,old_lu,isi)
  1187)                   end if
  1188)                   !------------------------------------------------------------------------!
  1189)                end do dist_to_loop
  1190)                !---------------------------------------------------------------------------!
  1191)             end do dist_from_loop
  1192)             !------------------------------------------------------------------------------!
  1193) 
  1194) 
  1195) 
  1196) 
  1197)             !----- Print the area loss and disturbance transitions. -----------------------!
  1198)             if (print_debug) then
  1199)                write(unit=*,fmt='(45a)')        ('-',i=1,45)
  1200)                write(unit=*,fmt='(a)')          ' '
  1201)             end if
  1202)             !------------------------------------------------------------------------------!
  1203) 
  1204) 
  1205) 
  1206)             !----- Free memory before re-allocating for the next site... ------------------!
  1207)             deallocate(disturb_mask   )
  1208)             deallocate(original_area  )
  1209)             deallocate(original_lu    )
  1210)             deallocate(harvestable_agb)
  1211)             deallocate(pot_area_harv  )
  1212)             deallocate(pot_area_loss  )
  1213)             deallocate(act_area_loss  )
  1214)             !------------------------------------------------------------------------------!
  1215)          end do siteloop
  1216)          !---------------------------------------------------------------------------------!
  1217)       end do polyloop
  1218)       !------------------------------------------------------------------------------------!
  1219) 
  1220)       !----- Free memory before leaving... ------------------------------------------------!
  1221)       deallocate(tsite)
  1222)       deallocate(pfts )
  1223)       !------------------------------------------------------------------------------------!
  1224) 
  1225)       return
  1226)    end subroutine apply_disturbances

  1237)    !     This sub-routine fills in the disturbance transition matrix.                      !
  1238)    !---------------------------------------------------------------------------------------!
  1239)    subroutine site_disturbance_rates(year, cgrid)
  1240) 
  1241)       use ed_state_vars, only : edtype                    & ! structure
  1242)                               , polygontype               & ! structure
  1243)                               , sitetype                  & ! structure
  1244)                               , patchtype                 ! ! structure
  1245)       use disturb_coms , only : treefall_disturbance_rate & ! intent(in)
  1246)                               , lutime                    & ! structure
  1247)                               , ianth_disturb             & ! intent(in)
  1248)                               , include_fire              & ! intent(in)
  1249)                               , plantation_year           & ! intent(in)
  1250)                               , plantation_rotation       & ! intent(in)
  1251)                               , mature_harvest_age        ! ! intent(in)
  1252)       use ed_max_dims  , only : n_pft                     & ! intent(in)
  1253)                               , n_dist_types              ! ! intent(in)
  1254)       use ed_misc_coms , only : current_time              ! ! intent(in)
  1255) 
  1256)       implicit none
  1257)       !----- Arguments. -------------------------------------------------------------------!
  1258)       type(edtype)     , target                  :: cgrid
  1259)       integer          , intent(in)              :: year
  1260)       !----- Local variables. -------------------------------------------------------------!
  1261)       type(polygontype), pointer                 :: cpoly
  1262)       type(sitetype)   , pointer                 :: csite
  1263)       type(patchtype)  , pointer                 :: cpatch
  1264)       type(lutime)     , pointer                 :: clutime
  1265)       integer                                    :: ipy
  1266)       integer                                    :: isi
  1267)       integer                                    :: ipa
  1268)       integer                                    :: ico
  1269)       integer                                    :: ipft
  1270)       integer                                    :: ilu
  1271)       integer                                    :: iyear
  1272)       integer                                    :: useyear
  1273)       real                                       :: weight
  1274)       real(kind=4)     , dimension(n_dist_types) :: sumweight
  1275)       real(kind=4)     , dimension(n_dist_types) :: pharvest
  1276)       real                                       :: fire_disturbance_rate
  1277)       logical                                    :: is_plantation
  1278)       !----- Local constants. -------------------------------------------------------------!
  1279)       real             , parameter               :: max_pharvest = 1.-2.*epsilon(1.)
  1280)       !------------------------------------------------------------------------------------!
  1281) 
  1282) 
  1283)       !------------------------------------------------------------------------------------!
  1284)       !      Loop over sites and polygons.                                                 !
  1285)       !------------------------------------------------------------------------------------!
  1286)       polyloop: do ipy = 1,cgrid%npolygons
  1287)          cpoly => cgrid%polygon(ipy)
  1288) 
  1289)          siteloop: do isi = 1,cpoly%nsites
  1290)             csite => cpoly%site(isi)
  1291) 
  1292)             !----- Test whether the new explored patch will be plantation or logged. ------!
  1293)             is_plantation = cpoly%plantation(isi) == 1               .and.                 &
  1294)                             current_time%year     >  plantation_year
  1295)             !------------------------------------------------------------------------------!
  1296) 
  1297) 
  1298) 
  1299) 
  1300)             !------------------------------------------------------------------------------!
  1301)             !      Reset the disturbance rates.  We only populate the transitions that are !
  1302)             ! may be positive.  The first index is the new land use type, and the second   !
  1303)             ! index is the old land use type.  Both use the following convention:          !
  1304)             !                                                                              !
  1305)             !  1.  Agricultural lands (cropland / pasture)                                 !
  1306)             !  2.  Forest plantation                                                       !
  1307)             !  3.  Tree fall                                                               !
  1308)             !  4.  Burnt                                                                   !
  1309)             !  5.  Abandoned                                                               !
  1310)             !  6.  Logged                                                                  !
  1311)             !------------------------------------------------------------------------------!
  1312)             cpoly%disturbance_rates(:,:,isi) = 0.0
  1313)             !------------------------------------------------------------------------------!
  1314) 
  1315) 
  1316) 
  1317) 
  1318)             !------------------------------------------------------------------------------!
  1319)             !    Calculate fire disturbance rates only when fire is on.                    !
  1320)             !------------------------------------------------------------------------------!
  1321)             select case (include_fire)
  1322)             case (0)
  1323)                fire_disturbance_rate = 0.0
  1324)             case default
  1325)                fire_disturbance_rate = sum(cpoly%lambda_fire(1:12,isi)) / 12.0
  1326)             end select
  1327)             !------------------------------------------------------------------------------!
  1328) 
  1329) 
  1330) 
  1331)             !------------------------------------------------------------------------------!
  1332)             !      Tree fall and fires also occur in plantations.  Treefall allows the     !
  1333)             ! plantation to continue, but fire leads to abandonment.                       !
  1334)             !------------------------------------------------------------------------------!
  1335)             cpoly%disturbance_rates(2,2,isi) = treefall_disturbance_rate
  1336)             cpoly%disturbance_rates(5,2,isi) = fire_disturbance_rate
  1337)             !------------------------------------------------------------------------------!
  1338) 
  1339) 
  1340) 
  1341)             !------------------------------------------------------------------------------!
  1342)             !      Disturbance that creates new "tree fall patches".  Only non-cultivated  !
  1343)             ! lands may suffer this disturbance.                                           !
  1344)             !------------------------------------------------------------------------------!
  1345)             cpoly%disturbance_rates(3,3,isi) = treefall_disturbance_rate
  1346)             cpoly%disturbance_rates(3,4,isi) = treefall_disturbance_rate
  1347)             cpoly%disturbance_rates(3,5,isi) = treefall_disturbance_rate
  1348)             cpoly%disturbance_rates(3,6,isi) = treefall_disturbance_rate
  1349)             !------------------------------------------------------------------------------!
  1350) 
  1351) 
  1352) 
  1353)             !------------------------------------------------------------------------------!
  1354)             !      Disturbance that creates new "burnt patches".  Only non-cultivated      !
  1355)             ! lands may suffer this disturbance.                                           !
  1356)             !------------------------------------------------------------------------------!
  1357)             cpoly%disturbance_rates(4,3,isi) = fire_disturbance_rate
  1358)             cpoly%disturbance_rates(4,4,isi) = fire_disturbance_rate
  1359)             cpoly%disturbance_rates(4,5,isi) = fire_disturbance_rate
  1360)             cpoly%disturbance_rates(4,6,isi) = fire_disturbance_rate
  1361)             !------------------------------------------------------------------------------!
  1362) 
  1363) 
  1364) 
  1365)             !------------------------------------------------------------------------------!
  1366)             !      Disturbance that creates new "burnt patches".  Only non-cultivated      !
  1367)             ! lands may suffer this disturbance.                                           !
  1368)             !------------------------------------------------------------------------------!
  1369)             cpoly%disturbance_rates(4,3:6,isi) = fire_disturbance_rate
  1370)             !------------------------------------------------------------------------------!
  1371) 
  1372) 
  1373) 
  1374)             !------------------------------------------------------------------------------!
  1375)             !      Check for anthropogenic disturbance rates.  We no longer need  to check !
  1376)             ! the year because all years in this simulation are assigned prescribed        !
  1377)             ! disturbance rates (even if that means zero disturbance).                     !
  1378)             !------------------------------------------------------------------------------!
  1379)             useyear = cpoly%num_landuse_years(isi)
  1380)             !----- Loop over years. -------------------------------------------------------!
  1381)             find_lu_year: do iyear = 1,cpoly%num_landuse_years(isi)
  1382)                if (year == cpoly%clutimes(iyear,isi)%landuse_year) then
  1383)                   useyear = iyear
  1384)                   exit find_lu_year
  1385)                end if
  1386)             end do find_lu_year
  1387)             !------------------------------------------------------------------------------!
  1388) 
  1389) 
  1390) 
  1391)             !------------------------------------------------------------------------------!
  1392)             !    Link to the current land use change year.                                 !
  1393)             !------------------------------------------------------------------------------!
  1394)             clutime => cpoly%clutimes(useyear,isi)
  1395)             !------------------------------------------------------------------------------!
  1396) 
  1397) 
  1398)             !------------------------------------------------------------------------------!
  1399)             !      For the time being conversion between pasture and croplands are not     !
  1400)             ! included.                                                                    !
  1401)             !------------------------------------------------------------------------------!
  1402)             ! cpoly%disturbance_rates(1,1,isi) = clutime%landuse(1) + clutime%landuse(2)
  1403)             !------------------------------------------------------------------------------!
  1404) 
  1405) 
  1406) 
  1407)             !------------------------------------------------------------------------------!
  1408)             !      Secondary forest (plantation, logged, burnt, abandoned) to agriculture. !
  1409)             !------------------------------------------------------------------------------!
  1410)             cpoly%disturbance_rates(1,2,isi) = clutime%landuse(7) + clutime%landuse(9)
  1411)             cpoly%disturbance_rates(1,4,isi) = clutime%landuse(7) + clutime%landuse(9)
  1412)             cpoly%disturbance_rates(1,5,isi) = clutime%landuse(7) + clutime%landuse(9)
  1413)             cpoly%disturbance_rates(1,6,isi) = clutime%landuse(7) + clutime%landuse(9)
  1414)             !------------------------------------------------------------------------------!
  1415) 
  1416) 
  1417) 
  1418)             !----- Primary forest to agriculture (3 => 1). --------------------------------!
  1419)             cpoly%disturbance_rates(1,3,isi) = clutime%landuse(4) + clutime%landuse(5)
  1420)             !------------------------------------------------------------------------------!
  1421) 
  1422) 
  1423) 
  1424)             !------------------------------------------------------------------------------!
  1425)             !     Agriculture to abandoned (1 => 5).   Here it depends on whether          !
  1426)             ! to establish plantations or abandoned lands.                                 !
  1427)             !------------------------------------------------------------------------------!
  1428)             select case (cpoly%plantation(isi))
  1429)             case (0)
  1430)                !----- Abandoned lands only. -----------------------------------------------!
  1431)                cpoly%disturbance_rates(5,1,isi) = clutime%landuse(8) + clutime%landuse(10) &
  1432)                                                 + clutime%landuse(3) + clutime%landuse(6)
  1433)                !---------------------------------------------------------------------------!
  1434)             case (1)
  1435)                !----- "Secondary" sends area to plantation, "Primary" to abandoned. -------!
  1436)                cpoly%disturbance_rates(2,1,isi) = clutime%landuse(8) + clutime%landuse(10)
  1437)                cpoly%disturbance_rates(5,1,isi) = clutime%landuse(3) + clutime%landuse(6)
  1438)                !---------------------------------------------------------------------------!
  1439)             end select
  1440)             !------------------------------------------------------------------------------!
  1441) 
  1442) 
  1443) 
  1444)             !------------------------------------------------------------------------------!
  1445)             !     Harvesting (either plantation -> plantation or logging) when a biomass   !
  1446)             ! target does not exist (e.g. SimAmazonia)).  Convert the    !
  1447)             ! harvest probability of being cut given that the DBH exceeds the minimum DBH. !
  1448)             ! This is done only when anthropogenic disturbance is on and we are not seek-  !
  1449)             ! ing the biomass target, otherwise we set it to zero.                         !
  1450)             !------------------------------------------------------------------------------!
  1451)             if (ianth_disturb == 1) then
  1452)                if (clutime%landuse(12) <= 0) then
  1453) 
  1454)                   !----- Loop over all patches, and find the harvest probability. ---------!
  1455)                   sumweight(:) = 0.
  1456)                   pharvest (:) = 0.
  1457)                   patchloop: do ipa=1,csite%npatches
  1458)                      cpatch => csite%patch(ipa)
  1459) 
  1460)                      !---------------------------------------------------------------------!
  1461)                      !     Check the land use of this patch, based on the definition we    !
  1462)                      ! define whether it is mature or not.                                 !
  1463)                      !---------------------------------------------------------------------!
  1464)                      ilu = csite%dist_type(ipa)
  1465)                      select case (ilu)
  1466)                      case (2)
  1467)                         !------------------------------------------------------------------!
  1468)                         !      Plantation.  Check whether the plantation is mature.  If    !
  1469)                         ! so, all trees may be harvested.                                  !
  1470)                         !------------------------------------------------------------------!
  1471)                         if (csite%age(ipa) > plantation_rotation) then
  1472)                            cohortloop_02: do ico=1,cpatch%ncohorts
  1473)                               ipft      = cpatch%pft(ico)
  1474)                               weight    = cpatch%nplant(ico) * cpatch%basarea(ico)         &
  1475)                                         * csite%area(ipa)
  1476)                               pharvest (ilu) = pharvest(ilu)                               &
  1477)                                              + cpoly%probharv_secondary(ipft,isi) * weight
  1478)                               sumweight(ilu) = sumweight(ilu) + weight
  1479)                            end do cohortloop_02
  1480)                            !---------------------------------------------------------------!
  1481)                         end if
  1482)                         !------------------------------------------------------------------!
  1483)                      case (6)
  1484)                         !------------------------------------------------------------------!
  1485)                         !      Logging.  Check whether the managed forest is mature.  If   !
  1486)                         ! so, all trees with sufficiently large DBH may be harvested.      !
  1487)                         !------------------------------------------------------------------!
  1488)                         if (csite%age(ipa) > mature_harvest_age) then
  1489)                            cohortloop_06: do ico=1,cpatch%ncohorts
  1490)                               ipft = cpatch%pft(ico)
  1491)                               if (cpatch%dbh(ico) >= cpoly%mindbh_secondary(ipft,isi)) then
  1492)                                  weight         = cpatch%nplant(ico) * cpatch%basarea(ico) &
  1493)                                                 * csite%area(ipa)
  1494)                                  pharvest (ilu) = pharvest(ilu)                               &
  1495)                                                 + cpoly%probharv_secondary(ipft,isi)       &
  1496)                                                 * weight
  1497)                                  sumweight(ilu) = sumweight(ilu) + weight
  1498)                               end if
  1499)                            end do cohortloop_06
  1500)                            !---------------------------------------------------------------!
  1501)                         end if
  1502)                         !------------------------------------------------------------------!
  1503)                      end select
  1504)                      !---------------------------------------------------------------------!
  1505)                   end do patchloop
  1506)                   !------------------------------------------------------------------------!
  1507) 
  1508) 
  1509) 
  1510)                   !----- Normalise the probability, unless it's zero. ---------------------!
  1511)                   where (sumweight(:) > 0.)
  1512)                      pharvest(:) = min(max_pharvest,pharvest(:)/sumweight(:))
  1513)                   elsewhere
  1514)                      pharvest(:) = 0.
  1515)                   end where
  1516)                   !------------------------------------------------------------------------!
  1517) 
  1518) 
  1519) 
  1520)                   !----- Convert the probability into disturbance rate. -------------------!
  1521)                   cpoly%disturbance_rates(6,6,isi) = - log(1.0 - pharvest(6))
  1522)                   cpoly%disturbance_rates(2,2,isi) = max( treefall_disturbance_rate        &
  1523)                                                         , - log(1.0 - pharvest(2)) )
  1524)                   !------------------------------------------------------------------------!
  1525) 
  1526) 
  1527)                   !------------------------------------------------------------------------!
  1528)                   !     Logging based on tree size, set the biomass target to zero.        !
  1529)                   !------------------------------------------------------------------------!
  1530)                   cpoly%secondary_harvest_target(isi) = 0.0

  2703)    subroutine insert_survivors(csite,np,cp,new_lu,area_fac,dist_path,mindbh_harvest)
  2704)       use update_derived_props_module
  2705)       use ed_state_vars, only : sitetype     & ! structure
  2706)                               , patchtype    ! ! structure
  2707)       use ed_max_dims  , only : n_pft        ! ! intent(in)

  2709)

  2714)       integer                         , intent(in)  :: dist_path
  2715)       integer                         , intent(in)  :: np
  2716)       integer                         , intent(in)  :: cp
  2717)       real          , dimension(n_pft), intent(in)  :: mindbh_harvest

  2719)       !----- Local variables. -------------------------------------------------------------!
  2720)       type(patchtype)                 , pointer     :: cpatch
  2721)       type(patchtype)                 , pointer     :: npatch
  2722)       type(patchtype)                 , pointer     :: tpatch
  2723)       logical        , dimension(:)   , allocatable :: mask
  2724)       integer                                       :: ico

  2726)       integer                                       :: addco

  2729)       !------------------------------------------------------------------------------------!
  2730) 
  2731)       !------------------------------------------------------------------------------------!
  2732)       ! cpatch => Old patches we are looping                                               !
  2733)       ! npatch => New transition patch                                                     !
  2734)       ! tpatch => temporary patch                                                          !
  2735)       !------------------------------------------------------------------------------------!

  2737)       npatch => csite%patch(np)

  2742)       if (cpatch%ncohorts > 0) then

  2744)          mask(:) = .false.
  2745) 
  2746)          survivalloop: do ico = 1,cpatch%ncohorts
  2747)             survival_fac = survivorship(new_lu,dist_path,mindbh_harvest,cpatch,ico)        &
  2748)                          * area_fac
  2749)             n_survivors  = cpatch%nplant(ico) * survival_fac

  2752)             mask(ico) = n_survivors > 0.0

  2754)          end do survivalloop

  2761)

  2767)       ! routine.                                                                           !

  2769)       if (npatch%ncohorts > 0) then
  2770)          nco = npatch%ncohorts

  2775)          nco = 0
  2776)          call allocate_patchtype(tpatch,addco)

  2779) 
  2780) 
  2781)       cohortloop: do ico = 1,cpatch%ncohorts
  2782) 
  2783)          survival_fac = survivorship(new_lu,dist_path,mindbh_harvest,cpatch,ico)           &
  2784)                       * area_fac
  2785)          n_survivors  = cpatch%nplant(ico) * survival_fac

  2790)             call copy_patchtype(cpatch,tpatch,ico,ico,nco,nco)
  2791) 
  2792)             !------------------------------------------------------------------------------!
  2793)             !    Scale the total area based on the new population density and new area.    !
  2794)             !------------------------------------------------------------------------------!
  2795)             call update_cohort_extensive_props(tpatch,nco,nco,survival_fac)
  2796)             !------------------------------------------------------------------------------!
  2797) 
  2798)             !----- Make mortality rate due to disturbance zero to avoid double counting. --!
  2799)             tpatch%mort_rate(5,nco) = 0.0
  2800)             !------------------------------------------------------------------------------!
  2801)          end if
  2802)          !---------------------------------------------------------------------------------!
  2803)       end do cohortloop
  2804)       !------------------------------------------------------------------------------------!
  2805) 
  2806)       !----- Copy the temporary patch into the newpatch. ----------------------------------!
  2807)       call allocate_patchtype(npatch,tpatch%ncohorts)
  2808)       call copy_patchtype(tpatch,npatch,1,tpatch%ncohorts,1,tpatch%ncohorts)
  2809)       call deallocate_patchtype(tpatch)
  2810) 
  2811)       deallocate(tpatch)
  2812)       if (allocated(mask)) deallocate(mask)

  2814)

  2830)       use ed_state_vars, only : sitetype     & ! structure
  2831)                               , patchtype    & ! structure
  2832)                               , polygontype  ! ! structure
  2833)       use decomp_coms  , only : f_labile     ! ! intent(in)

  2836)                               , c2n_leaf     & ! intent(in)
  2837)                               , c2n_stem     & ! intent(in)
  2838)                               , l2n_stem     ! ! intent(in)
  2839)       use pft_coms     , only : agf_bs       ! ! intent(in)
  2840)       use mortality    , only : survivorship ! ! function
  2841) 
  2842)       implicit none
  2843)       !----- Arguments. -------------------------------------------------------------------!
  2844)       type(sitetype)                   , target     :: csite
  2845)       integer                          , intent(in) :: np
  2846)       integer                          , intent(in) :: cp
  2847)       real           , dimension(n_pft), intent(in) :: mindbh_harvest
  2848)       integer                          , intent(in) :: new_lu
  2849)       real                             , intent(in) :: area_fac
  2850)       integer                          , intent(in) :: dist_path
  2851)       !----- Local variables. -------------------------------------------------------------!
  2852)       type(patchtype)                  , pointer    :: cpatch
  2853)       type(patchtype)                  , pointer    :: npatch
  2854)       integer                                       :: ico
  2855)       integer                                       :: ipft

  2879)          ipft = cpatch%pft(ico)
  2880)

  2885)          !---------------------------------------------------------------------------------!
  2886)          select case(new_lu)

  2892)          !---------------------------------------------------------------------------------!

  2903)          fast_litter_n = fast_litter_n                                                     &
  2904)                        + (1. - survival_fac)                                               &
  2905)                        * ( f_labile(ipft) * cpatch%balive(ico) / c2n_leaf(ipft)            &

  2910)                        * (1. - survival_fac)                                               &
  2911)                        * ( (1. - loss_fraction ) * cpatch%bdead(ico)                       &

  2913) 
  2914)          struct_litter = struct_litter + struct_cohort
  2915)          struct_lignin = struct_lignin + struct_cohort * l2n_stem / c2n_stem(ipft)

  2917)       !------------------------------------------------------------------------------------!
  2918) 
  2919) 
  2920) 
  2921) 
  2922)       !----- Load disturbance litter directly into carbon and N pools. --------------------!
  2923)       csite%fast_soil_C(np)       = csite%fast_soil_C(np)       + fast_litter   * area_fac
  2924)       csite%structural_soil_C(np) = csite%structural_soil_C(np) + struct_litter * area_fac
  2925)       csite%structural_soil_L(np) = csite%structural_soil_L(np) + struct_lignin * area_fac
  2926)       csite%fast_soil_N(np)       = csite%fast_soil_N(np)       + fast_litter_n * area_fac
  2927)       !------------------------------------------------------------------------------------!
  2928) 
  2929)       return
