    42)    ! ISOILBC -- This controls the soil moisture boundary condition at the bottom.  Choose  !
    43)    !            the option according to the site characteristics.                          !
    44)    !            0.  Flat bedrock.  Flux from the bottom of the bottommost layer is zero.   !
    45)    !            1.  Gravitational flow (free drainage).  The flux from the bottom of the   !
    46)    !                bottommost layer is due to gradient of height only.                    !
    47)    !            2.  Lateral drainage.  Similar to free drainage, but the gradient is       !
    48)    !                reduced by the slope not being completely vertical.  The reduction is  !
    49)    !                controlled by variable SLDRAIN.  In the future options 0, 1, and 2 may !
    50)    !                be combined into a single option.                                      !
    51)    !            3.  Aquifer.  Soil moisture of the ficticious layer beneath the bottom is  !
    52)    !                always at saturation.                                                  !
    53)    !---------------------------------------------------------------------------------------!
    54)    integer                                    :: isoilbc

    62)    !                 the equivalent slope that will slow down drainage.  If this is set to !
    63)    !                 zero, then lateral drainage reduces to flat bedrock, and if this is   !
    64)    !                 set to 90, then lateral drainage becomes free drainage.  SLDRAIN must !
    65)    !                 be between 0 and 90.                                                  !

    67)    ! SLDRAIN8     -- These are auxiliary variables derived from sldrain.  SIN_SLDRAIN is   !

    69)    ! SIN_SLDRAIN8    precision versions of SLDRAIN and SIN_SLDRAIN.                        !

   587)    !      This sub-routines converts the soil moisture index into soil moisture.           !
   588)    !  This scale is piece-wise linear, and dependent on the soil texture.                  !
   589)    !  -1. = dry air soil moisture (soilcp)                                                 !
   590)    !   0. = wilting point         (soilwp)                                                 !
   591)    !   1. = field capacity        (sfldcap)                                                !
   592)    !   2. = porosity/saturation   (slmsts)                                                 !
   593)    !---------------------------------------------------------------------------------------!
   594)    real(kind=4) function ed_soil_idx2water(soil_index,ntext)
   595)       implicit none
   596)       !----- Arguments. -------------------------------------------------------------------!
   597)       real(kind=4), intent(in) :: soil_index
   598)       integer     , intent(in) :: ntext
   599)       !----- Local variables. -------------------------------------------------------------!
   600)       real(kind=4)             :: soil_water
   601)       !------------------------------------------------------------------------------------!
   602) 
   603) 
   604)       if (soil_index < 0.0) then
   605)          soil_water = soil(ntext)%soilcp                                                   &
   606)                     + (soil_index + 1.0) * (soil(ntext)%soilwp  - soil(ntext)%soilcp )
   607)       elseif (soil_index < 1.0) then
   608)          soil_water = soil(ntext)%soilwp                                                   &
   609)                     +  soil_index        * (soil(ntext)%sfldcap - soil(ntext)%soilwp )
   610)       else
   611)          soil_water = soil(ntext)%sfldcap                                                  &
   612)                     + (soil_index - 1.0) * (soil(ntext)%slmsts  - soil(ntext)%sfldcap)
   613)       end if
   614)
