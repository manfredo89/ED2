     3) ! SUBROUTINE: VEGETATION_DYNAMICS
     4) !
     5) !> \brief   Handles vegetation growth, phenology, reproduction, and disturbance.
     6) !> \details This subroutine first handles any prescribed events, then moves on to phenology,
     7) !>          biomass growth, structural growth, reproduction, then disturbance.
     8) !> \author  Translated from ED1 by Ryan Knox and Marcos Longo
     9) !------------------------------------------------------------------------------------------!
    10) subroutine vegetation_dynamics(new_month,new_year)
    11)    use structural_growth_module
    12)    use reproduction_module
    13)    use structural_growth_module
    14)    use soil_respiration_module
    15)    use phenology_driv
    16)    use update_derived_props_module
    17)    use grid_coms        , only : ngrids
    18)    use ed_misc_coms     , only : current_time               & ! intent(in)
    19)                                , dtlsm                      & ! intent(in)
    20)                                , ibigleaf                   ! ! intent(in)
    21)    use disturbance_utils, only : apply_disturbances         & ! subroutine
    22)                                , site_disturbance_rates     ! ! subroutine
    23)    use fuse_fiss_utils  , only : fuse_patches               & ! subroutine
    24)                                , terminate_patches          & ! subroutine
    25)                                , rescale_patches            ! ! subroutine
    26)    use ed_state_vars    , only : edgrid_g                   & ! intent(inout)

    28)                                , polygontype                & ! variable type!
    29)                                , sitetype                   & ! structure
    30)                                , patchtype                  ! ! structure!*
    31)    use growth_balive    , only : dbalive_dt                 ! ! subroutine

    33)                                , yr_day                     ! ! intent(in)
    34)    use mem_polygons     , only : maxpatch                   ! ! intent(in)
    35)    use average_utils    , only : normalize_ed_today_vars    & ! sub-routine
    36)                                , normalize_ed_todaynpp_vars & ! sub-routine

    39)    !----- Arguments. ----------------------------------------------------------------------!
    40)    logical          , intent(in)   :: new_month             !< First dtlsm of a new month?
    41)    logical          , intent(in)   :: new_year              !< First dtlsm of a new year?
    42)    !----- Local variables. ----------------------------------------------------------------!
    43)    type(edtype)     , pointer      :: cgrid
    44)    type(polygontype), pointer      :: cpoly
    45)    real                            :: dtlsm_o_day
    46)    real                            :: one_o_year

    50)    integer          , external     :: julday
    51)    !---------------------------------------------------------------------------------------!
    52)    integer                       :: ipy
    53)    integer                       :: isi
    54) 
    55)    !----- Find the day of year. -----------------------------------------------------------!
    56)    doy = julday(current_time%month, current_time%date, current_time%year)
    57)   
    58)    !----- Time factor for normalizing daily variables updated on the DTLSM step. ----------!
    59)    dtlsm_o_day = dtlsm / day_sec
    60)    !----- Time factor for averaging dailies. ----------------------------------------------!
    61)    one_o_year  = 1.0 / yr_day
    62) 
    63)    !----- Apply events. -------------------------------------------------------------------!
    64)    call prescribed_event(current_time%year,doy)
    65) 
    66)   
    67)    !---------------------------------------------------------------------------------------!
    68)    !   Loop over all domains.                                                              !
    69)    !---------------------------------------------------------------------------------------!
    70)    do ifm=1,ngrids
    71) 
    72)       cgrid => edgrid_g(ifm) 
    73) 
    74)       !------------------------------------------------------------------------------------!
    75)       !     The following block corresponds to the daily time-step.                        !
    76)       !------------------------------------------------------------------------------------!
    77)       !----- Standardise the fast-scale uptake and respiration, for growth rates. ---------!
    78)       call normalize_ed_today_vars(cgrid)
    79)       !----- Update phenology and growth of live tissues. ---------------------------------!
    80)       call phenology_driver(cgrid,doy,current_time%month, dtlsm_o_day)
    81)       call dbalive_dt(cgrid,one_o_year)
    82)       !------------------------------------------------------------------------------------!
    83) 
    84) 
    85)       !------------------------------------------------------------------------------------!
    86)       !     The following block corresponds to the monthly time-step:                      !
    87)       !------------------------------------------------------------------------------------!
    88)       if (new_month) then
    89) 
    90)          !----- Update the mean workload counter. -----------------------------------------!
    91)          call update_workload(cgrid)
    92) 
    93)          !----- Update the growth of the structural biomass. ------------------------------!
    94)          call structural_growth(cgrid, current_time%month)
    95) 
    96)          !----- Solve the reproduction rates. ---------------------------------------------!
    97)          call reproduction(cgrid,current_time%month)
    98) 
    99)          !----- Update the fire disturbance rates. ----------------------------------------!
   100)          call fire_frequency(cgrid)
   101) 
   102)          !----- This is actually the yearly time-step, apply the disturbances. ------------!
   103)          if (new_year) then
   104)             !----- Update the disturbance rates. ------------------------------------------!
   105)             call site_disturbance_rates(current_time%year, cgrid)
   106)             call apply_disturbances(cgrid)
   107)             !------------------------------------------------------------------------------!
   108)          end if
   109)          !---------------------------------------------------------------------------------!
   110)       end if
   111)       !------------------------------------------------------------------------------------!
   112) 
   113)       !------  update dmean and mmean values for NPP allocation terms ---------------------!
   114)       call normalize_ed_todayNPP_vars(cgrid)
   115)       
   116)       !------------------------------------------------------------------------------------!
   117)       !     This should be done every day, but after the longer-scale steps.  We update    !
   118)       ! the carbon and nitrogen pools, and re-set the daily variables.                     !
   119)       !------------------------------------------------------------------------------------!
   120)       call update_C_and_N_pools(cgrid)
   121)       call zero_ed_today_vars(cgrid)
   122)       !------------------------------------------------------------------------------------!
   123) 
   124) 
   125)       !------------------------------------------------------------------------------------!
   126)       !      Patch dynamics.                                                               !
   127)       !------------------------------------------------------------------------------------!
   128)       if (new_year) then
   129)          select case(ibigleaf)
   130)          case (0)
   131)             !------------------------------------------------------------------------------!
   132)             !    Size and age structure.  Fuse patches last, after all updates have been   !
   133)             ! applied.  This reduces the number of patch variables that actually need to   !
   134)             ! be fused.  After fusing, we also check whether there are patches that are    !
   135)             ! too small, and terminate them.                                               !
   136)             !------------------------------------------------------------------------------!
   137)             if (maxpatch >= 0) call fuse_patches(cgrid,ifm,.false.)
   138)             do ipy = 1,cgrid%npolygons
   139)                cpoly => cgrid%polygon(ipy)
   140)                  
   141)                do isi = 1, cpoly%nsites
   142)                   call terminate_patches(cpoly%site(isi))
   143)                end do
   144)             end do
   145)             !------------------------------------------------------------------------------!
   146) 
   147)          case (1)
   148)             !------------------------------------------------------------------------------!
   149)             !    Big leaf.  All that we do is rescale the patches.                         !
   150)             !------------------------------------------------------------------------------!
   151)             do ipy = 1,cgrid%npolygons
   152)                cpoly => cgrid%polygon(ipy)
   153)                  
   154)                do isi = 1, cpoly%nsites
   155)                   call rescale_patches(cpoly%site(isi))
   156)                end do
   157)             end do
   158)             !------------------------------------------------------------------------------!
   159)          end select
   160)          !---------------------------------------------------------------------------------!
   161)       end if
   162)       !------------------------------------------------------------------------------------!
   163) 
   164) 
   165) 
   166)       !------------------------------------------------------------------------------------!
   167)       !     Update polygon-level properties that are derived from patches and cohorts.     !
   168)       !------------------------------------------------------------------------------------!
   169)       call update_polygon_derived_props(edgrid_g(ifm))
   170)       !------------------------------------------------------------------------------------!
   171) 
   172) 
   173) 
   174)       !----- Print the carbon and nitrogen budget. ----------------------------------------!
   175)       call print_C_and_N_budgets(cgrid)
   176)       !------------------------------------------------------------------------------------!
   177) 
   178)    end do
   179) 
   180)    return
   181) end subroutine vegetation_dynamics
