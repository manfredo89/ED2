     4) !==========================================================================================!
     5) !==========================================================================================!
     6) !     This subroutine calculates the site-level variables based on the polygon-level ones. !
     7) ! If needed, it will also apply the lapse rate on some variables, and recalculate the      !
     8) ! others in order to keep satisfying the ideal gas law and some thermodynamic properties.  !
     9) !------------------------------------------------------------------------------------------!
    10) subroutine calc_met_lapse(cgrid,ipy)
    11)   
    12)    use ed_state_vars         , only : edtype      & ! structure
    13)                                     , polygontype ! ! structure
    14)    use consts_coms           , only : toodry,pi1      ! ! intent(in)
    15)    implicit none
    16)    !----- Arguments. ----------------------------------------------------------------------!
    17)    type(edtype)     , target     :: cgrid
    18)    integer          , intent(in) :: ipy
    19)    !----- Local variables -----------------------------------------------------------------!
    20)    type(polygontype), pointer    :: cpoly
    21)    integer                       :: isi
    22)    real                          :: ebar   !! mean elevation
    23)    real                          :: delE   !! deviation from mean elevation
    24)    real                          :: aterr  !! terrestrial area
    25)    real                          :: hillshade
    26)    !----- Local constants -----------------------------------------------------------------!
    27)    real             , parameter  :: offset=tiny(1.)/epsilon(1.) !! Tiny offset to avoid FPE
    28)    logical          , parameter  :: bypass=.true.
    29)    !---------------------------------------------------------------------------------------!
    30) 
    31)    !----- Pass over sites once to calc preliminary stats. ---------------------------------!
    32)    cpoly => cgrid%polygon(ipy)
    33) 
    34)    ebar = 0.0
    35)    aterr = 0.0
    36) 
    37)    do isi=1,cpoly%nsites
    38)       ebar  = ebar + cpoly%area(isi)*cpoly%elevation(isi)
    39)       aterr = aterr + cpoly%area(isi)
    40)    end do
    41)    ebar = ebar/aterr
    42) 
    43)    if (bypass) then
    44)       do isi = 1,cpoly%nsites
    45)          hillshade = sin(pi1*cpoly%slope(isi)/180.)/2.
    46)          cpoly%met(isi)%geoht       = cgrid%met(ipy)%geoht
    47)          cpoly%met(isi)%atm_tmp     = cgrid%met(ipy)%atm_tmp
    48)          cpoly%met(isi)%atm_shv     = cgrid%met(ipy)%atm_shv
    49)          cpoly%met(isi)%prss        = cgrid%met(ipy)%prss
    50)          cpoly%met(isi)%pcpg        = cgrid%met(ipy)%pcpg
    51)          cpoly%met(isi)%par_diffuse = cgrid%met(ipy)%par_diffuse*(1.0-hillshade)
    52)          cpoly%met(isi)%atm_co2     = cgrid%met(ipy)%atm_co2
    53)          cpoly%met(isi)%rlong       = cgrid%met(ipy)%rlong
    54)          cpoly%met(isi)%par_beam    = cgrid%met(ipy)%par_beam
    55)          cpoly%met(isi)%nir_diffuse = cgrid%met(ipy)%nir_diffuse*(1.0-hillshade)
    56)          cpoly%met(isi)%nir_beam    = cgrid%met(ipy)%nir_beam
    57)          cpoly%met(isi)%vels        = cgrid%met(ipy)%vels
    58)          cpoly%met(isi)%atm_ustar   = cgrid%met(ipy)%atm_ustar
    59)       end do
    60)    else
    61)       
    62)       !----- Second pass, calculate lapse rate adjustment. --------------------------------!
    63)       do isi = 1,cpoly%nsites
    64)          
    65)          delE = cpoly%elevation(isi) - ebar
    66)          
    67)          !----- Perform linear adjustments. -----------------------------------------------!
    68)          cpoly%met(isi)%geoht   = cgrid%met(ipy)%geoht   + cgrid%lapse(ipy)%geoht   * delE
    69)          cpoly%met(isi)%atm_tmp = cgrid%met(ipy)%atm_tmp + cgrid%lapse(ipy)%atm_tmp * delE
    70)          cpoly%met(isi)%atm_shv = max(toodry, cgrid%met(ipy)%atm_shv                       &
    71)                                             + cgrid%lapse(ipy)%atm_shv * delE)
    72)          cpoly%met(isi)%prss    = cgrid%met(ipy)%prss    + cgrid%lapse(ipy)%prss    * delE
    73)          cpoly%met(isi)%atm_co2 = cgrid%met(ipy)%atm_co2 + cgrid%lapse(ipy)%atm_co2 * delE
    74)          cpoly%met(isi)%rlong   = cgrid%met(ipy)%rlong   + cgrid%lapse(ipy)%rlong   * delE
    75)          cpoly%met(isi)%par_diffuse = (cgrid%met(ipy)%par_diffuse                           &
    76)                                     + cgrid%lapse(ipy)%par_diffuse * delE)*(1.0-hillshade)
    77)          cpoly%met(isi)%par_beam    = cgrid%met(ipy)%par_beam                              &
    78)                                     + cgrid%lapse(ipy)%par_beam * delE
    79)          cpoly%met(isi)%nir_diffuse = (cgrid%met(ipy)%nir_diffuse                           &
    80)                                     + cgrid%lapse(ipy)%nir_diffuse * delE)*(1.0-hillshade)
    81)          cpoly%met(isi)%nir_beam    = cgrid%met(ipy)%nir_beam                              &
    82)                                     + cgrid%lapse(ipy)%nir_beam * delE
    83)          !---------------------------------------------------------------------------------!
    84)          ! Note: at this point VELS is vel^2.  Thus this lapse preserves mean wind ENERGY  !
    85)          !       not wind SPEED.                                                           !
    86)          !---------------------------------------------------------------------------------!
    87)          cpoly%met(isi)%vels      = cgrid%met(ipy)%vels                                    &
    88)                                   + cgrid%lapse(ipy)%vels*delE
    89)          cpoly%met(isi)%atm_ustar = cgrid%met(ipy)%atm_ustar                               &
    90)                                   + cgrid%lapse(ipy)%atm_ustar*delE
    91) 
    92)          !---------------------------------------------------------------------------------!
    93)          ! Note: Precipitation adjustment is based on proportional change rather than a    !
    94)          !       linear scaling.                                                           !
    95)          !---------------------------------------------------------------------------------!
    96)          cpoly%met(isi)%pcpg    = cgrid%met(ipy)%pcpg * cpoly%pptweight(isi)
    97) 
    98)       end do
    99)    end if
   100) 
   101)    !----- Check whether the radiation terms make sense... ---------------------------------!
   102)    call met_sanity_check(cgrid,ipy)
   103) 
   104)    return
   105) end subroutine calc_met_lapse

   116) !      Right now, this subroutine simply transfers lapse rates from ed_data.  In the       !
   117) ! future, it could set parmameters based on spatial maps of parameters.                    !
   118) !------------------------------------------------------------------------------------------!
   119) subroutine setLapseParms(cgrid)
   120)    use ed_state_vars  , only : edtype      & ! structure
   121)                              , polygontype ! ! structure
   122)    use met_driver_coms, only : lapse       ! ! structure
   123) 
   124)    implicit none
   125)    !----- Arguments. ----------------------------------------------------------------------!
   126)    type(edtype)     , target  :: cgrid
   127)    !----- Local variables. ----------------------------------------------------------------!
   128)    type(polygontype), pointer :: cpoly
   129)    real                       :: ebar
   130)    real                       :: aterr
   131)    integer                    :: ipy
   132)    integer                    :: isi
   133)    !---------------------------------------------------------------------------------------!
   134) 
   135)   
   136)    do ipy = 1,cgrid%npolygons
   137)       
   138)       cgrid%lapse(ipy)%geoht       = lapse%geoht
   139)       cgrid%lapse(ipy)%vels        = lapse%vels
   140)       cgrid%lapse(ipy)%atm_ustar   = lapse%atm_ustar
   141)       cgrid%lapse(ipy)%atm_tmp     = lapse%atm_tmp
   142)       cgrid%lapse(ipy)%atm_shv     = lapse%atm_shv
   143)       cgrid%lapse(ipy)%prss        = lapse%prss
   144)       cgrid%lapse(ipy)%pcpg        = lapse%pcpg
   145)       cgrid%lapse(ipy)%atm_co2     = lapse%atm_co2
   146)       cgrid%lapse(ipy)%rlong       = lapse%rlong
   147)       cgrid%lapse(ipy)%nir_beam    = lapse%nir_beam
   148)       cgrid%lapse(ipy)%nir_diffuse = lapse%nir_diffuse
   149)       cgrid%lapse(ipy)%par_beam    = lapse%par_beam
   150)       cgrid%lapse(ipy)%par_diffuse = lapse%par_diffuse
   151)       cgrid%lapse(ipy)%pptnorm     = lapse%pptnorm
   152) 
   153)       !----- Precipitation weight. --------------------------------------------------------!
   154)       cpoly => cgrid%polygon(ipy)
   155)       ebar  = 0.0
   156)       aterr = 0.0
   157)       do isi = 1,cpoly%nsites
   158)          ebar  = ebar  + cpoly%area(isi)*cpoly%elevation(isi)
   159)          aterr = aterr + cpoly%area(isi)
   160)       end do
   161) 
   162)       ebar = ebar/aterr
   163)       do isi = 1,cpoly%nsites
   164)          if (cgrid%lapse(ipy)%pptnorm /= 0.) then
   165)             cpoly%pptweight(isi) = ( cgrid%lapse(ipy)%pptnorm                              &
   166)                                    + cgrid%lapse(ipy)%pcpg*(cpoly%elevation(isi) - ebar) ) &
   167)                                  / cgrid%lapse(ipy)%pptnorm
   168)          else
   169)             cpoly%pptweight(isi) = 1.0
   170)          end if
   171)       end do
   172) 
   173)    end do
   174) 
   175)    return
   176) end subroutine setLapseParms

   187) !     This subroutine will check the meteorological drivers, and print some information    !
   188) ! on the standard output, then print the banner that makes any user shiver...              !
   189) !------------------------------------------------------------------------------------------!
   190) subroutine met_sanity_check(cgrid,ipy)
   191)    use ed_state_vars  , only : edtype       & ! structure
   192)                              , polygontype  ! ! structure
   193)    use met_driver_coms, only : rshort_min   & ! intent(in)
   194)                              , rshort_max   & ! intent(in)
   195)                              , rlong_min    & ! intent(in)
   196)                              , rlong_max    & ! intent(in)
   197)                              , atm_tmp_min  & ! intent(in)
   198)                              , atm_tmp_max  & ! intent(in)
   199)                              , atm_shv_min  & ! intent(in)
   200)                              , atm_shv_max  & ! intent(in)
   201)                              , atm_co2_min  & ! intent(in)
   202)                              , atm_co2_max  & ! intent(in)
   203)                              , prss_min     & ! intent(in)
   204)                              , prss_max     & ! intent(in)
   205)                              , pcpg_min     & ! intent(in)
   206)                              , pcpg_max     & ! intent(in)
   207)                              , vels_min     & ! intent(in)
   208)                              , vels_max     & ! intent(in)
   209)                              , geoht_min    & ! intent(in)
   210)                              , geoht_max    ! ! intent(in)
   211)    use ed_misc_coms   , only : current_time ! ! intent(in)
   212)    implicit none
   213)    !----- Arguments. ----------------------------------------------------------------------!
   214)    type(edtype)     , target     :: cgrid
   215)    integer          , intent(in) :: ipy
   216)    !----- Local variables. ----------------------------------------------------------------!
   217)    type(polygontype), pointer    :: cpoly
   218)    integer                       :: isi
   219)    integer                       :: ifaterr
   220)    character(len=10)             :: now_date
   221)    character(len=12)             :: now_time
   222)    !----- Local constants. ----------------------------------------------------------------!
   223)    logical          , parameter  :: fixRad = .true.
   224)    character(len=3) , parameter  :: fmtc = '(a)'
   225)    character(len=9) , parameter  :: fmti = '(a,1x,i6)'
   226)    character(len=13), parameter  :: fmtf = '(a,1x,es12.5)'
   227)    character(len=8) , parameter  :: fmtt = '(a,1x,a)'
   228)    !---------------------------------------------------------------------------------------!
   229) 
   230) 
   231)    !----- Reset the error counter. --------------------------------------------------------!
   232)    ifaterr = 0
   233)    cpoly => cgrid%polygon(ipy)
   234) 
   235)    !----- Write the current date and time. ------------------------------------------------!
   236)    write(now_date,fmt='(2(i2.2,a),i4.4)') current_time%month,'/',current_time%date ,'/'    &
   237)                                          ,current_time%year
   238)    write(now_time,fmt='(3(i2.2,a))'     ) current_time%hour ,':',current_time%min  ,':'    &
   239)                                          ,current_time%sec,' UTC'
   240)    siteloop: do isi=1,cpoly%nsites
   241)       if (cpoly%met(isi)%geoht < geoht_min .or.                                            &
   242)           cpoly%met(isi)%geoht > geoht_max) then
   243)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   244)          write (unit=*,fmt=fmtc) ' Reference height doesn''t make sense... '
   245)          write (unit=*,fmt=fmtc) ' '
   246)          write (unit=*,fmt=fmti) ' - Polygon        :',ipy
   247)          write (unit=*,fmt=fmti) ' - Site           :',isi
   248)          write (unit=*,fmt=fmtt) ' - Date           :',now_date
   249)          write (unit=*,fmt=fmtt) ' - Time           :',now_time
   250)          write (unit=*,fmt=fmtf) ' - Longitude      :',cgrid%lon(ipy)
   251)          write (unit=*,fmt=fmtf) ' - Latitude       :',cgrid%lat(ipy)
   252)          write (unit=*,fmt=fmtf) ' - Site height    :',cpoly%met(isi)%geoht
   253)          write (unit=*,fmt=fmtf) ' - Polygon height :',cgrid%met(ipy)%geoht
   254)          write (unit=*,fmt=fmtf) ' - Minimum OK     :',geoht_min
   255)          write (unit=*,fmt=fmtf) ' - Maximum OK     :',geoht_max
   256)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   257)          write (unit=*,fmt=fmtc) ' '
   258)          ifaterr = ifaterr + 1
   259)       end if
   260) 
   261)       if (cpoly%met(isi)%atm_tmp < atm_tmp_min .or.                                        &
   262)           cpoly%met(isi)%atm_tmp > atm_tmp_max) then
   263)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   264)          write (unit=*,fmt=fmtc) ' Air temperature doesn''t make sense... '
   265)          write (unit=*,fmt=fmtc) ' '
   266)          write (unit=*,fmt=fmti) ' - Polygon        :',ipy
   267)          write (unit=*,fmt=fmti) ' - Site           :',isi
   268)          write (unit=*,fmt=fmtt) ' - Date           :',now_date
   269)          write (unit=*,fmt=fmtt) ' - Time           :',now_time
   270)          write (unit=*,fmt=fmtf) ' - Longitude      :',cgrid%lon(ipy)
   271)          write (unit=*,fmt=fmtf) ' - Latitude       :',cgrid%lat(ipy)
   272)          write (unit=*,fmt=fmtf) ' - Site temp.     :',cpoly%met(isi)%atm_tmp
   273)          write (unit=*,fmt=fmtf) ' - Polygon temp.  :',cgrid%met(ipy)%atm_tmp
   274)          write (unit=*,fmt=fmtf) ' - Minimum OK     :',atm_tmp_min
   275)          write (unit=*,fmt=fmtf) ' - Maximum OK     :',atm_tmp_max
   276)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   277)          write (unit=*,fmt=fmtc) ' '
   278)          ifaterr = ifaterr + 1
   279)       end if
   280) 
   281)       if (cpoly%met(isi)%atm_shv < atm_shv_min .or.                                        &
   282)           cpoly%met(isi)%atm_shv > atm_shv_max) then
   283)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   284)          write (unit=*,fmt=fmtc) ' Air specific humidity doesn''t make sense... '
   285)          write (unit=*,fmt=fmtc) ' '
   286)          write (unit=*,fmt=fmti) ' - Polygon        :',ipy
   287)          write (unit=*,fmt=fmti) ' - Site           :',isi
   288)          write (unit=*,fmt=fmtt) ' - Date           :',now_date
   289)          write (unit=*,fmt=fmtt) ' - Time           :',now_time
   290)          write (unit=*,fmt=fmtf) ' - Longitude      :',cgrid%lon(ipy)
   291)          write (unit=*,fmt=fmtf) ' - Latitude       :',cgrid%lat(ipy)
   292)          write (unit=*,fmt=fmtf) ' - Site spec. hum.:',cpoly%met(isi)%atm_shv
   293)          write (unit=*,fmt=fmtf) ' - Polygon sp. h. :',cgrid%met(ipy)%atm_shv
   294)          write (unit=*,fmt=fmtf) ' - Minimum OK     :',atm_shv_min
   295)          write (unit=*,fmt=fmtf) ' - Maximum OK     :',atm_shv_max
   296)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   297)          write (unit=*,fmt=fmtc) ' '
   298)          ifaterr = ifaterr + 1
   299)       end if
   300) 
   301)       if (cpoly%met(isi)%atm_co2 < atm_co2_min .or.                                        &
   302)           cpoly%met(isi)%atm_co2 > atm_co2_max) then
   303)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   304)          write (unit=*,fmt=fmtc) ' Air carbon dioxide doesn''t make sense... '
   305)          write (unit=*,fmt=fmtc) ' '
   306)          write (unit=*,fmt=fmti) ' - Polygon        :',ipy
   307)          write (unit=*,fmt=fmti) ' - Site           :',isi
   308)          write (unit=*,fmt=fmtt) ' - Date           :',now_date
   309)          write (unit=*,fmt=fmtt) ' - Time           :',now_time
   310)          write (unit=*,fmt=fmtf) ' - Longitude      :',cgrid%lon(ipy)
   311)          write (unit=*,fmt=fmtf) ' - Latitude       :',cgrid%lat(ipy)
   312)          write (unit=*,fmt=fmtf) ' - Site CO2       :',cpoly%met(isi)%atm_co2
   313)          write (unit=*,fmt=fmtf) ' - Polygon CO2    :',cgrid%met(ipy)%atm_co2
   314)          write (unit=*,fmt=fmtf) ' - Minimum OK     :',atm_co2_min
   315)          write (unit=*,fmt=fmtf) ' - Maximum OK     :',atm_co2_max
   316)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   317)          write (unit=*,fmt=fmtc) ' '
   318)          ifaterr = ifaterr + 1
   319)       end if
   320) 
   321)       if (cpoly%met(isi)%prss < prss_min .or.                                              &
   322)           cpoly%met(isi)%prss > prss_max) then
   323)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   324)          write (unit=*,fmt=fmtc) ' Atmospheric pressure doesn''t make sense... '
   325)          write (unit=*,fmt=fmtc) ' '
   326)          write (unit=*,fmt=fmti) ' - Polygon        :',ipy
   327)          write (unit=*,fmt=fmti) ' - Site           :',isi
   328)          write (unit=*,fmt=fmtt) ' - Date           :',now_date
   329)          write (unit=*,fmt=fmtt) ' - Time           :',now_time
   330)          write (unit=*,fmt=fmtf) ' - Longitude      :',cgrid%lon(ipy)
   331)          write (unit=*,fmt=fmtf) ' - Latitude       :',cgrid%lat(ipy)
   332)          write (unit=*,fmt=fmtf) ' - Site press.    :',cpoly%met(isi)%prss
   333)          write (unit=*,fmt=fmtf) ' - Polygon press. :',cgrid%met(ipy)%prss
   334)          write (unit=*,fmt=fmtf) ' - Minimum OK     :',prss_min
   335)          write (unit=*,fmt=fmtf) ' - Maximum OK     :',prss_max
   336)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   337)          write (unit=*,fmt=fmtc) ' '
   338)          ifaterr = ifaterr + 1
   339)       end if
   340) 
   341)       if (cpoly%met(isi)%pcpg < pcpg_min .or.                                              &
   342)           cpoly%met(isi)%pcpg > pcpg_max) then
   343)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   344)          write (unit=*,fmt=fmtc) ' Precipitation rate doesn''t make sense... '
   345)          write (unit=*,fmt=fmtc) ' '
   346)          write (unit=*,fmt=fmti) ' - Polygon        :',ipy
   347)          write (unit=*,fmt=fmti) ' - Site           :',isi
   348)          write (unit=*,fmt=fmtt) ' - Date           :',now_date
   349)          write (unit=*,fmt=fmtt) ' - Time           :',now_time
   350)          write (unit=*,fmt=fmtf) ' - Longitude      :',cgrid%lon(ipy)
   351)          write (unit=*,fmt=fmtf) ' - Latitude       :',cgrid%lat(ipy)
   352)          write (unit=*,fmt=fmtf) ' - Site Precip    :',cpoly%met(isi)%pcpg
   353)          write (unit=*,fmt=fmtf) ' - Polygon Precip :',cgrid%met(ipy)%pcpg
   354)          write (unit=*,fmt=fmtf) ' - Minimum OK     :',pcpg_min
   355)          write (unit=*,fmt=fmtf) ' - Maximum OK     :',pcpg_max
   356)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   357)          write (unit=*,fmt=fmtc) ' '
   358)          ifaterr = ifaterr + 1
   359)       end if
   360) 
   361)       !------------------------------------------------------------------------------------!
   362)       !     At this point, vels is twice the kinetic energy.  Check the square root of     !
   363)       ! vels, so we are comparing apples to apples.                                        !
   364)       !------------------------------------------------------------------------------------!
   365)       if (cpoly%met(isi)%vels < vels_min * vels_min) then
   366)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   367)          write (unit=*,fmt=fmtc) ' Kinetic energy is negative...'
   368)          write (unit=*,fmt=fmtc) ' '
   369)          write (unit=*,fmt=fmti) ' - Polygon                :',ipy
   370)          write (unit=*,fmt=fmti) ' - Site                   :',isi
   371)          write (unit=*,fmt=fmtt) ' - Date                   :',now_date
   372)          write (unit=*,fmt=fmtt) ' - Time                   :',now_time
   373)          write (unit=*,fmt=fmtf) ' - Longitude              :',cgrid%lon(ipy)
   374)          write (unit=*,fmt=fmtf) ' - Latitude               :',cgrid%lat(ipy)
   375)          write (unit=*,fmt=fmtf) ' - Site kinetic energy    :',0.5 * cpoly%met(isi)%vels
   376)          write (unit=*,fmt=fmtf) ' - Polygon kinetic energy :',0.5 * cgrid%met(ipy)%vels
   377)          write (unit=*,fmt=fmtf) ' - Minimum OK             :',0.5 * vels_min * vels_min
   378)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   379)          write (unit=*,fmt=fmtc) ' '
   380)          ifaterr = ifaterr + 1
   381)       elseif (cpoly%met(isi)%vels > vels_max * vels_max) then
   382)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   383)          write (unit=*,fmt=fmtc) ' Wind speed doesn''t make sense... '
   384)          write (unit=*,fmt=fmtc) ' '
   385)          write (unit=*,fmt=fmti) ' - Polygon        :',ipy
   386)          write (unit=*,fmt=fmti) ' - Site           :',isi
   387)          write (unit=*,fmt=fmtt) ' - Date           :',now_date
   388)          write (unit=*,fmt=fmtt) ' - Time           :',now_time
   389)          write (unit=*,fmt=fmtf) ' - Longitude      :',cgrid%lon(ipy)
   390)          write (unit=*,fmt=fmtf) ' - Latitude       :',cgrid%lat(ipy)
   391)          write (unit=*,fmt=fmtf) ' - Site wind      :',sqrt(cpoly%met(isi)%vels)
   392)          write (unit=*,fmt=fmtf) ' - Polygon wind   :',sqrt(max(0.,cgrid%met(ipy)%vels))
   393)          write (unit=*,fmt=fmtf) ' - Minimum OK     :',vels_min
   394)          write (unit=*,fmt=fmtf) ' - Maximum OK     :',vels_max
   395)          write (unit=*,fmt=fmtc) '---------------------------------------------------'
   396)          write (unit=*,fmt=fmtc) ' '
   397)          ifaterr = ifaterr + 1
   398)       end if
