    30)    !---------------------------------------------------------------------------------------!
    31)    !    Set all cohorts to not have enough leaf area index or wood area index to be        !
    32)    ! resolved.  The code will update this every time step, but we must assign an initial   !
    33)    ! value so the debugger won't complain.                                                 !
    34)    !---------------------------------------------------------------------------------------!
    35)    cpatch%leaf_resolvable(ico) = .false.
    36)    cpatch%wood_resolvable(ico) = .false.

    41)    !---------------------------------------------------------------------------------------!
    42)    !     The leaf life span is initialised with the inverse of the turnover rate.  Notice  !
    43)    ! that the turnover rate is in years, but the life span is in months.  Also, some PFTs  !
    44)    ! do not define the turnover rate (temperate cold-deciduous for example), in which case !
    45)    ! we assign a meaningless number just to make sure the variable is initialised.         !
    46)    !---------------------------------------------------------------------------------------!
    47)    if (leaf_turnover_rate(cpatch%pft(ico)) > 0.0) then
    48)       cpatch%llspan(ico) = 12.0 / leaf_turnover_rate(cpatch%pft(ico))
    49)    else
    50)       cpatch%llspan(ico) = llspan_inf
    51)    end if
    52)    !---------------------------------------------------------------------------------------!
    53) 
    54) 
    55)    !---------------------------------------------------------------------------------------!
    56)    !      The maximum capacity of Rubisco to perform the carboxylase function (Vm) and the !
    57)    ! specific leaf area (SLA) must be assigned with the default values.  These numbers     !
    58)    ! will change only if the PFT uses a light-controlled phenology.                        !
    59)    !---------------------------------------------------------------------------------------!
    60)    select case(phenology(cpatch%pft(ico)))
    61)    case (3)
    62)       cpatch%vm_bar(ico) = vm0_amp / (1.0 + (cpatch%llspan(ico)/vm0_tran)**vm0_slope)      &
    63)                          + vm0_min
    64)    case default
    65)       cpatch%vm_bar(ico) = Vm0(cpatch%pft(ico))
    66)    end select
    67)    cpatch%sla(ico) = sla(cpatch%pft(ico))
    68)    !---------------------------------------------------------------------------------------!
    69) 
    70) 
    71)    !---------------------------------------------------------------------------------------!
    72)    !     Start the fraction of open stomata with 1., since this is the most likely value   !
    73)    ! at night time.  FS_open is initialised with 0., though.                               !
    74)    !---------------------------------------------------------------------------------------!
    75)    cpatch%fsw    (ico) = 1.0
    76)    cpatch%fsn    (ico) = 1.0
    77)    cpatch%fs_open(ico) = 0.0
    78)    !---------------------------------------------------------------------------------------!
    79)

    82)    ! life span in the light-controlled phenology.                                          !

    91)    ! element (current month integration) must be set to 0, though, otherwise the first     !
    92)    ! month carbon balance will be incorrect.                                               !
    93)    !---------------------------------------------------------------------------------------!

    95)    cpatch%cb_lightmax      (1:12,ico) = 1.0
    96)    cpatch%cb_moistmax      (1:12,ico) = 1.0
