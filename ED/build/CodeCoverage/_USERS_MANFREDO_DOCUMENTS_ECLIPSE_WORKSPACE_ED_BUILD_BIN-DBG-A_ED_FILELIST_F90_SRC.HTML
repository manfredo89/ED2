    20) subroutine ed_filelist(fnames,file_prefix,nfile)
    21)  
    22)    use ed_max_dims, only : str_len  & ! intent(in)
    23)                          , maxfiles ! ! intent(in)
    24) 
    25)    implicit none
    26)    !------ Arguments. ---------------------------------------------------------------------!
    27)    integer                        , intent(out)    :: nfile
    28)    character(len=*), dimension(*) , intent(inout)  :: fnames
    29)    character(len=str_len)         , intent(inout)  :: file_prefix
    30)    !------ Local variables. ---------------------------------------------------------------!
    31)    character(len=maxfiles*str_len)                 :: filelist
    32)    character(len=str_len)                          :: prefix
    33)    character(len=str_len)                          :: myform
    34)    integer, dimension(maxfiles)                    :: indices
    35)    integer                                         :: iprelen
    36)    integer                                         :: nf
    37)    integer                                         :: n
    38)    !---------------------------------------------------------------------------------------!
    39) 
    40) 
    41) 
    42)    !----- Initialise the file counter. ----------------------------------------------------!
    43)    nfile = 0
    44)    !---------------------------------------------------------------------------------------!
    45) 
    46) 
    47) 
    48)    !----- String format.  This works if str_len is between 100 and 999. -------------------!
    49)    write(myform,fmt='(a,i3.3,a)') '(a',str_len,')'
    50)    !---------------------------------------------------------------------------------------!
    51) 
    52) 
    53) 
    54)    !----- Determine the files based on the prefix -----------------------------------------!
    55)    write (unit=*,fmt='(a,1x,a)') ' [-] filelist_f: Checking prefix: ',trim(file_prefix)
    56)    !---------------------------------------------------------------------------------------!
    57) 
    58) 
    59) 
    60)    !----- Find the length of the prefix. --------------------------------------------------!
    61)    iprelen = len_trim(file_prefix)
    62)    if(iprelen == 0) iprelen=len(file_prefix)
    63)    !---------------------------------------------------------------------------------------!
    64) 
    65) 
    66) 
    67) !----- Use preprocessor tool. Windows require different way to list... --------------------!
    68) #if defined (PC_NT1)
    69)    !---------------------------------------------------------------------------------------!
    70)    ! First change all "/" to "\" so same namelist can be used for Unix/Linux/Windows       !
    71)    !---------------------------------------------------------------------------------------!
    72)    do nc=1,iprelen
    73)       if(file_prefix(nc:nc) == '/') file_prefix(nc:nc)='\'
    74)    enddo
    75)    command=  &
    76)      'dir /b '//file_prefix(1:len_trim(file_prefix))//' >c:\temp\MEVI_filelist'
    77)    call system(trim(command))
    78)    
    79)    !----- Open the directory list ---------------------------------------------------------!
    80)    iun=66
    81)    open(unit=iun,file='c:\temp\ED22_filelist',status='old',iostat=ierr)
    82)    if (ierr /= 0) call fatal_error('filelist_f: Error opening temporary MEVI_filelist'  &
    83)                                   ,'filelist_f','filelist.F90')
    84)    !---------------------------------------------------------------------------------------!
    85) 
    86) 
    87) 
    88)    !----- Read through the files. Windows doesn't put directory names on "dir", so... -----!
    89)    foundbslash = .false.
    90)    nameloop: do nc =len_trim(file_prefix),1,-1
    91)       if (file_prefix(nc:nc) == '\') then
    92)          lndir=nc
    93)          cdir=file_prefix(1:lndir)
    94)          foundbslash = .true.
    95)          exit nameloop
    96)       end if
    97)    end do nameloop
    98)    if (.not. foundbslash) then
    99)      lndir=2
   100)      cdir='.\'
   101)    end if
   102)    !---------------------------------------------------------------------------------------!
   103) 
   104)

   106)    !    Get the file names from the scratch file.                                          !
   107)    !---------------------------------------------------------------------------------------!
   108)    readloop: do
   109)       read(unit=iun,fmt=myform,iostat=ierr) file
   110)       if (ierr /= 0) exit readloop
   111)       fnames(nf) = cdir(1:lndir)//file
   112)    end do readloop
   113)    close(unit=iun,status='delete')
   114)    !---------------------------------------------------------------------------------------!
   115) 
   116) #else
   117) 
   118)    !----- Appending char(0), so C can handle the characters properly. ---------------------!
   119)    prefix = file_prefix(1:iprelen) // char(0)
   120) 
   121)    !---------------------------------------------------------------------------------------!
   122)    !    The following subroutine uses intrinsic C functions to return an vector of file    !

   125)    !---------------------------------------------------------------------------------------!
   126)    call filelist_c(n,indices,prefix,filelist)

   128) 
   129) 
   130) 
   131)    !---------------------------------------------------------------------------------------!
   132)    !      Print the first 10 files on screen, so the user can check whether it makes sense !
   133)    ! or not.                                                                               !
   134)    !---------------------------------------------------------------------------------------!
   135)    write (unit=*,fmt='(a)') ' +  Showing first 10 files:'

   137)       fnames(nf) = trim(filelist(indices(nf):indices(nf+1)-1))
   138) 
   139)       if (nf <= 10) then
   140)          write (unit=*,fmt='(a,1x,i5,1x,a)') '   [-] File #: ',nf,trim(fnames(nf))
   141)       end if
   142)    end do
   143)    !---------------------------------------------------------------------------------------!
   144) 
   145) 
   146) 
   147) #endif
   148)    
   149)    !------ Update the total number of files. ----------------------------------------------!
   150)    nfile=nf-1
   151)    !---------------------------------------------------------------------------------------!
   152) 
   153)    !---------------------------------------------------------------------------------------!
   154)    !      Crash in case there isn't a file with the sought prefix.                         !
   155)    !---------------------------------------------------------------------------------------!
   156)    if (nfile == 0) then
   157)       print *, 'No INPUT files for prefix:',file_prefix
   158)       call fatal_error('filelist_f: No files found...','filelist_f','filelist.F90')
   159)    end if
   160)    !---------------------------------------------------------------------------------------!
   161) 
   162)    return
   163) end subroutine ed_filelist

   174) subroutine ed1_fileinfo(text,nfiles,full_list,ntype,type_list,tlon_list,tlat_list)
   175)    use ed_max_dims, only: str_len,maxfiles,maxlist
   176)    implicit none
   177)    !----- Arguments -----------------------------------------------------------------------!
   178)    character(len=*)                           , intent(in)  :: text   ! Type file extension
   179)    integer                                    , intent(in)  :: nfiles ! # of input files
   180)    character(len=str_len), dimension(maxlist) , intent(in)  :: full_list 
   181)    integer                                    , intent(inout) :: ntype
   182)    character(len=str_len), dimension(maxfiles), intent(inout) :: type_list
   183)    real                  , dimension(maxfiles), intent(inout) :: tlon_list, tlat_list
   184)    !----- Local variables -----------------------------------------------------------------!
   185)    character(len=str_len)                                   :: this_file
   186)    integer                                                  :: n,posend,posdot,posextradot
   187)    integer                                                  :: latbeg,lonbeg,poslondot
   188)    integer                                                  :: okdot
   189)    !---------------------------------------------------------------------------------------!
   190) 
   191) 
   192)    !----- Safety check, just to avoid assigning the wrong position... ---------------------!
   193)    select case(text)
   194)    case ('.site')
   195)       okdot = 4
   196)    case ('.pss','.css','.txt')
   197)       okdot = 3
   198)    case ('.lu')
   199)       okdot = 2
   200)    end select
   201) 
   202) 
   203)    !---------------------------------------------------------------------------------------!
   204)    !     Counting how many files with the extension are there.  If there is no such file,  !
   205)    ! return without anything else assigned...                                              !
   206)    !---------------------------------------------------------------------------------------!
   207)    ntype = 0
   208)    listloop: do n=1,nfiles
   209)       !---- Checking whether this file is the type we are looking for ---------------------!
   210)       this_file = full_list(n)
   211)       posdot    = index(this_file,text,.true.)
   212)       posend    = len_trim(this_file)
   213)             
   214)       if (posdot == 0 .or. posend-posdot /= okdot) cycle listloop
   215)       
   216)       !----- Seeking for some possible dots after lon but before the file extension -------!
   217)       posextradot = index(this_file(1:(posdot-1)),'.',.true.)
   218) 
   219)       !---- It is! Adding one more to our count and assigning the file to this count. -----!
   220)       ntype=ntype+1
   221)       type_list(ntype) = this_file
   222)       
   223)       !---- Figuring out the latitude and longitude of this file. -------------------------!
   224)       latbeg = index(this_file,'lat',.true.) + 3
   225)       lonbeg = index(this_file,'lon',.true.) + 3
   226)       poslondot = index(this_file(lonbeg:(posdot-1)),'.',.false.)+lonbeg-1
   227) 
   228)       read(this_file(latbeg:lonbeg-4),fmt=*) tlat_list(ntype)
   229) 
   230) 
   231)       !---- If poslondot is the same as posextradot, no extra dot was found ---------------!
   232)       if (poslondot == posextradot) then
   233)          read(this_file(lonbeg:posdot-1),fmt=*) tlon_list(ntype)
   234)       else
   235)          read(this_file(lonbeg:posextradot-1),fmt=*) tlon_list(ntype)
   236)       end if
   237)       
   238)    end do listloop
   239)    !---------------------------------------------------------------------------------------!
   240) 
   241)    return
   242) end subroutine ed1_fileinfo

   253) !     This subroutine will go through a list of file and retain only those that are actual !
   254) ! ED-2.1 history files.                                                                    !
   255) !------------------------------------------------------------------------------------------!
   256) subroutine ed21_fileinfo(nfiles,full_list,nhisto,histo_list)
   257)    use ed_max_dims, only: str_len,maxfiles,maxlist
   258)    implicit none
   259)    !----- Arguments -----------------------------------------------------------------------!
   260)    integer                                    , intent(in)    :: nfiles ! # of input files
   261)    character(len=str_len), dimension(maxlist) , intent(in)    :: full_list 
   262)    integer                                    , intent(inout) :: nhisto
   263)    character(len=str_len), dimension(maxfiles), intent(inout) :: histo_list
   264)    !----- Local variables -----------------------------------------------------------------!
   265)    character(len=str_len)                                     :: this_file
   266)    integer                                                    :: n
   267)    integer                                                    :: posend
   268)    integer                                                    :: posdot
   269)    integer                                                    :: posdsd
   270)    !----- Local constants. ----------------------------------------------------------------!
   271)    integer                                    , parameter     :: okdot = 2
   272)    integer                                    , parameter     :: okdsd = 26
   273)    !---------------------------------------------------------------------------------------!
   274) 
   275) 
   276) 
   277)    !---------------------------------------------------------------------------------------!
   278)    !     Counting how many files with the extension are there.  If there is no such file,  !
   279)    ! return without anything else assigned...                                              !
   280)    !---------------------------------------------------------------------------------------!
   281)    nhisto        = 0
   282)    histo_list(:) = ''
   283)    listloop: do n=1,nfiles
   284)       !------------------------------------------------------------------------------------!
   285)       !     Checking whether this file is the type we are looking for.  To qualify as a    !
   286)       ! restart/history ED-2.1 file, the file must end with .h5, and must have a -S- right !
   287)       ! before the date information.                                                       !
   288)       !------------------------------------------------------------------------------------!
   289)       this_file = full_list(n)
   290)       posdot    = index(this_file,'.h5',.true.)
   291)       posdsd    = index(this_file,'-S-',.true.)
   292)       posend    = len_trim(this_file)
   293) 
   294)       !------------------------------------------------------------------------------------!
   295)       !     If anything is not in place, then this file will not be used for the history   !
   296)       ! initialisation, since it couldn't be considered an ED-2.1 history file.            !
   297)       !------------------------------------------------------------------------------------!
   298)       if (posdot == 0 .or. posend-posdot /= okdot) cycle listloop
   299)       if (posdsd == 0 .or. posend-posdsd /= okdsd) cycle listloop
   300) 
   301) 
   302)       !------------------------------------------------------------------------------------!
   303)       !     If we reach this point, it means that the filename at least looks like a       !
   304)       ! restart.  Adding one more to our count and assigning the file to this count.       !
   305)       !------------------------------------------------------------------------------------!
   306)       nhisto             = nhisto + 1
   307)       histo_list(nhisto) = this_file
   308) 
   309)    end do listloop
   310)    !---------------------------------------------------------------------------------------!
   311) 
   312)    return
   313) end subroutine ed21_fileinfo
