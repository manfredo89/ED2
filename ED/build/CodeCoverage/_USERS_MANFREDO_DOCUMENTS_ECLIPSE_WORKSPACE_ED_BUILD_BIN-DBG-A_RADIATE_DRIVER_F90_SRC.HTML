     6) !     This subroutine will control the two-stream radiation scheme.  This is called every  !
     7) ! step, but not every sub-step.                                                            !
     8) !------------------------------------------------------------------------------------------!
     9) subroutine radiate_driver(cgrid)
    10)    use ed_misc_coms          , only : current_time          & ! intent(in)
    11)                                     , radfrq                & ! intent(in)
    12)                                     , dtlsm                 ! ! intent(in)
    13)    use ed_state_vars         , only : edtype                & ! structure
    14)                                     , polygontype           & ! structure
    15)                                     , sitetype              & ! structure
    16)                                     , patchtype             ! ! structure
    17)    use canopy_radiation_coms , only : par_beam_norm         & ! intent(in)
    18)                                     , par_diff_norm         & ! intent(in)
    19)                                     , nir_beam_norm         & ! intent(in)
    20)                                     , nir_diff_norm         & ! intent(in)
    21)                                     , cosz_min              & ! intent(in)
    22)                                     , rshort_twilight_min   ! ! intent(in)
    23)    use consts_coms           , only : pio180                ! ! intent(in)
    24)    use grid_coms             , only : nzg                   & ! intent(in)
    25)                                     , nzs                   ! ! intent(in)
    26)    implicit none
    27)    !----- Argument. -----------------------------------------------------------------------!
    28)    type(edtype)     , target   :: cgrid
    29)    !----- Local variables. ----------------------------------------------------------------!
    30)    type(polygontype), pointer  :: cpoly
    31)    type(sitetype)   , pointer  :: csite
    32)    type(patchtype)  , pointer  :: cpatch
    33)    real                        :: rshort_tot
    34)    integer                     :: maxcohort
    35)    integer                     :: ipy
    36)    integer                     :: isi
    37)    integer                     :: ipa
    38)    integer                     :: tuco
    39)    logical                     :: daytime
    40)    logical                     :: twilight
    41)    real                        :: hrangl
    42)    real                        :: sloperad
    43)    real                        :: aspectrad
    44)    real                        :: sum_norm
    45)    !----- External functions. -------------------------------------------------------------!
    46)    real             , external :: ed_zen
    47)    !---------------------------------------------------------------------------------------!
    48) 
    49) 
    50)    !----- Check whether it is time to update radiative fluxes and heating rates -----------!
    51)    if (mod(current_time%time + .001,radfrq) < dtlsm) then
    52) 
    53)       !----- Loop over polygons and sites. ------------------------------------------------!
    54) 
    55)       polyloop: do ipy = 1,cgrid%npolygons
    56) 
    57)          !----- Find the solar zenith angle [cosz] -----------------------------------------!
    58)          cgrid%cosz(ipy) = ed_zen(cgrid%lon(ipy),cgrid%lat(ipy),current_time)
    59)          !---------------------------------------------------------------------------------!
    60) 
    61)          cpoly => cgrid%polygon(ipy)
    62) 
    63)          siteloop: do isi = 1,cpoly%nsites
    64) 
    65)             csite => cpoly%site(isi)
    66) 
    67)             !------------------------------------------------------------------------------!
    68)             !     Update angle of incidence.                                               !
    69)             !------------------------------------------------------------------------------!
    70)             hrangl    = 15. * pio180                                                       &
    71)                       * (mod(current_time%time + cgrid%lon(ipy) / 15. + 24., 24.) - 12.)
    72)             sloperad  = cpoly%slope(isi)  * pio180
    73)             aspectrad = cpoly%aspect(isi) * pio180
    74)             call angle_of_incid(cpoly%cosaoi(isi),cgrid%cosz(ipy),hrangl                   &
    75)                                ,sloperad,aspectrad)
    76)             !------------------------------------------------------------------------------!
    77) 
    78) 
    79)             !------------------------------------------------------------------------------!
    80)             !    Find the two logicals, that will tell which part of the day we are at     !
    81)             ! least.                                                                       !
    82)             !------------------------------------------------------------------------------!
    83)             daytime  = cpoly%cosaoi(isi) > cosz_min .and.                                  &
    84)                        cpoly%met(isi)%rshort > rshort_twilight_min
    85)             twilight = cpoly%met(isi)%rshort > rshort_twilight_min
    86)             !------------------------------------------------------------------------------!
    87) 
    88) 
    89)             !------------------------------------------------------------------------------!
    90)             !      Update the daylight length and nighttime flag.                          !
    91)             !------------------------------------------------------------------------------!
    92)             cpoly%nighttime(isi)              = .not. twilight
    93)             if (twilight) cpoly%daylight(isi) = cpoly%daylight(isi) + radfrq
    94)             !------------------------------------------------------------------------------!
    95) 
    96) 
    97)             !------------------------------------------------------------------------------!
    98)             !      In case the angle of incidence is too high (i.e., its cosine is too     !
    99)             ! close to zero), eliminate all direct radiation.   This is different from the !
   100)             ! cosine of the zenith angle because mountains can hide the sun even when it   !
   101)             ! is still above the horizon.                                                  !
   102)             !------------------------------------------------------------------------------!
   103)             if (daytime) then
   104)                rshort_tot    = cpoly%met(isi)%rshort
   105)                par_beam_norm = max( 1.d-5 , dble(cpoly%met(isi)%par_beam   )               &
   106)                                           / dble(cpoly%met(isi)%rshort     ) )
   107)                par_diff_norm = max( 1.d-5 , dble(cpoly%met(isi)%par_diffuse)               &
   108)                                           / dble(cpoly%met(isi)%rshort     ) )
   109)                nir_beam_norm = max( 1.d-5 , dble(cpoly%met(isi)%nir_beam   )               &
   110)                                           / dble(cpoly%met(isi)%rshort     ) )
   111)                nir_diff_norm = max( 1.d-5 , dble(cpoly%met(isi)%nir_diffuse)               &
   112)                                           / dble(cpoly%met(isi)%rshort     ) )
   113)                sum_norm      = par_beam_norm + par_diff_norm                               &
   114)                              + nir_beam_norm + nir_diff_norm
   115)             elseif (twilight) then
   116)                rshort_tot    = cpoly%met(isi)%rshort_diffuse
   117)                par_beam_norm = 1.d-5
   118)                par_diff_norm = 1.d-5
   119)                nir_beam_norm = max(1.d-5, dble(cpoly%met(isi)%nir_beam   )                 &
   120)                                         / dble(cpoly%met(isi)%rshort     ) )
   121)                nir_diff_norm = max(1.d-5, dble(cpoly%met(isi)%nir_diffuse)                 &
   122)                                         / dble(cpoly%met(isi)%rshort     ) )
   123)                sum_norm      = par_beam_norm + par_diff_norm                               &
   124)                              + nir_beam_norm + nir_diff_norm
   125)             else
   126)                !---------------------------------------------------------------------------!
   127)                !     Night-time, nothing will happen, fill split equally to the 4          !
   128)                ! components.                                                               !
   129)                !---------------------------------------------------------------------------!
   130)                rshort_tot    = 0.0
   131)                par_beam_norm = 2.5d-1
   132)                par_diff_norm = 2.5d-1
   133)                nir_beam_norm = 2.5d-1
   134)                nir_diff_norm = 2.5d-1
   135)                sum_norm      = 1.d0
   136)             end if
   137)             !------------------------------------------------------------------------------!
   138)             !     Because we must tweak the radiation so none of the terms are zero, we    !
   139)             ! must correct the normalised radiation variables so they add up to one.       !
   140)             !------------------------------------------------------------------------------!
   141)             par_beam_norm = par_beam_norm / sum_norm
   142)             par_diff_norm = par_diff_norm / sum_norm
   143)             nir_beam_norm = nir_beam_norm / sum_norm
   144)             nir_diff_norm = nir_diff_norm / sum_norm
   145)             !------------------------------------------------------------------------------!
   146) 
   147)             !------------------------------------------------------------------------------!
   148)             !    Loop over subgrid-scale patches.  These routines can be done as arrays.   !
   149)             !------------------------------------------------------------------------------!
   150)             maxcohort = 1
   151)             do ipa = 1,csite%npatches
   152)                cpatch=>csite%patch(ipa)
   153)                if ( cpatch%ncohorts>maxcohort ) maxcohort = cpatch%ncohorts
   154)             end do
   155)             !------------------------------------------------------------------------------!
   156) 
   157) 
   158)             !----- Get unnormalized radiative transfer information. -----------------------!
   159)             call sfcrad_ed(cpoly%cosaoi(isi),csite,nzg,nzs,cpoly%ntext_soil(:,isi)         &
   160)                            ,cpoly%ncol_soil(isi),tuco,cpoly%met(isi)%rlong,twilight)
   161)             !------------------------------------------------------------------------------!
   162) 
   163) 
   164) 
   165)             !----- Normalize the absorbed radiations. -------------------------------------!
   166)             call scale_ed_radiation(rshort_tot,cpoly%met(isi)%rlong                   &
   167)                                     ,cpoly%nighttime(isi),csite)
   168)             !------------------------------------------------------------------------------!
   169) 
   170)          end do siteloop
   171)       end do polyloop
   172) 
   173)       !----- Update the average radiation for phenology. ----------------------------------!
   174)       call update_rad_avg(cgrid)
   175)       !------------------------------------------------------------------------------------!
   176) 
   177)    end if
   178) 
   179)    return
   180) end subroutine radiate_driver

   194) subroutine sfcrad_ed(cosaoi,csite,mzg,mzs,ntext_soil,ncol_soil,tuco,rlong,twilight)
   195) 
   196)    use ed_state_vars        , only : sitetype             & ! structure
   197)                                    , patchtype            ! ! structure
   198)    use canopy_layer_coms    , only : crown_mod            & ! intent(in)
   199)                                    , tai_lyr_max          ! ! intent(in)
   200)    use canopy_radiation_coms, only : icanrad              & ! intent(in)
   201)                                    , clumping_factor      & ! intent(in)
   202)                                    , par_beam_norm        & ! intent(in)
   203)                                    , par_diff_norm        & ! intent(in)
   204)                                    , nir_beam_norm        & ! intent(in)
   205)                                    , nir_diff_norm        & ! intent(in)
   206)                                    , leaf_trans_vis       & ! intent(in)
   207)                                    , leaf_reflect_vis     & ! intent(in)
   208)                                    , leaf_scatter_vis     & ! intent(out)
   209)                                    , leaf_backscatter_vis & ! intent(out)
   210)                                    , wood_trans_vis       & ! intent(in)
   211)                                    , wood_reflect_vis     & ! intent(in)
   212)                                    , wood_scatter_vis     & ! intent(out)
   213)                                    , wood_backscatter_vis & ! intent(out)
   214)                                    , leaf_trans_nir       & ! intent(in)
   215)                                    , leaf_reflect_nir     & ! intent(in)
   216)                                    , leaf_scatter_nir     & ! intent(out)
   217)                                    , leaf_backscatter_nir & ! intent(out)
   218)                                    , wood_trans_nir       & ! intent(in)
   219)                                    , wood_reflect_nir     & ! intent(in)
   220)                                    , wood_scatter_nir     & ! intent(out)
   221)                                    , wood_backscatter_nir & ! intent(out)
   222)                                    , leaf_emiss_tir       & ! intent(in)
   223)                                    , wood_emiss_tir       & ! intent(in)
   224)                                    , snow_albedo_vis      & ! intent(in)
   225)                                    , snow_albedo_nir      & ! intent(in)
   226)                                    , snow_emiss_tir       & ! intent(in)
   227)                                    , orient_factor        & ! intent(out)
   228)                                    , phi1                 & ! intent(out)
   229)                                    , phi2                 & ! intent(out)
   230)                                    , mu_bar               & ! intent(out)
   231)                                    , radscr
   232)    use soil_coms            , only : soil                 & ! intent(in)
   233)                                    , soilcol              ! ! intent(in)
   234)    use consts_coms          , only : stefan               & ! intent(in)
   235)                                    , lnexp_max            ! ! intent(in)
   236)    use ed_max_dims          , only : n_pft                & ! intent(in)
   237)                                    , n_radprof            ! ! intent(in)
   238)    use allometry            , only : h2crownbh            ! ! intent(in)
   239)    use ed_misc_coms         , only : ibigleaf             ! ! intent(in)
   240)    !$ use omp_lib
   241) 
   242)    implicit none
   243)    !----- Arguments. ----------------------------------------------------------------------!
   244)    type(sitetype)                  , target      :: csite
   245)    integer                         , intent(in)  :: mzg
   246)    integer                         , intent(in)  :: mzs
   247)    integer         , dimension(mzg), intent(in)  :: ntext_soil
   248)    integer                         , intent(in)  :: ncol_soil
   249)    real                            , intent(in)  :: rlong
   250)    real                            , intent(in)  :: cosaoi
   251)    logical                         , intent(in)  :: twilight
   252)    integer                         , intent(out) :: tuco
   253)    !----- Local variables. ----------------------------------------------------------------!
   254)    type(patchtype) , pointer                     :: cpatch
   255)    integer                                       :: il
   256)    integer                                       :: ipa
   257)    integer                                       :: ico
   258)    integer                                       :: ipft
   259)    integer                                       :: cohort_count
   260)    integer                                       :: nsoil
   261)    integer                                       :: colour
   262)    integer                                       :: k
   263)    integer                                       :: ksn
   264)    integer                                       :: tuco_leaf
   265)    real                                          :: fcpct
   266)    real                                          :: albedo_soil_par
   267)    real                                          :: albedo_soil_nir
   268)    real                                          :: albedo_damp_par
   269)    real                                          :: albedo_damp_nir
   270)    real                                          :: albedo_sfcw_par
   271)    real                                          :: albedo_sfcw_nir
   272)    real                                          :: rad_sfcw_par
   273)    real                                          :: rad_sfcw_nir
   274)    real                                          :: sfcw_odepth
   275)    real                                          :: fractrans_par
   276)    real                                          :: fractrans_nir
   277)    real            , dimension(mzs)              :: abs_sfcw_par
   278)    real            , dimension(mzs)              :: abs_sfcw_nir
   279)    real                                          :: abs_ground_par
   280)    real                                          :: abs_ground_nir
   281)    real                                          :: albedo_ground_par
   282)    real                                          :: albedo_ground_nir
   283)    real                                          :: downward_par_below_beam
   284)    real                                          :: downward_nir_below_beam
   285)    real                                          :: downward_par_below_diffuse
   286)    real                                          :: upward_par_above_diffuse
   287)    real                                          :: downward_nir_below_diffuse
   288)    real                                          :: upward_nir_above_diffuse
   289)    real                                          :: T_surface
   290)    real                                          :: emissivity
   291)    real                                          :: downward_lw_below
   292)    real                                          :: upward_lw_below
   293)    real                                          :: upward_lw_above
   294)    real                                          :: downward_rshort_below_beam
   295)    real                                          :: downward_rshort_below_diffuse
   296)    real                                          :: upward_rshort_above_diffuse
   297)    real                                          :: surface_netabs_longwave
   298)    real                                          :: nir_v_beam
   299)    real                                          :: nir_v_diffuse
   300)    real                                          :: wleaf_vis
   301)    real                                          :: wleaf_nir
   302)    real                                          :: wleaf_tir
   303)    real                                          :: wwood_vis
   304)    real                                          :: wwood_nir
   305)    real                                          :: wwood_tir
   306)    real                                          :: bl_lai_each
   307)    real                                          :: bl_wai_each
   308)    integer                                       :: ibuff
   309)    !----- External function. --------------------------------------------------------------!
   310)    real            , external                    :: sngloff
   311)    !----- Local constants. ----------------------------------------------------------------!
   312)    real(kind=8)    , parameter                   :: tiny_offset = 1.d-20
   313)    !---------------------------------------------------------------------------------------!
   314) 
   315)    !---------------------------------------------------------------------------------------!
   316)    !     Scattering coefficients.  Contrary to ED-2.1, these values are based on the       !
   317)    ! description by by Sellers (1985) and the CLM technical manual, which includes the     !
   318)    ! leaf orientation factor in the backscattering.  This DOES NOT reduce to ED-2.1 case   !
   319)    ! when the leaf orientation is random.                                                  !
   320)    !---------------------------------------------------------------------------------------!
   321) 
   322)    !---------------------------------------------------------------------------------------!
   323)    !     Forward scattering.                                                               !
   324)    !---------------------------------------------------------------------------------------!
   325)    !----- Visible (PAR). ------------------------------------------------------------------!
   326)    leaf_scatter_vis(:) = leaf_reflect_vis(:) + leaf_trans_vis(:)
   327)    wood_scatter_vis(:) = wood_reflect_vis(:) + wood_trans_vis(:)
   328)    !----- Near infrared (NIR). ------------------------------------------------------------!
   329)    leaf_scatter_nir(:) = leaf_reflect_nir(:) + leaf_trans_nir(:)
   330)    wood_scatter_nir(:) = wood_reflect_nir(:) + wood_trans_nir(:)
   331)    !---------------------------------------------------------------------------------------!
   332) 
   333)    !---------------------------------------------------------------------------------------!
   334)    !      Back-scattering coefficients following CLM.                                      !
   335)    !---------------------------------------------------------------------------------------!
   336)    !----- Visible (PAR). ------------------------------------------------------------------!
   337)    leaf_backscatter_vis = ( leaf_scatter_vis                                               &
   338)                           + 2.5d-1 * ( leaf_reflect_vis - leaf_trans_vis   )               &
   339)                           * ( 1.d0 + orient_factor) ** 2 )                                 &
   340)                           / ( 2.d0 * leaf_scatter_vis )
   341)    wood_backscatter_vis = ( wood_scatter_vis                                               &
   342)                           + 2.5d-1                                                         &
   343)                           * ( wood_reflect_vis - wood_trans_vis   )                        &
   344)                           * ( 1.d0 + orient_factor) ** 2 )                                 &
   345)                           / ( 2.d0 * wood_scatter_vis )
   346)    !----- Near infrared (NIR). ------------------------------------------------------------!
   347)    leaf_backscatter_nir = ( leaf_scatter_nir                                               &
   348)                           + 2.5d-1                                                         &
   349)                           * ( leaf_reflect_nir - leaf_trans_nir   )                        &
   350)                           * ( 1.d0 + orient_factor) ** 2 )                                 &
   351)                           / ( 2.d0 * leaf_scatter_nir )
   352)    wood_backscatter_nir = ( wood_scatter_nir                                               &
   353)                           + 2.5d-1                                                         &
   354)                           * ( wood_reflect_nir - wood_trans_nir   )                        &
   355)                           * ( 1.d0 + orient_factor) ** 2 )                                 &
   356)                           / ( 2.d0 * wood_scatter_nir )
   357) 
   358)    !---------------------------------------------------------------------------------------!
   359)    !     Light extinction coefficients.   These are found following CLM technical manual,  !
   360)    ! and the values fall back to ED-2.0 defaults when orient_factor is zero.               !
   361)    !---------------------------------------------------------------------------------------!
   362)    phi1 = 5.d-1 - orient_factor * ( 6.33d-1 + 3.3d-1 * orient_factor )
   363)    phi2 = 8.77d-1 * (1.d0 - 2.d0 * phi1)
   364) 
   365)    !---------------------------------------------------------------------------------------!
   366)    !     Find the average inverse diffuse optical depth per unit leaf and stem area.       !
   367)    ! We follow CLM technical manual, equation 3.4 only when the orientation factor is      !
   368)    ! non-zero.   Otherwise, we make it 1.d0, which is the limit of that equation when      !
   369)    ! phi2 approaches zero.                                                                 !
   370)    !---------------------------------------------------------------------------------------!
   371)    do ipft = 1, n_pft
   372)       if (orient_factor(ipft) == 0.d0) then
   373)          mu_bar(ipft) = 1.d0
   374)       else
   375)          mu_bar(ipft) = ( 1.d0                                                             &
   376)                         - phi1(ipft) * log(1.d0 + phi2(ipft) / phi1(ipft)) / phi2(ipft) )  &
   377)                         / phi2(ipft)
   378)       end if
   379)    end do
   380)    !---------------------------------------------------------------------------------------!
   381) 
   382) 
   383)    !$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(         &
   384)    !$OMP ibuff,cpatch,cohort_count,tuco,tuco_leaf,    &
   385)    !$OMP ico,bl_lai_each,bl_wai_each,nsoil,colour,    &
   386)    !$OMP albedo_soil_par,albedo_soil_nir,             &
   387)    !$OMP albedo_damp_par,albedo_damp_nir,fcpct,       &
   388)    !$OMP rad_sfcw_par,rad_sfcw_nir,albedo_ground_par, &
   389)    !$OMP albedo_ground_nir,abs_sfcw_par,abs_sfcw_nir, &
   390)    !$OMP emissivity,T_surface,ksn,albedo_sfcw_par,    &
   391)    !$OMP albedo_sfcw_nir,k,sfcw_odepth,fractrans_par, &
   392)    !$OMP fractrans_nir,downward_lw_below,             &
   393)    !$OMP upward_lw_below,upward_lw_above,             &
   394)    !$OMP surface_netabs_longwave,                     &
   395)    !$OMP downward_par_below_beam,                     &
   396)    !$OMP downward_par_below_diffuse,                  &
   397)    !$OMP upward_par_above_diffuse,                    &
   398)    !$OMP downward_nir_below_beam,                     &
   399)    !$OMP downward_nir_below_diffuse,                  &
   400)    !$OMP upward_nir_above_diffuse,                    &
   401)    !$OMP ipft,wleaf_vis,wleaf_nir,wleaf_tir,          &
   402)    !$OMP wwood_vis,wwood_nir,wwood_tir,               &
   403)    !$OMP upward_rshort_above_diffuse,                 &
   404)    !$OMP downward_rshort_below_beam,                  &
   405)    !$OMP downward_rshort_below_diffuse,               &
   406)    !$OMP il,nir_v_beam,nir_v_diffuse,                 &
   407)    !$OMP abs_ground_par,abs_ground_nir )
   408) 
   409) 
   410)    !----- Loop over the patches -----------------------------------------------------------!
   411)    do ipa = 1,csite%npatches
   412)       cpatch => csite%patch(ipa)
   413) 
   414)       ibuff = 1
   415)       !$ ibuff = OMP_get_thread_num()+1
   416) 
   417)       !------------------------------------------------------------------------------------!
   418)       !     Initialise the variables.                                                      !
   419)       !------------------------------------------------------------------------------------!
   420)       radscr(ibuff)%pft_array(:)                 = -1
   421)       radscr(ibuff)%leaf_temp_array(:)           = 0.d0
   422)       radscr(ibuff)%wood_temp_array(:)           = 0.d0
   423)       radscr(ibuff)%lai_array(:)                 = 0.d0
   424)       radscr(ibuff)%wai_array(:)                 = 0.d0
   425)       radscr(ibuff)%CA_array(:)                  = 0.d0
   426)       radscr(ibuff)%htop_array(:)                = 0.d0
   427)       radscr(ibuff)%hbot_array(:)                = 0.d0
   428)       radscr(ibuff)%par_level_beam(:)            = 0.d0
   429)       radscr(ibuff)%par_level_diffd(:)           = 0.d0
   430)       radscr(ibuff)%par_level_diffu(:)           = 0.d0
   431)       radscr(ibuff)%light_level_array(:)         = 0.d0
   432)       radscr(ibuff)%light_beam_level_array(:)    = 0.d0
   433)       radscr(ibuff)%light_diff_level_array(:)    = 0.d0
   434)       radscr(ibuff)%par_v_beam_array(:)          = 0.
   435)       radscr(ibuff)%rshort_v_beam_array(:)       = 0.
   436)       radscr(ibuff)%par_v_diffuse_array(:)       = 0.
   437)       radscr(ibuff)%rshort_v_diffuse_array(:)    = 0.
   438)       radscr(ibuff)%lw_v_array(:)                = 0.
   439)       radscr(ibuff)%radprof_array(:,:)           = 0.
   440)       !------------------------------------------------------------------------------------!
   441) 
   442)       !------------------------------------------------------------------------------------!
   443)       !     Cohort_count is the number of cohorts that affect the radiation balance (i.e.  !
   444)       ! those which are flagged as resolvable.                                             !
   445)       !------------------------------------------------------------------------------------!
   446)       cohort_count = 0
   447)       tuco         = 0
   448)       tuco_leaf    = 0
   449)       !------------------------------------------------------------------------------------!
   450) 
   451) 
   452) 
   453)       !------------------------------------------------------------------------------------!
   454)       !     Initialise par_l_ as zero just in case it is night time or if there is no      !
   455)       ! resolvable cohort.                                                                 !
   456)       !------------------------------------------------------------------------------------!
   457)       csite%par_l_beam_max   (ipa) = 0.0
   458)       csite%par_l_diffuse_max(ipa) = 0.0
   459)       csite%par_l_max        (ipa) = 0.0
   460)       !------------------------------------------------------------------------------------!
   461) 
   462) 
   463)       !------------------------------------------------------------------------------------!
   464)       !     Transfer information from linked lists to arrays.  Here we must check          !
   465)       ! whether is running as a true size-and-age structure model, or as big leaf.         !
   466)       !------------------------------------------------------------------------------------!
   467)       select case (ibigleaf)
   468)       case (0)
   469) 
   470)          !---------------------------------------------------------------------------------!
   471)          !     Size- and age-structure.  Each layer in the radiation will correspond to    !
   472)          ! one cohort.  Unusually, here we go from shortest to tallest, as required by the !
   473)          ! radiation schemes.                                                              !
   474)          !---------------------------------------------------------------------------------!
   475)          do ico = cpatch%ncohorts,1,-1
   476) 
   477)             !----- Initialize values. -----------------------------------------------------!
   478)             cpatch%par_l             (ico)    = 0.0
   479)             cpatch%par_l_beam        (ico)    = 0.0
   480)             cpatch%par_l_diffuse     (ico)    = 0.0
   481) 
   482)             cpatch%rshort_l          (ico)    = 0.0
   483)             cpatch%rshort_l_beam     (ico)    = 0.0
   484)             cpatch%rshort_l_diffuse  (ico)    = 0.0
   485) 
   486)             cpatch%rlong_l           (ico)    = 0.0
   487) 
   488)             cpatch%rshort_w          (ico)    = 0.0
   489)             cpatch%rshort_w_beam     (ico)    = 0.0
   490)             cpatch%rshort_w_diffuse  (ico)    = 0.0
   491) 
   492)             cpatch%rlong_w           (ico)    = 0.0
   493) 
   494)             cpatch%light_level       (ico)    = 0.0
   495)             cpatch%light_level_beam  (ico)    = 0.0
   496)             cpatch%light_level_diff  (ico)    = 0.0
   497) 
   498)             cpatch%rad_profile     (:,ico)    = 0.0
   499) 
   500)             if (cpatch%leaf_resolvable(ico) .or. cpatch%wood_resolvable(ico)) then
   501)                !----- This will eventually have the index of the tallest used cohort. -----!
   502)                tuco = ico
   503)                !---------------------------------------------------------------------------!
   504) 
   505)                cohort_count                          = cohort_count + 1
   506)                radscr(ibuff)%pft_array(cohort_count) = cpatch%pft(ico)
   507) 
   508)                !---------------------------------------------------------------------------!
   509)                !     Here we only tell the true LAI if the leaf is resolvable, and the     !
   510)                ! true WAI if the wood is resolvable.  Also, for photosynthesis, we must    !
   511)                ! keep track of the tallest cohort that has leaves (we track the array      !
   512)                ! counters because we will extract the information directly from the        !
   513)                ! arrays.                                                                   !
   514)                !---------------------------------------------------------------------------!
   515)                if (cpatch%leaf_resolvable(ico)) then
   516)                   tuco_leaf                             = cohort_count
   517)                   radscr(ibuff)%lai_array(cohort_count) = dble(cpatch%lai(ico))
   518)                else
   519)                   radscr(ibuff)%lai_array(cohort_count) = 0.d0
   520)                end if
   521)                if (cpatch%wood_resolvable(ico)) then
   522)                   radscr(ibuff)%wai_array(cohort_count) = dble(cpatch%wai(ico))
   523)                else
   524)                   radscr(ibuff)%wai_array(cohort_count) = 0.d0
   525)                end if
   526)                !---------------------------------------------------------------------------!
   527) 
   528)                ! TEST REMOVING THESE ZERO CALLS (REDUNDANT)
   529) 
   530)                radscr(ibuff)%leaf_temp_array(cohort_count) = dble(cpatch%leaf_temp(ico))
   531)                radscr(ibuff)%wood_temp_array(cohort_count) = dble(cpatch%wood_temp(ico))
   532)                radscr(ibuff)%rshort_v_beam_array   (  cohort_count) = 0.0
   533)                radscr(ibuff)%par_v_beam_array      (  cohort_count) = 0.0
   534)                radscr(ibuff)%rshort_v_diffuse_array(  cohort_count) = 0.0
   535)                radscr(ibuff)%par_v_diffuse_array   (  cohort_count) = 0.0
   536)                radscr(ibuff)%lw_v_array            (  cohort_count) = 0.0
   537)                radscr(ibuff)%radprof_array         (:,cohort_count) = 0.0
   538)                radscr(ibuff)%par_level_beam        (  cohort_count) = 0.d0
   539)                radscr(ibuff)%par_level_diffd       (  cohort_count) = 0.d0
   540)                radscr(ibuff)%par_level_diffu       (  cohort_count) = 0.d0
   541)                radscr(ibuff)%light_level_array     (  cohort_count) = 0.d0
   542)                radscr(ibuff)%light_beam_level_array(  cohort_count) = 0.d0
   543)                radscr(ibuff)%light_diff_level_array(  cohort_count) = 0.d0
   544)                !---------------------------------------------------------------------------!
   545) 
   546)                !---------------------------------------------------------------------------!
   547)                !      Decide whether to assume infinite crown, or the crown area allometry !
   548)                ! method as in Dietze and Clark (2008).                                     !
   549)                !---------------------------------------------------------------------------!
   550)                select case (crown_mod)
   551)                case (0)
   552)                   radscr(ibuff)%CA_array(cohort_count) = 1.d0
   553)                case (1)
   554)                   radscr(ibuff)%CA_array(cohort_count) = dble(cpatch%crown_area(ico))
   555)                end select
   556)                !---------------------------------------------------------------------------!
   557)             end if
   558)             !------------------------------------------------------------------------------!
   559)          end do
   560)          !---------------------------------------------------------------------------------!
   561)       case (1)
   562)          !---------------------------------------------------------------------------------!
   563)          !     Big-leaf solver.  We have either 0 (desert) or 1 cohort.  In case of zero,  !
   564)          ! we bypass the entire cohort loop, otherwise we check how many layers we will    !
   565)          ! create.                                                                         !
   566)          !---------------------------------------------------------------------------------!
   567)          if (cpatch%ncohorts > 0) then
   568)             !----- Initialize values. -----------------------------------------------------!
   569)             cpatch%par_l(1)                 = 0.0
   570)             cpatch%par_l_beam(1)            = 0.0
   571)             cpatch%par_l_diffuse(1)         = 0.0
   572)             cpatch%rshort_l(1)              = 0.0
   573)             cpatch%rshort_l_beam(1)         = 0.0
   574)             cpatch%rshort_l_diffuse(1)      = 0.0
   575)             cpatch%rlong_l(1)               = 0.0
   576)             cpatch%rshort_w(1)              = 0.0
   577)             cpatch%rshort_w_beam(1)         = 0.0
   578)             cpatch%rshort_w_diffuse(1)      = 0.0
   579)             cpatch%rlong_w(1)               = 0.0
   580)             cpatch%rad_profile     (:,1)    = 0.0
   581)             cpatch%light_level     (1)      = 0.0
   582)             cpatch%light_level_beam(1)      = 0.0
   583)             cpatch%light_level_diff(1)      = 0.0
   584) 
   585)             !------------------------------------------------------------------------------!
   586)             !     Check whether the cohort is resolvable or not.                           !
   587)             !------------------------------------------------------------------------------!
   588)            if (cpatch%leaf_resolvable(1) .or. cpatch%wood_resolvable(1)) then
   589)                tuco = 1 !---- Dummy variable. ---------------------------------------------!
   590) 
   591)                !---------------------------------------------------------------------------!
   592)                !     Here we only tell the true LAI if the leaf is resolvable, and the     !
   593)                ! true WAI if the wood is resolvable.  Also, for photosynthesis, we must    !
   594)                ! keep track of the tallest cohort that has leaves (we track the array      !
   595)                ! counters because we will extract the information directly from the        !
   596)                ! arrays.                                                                   !
   597)                !---------------------------------------------------------------------------!
   598)                if (cpatch%leaf_resolvable(1) .and. cpatch%wood_resolvable(1)) then
   599)                   cohort_count = ceiling( (cpatch%lai(1) + cpatch%wai(1)) / tai_lyr_max )
   600)                   bl_lai_each  = cpatch%lai(1) / real(cohort_count)
   601)                   bl_wai_each  = cpatch%wai(1) / real(cohort_count)
   602)                   tuco_leaf    = cohort_count
   603)                elseif (cpatch%leaf_resolvable(1)) then
   604)                   cohort_count = ceiling( cpatch%lai(1) / tai_lyr_max )
   605)                   bl_lai_each  = cpatch%lai(1) / real(cohort_count)
   606)                   bl_wai_each  = 0.0
   607)                   tuco_leaf    = cohort_count
   608)                elseif (cpatch%wood_resolvable(1)) then
   609)                   cohort_count = ceiling( cpatch%wai(1) / tai_lyr_max )
   610)                   bl_lai_each  = 0.0
   611)                   bl_wai_each  = cpatch%wai(1) / real(cohort_count)
   612)                end if
   613)                !---------------------------------------------------------------------------!
   614) 
   615) 
   616) 
   617)                !---------------------------------------------------------------------------!
   618)                !     Loop over all layers, and assign equal amounts of LAI and WAI such    !
   619)                ! that they add back to the total amount.  We impose crown model off for    !
   620)                ! big leaf, so CA_array must be always set to 1.                            !
   621)                !---------------------------------------------------------------------------!
   622)                do ico = 1, cohort_count
   623)                   radscr(ibuff)%pft_array                (ico) = cpatch%pft(1)
   624)                   radscr(ibuff)%lai_array                (ico) = dble(bl_lai_each)
   625)                   radscr(ibuff)%wai_array                (ico) = dble(bl_wai_each)
   626)                   radscr(ibuff)%CA_array                 (ico) = 1.d0
   627)                   radscr(ibuff)%leaf_temp_array          (ico) = dble(cpatch%leaf_temp(1))
   628)                   radscr(ibuff)%wood_temp_array          (ico) = dble(cpatch%wood_temp(1))
   629)                   radscr(ibuff)%rshort_v_beam_array      (ico) = 0.0
   630)                   radscr(ibuff)%par_v_beam_array         (ico) = 0.0
   631)                   radscr(ibuff)%rshort_v_diffuse_array   (ico) = 0.0
   632)                   radscr(ibuff)%lw_v_array             (  ico) = 0.0
   633)                   radscr(ibuff)%radprof_array          (:,ico) = 0.0
   634)                   radscr(ibuff)%par_v_diffuse_array      (ico) = 0.0
   635)                   radscr(ibuff)%par_level_beam           (ico) = 0.d0
   636)                   radscr(ibuff)%par_level_diffu          (ico) = 0.d0
   637)                   radscr(ibuff)%par_level_diffd          (ico) = 0.d0
   638)                   radscr(ibuff)%light_level_array        (ico) = 0.d0
   639)                   radscr(ibuff)%light_beam_level_array   (ico) = 0.d0
   640)                   radscr(ibuff)%light_diff_level_array   (ico) = 0.d0
   641)                end do
   642)                !---------------------------------------------------------------------------!
   643)             end if
   644)             !------------------------------------------------------------------------------!
   645)          end if
   646)          !---------------------------------------------------------------------------------!
   647)       end select
   648)       !------------------------------------------------------------------------------------!
   649) 
   650) 
   651)       !------------------------------------------------------------------------------------!
   652)       csite%rshort_s_diffuse(:,ipa) = 0.0
   653)       csite%rshort_s_beam   (:,ipa) = 0.0
   654)       csite%par_s_diffuse   (:,ipa) = 0.0
   655)       csite%par_s_beam      (:,ipa) = 0.0
   656)       !------------------------------------------------------------------------------------!
   657) 
   658)       !------------------------------------------------------------------------------------!
   659)       !     Find the ground albedo as a function of soil water relative moisture of the    !
   660)       ! top layer.                                                                         !
   661)       !------------------------------------------------------------------------------------!
   662)       nsoil  = ntext_soil(mzg)
   663)       colour = ncol_soil
   664)       select case (nsoil)
   665)       case (13)
   666)          !----- Bedrock, use constants soil value for granite. ----------------------------!
   667)          albedo_soil_par = soilcol(colour)%alb_vis_dry
   668)          albedo_soil_nir = soilcol(colour)%alb_nir_dry
   669)          !----- Damp soil, for temporary surface water albedo. ----------------------------!
   670)          albedo_damp_par = albedo_soil_par
   671)          albedo_damp_nir = albedo_damp_nir
   672)       case (12)
   673)          !----- Peat, follow McCumber and Pielke (1981). ----------------------------------!
   674)          fcpct = csite%soil_water(mzg,ipa) / soil(nsoil)%slmsts
   675)          albedo_soil_par = max (0.07, 0.14 * (1.0 - fcpct))
   676)          albedo_soil_nir = albedo_soil_par
   677)          !----- Damp soil, for temporary surface water albedo. ----------------------------!
   678)          albedo_damp_par = 0.14
   679)          albedo_damp_nir = 0.14
   680)       case default
   681)          select case (colour)
   682)          case (21)
   683)             !------------------------------------------------------------------------------!
   684)             !     ED-2.1 soil colour.  Also, we use the ED-2.1 default method to determine !
   685)             ! the albedo.                                                                  !
   686)             !------------------------------------------------------------------------------!
   687)             fcpct           = csite%soil_water(mzg,ipa) / soil(nsoil)%slmsts
   688)             albedo_soil_par = max(0.14,0.31-0.34*fcpct)
   689)             albedo_soil_nir = albedo_soil_par
   690)             !----- Damp soil, for temporary surface water albedo. -------------------------!
   691)             albedo_damp_par = 0.14
   692)             albedo_damp_nir = 0.14
   693)          case default
   694)             !------------------------------------------------------------------------------!
   695)             !      Other soils, we use the soil numbers from CLM-4.  The colour class must !
   696)             ! be given at RAMSIN.  At this point the value is the same for all points, but !
   697)             ! in the future we may read their files if the results are promising.          !
   698)             !------------------------------------------------------------------------------!
   699)             fcpct           = max(0., 0.11 - 0.40 * csite%soil_water(mzg,ipa))
   700)             albedo_soil_par = min(soilcol(colour)%alb_vis_dry                              &
   701)                                  ,soilcol(colour)%alb_vis_wet  + fcpct)
   702)             albedo_soil_nir = min(soilcol(colour)%alb_nir_dry                              &
   703)                                  ,soilcol(colour)%alb_nir_wet  + fcpct)
   704)             !----- Damp soil, for temporary surface water albedo. -------------------------!
   705)             fcpct           = max(0., 0.11 - 0.40 * soil(nsoil)%slmsts)
   706)             albedo_damp_par = min(soilcol(colour)%alb_vis_dry                              &
   707)                                  ,soilcol(colour)%alb_vis_wet  + fcpct)
   708)             albedo_damp_nir = min(soilcol(colour)%alb_nir_dry                              &
   709)                                  ,soilcol(colour)%alb_nir_wet  + fcpct)
   710)             !------------------------------------------------------------------------------!
   711)          end select
   712)          !---------------------------------------------------------------------------------!
   713)       end select
   714)       !------------------------------------------------------------------------------------!
   715) 
   716) 
   717) 
   718) 
   719)       !------------------------------------------------------------------------------------!
   720)       !     Decide what is our surface temperature.  When the soil is exposed, then that   !
   721)       ! is the surface temperature.  Otherwise, we pick the temporary surface water or     !
   722)       ! snow layer.                                                                        !
   723)       !------------------------------------------------------------------------------------!
   724)       rad_sfcw_par      = 1.0
   725)       rad_sfcw_nir      = 1.0
   726)       albedo_ground_par = 1.0
   727)       albedo_ground_nir = 1.0
   728)       abs_sfcw_par      = 0.0
   729)       abs_sfcw_nir      = 0.0
   730)       ksn               = csite%nlev_sfcwater(ipa)
   731)       if (ksn == 0) then
   732)          emissivity = soilcol(colour)%emiss_tir
   733)          T_surface  = csite%soil_tempk(mzg,ipa)
   734)       else
   735)          !---------------------------------------------------------------------------------!
   736)          !      Sfcwater albedo ALS ranges from wet-soil value for all-liquid to typical   !
   737)          ! snow albedo for ice.  In the future, we should consider a more realistic snow   !
   738)          ! albedo model that takes snow age and snow melt into account.                    !
   739)          !                                                                                 !
   740)          !  Potential starting points:                                                     !
   741)          !                                                                                 !
   742)          !  Roesch, A., et al., 2002: Comparison of spectral surface albedos and their     !
   743)          !      impact on the general circulation model simulated surface climate.  J.     !
   744)          !      Geophys. Res.-Atmosph., 107(D14), 4221, 10.1029/2001JD000809.              !
   745)          !                                                                                 !
   746)          !  Oleson, K.W., et al., 2010: Technical description of version 4.0 of the        !
   747)          !      Community Land Model (CLM). NCAR Technical Note NCAR/TN-478+STR.           !
   748)          !                                                                                 !
   749)          !---------------------------------------------------------------------------------!
   750)          albedo_sfcw_par = snow_albedo_vis + csite%sfcwater_fracliq(ksn,ipa)               &
   751)                                            * ( albedo_damp_par - snow_albedo_vis )
   752)          albedo_sfcw_nir = snow_albedo_nir + csite%sfcwater_fracliq(ksn,ipa)               &
   753)                                            * ( albedo_damp_nir - snow_albedo_nir )
   754) 
   755)          !---------------------------------------------------------------------------------!
   756) 
   757) 
   758)          !----- Fraction shortwave absorbed into sfcwater + soil. -------------------------!
   759)          rad_sfcw_par = 1.0 - albedo_sfcw_par
   760)          rad_sfcw_nir = 1.0 - albedo_sfcw_nir
   761)          do k = ksn,1,-1
   762)             !------------------------------------------------------------------------------!
   763)             !      Fractrans is fraction of shortwave entering each sfcwater layer that    !
   764)             ! gets transmitted through that layer.                                         !
   765)             !------------------------------------------------------------------------------!
   766)             sfcw_odepth   = min( lnexp_max,   20.0 * csite%sfcwater_depth(k,ipa)           &
   767)                                                    / csite%snowfac         (ipa) )
   768)             fractrans_par = exp( - sfcw_odepth )
   769)             fractrans_nir = fractrans_par
   770)             !------------------------------------------------------------------------------!
   771) 
   772)             !------------------------------------------------------------------------------!
   773)             !      abs_sfcw_???(k) is fraction of total incident shortwave (at top of top  !
   774)             ! sfcwater layer) that is absorbed in each sfcwater layer.                     !
   775)             !------------------------------------------------------------------------------!
   776)             abs_sfcw_par(k) = rad_sfcw_par * (1.0 - fractrans_par) * csite%snowfac(ipa)
   777)             abs_sfcw_nir(k) = rad_sfcw_nir * (1.0 - fractrans_nir) * csite%snowfac(ipa)
   778)             !------------------------------------------------------------------------------!
   779) 
   780)             !------------------------------------------------------------------------------!
   781)             !      Rad is fraction of total incident shortwave (at top of top sfcwater     !
   782)             ! layer) that remains at bottom of current sfcwater layer.                     !
   783)             !------------------------------------------------------------------------------!
   784)             rad_sfcw_par = rad_sfcw_par * fractrans_par
   785)             rad_sfcw_nir = rad_sfcw_nir * fractrans_nir
   786)             !------------------------------------------------------------------------------!
   787) 
   788)             !------------------------------------------------------------------------------!
   789)             !      Albedo_ground will ultimately be the albedo of the soil+sfcwater.  So   !
   790)             ! subtract out whatever is being absorbed by sfcwater.                         !
   791)             !------------------------------------------------------------------------------!
   792)             albedo_ground_par = albedo_ground_par - abs_sfcw_par(k)
   793)             albedo_ground_nir = albedo_ground_nir - abs_sfcw_nir(k)
   794)             !------------------------------------------------------------------------------!
   795)          end do
   796)          !---------------------------------------------------------------------------------!
   797) 
   798) 
   799)          !----- Long wave parameter if sfcwater exists. -----------------------------------!
   800)          emissivity = snow_emiss_tir            *        csite%snowfac(ipa)                &
   801)                     + soilcol(colour)%emiss_tir * ( 1. - csite%snowfac(ipa) )
   802)          T_surface  = sqrt(sqrt( ( csite%sfcwater_tempk (ksn,ipa)** 4                      &
   803)                                  * snow_emiss_tir            *        csite%snowfac(ipa)   &
   804)                                  + csite%soil_tempk     (mzg,ipa)** 4                      &
   805)                                  * soilcol(colour)%emiss_tir * ( 1. - csite%snowfac(ipa))) &
   806)                                / emissivity ) )
   807)          !---------------------------------------------------------------------------------!
   808)       end if
   809)       !------------------------------------------------------------------------------------!
   810) 
   811) 
   812) 
   813)       !------------------------------------------------------------------------------------!
   814)       !     This is the fraction of below-canopy radiation that is absorbed by the ground. !
   815)       !------------------------------------------------------------------------------------!
   816)       abs_ground_par = (1.0 - albedo_soil_par)                                             &
   817)                      * (1.0 - csite%snowfac(ipa) + csite%snowfac(ipa) * rad_sfcw_par )
   818)       abs_ground_nir = (1.0 - albedo_soil_nir)                                             &
   819)                      * (1.0 - csite%snowfac(ipa) + csite%snowfac(ipa) * rad_sfcw_nir )
   820)       !------------------------------------------------------------------------------------!
   821) 
   822)       !----- Subtract off ground absorption to obtain the soil+sfcwater albedo. -----------!
   823)       albedo_ground_par = albedo_ground_par - abs_ground_par
   824)       albedo_ground_nir = albedo_ground_nir - abs_ground_nir
   825)       !------------------------------------------------------------------------------------!
   826) 
   827)       !------------------------------------------------------------------------------------!
   828)       !     Decide whether to call the radiation or not.  If there is no cohort, we can    !
   829)       ! bypass it entirely.                                                                !
   830)       !------------------------------------------------------------------------------------!
   831)       if (cohort_count > 0) then
   832) 
   833)          !---------------------------------------------------------------------------------!
   834)          !     First, let's solve the long-wave.  Here we must check which model to use,   !
   835)          ! two-stream or multiple scattering.                                              !
   836)          !---------------------------------------------------------------------------------!
   837)          select case (icanrad)
   838)          case (0)
   839)             !------------------------------------------------------------------------------!
   840)             !    Original two-stream model.                                                !
   841)             !------------------------------------------------------------------------------!
   842)             call old_lw_two_stream(emissivity,T_surface,rlong,cohort_count,                &
   843)                                    radscr(ibuff)%pft_array(1:cohort_count),                &
   844)                                    radscr(ibuff)%LAI_array(1:cohort_count),                &
   845)                                    radscr(ibuff)%WAI_array(1:cohort_count),                &
   846)                                    radscr(ibuff)%leaf_temp_array(1:cohort_count),          &
   847)                                    radscr(ibuff)%wood_temp_array(1:cohort_count),          &
   848)                                    radscr(ibuff)%radprof_array(1:n_radprof,1:cohort_count),&
   849)                                    radscr(ibuff)%lw_v_array(1:cohort_count),               &
   850)                                    downward_lw_below,                                      &
   851)                                    upward_lw_below,                                        &
   852)                                    upward_lw_above)
   853)             !------------------------------------------------------------------------------!
   854) 
   855)          case (1)
   856)             !------------------------------------------------------------------------------!
   857)             !      Multiple-scatter model.  Here there is one important difference: we do  !
   858)             ! NOT scale longwave radiation, and we do NOT split longwave radiation coming  !
   859)             ! from the sky and coming from the surface, as they interact in the middle     !
   860)             ! layers.  We save all radiation in the "incid" arrays, and scale them after   !
   861)             ! the solution, so the code called after this step can be used the same way    !
   862)             ! for both radiations.                                                         !
   863)             !------------------------------------------------------------------------------!
   864)             call lw_multiple_scatter(emissivity,T_surface,rlong,cohort_count,              &
   865)                                    radscr(ibuff)%pft_array(1:cohort_count),                &
   866)                                    radscr(ibuff)%LAI_array(1:cohort_count),                &
   867)                                    radscr(ibuff)%WAI_array(1:cohort_count),                &
   868)                                    radscr(ibuff)%CA_array(1:cohort_count),                 &
   869)                                    radscr(ibuff)%leaf_temp_array(1:cohort_count),          &

  1253)                   !------ Visible (PAR), only leaves need this (photsynthesis model). -----!
  1254)                   cpatch%par_l_beam       (1) = cpatch%par_l_beam(1)                       &
  1255)                                               + radscr(ibuff)%par_v_beam_array(il)*wleaf_vis
  1256)                   cpatch%par_l_diffuse    (1) = cpatch%par_l_diffuse(1)                    &
  1257)                                               + radscr(ibuff)%par_v_diffuse_array(il)*wleaf_vis
  1258)                   !------ Total short wave radiation (PAR+NIR). ---------------------------!
  1259)                   cpatch%rshort_l_beam    (1) = cpatch%rshort_l_beam(1)                &
  1260)                                               + radscr(ibuff)%par_v_beam_array(il)*wleaf_vis &
  1261)                                               + nir_v_beam                  * wleaf_nir
  1262)                   cpatch%rshort_l_diffuse (1) = cpatch%rshort_l_diffuse (1)                &
  1263)                                               + radscr(ibuff)%par_v_diffuse_array    (il) * wleaf_vis    &
  1264)                                               + nir_v_diffuse               * wleaf_nir
  1265)                   cpatch%rshort_w_beam    (1) = cpatch%rshort_w_beam    (1)                &
  1266)                                               + radscr(ibuff)%par_v_beam_array       (il) * wwood_vis    &
  1267)                                               + nir_v_beam                  * wwood_nir
  1268)                   cpatch%rshort_w_diffuse (1) = cpatch%rshort_w_diffuse (1)                &
  1269)                                               + radscr(ibuff)%par_v_diffuse_array    (il) * wwood_vis    &
  1270)                                               + nir_v_diffuse               * wwood_nir
  1271)                   !----- Thermal infra-red (long wave). -----------------------------------!
  1272)                   cpatch%rlong_l          (1) = cpatch%rlong_l (1)                         &
  1273)                                               + radscr(ibuff)%lw_v_array    (il) * wleaf_tir
  1274)                   cpatch%rlong_w          (1) = cpatch%rlong_w (1)                         &
  1275)                                               + radscr(ibuff)%lw_v_array    (il) * wwood_tir
  1276)                   !------------------------------------------------------------------------!
  1277)                end do
  1278)                !---------------------------------------------------------------------------!
  1279) 
  1280) 
  1281) 
  1282)                !----- Save the light levels as the median level. --------------------------!
  1283)                il = ceiling(real(cohort_count)/2.0)
  1284)                cpatch%light_level      (1) = sngloff(radscr(ibuff)%light_level_array     (il)            &

  1287)                                                       ,tiny_offset )

  1289)                                                       ,tiny_offset )
  1290)

  1294) 
  1295) 
  1296) 
  1297)                !---------------------------------------------------------------------------!
  1298)             end if
  1299)             !------------------------------------------------------------------------------!
  1300)          end select
  1301)          !---------------------------------------------------------------------------------!
  1302) 
  1303)       else
  1304) 
  1305) 
  1306)          !----- This is the case where there is no vegetation. ----------------------------!
  1307)          downward_par_below_beam         = par_beam_norm
  1308)          downward_par_below_diffuse      = par_diff_norm
  1309)          downward_nir_below_beam         = nir_beam_norm
  1310)          downward_nir_below_diffuse      = nir_diff_norm
  1311)          downward_rshort_below_beam      = par_beam_norm + nir_beam_norm
  1312)          downward_rshort_below_diffuse   = par_diff_norm + nir_diff_norm
  1313) 
  1314)          upward_par_above_diffuse        = albedo_ground_par * par_diff_norm
  1315)          upward_nir_above_diffuse        = albedo_ground_nir * nir_diff_norm
  1316)          upward_rshort_above_diffuse     = upward_par_above_diffuse                        &
  1317)                                          + upward_nir_above_diffuse
  1318) 
  1319)          surface_netabs_longwave         = emissivity * (rlong - stefan * T_surface**4 )
  1320)

  1322)                                          / ( par_beam_norm + par_diff_norm )
  1323)          csite%albedo_nir          (ipa) = upward_nir_above_diffuse                        &
  1324)                                          / ( nir_beam_norm + nir_diff_norm )
  1325)          csite%albedo              (ipa) = upward_rshort_above_diffuse
  1326)          csite%rlongup             (ipa) = rlong - surface_netabs_longwave
  1327)          csite%rlong_albedo        (ipa) = csite%rlongup(ipa) / rlong
  1328) 
  1329)       end if
  1330) 
  1331) 
  1332)       !----- Absorption rate of short wave by the soil. -----------------------------------!
  1333)       csite%rshort_g_beam   (ipa) = downward_par_below_beam    * abs_ground_par            &
  1334)                                   + downward_nir_below_beam    * abs_ground_nir
  1335)       csite%rshort_g_diffuse(ipa) = downward_par_below_diffuse * abs_ground_par            &
  1336)                                   + downward_nir_below_diffuse * abs_ground_nir
  1337)       csite%par_g_beam      (ipa) = downward_par_below_beam    * abs_ground_par
  1338)       csite%par_g_diffuse   (ipa) = downward_par_below_diffuse * abs_ground_par
  1339)       csite%parup           (ipa) = upward_par_above_diffuse
  1340)       csite%nirup           (ipa) = upward_nir_above_diffuse
  1341) 
  1342)       !----- Incident radiation at the ground. --------------------------------------------!
  1343)       csite%par_b_beam      (ipa) = downward_par_below_beam
  1344)       csite%nir_b_beam      (ipa) = downward_nir_below_beam
  1345)       csite%par_b_diffuse   (ipa) = downward_par_below_diffuse
  1346)       csite%nir_b_diffuse   (ipa) = downward_nir_below_diffuse
  1347) 
  1348)       !----- Absorption rate of short wave by the surface water. --------------------------!
  1349)       do k=1,csite%nlev_sfcwater(ipa)
  1350)          csite%rshort_s_beam   (k,ipa) = downward_par_below_beam    * abs_sfcw_par(k)      &
  1351)                                        + downward_nir_below_beam    * abs_sfcw_nir(k)
  1352)          csite%rshort_s_diffuse(k,ipa) = downward_par_below_diffuse * abs_sfcw_par(k)      &
  1353)                                        + downward_nir_below_beam    * abs_sfcw_nir(k)
  1354)          csite%par_s_beam      (k,ipa) = downward_par_below_beam    * abs_sfcw_par(k)
  1355)          csite%par_s_diffuse   (k,ipa) = downward_par_below_diffuse * abs_sfcw_par(k)
  1356)       end do
  1357) 
  1358)       !----- Long wave absorption rate at the surface. ------------------------------------!
  1359)       if (csite%nlev_sfcwater(ipa) == 0) then
  1360)          csite%rlong_s (ipa) = 0.0
  1361)          csite%rlong_g (ipa) = surface_netabs_longwave
  1362)       else
  1363)          csite%rlong_s (ipa) = surface_netabs_longwave
  1364)          csite%rlong_g (ipa) = 0.0
  1365)       end if
  1366)       !------------------------------------------------------------------------------------!
  1367) 
  1368)    end do
  1369)    !---------------------------------------------------------------------------------------!
  1370)    !$OMP END PARALLEL DO
  1371) 
  1372)    return
