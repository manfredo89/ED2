    49)    subroutine compute_budget(csite,lsl,pcpg,qpcpg,ipa,wcurr_loss2atm,ecurr_netrad          &
    50)                             ,ecurr_loss2atm,co2curr_loss2atm,wcurr_loss2drainage           &
    51)                             ,ecurr_loss2drainage,wcurr_loss2runoff,ecurr_loss2runoff       &
    52)                             ,site_area,cbudget_nep,old_can_enthalpy,old_can_shv            &
    53)                             ,old_can_co2,old_can_rhos,old_can_prss)
    54)       use ed_state_vars, only : sitetype           & ! structure
    55)                               , patchtype          ! ! structure
    56)       use ed_max_dims  , only : str_len            ! ! intent(in)
    57)       use ed_misc_coms , only : dtlsm              & ! intent(in)
    58)                               , current_time       ! ! intent(in)
    59)       use ed_max_dims  , only : n_dbh              ! ! intent(in)
    60)       use consts_coms  , only : umol_2_kgC         & ! intent(in)
    61)                               , day_sec            & ! intent(in)
    62)                               , rdry               & ! intent(in)
    63)                               , mmdryi             & ! intent(in)
    64)                               , epim1              ! ! intent(in)
    65)       use rk4_coms     , only : rk4eps             & ! intent(in)
    66)                               , print_budget       & ! intent(in)
    67)                               , budget_pref        & ! intent(in)
    68)                               , checkbudget        ! ! intent(in)
    69)       use therm_lib    , only : tq2enthalpy        ! ! function
    70)       implicit none
    71)       !----- Arguments --------------------------------------------------------------------!
    72)       type(sitetype)                          , target        :: csite
    73)       real                                    , intent(inout) :: pcpg
    74)       real                                    , intent(inout) :: qpcpg
    75)       real                                    , intent(inout) :: co2curr_loss2atm
    76)       real                                    , intent(inout) :: ecurr_netrad
    77)       real                                    , intent(inout) :: ecurr_loss2atm
    78)       real                                    , intent(inout) :: ecurr_loss2drainage
    79)       real                                    , intent(inout) :: ecurr_loss2runoff
    80)       real                                    , intent(inout) :: wcurr_loss2atm
    81)       real                                    , intent(inout) :: wcurr_loss2drainage
    82)       real                                    , intent(inout) :: wcurr_loss2runoff
    83)       integer                                 , intent(in)    :: lsl
    84)       integer                                 , intent(in)    :: ipa
    85)       real                                    , intent(in)    :: site_area
    86)       real                                    , intent(inout) :: cbudget_nep
    87)       real                                    , intent(in)    :: old_can_enthalpy
    88)       real                                    , intent(in)    :: old_can_shv
    89)       real                                    , intent(in)    :: old_can_co2
    90)       real                                    , intent(in)    :: old_can_rhos
    91)       real                                    , intent(in)    :: old_can_prss
    92)       !----- Local variables --------------------------------------------------------------!
    93)       type(patchtype)                         , pointer       :: cpatch
    94)       character(len=str_len)                                  :: budget_fout
    95)       real                                                    :: co2budget_finalstorage
    96)       real                                                    :: co2budget_deltastorage
    97)       real                                                    :: co2curr_gpp
    98)       real                                                    :: co2curr_leafresp
    99)       real                                                    :: co2curr_rootresp
   100)       real                                                    :: co2curr_leafgrowthresp
   101)       real                                                    :: co2curr_rootgrowthresp
   102)       real                                                    :: co2curr_sapagrowthresp
   103)       real                                                    :: co2curr_sapbgrowthresp
   104)       real                                                    :: co2curr_leafstorageresp
   105)       real                                                    :: co2curr_rootstorageresp
   106)       real                                                    :: co2curr_sapastorageresp
   107)       real                                                    :: co2curr_sapbstorageresp
   108)       real                                                    :: co2curr_hetresp
   109)       real                                                    :: co2curr_nep
   110)       real                                                    :: co2curr_denseffect
   111)       real                                                    :: co2curr_residual
   112)       real                                                    :: ebudget_finalstorage
   113)       real                                                    :: ebudget_deltastorage
   114)       real                                                    :: ecurr_precipgain
   115)       real                                                    :: ecurr_denseffect
   116)       real                                                    :: ecurr_prsseffect
   117)       real                                                    :: ecurr_residual
   118)       real                                                    :: wbudget_finalstorage
   119)       real                                                    :: wbudget_deltastorage
   120)       real                                                    :: wcurr_precipgain
   121)       real                                                    :: wcurr_denseffect
   122)       real                                                    :: wcurr_residual
   123)       real                                                    :: curr_can_enthalpy
   124)       real                                                    :: gpp
   125)       real                                                    :: leaf_resp
   126)       real                                                    :: root_resp
   127)       real                                                    :: leaf_growth_resp
   128)       real                                                    :: root_growth_resp
   129)       real                                                    :: sapa_growth_resp
   130)       real                                                    :: sapb_growth_resp
   131)       real                                                    :: leaf_storage_resp
   132)       real                                                    :: root_storage_resp
   133)       real                                                    :: sapa_storage_resp
   134)       real                                                    :: sapb_storage_resp
   135)       real                                                    :: co2_factor
   136)       real                                                    :: ene_factor
   137)       real                                                    :: h2o_factor
   138)       real                                                    :: patch_lai
   139)       real                                                    :: patch_wai
   140)       integer                                                 :: jpa
   141)       integer                                                 :: ico
   142)       logical                                                 :: isthere
   143)       logical                                                 :: co2_ok
   144)       logical                                                 :: energy_ok
   145)       logical                                                 :: water_ok
   146)       !----- Local constants. -------------------------------------------------------------!
   147)       character(len=13)     , parameter     :: fmtf='(a,1x,es14.7)'
   148)       character(len=10)     , parameter     :: bhfmt='(31(a,1x))'
   149)       character(len=48)     , parameter     :: bbfmt='(3(i13,1x),28(es13.6,1x))'
   150)       !----- Locally saved variables. -----------------------------------------------------!
   151)       logical               , save          :: first_time = .true.
   152)       !------------------------------------------------------------------------------------!
   153) 
   154) 
   155) 
   156)       !------------------------------------------------------------------------------------!
   157)       !      If this is the first time, we initialise all files with their headers.        !
   158)       !------------------------------------------------------------------------------------!
   159)       if (first_time) then
   160)          do jpa = 1, csite%npatches
   161)             write(budget_fout,fmt='(2a,i4.4,a)') trim(budget_pref),'patch_',jpa,'.txt'
   162)             inquire(file=trim(budget_fout),exist=isthere)
   163)             if (isthere) then
   164)                !---- Open the file to delete when closing. --------------------------------!
   165)                open (unit=86,file=trim(budget_fout),status='old',action='write')
   166)                close(unit=86,status='delete')
   167)             end if
   168)             !------------------------------------------------------------------------------!
   169) 
   170)             if (print_budget) then
   171)                !---------------------------------------------------------------------------!
   172)                open (unit=86,file=trim(budget_fout),status='replace',action='write')
   173)                write(unit=86,fmt=bhfmt)   '         YEAR' , '        MONTH'                &
   174)                                         , '          DAY' , '         TIME'                &
   175)                                         , '          LAI' , '          WAI'                &
   176)                                         , '       HEIGHT' , '  CO2.STORAGE'                &
   177)                                         , ' CO2.RESIDUAL' , ' CO2.DSTORAGE'                &
   178)                                         , '      CO2.NEP' , ' CO2.DENS.EFF'                &
   179)                                         , ' CO2.LOSS2ATM' , '  ENE.STORAGE'                &
   180)                                         , ' ENE.RESIDUAL' , ' ENE.DSTORAGE'                &
   181)                                         , '   ENE.PRECIP' , '   ENE.NETRAD'                &
   182)                                         , ' ENE.DENS.EFF' , ' ENE.PRSS.EFF'                &
   183)                                         , ' ENE.LOSS2ATM' , ' ENE.DRAINAGE'                &
   184)                                         , '   ENE.RUNOFF' , '  H2O.STORAGE'                &
   185)                                         , ' H2O.RESIDUAL' , ' H2O.DSTORAGE'                &
   186)                                         , '   H2O.PRECIP' , ' H2O.DENS.EFF'                &
   187)                                         , ' H2O.LOSS2ATM' , ' H2O.DRAINAGE'                &
   188)                                         , '   H2O.RUNOFF'
   189)                                         
   190)                close(unit=86,status='keep')
   191)             end if
   192)          end do
   193)          first_time = .false.
   194)       end if
   195)       !------------------------------------------------------------------------------------!
   196) 
   197) 
   198) 
   199)       !------------------------------------------------------------------------------------!
   200)       !     Compute gain in water and energy due to precipitation.                         !
   201)       !------------------------------------------------------------------------------------!
   202)       wcurr_precipgain = pcpg  * dtlsm
   203)       ecurr_precipgain = qpcpg * dtlsm
   204)       !------------------------------------------------------------------------------------!
   205) 
   206) 
   207)       !------------------------------------------------------------------------------------!
   208)       !     Compute the density and pressure effects.  We seek the conservation of the     !
   209)       ! extensive properties [X/m2], but the canopy air space solves the intensive         !
   210)       ! quantities instead [X/mol or X/kg].  Because density is not constant within the    !
   211)       ! time step, and during the integration we solve the intensive form for the canopy   !
   212)       ! air space, we must subtract the density effect from the residual.  The derivation  !
   213)       ! is shown below.  The storage term is what we aim at, but in reality we solve the   !
   214)       ! equations after the >>>.                                                           !
   215)       !                                                                                    !
   216)       !  dM              d(rho*z*m)                         dm                     d(rho)  !
   217)       ! ---- = I - L -> ------------ = I - L >>> rho * z * ---- = I - L - m * z * -------- !
   218)       !  dt                  dt                             dt                       dt    !
   219)       !                                                                                    !
   220)       ! where M is the extensive propery, I is the input flux, L is the loss flux, z is    !
   221)       ! the canopy air space depth, and rho is the canopy air space density.               !
   222)       !    For the specific case of enthalpy, we also compute the pressure effect between  !
   223)       ! time steps.  We cannot guarantee conservation of enthalpy when we update pressure, !
   224)       ! because of the first law of thermodynamics (the way to address this would be to    !
   225)       ! use equivalent potential temperature, which is enthalpy plus pressure effect).     !
   226)       ! Enthalpy is preserved within one time step, once pressure is updated and remains   !
   227)       ! constant.                                                                          !
   228)       !                                                                                    !
   229)       !   dH         dp                     dh             dp             d(rho)           !
   230)       !  ---- - V * ---- = Q >>> rho * z * ---- = Q + z * ---- - m * z * --------          !
   231)       !   dt         dt                     dt             dt               dt             !
   232)       !                                                                                    !
   233)       ! where p is the canopy air space pressure, Q is the net heat exchange, V is the     !
   234)       ! volume of the canopy air space.                                                    !
   235)       !------------------------------------------------------------------------------------!
   236)       !------ CO2.  Density effect only. --------------------------------------------------!
   237)       co2curr_denseffect  = ddens_dt_effect(old_can_rhos,csite%can_rhos(ipa)               &
   238)                                            ,old_can_co2,csite%can_co2(ipa)                 &
   239)                                            ,csite%can_depth(ipa),mmdryi)
   240)       !------ Water. Density effect only. -------------------------------------------------!
   241)       wcurr_denseffect    = ddens_dt_effect(old_can_rhos,csite%can_rhos(ipa)               &
   242)                                            ,old_can_shv,csite%can_shv(ipa)                 &
   243)                                            ,csite%can_depth(ipa),1.)
   244)       !------ Enthalpy.  Density and pressure effects. ------------------------------------!
   245)       curr_can_enthalpy   = tq2enthalpy(csite%can_temp(ipa),csite%can_shv(ipa),.true.)
   246)       ecurr_denseffect    = ddens_dt_effect(old_can_rhos,csite%can_rhos(ipa)               &
   247)                                            ,old_can_enthalpy, curr_can_enthalpy            &
   248)                                            ,csite%can_depth(ipa),1.0)
   249)       ecurr_prsseffect    = csite%can_depth(ipa) * (csite%can_prss(ipa) - old_can_prss)
   250)       !------------------------------------------------------------------------------------!
   251) 
   252) 
   253)       !------------------------------------------------------------------------------------!
   254)       !     Compute the carbon flux components.                                            !
   255)       !------------------------------------------------------------------------------------!
   256)       call sum_plant_cfluxes(csite,ipa,gpp,leaf_resp,root_resp,leaf_growth_resp            &
   257)                             ,root_growth_resp,sapa_growth_resp,sapb_growth_resp            &
   258)                             ,leaf_storage_resp,root_storage_resp,sapa_storage_resp         &
   259)                             ,sapb_storage_resp)
   260)       co2curr_gpp             = gpp              * dtlsm
   261)       co2curr_leafresp        = leaf_resp        * dtlsm
   262)       co2curr_rootresp        = root_resp        * dtlsm
   263)       co2curr_leafstorageresp = leaf_storage_resp* dtlsm
   264)       co2curr_rootstorageresp = root_storage_resp* dtlsm
   265)       co2curr_sapastorageresp = sapa_storage_resp* dtlsm
   266)       co2curr_sapbstorageresp = sapb_storage_resp* dtlsm
   267)       co2curr_hetresp         = csite%rh(ipa)    * dtlsm
   268)       co2curr_leafgrowthresp  = leaf_growth_resp * dtlsm
   269)       co2curr_rootgrowthresp  = root_growth_resp * dtlsm
   270)       co2curr_sapagrowthresp  = sapa_growth_resp * dtlsm
   271)       co2curr_sapbgrowthresp  = sapb_growth_resp * dtlsm
   272)       
   273)       co2curr_nep         = co2curr_gpp - co2curr_leafresp - co2curr_rootresp              &
   274)                           - co2curr_leafgrowthresp  - co2curr_rootgrowthresp               &
   275)                           - co2curr_sapagrowthresp  - co2curr_sapbgrowthresp               &
   276)                           - co2curr_leafstorageresp - co2curr_rootstorageresp              &
   277)                           - co2curr_sapastorageresp - co2curr_sapbstorageresp              &
   278)                           - co2curr_hetresp
   279)       cbudget_nep         = cbudget_nep + site_area * csite%area(ipa) * co2curr_nep        &
   280)                                         * umol_2_kgC
   281) 
   282) 
   283)       !----- Compute current storage terms. -----------------------------------------------!
   284)       co2budget_finalstorage = compute_co2_storage(csite,ipa)
   285)       wbudget_finalstorage   = compute_water_storage(csite,lsl,ipa)
   286)       ebudget_finalstorage   = compute_energy_storage(csite,lsl,ipa)
   287) 
   288)       !----- Compute the change in storage. -----------------------------------------------!
   289)       co2budget_deltastorage = co2budget_finalstorage - csite%co2budget_initialstorage(ipa)
   290)       wbudget_deltastorage   = wbudget_finalstorage   - csite%wbudget_initialstorage(ipa)
   291)       ebudget_deltastorage   = ebudget_finalstorage   - csite%ebudget_initialstorage(ipa)
   292)       !------------------------------------------------------------------------------------!
   293)       !     Compute residuals.                                                             !
   294)       !------------------------------------------------------------------------------------!
   295)       !----- 1. Canopy CO2. ---------------------------------------------------------------!
   296)       co2curr_residual = co2budget_deltastorage                                            &
   297)                        - ( - co2curr_nep       - co2curr_loss2atm  )                       &
   298)                        - co2curr_denseffect
   299)       !----- 2. Energy. -------------------------------------------------------------------!
   300)       ecurr_residual   = ebudget_deltastorage                                              &
   301)                        - ( ecurr_precipgain    - ecurr_loss2atm  - ecurr_loss2drainage     &
   302)                          - ecurr_loss2runoff   + ecurr_netrad    + ecurr_prsseffect    )   &
   303)                        - ecurr_denseffect
   304)       !----- 3. Water. --------------------------------------------------------------------!
   305)       wcurr_residual   = wbudget_deltastorage                                              &
   306)                        - ( wcurr_precipgain    - wcurr_loss2atm                            &
   307)                          - wcurr_loss2drainage - wcurr_loss2runoff )                       &
   308)                        - wcurr_denseffect
   309)       !------------------------------------------------------------------------------------!
   310) 
   311) 
   312)       !------------------------------------------------------------------------------------!
   313)       !     Integrate residuals.                                                           !
   314)       !------------------------------------------------------------------------------------!
   315)       !----- 1. Canopy CO2. ---------------------------------------------------------------!
   316)       csite%co2budget_residual(ipa) = csite%co2budget_residual(ipa)  + co2curr_residual
   317)       !----- 2. Energy. -------------------------------------------------------------------!
   318)       csite%ebudget_residual(ipa) = csite%ebudget_residual(ipa)      + ecurr_residual
   319)       !----- 3. Water. --------------------------------------------------------------------!
   320)       csite%wbudget_residual(ipa) = csite%wbudget_residual(ipa)      + wcurr_residual
   321)       !------------------------------------------------------------------------------------!
   322) 
   323)       !------------------------------------------------------------------------------------!
   324)       !    Integrate the terms that are part of the budget.                                !
   325)       !------------------------------------------------------------------------------------!
   326)       !----- 1. Carbon dioxide. -----------------------------------------------------------!
   327)       csite%co2budget_gpp(ipa)         = csite%co2budget_gpp(ipa)       + gpp       *dtlsm
   328)       csite%co2budget_plresp(ipa)      = csite%co2budget_plresp(ipa)                       &
   329)                                        + ( leaf_resp + root_resp + leaf_growth_resp        &
   330)                                          + root_growth_resp  + sapa_growth_resp            &
   331)                                          + sapb_growth_resp  + leaf_storage_resp           &
   332)                                          + root_storage_resp + sapa_storage_resp           &
   333)                                          + sapb_storage_resp ) * dtlsm
   334)       csite%co2budget_rh(ipa)          = csite%co2budget_rh(ipa)                           &
   335)                                        + csite%rh(ipa) * dtlsm
   336)       csite%co2budget_denseffect(ipa)  = csite%co2budget_denseffect(ipa)                   &
   337)                                        + co2curr_denseffect
   338)       csite%co2budget_loss2atm(ipa)    = csite%co2budget_loss2atm(ipa)                     &
   339)                                        + co2curr_loss2atm
   340)       !----- 2. Energy. -------------------------------------------------------------------!
   341)       csite%ebudget_precipgain(ipa)    = csite%ebudget_precipgain(ipa)   + ecurr_precipgain
   342)       csite%ebudget_netrad(ipa)        = csite%ebudget_netrad(ipa)       + ecurr_netrad
   343)       csite%ebudget_prsseffect(ipa)    = csite%ebudget_prsseffect(ipa)   + ecurr_prsseffect
   344)       csite%ebudget_denseffect(ipa)    = csite%ebudget_denseffect(ipa)   + ecurr_denseffect
   345)       csite%ebudget_loss2atm(ipa)      = csite%ebudget_loss2atm(ipa)     + ecurr_loss2atm
   346)       csite%ebudget_loss2drainage(ipa) = csite%ebudget_loss2drainage(ipa)                  &
   347)                                        + ecurr_loss2drainage
   348)       csite%ebudget_loss2runoff(ipa)   = csite%ebudget_loss2runoff(ipa)                    &
   349)                                        + ecurr_loss2runoff
   350)       !----- 3. Water. --------------------------------------------------------------------!
   351)       csite%wbudget_precipgain(ipa)    = csite%wbudget_precipgain(ipa) + wcurr_precipgain
   352)       csite%wbudget_denseffect(ipa)    = csite%wbudget_denseffect(ipa) + wcurr_denseffect
   353)       csite%wbudget_loss2atm(ipa)      = csite%wbudget_loss2atm(ipa)   + wcurr_loss2atm
   354)       csite%wbudget_loss2drainage(ipa) = csite%wbudget_loss2drainage(ipa)                  &
   355)                                        + wcurr_loss2drainage
   356)       csite%wbudget_loss2runoff(ipa)   = csite%wbudget_loss2runoff(ipa)                    &
   357)                                        + wcurr_loss2runoff
   358)       !------------------------------------------------------------------------------------!
   359) 
   360) 
   361)       !------------------------------------------------------------------------------------!
   362)       !     Update density and initial storage for next step.                              !
   363)       !------------------------------------------------------------------------------------!
   364)       csite%wbudget_initialstorage(ipa)   = wbudget_finalstorage
   365)       csite%ebudget_initialstorage(ipa)   = ebudget_finalstorage
   366)       csite%co2budget_initialstorage(ipa) = co2budget_finalstorage
   367) 
   368) 
   369)       !------------------------------------------------------------------------------------!
   370)       !    If the "check budget" option is activated (you can turn on and turn off by set- !
   371)       ! ting checkbudget in ed_params.f90), then the model will crash whenever there is    !
   372)       ! some significant leak of CO2, water, or energy.                                    !
   373)       !------------------------------------------------------------------------------------!
   374)       if (checkbudget) then
   375)          co2_ok  = abs(co2curr_residual) <= rk4eps * ( abs(co2budget_finalstorage)         &
   376)                                                      + abs(co2budget_deltastorage) )
   377)          energy_ok = abs(ecurr_residual) <= rk4eps * ( abs(ebudget_finalstorage)           &
   378)                                                      + abs(ebudget_deltastorage)   )
   379)          water_ok  = abs(wcurr_residual) <= rk4eps * ( abs(wbudget_finalstorage)           &
   380)                                                      + abs(wbudget_deltastorage)   )
   381) 
   382) 
   383)          !---------------------------------------------------------------------------------!
   384)          !     Find the patch LAI and WAI for output, only if it is needed.                !
   385)          !---------------------------------------------------------------------------------!
   386)          if (.not. (co2_ok .and. energy_ok .and. water_ok) .or. print_budget) then
   387)             cpatch => csite%patch(ipa)
   388)             patch_lai = 0.0
   389)             patch_wai = 0.0
   390)             do ico=1,cpatch%ncohorts
   391)                patch_lai = patch_lai + cpatch%lai(ico)
   392)                patch_wai = patch_wai + cpatch%wai(ico)
   393)             end do
   394)          end if
   395)          !---------------------------------------------------------------------------------!
   396) 
   397) 
   398)          if (.not. co2_ok) then
   399)             write (unit=*,fmt='(a)') '|--------------------------------------------------|'
   400)             write (unit=*,fmt='(a)') '|           !!! ): CO2 budget failed :( !!!        |'
   401)             write (unit=*,fmt='(a)') '|--------------------------------------------------|'
   402)             write (unit=*,fmt='(a,i4.4,2(1x,i2.2),1x,f6.0)') ' TIME           : ',         &
   403)                current_time%year,current_time%month,current_time%date ,current_time%time
   404)             write (unit=*,fmt=fmtf ) ' LAI              : ',patch_lai
   405)             write (unit=*,fmt=fmtf ) ' VEG_HEIGHT       : ',csite%veg_height(ipa)
   406)             write (unit=*,fmt=fmtf ) ' CAN_RHOS         : ',csite%can_rhos(ipa)
   407)             write (unit=*,fmt=fmtf ) ' OLD_CAN_RHOS     : ',old_can_rhos
   408)             write (unit=*,fmt=fmtf ) ' CAN_DEPTH        : ',csite%can_depth(ipa)

   560)    !   This function computes the total water stored in the system, in kg/m2.              !
   561)    !   (soil + temporary pools + canopy air space + leaf surface).                         !
   562)    !---------------------------------------------------------------------------------------!
   563)    real function compute_water_storage(csite, lsl,ipa)
   564)       use ed_state_vars , only  : sitetype              & ! structure
   565)                                 , patchtype             ! ! structure
   566)       use grid_coms      , only : nzg                   ! ! intent(in)
   567)       use soil_coms      , only : dslz                  ! ! intent(in)
   568)       use consts_coms    , only : wdns                  ! ! intent(in)
   569)       implicit none
   570)       !----- Arguments --------------------------------------------------------------------!
   571)       type(sitetype) , target     :: csite
   572)       integer        , intent(in) :: ipa
   573)       integer        , intent(in) :: lsl
   574)       !----- Local variables --------------------------------------------------------------!
   575)       type(patchtype), pointer    :: cpatch
   576)       integer                     :: k
   577)       integer                     :: ico
   578)       !------------------------------------------------------------------------------------!
   579) 
   580) 
   581)       compute_water_storage = 0.0
   582)       cpatch => csite%patch(ipa)
   583) 
   584)       !----- 1. Add the water stored in the soil. -----------------------------------------!
   585)       do k = lsl, nzg
   586)          compute_water_storage = compute_water_storage                                     &
   587)                                + csite%soil_water(k,ipa) * dslz(k) * wdns
   588)       end do
   589)       !----- 2. Add the water stored in the temporary surface water/snow. -----------------!
   590)       do k = 1, csite%nlev_sfcwater(ipa)
   591)          compute_water_storage = compute_water_storage + csite%sfcwater_mass(k,ipa)
   592)       end do
   593)       !----- 3. Add the water vapour floating in the canopy air space. --------------------!
   594)       compute_water_storage = compute_water_storage                                        &
   595)                             + csite%can_shv(ipa) * csite%can_depth(ipa)                    &
   596)                             * csite%can_rhos(ipa)
   597)       !----- 4. Add the water on the leaf and wood surfaces. ------------------------------!
   598)       do ico = 1,cpatch%ncohorts
   599)          compute_water_storage = compute_water_storage + cpatch%leaf_water(ico)
   600)          compute_water_storage = compute_water_storage + cpatch%wood_water(ico)
   601)       end do
   602) 
   603)       return
   604)    end function compute_water_storage

   615)    !    This function computs the total net radiation, by adding the radiation that        !
   616)    ! interacts with the different surfaces.                                                !
   617)    !---------------------------------------------------------------------------------------!
   618)    real function compute_netrad(csite,ipa)
   619)       use ed_state_vars , only : sitetype  & ! structure
   620)                                , patchtype ! ! structure
   621)       implicit none
   622)       !----- Arguments --------------------------------------------------------------------!
   623)       type(sitetype) , target     :: csite
   624)       integer        , intent(in) :: ipa
   625)       !----- Local variables --------------------------------------------------------------!
   626)       type(patchtype), pointer    :: cpatch
   627)       integer                     :: k
   628)       integer                     :: ico
   629)       !------------------------------------------------------------------------------------!
   630) 
   631)       cpatch => csite%patch(ipa)
   632) 
   633)       compute_netrad = 0.0
   634)       !----- 1. Add the ground components -------------------------------------------------!
   635)       compute_netrad = csite%rshort_g(ipa) + csite%rlong_g(ipa) + csite%rlong_s(ipa)
   636)       !----- 2. Add the shortwave radiation that reaches each snow/water layer ------------!
   637)       do k = 1, csite%nlev_sfcwater(ipa)
   638)          compute_netrad = compute_netrad + csite%rshort_s(k,ipa)
   639)       end do
   640)       !----- 3. Add the radiation components that is absorbed by leaves and branches. -----!
   641)       do ico = 1,cpatch%ncohorts
   642)          compute_netrad = compute_netrad + cpatch%rshort_l(ico) + cpatch%rlong_l(ico)
   643)          compute_netrad = compute_netrad + cpatch%rshort_w(ico) + cpatch%rlong_w(ico)
   644)       end do
   645)       return
   646)    end function compute_netrad

   655)    !=======================================================================================!
   656)    !=======================================================================================!
   657)    !    This function computs the total net radiation, by adding the radiation that        !
   658)    ! interacts with the different surfaces.  The result is given in J/m2.                  !
   659)    !---------------------------------------------------------------------------------------!
   660)    real function compute_energy_storage(csite, lsl, ipa)
   661)       use ed_state_vars        , only : sitetype              & ! structure
   662)                                       , patchtype             ! ! structure
   663)       use grid_coms            , only : nzg                   ! ! intent(in)
   664)       use soil_coms            , only : dslz                  ! ! intent(in)
   665)       use therm_lib            , only : tq2enthalpy           ! ! function
   666)       implicit none
   667)       !----- Arguments --------------------------------------------------------------------!
   668)       type(sitetype) , target     :: csite
   669)       integer        , intent(in) :: ipa
   670)       integer        , intent(in) :: lsl
   671)       !----- Local variables --------------------------------------------------------------!
   672)       type(patchtype), pointer    :: cpatch
   673)       integer                     :: k
   674)       integer                     :: ico
   675)       real                        :: soil_storage
   676)       real                        :: sfcwater_storage
   677)       real                        :: cas_storage
   678)       real                        :: veg_storage
   679)       real                        :: can_enthalpy
   680)       !------------------------------------------------------------------------------------!
   681) 
   682) 
   683)       cpatch => csite%patch(ipa)
   684) 
   685)       !----- 1. Computing internal energy stored at the soil. -----------------------------!
   686)       soil_storage = 0.0
   687)       do k = lsl, nzg
   688)          soil_storage = soil_storage + csite%soil_energy(k,ipa) * dslz(k)
   689)       end do
   690)       !------------------------------------------------------------------------------------!
   691) 
   692) 
   693)       !------------------------------------------------------------------------------------!
   694)       !   2. Computing internal energy stored at the temporary snow/water sfc. layer.      !
   695)       !      Converting it to J/m2. 
   696)       !------------------------------------------------------------------------------------!
   697)       sfcwater_storage = 0.0
   698)       do k = 1, csite%nlev_sfcwater(ipa)
   699)          sfcwater_storage = sfcwater_storage                                               &
   700)                           + csite%sfcwater_energy(k,ipa) * csite%sfcwater_mass(k,ipa)
   701)       end do
   702)       !------------------------------------------------------------------------------------!
   703) 
   704) 
   705)       !------------------------------------------------------------------------------------!
   706)       ! 3. Find and value for canopy air total enthalpy .                                  !
   707)       !------------------------------------------------------------------------------------!
   708)       can_enthalpy = tq2enthalpy(csite%can_temp(ipa),csite%can_shv(ipa),.true.)
   709)       cas_storage  = csite%can_rhos(ipa) * csite%can_depth(ipa) * can_enthalpy
   710)       !------------------------------------------------------------------------------------!
   711) 
   712) 
   713)       !------------------------------------------------------------------------------------!
   714)       ! 4. Compute the internal energy stored in the plants.                               !
   715)       !------------------------------------------------------------------------------------!
   716)       veg_storage = 0.0
   717)       do ico = 1,cpatch%ncohorts
   718)          veg_storage = veg_storage + cpatch%leaf_energy(ico) + cpatch%wood_energy(ico)
   719)       end do
   720)       !------------------------------------------------------------------------------------!
   721) 
   722) 
   723)       !----- 5. Integrating the total energy in ED. ---------------------------------------!
   724)       compute_energy_storage = soil_storage + sfcwater_storage + cas_storage + veg_storage
   725)       !------------------------------------------------------------------------------------!
   726) 
   727)       return
   728)    end function compute_energy_storage

   739)    !    This subroutine computes the carbon flux terms.                                    !
   740)    !---------------------------------------------------------------------------------------!
   741)    subroutine sum_plant_cfluxes(csite,ipa, gpp, leaf_resp,root_resp,leaf_growth_resp       &
   742)                                ,root_growth_resp,sapa_growth_resp,sapb_growth_resp         &
   743)                                ,leaf_storage_resp,root_storage_resp,sapa_storage_resp      &
   744)                                ,sapb_storage_resp)
   745)       use ed_state_vars        , only : sitetype    & ! structure
   746)                                       , patchtype   ! ! structure
   747)       use consts_coms          , only : day_sec     & ! intent(in)
   748)                                       , umol_2_kgC  ! ! intent(in)
   749)       use ed_max_dims          , only : n_dbh       ! ! intent(in)
   750)       implicit none
   751)       !----- Arguments --------------------------------------------------------------------!
   752)       type(sitetype)        , target      :: csite
   753)       integer               , intent(in)  :: ipa
   754)       real                  , intent(out) :: gpp
   755)       real                  , intent(out) :: leaf_resp
   756)       real                  , intent(out) :: root_resp
   757)       real                  , intent(out) :: leaf_growth_resp
   758)       real                  , intent(out) :: root_growth_resp
   759)       real                  , intent(out) :: sapa_growth_resp
   760)       real                  , intent(out) :: sapb_growth_resp
   761)       real                  , intent(out) :: leaf_storage_resp
   762)       real                  , intent(out) :: root_storage_resp
   763)       real                  , intent(out) :: sapa_storage_resp
   764)       real                  , intent(out) :: sapb_storage_resp
   765)       !----- Local variables --------------------------------------------------------------!
   766)       type(patchtype), pointer            :: cpatch
   767)       integer                             :: ico
   768)       !------------------------------------------------------------------------------------!
   769) 
   770)       !----- Initializing some variables. -------------------------------------------------!
   771)       gpp          = 0.0
   772)       leaf_resp    = 0.0
   773)       root_resp    = 0.0
   774)       leaf_storage_resp = 0.0
   775)       root_storage_resp = 0.0
   776)       sapa_storage_resp = 0.0
   777)       sapb_storage_resp = 0.0
   778)       leaf_growth_resp  = 0.0
   779)       root_growth_resp  = 0.0
   780)       sapa_growth_resp  = 0.0
   781)       sapb_growth_resp  = 0.0
   782)       cpatch => csite%patch(ipa)
   783) 
   784)       !------------------------------------------------------------------------------------!
   785)       !     Looping over cohorts.                                                          !
   786)       !------------------------------------------------------------------------------------!
   787)       do ico = 1,cpatch%ncohorts
   788)          !----- Add GPP and leaf respiration only for those cohorts with enough leaves. ---!
   789)          if (cpatch%leaf_resolvable(ico)) then
   790)             gpp       = gpp       + cpatch%gpp(ico)
   791)             leaf_resp = leaf_resp + cpatch%leaf_respiration(ico)
   792) 
   793)          end if
   794)          !----- Root respiration happens even when the LAI is tiny ------------------------!
   795)          root_resp = root_resp + cpatch%root_respiration(ico)
   796) 
   797)          !---------------------------------------------------------------------------------!
   798)          !      Structural terms are "intensive", we must convert them from kgC/plant/day  !
   799)          ! to umol/m2/s.                                                                   !
   800)          !---------------------------------------------------------------------------------!
   801)          leaf_growth_resp  = leaf_growth_resp                                              &
   802)                            + cpatch%leaf_growth_resp(ico)  * cpatch%nplant(ico)            &
   803)                            / (day_sec * umol_2_kgC)
   804)          root_growth_resp  = root_growth_resp                                              &
   805)                            + cpatch%root_growth_resp(ico)  * cpatch%nplant(ico)            &
   806)                            / (day_sec * umol_2_kgC)
   807)          sapa_growth_resp  = sapa_growth_resp                                              &
   808)                            + cpatch%sapa_growth_resp(ico)  * cpatch%nplant(ico)            &
   809)                            / (day_sec * umol_2_kgC)
   810)          sapb_growth_resp  = sapb_growth_resp                                              &
   811)                            + cpatch%sapb_growth_resp(ico)  * cpatch%nplant(ico)            &
   812)                            / (day_sec * umol_2_kgC)
   813)          leaf_storage_resp = leaf_storage_resp                                             &
   814)                            + cpatch%leaf_storage_resp(ico) * cpatch%nplant(ico)            &
   815)                            / (day_sec * umol_2_kgC)
   816)          root_storage_resp = root_storage_resp                                             &
   817)                            + cpatch%root_storage_resp(ico) * cpatch%nplant(ico)            &
   818)                            / (day_sec * umol_2_kgC)
   819)          sapa_storage_resp = sapa_storage_resp                                             &
   820)                            + cpatch%sapa_storage_resp(ico) * cpatch%nplant(ico)            &
   821)                            / (day_sec * umol_2_kgC)
   822)          sapb_storage_resp = sapb_storage_resp                                             &
   823)                            + cpatch%sapb_storage_resp(ico) * cpatch%nplant(ico)            &
   824)                            / (day_sec * umol_2_kgC)
   825)       end do
   826) 
   827)       return
   828)    end subroutine sum_plant_cfluxes
