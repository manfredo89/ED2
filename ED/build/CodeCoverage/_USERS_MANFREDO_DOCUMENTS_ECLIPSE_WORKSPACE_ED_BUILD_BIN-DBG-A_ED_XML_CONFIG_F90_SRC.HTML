    18) subroutine count_pft_xml_config(filename,maxpft)
    19)   integer(4) :: npft,i,myPFT
    20)   integer :: maxpft
    21)   logical(4) :: texist = .false.
    22)   character*(*) :: filename
    23)   character(1),allocatable    ::  readstring(:)
    24)   allocate(readstring(10))
    25)   deallocate(readstring)
    26) print*,'count xml pft : ',trim(filename)
    27)   !! Open File and Init
    28)   call libxml2f90__setformat(1) !set to pure XML but with blank removal
    29)   call libxml2f90__readin_file(trim(filename),'CONFIG')
    30) 
    31)   call libxml2f90__ll_selectlist('CONFIG')       
    32)   call libxml2f90__ll_selecttag('ACT','config',1) !select upper level tag
    33)   call libxml2f90__ll_exist('DOWN','pft',npft)    !get number of pft tags
    34)   print*,"NPFT = ",npft
    35)   maxpft = 0
    36)   if(npft .ge. 1) then
    37)      do i=1,npft
    38)         call libxml2f90__ll_selecttag('DOWN','pft',i) !select pft
    39)         call getConfigINT   ('num','pft',i,myPFT,texist)
    40)         if(texist .and. myPFT > maxpft) maxpft = myPFT
    41)         call libxml2f90__ll_selecttag('UP','config',1) !move back up to top level
    42)      enddo
    43)   endif
    44) 
    45)   !! next, check for a MAXPFT flag so don't have to read recursive files
    46)   call libxml2f90__ll_exist('DOWN','maxpft',npft)    !get number of pft tags
    47)   if(npft .ge. 1) then
    48)      do i= 1,npft
    49)         call getConfigINT   ('maxpft','config',i,myPFT,texist)  
    50)         if(texist .and. myPFT > maxpft) maxpft = myPFT
    51)      enddo
    52)   endif
    53) 
    54)   print*,"MAXPFT = ",maxpft
    55) 
    56)   return
    57) 
    58) end subroutine count_pft_xml_config
