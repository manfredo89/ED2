     3) !     This sub-routine solves the long-wave radiation method using the multiple-scatter-   !
     4) ! ing model.  We consider the finite crown area when computing the layer transmittance.    !
     5) ! Contrary to short-wave radiation, we don't normalise long-wave radiation.                !
     6) !                                                                                          !
     7) ! References:                                                                              !
     8) !                                                                                          !
     9) ! Zhao, W., R. J. Qualls, 2006: Modeling of long-wave and net radiation energy             !
    10) !    distribution within a homogeneous plant canopy via multiple scattering processes.     !
    11) !    Water Resources Res., 42, W08435, doi: 10.1029/2005WR004581. (ZQ06)                   !
    12) !                                                                                          !
    13) ! Oleson, K. W., and co-authors, 2004: Technical description of the community land model   !
    14) !   (CLM). NCAR Technical note NCAR/TN-461+STR. 186pp. (CLM04)                             !
    15) !                                                                                          !
    16) ! Sellers, P. J., 1985: Canopy reflectance, photosynthesis and transpiration. Intl. J.     !
    17) !    of remote sensing, 6, 1335-1372. (S85)                                                !
    18) !                                                                                          !
    19) !                                                                                          !
    20) !------------------------------------------------------------------------------------------!
    21) subroutine lw_multiple_scatter(grnd_emiss4,grnd_temp4,rlong_top4,ncoh,pft,lai,wai,cai      &
    22)                               ,leaf_temp,wood_temp,radprof_flip,tir_flip                   &
    23)                               ,dw_tirlo,uw_tirlo,uw_tirhi)
    24)    use ed_max_dims          , only : n_pft                   & ! intent(in)
    25)                                    , n_radprof               ! ! intent(in)
    26)    use rk4_coms             , only : tiny_offset             ! ! intent(in)
    27)    use canopy_radiation_coms, only : clumping_factor         & ! intent(in)
    28)                                    , leaf_emiss_tir          & ! intent(in)
    29)                                    , wood_emiss_tir          & ! intent(in)
    30)                                    , phi1                    & ! intent(in)
    31)                                    , phi2                    & ! intent(in)
    32)                                    , leaf_backscatter_tir    & ! intent(in)
    33)                                    , wood_backscatter_tir    ! ! intent(in)
    34)    use consts_coms          , only : stefan8                 ! ! intent(in)
    35) 
    36)    implicit none
    37) 
    38)    !----- Arguments. ----------------------------------------------------------------------!
    39)    real(kind=4)                              , intent(in)    :: grnd_emiss4
    40)    real(kind=4)                              , intent(in)    :: grnd_temp4
    41)    real(kind=4)                              , intent(in)    :: rlong_top4
    42)    integer                                   , intent(in)    :: ncoh
    43)    integer     , dimension(ncoh)             , intent(in)    :: pft
    44)    real(kind=8), dimension(ncoh)             , intent(in)    :: lai
    45)    real(kind=8), dimension(ncoh)             , intent(in)    :: wai
    46)    real(kind=8), dimension(ncoh)             , intent(in)    :: cai
    47)    real(kind=8), dimension(ncoh)             , intent(in)    :: leaf_temp
    48)    real(kind=8), dimension(ncoh)             , intent(in)    :: wood_temp
    49)    real(kind=4), dimension(n_radprof,ncoh)   , intent(inout) :: radprof_flip
    50)    real(kind=4), dimension(ncoh)             , intent(out)   :: tir_flip
    51)    real(kind=4)                              , intent(out)   :: dw_tirlo
    52)    real(kind=4)                              , intent(out)   :: uw_tirlo
    53)    real(kind=4)                              , intent(out)   :: uw_tirhi
    54)    !----- Local variables. ----------------------------------------------------------------!
    55)    integer                                                   :: ipft
    56)    integer                                                   :: nsiz
    57)    integer                                                   :: i
    58)    integer                                                   :: ip1
    59)    integer                                                   :: im1
    60)    integer                                                   :: i2
    61)    integer                                                   :: i2p1
    62)    integer                                                   :: i2m1
    63)    integer                                                   :: i2p2
    64)    logical                                                   :: sing
    65)    real(kind=8), dimension(ncoh)                             :: locetai
    66)    real(kind=8), dimension(ncoh)                             :: elai
    67)    real(kind=8), dimension(ncoh)                             :: etai
    68)    real(kind=8), dimension(ncoh)                             :: tai
    69)    real(kind=8), dimension(ncoh)                             :: leaf_weight
    70)    real(kind=8), dimension(ncoh)                             :: wood_weight
    71)    real(kind=8), dimension(ncoh)                             :: source_lw
    72)    real(kind=8), dimension(0:ncoh+1)                         :: tau      ! tau_i
    73)    real(kind=8), dimension(0:ncoh+1)                         :: omt      ! 1-tau_i
    74)    real(kind=8), dimension(0:ncoh+1)                         :: omr      ! 1-r_i
    75)    real(kind=8), dimension(0:ncoh+1)                         :: r        ! r_i
    76)    real(kind=8), dimension(0:ncoh+1)                         :: epsil    ! epsilon_i
    77)    real(kind=8), dimension(0:ncoh+1)                         :: ome      ! 1-epsilon_i
    78)    real(kind=8), dimension(  ncoh+1)                         :: lwd0
    79)    real(kind=8), dimension(0:ncoh  )                         :: lwu0
    80)    real(kind=8), dimension(  ncoh+1)                         :: lwd
    81)    real(kind=8), dimension(0:ncoh  )                         :: lwu
    82)    real(kind=8), dimension(2*ncoh+2)                         :: lwvec
    83)    real(kind=8), dimension(2*ncoh+2)                         :: cvec
    84)    real(kind=8), dimension(2*ncoh+2,2*ncoh+2)                :: amat
    85)    real(kind=8)                                              :: temiss_four
    86)    real(kind=8)                                              :: grnd_emiss
    87)    real(kind=8)                                              :: grnd_temp
    88)    real(kind=8)                                              :: rlong_top
    89)    real(kind=8)                                              :: ext_diff1
    90)    real(kind=8)                                              :: ext_diff2
    91)    !----- External functions. -------------------------------------------------------------!
    92)    real(kind=8)                              , external      :: eifun8
    93)    real(kind=4)                              , external      :: sngloff
    94)    !---------------------------------------------------------------------------------------!
    95) 
    96) 
    97)    !---------------------------------------------------------------------------------------!
    98)    !     Find the size of the solution matrix and array.                                   !
    99)    !---------------------------------------------------------------------------------------!
   100)    nsiz = 2*ncoh + 2
   101)    !---------------------------------------------------------------------------------------!
   102) 
   103) 
   104) 
   105)    !---------------------------------------------------------------------------------------!
   106)    !     Convert some properties to double precision.                                      !
   107)    !---------------------------------------------------------------------------------------!
   108)    grnd_emiss = dble(grnd_emiss4)
   109)    grnd_temp  = dble(grnd_temp4 )
   110)    rlong_top  = dble(rlong_top4 )
   111)    !---------------------------------------------------------------------------------------!
   112) 
   113) 
   114) 
   115)    !---------------------------------------------------------------------------------------!
   116)    !     Convert some leaf variables to double precision, then find some general           !
   117)    ! properties of leaves/branches (i.e., properties that do not depend on which band we   !
   118)    ! are solving):  the extinction coefficient lambda, the layer transmittance             !
   119)    ! coefficients tau_beam and tau_diff, and the weight of LAI and WAI relative to the     !
   120)    ! total area.                                                                           !
   121)    !---------------------------------------------------------------------------------------!
   122)    do i=1,ncoh
   123)       ipft      = pft(i)
   124)       !----- Area indices, correct LAI by clumping factor. --------------------------------!
   125)       elai(i) = clumping_factor(ipft) * lai(i)
   126)       tai(i)  =  lai(i) + wai(i)
   127)       etai(i) = elai(i) + wai(i)
   128)       !------------------------------------------------------------------------------------!
   129) 
   130) 
   131) 
   132)       !----- LOCETAI is the local exposed LAI. --------------------------------------------!
   133)       locetai(i) = etai(i) / cai(i)
   134)       !------------------------------------------------------------------------------------!
   135) 
   136) 
   137) 
   138)       !------------------------------------------------------------------------------------!
   139)       !    The weighting factors are defined by the relative LAI and WAI areas.            !
   140)       !------------------------------------------------------------------------------------!
   141)       leaf_weight(i)  =  lai(i) / tai(i)
   142)       wood_weight(i)  = 1.d0 -  leaf_weight(i)
   143)       !------------------------------------------------------------------------------------!
   144) 
   145) 
   146) 
   147)       !------------------------------------------------------------------------------------!
   148)       !     The layer transmittance coefficient for diffuse radiation is defined in a      !
   149)       ! similar way from equation (2) of ZQ06.  We must integrate the contribution coming  !
   150)       ! from each hemispheric direction.  Again, we must consider the finite crown area,   !
   151)       ! and use the same extinction function we use for shortwave radiation, which comes   !
   152)       ! from CLM04 equation 3.3.                                                           !
   153)       !     The integral can be solved analytically for our case, with the help of the     !
   154)       ! handy http://www.integrals.com website, of course ;-).                             !
   155)       !------------------------------------------------------------------------------------!
   156)       ext_diff1   = phi1(ipft) * locetai(i)
   157)       ext_diff2   = phi2(ipft) * locetai(i)
   158)       tau(i)      = (1.d0 - cai(i))                                                        &
   159)                   - cai(i) * exp(- ext_diff1 - ext_diff2)                                  &
   160)                   * ( ext_diff1 * ext_diff1 * exp(ext_diff1) * eifun8(-ext_diff1)          &
   161)                     + (ext_diff1 - 1.d0) )
   162)       !------------------------------------------------------------------------------------!
   163) 
   164) 
   165) 
   166)       !----- Backward scattering of diffuse radiation. ------------------------------------!
   167)       r(i)     = leaf_weight(i) * leaf_backscatter_tir(ipft)                               &
   168)                + wood_weight(i) * wood_backscatter_tir(ipft)
   169)       !------------------------------------------------------------------------------------!
   170) 
   171) 
   172) 
   173)       !----- Layer emissivity. ------------------------------------------------------------!
   174)       epsil(i) = leaf_weight(i) * leaf_emiss_tir(ipft)                                     &
   175)                + wood_weight(i) * wood_emiss_tir(ipft)
   176)       !------------------------------------------------------------------------------------!
   177) 
   178) 
   179) 
   180)       !------------------------------------------------------------------------------------!
   181)       !      Thermal emission.  The temperature is the weighted average where the weights  !
   182)       ! are the product between the area and the emissivity.                               !
   183)       !------------------------------------------------------------------------------------!
   184)       temiss_four  = ( leaf_weight(i) * leaf_emiss_tir(ipft) * leaf_temp(i) ** 4           &
   185)                      + wood_weight(i) * wood_emiss_tir(ipft) * wood_temp(i) ** 4 )         &
   186)                    / ( leaf_weight(i) * leaf_emiss_tir(ipft)                               &
   187)                      + wood_weight(i) * wood_emiss_tir(ipft) )
   188)       source_lw(i) = epsil(i) * stefan8 * temiss_four
   189)       !------------------------------------------------------------------------------------!
   190) 
   191) 
   192)    end do
   193)    !---------------------------------------------------------------------------------------!
   194) 
   195) 
   196) 
   197)    !---------------------------------------------------------------------------------------!
   198)    !     Define some boundary conditions for transmission, emission, and backscattering,   !
   199)    ! and the complement of the three properties.  The boundary conditions come from ZQ06,  !
   200)    ! equations 16-22.                                                                      !
   201)    !---------------------------------------------------------------------------------------!
   202)    tau       (0) = 0.d0
   203)    tau  (ncoh+1) = 1.d0
   204)    r         (0) = 1.d0
   205)    r    (ncoh+1) = 0.d0
   206)    epsil     (0) = grnd_emiss
   207)    epsil(ncoh+1) = 0.d0
   208)    omr(:)        = 1.d0 - r    (:)
   209)    omt(:)        = 1.d0 - tau  (:)
   210)    ome(:)        = 1.d0 - epsil(:)
   211)    !---------------------------------------------------------------------------------------!
   212) 
   213) 
   214) 
   215)    !---------------------------------------------------------------------------------------!
   216)    !    Find the top diffuse radiation, using equations (19) and (22) from ZQ06.           !
   217)    !---------------------------------------------------------------------------------------!
   218)    lwd0     (ncoh+1) = rlong_top
   219)    lwu0     (     0) = epsil(0) * stefan8 * grnd_temp ** 4
   220)    !---------------------------------------------------------------------------------------!
   221) 
   222) 
   223) 
   224) 
   225)    !---------------------------------------------------------------------------------------!
   226)    !     Fill in the right hand side vector (C) and the matrix (A) described by            !
   227)    ! equations (25) and ("25 1/2") of ZQ06.                                                !
   228)    !---------------------------------------------------------------------------------------!
   229)    !------ Initialise the vector and the matrix.  The matrix is a sparse one... -----------!
   230)    amat(:,:)       = 0.d0
   231)    cvec(:)         = 0.d0
   232)    !------ Add the edge values first. -----------------------------------------------------!
   233)    amat(1,1)       = 1.d0
   234)    amat(nsiz,nsiz) = 1.d0
   235)    cvec(1)         = lwu0(0)
   236)    cvec(nsiz)      = lwd0(ncoh+1)
   237)    do i=1,ncoh
   238)       !----- Find auxiliary indices. ------------------------------------------------------!
   239)       ip1  =  i + 1
   240)       im1  =  i - 1
   241)       i2   =  2 * i
   242)       i2p1 = i2 + 1
   243)       i2m1 = i2 - 1
   244)       i2p2 = i2 + 2
   245)       !------------------------------------------------------------------------------------!
   246) 
   247)       !----- Make elements of C vector. ---------------------------------------------------!
   248)       cvec(i2)   = ( 1.d0 - r(im1) * ome(im1) * omt(im1) * r(i)   * ome(i)   * omt(i))     &
   249)                    * omt(i) * source_lw(i)                                                
   250)       cvec(i2p1) = ( 1.d0 - r(i)   * ome(i)   * omt(i) * r(ip1) * ome(ip1) * omt(ip1))     &
   251)                    * omt(i) * source_lw(i)                                                
   252)       !------------------------------------------------------------------------------------!
   253) 
   254) 
   255) 
   256)       !------------------------------------------------------------------------------------!
   257)       !     Make elements of the tri-diagonal A array.                                     !
   258)       !------------------------------------------------------------------------------------!
   259)       amat(i2  ,i2m1) = - (tau(i) + omr(i) * ome(i) * omt(i))
   260)       amat(i2  ,i2  ) = - r(im1) * ome(im1) * omt(im1) * (tau(i) + omt(i) * ome(i) * omr(i))
   261)       amat(i2  ,i2p1) = 1.d0 - r(im1) * ome(im1) * omt(im1) * r(i)   * ome(i)   * omt(i)
   262)       amat(i2p1,i2  ) = 1.d0 - r(i)   * ome(i)   * omt(i) * r(ip1) * ome(ip1) * omt(ip1)
   263)       amat(i2p1,i2p1) = - r(ip1) * ome(ip1) * omt(ip1) * (tau(i) + omt(i) * ome(i) * omr(i))
   264)       amat(i2p1,i2p2) = - (tau(i) + omt(i) * ome(i) * omr(i))
   265)    end do
   266)    !---------------------------------------------------------------------------------------!
   267) 
   268) 
   269) 
   270)    !---------------------------------------------------------------------------------------!
   271)    !       Solve the linear system.  In the future we could use a tridiagonal solver,      !
   272)    ! which is a lot cheaper than the regular Gauss elimination, but for the time being, we !
   273)    ! go with a tested method.                                                              !
   274)    !---------------------------------------------------------------------------------------!
   275)    call lisys_solver8(nsiz,amat,cvec,lwvec,sing)
   276)    if (sing) then
   277)       call fatal_error('LW radiation failed... The matrix is singular!'                    &
   278)                       ,'lw_multiple_scatter','multiple_scatter.f90')
   279)    end if
   280)    !---------------------------------------------------------------------------------------!
   281) 
   282) 
   283) 
   284)    !---------------------------------------------------------------------------------------!
   285)    !     Copy the solution without considering the multiple scattering to the vectors.     !
   286)    !---------------------------------------------------------------------------------------!
   287)    lwu0(0)      = lwvec(1)
   288)    lwd0(ncoh+1) = lwvec(nsiz)
   289)    do i=1,ncoh
   290) 
   291)       !----- Auxiliary indices. -----------------------------------------------------------!
   292)       i2   = i * 2
   293)       i2p1 = i * 2 + 1
   294)       !------------------------------------------------------------------------------------!
   295) 
   296) 
   297) 
   298)       !------------------------------------------------------------------------------------!
   299)       !      Retrieve the downward and upward diffuse (hemispheric) radiation, using       !
   300)       ! equation (24) from ZQ06.                                                           !
   301)       !------------------------------------------------------------------------------------!
   302)       lwd0(i) = lwvec(i2)
   303)       lwu0(i) = lwvec(i2p1)
   304)       !------------------------------------------------------------------------------------!
   305) 
   306)    end do
   307)    !---------------------------------------------------------------------------------------!
   308) 
   309) 
   310) 
   311)    !---------------------------------------------------------------------------------------!
   312)    !     Find the diffuse radiation when multiple scattering is considered.  This is based !
   313)    ! on equations (8) and (9) of ZQ06, with the difference that we find LWd_i instead of   !
   314)    ! LWd_i+1.                                                                              !
   315)    !---------------------------------------------------------------------------------------!
   316)    !----- Downward diffuse (hemispheric) radiation. ---------------------------------------!
   317)    do i=1,ncoh+1
   318)       im1 = i - 1
   319)       ip1 = i + 1
   320)       lwd(i) = ( lwd0(i) + r(i) * ome(i) * omt(i) * lwu0(im1) )                            &
   321)              / ( 1.d0 - r(im1) * ome(im1) * omt(im1) * r(i)   * ome(i)   * omt(i)   )
   322)    end do
   323)    !----- Upward diffuse (hemispheric) radiation. -----------------------------------------!
   324)    do i=0,ncoh
   325)       im1 = i - 1
   326)       ip1 = i + 1
   327)       lwu(i) = ( lwu0(i) + r(i) * ome(i) * omt(i) * lwd0(ip1) )                            &
   328)              / ( 1.d0 - r(i)   * ome(i)   * omt(i) * r(ip1) * ome(ip1) * omt(ip1)   )
   329)    end do
   330)    !---------------------------------------------------------------------------------------!
   331) 
   332) 
   333) 
   334)    !---------------------------------------------------------------------------------------!
   335)    !     Save the fluxes that we will use outside this sub-routine.                        !
   336)    !---------------------------------------------------------------------------------------!
   337) 
   338) 
   339) 
   340)    !------ Save the fluxes reaching the surface and leaving the top. ----------------------!
   341)    dw_tirlo = sngloff(lwd          (1), tiny_offset)
   342)    uw_tirlo = sngloff(lwu          (0), tiny_offset)
   343)    uw_tirhi = sngloff(lwu       (ncoh), tiny_offset)
   344)    !---------------------------------------------------------------------------------------!
   345) 
   346) 
   347)    !---------------------------------------------------------------------------------------!
   348)    !     Save the radiation fluxes to the output variable.                                 !
   349)    !---------------------------------------------------------------------------------------!
   350)    do i=1,ncoh
   351)       im1 = i - 1
   352)       ip1 = i + 1
   353)       radprof_flip( 9,i) = sngloff(lwd  (i), tiny_offset)
   354)       radprof_flip(10,i) = sngloff(lwu(im1), tiny_offset)
   355)       tir_flip       (i) = sngloff(lwd(ip1) - lwd(i) + lwu(im1) - lwu(i), tiny_offset)
   356)    end do
   357)    !---------------------------------------------------------------------------------------!
   358) 
   359)    return
   360) end subroutine lw_multiple_scatter

   371) !     This sub-routine solves the short-wave radiation method using the multiple-scatter-  !
   372) ! ing model.  We consider the finite crown area when computing the transmittance of both   !
   373) ! direct and diffuse radiation.  References:                                               !
   374) !                                                                                          !
   375) ! Zhao, W., R. J. Qualls, 2005: A multiple-layer canopy scattering model to simulate       !
   376) !    shortwave radiation distribution within a homogeneous plant canopy.  Water Resources  !
   377) !    Res., 41, W08409, doi: 10.1029/2005WR004016. (ZQ05)                                   !
   378) !                                                                                          !
   379) ! Oleson, K. W., and co-authors, 2004: Technical description of the community land model   !
   380) !   (CLM). NCAR Technical note NCAR/TN-461+STR. 186pp. (CLM04)                             !
   381) !                                                                                          !
   382) ! Sellers, P. J., 1985: Canopy reflectance, photosynthesis and transpiration. Intl. J.     !
   383) !    of remote sensing, 6, 1335-1372. (S85)                                                !
   384) !                                                                                          !
   385) !                                                                                          !
   386) !------------------------------------------------------------------------------------------!
   387) subroutine sw_multiple_scatter(grnd_alb_par4,grnd_alb_nir4,cosaoi4,ncoh,pft,lai,wai,cai    &
   388)                               ,radprof_flip,par_beam_flip,par_diff_flip,sw_abs_beam_flip   &
   389)                               ,sw_abs_diff_flip,dw_parlo_beam,dw_parlo_diff                &
   390)                               ,uw_parhi_diff,dw_nirlo_beam,dw_nirlo_diff,uw_nirhi_diff     &
   391)                               ,par_level_beam,par_level_diffd,par_level_diffu              &
   392)                               ,light_level,light_beam_level  &
   393)                               ,light_diff_level)
   394)    use ed_max_dims          , only : n_pft                   & ! intent(in)
   395)                                    , n_radprof               ! ! intent(in)
   396)    use rk4_coms             , only : tiny_offset             ! ! intent(in)
   397)    use canopy_radiation_coms, only : clumping_factor         & ! intent(in)
   398)                                    , phi1                    & ! intent(in)
   399)                                    , phi2                    & ! intent(in)
   400)                                    , mu_bar                  & ! intent(in)
   401)                                    , leaf_scatter_vis        & ! intent(in)
   402)                                    , leaf_scatter_nir        & ! intent(in)
   403)                                    , leaf_backscatter_nir    & ! intent(in)
   404)                                    , leaf_backscatter_vis    & ! intent(in)
   405)                                    , wood_scatter_nir        & ! intent(in)
   406)                                    , wood_scatter_vis        & ! intent(in)
   407)                                    , wood_backscatter_nir    & ! intent(in)
   408)                                    , wood_backscatter_vis    & ! intent(in)
   409)                                    , par_beam_norm           & ! intent(in)
   410)                                    , par_diff_norm           & ! intent(in)
   411)                                    , nir_beam_norm           & ! intent(in)
   412)                                    , nir_diff_norm           & ! intent(in)
   413)                                    , cosz_min8               ! ! intent(in)
   414) 
   415)    implicit none
   416) 
   417)    !----- Arguments. ----------------------------------------------------------------------!
   418)    real(kind=4)                              , intent(in)    :: grnd_alb_par4
   419)    real(kind=4)                              , intent(in)    :: grnd_alb_nir4
   420)    real(kind=4)                              , intent(in)    :: cosaoi4
   421)    integer                                   , intent(in)    :: ncoh
   422)    integer     , dimension(ncoh)             , intent(in)    :: pft
   423)    real(kind=8), dimension(ncoh)             , intent(in)    :: lai
   424)    real(kind=8), dimension(ncoh)             , intent(in)    :: wai
   425)    real(kind=8), dimension(ncoh)             , intent(in)    :: cai
   426)    real(kind=4), dimension(n_radprof,ncoh)   , intent(inout) :: radprof_flip
   427)    real(kind=4), dimension(ncoh)             , intent(out)   :: par_beam_flip
   428)    real(kind=4), dimension(ncoh)             , intent(out)   :: par_diff_flip
   429)    real(kind=4), dimension(ncoh)             , intent(out)   :: sw_abs_beam_flip
   430)    real(kind=4), dimension(ncoh)             , intent(out)   :: sw_abs_diff_flip
   431)    real(kind=4)                              , intent(out)   :: uw_parhi_diff
   432)    real(kind=4)                              , intent(out)   :: uw_nirhi_diff
   433)    real(kind=4)                              , intent(out)   :: dw_parlo_beam
   434)    real(kind=4)                              , intent(out)   :: dw_parlo_diff
   435)    real(kind=4)                              , intent(out)   :: dw_nirlo_beam
   436)    real(kind=4)                              , intent(out)   :: dw_nirlo_diff
   437)    real(kind=8), dimension(ncoh)             , intent(out)   :: par_level_beam
   438)    real(kind=8), dimension(ncoh)             , intent(out)   :: par_level_diffu
   439)    real(kind=8), dimension(ncoh)             , intent(out)   :: par_level_diffd
   440)    real(kind=8), dimension(ncoh)             , intent(out)   :: light_level
   441)    real(kind=8), dimension(ncoh)             , intent(out)   :: light_beam_level
   442)    real(kind=8), dimension(ncoh)             , intent(out)   :: light_diff_level
   443)    !----- Local variables. ----------------------------------------------------------------!
   444)    integer                                                 :: ipft
   445)    integer                                                 :: iband
   446)    integer                                                 :: nsiz
   447)    integer                                                 :: i
   448)    integer                                                 :: ip1
   449)    integer                                                 :: im1
   450)    integer                                                 :: i2
   451)    integer                                                 :: i2p1
   452)    integer                                                 :: i2m1
   453)    integer                                                 :: i2p2
   454)    logical                                                 :: sing
   455)    real(kind=8)                                            :: alb_par
   456)    real(kind=8)                                            :: alb_nir
   457)    real(kind=8)                                            :: mu
   458)    real(kind=4), dimension(ncoh)                           :: nir_beam_flip
   459)    real(kind=4), dimension(ncoh)                           :: nir_diff_flip
   460)    real(kind=8), dimension(ncoh)                           :: locetai
   461)    real(kind=8), dimension(ncoh)                           :: etai
   462)    real(kind=8), dimension(ncoh)                           :: elai
   463)    real(kind=8), dimension(ncoh)                           :: leaf_weight
   464)    real(kind=8), dimension(ncoh)                           :: wood_weight
   465)    real(kind=8), dimension(ncoh)                           :: beam_backscatter
   466)    real(kind=8), dimension(ncoh)                           :: lambda
   467)    real(kind=8), dimension(0:ncoh+1)                       :: tau_beam ! tau_i(psi)
   468)    real(kind=8), dimension(0:ncoh+1)                       :: tau_diff ! tau_i
   469)    real(kind=8), dimension(0:ncoh+1)                       :: omt_beam ! 1-tau_i(psi)
   470)    real(kind=8), dimension(0:ncoh+1)                       :: omt_diff ! 1-tau_i
   471)    real(kind=8), dimension(0:ncoh+1)                       :: omr_beam ! 1-r_i(psi)
   472)    real(kind=8), dimension(0:ncoh+1)                       :: omr_diff ! 1-r_i
   473)    real(kind=8), dimension(0:ncoh+1)                       :: r_beam   ! r_i(psi)
   474)    real(kind=8), dimension(0:ncoh+1)                       :: r_diff   ! r_i
   475)    real(kind=8), dimension(0:ncoh+1)                       :: alpha    ! alpha_i
   476)    real(kind=8), dimension(0:ncoh+1)                       :: oma      ! 1-alpha_i
   477)    real(kind=8), dimension(  ncoh+1)                       :: beam_down
   478)    real(kind=8), dimension(  ncoh+1)                       :: swd0
   479)    real(kind=8), dimension(0:ncoh  )                       :: swu0
   480)    real(kind=8), dimension(  ncoh+1)                       :: swd
   481)    real(kind=8), dimension(0:ncoh  )                       :: swu
   482)    real(kind=8), dimension(2*ncoh+2)                       :: swvec
   483)    real(kind=8), dimension(2*ncoh+2)                       :: cvec
   484)    real(kind=8), dimension(2*ncoh+2,2*ncoh+2)              :: amat
   485)    real(kind=8)                                            :: proj_area
   486)    real(kind=8)                                            :: ext_diff1
   487)    real(kind=8)                                            :: ext_diff2
   488)    real(kind=8)                                            :: snglscat_alb
   489)    !----- External functions. -------------------------------------------------------------!
   490)    real(kind=8)                              , external    :: eifun8
   491)    real(kind=4)                              , external    :: sngloff
   492)    !---------------------------------------------------------------------------------------!
   493) 
   494) 
   495)    !---------------------------------------------------------------------------------------!
   496)    !     Find the size of the solution matrix and array.                                   !
   497)    !---------------------------------------------------------------------------------------!
   498)    nsiz = 2*ncoh + 2
   499)    !---------------------------------------------------------------------------------------!
   500) 
   501) 
   502) 
   503)    !---------------------------------------------------------------------------------------!
   504)    !     Convert cosine of angle of incidence to double precision.                         !
   505)    !---------------------------------------------------------------------------------------!
   506)    mu      = max(cosz_min8,dble(cosaoi4))
   507)    alb_par = dble(grnd_alb_par4)
   508)    alb_nir = dble(grnd_alb_nir4)
   509)    !---------------------------------------------------------------------------------------!
   510) 
   511) 
   512) 
   513)    !---------------------------------------------------------------------------------------!
   514)    !     Convert some leaf variables to double precision, then find some general           !
   515)    ! properties of leaves/branches (i.e., properties that do not depend on which band we   !
   516)    ! are solving):  the extinction coefficient lambda, the layer transmittance             !
   517)    ! coefficients tau_beam and tau_diff, and the weight of LAI and WAI relative to the     !
   518)    ! total area.                                                                           !
   519)    !---------------------------------------------------------------------------------------!
   520)    do i=1,ncoh
   521)       ipft      = pft(i)
   522)       !----- Area indices, correct LAI by clumping factor. --------------------------------!
   523)       elai(i) = clumping_factor(ipft) * lai(i)
   524)       etai(i) = elai(i) + wai(i)
   525)       !------------------------------------------------------------------------------------!
   526) 
   527) 
   528) 
   529)       !----- LOCETAI is the local exposed LAI. --------------------------------------------!
   530)       locetai(i) = etai(i) / cai(i)
   531)       !------------------------------------------------------------------------------------!
   532) 
   533) 
   534) 
   535)       !------------------------------------------------------------------------------------!
   536)       !    The weighting factors are defined by the relative LAI and WAI areas.            !
   537)       !------------------------------------------------------------------------------------!
   538)       leaf_weight(i) = elai(i) / etai(i)
   539)       wood_weight(i) = 1.d0 - leaf_weight(i)
   540)       !------------------------------------------------------------------------------------!
   541) 
   542) 
   543) 
   544)       !------------------------------------------------------------------------------------!
   545)       !     We find the optical depth of the direct beam (lambda), following CLM04         !
   546)       ! (equation 3.3 and text after equation 3.2).                                        !
   547)       !------------------------------------------------------------------------------------!
   548)       proj_area     = phi1(ipft) + phi2(ipft) * mu
   549)       lambda    (i) = proj_area / mu
   550)       !------------------------------------------------------------------------------------!
   551) 
   552) 
   553) 
   554)       !------------------------------------------------------------------------------------!
   555)       !     The layer transmittance coefficient for direct radiation is defined from       !
   556)       ! equation (1) of ZQ05, with the difference that the crown area is taken into        !
   557)       ! account.                                                                           !
   558)       !------------------------------------------------------------------------------------!
   559)       tau_beam(i) = (1.d0 - cai(i)) + cai(i) * exp (- lambda(i) * locetai(i))
   560)       !------------------------------------------------------------------------------------!
   561) 
   562) 
   563)       !------------------------------------------------------------------------------------!
   564)       !     The layer transmittance coefficient for diffuse radiation is defined in a      !
   565)       ! similar way from equation (2) of ZQ06.  We must integrate the contribution coming  !
   566)       ! from each hemispheric direction.  Again, we must consider the finite crown area.   !
   567)       !     The integral can be solved analytically for our case, with the help of the     !
   568)       ! handy http://www.integrals.com website, of course ;-).                             !
   569)       !------------------------------------------------------------------------------------!
   570)       ext_diff1   = phi1(ipft) * locetai(i)
   571)       ext_diff2   = phi2(ipft) * locetai(i)
   572)       tau_diff(i) = (1.d0 - cai(i))                                                        &
   573)                   - cai(i) * exp(- ext_diff1 - ext_diff2)                                  &
   574)                   * ( ext_diff1 * ext_diff1 * exp(ext_diff1) * eifun8(-ext_diff1)          &
   575)                     + (ext_diff1 - 1.d0) )
   576)       !------------------------------------------------------------------------------------!
   577) 
   578) 
   579) 
   580)       !------------------------------------------------------------------------------------!
   581)       !      Find the backscatter for direct beam radiation, following CLM04, equations    !
   582)       ! (3.14) and (3.15).  The forward scattering was omitted on both equations because   !
   583)       ! they are different for PAR and NIR, but in both cases they would cancel out.       !
   584)       !------------------------------------------------------------------------------------!
   585)       snglscat_alb        = 5.d-1 * proj_area / (phi2(ipft) * mu + proj_area)              &
   586)                           * ( 1.d0 - phi1(ipft) * mu / (phi2(ipft) * mu + proj_area)       &
   587)                             * log (1.d0  + (phi2(ipft) * mu + proj_area)                   &
   588)                                          / (phi1(ipft) * mu)))
   589)       beam_backscatter(i) = ( 1.d0 + mu_bar(ipft) * lambda(i) ) * snglscat_alb             &
   590)                              / ( mu_bar(ipft) * lambda(i) )
   591)       !------------------------------------------------------------------------------------!
   592)    end do
   593)    !---------------------------------------------------------------------------------------!
   594) 
   595) 
   596) 
   597)    !---------------------------------------------------------------------------------------!
   598)    !     Define some boundary conditions for transmission, and the complement of           !
   599)    ! transmission.  The boundary conditions come from ZQ05, equations 32-37.               !
   600)    !---------------------------------------------------------------------------------------!
   601)    tau_beam(0)      = 0.d0
   602)    tau_beam(ncoh+1) = 1.d0
   603)    tau_diff(0)      = 0.d0
   604)    tau_diff(ncoh+1) = 1.d0
   605)    omt_beam(:)      = 1.d0 - tau_beam(:)
   606)    omt_diff(:)      = 1.d0 - tau_diff(:)
   607)    !---------------------------------------------------------------------------------------!
   608) 
   609) 
   610) 
   611)    !---------------------------------------------------------------------------------------!
   612)    !     Initialise the light level variables.                                             !
   613)    !---------------------------------------------------------------------------------------!
   614)    light_level     (:) = 0.d0
   615)    light_beam_level(:) = 0.d0
   616)    light_diff_level(:) = 0.d0
   617)    !---------------------------------------------------------------------------------------!
   618) 
   619) 
   620) 
   621)    !---------------------------------------------------------------------------------------!
   622)    !     Here we loop over the two bands (band 1 is PAR and band 2 is NIR), following the  !
   623)    ! algorithm proposed by ZQ05.                                                           !
   624)    !---------------------------------------------------------------------------------------!
   625)    bandloop: do iband = 1,2
   626)       select case (iband)
   627)       case (1)
   628)          !---------------------------------------------------------------------------------!
   629)          !     Visible (PAR).                                                              !
   630)          !---------------------------------------------------------------------------------!
   631)          do i=1,ncoh
   632)             !----- Alias for PFT of this layer. -------------------------------------------!
   633)             ipft = pft(i)
   634) 
   635)             !----- Absorptance: the fraction that is not scattered. -----------------------!
   636)             alpha(i)    = 1.d0 - leaf_weight(i) * leaf_scatter_vis(ipft)                   &
   637)                                - wood_weight(i) * wood_scatter_vis(ipft)
   638) 
   639)             !----- Backward scattering of diffuse radiation. ------------------------------!
   640)             r_diff(i)   = leaf_weight(i) * leaf_backscatter_vis(ipft)                      &
   641)                         + wood_weight(i) * wood_backscatter_vis(ipft)
   642) 
   643)             !----- Backward scattering of direct radiation. -------------------------------!
   644)             r_beam(i)   = beam_backscatter(i)
   645) 
   646)          end do
   647)          !---------------------------------------------------------------------------------!
   648) 
   649) 
   650) 
   651)          !---------------------------------------------------------------------------------!
   652)          !     Find the boundary conditions following ZQ05, equations (32-37).             !
   653)          !---------------------------------------------------------------------------------!
   654)          alpha (0)      = 1.d0 - alb_par
   655)          alpha (ncoh+1) = 0.d0
   656)          r_beam(0)      = 1.d0
   657)          r_diff(0)      = 1.d0
   658)          r_beam(ncoh+1) = 0.d0
   659)          r_diff(ncoh+1) = 0.d0
   660)          !---------------------------------------------------------------------------------!
   661) 
   662) 
   663) 
   664)          !---------------------------------------------------------------------------------!
   665)          !    Find the top beam and diffuse radiation.  For the direct radiation, we       !
   666)          ! follow a notation that is slightly different from ZQ05 notation, in which       !
   667)          ! the beam radiation at level i is the amount that reaches the bottom of the      !
   668)          ! layer.                                                                          !
   669)          !---------------------------------------------------------------------------------!
   670)          beam_down(ncoh+1) = par_beam_norm
   671)          swd0     (ncoh+1) = par_diff_norm
   672)          !---------------------------------------------------------------------------------!
   673) 
   674) 
   675)       case (2)
   676)          !---------------------------------------------------------------------------------!
   677)          !     Near-Infrared (NIR).                                                        !
   678)          !---------------------------------------------------------------------------------!
   679)          do i=1,ncoh
   680)             !----- Alias for PFT of this layer. -------------------------------------------!
   681)             ipft = pft(i)
   682) 
   683)             !----- Absorptance: the fraction that is not scattered. -----------------------!
   684)             alpha(i)    = 1.d0 - leaf_weight(i) * leaf_scatter_nir(ipft)                   &
   685)                                - wood_weight(i) * wood_scatter_nir(ipft)
   686) 
   687)             !----- Backward scattering of diffuse radiation. ------------------------------!
   688)             r_diff(i)   = leaf_weight(i) * leaf_backscatter_nir(ipft)                      &
   689)                         + wood_weight(i) * wood_backscatter_nir(ipft)
   690) 
   691)             !----- Backward scattering of direct radiation. -------------------------------!
   692)             r_beam(i)   = beam_backscatter(i)
   693) 
   694)          end do
   695)          !---------------------------------------------------------------------------------!
   696) 
   697) 
   698) 
   699)          !---------------------------------------------------------------------------------!
   700)          !     Find the boundary conditions following ZQ05, equations (32-37).             !
   701)          !---------------------------------------------------------------------------------!
   702)          alpha (0)      = 1.d0 - alb_nir
   703)          alpha (ncoh+1) = 0.d0
   704)          r_beam(0)      = 1.d0
   705)          r_diff(0)      = 1.d0
   706)          r_beam(ncoh+1) = 0.d0
   707)          r_diff(ncoh+1) = 0.d0
   708)          !---------------------------------------------------------------------------------!
   709) 
   710) 
   711) 
   712)          !---------------------------------------------------------------------------------!
   713)          !    Find the top beam and diffuse radiation.  For the direct radiation, we       !
   714)          ! follow a notation that is slightly different from ZQ05 notation, in which       !
   715)          ! the beam radiation at level i is the amount that reaches the bottom of the      !
   716)          ! layer.                                                                          !
   717)          !---------------------------------------------------------------------------------!
   718)          beam_down(ncoh+1) = nir_beam_norm
   719)          swd0     (ncoh+1) = nir_diff_norm
   720)          !---------------------------------------------------------------------------------!
   721)       end select
   722) 
   723) 
   724) 
   725)       !----- Find the complement of the absorptance and scattering. -----------------------!
   726)       oma     (:) = 1.d0 - alpha (:)
   727)       omr_diff(:) = 1.d0 - r_diff(:)
   728)       omr_beam(:) = 1.d0 - r_beam(:)
   729)       !------------------------------------------------------------------------------------!
   730) 
   731) 
   732) 
   733)       !------------------------------------------------------------------------------------!
   734)       !    Find the profile of downward direct radiation using the typical atenuation      !
   735)       ! function.  We define the beam_down vector in a slightly different way from ZQ05,   !
   736)       ! because here we will define S_i as the radiation that reaches the bottom of the    !
   737)       ! layer, whereas they defined it as the radiation that reaches the top.              !
   738)       !------------------------------------------------------------------------------------!
   739)       do i=ncoh,1,-1
   740)          beam_down(i) = beam_down(i+1) * tau_beam(i)
   741)       end do
   742)       !------------------------------------------------------------------------------------!
   743) 
   744) 
   745)       !------------------------------------------------------------------------------------!
   746)       !     Fill in the right hand side vector (C) and the matrix (A) described by         !
   747)       ! equations (41) and (42) of ZQ05.                                                   !
   748)       !------------------------------------------------------------------------------------!
   749)       !------ Initialise the vector and the matrix.  The matrix is a sparse one... --------!
   750)       amat(:,:)       = 0.d0
   751)       cvec(:)         = 0.d0
   752)       !------ Add the edge values first. --------------------------------------------------!
   753)       amat(1,1)       = 1.d0
   754)       amat(nsiz,nsiz) = 1.d0
   755)       cvec(1)         = oma(0) * beam_down(1)
   756)       cvec(nsiz)      = swd0(ncoh+1)
   757)       do i=1,ncoh
   758)          !----- Find auxiliary indices. ---------------------------------------------------!
   759)          ip1  =  i + 1
   760)          im1  =  i - 1
   761)          i2   =  2 * i
   762)          i2p1 = i2 + 1
   763)          i2m1 = i2 - 1
   764)          i2p2 = i2 + 2
   765)          !---------------------------------------------------------------------------------!
   766) 
   767)          !----- Make elements of C vector. ------------------------------------------------!
   768)          cvec(i2)      = ( 1.d0                                                            &
   769)                          - r_diff(im1) * oma(im1) * omt_diff(im1)                          &
   770)                          * r_diff(i)   * oma(i)   * omt_diff(i)   )                        &
   771)                          * oma(i) * omt_beam(i) * r_beam(i) * beam_down(ip1)
   772)          cvec(i2p1)    = ( 1.d0                                                            &
   773)                          - r_diff(i)   * oma(i)   * omt_diff(i)                            &
   774)                          * r_diff(ip1) * oma(ip1) * omt_diff(ip1)   )                      &
   775)                          * oma(i) * omt_beam(i) * omr_beam(i) * beam_down(ip1)
   776)          !---------------------------------------------------------------------------------!
   777) 
   778) 
   779) 
   780)          !---------------------------------------------------------------------------------!
   781)          !     Make elements of the tri-diagonal A array.                                  !
   782)          !---------------------------------------------------------------------------------!
   783)          amat(i2  ,i2m1)  = - (tau_diff(i) + omt_diff(i) * oma(i) * omr_diff(i))
   784)          amat(i2  ,i2  )  = - r_diff(im1) * oma(im1) * omt_diff(im1)                       &
   785)                           * (tau_diff(i) + omt_diff(i) * oma(i) * omr_diff(i) )
   786)          amat(i2  ,i2p1)  = 1.d0 - r_diff(im1) * oma(im1) * omt_diff(im1)                  &
   787)                                  * r_diff(i)   * oma(i)   * omt_diff(i)
   788)          amat(i2p1,i2  )  = 1.d0 - r_diff(i)   * oma(i)   * omt_diff(i)                    &
   789)                                  * r_diff(ip1) * oma(ip1) * omt_diff(ip1)
   790)          amat(i2p1,i2p1)  = - r_diff(ip1) * oma(ip1) * omt_diff(ip1)                       &
   791)                             * (tau_diff(i) + omt_diff(i) * oma(i) * omr_diff(i))
   792)          amat(i2p1,i2p2)  = - (tau_diff(i) + omt_diff(i) * oma(i) * omr_diff(i))
   793)       end do
   794)       !------------------------------------------------------------------------------------!
   795) 
   796) 
   797) 
   798)       !------------------------------------------------------------------------------------!
   799)       !       Solve the linear system.  In the future we could use a tridiagonal solver,   !
   800)       ! which is a lot cheaper than the regular Gauss elimination, but for the time being, !
   801)       ! we go with a tested method.                                                        !
   802)       !------------------------------------------------------------------------------------!
   803)       call lisys_solver8(nsiz,amat,cvec,swvec,sing)
   804)       if (sing) then
   805)          call fatal_error('SW radiation failed... The matrix is singular!'                 &
   806)                          ,'sw_multiple_scatter','multiple_scatter.f90')
   807)       end if
   808)       !------------------------------------------------------------------------------------!
   809) 
   810) 
   811) 
   812)       !------------------------------------------------------------------------------------!
   813)       !     Copy the solution without considering the multiple scattering to the vectors.  !
   814)       !------------------------------------------------------------------------------------!
   815)       swu0(0)      = swvec(1)
   816)       swd0(ncoh+1) = swvec(nsiz)
   817)       do i=1,ncoh
   818) 
   819)          !----- Auxiliary indices. --------------------------------------------------------!
   820)          i2   = i * 2
   821)          i2p1 = i * 2 + 1
   822)          !---------------------------------------------------------------------------------!
   823) 
   824) 
   825) 
   826)          !---------------------------------------------------------------------------------!
   827)          !      Retrieve the downward and upward diffuse (hemispheric) radiation, using    !
   828)          ! equation (40) from ZQ05.                                                        !
   829)          !---------------------------------------------------------------------------------!
   830)          swd0(i) = swvec(i2)
   831)          swu0(i) = swvec(i2p1)
   832)          !---------------------------------------------------------------------------------!
   833) 
   834)       end do
   835)       !------------------------------------------------------------------------------------!
   836) 
   837) 
   838) 
   839)       !------------------------------------------------------------------------------------!
   840)       !     Find the diffuse radiation when multiple scattering is considered.  This is    !
   841)       ! based on equations (24) and (25) of ZQ05, with the difference that we find SWd_i   !
   842)       ! instead of SWd_i+1.                                                                !
   843)       !------------------------------------------------------------------------------------!
   844)       !----- Downward diffuse (hemispheric) radiation. ------------------------------------!
   845)       do i=1,ncoh+1
   846)          im1 = i - 1
   847)          ip1 = i + 1
   848)          swd(i) = ( swd0(i) + r_diff(i) * oma(i) * omt_diff(i) * swu0(im1) )               &
   849)                 / ( 1.d0 - r_diff(im1) * oma(im1) * omt_diff(im1)                          &
   850)                          * r_diff(i)   * oma(i)   * omt_diff(i)   )
   851)       end do
   852)       !----- Upward diffuse (hemispheric) radiation. --------------------------------------!
   853)       do i=0,ncoh
   854)          im1 = i - 1
   855)          ip1 = i + 1
   856)          swu(i) = ( swu0(i) + r_diff(i) * oma(i) * omt_diff(i) * swd0(ip1) )               &
   857)                 / ( 1.d0 - r_diff(i)   * oma(i)   * omt_diff(i)                            &
   858)                          * r_diff(ip1) * oma(ip1) * omt_diff(ip1)   )
   859)       end do
   860)       !------------------------------------------------------------------------------------!
   861) 
   862) 
   863)       !------------------------------------------------------------------------------------!
   864)       !     Integrate the light levels.                                                    !
   865)       !------------------------------------------------------------------------------------!
   866)       do i=1,ncoh
   867)          ip1 = i + 1
   868)          light_level     (i) = light_level     (i)                                         &
   869)                              + 5.d-1 * (swd(i) + swd(ip1) + beam_down(i) + beam_down(ip1))
   870)          light_beam_level(i) = light_beam_level(i) + 5.d-1 * (beam_down(i) + beam_down(ip1))
   871)          light_diff_level(i) = light_diff_level(i) + 5.d-1 * (swd(i) + swd(ip1))
   872)       end do
   873)       !------------------------------------------------------------------------------------!
   874) 
   875) 
   876) 
   877)       !------------------------------------------------------------------------------------!
   878)       !     Save the fluxes that we will use outside this sub-routine.                     !
   879)       !------------------------------------------------------------------------------------!
   880)       select case (iband)
   881)       case (1)
   882)          !---------------------------------------------------------------------------------!
   883)          !     Visible (PAR).                                                              !
   884)          !---------------------------------------------------------------------------------!
   885) 
   886)          !------ Integrate the visible light levels. --------------------------------------!
   887)          do i=1,ncoh
   888)             ip1 = i + 1
   889)             im1 = i - 1
   890)             par_level_diffd(i) = 5.d-1 * (swd(i) + swd(ip1)) /                             &
   891)                  (par_diff_norm + par_beam_norm)
   892)             par_level_diffu(i) = 5.d-1 * (swu(i) + swu(im1)) /                             &
   893)                  (par_diff_norm + par_beam_norm)
   894)             par_level_beam (i) = 5.d-1 * (beam_down(i) + beam_down(ip1)) /                 &
   895)                  (par_diff_norm+par_beam_norm)
   896)          end do
   897)          !---------------------------------------------------------------------------------!
   898) 
   899)          !------ Save the fluxes reaching the surface and leaving the top. ----------------!
   900)          dw_parlo_beam = sngloff(beam_down      (1), tiny_offset)
   901)          dw_parlo_diff = sngloff(swd            (1), tiny_offset)
   902)          uw_parhi_diff = sngloff(swu         (ncoh), tiny_offset)
   903)          !---------------------------------------------------------------------------------!
   904) 
   905) 
   906)          !---------------------------------------------------------------------------------!
   907)          !     Save the radiation fluxes to the output variable.                           !
   908)          !---------------------------------------------------------------------------------!
   909)          do i=1,ncoh
   910)             im1 = i - 1
   911)             ip1 = i + 1
   912)             par_beam_flip  (i) = sngloff(beam_down(ip1) - beam_down(i)        ,tiny_offset)
