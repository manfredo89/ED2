     3) ! Subroutine leaf_derivs                                                                   !
     4) !                                                                                          !
     5) !     This subroutine finds the fast-scale derivatives at canopy, soil, and leaf surface.  !
     6) ! This subroutine is based on LEAF-3, except that here only the derivative is computed,    !
     7) ! whereas in LEAF-3 the actual step is done at once. This derivative will be used for the  !
     8) ! Runge-Kutta integration step.                                                            !
     9) !------------------------------------------------------------------------------------------!
    10) subroutine leaf_derivs(initp,dinitp,csite,ipa,dt,is_hybrid)
    11) 
    12)    use rk4_coms               , only : rk4patchtype       ! ! structure
    13)    use ed_state_vars          , only : sitetype           & ! structure
    14)                                      , polygontype        ! ! structure
    15)    use grid_coms              , only : nzg                ! ! intent(in)
    16)    implicit none
    17)    !----- Arguments -----------------------------------------------------------------------!
    18)    type(rk4patchtype) , target     :: initp     ! Structure with RK4 intermediate state
    19)    type(rk4patchtype) , target     :: dinitp    ! Structure with RK4 derivatives
    20)    type(sitetype)     , target     :: csite     ! This site (with previous values);
    21)    integer            , intent(in) :: ipa       ! Patch ID
    22)    real(kind=8)       , intent(in) :: dt        ! Current time step
    23)    logical            , intent(in) :: is_hybrid ! Flag to tell whether it is a hybrid
    24)                                                 !    solver solution.
    25)    !---------------------------------------------------------------------------------------!
    26) 
    27)    !---------------------------------------------------------------------------------------!
    28)    ! Depending on the type of compilation, interfaces must be explicitly declared.         !
    29)    !---------------------------------------------------------------------------------------!
    30) #if USE_INTERF
    31)    interface
    32)       !------------------------------------------------------------------------------------!
    33)       !    Subroutine that computes the canopy and leaf fluxes.                            !
    34)       !------------------------------------------------------------------------------------!
    35)       subroutine leaftw_derivs(mzg,initp,dinitp,csite,ipa,dt,is_hybrid)
    36)          use rk4_coms      , only : rk4patchtype ! ! structure
    37)          use ed_state_vars , only : sitetype     & ! structure
    38)                                   , polygontype  ! ! structure
    39)          implicit none
    40)          !----- Arguments -----------------------------------------------------------------!
    41)          type(rk4patchtype)  , target     :: initp     ! RK4 structure, intermediate step
    42)          type(rk4patchtype)  , target     :: dinitp    ! RK4 structure, derivatives
    43)          type(sitetype)      , target     :: csite     ! Current site (before integration)
    44)          integer             , intent(in) :: ipa       ! Current patch ID
    45)          integer             , intent(in) :: mzg       ! Number of ground layers
    46)          real(kind=8)        , intent(in) :: dt        ! Time step
    47)          logical             , intent(in) :: is_hybrid ! Hybrid solver?
    48)       end subroutine leaftw_derivs
    49)       !------------------------------------------------------------------------------------!
    50)    end interface
    51) #endif
    52)    !---------------------------------------------------------------------------------------!
    53) 
    54)    !----- Ensure that theta_Eiv and water storage derivatives are both zero. --------------!
    55)    dinitp%ebudget_storage   = 0.d0
    56)    dinitp%wbudget_storage   = 0.d0
    57)    dinitp%co2budget_storage = 0.d0
    58)    !---------------------------------------------------------------------------------------!
    59) 
    60) 
    61)    !----- Find the derivatives. -----------------------------------------------------------!
    62)    call leaftw_derivs(nzg,initp,dinitp,csite,ipa,dt,is_hybrid)
    63)    !---------------------------------------------------------------------------------------!
    64) 
    65)    return
    66) end subroutine leaf_derivs

    77) subroutine leaftw_derivs(mzg,initp,dinitp,csite,ipa,dt,is_hybrid)
    78)    use ed_max_dims          , only : nzgmax                & ! intent(in)
    79)                                    , nzsmax                ! ! intent(in)
    80)    use consts_coms          , only : cliq8                 & ! intent(in)
    81)                                    , cph2o8                & ! intent(in)
    82)                                    , wdns8                 & ! intent(in)
    83)                                    , wdnsi8                & ! intent(in)
    84)                                    , lnexp_min8            ! ! intent(in)
    85)    use soil_coms            , only : soil8                 & ! intent(in)
    86)                                    , dslz8                 & ! intent(in)
    87)                                    , dslzi8                & ! intent(in)
    88)                                    , infiltration_method   & ! intent(in)
    89)                                    , dslzti8               & ! intent(in)
    90)                                    , slcons18              & ! intent(in)
    91)                                    , slzt8                 & ! intent(in)
    92)                                    , dslzt8                & ! intent(in)
    93)                                    , ss                    & ! intent(in)
    94)                                    , isoilbc               & ! intent(in)
    95)                                    , sin_sldrain8          & ! intent(in)
    96)                                    , matric_potential8     & ! function
    97)                                    , hydr_conduct8         ! ! function
    98)    use rk4_coms             , only : checkbudget           & ! intent(in)
    99)                                    , rk4site               & ! intent(in)
   100)                                    , rk4patchtype          & ! structure
   101)                                    , rk4aux                & ! intent(out)
   102)                                    , zero_rk4_aux          ! ! intent(in)
   103)    use ed_state_vars        , only : sitetype              & ! structure
   104)                                    , patchtype             & ! structure
   105)                                    , polygontype           ! ! structure
   106)    use therm_lib8           , only : tl2uint8              ! ! functions
   107)    use physiology_coms      , only : h2o_plant_lim         ! ! intent(in)
   108)    !$ use omp_lib
   109) 
   110)    implicit none
   111)    !----- Arguments -----------------------------------------------------------------------!
   112)    type(rk4patchtype)  , target     :: initp            ! RK4 structure, intermediate step
   113)    type(rk4patchtype)  , target     :: dinitp           ! RK4 structure, derivatives
   114)    type(sitetype)      , target     :: csite            ! Current site (before integration)
   115)    integer             , intent(in) :: ipa              ! Current patch number
   116)    integer             , intent(in) :: mzg              ! Number of ground layers
   117)    real(kind=8)        , intent(in) :: dt               ! Timestep
   118)    logical             , intent(in) :: is_hybrid        ! Hybrid solver?
   119)    !----- Local variables -----------------------------------------------------------------!
   120)    type(patchtype)     , pointer    :: cpatch           ! Current patch
   121)    integer                          :: ico              ! Cohort counter
   122)    integer                          :: k                ! Level counter
   123)    integer                          :: k1               ! Level counter
   124)    integer                          :: k2               ! Level counter
   125)    integer                          :: klsl             ! Alias for rk4site%lsl
   126)    integer                          :: kben             ! Alias for layer beneath bottom
   127)    integer                          :: ksn              ! # of temporary water/snow layers
   128)    integer                          :: nsoil            ! Short for csite%soil_text(k,ipa)
   129)    real(kind=8)                     :: snden            ! Snow/water density
   130)    real(kind=8)                     :: hflxsc           ! TSW -> canopy heat flux
   131)    real(kind=8)                     :: wflxsc           ! TSW -> canopy water flux
   132)    real(kind=8)                     :: qwflxsc          ! TSW -> canopy latent heat flux
   133)    real(kind=8)                     :: hflxgc           ! Soil -> canopy heat flux
   134)    real(kind=8)                     :: wflxgc           ! Soil -> canopy water flux
   135)    real(kind=8)                     :: qwflxgc          ! Soil -> canopy latent heat flux
   136)    real(kind=8)                     :: dewgnd           ! Dew/frost flux to ground
   137)    real(kind=8)                     :: qdewgnd          ! Dew/frost heat flux to ground
   138)    real(kind=8)                     :: ddewgnd          ! Dew/frost density flux to ground
   139)    real(kind=8)                     :: wshed_tot        ! Water shedding flux
   140)    real(kind=8)                     :: qwshed_tot       ! Energy flux due to water shedding
   141)    real(kind=8)                     :: dwshed_tot       ! Depth flux due to water shedding
   142)    real(kind=8)                     :: throughfall_tot  ! Water shedding flux
   143)    real(kind=8)                     :: qthroughfall_tot ! Energy flux due to water shedding
   144)    real(kind=8)                     :: dthroughfall_tot ! Depth flux due to water shedding
   145)    real(kind=8)                     :: wilting_factor   ! Wilting factor
   146)    real(kind=8)                     :: ext_weight       ! Layer weight for transpiration
   147)    real(kind=8)                     :: wloss            ! Water loss due to transpiration
   148)    real(kind=8)                     :: wvlmeloss        ! Water loss due to transpiration
   149)    real(kind=8)                     :: wloss_tot        ! Total water loss amongst cohorts
   150)    real(kind=8)                     :: wvlmeloss_tot    ! Total water loss amongst cohorts
   151)    real(kind=8)                     :: qloss            ! Energy loss due to transpiration
   152)    real(kind=8)                     :: qvlmeloss        ! Energy loss due to transpiration
   153)    real(kind=8)                     :: qloss_tot        ! Total energy loss amongst cohorts
   154)    real(kind=8)                     :: qvlmeloss_tot    ! Total energy loss amongst cohorts
   155)    real(kind=8)                     :: infilt           ! Surface infiltration rate
   156)    real(kind=8)                     :: qinfilt          ! Surface infiltration heat rate
   157)    real(kind=8)                     :: surface_water    ! Temp. variable. Available liquid
   158)    real(kind=8)                     :: avg_th_cond      ! Mean thermal conductivity
   159)    real(kind=8)                     :: avg_hydcond      ! Mean thermal conductivity
   160)    real(kind=8)                     :: wloss_tot_k1     ! MdP temporal parameter
   161)    real(kind=8)                     :: wloss_tot_k2     ! MdP temporal parameter
   162)    real(kind=8)                     :: uint_here        ! MdP temporal parameter
   163)    real(kind=8)                     :: uint_here1       ! MdP temporal parameter
   164)    real(kind=8)                     :: rk4aux_k1        ! MdP temporal parameter
   165)    integer                          :: ibuff            ! The shared memory processor index
   166)                                                         ! for the buffer space (privatize)
   167)    !---------------------------------------------------------------------------------------!
   168) 
   169) 
   170)    !---------------------------------------------------------------------------------------!
   171)    ! Depending on the type of compilation, interfaces must be explicitly declared.         !
   172)    !---------------------------------------------------------------------------------------!
   173) #if USE_INTERF
   174)    interface
   175)       subroutine canopy_derivs_two(mzg,initp,dinitp,csite,ipa,hflxsc,wflxsc,qwflxsc        &
   176)                                   ,hflxgc,wflxgc,qwflxgc,dewgndflx,qdewgndflx,ddewgndflx   &
   177)                                   ,throughfall_tot,qthroughfall_tot,dthroughfall_tot       &
   178)                                   ,wshed_tot,qwshed_tot,dwshed_tot,dt,is_hybrid)
   179)          use rk4_coms     , only: rk4patchtype  ! ! structure
   180)          use ed_state_vars, only: sitetype      & ! structure
   181)                                 , patchtype     & ! structure
   182)                                 , polygontype   ! ! structure
   183)          implicit none
   184)          type (rk4patchtype), target      :: initp, dinitp
   185)          type (sitetype)    , target      :: csite
   186)          integer            , intent(in)  :: ipa
   187)          integer            , intent(in)  :: mzg
   188)          real(kind=8)       , intent(in)  :: dt
   189)          logical            , intent(in)  :: is_hybrid
   190)          real(kind=8)       , intent(out) :: hflxsc
   191)          real(kind=8)       , intent(out) :: wflxsc
   192)          real(kind=8)       , intent(out) :: qwflxsc
   193)          real(kind=8)       , intent(out) :: hflxgc
   194)          real(kind=8)       , intent(out) :: wflxgc
   195)          real(kind=8)       , intent(out) :: qwflxgc
   196)          real(kind=8)       , intent(out) :: dewgndflx
   197)          real(kind=8)       , intent(out) :: qdewgndflx
   198)          real(kind=8)       , intent(out) :: ddewgndflx
   199)          real(kind=8)       , intent(out) :: wshed_tot
   200)          real(kind=8)       , intent(out) :: qwshed_tot
   201)          real(kind=8)       , intent(out) :: dwshed_tot
   202)          real(kind=8)       , intent(out) :: throughfall_tot
   203)          real(kind=8)       , intent(out) :: qthroughfall_tot
   204)          real(kind=8)       , intent(out) :: dthroughfall_tot
   205)       end subroutine canopy_derivs_two
   206)    end interface
   207) #endif
   208)    !---------------------------------------------------------------------------------------!
   209) 
   210)    ibuff = 1
   211)    !$ ibuff = OMP_get_thread_num()+1
   212) 
   213)    !----- Set the pointer to the current patch. -------------------------------------------!
   214)    cpatch => csite%patch(ipa)
   215)    !---------------------------------------------------------------------------------------!
   216) 
   217) 
   218)    !----- Copy the # of surface water/snow layers and bottom layer to aliases -------------!
   219)    ksn  = initp%nlev_sfcwater
   220)    klsl = rk4site%lsl
   221)    kben = klsl - 1
   222)    !---------------------------------------------------------------------------------------!
   223) 
   224) 
   225)    !---- Flush auxiliary variables to zero. -----------------------------------------------!
   226)    call zero_rk4_aux(rk4aux(ibuff))
   227)    !---------------------------------------------------------------------------------------!
   228) 
   229) 
   230) 
   231)    !----- Make sure derivatives are flushed to zero. --------------------------------------!
   232)    dinitp%soil_energy(:)     = 0.0d0
   233)    dinitp%soil_water(:)      = 0.0d0
   234)    dinitp%sfcwater_depth(:)  = 0.0d0
   235)    dinitp%sfcwater_energy(:) = 0.0d0
   236)    dinitp%sfcwater_mass(:)   = 0.0d0
   237)    dinitp%virtual_energy     = 0.0d0
   238)    dinitp%virtual_water      = 0.0d0
   239)    dinitp%virtual_depth      = 0.0d0
   240)    dinitp%avg_transloss(:)   = 0.0d0
   241)    !---------------------------------------------------------------------------------------!
   242) 
   243) 
   244) 
   245) 
   246)    !---------------------------------------------------------------------------------------!
   247)    !     Compute the following variables:                                                  !
   248)    !                                                                                       !
   249)    ! TH_COND_S              -- thermal conductivity - soil                        [ W/m/K] !
   250)    ! TH_COND_P              -- thermal conductivity - pounding water or snow pack [ W/m/K] !
   251)    ! HYDCOND                -- hydraulic conductivity                             [   m/s] !
   252)    ! PSIPLUSZ               -- the total potential (water + gravitational)        [     m] !
   253)    ! DRYSOIL                -- flag that tells whether this layer is totally dry  [   T|F] !
   254)    ! SATSOIL                -- flag that tells whether this layer is saturated    [   T|F] !
   255)    !---------------------------------------------------------------------------------------!
   256)    do k = klsl, mzg
   257)       nsoil                      = rk4site%ntext_soil(k)
   258)       rk4aux(ibuff)%th_cond_s(k) = ( soil8(nsoil)%thcond0                                  &
   259)                                    + soil8(nsoil)%thcond1 * initp%soil_water(k) )          &
   260)                                  / ( soil8(nsoil)%thcond2                                  &
   261)                                    + soil8(nsoil)%thcond3 * initp%soil_water(k) )
   262) 
   263) 
   264)       !----- Find the correction for (partially) frozen soil layers. ----------------------!
   265)       rk4aux(ibuff)%hydcond (k) = hydr_conduct8(k,nsoil,initp%soil_water(k)                &
   266)                                                ,initp%soil_fracliq(k))
   267)       !------------------------------------------------------------------------------------!
   268) 
   269) 
   270)       rk4aux(ibuff)%psiplusz(k) = slzt8(k) + initp%soil_mstpot(k)
   271)       rk4aux(ibuff)%drysoil (k) = (initp%soil_water(k) - soil8(nsoil)%soilcp)              &
   272)                                 * initp%soil_fracliq(k)                        <= 0.d0
   273)       rk4aux(ibuff)%satsoil (k) = initp%soil_water(k) >= soil8(nsoil)%slmsts
   274)    end do
   275)    !---------------------------------------------------------------------------------------!
   276) 
   277)    !---------------------------------------------------------------------------------------!
   278)    !      Find the thermal conductivity of the temporary surface water/snow.               !
   279)    !---------------------------------------------------------------------------------------!
   280)    do k = 1, ksn
   281)       if (initp%sfcwater_depth(k) > 0.d0 .and. initp%sfcwater_mass(k) > 0.d0) then
   282)          snden = initp%sfcwater_mass(k) / initp%sfcwater_depth(k)
   283)          rk4aux(ibuff)%th_cond_p(k) = ss(1) * exp(ss(2) * initp%sfcwater_tempk(k))         &
   284)                                 * (ss(3) + snden * (ss(4) + snden * (ss(5) + snden*ss(6))))
   285)       else
   286)          rk4aux(ibuff)%th_cond_p(k) = 0.d0
   287)       end if
   288)    end do
   289)    !---------------------------------------------------------------------------------------!
   290) 
   291) 
   292) 
   293) 
   294)    !---------------------------------------------------------------------------------------!
   295)    !     Compute the following variables:                                                  !
   296)    !                                                                                       !
   297)    ! AVAIL_H2O_LYR          -- the available water factor for this layer          [ kg/m�] !
   298)    ! AVAIL_H2O_INT          -- the integral of AVAIL_H2O down to the layer        [ kg/m�] !
   299)    !                                                                                       !
   300)    ! Both AVAIL_H2O_LYR and AVAIL_H2O_INT depend on which water limitation we are using.   !
   301)    !---------------------------------------------------------------------------------------!
   302)    select case (h2o_plant_lim)
   303)    case (0,1)
   304)       !------------------------------------------------------------------------------------!
   305)       !     The available water factor is the fraction of water mass above wilting point,  !
   306)       ! scaled by the liquid fraction.                                                     !
   307)       !------------------------------------------------------------------------------------!
   308)       do k = mzg, klsl, -1
   309)          nsoil                   = rk4site%ntext_soil(k)
   310) 
   311)          !----- Find the available water factor for this layer. ---------------------------!
   312)          rk4aux(ibuff)%avail_h2o_lyr(k) = max( 0.d0                                        &
   313)                                              , (initp%soil_water(k)-soil8(nsoil)%soilwp))  &
   314)                                         * initp%soil_fracliq(k) * wdns8 * dslz8(k)
   315)          !---------------------------------------------------------------------------------!
   316) 
   317)          !----- Add the factor from this layer to the integral. ---------------------------!
   318)          rk4aux(ibuff)%avail_h2o_int(k) = rk4aux(ibuff)%avail_h2o_int(k+1)                 &
   319)                                         + rk4aux(ibuff)%avail_h2o_lyr(k)
   320)          !---------------------------------------------------------------------------------!
   321)       end do
   322)       !------------------------------------------------------------------------------------!
   323) 
   324)    case (2)
   325)       !------------------------------------------------------------------------------------!
   326)       !     The available water factor is the soil moisture at field capacity minus wilt-  !
   327)       ! ing, scaled by the wilting factor, defined as a function of soil potential.        !
   328)       !------------------------------------------------------------------------------------!
   329)       do k = mzg, klsl, -1
   330)          nsoil                   = rk4site%ntext_soil(k)
   331) 
   332)          !----- Find the available water factor for this layer. ---------------------------!
   333)          wilting_factor          = (rk4aux(ibuff)%psiplusz(k) - soil8(nsoil)%slpotwp)      &
   334)                                  / (soil8(nsoil)%slpotfc - soil8(nsoil)%slpotwp)
   335)          rk4aux(ibuff)%avail_h2o_lyr(k) = min( 1.d0, max( 0.d0, wilting_factor ) )         &
   336)                                         * initp%soil_fracliq(k)                            &
   337)                                         * ( soil8(nsoil)%sfldcap - soil8(nsoil)%soilwp )   &
   338)                                         * wdns8 * dslz8(k)
   339)          !---------------------------------------------------------------------------------!
   340) 
   341) 
   342)          !----- Add the factor from this layer to the integral. ---------------------------!
   343)          rk4aux(ibuff)%avail_h2o_int(k) = rk4aux(ibuff)%avail_h2o_int(k+1)                 &
   344)                                         + rk4aux(ibuff)%avail_h2o_lyr(k)
   345)          !---------------------------------------------------------------------------------!
   346)       end do
   347)       !------------------------------------------------------------------------------------!
   348)    end select
   349)    !---------------------------------------------------------------------------------------!
   350) 
   351)    !---------------------------------------------------------------------------------------!
   352)    !    Find the boundary condition for total potential beneath the bottom layer.          !
   353)    !---------------------------------------------------------------------------------------!
   354)    nsoil = rk4site%ntext_soil(klsl)
   355)    select case (isoilbc)
   356)    case (0)
   357)       !------------------------------------------------------------------------------------!
   358)       !    Bedrock.  Make the potential exactly the same as the bottom layer, and the flux !
   359)       ! will be zero.                                                                      !
   360)       !------------------------------------------------------------------------------------!
   361)       initp%soil_water       (kben) = initp%soil_water   (klsl)
   362)       initp%soil_mstpot      (kben) = initp%soil_mstpot  (klsl)
   363)       initp%soil_fracliq     (kben) = initp%soil_fracliq (klsl)
   364)       rk4aux(ibuff)%th_cond_s(kben) = rk4aux(ibuff)%th_cond_s   (klsl)
   365)       rk4aux(ibuff)%hydcond  (kben) = rk4aux(ibuff)%hydcond     (klsl)
   366)       rk4aux(ibuff)%psiplusz (kben) = rk4aux(ibuff)%psiplusz    (klsl)
   367)       rk4aux(ibuff)%drysoil  (kben) = .true.
   368)       rk4aux(ibuff)%satsoil  (kben) = .true.
   369)       !------------------------------------------------------------------------------------!
   370) 
   371)    case (1)
   372)       !------------------------------------------------------------------------------------!
   373)       !     Free drainage.   Make the water potential at the layer beneath to be at the    !
   374)       ! same soil moisture as the bottom layer.                                            !
   375)       !------------------------------------------------------------------------------------!
   376)       initp%soil_water       (kben) = initp%soil_water   (klsl)
   377)       initp%soil_mstpot      (kben) = initp%soil_mstpot  (klsl)
   378)       initp%soil_fracliq     (kben) = initp%soil_fracliq (klsl)
   379)       rk4aux(ibuff)%th_cond_s(kben) = rk4aux(ibuff)%th_cond_s   (klsl)
   380)       rk4aux(ibuff)%hydcond  (kben) = rk4aux(ibuff)%hydcond     (klsl)
   381)       rk4aux(ibuff)%psiplusz (kben) = slzt8(kben) + initp%soil_mstpot(kben)
   382)       rk4aux(ibuff)%drysoil  (kben) = .false.
   383)       rk4aux(ibuff)%satsoil  (kben) = .false.
   384)       !------------------------------------------------------------------------------------!
   385) 
   386)    case (2)
   387)       !------------------------------------------------------------------------------------!
   388)       !     Lateral drainage (or reduced drainage).  Find the equivalent depth of the      !
   389)       ! layer beneath as a function of the slope (sldrain), and assume the soil moisture   !
   390)       ! and matric potential to be the same as the bottom layer.  Notice that when sldrain !
   391)       ! is zero this becomes the flat bedrock condition, and when sldrain is 90 degrees,   !
   392)       ! then it becomes free drainage.                                                     !
   393)       !------------------------------------------------------------------------------------!
   394)       initp%soil_water       (kben) = initp%soil_water   (klsl)
   395)       initp%soil_mstpot      (kben) = initp%soil_mstpot  (klsl)
   396)       initp%soil_fracliq     (kben) = initp%soil_fracliq (klsl)
   397)       rk4aux(ibuff)%th_cond_s(kben) = rk4aux(ibuff)%th_cond_s   (klsl)
   398)       rk4aux(ibuff)%hydcond  (kben) = rk4aux(ibuff)%hydcond     (klsl)
   399)       rk4aux(ibuff)%psiplusz (kben) = slzt8(klsl) - dslzt8(klsl) * sin_sldrain8            &
   400)                                     + initp%soil_mstpot(kben)
   401)       rk4aux(ibuff)%drysoil  (kben) = .false.
   402)       rk4aux(ibuff)%satsoil  (kben) = .false.
   403)       !------------------------------------------------------------------------------------!
   404) 
   405)    case (3)
   406)       !------------------------------------------------------------------------------------!
   407)       !     Aquifer.   Make the soil moisture in the layer beneath to be always saturated. !
   408)       !------------------------------------------------------------------------------------!
   409)       initp%soil_water       (kben) = soil8(nsoil)%slmsts
   410)       initp%soil_mstpot      (kben) = soil8(nsoil)%slpots
   411)       initp%soil_fracliq     (kben) = initp%soil_fracliq (klsl)
   412)       rk4aux(ibuff)%th_cond_s(kben) = ( soil8(nsoil)%thcond0                               &
   413)                                       + soil8(nsoil)%thcond1 * initp%soil_water(kben) )    &
   414)                                     / ( soil8(nsoil)%thcond2                               &
   415)                                       + soil8(nsoil)%thcond3 * initp%soil_water(kben) )
   416)       rk4aux(ibuff)%hydcond  (kben) = slcons18(kben,nsoil)
   417)       rk4aux(ibuff)%psiplusz (kben) = slzt8(kben) + initp%soil_mstpot(kben)
   418)       rk4aux(ibuff)%drysoil  (kben) = .false.
   419)       rk4aux(ibuff)%satsoil  (kben) = .false.
   420) 
   421)    end select
   422)    !---------------------------------------------------------------------------------------!
   423) 
   424) 
   425)    !---------------------------------------------------------------------------------------!
   426)    !      Get derivatives of vegetation and canopy air space variables, plus some fluxes   !
   427)    ! that will be used for soil top boundary conditions and for transpiration.             !
   428)    !---------------------------------------------------------------------------------------!
   429)    call canopy_derivs_two(mzg,initp,dinitp,csite,ipa,hflxsc,wflxsc,qwflxsc,hflxgc,wflxgc   &
   430)                          ,qwflxgc,dewgnd,qdewgnd,ddewgnd,throughfall_tot,qthroughfall_tot  &
   431)                          ,dthroughfall_tot,wshed_tot,qwshed_tot,dwshed_tot,dt,is_hybrid)
   432)    !---------------------------------------------------------------------------------------!
   433) 
   434) 
   435) 
   436)    !---------------------------------------------------------------------------------------!
   437)    !     Calculate the sensible heat fluxes find soil and sfcwater internal sensible heat  !
   438)    ! fluxes (h_flux_g and h_flux_s) [W/m2].  The convention is that fluxes are positive    !
   439)    ! when they are upwards.                                                                !
   440)    !     The average thermal conductivity is found using a log-linear interpolation        !
   441)    ! between the mid-points of the consecutive layers.                                     !
   442)    !---------------------------------------------------------------------------------------!
   443)    do k = klsl+1, mzg
   444)       avg_th_cond                 =  rk4aux(ibuff)%th_cond_s(k-1)                          &
   445)                                   *  ( rk4aux(ibuff)%th_cond_s(k)                          &
   446)                                      / rk4aux(ibuff)%th_cond_s(k-1) )                      &
   447)                                   ** ( dslz8(k-1) / ( dslz8(k-1) + dslz8(k) ) )
   448)       rk4aux(ibuff)%h_flux_g(k)   = - avg_th_cond                                          &
   449)                                     * (initp%soil_tempk(k) - initp%soil_tempk(k-1))        &
   450)                                     * dslzti8(k)
   451)       !------ Diagnostic sensible heat flux. ----------------------------------------------!
   452)       dinitp%avg_sensible_gg(k-1) = rk4aux(ibuff)%h_flux_g(k)
   453)       !------------------------------------------------------------------------------------!
   454)    end do
   455)    !---------------------------------------------------------------------------------------!
   456) 
   457) 
   458) 
   459)    !---------------------------------------------------------------------------------------!
   460)    !     If temporary water/snow layers exist, we compute them now.  Because the layers    !
   461)    ! may not cover the full area, we scale the fluxes by the fraction.                     !
   462)    !---------------------------------------------------------------------------------------!
   463)    if (ksn >= 1) then
   464)       !------------------------------------------------------------------------------------!
   465)       !     The first layer is the interface between soil and TSW.  We account for the     !
   466)       ! fluxes twice.                                                                      !
   467)       !------------------------------------------------------------------------------------!
   468)       avg_th_cond                   =  rk4aux(ibuff)%th_cond_s(mzg)                        &
   469)                                     *  ( rk4aux(ibuff)%th_cond_p(1)                        &
   470)                                        / rk4aux(ibuff)%th_cond_s(mzg) )                    &
   471)                                     ** ( dslz8(mzg)                                        &
   472)                                        / (initp%sfcwater_depth(1)+ dslz8(mzg)))
   473)       rk4aux(ibuff)%h_flux_g(mzg+1) = - avg_th_cond                                        &
   474)                                     * (initp%sfcwater_tempk(1) - initp%soil_tempk(mzg))    &
   475)                                     / (5.d-1 * initp%sfcwater_depth(1) - slzt8(mzg) )
   476)       rk4aux(ibuff)%h_flux_s(1)     = rk4aux(ibuff)%h_flux_g(mzg+1)
   477)       do k = 2,ksn
   478)          avg_th_cond                =  rk4aux(ibuff)%th_cond_p(k-1)                        &
   479)                                     *  ( rk4aux(ibuff)%th_cond_p(k)                        &
   480)                                        / rk4aux(ibuff)%th_cond_p(k-1))                     &
   481)                                     ** ( initp%sfcwater_depth(k-1)                         &
   482)                                        / ( initp%sfcwater_depth(k-1)                       &
   483)                                          + initp%sfcwater_depth(k) ) )
   484)          rk4aux(ibuff)%h_flux_s(k)  = - 2.d0 * avg_th_cond                                 &
   485)                                     * ( initp%sfcwater_tempk(k)-initp%sfcwater_tempk(k-1)) &
   486)                                     / ( initp%sfcwater_depth(k)+initp%sfcwater_depth(k-1))
   487)       end do
   488)       !------------------------------------------------------------------------------------!
   489) 
   490)       ! This is just sensible heat flux loss from top surface layer
   491)       ! The layer's energy budget (shown below) will include the other terms
   492)       ! Convention is positive up
   493)       rk4aux(ibuff)%h_flux_s        (ksn+1) = rk4aux(ibuff)%h_flux_s(ksn+1) + hflxsc
   494) 
   495)    end if
   496)    !---------------------------------------------------------------------------------------!
   497) 
   498) 
   499)    !---------------------------------------------------------------------------------------!
   500)    !      Add the irradiance and canopy fluxes.                                            !
   501)    !---------------------------------------------------------------------------------------!
   502)    dinitp%avg_sensible_gg(mzg)   = hflxgc + qwflxgc - dble(csite%rlong_g(ipa))             &
   503)                                  - dble(csite%rshort_g(ipa))
   504)    rk4aux(ibuff)%h_flux_g(mzg+1) = rk4aux(ibuff)%h_flux_g(mzg+1)                           &
   505)                                  + dinitp%avg_sensible_gg (mzg)
   506)    !---------------------------------------------------------------------------------------!
   507) 
   508) 
   509) 
   510)    !---------------------------------------------------------------------------------------!
   511)    !    Update soil U values [J/m�/s] from sensible heat, upward water vapor (latent heat) !
   512)    ! and longwave fluxes. This excludes effects of dew/frost formation, precipitation,     !
   513)    ! shedding, and percolation.                                                            !
   514)    !---------------------------------------------------------------------------------------!
   515)    do k = klsl,mzg
   516)       dinitp%soil_energy(k) = dslzi8(k) * ( rk4aux(ibuff)%h_flux_g(k)                      &
   517)                                           - rk4aux(ibuff)%h_flux_g(k+1) )
   518)    end do
   519)    !---------------------------------------------------------------------------------------!
   520) 
   521)    !---------------------------------------------------------------------------------------!
   522)    ! Update surface water U values [J/m�/s] from sensible heat and shortwave               !
   523)    ! fluxes.  This excludes effects of dew/frost, latent heat flux, precipitation,         !
   524)    ! shedding and percolation (and any mass fluxes). Right now, thermal radiation only     !
   525)    ! affects the top layer.                                                                !
   526)    !---------------------------------------------------------------------------------------!
   527)    do k = 1,ksn
   528)      dinitp%sfcwater_energy(k) = rk4aux(ibuff)%h_flux_s(k) - rk4aux(ibuff)%h_flux_s(k+1)   &
   529)                                + dble(csite%rshort_s(k,ipa))
   530)    end do
   531)    !---------------------------------------------------------------------------------------!
   532) 
   533) 
   534) 
   535) 
   536)    !---------------------------------------------------------------------------------------!
   537)    !     Calculate the fluxes of water with their associated heat fluxes. Update top soil  !
   538)    ! or snow moisture from evaporation only.                                               !
   539)    !---------------------------------------------------------------------------------------!
   540)    if ( ksn > 0 ) then
   541)       dinitp%sfcwater_mass  (ksn) =  dewgnd +  wshed_tot +  throughfall_tot -  wflxsc
   542)       dinitp%sfcwater_energy(ksn) = dinitp%sfcwater_energy(ksn) + dble(csite%rlong_s(ipa)) &
   543)                                   + qdewgnd + qwshed_tot + qthroughfall_tot - qwflxsc
   544)       dinitp%sfcwater_depth (ksn) = ddewgnd + dwshed_tot + dthroughfall_tot
   545)    else
   546)       dinitp%virtual_water        =  dewgnd +  wshed_tot +  throughfall_tot -  wflxsc
   547)       dinitp%virtual_energy       = qdewgnd + qwshed_tot + qthroughfall_tot - qwflxsc
   548)       dinitp%virtual_depth        = ddewgnd + dwshed_tot + dthroughfall_tot
   549)    end if
   550)    !---------------------------------------------------------------------------------------!
   551) 
   552) 
   553) 
   554)    !------ Diagnostic variable for water flux, bypass the virtual/sfcw layers. ------------!
   555)    dinitp%avg_smoist_gg(mzg) = rk4aux(ibuff)%w_flux_g(mzg+1)                               &
   556)                              + dewgnd +  wshed_tot +  throughfall_tot -  wflxsc -  wflxgc
   557)    !---------------------------------------------------------------------------------------!
   558) 
   559) 
   560) 
   561) 
   562)    !---------------------------------------------------------------------------------------!
   563)    !     Find amount of water transferred between soil layers (w_flux) [m] modulated by    !
   564)    ! the liquid water fraction.                                                            !
   565)    !---------------------------------------------------------------------------------------!
   566)    rk4aux(ibuff)%w_flux_g(mzg+1) = wflxgc * wdnsi8 ! now in m/s
   567)    !---------------------------------------------------------------------------------------!
   568) 
   569) 
   570) 
   571) 
   572)    !---------------------------------------------------------------------------------------!
   573)    !---------------------------------------------------------------------------------------!
   574)    !---------------------------------------------------------------------------------------!
   575)    !---------------------------------------------------------------------------------------!
   576)    !     Alternate surface infiltration (MCD) based on surface conductivity not capacity.  !
   577)    !---------------------------------------------------------------------------------------!
   578)    if (infiltration_method /= 0) then
   579)       call fatal_error ('Running alt infiltation when we shouldn''t be'                    &
   580)                        ,'leaftw_derivs','rk4_derivs.F90')
   581) 
   582)       if (initp%virtual_water /= 0.d0) then  !!process "virtural water" pool
   583)          nsoil = rk4site%ntext_soil(mzg)
   584)          if (nsoil /= 13) then
   585)             infilt = - dslzi8(mzg) * 5.d-1                                                 &
   586)                      * hydr_conduct8(mzg,nsoil,initp%soil_water(mzg)                       &
   587)                                     ,initp%soil_fracliq(mzg))                              &
   588)                      * (rk4aux(ibuff)%psiplusz(mzg)-initp%virtual_water/2.d3)              & ! diff. in pot.
   589)                      * 5.d-1 * (initp%soil_fracliq(mzg)+ initp%virtual_fracliq)            ! ! mean liquid fraction
   590)             qinfilt = infilt * wdns8 * tl2uint8(initp%virtual_tempk,1.d0)
   591)             !----- Adjust other rates accordingly -----------------------------------------!
   592)             rk4aux(ibuff)%w_flux_g (mzg+1) = rk4aux(ibuff)%w_flux_g(mzg+1)  + infilt
   593)             rk4aux(ibuff)%qw_flux_g(mzg+1) = rk4aux(ibuff)%qw_flux_g(mzg+1) + qinfilt
   594)             dinitp%virtual_water    = dinitp%virtual_water    - infilt*wdns8
   595)             dinitp%virtual_energy   = dinitp%virtual_energy   - qinfilt
   596)          end if
   597)       end if  !! end virtual water pool
   598)       if (initp%nlev_sfcwater >= 1) then !----- Process "snow" water pool -----------------!
   599)          surface_water = initp%sfcwater_mass(1)*initp%sfcwater_fracliq(1)*wdnsi8 !(m/m2)
   600)          nsoil = rk4site%ntext_soil(mzg)
   601)          if (nsoil /= 13) then
   602)             !----- Calculate infiltration rate (m/s) --------------------------------------!
   603)             infilt = - dslzi8(mzg) * 5.d-1                                                 &
   604)                      * hydr_conduct8(mzg,nsoil,initp%soil_water(mzg)                       &
   605)                                     ,initp%soil_fracliq(mzg))                              &
   606)                      * (rk4aux(ibuff)%psiplusz(mzg) - surface_water/2.d0)                  & ! difference in potentials
   607)                      * 5.d-1 * (initp%soil_fracliq(mzg) + initp%sfcwater_fracliq(1))       ! ! mean liquid fraction
   608)             qinfilt = infilt * wdns8 * tl2uint8(initp%sfcwater_tempk(1),1.d0)
   609)             !----- Adjust other rates accordingly -----------------------------------------!
   610)             rk4aux(ibuff)%w_flux_g(mzg+1)    = rk4aux(ibuff)%w_flux_g(mzg+1)    + infilt
   611)             rk4aux(ibuff)%qw_flux_g(mzg+1)   = rk4aux(ibuff)%qw_flux_g(mzg+1)   + qinfilt
   612)             dinitp%sfcwater_mass(1)   = dinitp%sfcwater_mass(1)   - infilt*wdns8
   613)             dinitp%sfcwater_energy(1) = dinitp%sfcwater_energy(1) - qinfilt
   614)             dinitp%sfcwater_depth(1)  = dinitp%sfcwater_depth(1)  - infilt
   615)          end if
   616)       end if  ! End snow water pool
   617)    end if  !! End alternate infiltration
   618)    !---------------------------------------------------------------------------------------!
   619)    !---------------------------------------------------------------------------------------!
   620)    !---------------------------------------------------------------------------------------!
   621)    !---------------------------------------------------------------------------------------!
   622) 
   623) 
   624) 
   625) 
   626)    !---------------------------------------------------------------------------------------!
   627)    !     Here we solve for all layers including the bottommost one, which is now prepared  !
   628)    ! according to the chosen boundary condition.  In this block, units are:                !
   629)    ! W_Flux  = m/s                                                                         !
   630)    ! QW_Flux = W/m2                                                                        !
   631)    !---------------------------------------------------------------------------------------!
   632)    do k = klsl, mzg
   633)       nsoil = rk4site%ntext_soil(k)
   634)       if (nsoil /= 13) then
   635) 
   636)          !----- Log-linear interpolation of hydraulic conductivity to layer interface. ----!
   637)          avg_hydcond =  rk4aux(ibuff)%hydcond(k-1)                                         &
   638)                      *  ( rk4aux(ibuff)%hydcond(k) / rk4aux(ibuff)%hydcond(k-1) )          &
   639)                      ** ( dslz8(k-1) / ( dslz8(k-1) + dslz8(k) ) )
   640)          !---------------------------------------------------------------------------------!
   641) 
   642) 
   643) 
   644)          !----- Find the potential flux. --------------------------------------------------!
   645)          rk4aux(ibuff)%w_flux_g(k) = - avg_hydcond * ( rk4aux(ibuff)%psiplusz(k)           &
   646)                                                      - rk4aux(ibuff)%psiplusz(k-1) )       &
   647)                                    * dslzti8(k)
   648)          !---------------------------------------------------------------------------------!
   649) 
   650) 
   651) 
   652)          !----- Limit water transfers to prevent over-saturation and over-depletion. ------!
   653)          if ( rk4aux(ibuff)%w_flux_g(k) >= 0. .and.                                        &
   654)              (rk4aux(ibuff)%drysoil(k-1) .or. rk4aux(ibuff)%satsoil(k)) )    then
   655)             rk4aux(ibuff)%w_flux_g(k) = 0.d0
   656) 
   657)          elseif( rk4aux(ibuff)%w_flux_g(k) < 0. .and.                                      &
   658)                 (rk4aux(ibuff)%satsoil(k-1) .or. rk4aux(ibuff)%drysoil(k)) ) then
   659)             rk4aux(ibuff)%w_flux_g(k) = 0.d0
   660) 
   661)          end if
   662)          !---------------------------------------------------------------------------------!
   663) 
   664)       else
   665)          rk4aux(ibuff)%w_flux_g(k) = 0.d0
   666)       end if
   667)       !------------------------------------------------------------------------------------!
   668) 
   669) 
   670)       !------------------------------------------------------------------------------------!
   671)       !      Find the internal energy flux associated with the water flux.  This is sign-  !
   672)       ! -dependent because the temperature must be the source temperature.                 !
   673)       !------------------------------------------------------------------------------------!
   674)       if (rk4aux(ibuff)%w_flux_g(k) > 0) then
   675)          rk4aux(ibuff)%qw_flux_g(k) = rk4aux(ibuff)%w_flux_g(k) * wdns8                    &
   676)                                     * tl2uint8(initp%soil_tempk(k-1),1.d0)
   677)       else
   678)          rk4aux(ibuff)%qw_flux_g(k) = rk4aux(ibuff)%w_flux_g(k) * wdns8                    &
   679)                                     * tl2uint8(initp%soil_tempk(k)  ,1.d0)
   680)       end if
   681)       !------------------------------------------------------------------------------------!
   682) 
   683) 
   684)       !----- Save the moisture flux in kg/m2/s. -------------------------------------------!
   685)       if (k /= 1) dinitp%avg_smoist_gg(k-1) = rk4aux(ibuff)%w_flux_g(k) * wdns8 ! Diagnostic
   686)       !------------------------------------------------------------------------------------!
   687)    end do
   688) 
   689)    !---------------------------------------------------------------------------------------!
   690)    !     Find the drainage flux.  This is simply the flux at the bottom of the bottom-     !
   691)    ! most layer.  Units are kg/m2/s for water flux and W/m2 for the associated internal    !
   692)    ! energy leak.  Notice that for the aquifer case we may actually have negative drain-   !
   693)    ! age, but that shouldn't affect the budget in any way (except that we are adding water !
   694)    ! to the system).                                                                       !
   695)    !---------------------------------------------------------------------------------------!
   696)    dinitp%avg_drainage  = - rk4aux(ibuff)%w_flux_g (klsl) * wdns8
   697)    dinitp%avg_qdrainage = - rk4aux(ibuff)%qw_flux_g(klsl)
   698)    !----- Copy the variables to the budget arrays. ----------------------------------------!
   699)    if (checkbudget) then
   700)       dinitp%wbudget_loss2drainage = dinitp%avg_drainage
   701)       dinitp%ebudget_loss2drainage = dinitp%avg_qdrainage
   702) 
   703)       dinitp%wbudget_storage = dinitp%wbudget_storage - dinitp%avg_drainage
   704)       dinitp%ebudget_storage = dinitp%ebudget_storage - dinitp%avg_qdrainage
   705)    end if
   706)    !---------------------------------------------------------------------------------------!
   707) 
   708) 
   709) 
   710) 
   711)    !----- Finally, update soil moisture and soil energy. ----------------------------------!
   712)    do k = klsl,mzg
   713)       dinitp%soil_water(k)  = dinitp%soil_water(k)                                         &
   714)                             + dslzi8(k) * (  rk4aux(ibuff)%w_flux_g(k)                     &
   715)                                           -  rk4aux(ibuff)%w_flux_g(k+1) )
   716)       dinitp%soil_energy(k) = dinitp%soil_energy(k)                                        &
   717)                             + dslzi8(k) * ( rk4aux(ibuff)%qw_flux_g(k)                     &
   718)                                           - rk4aux(ibuff)%qw_flux_g(k+1) )
   719)    end do
   720)    !---------------------------------------------------------------------------------------!
   721) 
   722) 
   723) 
   724)    !---- Update soil moisture and energy from transpiration/root uptake. ------------------!
   725)    if (rk4aux(ibuff)%any_resolvable) then
   726)       do k1 = klsl, mzg    ! loop over extracted water
   727) 
   728)          !---------------------------------------------------------------------------!
   729)          !     Transpiration happens only when there is some water left down to this !
   730)          ! layer.                                                                    !
   731)          !---------------------------------------------------------------------------!
   732) 
   733)          if (rk4aux(ibuff)%avail_h2o_int(k1) > 0.d0) then
   734) 
   735)             wloss_tot_k1 = 0.d0
   736) 
   737)             !-------- Integrate the total to be removed from this layer. ------------!
   738)             do ico=1,cpatch%ncohorts
   739)                wloss_tot_k1 = wloss_tot_k1 + rk4aux(ibuff)%extracted_water(ico,k1)
   740)             end do
   741)             !------------------------------------------------------------------------!
   742) 
   743)             rk4aux_k1 = rk4aux(ibuff)%avail_h2o_int(k1)
   744) 
   745)             do k2=k1,mzg
   746)                if (rk4site%ntext_soil(k2) /= 13) then
   747)                   !------------------------------------------------------------------------!
   748)                   !    Find the contribution of layer k2 for the transpiration from        !
   749)                   ! cohorts that reach layer k1.                                           !
   750)                   !------------------------------------------------------------------------!
   751)                   ext_weight = rk4aux(ibuff)%avail_h2o_lyr(k2) / rk4aux_k1
   752) 
   753)                   !------------------------------------------------------------------------!
   754)                   !    Find the loss of water from layer k2 due to cohorts that reach at   !
   755)                   ! least layer k1.  Here we convert extracted water which is in kg/m2/s   !
   756)                   ! (tloss) to m3/m3/s (wloss).  Also, find the internal energy loss       !
   757)                   ! (qwloss) associated with the water loss.  Since plants can extract     !
   758)                   ! liquid water only, the internal energy is assumed to be entirely in    !
   759)                   ! liquid phase.  Because the actual conversion from liquid phase to      !
   760)                   ! vapour happens at the leaf level, the internal energy must stay with   !
   761)                   ! the leaves so energy is preserved.                                     !
   762)                   !------------------------------------------------------------------------!
   763) 
   764)                   uint_here     = tl2uint8(initp%soil_tempk(k2),1.d0)
   765) 
   766)                   !------------------------------------------------------------------------!
   767)                   !      Add the internal energy to the cohort.  This energy will be       !
   768)                   ! eventually lost to the canopy air space because of transpiration,      !
   769)                   ! but we will do it in two steps so we ensure energy is conserved.       !
   770)                   !------------------------------------------------------------------------!
   771)                   do ico=1,cpatch%ncohorts
   772)                      wloss         = rk4aux(ibuff)%extracted_water(ico,k1) * ext_weight
   773)                      uint_here1    = wloss * uint_here
   774) 
   775)                      dinitp%leaf_energy(ico) = dinitp%leaf_energy(ico) + uint_here1
   776)                      dinitp%veg_energy(ico)  = dinitp%veg_energy(ico)  + uint_here1
   777)                      initp%hflx_lrsti(ico)   = initp%hflx_lrsti(ico)   + uint_here1
   778)                   end do
   779)                   !------------------------------------------------------------------------!
   780) 
   781)                   !--------- Derive the total to be removed from tthis layer --------------!
   782)                   wloss_tot     = wloss_tot_k1 * ext_weight
   783)                   wloss_tot_k2  = wloss_tot    * dslzi8(k2)
   784)                   wvlmeloss_tot = wloss_tot_k2 * wdnsi8
   785)                   qvlmeloss_tot = wloss_tot_k2 * uint_here
   786)                   !------------------------------------------------------------------------!
   787) 
   788)                   !----- Update derivatives of water, energy, and transpiration. ----------!
   789)                   dinitp%soil_water   (k2) = dinitp%soil_water(k2)    - wvlmeloss_tot
   790)                   dinitp%soil_energy  (k2) = dinitp%soil_energy(k2)   - qvlmeloss_tot
   791)                   dinitp%avg_transloss(k2) = dinitp%avg_transloss(k2) - wloss_tot
   792)                   !------------------------------------------------------------------------!
   793) 
   794) 
   795)                end if
   796)                !---------------------------------------------------------------------------!
   797)             end do
   798)             !------------------------------------------------------------------------------!
   799)          end if
   800)          !---------------------------------------------------------------------------------!
   801)       end do
   802)       !------------------------------------------------------------------------------------!
   803)    end if
   804)    !---------------------------------------------------------------------------------------!
   805) 
   806) 
   807)    return
   808) end subroutine leaftw_derivs

   892)    logical                          :: is_dew_cs         ! Test whether to add dew to soil
   893)    integer                          :: ico               ! Current cohort ID
   894)    integer                          :: ksn               ! Number of TSW layers
   895)    integer                          :: ipft              ! Shortcut for PFT type

   897)    real(kind=8)                     :: closedcan_frac    ! total fractional canopy coverage
   898)    real(kind=8)                     :: transp            ! Cohort transpiration
   899)    real(kind=8)                     :: cflxac            ! Atm->canopy carbon flux
   900)    real(kind=8)                     :: wflxac            ! Atm->canopy water flux
   901)    real(kind=8)                     :: hflxac            ! Atm->canopy sensible heat flux
   902)    real(kind=8)                     :: eflxac            ! Atm->canopy Eq. Pot. temp flux
   903)    real(kind=8)                     :: wflxlc_try        ! Intended flux leaf sfc -> canopy
   904)    real(kind=8)                     :: wflxwc_try        ! Intended flux wood sfc -> canopy
   905)    real(kind=8)                     :: shv_gradient      ! Term for psi_open/psi_closed
   906)    real(kind=8)                     :: gleaf_open        ! Net leaf conductance (open)
   907)    real(kind=8)                     :: gleaf_closed      ! Net leaf conductance (closed)
   908)    real(kind=8)                     :: hflxlc            ! Leaf->canopy heat flux
   909)    real(kind=8)                     :: hflxwc            ! Wood->canopy heat flux
   910)    real(kind=8)                     :: sigmaw            !
   911)    real(kind=8)                     :: wflxlc            ! Leaf sfc -> canopy water flux
   912)    real(kind=8)                     :: wflxwc            ! Wood sfc -> canopy water flux
   913)    real(kind=8)                     :: cflxgc            !
   914)    real(kind=8)                     :: wshed             ! Water shed from leaves
   915)    real(kind=8)                     :: qwshed            ! Internal energy of water shed
   916)    real(kind=8)                     :: dwshed            ! Depth of water shed
   917)    real(kind=8)                     :: throughfall       ! Extra throughfall from full coh.
   918)    real(kind=8)                     :: qthroughfall      ! Its internal energy
   919)    real(kind=8)                     :: dthroughfall      ! Its depth
   920)    real(kind=8)                     :: taii              !
   921)    real(kind=8)                     :: wood_shv          ! Sat. sp. hum. at wood sfc.
   922)    real(kind=8)                     :: hflxlc_tot        ! Total leaf -> CAS heat flux
   923)    real(kind=8)                     :: hflxwc_tot        ! Total wood -> CAS heat flux
   924)    real(kind=8)                     :: transp_tot        ! Total transpiration (water)
   925)    real(kind=8)                     :: qtransp_tot       ! Total transpiration (energy)
   926)    real(kind=8)                     :: cflxlc_tot        ! Total leaf -> CAS CO2 flux
   927)    real(kind=8)                     :: cflxwc_tot        ! Total wood -> CAS CO2 flux
   928)    real(kind=8)                     :: wflxlc_tot        ! Leaf -> CAS evaporation (water)
   929)    real(kind=8)                     :: wflxwc_tot        ! Wood -> CAS evaporation (water)
   930)    real(kind=8)                     :: qwflxlc_tot       ! Leaf -> CAS evaporation (energy)
   931)    real(kind=8)                     :: qwflxwc_tot       ! Wood -> CAS evaporation (energy)
   932)    real(kind=8)                     :: rho_ustar         !
   933)    real(kind=8)                     :: leaf_flux         !
   934)    real(kind=8)                     :: min_leaf_water    !
   935)    real(kind=8)                     :: max_leaf_water    !
   936)    real(kind=8)                     :: min_wood_water    !
   937)    real(kind=8)                     :: max_wood_water    !
   938)    real(kind=8)                     :: dew_now           !
   939)    real(kind=8)                     :: sfcwater_ssh      ! Specific humidity at top layer
   940)    real(kind=8)                     :: intercepted_max   ! Pot. interecepted rainfall
   941)    real(kind=8)                     :: qintercepted_max  ! Int. energy of pot. intercept.
   942)    real(kind=8)                     :: dintercepted_max  ! Depth of pot. interception
   943)    real(kind=8)                     :: intercepted_tot   ! Actual intercepted rainfall
   944)    real(kind=8)                     :: qintercepted_tot  ! Int. energy of act. intercept.
   945)    real(kind=8)                     :: leaf_intercepted  ! Leaf interception
   946)    real(kind=8)                     :: leaf_qintercepted ! Int. energy of leaf intercept.
   947)    real(kind=8)                     :: wood_intercepted  ! Leaf interception
   948)    real(kind=8)                     :: wood_qintercepted ! Int. energy of leaf intercept.
   949)    real(kind=8)                     :: qwflxlc           ! Leaf -> CAS evaporation (energy)
   950)    real(kind=8)                     :: qwflxwc           ! Wood -> CAS evaporation (energy)
   951)    real(kind=8)                     :: qtransp           ! Transpiration (energy)
   952)    real(kind=8)                     :: flux_area         ! Area between canopy and plant
   953)    real(kind=8)                     :: a,b,c0            ! Temporary variables for solving
   954)                                                          ! the CO2 ODE
   955)    real(kind=8)                     :: max_dwdt          ! Used for capping leaf evap
   956)    integer                          :: ibuff
   957)    !----- Functions -----------------------------------------------------------------------!
   958)    real(kind=4), external           :: sngloff           ! Safe dble 2 single precision
   959)    !---------------------------------------------------------------------------------------!
   960) 
   961)    ibuff = 1
   962)    !$ ibuff = OMP_get_thread_num()+1
   963) 
   964)    !----- First step, we assign the pointer for the current patch. ------------------------!
   965)    cpatch => csite%patch(ipa)
   966)    !---------------------------------------------------------------------------------------!
   967) 
   968)    !---------------------------------------------------------------------------------------!
   969)    !    Computing the fluxes from atmosphere to canopy.                                    !
   970)    !---------------------------------------------------------------------------------------!
   971)    rho_ustar = initp%can_rhos * initp%ustar                   ! Aux. variable
   972)    wflxac    = rho_ustar      * initp%qstar                   ! Water flux
   973)    eflxac    = rho_ustar      * initp%estar                   ! Enthalpy flux
   974)    cflxac    = rho_ustar      * initp%cstar * mmdryi8         ! CO2 flux [umol/m2/s]
   975)    !------ Sensible heat flux. ------------------------------------------------------------!
   976)    hflxac    = eflxac + wflxac * cph2o8 * tsupercool_vap8
   977)    !---------------------------------------------------------------------------------------!
   978) 
   979) 
   980) 
   981)    !---------------------------------------------------------------------------------------!
   982)    !     Calculate fraction of closed canopy.                                              !
   983)    !---------------------------------------------------------------------------------------!
   984)    closedcan_frac = max(0.d0,min(1.d0,1.d0-initp%opencan_frac))
   985)    !---------------------------------------------------------------------------------------!
   986) 
   987) 
   988)    !---------------------------------------------------------------------------------------!
   989)    !     Set all heat fluxes to zero, we only add them if they occur and are allowed.      !
   990)    !---------------------------------------------------------------------------------------!
   991)    hflxsc         = 0.d0
   992)    hflxgc         = 0.d0
   993)    !---------------------------------------------------------------------------------------!
   994) 
   995) 
   996)    !---------------------------------------------------------------------------------------!
   997)    !     Here we will determine the initial guess for throughfall precipitation and the    !
   998)    ! potential amount of interception.  Later we will adjust both values depending on      !
   999)    ! whether some cohorts are already full or not, or if it is fine to exceed the maximum  !
  1000)    ! amount of water that a cohort can hold.                                               !
  1001)    !---------------------------------------------------------------------------------------!
  1002)    if (rk4aux(ibuff)%any_resolvable) then
  1003)       taii = 0.d0
  1004)       cpatch => csite%patch(ipa)
  1005)       do ico = 1,cpatch%ncohorts
  1006)          taii = taii + initp%tai(ico)
  1007)       end do
  1008)       taii = 1.d0/taii
  1009) 
  1010)       !------------------------------------------------------------------------------------!
  1011)       !    If the canopy does not cover all of the ground, then it should not intercept    !
  1012)       ! all of the water.                                                                  !
  1013)       !------------------------------------------------------------------------------------!
  1014)       if (rk4site%pcpg > 0.d0) then
  1015)          if (leaf_intercept) then
  1016)             !----- Scale interception by canopy openess (MCD 01-12-09). -------------------!
  1017)             intercepted_max  = rk4site%pcpg  * closedcan_frac
  1018)             qintercepted_max = rk4site%qpcpg * closedcan_frac
  1019)             dintercepted_max = rk4site%dpcpg * closedcan_frac
  1020)          else
  1021)             !----- No interception (developer only). --------------------------------------!
  1022)             intercepted_max  = 0.d0
  1023)             qintercepted_max = 0.d0
  1024)             dintercepted_max = 0.d0
  1025)          end if
  1026)          !---------------------------------------------------------------------------------!
  1027) 
  1028) 
  1029)          !---------------------------------------------------------------------------------!
  1030)          !    The first guess for through fall is the rainfall minus the maximum           !
  1031)          ! interception.  If some of the water can't be intercepted, we will add to the    !
  1032)          ! through fall later.                                                             !
  1033)          !---------------------------------------------------------------------------------!
  1034)          throughfall_tot  = rk4site%pcpg  - intercepted_max
  1035)          qthroughfall_tot = rk4site%qpcpg - qintercepted_max
  1036)          dthroughfall_tot = rk4site%dpcpg - dintercepted_max
  1037)          !---------------------------------------------------------------------------------!
  1038)       else
  1039)          !----- No precipitation, nothing to be intercepted... ----------------------------!
  1040)          intercepted_max  = 0.d0
  1041)          qintercepted_max = 0.d0
  1042)          dintercepted_max = 0.d0
  1043)          throughfall_tot  = 0.d0
  1044)          qthroughfall_tot = 0.d0
  1045)          dthroughfall_tot = 0.d0
  1046)          !---------------------------------------------------------------------------------!
  1047)       end if
  1048)       !------------------------------------------------------------------------------------!
  1049) 
  1050)    else
  1051)       !------------------------------------------------------------------------------------!
  1052)       !     If the TAI is very small or total patch vegetation heat capacity is too        !
  1053)       ! small, bypass vegetation computations.  Set throughfall precipitation heat and     !
  1054)       ! moisture to unintercepted values.                                                  !
  1055)       !------------------------------------------------------------------------------------!
  1056)       intercepted_max  = 0.d0
  1057)       qintercepted_max = 0.d0
  1058)       dintercepted_max = 0.d0
  1059)       throughfall_tot  = rk4site%pcpg
  1060)       qthroughfall_tot = rk4site%qpcpg
  1061)       dthroughfall_tot = rk4site%dpcpg
  1062) 
  1063)       !------------------------------------------------------------------------------------!
  1064)       ! Note: If the condition of low TAI for the entire patch was met, then it does not   !
  1065)       !       matter what the individual cohorts are normalized by, because they are       !
  1066)       !       effectively zero. So make sure the inverse patch TAI is a nominal non-zero/  !
  1067)       !       non-infinite number. This will only be used when parsing out intercepted     !
  1068)       !       leaf water into shed water; in which case the intercepted water is zero      !
  1069)       !       anyway. So this is just to prevent FPEs.                                     !
  1070)       !------------------------------------------------------------------------------------!
  1071)       taii = 0.d0
  1072)       do ico = 1,cpatch%ncohorts
  1073)          taii = taii + initp%tai(ico)
  1074)       end do
  1075)       if (taii == 0.d0) then
  1076)          taii = 1.d0
  1077)       else
  1078)          taii = 1.d0/taii
  1079)       end if
  1080)    end if
  1081)    !---------------------------------------------------------------------------------------!
  1082)    !     Compute sensible heat and moisture fluxes between the ground and the canopy air   !
  1083)    ! space.  The ground may be either the top soil layer or the temporary surface water    !
  1084)    ! snow surface.  First we check whether the vapour fluxes to ground are positive or     !
  1085)    ! negative.  Later, three variables will define the flux between ground and canopy air  !
  1086)    ! space.  They are either positive or zero:                                             !
  1087)    !                                                                                       !
  1088)    ! - wflxsc [kg/m2/s] is going to be the evaporation flux from pounding water or         !
  1089)    !                    snowpack to canopy.                                                !
  1090)    ! - wflxgc [kg/m2/s] is going to be the evaporation flux from top soil layer to canopy. !
  1091)    ! - dewgnd [kg/m2/s] is going to be the dew/frost flux from canopy to ground (this      !
  1092)    !                    encompasses dew formation on both because dew always goes to the   !
  1093)    !                    virtual layer).                                                    !
  1094)    !                                                                                       !
  1095)    !     Sensible heat is defined by two variables, hflxgc (soil) and hflxsc (TSW)         !
  1096)    ! [J/m2/s], which can be either positive or negative.                                   !
  1097)    !---------------------------------------------------------------------------------------!
  1098)    ksn = initp%nlev_sfcwater
  1099)    if (ksn > 0) then
  1100)       hflxsc       = initp%snowfac * initp%ggnet * initp%can_rhos * initp%can_cp           &
  1101)                    * (initp%sfcwater_tempk(ksn) - initp%can_temp)
  1102)       sfcwater_ssh = qslif8(initp%can_prss,initp%sfcwater_tempk(ksn))
  1103)       is_dew_cp    = sfcwater_ssh <= initp%can_shv
  1104)    else
  1105)       hflxsc       = 0.d0
  1106)       sfcwater_ssh = initp%can_shv
  1107)       is_dew_cp    = .false.
  1108)    end if
  1109)    hflxgc    = (1.d0 - initp%snowfac) * initp%ggnet * initp%can_rhos                       &
  1110)              * initp%can_cp * (initp%ground_temp - initp%can_temp)
  1111)    is_dew_cs = initp%ground_ssh <= initp%can_shv
  1112)    !---------------------------------------------------------------------------------------!
  1113) 
  1114) 
  1115) 
  1116)    !---------------------------------------------------------------------------------------!
  1117)    !     Set all water fluxes to zero, we only add them if they occur and are allowed.     !
  1118)    !---------------------------------------------------------------------------------------!
  1119)    dewgndflx         = 0.d0
  1120)    qdewgndflx        = 0.d0
  1121)    ddewgndflx        = 0.d0
  1122)    wflxsc            = 0.d0
  1123)    qwflxsc           = 0.d0
  1124)    wflxgc            = 0.d0
  1125)    qwflxgc           = 0.d0
  1126)    initp%flag_wflxgc = 0
  1127)    !---------------------------------------------------------------------------------------!
  1128) 
  1129) 
  1130) 
  1131)    !---------------------------------------------------------------------------------------!
  1132)    !     Here we will decide how to compute the evaporation and condensation fluxes        !
  1133)    ! between temporary surface water and canopy air space, based on the sign of water      !
  1134)    ! flux.                                                                                 !
  1135)    !---------------------------------------------------------------------------------------!
  1136)    if (is_dew_cp) then
  1137)       !----- Dew should be formed due to contact between top TSW and canopy air space. ----!
  1138)       dew_now           = initp%snowfac * initp%ggnet * initp%can_rhos                     &
  1139)                         * ( initp%can_shv - sfcwater_ssh )
  1140)       dewgndflx         = dewgndflx  + dew_now
  1141)       qdewgndflx        = qdewgndflx                                                       &
  1142)                         + dew_now * tq2enthalpy8(initp%sfcwater_tempk(ksn),1.d0,.true.)
  1143)       ddewgndflx        = ddewgndflx                                                       &
  1144)                         + dew_now * (          initp%sfcwater_fracliq(ksn)   * wdnsi8      &
  1145)                                     + ( 1.d0 - initp%sfcwater_fracliq(ksn) ) * fdnsi8 )
  1146)       !----- Set flux flag. ---------------------------------------------------------------!
  1147)       initp%flag_wflxgc = initp%flag_wflxgc + 1
  1148)       !------------------------------------------------------------------------------------!
  1149)    elseif (ksn > 0 .and. ( initp%can_rhv < 1.d0 .or. supersat_ok )) then
  1150)       !------------------------------------------------------------------------------------!
  1151)       !     Evaporation should occur, and canopy air space is not yet super-saturated (or  !
  1152)       ! the user is fine with super-saturation).                                           !
  1153)       !------------------------------------------------------------------------------------!
  1154)       wflxsc            = initp%snowfac * initp%ggnet * initp%can_rhos                     &
  1155)                         * ( sfcwater_ssh - initp%can_shv )
  1156)       qwflxsc           = wflxsc * tq2enthalpy8(initp%sfcwater_tempk(ksn),1.d0,.true.)
  1157)       !----- Set flux flag. ---------------------------------------------------------------!
  1158)       initp%flag_wflxgc = initp%flag_wflxgc + 2
  1159)       !------------------------------------------------------------------------------------!
  1160)    end if
  1161)    !---------------------------------------------------------------------------------------!
  1162) 
  1163) 
  1164) 
  1165)    !---------------------------------------------------------------------------------------!
  1166)    !     Similarly, we will decide how to compute the evaporation and condensation fluxes  !
  1167)    ! between top soil layer and canopy air space, based on the sign of water flux.         !
  1168)    !---------------------------------------------------------------------------------------!
  1169)    if (is_dew_cs) then
  1170)       !----- Dew should be formed due to contact between top soil and canopy air space. ---!
  1171)       dew_now           = ( 1.d0 - initp%snowfac ) * initp%ggnet * initp%can_rhos          &
  1172)                         * ( initp%can_shv - initp%ground_ssh )
  1173)       dewgndflx         = dewgndflx  + dew_now
  1174)       qdewgndflx        = qdewgndflx                                                       &
  1175)                         + dew_now * tq2enthalpy8(initp%ground_temp,1.d0,.true.)
  1176)       ddewgndflx        = ddewgndflx                                                       &
  1177)                         + dew_now * (          initp%ground_fliq   * wdnsi8                &
  1178)                                     + ( 1.d0 - initp%ground_fliq ) * fdnsi8 )
  1179)       !----- Set flux flag. ---------------------------------------------------------------!
  1180)       initp%flag_wflxgc = initp%flag_wflxgc + 4
  1181)       !------------------------------------------------------------------------------------!
  1182) 
  1183)    else if (rk4aux(ibuff)%drysoil(mzg)) then
  1184)       !------------------------------------------------------------------------------------!
  1185)       !    There should be evaporation, except that there is no water left to be extracted !
  1186)       ! from the ground... Set both evaporation and condensation fluxes to zero.           !
  1187)       !------------------------------------------------------------------------------------!
  1188)       initp%flag_wflxgc = initp%flag_wflxgc + 7
  1189)       !------------------------------------------------------------------------------------!
  1190) 
  1191)    elseif (initp%can_rhv < 1.d0 .or. supersat_ok) then
  1192)       !------------------------------------------------------------------------------------!
  1193)       !     Evaporation should occur, and canopy air space is not yet super-saturated (or  !
  1194)       ! the user is fine with super-saturation).                                           !
  1195)       !------------------------------------------------------------------------------------!
  1196)       wflxgc            = ( 1.d0 - initp%snowfac ) * initp%ggnet * initp%can_rhos          &
  1197)                         * ( initp%ground_ssh - initp%can_shv )                             &
  1198)                         * ( 1.d0 / (1.d0 + initp%ggnet / initp%ggsoil) )
  1199)       qwflxgc           = wflxgc * tq2enthalpy8(initp%ground_temp,1.d0,.true.)
  1200)       !----- Set flux flag. ---------------------------------------------------------------!
  1201)       initp%flag_wflxgc = initp%flag_wflxgc + 10
  1202)       !------------------------------------------------------------------------------------!
  1203) 
  1204)    end if
  1205)    !---------------------------------------------------------------------------------------!
  1206) 
  1207) 
  1208) 
  1209) 
  1210)    !---------------------------------------------------------------------------------------!
  1211)    ! The implicit solver needs to know the mass flux from ground to canopy.                !
  1212)    !---------------------------------------------------------------------------------------!
  1213)    initp%hflxsc  = hflxsc
  1214)    initp%wflxsc  = wflxsc
  1215)    initp%qwflxsc = qwflxsc
  1216)    initp%hflxgc  = hflxgc
  1217)    initp%wflxgc  = wflxgc
  1218)    initp%qwflxgc = qwflxgc
  1219)    !---------------------------------------------------------------------------------------!
  1220) 
  1221) 
  1222)    !---------------------------------------------------------------------------------------!
  1223)    !     Loop over the cohorts in the patch. Calculate energy fluxes with surrounding      !
  1224)    ! canopy air space, integrate cohort energy, calculate precipitation throughfall and    !
  1225)    ! sum fluxes to the patch level. Initialize variables used to store sums over cohorts.  !
  1226)    !---------------------------------------------------------------------------------------!
  1227)    hflxlc_tot       = 0.d0
  1228)    wflxlc_tot       = 0.d0
  1229)    qwflxlc_tot      = 0.d0
  1230)    transp_tot       = 0.d0
  1231)    qtransp_tot      = 0.d0
  1232)    wshed_tot        = 0.d0
  1233)    qwshed_tot       = 0.d0
