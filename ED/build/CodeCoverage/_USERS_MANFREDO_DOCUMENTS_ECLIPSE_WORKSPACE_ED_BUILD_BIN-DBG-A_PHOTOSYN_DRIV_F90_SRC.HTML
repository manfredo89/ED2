     1) module photosyn_driv
     2)   contains
     3) 
     4) !==========================================================================================!
     5) !==========================================================================================!
     6) !     This subroutine will control the photosynthesis scheme (Farquar and Leuning).  This  !
     7) ! is called every step, but not every sub-step.                                            !
     8) !------------------------------------------------------------------------------------------!
     9) subroutine canopy_photosynthesis(csite,cmet,mzg,ipa,ntext_soil                         &
    10)                                 ,leaf_aging_factor,green_leaf_factor)
    11)    use ed_state_vars  , only : sitetype           & ! structure
    12)                              , patchtype          ! ! structure
    13)    use ed_max_dims    , only : n_pft              ! ! intent(in)
    14)    use pft_coms       , only : water_conductance  & ! intent(in)
    15)                              , include_pft        & ! intent(in)
    16)                              , vm0                & ! intent(in)
    17)                              , leaf_turnover_rate ! ! intent(in)
    18)    use soil_coms      , only : soil               & ! intent(in)
    19)                              , slzt               & ! intent(in)
    20)                              , dslz               ! ! intent(in)
    21)    use consts_coms    , only : t00                & ! intent(in)
    22)                              , epi                & ! intent(in)
    23)                              , wdnsi              & ! intent(in)
    24)                              , wdns               & ! intent(in)
    25)                              , umols_2_kgCyr      & ! intent(in)
    26)                              , yr_day             & ! intent(in)
    27)                              , lnexp_min          & ! intent(in)
    28)                              , tiny_num           ! ! intent(in)
    29)    use ed_misc_coms   , only : dtlsm              & ! intent(in)
    30)                              , frqsum             ! ! intent(in)
    31)    use met_driver_coms, only : met_driv_state     ! ! structure
    32)    use physiology_coms, only : print_photo_debug  & ! intent(in)
    33)                              , h2o_plant_lim      ! ! intent(in)
    34)    use phenology_coms , only : llspan_inf         ! ! intent(in)
    35)    use farq_leuning   , only : lphysiol_full      ! ! sub-routine
    36)    use allometry      , only : h2crownbh          ! ! function
    37)    use therm_lib      , only : qslif              ! ! function
    38)    implicit none
    39)    !----- Arguments -----------------------------------------------------------------------!
    40)    type(sitetype)            , target      :: csite             ! Current site
    41)    type(met_driv_state)      , target      :: cmet              ! Current met. conditions.
    42)    integer                   , intent(in)  :: ipa               ! Current patch #
    43)    integer                   , intent(in)  :: mzg               ! Number of soil layers
    44)    integer, dimension(mzg)   , intent(in)  :: ntext_soil        ! Soil class
    45)    real   , dimension(n_pft) , intent(in)  :: leaf_aging_factor !
    46)    real   , dimension(n_pft) , intent(in)  :: green_leaf_factor !
    47)    !----- Local variables -----------------------------------------------------------------!
    48)    type(patchtype)           , pointer     :: cpatch             ! Current site
    49)    integer                                 :: ico                ! Current cohort #
    50)    integer                                 :: tuco               ! Tallest used cohort
    51)    integer                                 :: ipft
    52)    integer                                 :: tpft
    53)    integer                                 :: k
    54)    integer                                 :: kroot
    55)    integer                                 :: nsoil
    56)    integer                                 :: limit_flag
    57)    logical, dimension(mzg+1)               :: root_depth_indices !
    58)    logical                                 :: las
    59)    real   , dimension(:)    , allocatable  :: avail_h2o_coh
    60)    real                                    :: leaf_par
    61)    real                                    :: leaf_resp
    62)    real                                    :: d_A_light_max
    63)    real                                    :: d_A_rubp_max
    64)    real                                    :: d_A_co2_max
    65)    real                                    :: d_gsw_open
    66)    real                                    :: d_gsw_closed
    67)    real                                    :: d_lsfc_shv_open
    68)    real                                    :: d_lsfc_shv_closed
    69)    real                                    :: d_lsfc_co2_open
    70)    real                                    :: d_lsfc_co2_closed
    71)    real                                    :: d_lint_co2_open
    72)    real                                    :: d_lint_co2_closed
    73)    real                                    :: vm
    74)    real                                    :: mcheight
    75)    real                                    :: compp
    76)    real                                    :: broot_tot
    77)    real                                    :: broot_loc
    78)    real                                    :: water_demand
    79)    real                                    :: psiplusz
    80)    real                                    :: avail_h2o_lyr
    81)    real                                    :: wilting_factor
    82)    real                                    :: pss_available_water
    83)    real                                    :: vm0_tuco
    84)    real                                    :: llspan_tuco
    85)    real                                    :: can_ssh
    86)    integer, dimension(n_pft)               :: tuco_pft
    87)    !----- Locally saved variables. --------------------------------------------------------!
    88)    real                          , save    :: dtlsm_o_frqsum
    89)    logical                       , save    :: first_time = .true.
    90)    !---------------------------------------------------------------------------------------!
    91) 
    92) 
    93)    !----- Assign the constant scaling factor. ---------------------------------------------!
    94)    if (first_time) then
    95)       first_time     = .false.
    96)       dtlsm_o_frqsum = dtlsm / frqsum
    97)    end if
    98)    !---------------------------------------------------------------------------------------!
    99) 
   100) 
   101)    !----- Point to the cohort structures --------------------------------------------------!
   102)    cpatch => csite%patch(ipa)
   103)    !---------------------------------------------------------------------------------------!
   104) 
   105) 
   106) 
   107)    !----- Allocate the available water function for plants. -------------------------------!
   108)    if (cpatch%ncohorts > 0) then
   109)       allocate (avail_h2o_coh(cpatch%ncohorts))
   110)    end if
   111)    !---------------------------------------------------------------------------------------!
   112) 
   113) 
   114)    !---------------------------------------------------------------------------------------!
   115)    !     Calculate liquid water available for transpiration.   The way this is done        !
   116)    ! depends on how the water limitation is to be solved.                                  !
   117)    !---------------------------------------------------------------------------------------!
   118)    select case (h2o_plant_lim)
   119)    case (0,1)
   120)       !------------------------------------------------------------------------------------!
   121)       !     Available water is defined as the soil moisture (mass) above wilting point,    !
   122)       ! scaled by liquid water fraction.                                                   !
   123)       !------------------------------------------------------------------------------------!
   124)       do ico = 1, cpatch%ncohorts
   125)          !---- Aliases for rooting depth and PFT. -----------------------------------------!
   126)          kroot = cpatch%krdepth(ico)
   127)          ipft  = cpatch%pft(ico)
   128)          !---------------------------------------------------------------------------------!
   129) 
   130) 
   131)          !---------------------------------------------------------------------------------!
   132)          !     Find the available water for each layer.                                    !
   133)          !---------------------------------------------------------------------------------!
   134)          avail_h2o_coh(ico) = 0.
   135)          do k=mzg,kroot,-1
   136)             !----- Alias for soil type. ---------------------------------------------------!
   137)             nsoil = ntext_soil(k)
   138)             !------------------------------------------------------------------------------!
   139) 
   140) 
   141) 
   142)             !----- Find the available water factor for this layer. ------------------------!
   143)             avail_h2o_lyr = max(0.0, (csite%soil_water(k,ipa) - soil(nsoil)%soilwp))       &
   144)                           * csite%soil_fracliq(k,ipa) * wdns * dslz(k)
   145)             !------------------------------------------------------------------------------!
   146) 
   147) 
   148) 
   149)             !----- Add the factor from this layer to the integral. ------------------------!
   150)             avail_h2o_coh(ico) = avail_h2o_coh(ico) + avail_h2o_lyr
   151)             !------------------------------------------------------------------------------!
   152)          end do
   153)       end do
   154)       !------------------------------------------------------------------------------------!
   155) 
   156)    case (2)
   157)       !------------------------------------------------------------------------------------!
   158)       !     The available water factor is the soil moisture at field capacity minus wilt-  !
   159)       ! ing point, scaled by the wilting factor, defined as a function of soil potential   !
   160)       ! and height between roots and mid-crown.                                            !
   161)       !------------------------------------------------------------------------------------!
   162)       do ico = 1,cpatch%ncohorts
   163)          !---- Aliases for rooting depth and PFT. -----------------------------------------!
   164)          kroot = cpatch%krdepth(ico)
   165)          ipft  = cpatch%pft(ico)
   166)          !---------------------------------------------------------------------------------!
   167) 
   168) 
   169) 
   170)          !---------------------------------------------------------------------------------!
   171)          !     Find the mean height of the crown (to represent the distance between        !
   172)          ! the ground and the leaves.                                                      !
   173)          !---------------------------------------------------------------------------------!
   174)          mcheight = 0.5 * (cpatch%hite(ico) + h2crownbh(cpatch%hite(ico),ipft))
   175)          !---------------------------------------------------------------------------------!
   176) 
   177) 
   178)          !---------------------------------------------------------------------------------!
   179)          !     Find the available water for each layer.                                    !
   180)          !---------------------------------------------------------------------------------!
   181)          avail_h2o_coh(ico) = 0.
   182)          do k = mzg, kroot, -1
   183)             !----- Alias for soil type. ---------------------------------------------------!
   184)             nsoil = ntext_soil(k)
   185)             !------------------------------------------------------------------------------!
   186) 
   187) 
   188)             !----- Find the potential for this layer. -------------------------------------!
   189)             psiplusz = slzt(k) - mcheight + csite%soil_mstpot(k,ipa)
   190)             !------------------------------------------------------------------------------!
   191) 
   192) 
   193)             !----- Find the available water factor for this layer. ------------------------!
   194)             wilting_factor   = (psiplusz - soil(nsoil)%slpotwp)                            &
   195)                              / (soil(nsoil)%slpotfc - soil(nsoil)%slpotwp)
   196)             avail_h2o_lyr    = min( 1.0, max( 0.0, wilting_factor ) )                      &
   197)                              * csite%soil_fracliq(k,ipa)                                   &
   198)                              * ( soil(nsoil)%sfldcap - soil(nsoil)%soilwp )                &
   199)                              * wdns * dslz(k)
   200)             !------------------------------------------------------------------------------!
   201) 
   202) 
   203) 
   204)             !----- Add the factor from this layer to the integral. ------------------------!
   205)             avail_h2o_coh(ico) = avail_h2o_coh(ico) + avail_h2o_lyr
   206)             !------------------------------------------------------------------------------!
   207)          end do
   208)          !---------------------------------------------------------------------------------!
   209)       end do
   210)       !------------------------------------------------------------------------------------!
   211)    end select
   212)    !---------------------------------------------------------------------------------------!
   213) 
   214) 
   215) 
   216)    !---------------------------------------------------------------------------------------!
   217)    !     Initialize the array of maximum photosynthesis rates used in the mortality        !
   218)    ! function.                                                                             !
   219)    !---------------------------------------------------------------------------------------!
   220)    csite%A_o_max(1:n_pft,ipa) = 0.0
   221)    csite%A_c_max(1:n_pft,ipa) = 0.0
   222)    !---------------------------------------------------------------------------------------!
   223) 
   224) 
   225) 
   226)    !---------------------------------------------------------------------------------------!
   227)    !     Find the tallest cohort amongst all cohorts in this patch that is resolved        !
   228)    ! (tuco).  In addition, we must find the tallest cohort for each PFT, so in case the    !
   229)    ! we are using the light phenology, we use that value for Vm0 and leaf life span.       !
   230)    !---------------------------------------------------------------------------------------!
   231)    las         = .false.
   232)    tuco_pft(:) = 0
   233)    do ico = 1,cpatch%ncohorts
   234)       ipft = cpatch%pft(ico)
   235) 
   236)       !----- If this is the tallest cohort to be used, we save its index. -----------------!
   237)       if (.not. las .and. cpatch%leaf_resolvable(ico)) then
   238)          las  = .true.
   239)          tuco = ico
   240)       end if
   241)       !------------------------------------------------------------------------------------!
   242) 
   243) 
   244) 
   245)       !----- If this is the tallest cohort for this specific PFT, we save the index too. --!
   246)       if (tuco_pft(ipft) == 0 .and. cpatch%leaf_resolvable(ico)) then
   247)          tuco_pft(ipft) = ico
   248)       end if
   249)       !------------------------------------------------------------------------------------!
   250)    end do
   251)    !---------------------------------------------------------------------------------------!
   252) 
   253) 
   254) 
   255)    !---------------------------------------------------------------------------------------!
   256)    !    There is at least one cohort that meet requirements.  And this is tallest one, so  !
   257)    ! we can use it to compute the maximum photosynthetic rates, i.e., the rate the cohort  !
   258)    ! would have if it were at the top of the canopy.  This is used for the mortality       !
   259)    ! function.                                                                             !
   260)    !---------------------------------------------------------------------------------------!
   261)    if (las) then
   262)       !----- We now loop over PFTs, not cohorts, skipping those we are not using. ---------!
   263)       do ipft = 1, n_pft
   264) 
   265)          if (include_pft(ipft)) then
   266) 
   267)             !------------------------------------------------------------------------------!
   268)             !      Find the tallest cohort for this PFT.  In case the patch no longer has  !
   269)             ! the PFT, then we just the default Vm0 and leaf life span.  This only matters !
   270)             ! for light-controlled phenology, not the standard cases.                      !
   271)             !------------------------------------------------------------------------------!
   272)             tpft = tuco_pft(ipft)
   273)             if (tpft == 0) then
   274)                !---------------------------------------------------------------------------!
   275)                !    This patch doesn't have any cohort of this PFT left, use default       !
   276)                ! values.                                                                   !
   277)                !---------------------------------------------------------------------------!
   278)                vm0_tuco    = Vm0(ipft)
   279)                if (leaf_turnover_rate(ipft) == 0.) then
   280)                   llspan_tuco = llspan_inf
   281)                else
   282)                   llspan_tuco = 12. / leaf_turnover_rate(ipft)
   283)                end if
   284)             else
   285)                !---------------------------------------------------------------------------!
   286)                !    Use Vm0 and leaf life span of the tallest cohort of this PFT, so we    !
   287)                ! avoid punishing or helping the plants too much in case the PFTs don't     !
   288)                ! match.                                                                    !
   289)                !---------------------------------------------------------------------------!
   290)                vm0_tuco    = cpatch%vm_bar(tpft)
   291)                llspan_tuco = cpatch%llspan(tpft)
   292)             end if
   293)             !------------------------------------------------------------------------------!
   294) 
   295) 
   296)             !------------------------------------------------------------------------------!
   297)             !    Find the 100% relative humidity.  This is a temporary test to make the    !
   298)             ! maximum carbon balance less negative.                                        !
   299)             !------------------------------------------------------------------------------!
   300)             can_ssh = csite%can_shv(ipa)
   301)             !------------------------------------------------------------------------------!
   302) 
   303) 
   304)             !------------------------------------------------------------------------------!
   305)             !    Call the photosynthesis for maximum photosynthetic rates.  The units      !
   306)             ! of the input and output are the standard in most of ED modules, but many of  !
   307)             ! them are converted inside the photosynthesis model.                          !
   308)             !    Notice that the units that are per unit area are per m� of leaf, not the  !
   309)             ! patch area.                                                                  !
   310)             !------------------------------------------------------------------------------!
   311)             call lphysiol_full(            & !
   312)                csite%can_prss(ipa)         & ! Canopy air pressure              [       Pa]
   313)              , csite%can_rhos(ipa)         & ! Canopy air density               [    kg/m�]
   314)              , can_ssh                     & ! Canopy air sp. humidity          [    kg/kg]
   315)              , csite%can_co2(ipa)          & ! Canopy air CO2 mixing ratio      [ �mol/mol]
   316)              , ipft                        & ! Plant functional type            [      ---]
   317)              , csite%par_l_max(ipa)        & ! Absorbed photos. active rad.     [ W/m�leaf]
   318)              , cpatch%leaf_temp(tuco)      & ! Leaf temperature                 [        K]
   319)              , cpatch%lint_shv(tuco)       & ! Leaf intercellular spec. hum.    [    kg/kg]
   320)              , green_leaf_factor(ipft)     & ! Greenness rel. to on-allometry   [      ---]
   321)              , leaf_aging_factor(ipft)     & ! Ageing parameter to scale VM     [      ---]
   322)              , llspan_tuco                 & ! Leaf life span                   [       yr]
   323)              , vm0_tuco                    & ! Average Vm function              [�mol/m�/s]
   324)              , cpatch%leaf_gbw(tuco)       & ! Aerodyn. condct. of water vapour [  kg/m�/s]
   325)              , csite%A_o_max(ipft,ipa)     & ! Photosynthesis rate     (open)   [�mol/m�/s]
   326)              , csite%A_c_max(ipft,ipa)     & ! Photosynthesis rate     (closed) [�mol/m�/s]
   327)              , d_A_light_max               & ! Photosynthesis rate     (light)  [�mol/m�/s]
   328)              , d_A_rubp_max                & ! Photosynthesis rate     (RuBP)   [�mol/m�/s]
   329)              , d_A_co2_max                 & ! Photosynthesis rate     (CO2)    [�mol/m�/s]
   330)              , d_gsw_open                  & ! Stom. condct. of water  (open)   [  kg/m�/s]
   331)              , d_gsw_closed                & ! Stom. condct. of water  (closed) [  kg/m�/s]
   332)              , d_lsfc_shv_open             & ! Leaf sfc. sp. humidity  (open)   [    kg/kg]
   333)              , d_lsfc_shv_closed           & ! Leaf sfc. sp. humidity  (closed) [    kg/kg]
   334)              , d_lsfc_co2_open             & ! Leaf sfc. CO2 mix. rat. (open)   [ �mol/mol]
   335)              , d_lsfc_co2_closed           & ! Leaf sfc. CO2 mix. rat. (closed) [ �mol/mol]
   336)              , d_lint_co2_open             & ! Intercellular CO2       (open)   [ �mol/mol]
   337)              , d_lint_co2_closed           & ! Intercellular CO2       (closed) [ �mol/mol]
   338)              , leaf_resp                   & ! Leaf respiration rate            [�mol/m�/s]
   339)              , vm                          & ! Max. capacity of Rubisco         [�mol/m�/s]
   340)              , compp                       & ! Gross photo. compensation point  [ �mol/mol]
   341)              , limit_flag                  & ! Photosynthesis limitation flag   [      ---]
   342)              )
   343)          end if
   344)       end do
   345) 
   346)    else
   347)       !---- There is no "active" cohort. --------------------------------------------------!
   348)       csite%A_o_max(1:n_pft,ipa) = 0.0
   349)       csite%A_c_max(1:n_pft,ipa) = 0.0
   350)    end if
   351)    !---------------------------------------------------------------------------------------!
   352) 
   353) 
   354) 
   355)    !---------------------------------------------------------------------------------------!
   356)    !    Initialize some variables.                                                         !
   357)    !---------------------------------------------------------------------------------------!
   358)    !----- Total root biomass (in kgC/m2) and patch sum available water. -------------------!
   359)    pss_available_water = 0.0
   360)    broot_tot           = 0.0
   361)    !----- Initialize variables for transpiration calculation. -----------------------------!
   362)    root_depth_indices(:) = .false.
   363)    !---------------------------------------------------------------------------------------!
   364) 
   365)    !---------------------------------------------------------------------------------------!
   366)    !    Loop over all cohorts, from tallest to shortest.                                   !
   367)    !---------------------------------------------------------------------------------------!
   368)    cohortloop: do ico = 1,cpatch%ncohorts
   369) 
   370)       !------------------------------------------------------------------------------------!
   371)       !     Only need to worry about photosyn if radiative transfer has been  done for     !
   372)       ! this cohort.                                                                       !
   373)       !------------------------------------------------------------------------------------!
   374)       if (cpatch%leaf_resolvable(ico)) then
   375) 
   376)             !----- Alias for PFT and root layer. ------------------------------------------!
   377)             ipft  = cpatch%pft(ico)
   378)             kroot = cpatch%krdepth(ico)
   379) 
   380)             !------------------------------------------------------------------------------!
   381)             !    Scale photosynthetically active radiation per unit of leaf.               !
   382)             !------------------------------------------------------------------------------!
   383)             leaf_par = cpatch%par_l(ico) / cpatch%lai(ico)
   384)             !------------------------------------------------------------------------------!
   385) 
   386) 
   387)             !------------------------------------------------------------------------------!
   388)             !    Call the photosynthesis for actual photosynthetic rates.  The units       !
   389)             ! of the input and output are the standard in most of ED modules, but many of  !
   390)             ! them are converted inside the photosynthesis model.                          !
   391)             !    Notice that the units that are per unit area are per m� of leaf, not the  !
   392)             ! patch area.                                                                  !
   393)             !------------------------------------------------------------------------------!
   394)             call lphysiol_full(            & !
   395)                csite%can_prss(ipa)         & ! Canopy air pressure              [       Pa]
   396)              , csite%can_rhos(ipa)         & ! Canopy air density               [    kg/m�]
   397)              , csite%can_shv(ipa)          & ! Canopy air sp. humidity          [    kg/kg]
   398)              , csite%can_co2(ipa)          & ! Canopy air CO2 mixing ratio      [ �mol/mol]
   399)              , ipft                        & ! Plant functional type            [      ---]
   400)              , leaf_par                    & ! Absorbed photos. active rad.     [ W/m�leaf]
   401)              , cpatch%leaf_temp(ico)       & ! Leaf temperature                 [        K]
   402)              , cpatch%lint_shv(ico)        & ! Leaf intercellular spec. hum.    [    kg/kg]
   403)              , green_leaf_factor(ipft)     & ! Greenness rel. to on-allometry   [      ---]
   404)              , leaf_aging_factor(ipft)     & ! Ageing parameter to scale VM     [      ---]
   405)              , cpatch%llspan(ico)          & ! Leaf life span                   [       yr]
   406)              , cpatch%vm_bar(ico)          & ! Average Vm function              [�mol/m�/s]
   407)              , cpatch%leaf_gbw(ico)        & ! Aerodyn. condct. of water vapour [  kg/m�/s]
   408)              , cpatch%A_open(ico)          & ! Photosynthesis rate     (open)   [�mol/m�/s]
   409)              , cpatch%A_closed(ico)        & ! Photosynthesis rate     (closed) [�mol/m�/s]
   410)              , cpatch%A_light(ico)         & ! Photosynthesis rate     (light)  [�mol/m�/s]
   411)              , cpatch%A_rubp(ico)          & ! Photosynthesis rate     (RuBP)   [�mol/m�/s]
   412)              , cpatch%A_co2(ico)           & ! Photosynthesis rate     (CO2)    [�mol/m�/s]
   413)              , cpatch%gsw_open(ico)        & ! Stom. condct. of water  (open)   [  kg/m�/s]
   414)              , cpatch%gsw_closed(ico)      & ! Stom. condct. of water  (closed) [  kg/m�/s]
   415)              , cpatch%lsfc_shv_open(ico)   & ! Leaf sfc. sp. humidity  (open)   [    kg/kg]
   416)              , cpatch%lsfc_shv_closed(ico) & ! Leaf sfc. sp. humidity  (closed) [    kg/kg]
   417)              , cpatch%lsfc_co2_open(ico)   & ! Leaf sfc. CO2 mix. rat. (open)   [ �mol/mol]
   418)              , cpatch%lsfc_co2_closed(ico) & ! Leaf sfc. CO2 mix. rat. (closed) [ �mol/mol]
   419)              , cpatch%lint_co2_open(ico)   & ! Intercellular CO2       (open)   [ �mol/mol]
   420)              , cpatch%lint_co2_closed(ico) & ! Intercellular CO2       (closed) [ �mol/mol]
   421)              , leaf_resp                   & ! Leaf respiration rate            [�mol/m�/s]
   422)              , vm                          & ! Max. capacity of Rubisco         [�mol/m�/s]
   423)              , compp                       & ! Gross photo. compensation point  [ �mol/mol]
   424)              , limit_flag                  & ! Photosynthesis limitation flag   [      ---]
   425)              )
   426) 
   427)             !----- Convert leaf respiration to [�mol/m�ground/s] --------------------------!
   428)             cpatch%leaf_respiration(ico) = leaf_resp * cpatch%lai (ico)
   429)             cpatch%today_leaf_resp(ico)  = cpatch%today_leaf_resp (ico)                    &
   430)                                          + cpatch%leaf_respiration(ico)
   431)             !----- The output variable must be in [kgC/plant/yr]. -------------------------!
   432)             cpatch%fmean_leaf_resp(ico)  = cpatch%fmean_leaf_resp (ico)                    &
   433)                                          + cpatch%leaf_respiration(ico)                    &
   434)                                          * dtlsm_o_frqsum * umols_2_kgCyr                  &
   435)                                          / cpatch%nplant          (ico)
   436) 
   437)             !----- Root biomass [kg/m2]. --------------------------------------------------!
   438)             broot_loc = cpatch%broot(ico)  * cpatch%nplant(ico)
   439) 
   440)             !----- Supply of water. -------------------------------------------------------!
   441)             cpatch%water_supply      (ico) = water_conductance       (ipft) * broot_loc    &
   442)                                            * avail_h2o_coh            (ico)
   443)             cpatch%fmean_water_supply(ico) = cpatch%fmean_water_supply(ico)                &
   444)                                            + cpatch%water_supply      (ico)                &
   445)                                            * dtlsm_o_frqsum
   446) 
   447)             root_depth_indices(kroot) = .true.
   448)             broot_tot                 = broot_tot + broot_loc
   449)             pss_available_water       = pss_available_water                                &
   450)                                       + avail_h2o_coh(ico) * broot_loc
   451) 
   452)             !------------------------------------------------------------------------------!
   453)             !     Determine the fraction of open stomata due to water limitation.          !
   454)             ! This is a function of the ratio between the potential water demand           !
   455)             ! (cpatch%psi_open, which is the average over the last time step), and the     !
   456)             ! supply (cpatch%water_supply).                                                !
   457)             !------------------------------------------------------------------------------!
   458)             select case (h2o_plant_lim)
   459)             case (0)
   460)                !---- No water limitation, fsw is always 1.0. ------------------------------!
   461)                cpatch%fsw(ico) = 1.0
   462) 
   463)             case (1,2)
   464)                water_demand    = cpatch%psi_open(ico) * cpatch%lai(ico)
   465)                if (cpatch%water_supply (ico) < tiny_num) then
   466)                   cpatch%fsw(ico) = 0.0
   467)                else
   468)                   cpatch%fsw(ico) = 1.0 / (1.0 + water_demand / cpatch%water_supply(ico))
   469)                end if
   470)             end select
   471)             !------------------------------------------------------------------------------!
   472) 
   473) 
   474) 
   475)             !------------------------------------------------------------------------------!
   476)             !      Photorespiration can become important at high temperatures.  If so,     !
   477)             ! close down the stomata.                                                      !
   478)             !------------------------------------------------------------------------------!
   479)             if (cpatch%A_open(ico) < cpatch%A_closed(ico)) then
   480)                cpatch%fs_open(ico) = 0.0
   481)             else
   482)                cpatch%fs_open(ico) = cpatch%fsw(ico) * cpatch%fsn(ico)
   483)             end if
   484) 
   485)             !----- Net stomatal conductance. ----------------------------------------------!
   486)             cpatch%leaf_gsw(ico) =        cpatch%fs_open(ico)  * cpatch%gsw_open(ico)      &
   487)                                  + (1.0 - cpatch%fs_open(ico)) * cpatch%gsw_closed(ico)
   488)             !------------------------------------------------------------------------------!
   489) 
   490) 
   491)             !----- GPP, averaged over frqstate. -------------------------------------------!
   492)             cpatch%gpp(ico) = max(0., cpatch%lai(ico)                                      &
   493)                                     * ( cpatch%fs_open(ico) * cpatch%A_open(ico)           &
   494)                                       + (1. - cpatch%fs_open(ico)) * cpatch%A_closed(ico)) &
   495)                                     + cpatch%leaf_respiration(ico) )
   496)             !----- The average must be in [kgC/plant/yr]. ---------------------------------!
   497)             cpatch%fmean_gpp(ico) = cpatch%fmean_gpp(ico)                                  &
   498)                                   + cpatch%gpp      (ico) * umols_2_kgCyr * dtlsm_o_frqsum &
   499)                                   / cpatch%nplant(ico)
   500)             !------------------------------------------------------------------------------!
   501) 
   502) 
   503)             !----- GPP, summed over 1 day. [�mol/m�ground] --------------------------------!
   504)             cpatch%today_gpp(ico) = cpatch%today_gpp(ico) + cpatch%gpp(ico)
   505)             !------------------------------------------------------------------------------!
   506) 
   507) 
   508)             !----- Potential GPP if no N limitation. [�mol/m�ground] ----------------------!
   509)             cpatch%today_gpp_pot(ico) = cpatch%today_gpp_pot(ico)                          &
   510)                                       + cpatch%lai(ico)                                    &
   511)                                       * ( cpatch%fsw(ico) * cpatch%A_open(ico)             &
   512)                                         + (1.0 - cpatch%fsw(ico)) * cpatch%A_closed(ico))  &
   513)                                       + cpatch%leaf_respiration(ico)
   514)             !------------------------------------------------------------------------------!
   515) 
   516) 
   517) 
   518)             !------------------------------------------------------------------------------!
   519)             !     Find the maximum productivities:                                         !
   520)             !                                                                              !
   521)             !     - today_gpp_lightmax: productivity of this cohort if it were at the top  !
   522)             !                           of the canopy (full light), with the actual fsw.   !
   523)             !     - today_gpp_moistmax: productivity of this cohort if the soil moisture   !
   524)             !                           was such that fsw would be 1 (full moisture), with !
   525)             !                           the actual light.                                  !
   526)             !     - today_gpp_mlmax:    productivity of this cohort if it was at the top   !
   527)             !                           of the canopy (full light) AND the soil moisture   !
   528)             !                           was such that fsw would be 1 (full moisture).      !
   529)             !                                                                              !
   530)             !     These productivites are used to scale the relative carbon balance, which !
   531)             ! will control density-dependent mortality.                                    !
   532)             !------------------------------------------------------------------------------!
   533)             cpatch%today_gpp_lightmax(ico) = cpatch%today_gpp_lightmax(ico)                &
   534)                                            + cpatch%lai(ico)                               &
   535)                                            * ( cpatch%fs_open(ico)                         &
   536)                                              * csite%A_o_max(ipft,ipa)                     &
   537)                                              + (1.0 - cpatch%fs_open(ico))                 &
   538)                                              * csite%A_c_max(ipft,ipa) )                   &
   539)                                            + cpatch%leaf_respiration(ico)
   540)             cpatch%today_gpp_moistmax(ico) = cpatch%today_gpp_moistmax(ico)                &
   541)                                            + cpatch%lai(ico) * cpatch%A_open(ico)          &
   542)                                            + cpatch%leaf_respiration(ico)
   543)             cpatch%today_gpp_mlmax(ico)    = cpatch%today_gpp_mlmax(ico)                   &
   544)                                            + cpatch%lai(ico) * csite%A_o_max(ipft,ipa)     &
   545)                                            + cpatch%leaf_respiration(ico)
   546)             !------------------------------------------------------------------------------!
   547) 
   548)       else
   549)          !----- If the cohort wasn't solved, we must assign some zeroes. ------------------!
   550)          cpatch%A_open(ico)               = 0.0
   551)          cpatch%A_closed(ico)             = 0.0
   552)          cpatch%psi_open(ico)             = 0.0
   553)          cpatch%psi_closed(ico)           = 0.0
   554)          cpatch%water_supply(ico)         = 0.0
   555)          cpatch%gsw_open(ico)             = 0.0
   556)          cpatch%gsw_closed(ico)           = 0.0
   557)          cpatch%leaf_gbh(ico)             = 0.0
   558)          cpatch%leaf_gbw(ico)             = 0.0
   559)          cpatch%leaf_gsw(ico)             = 0.0
   560)          cpatch%gpp(ico)                  = 0.0
   561)          cpatch%leaf_respiration(ico)     = 0.0
   562)          vm                               = 0.0
   563)          limit_flag                       = 0
   564)       end if
   565) 
   566)       !------------------------------------------------------------------------------------!
   567)       !    Not really a part of the photosynthesis scheme, but this will do it.  We must   !
   568)       ! integrate the "mean" of the remaining respiration terms, except for the root one.  !
   569)       ! This is done regardless on whether the cohort is doing photosynthesis.             !
   570)       !                                                                                    !
   571)       !    The "_respiration(ico) terms are in kgC/plant/day, so we must also multiply     !
   572)       ! them by the number of years per day so the output is in kgC/plant/yr.  High time   !
   573)       ! we switched everything to SI...                                                    !
   574)       !------------------------------------------------------------------------------------!
   575)       cpatch%fmean_leaf_growth_resp (ico) = cpatch%fmean_leaf_growth_resp  (ico)           &
   576)                                           + cpatch%leaf_growth_resp        (ico)           &
   577)                                           *  dtlsm_o_frqsum * yr_day
   578)       cpatch%fmean_root_growth_resp (ico) = cpatch%fmean_root_growth_resp  (ico)           &
   579)                                           + cpatch%root_growth_resp        (ico)           &
   580)                                           *  dtlsm_o_frqsum * yr_day
   581)       cpatch%fmean_sapa_growth_resp (ico) = cpatch%fmean_sapa_growth_resp  (ico)           &
   582)                                           + cpatch%sapa_growth_resp        (ico)           &
   583)                                           *  dtlsm_o_frqsum * yr_day
   584)       cpatch%fmean_sapb_growth_resp (ico) = cpatch%fmean_sapb_growth_resp  (ico)           &
   585)                                           + cpatch%sapb_growth_resp        (ico)           &
   586)                                           *  dtlsm_o_frqsum * yr_day
   587)       cpatch%fmean_leaf_storage_resp(ico) = cpatch%fmean_leaf_storage_resp (ico)           &
   588)                                           + cpatch%leaf_storage_resp(ico) * dtlsm_o_frqsum &
   589)                                           * yr_day
   590)       cpatch%fmean_root_storage_resp(ico) = cpatch%fmean_root_storage_resp (ico)           &
   591)                                           + cpatch%root_storage_resp(ico) * dtlsm_o_frqsum &
   592)                                           * yr_day
   593)       cpatch%fmean_sapa_storage_resp(ico) = cpatch%fmean_sapa_storage_resp (ico)           &
   594)                                           + cpatch%sapa_storage_resp(ico) * dtlsm_o_frqsum &
   595)                                           * yr_day
   596)       cpatch%fmean_sapb_storage_resp(ico) = cpatch%fmean_sapb_storage_resp (ico)           &
   597)                                           + cpatch%sapb_storage_resp(ico) * dtlsm_o_frqsum &
   598)                                           * yr_day
   599)       !------------------------------------------------------------------------------------!
   600) 
   601)       if (print_photo_debug) then
   602)          call print_photo_details(cmet,csite,ipa,ico,limit_flag,vm,compp)
   603)       end if
   604)    end do cohortloop
   605)    !---------------------------------------------------------------------------------------!
   606) 
   607) 
   608) 
   609)    !---------------------------------------------------------------------------------------!
   610)    !     Add the contribution of this time step to the average available water.  This is   !
   611)    ! done only if there is some cohort transpiring.                                        !
   612)    !---------------------------------------------------------------------------------------!
   613)    if (broot_tot > 1.e-20) then
   614)       csite%fmean_available_water(ipa) = csite%fmean_available_water(ipa)                  &
   615)                                        + pss_available_water * dtlsm_o_frqsum / broot_tot
   616)    end if
   617)    !---------------------------------------------------------------------------------------!
   618) 
   619) 
   620) 
   621)    !---------------------------------------------------------------------------------------!
   622)    !    De-allocate the temporary vector.                                                  !
   623)    !---------------------------------------------------------------------------------------!
   624)    if (cpatch%ncohorts > 0) then
   625)       deallocate(avail_h2o_coh)
   626)    end if
   627)    !---------------------------------------------------------------------------------------!
   628) 
   629)    return
   630) end subroutine canopy_photosynthesis
