     9) subroutine read_met_driver_head()
    10)    use ed_max_dims    , only : max_met_vars     ! ! intent(in)
    11)    use met_driver_coms, only : nformats         & ! intent(in)
    12)                              , met_names        & ! intent(out)
    13)                              , met_nlon         & ! intent(out)
    14)                              , met_nlat         & ! intent(out)
    15)                              , met_dx           & ! intent(out)
    16)                              , met_dy           & ! intent(out)
    17)                              , met_xmin         & ! intent(out)
    18)                              , met_ymin         & ! intent(out)
    19)                              , met_nv           & ! intent(out)
    20)                              , met_vars         & ! intent(out)
    21)                              , met_frq          & ! intent(out)
    22)                              , met_interp       & ! intent(out)
    23)                              , ed_met_driver_db & ! intent(out)
    24)                              , no_ll            ! ! intent(out)
    25)    implicit none  
    26)    !----- Local variables -----------------------------------------------------------------!
    27)    logical :: l1
    28)    logical :: yes_lat     ! Logical for determining whether latitude grids are present
    29)    logical :: yes_lon     ! Logical for determining whether longitude grids are present
    30)    integer :: iformat
    31)    integer :: n
    32)    !---------------------------------------------------------------------------------------!
    33) 
    34) 
    35)    !----- First thing, let's check whether the meterological data metafile exists. --------!
    36)    inquire(file=trim(ed_met_driver_db),exist=l1)
    37)    if (.not. l1) then
    38)       write (unit=*,fmt='(a)') 'File '//trim(ed_met_driver_db)//' not found!'
    39)       write (unit=*,fmt='(a)') 'Specify ED_MET_DRIVER_DB properly in ED namelist.'
    40)       call fatal_error('Ed_met_driver_db not found!','read_met_driver_head'                &
    41)                       &,'ed_met_driver.f90')
    42)    end if
    43)    !---------------------------------------------------------------------------------------!
    44) 
    45) 
    46)    !----- Loading the meterorological data metafile information. --------------------------!
    47)    open(unit=12,file=trim(ed_met_driver_db),form='formatted',status='old')
    48)    read(unit=12,fmt=*)  ! skip header
    49) 
    50)    !------ Read the number of different file formats. -------------------------------------!
    51)    read(unit=12,fmt=*) nformats
    52)    !------ Allocate the header information for each format --------------------------------!
    53)    allocate(met_names (nformats)              )
    54)    allocate(met_nlon  (nformats)              )
    55)    allocate(met_nlat  (nformats)              )
    56)    allocate(met_dx    (nformats)              )
    57)    allocate(met_dy    (nformats)              )
    58)    allocate(met_xmin  (nformats)              )
    59)    allocate(met_ymin  (nformats)              )
    60)    allocate(met_nv    (nformats)              )
    61)    allocate(met_vars  (nformats, max_met_vars))
    62)    allocate(met_frq   (nformats, max_met_vars))
    63)    allocate(met_interp(nformats, max_met_vars))
    64)    allocate(no_ll     (nformats)              )
    65) 
    66)    !----- Just to initialize, if lon/lat are both found, it will become .false. -----------!
    67)    no_ll(:) = .true.    
    68)    !----- Read the information for each format. -------------------------------------------!
    69)    do iformat = 1,nformats
    70)       read(unit=12,fmt='(a)')  met_names(iformat)
    71)       read(unit=12,fmt=*)      met_nlon(iformat), met_nlat(iformat), met_dx(iformat)       &
    72)                              , met_dy(iformat)  , met_xmin(iformat), met_ymin(iformat)
    73)       read(unit=12,fmt=*)      met_nv(iformat)
    74)       read(unit=12,fmt=*)      (met_vars(iformat,n)  ,n=1,met_nv(iformat))
    75)       read(unit=12,fmt=*)      (met_frq(iformat,n)   ,n=1,met_nv(iformat))
    76)       read(unit=12,fmt=*)      (met_interp(iformat,n),n=1,met_nv(iformat))
    77)       
    78)       !----- Just making sure that the variable list is case insensitive. -----------------!
    79)       call tolower(met_vars(iformat,1:met_nv(iformat)),met_nv(iformat))
    80)       
    81)       !----- First check - see if lat/lon data are there. ---------------------------------!
    82)       yes_lon = any(met_vars(iformat,1:met_nv(iformat)) == 'lon')
    83)       yes_lat = any(met_vars(iformat,1:met_nv(iformat)) == 'lat')
    84)       
    85) 
    86)       !----- Check to see if you have both, none or one of each. --------------------------!
    87)       if (yes_lat .and. yes_lon) then
    88)          no_ll (iformat) = .false.
    89)       elseif (yes_lat .neqv. yes_lon) then
    90)          call fatal_error('You are missing a lat or a lon variable in the met nl'          &
    91)                          ,'read_met_driver_head','ed_met_driver.f90')
    92)       end if
    93)    end do
    94) 
    95)    close (unit=12,status='keep')
    96) 
    97)    return
    98) end subroutine read_met_driver_head

   109) !    This subroutine will initialise the arrays that will temporarily receive the meteoro- !
   110) ! logical dataset. It will nullify the pointers, allocate them, then assign a very strange !
   111) ! number so in case the variable is not properly initialised the user will notice some-    !

   113) !------------------------------------------------------------------------------------------!
   114) subroutine init_met_drivers
   115)    use lapse
   116)    use ed_max_dims     , only : max_met_vars
   117)    use met_driver_coms , only : nformats          & ! intent(in)
   118)                               , met_names         & ! intent(out)
   119)                               , met_nlon          & ! intent(out)
   120)                               , met_nlat          & ! intent(out)
   121)                               , met_dx            & ! intent(out)
   122)                               , met_dy            & ! intent(out)
   123)                               , met_xmin          & ! intent(out)
   124)                               , met_ymin          & ! intent(out)
   125)                               , met_nv            & ! intent(out)
   126)                               , met_vars          & ! intent(out)
   127)                               , met_frq           & ! intent(out)
   128)                               , met_interp        & ! intent(out)
   129)                               , no_ll             & ! intent(out)
   130)                               , has_co2           & ! intent(out)
   131)                               , has_ustar         ! ! intent(out)
   132)    use canopy_air_coms , only : isfclyrm          ! ! intent(in)
   133)    use ed_state_vars   , only : edgrid_g          & ! structure
   134)                               , edtype            & ! structure
   135)                               , polygontype       ! ! structure
   136)    use grid_coms       , only : ngrids            ! ! intent(in)
   137)    use consts_coms     , only : day_sec           ! ! intent(in)
   138)    implicit none
   139)    !----- Local variables -----------------------------------------------------------------!
   140)    type(polygontype), pointer :: cpoly
   141)    type(edtype)     , pointer :: cgrid
   142)    integer                    :: ipy,igr
   143)    integer                    :: iformat
   144)    integer                    :: iv
   145)    integer                    :: mem_size
   146)    real                       :: westedge,eastedge,southedge,northedge
   147)    !---------------------------------------------------------------------------------------!
   148) 
   149) 
   150)    !----- Set the lapse rates. ------------------------------------------------------------!
   151)    do igr=1,ngrids
   152)       call setLapseParms(edgrid_g(igr))
   153)    end do
   154) 
   155)    !----- Read the information for each format. -------------------------------------------!
   156)    has_co2   = .false.
   157)    has_ustar = .false.
   158)    formloop: do iformat = 1,nformats
   159)       !----- Finding the met driver boundaries. -------------------------------------------!
   160)       westedge  = met_xmin(iformat) - 0.5 * met_dx(iformat)
   161)       eastedge  = met_xmin(iformat) + (real(met_nlon(iformat))-0.5) * met_dx(iformat)
   162)       southedge = met_ymin(iformat) - 0.5 * met_dy(iformat)
   163)       northedge = met_ymin(iformat) + (real(met_nlat(iformat))-0.5) * met_dy(iformat)
   164)       gridloop: do igr = 1,ngrids
   165)          cgrid => edgrid_g(igr)
   166) 
   167)          polyloop: do ipy = 1,cgrid%npolygons
   168) 
   169)             cpoly => cgrid%polygon(ipy)
   170) 
   171)             !----- Make sure site falls within file domain. -------------------------------!
   172)             if(no_ll(iformat) .and.                                                        &
   173)                ( cgrid%lon(ipy) < westedge .or. cgrid%lat(ipy) < southedge .or.            &
   174)                  cgrid%lon(ipy) > eastedge .or. cgrid%lat(ipy) > northedge)     ) then
   175)                write(unit=*,fmt='(a)') '=================================================='
   176)                write(unit=*,fmt='(a)') ' Polygon lies outside the met driver domain!!!'
   177)                write(unit=*,fmt='(a)') '=================================================='
   178)                write(unit=*,fmt='(a)')           ' + Polygon: '
   179)                write(unit=*,fmt='(a,1x,es12.5)') '   - Longitude : ',cgrid%lon(ipy)
   180)                write(unit=*,fmt='(a,1x,es12.5)') '   - Latitude  : ',cgrid%lat(ipy)
   181)                write(unit=*,fmt='(a,1x,a)')      ' + File: : ',trim(met_names(iformat))
   182)                write(unit=*,fmt='(a)')           ' + Meteorological data boundaries: '
   183)                write(unit=*,fmt='(a,1x,es12.5)') '   - West      : ',westedge
   184)                write(unit=*,fmt='(a,1x,es12.5)') '   - East      : ',eastedge
   185)                write(unit=*,fmt='(a,1x,es12.5)') '   - South     : ',southedge
   186)                write(unit=*,fmt='(a,1x,es12.5)') '   - North     : ',northedge
   187)                call fatal_error('Polygon outside the domain of the meteorological drivers' &
   188)                                ,'init_met_drivers','ed_met_driver.f90')
   189)             end if
   190)             
   191)             !----- Loop over variables. ---------------------------------------------------!
   192)             varloop: do iv = 1,met_nv(iformat)
   193)                
   194)                !---------------------------------------------------------------------------!
   195)                !  Either this is a constant or variable in time.                           !
   196)                !---------------------------------------------------------------------------!
   197)                select case(met_interp(iformat,iv))
   198)                case(0,3)    !----- Variable. ----------------------------------------------!
   199)                   mem_size = nint(day_sec / met_frq(iformat,iv)) * 31
   200)                case(1,5)    !----- Interpolated.   Read two months in. ---------------------!
   201)                   mem_size = 2 * nint(day_sec / met_frq(iformat,iv)) * 31
   202)                case (2,4)  !----- Constant in time. ---------------------------------------!
   203)                   mem_size = 1
   204)                case default
   205)                   call fatal_error('Invalid met_interp! It should be between 0 and 5.'     &
   206)                                   ,'init_met_drivers','ed_met_driver.f90')
   207)                end select
   208) 
   209)                !---------------------------------------------------------------------------!
   210)                !     Allocate variables accordingly, and initialise them with a crazy      !
   211)                ! number to make the model crash in case it attempts to use them without    !
   212)                ! assigning an appropriate number.                                          !
   213)                !---------------------------------------------------------------------------!
   214)                select case (trim(met_vars(iformat,iv)))
   215)                case ('nbdsf')   !----- Near IR beam downward shortwave flux. -- [   W/m�] -!
   216)                   nullify(cgrid%metinput(ipy)%nbdsf)
   217)                   allocate(cgrid%metinput(ipy)%nbdsf(mem_size))
   218)                   cgrid%metinput(ipy)%nbdsf = huge(1.)
   219) 
   220)                case ('nddsf')   !----- Near IR diffuse downward shortwave flux. [   W/m�] -!
   221)                   nullify(cgrid%metinput(ipy)%nddsf)
   222)                   allocate(cgrid%metinput(ipy)%nddsf(mem_size))
   223)                   cgrid%metinput(ipy)%nddsf = huge(1.)
   224) 
   225)                case ('vbdsf')   !----- Visible beam downward shortwave flux. -- [   W/m�] -!
   226)                   nullify(cgrid%metinput(ipy)%vbdsf)
   227)                   allocate(cgrid%metinput(ipy)%vbdsf(mem_size))
   228)                   cgrid%metinput(ipy)%vbdsf = huge(1.)
   229) 
   230)                case ('vddsf')   !----- Visible diffuse downward shortwave flux. [   W/m�] -!
   231)                   nullify(cgrid%metinput(ipy)%vddsf)
   232)                   allocate(cgrid%metinput(ipy)%vddsf(mem_size))
   233)                   cgrid%metinput(ipy)%vddsf = huge(1.)
   234) 
   235)                case ('prate')   !----- Precipitation rate. -------------------- [kg/m�/s] -!
   236)                   nullify(cgrid%metinput(ipy)%prate)
   237)                   allocate(cgrid%metinput(ipy)%prate(mem_size))
   238)                   cgrid%metinput(ipy)%prate = huge(1.)
   239) 
   240)                case ('dlwrf')   !----- Downwelling longwave radiation. -------- [   W/m�] -!
   241)                   nullify(cgrid%metinput(ipy)%dlwrf)
   242)                   allocate(cgrid%metinput(ipy)%dlwrf(mem_size))
   243)                   cgrid%metinput(ipy)%dlwrf = huge(1.)
   244) 
   245)                case ('pres')   !----- Air pressure. --------------------------- [   N/m�] -!
   246)                   nullify(cgrid%metinput(ipy)%pres)
   247)                   allocate(cgrid%metinput(ipy)%pres(mem_size))
   248)                   cgrid%metinput(ipy)%pres = huge(1.)
   249) 
   250)                case ('hgt')     !----- Air pressure. -------------------------- [      m] -!
   251)                   nullify(cgrid%metinput(ipy)%hgt)
   252)                   allocate(cgrid%metinput(ipy)%hgt(mem_size))
   253)                   cgrid%metinput(ipy)%hgt = huge(1.)
   254) 
   255)                case ('ugrd')    !----- Zonal wind. ---------------------------- [    m/s] -!
   256)                   nullify(cgrid%metinput(ipy)%ugrd)
   257)                   allocate(cgrid%metinput(ipy)%ugrd(mem_size))
   258)                   cgrid%metinput(ipy)%ugrd = huge(1.)
   259) 
   260)                case ('vgrd')    !----- Meridional wind. ----------------------- [    m/s] -!
   261)                   nullify(cgrid%metinput(ipy)%vgrd)
   262)                   allocate(cgrid%metinput(ipy)%vgrd(mem_size))
   263)                   cgrid%metinput(ipy)%vgrd = huge(1.)
   264) 
   265)                case ('sh')      !----- Specific humidity. --------------------- [kg/kg_a] -!
   266)                   nullify(cgrid%metinput(ipy)%sh)
   267)                   allocate(cgrid%metinput(ipy)%sh(mem_size))
   268)                   cgrid%metinput(ipy)%sh = huge(1.)
   269) 
   270)                case ('tmp')     !----- Air temperature. ----------------------- [      K] -!
   271)                   nullify(cgrid%metinput(ipy)%tmp)
   272)                   allocate(cgrid%metinput(ipy)%tmp(mem_size))
   273)                   cgrid%metinput(ipy)%tmp = huge(1.)
   274) 
   275)                case ('co2')     !----- CO2 mixing ratio. ---------------------- [    ppm] -!
   276)                   has_co2 = .true.
   277)                   nullify(cgrid%metinput(ipy)%co2)
   278)                   allocate(cgrid%metinput(ipy)%co2(mem_size))
   279)                   cgrid%metinput(ipy)%co2 = huge(1.)
   280) 
   281)                case ('ustar')   !----- Friction velocity. --------------------- [    m/s] -!
   282)                   has_ustar = .true.
   283)                   nullify(cgrid%metinput(ipy)%atm_ustar)
   284)                   allocate(cgrid%metinput(ipy)%atm_ustar(mem_size))
   285)                   cgrid%metinput(ipy)%atm_ustar = huge(1.)
   286) 
   287)                case ('lat','lon') !---- Latitude and longitude: skip them. ----------------!
   288)                case default
   289)                   call fatal_error('Invalid met variable'//trim(met_vars(iformat,iv))//'!' &
   290)                                   ,'init_met_drivers','ed_met_driver.f90')
   291)                end select
   292) 
   293)             end do varloop
   294)          end do polyloop
   295)       end do gridloop
   296)    end do formloop
   297) 
   298) 
   299)    !---------------------------------------------------------------------------------------!
   300)    !     Friction velocity is not necessary (and is actually deprecated), but if the user  !
   301)    ! wants to run the model with prescribed u*, then they must provide it.                 !
   302)    !---------------------------------------------------------------------------------------!
   303)    if ( isfclyrm == 0 .and. .not. has_ustar) then
   304)       call fatal_error('You must provide u* if you want to run with prescribed u*!!!'      &
   305)                       ,'init_met_drivers','ed_met_driver.f90')
   306)    end if 
   307)    !---------------------------------------------------------------------------------------!
   308) 
   309)    return
   310) end subroutine init_met_drivers

   322) !------------------------------------------------------------------------------------------!
   323) subroutine read_met_drivers_init
   324) 
   325)    use ed_max_dims       , only : str_len        ! ! intent(in)
   326)    use ed_state_vars     , only : edgrid_g       & ! structure
   327)                                 , edtype         & ! structure
   328)                                 , polygontype    ! ! structure
   329)    use hdf5_utils        , only : shdf5_open_f   & ! subroutine
   330)                                 , shdf5_close_f  ! ! subroutine
   331)    use met_driver_coms   , only : nformats       & ! intent(in)
   332)                                 , ishuffle       & ! intent(in)
   333)                                 , met_names      & ! intent(in)
   334)                                 , met_nv         & ! intent(in)
   335)                                 , met_interp     & ! intent(in)
   336)                                 , met_frq        & ! intent(in)
   337)                                 , metcyc1        & ! intent(in)
   338)                                 , metcycf        & ! intent(in)
   339)                                 , metyears       ! ! intent(inout)
   340)    use ed_misc_coms      , only : current_time   & ! intent(in)
   341)                                 , iyeara         & ! intent(in)
   342)                                 , iyearz         & ! intent(in)
   343)                                 , imonthz        ! ! intent(in)
   344)    use grid_coms         , only : ngrids         ! ! intent(in)
   345)    use consts_coms       , only : day_sec        ! ! intent(in)
   346)    use ed_node_coms      , only : mynum          ! ! intent(in)
   347)    implicit none
   348)    !----- Local variables -----------------------------------------------------------------!
   349)    type(edtype)          , pointer :: cgrid
   350)    character(len=str_len)          :: infile
   351)    integer                         :: igr
   352)    integer                         :: year_use
   353)    integer                         :: iformat
   354)    integer                         :: iv
   355)    integer                         :: offset
   356)    integer                         :: m2
   357)    integer                         :: y2
   358)    integer                         :: year_use_2
   359)    integer                         :: nyears
   360)    integer                         :: ncyc
   361)    integer                         :: nfullcyc
   362)    integer                         :: i1stfull
   363)    integer                         :: iyear
   364)    integer, dimension(8)           :: seedtime
   365)    real                            :: runif
   366)    logical                         :: exans
   367)    !----- Local constants -----------------------------------------------------------------!
   368)    character(len=3), dimension(12), parameter :: mname = (/ 'JAN', 'FEB', 'MAR', 'APR'     &
   369)                                                           , 'MAY', 'JUN', 'JUL', 'AUG'     &
   370)                                                           , 'SEP', 'OCT', 'NOV', 'DEC'   /) 
   371)    !----- External functions --------------------------------------------------------------!
   372)    logical         , external  :: isleap
   373)    !----- Variables to be saved. ----------------------------------------------------------!
   374)    logical         , save      :: first_time=.true.
   375)    !---------------------------------------------------------------------------------------!
   376)  
   377) 
   378) 
   379)    !---------------------------------------------------------------------------------------!
   380)    !    If this is the first time this subroutine is called, we build the meteorological   !
   381)    ! driver sequence of years.  Often is the case that the user has a meteorological data- !
   382)    ! set that is shorter than the time span he or she is intending to run. In this case we !
   383)    ! need to fill the other years with some data.                                          !
   384)    !---------------------------------------------------------------------------------------!
   385)    if (first_time) then
   386)       first_time = .false.
   387) 
   388)       !------------------------------------------------------------------------------------!
   389)       !     Define the number of years we have meteorological information available.  In   !
   390)       ! case the simulation is supposed to end in December, we must add an extra year, be- !
   391)       ! cause the meteorological forcing may need to access the following month.           !
   392)       !------------------------------------------------------------------------------------!
   393)       ncyc   = metcycf - metcyc1 + 1
   394)       if (imonthz == 12) then
   395)          nyears = iyearz  - iyeara  + 2
   396)       else
   397)          nyears = iyearz  - iyeara  + 1
   398)       end if
   399) 
   400)       !------ Allocating the sequence of years. -------------------------------------------!
   401)       allocate(metyears(nyears))      
   402)       
   403)       !----- Initialising the seed for random number generation. --------------------------!
   404)       call random_seed()
   405) 
   406)       !------------------------------------------------------------------------------------!
   407)       !     Now we must decide how we are going to select the meteorological driver years, !
   408)       ! in particular, the years outside the meteorological driver range.                  !
   409)       !------------------------------------------------------------------------------------!
   410)       select case (ishuffle)
   411)       case (0)
   412)          !---------------------------------------------------------------------------------!
   413)          !     Make a loop over the years, repeating the loop as many times as we need.    !
   414)          ! First we determine how many full cycles fit between iyeara and metcyc1.  Note   !
   415)          ! that this can be negative in case metcyc1 is less than iyeara, but this will    !
   416)          ! work too.
   417)          !---------------------------------------------------------------------------------!
   418)          nfullcyc = floor(real(metcyc1-iyeara)/real(ncyc))
   419)          i1stfull = metcyc1 - nfullcyc * ncyc
   420)          
   421)          year_use   = metcycf - (i1stfull - iyeara)
   422)          if (mynum == 1) then
   423)             write (unit=*,fmt='(a)'       )  '------------------------------'
   424)             write (unit=*,fmt='(a,1x,i12)')  ' - METCYC1 =',metcyc1
   425)             write (unit=*,fmt='(a,1x,i12)')  ' - METCYCF =',metcycf
   426)             write (unit=*,fmt='(a,1x,i12)')  ' - NYEARS  =',nyears
   427)             write (unit=*,fmt='(a)'       )  '------------------------------'
   428)             write (unit=*,fmt='(2(1x,a))') '       IYEAR','    YEAR_USE'
   429)          end if
   430)          do iyear=1,nyears
   431)             if (year_use == metcycf) then
   432)                year_use = metcyc1
   433)             else
   434)                year_use = year_use + 1
   435)             end if
   436)             metyears(iyear) = year_use
   437)             if (mynum == 1) then
   438)                write (unit=*,fmt='(2(1x,i12))') iyear,metyears(iyear)
   439)             end if
   440)          end do
   441)          if (mynum == 1) then
   442)             write (unit=*,fmt='(a)'       )  '------------------------------'
   443)             write (unit=*,fmt='(a)'       )  ' '
   444)          end if
   445) 
   446)       !------------------------------------------------------------------------------------!
   447)       !    For years outside the met driver range, we pick up a random year.  Because we   !
   448)       ! use the actual sequence of random years given by the code random generator, this   !
   449)       ! sequence will be always the same for a given met driver and first year.            !
   450)       !------------------------------------------------------------------------------------!
   451)       case (1)
   452) 
   453)          if (mynum == 1) then
   454)             write (unit=*,fmt='(a)'       )  '------------------------------'
   455)             write (unit=*,fmt='(a,1x,i12)')  ' - METCYC1 =',metcyc1

   904)

   938)                ! tropical regions because the zenith angle changes very quickly and the    !
   939)                ! radiation could be too skewed even with hourly data.                      !

   941)                prevmet_timea = current_time

   952)                   !------------------------------------------------------------------------!
   953)

   957)                   !------------------------------------------------------------------------!
   958) 
   959)                case (3)
   960)                   !----- Averages centred at the reference time. --------------------------!
   961)                   call update_model_time_dm(prevmet_timea                                  &

  2382)             !------------------------------------------------------------------------------!
  2383)          else
  2384)             !----- No u* from met driver, assign a dummy value. ---------------------------!
  2385)             cpoly%met(isi)%atm_ustar = ustmin
  2386)             !------------------------------------------------------------------------------!
  2387)          end if
  2388)          !---------------------------------------------------------------------------------!
  2389) 
  2390) 
  2391) 
  2392)          !---------------------------------------------------------------------------------!
  2393)          !     We now find some derived properties.  In case several sites exist, the      !
  2394)          ! lapse rate was applied to pressure, temperature, and mixing ratio.  Then we     !
  2395)          ! calculate the Exner function, potential temperature and equivalent potential    !
  2396)          ! temperature, so it will respect the ideal gas law and first law of thermo-      !
  2397)          ! dynamics.                                                                       !
  2398)          !---------------------------------------------------------------------------------!
  2399)          cpoly%met(isi)%exner     = press2exner(cpoly%met(isi)%prss)
  2400)          cpoly%met(isi)%atm_theta = extemp2theta( cpoly%met(isi)%exner                     &
  2401)                                                 , cpoly%met(isi)%atm_tmp )
  2402)          !---------------------------------------------------------------------------------!
  2403) 
  2404) 
  2405) 
  2406)          !---------------------------------------------------------------------------------!
  2407)          !     Check the relative humidity associated with the current pressure, temper-   !
  2408)          ! ature, and specific humidity.  Impose lower and upper bounds as prescribed by   !
  2409)          ! the variables atm_rhv_min and atm_rhv_max (from met_driver_coms.f90, and        !
  2410)          ! defined at the init_met_params subroutine in ed_params.f90).                    !
  2411)          !---------------------------------------------------------------------------------!
  2412)          relhum = rehuil(cpoly%met(isi)%prss,cpoly%met(isi)%atm_tmp,cpoly%met(isi)%atm_shv &
  2413)                         ,.true.)
  2414)          !---------------------------------------------------------------------------------!
  2415)          !      Check whether the relative humidity is off-bounds.  If it is, then we re-  !
  2416)          ! calculate the mixing ratio and convert to specific humidity.                    !
  2417)          !---------------------------------------------------------------------------------!
  2418)          if (relhum < atm_rhv_min) then
  2419)             relhum = atm_rhv_min
  2420)             cpoly%met(isi)%atm_shv = ptrh2rvapil( relhum                                      &
  2421)                                                 , cgrid%met(ipy)%prss                      &
  2422)                                                 , cgrid%met(ipy)%atm_tmp                   &
  2423)                                                 , .true.                                   )
  2424)          elseif (relhum > atm_rhv_max) then
  2425)             relhum                 = atm_rhv_max
  2426)             cpoly%met(isi)%atm_shv = ptrh2rvapil( relhum                                      &
  2427)                                                 , cgrid%met(ipy)%prss                      &
  2428)                                                 , cgrid%met(ipy)%atm_tmp                   &
  2429)                                                 , .true.                                   )
  2430)          end if
  2431) 
  2432)          !----- Find the atmospheric equivalent potential temperature. --------------------!
  2433)          rvaux                    = cgrid%met(ipy)%atm_shv / (1.0 - cgrid%met(ipy)%atm_shv)
  2434)          cpoly%met(isi)%atm_theiv = thetaeiv(cpoly%met(isi)%atm_theta,cpoly%met(isi)%prss  &
  2435)                                             ,cpoly%met(isi)%atm_tmp,rvaux,rvaux)
  2436)          !---------------------------------------------------------------------------------!
  2437) 
  2438)          !----- Find the vapour pressure deficit. -----------------------------------------!
  2439)          cpoly%met(isi)%atm_vpdef = vpdefil (cpoly%met(isi)%prss,cpoly%met(isi)%atm_tmp    &
  2440)                                             ,cpoly%met(isi)%atm_shv,.true.)
  2441)          !---------------------------------------------------------------------------------!
  2442) 
  2443)          !----- Solar radiation -----------------------------------------------------------!
  2444)          cpoly%met(isi)%rshort_diffuse = cpoly%met(isi)%par_diffuse                        &
  2445)                                        + cpoly%met(isi)%nir_diffuse
  2446)          cpoly%met(isi)%rshort         = cpoly%met(isi)%rshort_diffuse                     &
  2447)                                        + cpoly%met(isi)%par_beam + cpoly%met(isi)%nir_beam
  2448)          !---------------------------------------------------------------------------------!
  2449) 
  2450) 
  2451) 
  2452)          !---------------------------------------------------------------------------------!
  2453)          !  Precipitation internal energy and "depth".                                     !
  2454)          !  fcns derived from                                                              !
  2455)          !  Jin et al 1999 Hydrol Process. 13:2467-2482 Table 2                            !
  2456)          !  [[modified 11/16/09 by MCD]]                                                   !
  2457)          !---------------------------------------------------------------------------------!
  2458)          if (cpoly%met(isi)%atm_tmp > (t3ple + 2.5)) then
  2459)             !----- Rain only. -------------------------------------------------------------!
  2460)             fliq = 1.0
  2461)             fice = 0.0
  2462)             cpoly%met(isi)%dpcpg = max(0.0, cpoly%met(isi)%pcpg) *wdnsi
  2463) 
  2464)          elseif (cpoly%met(isi)%atm_tmp <= (t3ple + 2.5) .and.                             &
  2465)                  cpoly%met(isi)%atm_tmp  > (t3ple + 2.0) ) then
  2466)             !------------------------------------------------------------------------------!
  2467)             !     Slight modification so the liquid fraction becomes continuous.           !
  2468)             !------------------------------------------------------------------------------!
  2469)             fliq  = min(1.0, 0.4 + 1.2 * (cpoly%met(isi)%atm_tmp - t3ple - 2.0))
  2470)             fice  = 1.0 - fliq
  2471)             snden = 189.0
  2472)             cpoly%met(isi)%dpcpg = max(0.0, cpoly%met(isi)%pcpg)                           &
  2473)                                  * ((1.0-fice) * wdnsi + fice / snden)
  2474) 
  2475)          elseif (cpoly%met(isi)%atm_tmp <= (t3ple + 2.0) .and.                             &
  2476)                  cpoly%met(isi)%atm_tmp > t3ple          ) then
  2477)             !------------------------------------------------------------------------------!
  2478)             !     Increasing the fraction of snow. (N.B. May not be appropriate for        !
  2479)             ! sub-tropical regions where the level of the melting layer is higher...).     !
  2480)             !------------------------------------------------------------------------------!
  2481)             fliq  = max(0.0, 0.2 * (cpoly%met(isi)%atm_tmp - t3ple))
  2482)             fice  = 1.0 - fliq
  2483)             snden = (50.0+1.7*(cpoly%met(isi)%atm_tmp-258.15)**1.5 )
  2484)             cpoly%met(isi)%dpcpg = max(0.0, cpoly%met(isi)%pcpg)                           &
  2485)                                  * ((1.0-fice) * wdnsi + fice / snden)
  2486) 
  2487)          elseif (cpoly%met(isi)%atm_tmp <= t3ple         .and.                             &
  2488)                  cpoly%met(isi)%atm_tmp > (t3ple - 15.0) ) then
  2489)             !----- Below freezing point, snow only. ---------------------------------------!
  2490)             fliq  = 0.0
  2491)             fice  = 1.0
  2492)             snden = (50.0+1.7*(cpoly%met(isi)%atm_tmp-258.15)**1.5 )
  2493)             cpoly%met(isi)%dpcpg = max(0.0, cpoly%met(isi)%pcpg) / snden
  2494) 
  2495)          else ! if (copy%met(isi)%atm_tmp < (t3ple - 15.0)) then
  2496)             !----- Below freezing point, snow only. ---------------------------------------!
  2497)             fliq  = 0.0
  2498)             fice  = 1.0
  2499)             snden = 50.
  2500)             cpoly%met(isi)%dpcpg = max(0.0, cpoly%met(isi)%pcpg) / snden
  2501) 
  2502)          end if
  2503)          !---------------------------------------------------------------------------------!
  2504) 
  2505)          !---------------------------------------------------------------------------------!
  2506)          !     Set internal energy.  This will be the precipitation times the specific     !
  2507)          ! internal energy of water (above or at triple point) multiplied by the liquid    !
  2508)          ! fraction plus the specific internal energy of ice (below or at the triple       !
  2509)          ! point) multiplied by the ice fraction.                                          !
  2510)          !---------------------------------------------------------------------------------!
  2511)          cpoly%met(isi)%qpcpg = max(0.0, cpoly%met(isi)%pcpg)                              &
  2512)                               * ( fliq * tl2uint(max(t3ple,cpoly%met(isi)%atm_tmp),1.0)    &
  2513)                                 + fice * tl2uint(min(t3ple,cpoly%met(isi)%atm_tmp),0.0) )
  2514)          !---------------------------------------------------------------------------------!
  2515)       end do siteloop
  2516)       !------------------------------------------------------------------------------------!
  2517)    end do polyloop
  2518)    !---------------------------------------------------------------------------------------!
  2519) 
  2520)    return
  2521) end subroutine update_met_drivers

  2531) !==========================================================================================!
  2532) !      This subroutine reads in the meteorological data.  Subroutine shdf5_irec_f uses     !
  2533) ! HDF5 routines to extract the data from the archive.  The rest of this code manages the   !
  2534) ! allocation and storage of that data.  The extracted data may be from a year, other than  !
  2535) ! the current simulation year.  This happens when we use a range of years to loop data     !
  2536) ! over, typically during very long simulations.  In such cases, the number of days in      !
  2537) ! February will change during leap years, and the number of days in February during the    !
  2538) ! model year may not match that of the simulation year, therefore this must be taken into  !
  2539) ! account.                                                                                 !
  2540) !==========================================================================================!
  2541) subroutine read_ol_file(infile,iformat, iv, mname, year, offset, cgrid)
  2542)    use ed_max_dims    , only : str_len       ! ! intent(in)
  2543)    use ed_state_vars  , only : edtype        & ! structure
  2544)                              , polygontype   ! ! structure
  2545)    use met_driver_coms, only : met_frq       & ! intent(in)
  2546)                              , met_nlon      & ! intent(in)
  2547)                              , met_nlat      & ! intent(in)
  2548)                              , met_vars      & ! intent(in)
  2549)                              , met_xmin      & ! intent(in)
  2550)                              , met_dx        & ! intent(in)
  2551)                              , met_ymin      & ! intent(in)
  2552)                              , met_dy        & ! intent(in)
  2553)                              , met_interp    & ! intent(in)
  2554)                              , no_ll         ! ! intent(in)
  2555)    use hdf5_utils     , only : shdf5_irec_f  & ! subroutine
  2556)                              , shdf5_info_f  ! ! subroutine
  2557)    use consts_coms    , only : day_sec       ! ! intent(in)
  2558)   
  2559)    implicit none
  2560)    !----- Arguments. ----------------------------------------------------------------------!
  2561)    character(len=str_len), intent(in)  :: infile
  2562)    type(edtype)          , target      :: cgrid
  2563)    integer               , intent(in)  :: iformat
  2564)    integer               , intent(in)  :: iv
  2565)    character(len=3)      , intent(in)  :: mname
  2566)    integer               , intent(in)  :: year
  2567)    integer               , intent(in)  :: offset
  2568)    !----- Local variables. ----------------------------------------------------------------!
  2569)    integer                             :: ipy
  2570)    integer                             :: points_per_day
  2571)    integer                             :: nday
  2572)    integer                             :: nday_dset
  2573)    integer                             :: np
  2574)    integer                             :: ip
  2575)    integer                             :: np_dset
  2576)    real, dimension(:,:,:), allocatable :: metvar
  2577)    real, dimension(:,:,:), allocatable :: metvar_dset
  2578)    real, dimension(:)    , allocatable :: metscalar
  2579)    integer                             :: ndims
  2580)    integer, dimension(3)               :: idims
  2581)    integer                             :: ilon
  2582)    integer                             :: ilat
  2583)    integer                             :: ioa
  2584)    integer                             :: ioz
  2585)    integer                             :: d
  2586)    !----- External functions. -------------------------------------------------------------!
  2587)    logical, external                   :: isleap
  2588)    !---------------------------------------------------------------------------------------!
  2589) 
  2590) 
  2591)    !----- Geoposition variables do not apply. ---------------------------------------------!
  2592)    if (met_vars(iformat,iv).eq.'lat' .or. met_vars(iformat,iv).eq.'lon') return
  2593)   
  2594) 
  2595)    !---------------------------------------------------------------------------------------!
  2596)    !  Determine how many times to read.                                                    !
  2597)    !---------------------------------------------------------------------------------------!
  2598)   
  2599)    select case (met_interp(iformat,iv))
  2600)    case (2,4)
  2601)       !------------------------------------------------------------------------------------!
  2602)       !     In cases 2 and 4 we are dealing with constant time, only 1 point.              !
  2603)       !------------------------------------------------------------------------------------!
  2604)       np = 1
  2605)    case default
  2606) 
  2607)       !------------------------------------------------------------------------------------!
  2608)       !      Allocate for the number of points in the current year. In case this month is  !
  2609)       ! February, we must make sure that we account for the possibility of having a mis-   !
  2610)       ! match in the number of days in the model and in the dataset.                       !
  2611)       ! 1.  If the model year is leap but the dataset is not, we repeat February 28 twice. !
  2612)       ! 2.  If the model year is not leap but the dataset is, we drop February 29.         !
  2613)       !------------------------------------------------------------------------------------!
  2614) 
  2615)       points_per_day = nint(day_sec/met_frq(iformat,iv))
  2616) 
  2617)       !----- Determine the number of points necessary for the model year. -----------------!
  2618)       select case (trim(mname))
  2619)       case ('APR','JUN','SEP','NOV')
  2620)          nday = 30
  2621)          nday_dset = 30
  2622)       case ('JAN','MAR','MAY','JUL','AUG','OCT','DEC')
  2623)          nday = 31
  2624)          nday_dset = 31
  2625)       case ('FEB')
  2626)          if (isleap(year)) then
  2627)             nday = 29
  2628)          else 
  2629)             nday = 28
  2630)          end if
  2631) 
  2632)       !!NEVER ASSUME ANYTHING ABOUT THE DATASET
  2633) 
  2634)          call shdf5_info_f(met_vars(iformat,iv),ndims,idims)
  2635)          nday_dset=idims(1)/points_per_day
  2636)  !        write (unit=*,fmt='(2(a,1x))') ' Reading Dataset : ', trim(met_vars(iformat,iv))
  2637)  !        write (unit=*,fmt='(2(a,i5))') ' for year        : ', year
  2638)  !        write (unit=*,fmt='(2(a,i5))') ' for dataset year: ', year_use
  2639)  !        write(unit=*,fmt='(a,1x,i5)')  ' NUMBER DAYS in FEB set to  = ', nday_dset
  2640)          
  2641) 
  2642)       end select 
  2643) 
  2644)       np      = nday      * points_per_day
  2645)       np_dset = nday_dset * points_per_day
  2646) 
  2647)    end select
  2648) 
  2649)    !----- Allocate the buffer space. ------------------------------------------------------!
  2650)   
  2651)    select case (met_interp(iformat,iv))
  2652)    case (3,5) 
  2653)       !------------------------------------------------------------------------------------!
  2654)       !     Cases 3 and 5, reading in one value representing the entire domain, though     !
  2655)       ! they change over time.                                                             !
  2656)       !------------------------------------------------------------------------------------!
  2657)       ndims = 1
  2658)       idims(1) = np_dset
  2659)       
  2660)       allocate(metvar(np,met_nlon(iformat),met_nlat(iformat)))
  2661)       allocate(metscalar(np_dset))
  2662)       call shdf5_irec_f(ndims,idims,trim(met_vars(iformat,iv)),rvara = metscalar)
  2663)       
  2664)       if (np < np_dset) then     
  2665)          !----- No interpolation is needed, but we need two buffers. ----------------------!
  2666)          do ip=1,np
  2667)             metvar(ip,:,:) = metscalar(ip)
  2668)          end do
  2669) 
  2670)       elseif(np == np_dset) then
  2671)          !----- Datasets are the same, no double allocation. ------------------------------!
  2672)          do ip=1,np
  2673)             metvar(ip,:,:) = metscalar(ip)
  2674)          end do
  2675) 
  2676)       else
  2677)          !---------------------------------------------------------------------------------!
  2678)          !     The dataset buffer is smaller than the models buffer, so we recycle the     !
  2679)          ! last day.                                                                       !
  2680)          !---------------------------------------------------------------------------------!
  2681)          do ip=1,np_dset
  2682)             metvar(ip,:,:) = metscalar(ip)
  2683)          end do
  2684) 
  2685)          do ip=np_dset+1,np
  2686)             !----- Then reuse the 28th for the 29th day. ----------------------------------!
  2687)             metvar(ip,:,:) = metscalar(ip-points_per_day+1)
  2688)          end do
  2689)       end if
  2690) 
  2691)       deallocate(metscalar)
  2692) 
  2693)    case (4)
  2694)       !------------------------------------------------------------------------------------!
  2695)       !      Case 4, we don't read, instead we use the value stored in met_freq as the     !
  2696)       ! scalar value.                                                                      !
  2697)       !------------------------------------------------------------------------------------!
  2698)       allocate (metvar(1,met_nlon(iformat),met_nlat(iformat)))
  2699)       metvar(1,:,:) = met_frq(iformat,iv)
  2700)       
  2701)    case (2)
  2702)       !----- Case 2, 2D grid is read in, ie. np = 1. --------------------------------------!
  2703) 
  2704)       allocate(metvar(1,met_nlon(iformat),met_nlat(iformat)))
  2705) 
  2706)       call shdf5_info_f(trim(met_vars(iformat,iv)),ndims,idims)
  2707)       
  2708)       !----- Check to see if the dimensions are consistent. -------------------------------!
  2709)       if (ndims /= 2 .or. idims(1) /= met_nlon(iformat)                                    &
  2710)                      .or. idims(2) /= met_nlat(iformat)) then
  2711)          write (unit=*,fmt='(2(a,1x))') ' In file: ',trim(infile)
  2712)          write (unit=*,fmt='(2(a,1x))') ' Dataset: ', trim(met_vars(iformat,iv))
  2713)          write (unit=*,fmt='(a)')       ' doesn''t have dimensions that match the'
  2714)          write (unit=*,fmt='(a)')       ' specified input, or latitude/longitude grid...'
  2715)          write (unit=*,fmt='(a,1x,i5)') ' MET_NLON = ',met_nlon(iformat)
  2716)          write (unit=*,fmt='(a,1x,i5)') ' MET_NLAT = ',met_nlat(iformat)
  2717)          write (unit=*,fmt='(a,1x,i5)') ' NDIMS    = ',ndims
  2718)          do d=1,ndims
  2719)             write(unit=*,fmt='(a,i5,a,1x,i5)') 'IDIMS (',d,') =', idims(d)
  2720)          end do
  2721)          call fatal_error('Mismatch between dataset and specified input'                   &
  2722)                          ,'read_ol_file','ed_met_driver.f90')
  2723)       end if
  2724) 
  2725)       call shdf5_irec_f(ndims,idims,trim(met_vars(iformat,iv)),rvara = metvar)
  2726) 
  2727)    case default
  2728)       !----- Case 1 or 0, we must read time-and-space variable array. ---------------------!
  2729)       call shdf5_info_f(trim(met_vars(iformat,iv)),ndims,idims)
  2730)       
  2731)       ! Check to see if the dimensions are consistent
  2732)       if (ndims /= 3 .or. idims(1).ne.np_dset .or. idims(2).ne.met_nlon(iformat)           &
  2733)                                               .or. idims(3).ne.met_nlat(iformat)) then
  2734)          print*,"DATASET: ",trim(met_vars(iformat,iv))
  2735)          print*,"DOES NOT HAVE DIMENSIONS THAT MATCH THE"
  2736)          print*,"SPECIFIED INPUT, OR LAT/LON GRID"
  2737)          print*,ndims
  2738)          print*,np_dset,met_nlon(iformat),met_nlat(iformat)
  2739)          print*,idims
  2740)          write (unit=*,fmt='(2(a,1x))') ' In file: ',trim(infile)
  2741)          write (unit=*,fmt='(2(a,1x))') ' Dataset: ', trim(met_vars(iformat,iv))
  2742)          write (unit=*,fmt='(a)')       ' doesn''t have dimensions that match the'
  2743)          write (unit=*,fmt='(a)')       ' specified input, or latitude/longitude grid...'
  2744)          write (unit=*,fmt='(a,1x,i5)') ' NP_DSET  = ', np_dset
  2745)          write (unit=*,fmt='(a,1x,i5)') ' MET_NLON = ', met_nlon(iformat)
  2746)          write (unit=*,fmt='(a,1x,i5)') ' MET_NLAT = ', met_nlat(iformat)
  2747)          write (unit=*,fmt='(a,1x,i5)') ' NDIMS    = ', ndims
  2748)          do d=1,ndims
  2749)             write(unit=*,fmt='(a,i5,a,1x,i5)') 'IDIMS (',d,') =', idims(d)
  2750)          end do
  2751)          call fatal_error('Mismatch between dataset and specified input' &
  2752)                          ,'read_ol_file','ed_met_driver.f90')
  2753)       end if
  2754) 
  2755)       if (np < np_dset) then
  2756)          !----- No interpolation is needed, but we need two buffers. ----------------------!
  2757)          allocate(metvar(np,met_nlon(iformat),met_nlat(iformat)))
  2758)          allocate(metvar_dset(np_dset,met_nlon(iformat),met_nlat(iformat)))
  2759)          
  2760)          call shdf5_irec_f(ndims, idims, trim(met_vars(iformat,iv)), rvara = metvar_dset)
  2761)          metvar(1:np,:,:) = metvar_dset(1:np,:,:)
  2762)          deallocate(metvar_dset)
  2763)          
  2764)          
  2765)       elseif (np == np_dset) then
  2766)          !----- Datasets are the same, no double allocation needed. -----------------------!
  2767)          allocate(metvar(np,met_nlon(iformat),met_nlat(iformat)))
  2768)          call shdf5_irec_f(ndims, idims, trim(met_vars(iformat,iv)), rvara = metvar)
  2769) 
  2770)       else
  2771)          !---------------------------------------------------------------------------------!
  2772)          !     The dataset buffer is smaller than the models buffer, so we recycle the     !
  2773)          ! last day.                                                                       !
  2774)          !---------------------------------------------------------------------------------!
  2775)          allocate(metvar(np,met_nlon(iformat),met_nlat(iformat)))
  2776)          allocate(metvar_dset(np_dset,met_nlon(iformat),met_nlat(iformat)))
  2777) 
  2778)          call shdf5_irec_f(ndims,idims,trim(met_vars(iformat,iv)),rvara=metvar_dset)
  2779)          metvar(1:np_dset,:,:) = metvar_dset(1:np_dset,:,:)
  2780)          
  2781)          !----- Then reuse the 28th for the 29th day. -------------------------------------!
  2782)          metvar(np_dset+1:np,:,:) = metvar_dset(np_dset-points_per_day+1:np_dset,:,:)
  2783)          deallocate(metvar_dset)
  2784)       end if
  2785)    end select
  2786) 
  2787)    !----- Define some aliases for the indices. --------------------------------------------!
  2788)    ioa = offset + 1
  2789)    ioz = offset + np
  2790) 
  2791)    !---------------------------------------------------------------------------------------!
  2792)    !      The dataset for that variable has been read into an array.  Now, the polygons    !
  2793)    ! are cycled through, and they are associated with an index in the grid.  If this is a  !
  2794)    ! lat-lon grid, the procedure is a simple interpolation between the domain endpoints,   !
  2795)    ! if this is a polar stereo-graphic projection, we make use of the associated indices   !
  2796)    ! saved in memory, ie. (cgrid%ilon(ipy)).                                               !
  2797)    !---------------------------------------------------------------------------------------!
  2798)    do ipy = 1,cgrid%npolygons
  2799)       
  2800)       !----- Get the indices.  Remember, latitude is flipped. -----------------------------!
