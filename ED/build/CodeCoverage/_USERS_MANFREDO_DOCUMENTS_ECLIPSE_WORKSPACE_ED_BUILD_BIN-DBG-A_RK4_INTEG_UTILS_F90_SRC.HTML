     6) ! Subroutine odeint                                                                        !
     7) !                                                                                          !
     8) !     This subroutine will drive the integration of several ODEs that drive the fast-scale !
     9) ! state variables.                                                                         !
    10) !------------------------------------------------------------------------------------------!
    11) subroutine odeint(h1,csite,ipa,isi,nsteps)
    12)    use rk4_copy_patch
    13)    use rk4_misc
    14)    use ed_state_vars  , only : sitetype               & ! structure
    15)                              , patchtype              & ! structure
    16)                              , polygontype
    17)    use rk4_coms       , only : integration_vars       & ! structure
    18)                              , integration_buff       & ! intent(inout)
    19)                              , maxstp                 & ! intent(in)
    20)                              , tbeg                   & ! intent(in)
    21)                              , tend                   & ! intent(in)
    22)                              , dtrk4                  & ! intent(in)
    23)                              , tiny_offset            & ! intent(in)
    24)                              , checkbudget            & ! intent(in)
    25)                              , norm_rk4_fluxes        ! ! sub-routine
    26)    use ed_misc_coms   , only : fast_diagnostics       ! ! intent(in)
    27)    use grid_coms      , only : nzg                    & ! intent(in)
    28)                              , nzs                    ! ! intent(in)
    29)    use soil_coms      , only : runoff_time_i          &
    30)                              , simplerunoff
    31)    use consts_coms    , only : wdnsi8                 ! ! intent(in)
    32)    use therm_lib8     , only : tl2uint8               ! ! intent(in)
    33)    !$ use omp_lib
    34) 
    35)    implicit none
    36)    !----- Arguments -----------------------------------------------------------------------!
    37)    type(sitetype)            , target      :: csite            ! Current site
    38)    integer                   , intent(in)  :: ipa              ! Current patch ID
    39)    integer                   , intent(in)  :: isi              ! Current site ID
    40)    real(kind=8)              , intent(in)  :: h1               ! First guess of delta-t
    41)    integer                   , intent(out) :: nsteps           ! Number of steps taken.
    42)    !----- Local variables -----------------------------------------------------------------!
    43)    type(patchtype)           , pointer     :: cpatch           ! Current patch
    44)    integer                                 :: i                ! Step counter
    45)    integer                                 :: ksn              ! # of snow/water layers
    46)    real(kind=8)                            :: x                ! Elapsed time
    47)    real(kind=8)                            :: h                ! Current delta-t attempt
    48)    real(kind=8)                            :: hnext            ! Next delta-t
    49)    real(kind=8)                            :: hdid             ! delta-t that worked (???)
    50)    real(kind=8)                            :: qwfree           ! Free water internal energy
    51)    real(kind=8)                            :: wfreeb           ! Free water 
    52)    integer                                 :: ibuff
    53) 
    54)    !----- External function. --------------------------------------------------------------!
    55)    real                      , external    :: sngloff
    56)    
    57)    !---------------------------------------------------------------------------------------!
    58) 
    59)    ibuff = 1
    60)    !$ ibuff = OMP_get_thread_num()+1
    61) 
    62)    cpatch => csite%patch(ipa)
    63) 
    64)    !---------------------------------------------------------------------------------------!
    65)    !     Copy the initial patch to the one we use for integration.                         !
    66)    !---------------------------------------------------------------------------------------!
    67)    call copy_rk4_patch(integration_buff(ibuff)%initp, integration_buff(ibuff)%y,cpatch)
    68)    !---------------------------------------------------------------------------------------!
    69) 
    70) 
    71)    !---------------------------------------------------------------------------------------!
    72)    ! Set initial time and stepsize.                                                        !
    73)    !---------------------------------------------------------------------------------------!
    74)    x = tbeg
    75)    h = h1
    76)    if (dtrk4 < 0.d0) h = -h1
    77)    !---------------------------------------------------------------------------------------!
    78) 
    79) 
    80) 
    81)    !---------------------------------------------------------------------------------------!
    82)    ! Begin timestep loop                                                                   !
    83)    !---------------------------------------------------------------------------------------!
    84)    timesteploop: do i=1,maxstp
    85) 
    86)       !----- Get initial derivatives ------------------------------------------------------!
    87)       call leaf_derivs(integration_buff(ibuff)%y,integration_buff(ibuff)%dydx,csite,ipa,h,.false.)
    88) 
    89)       !----- Get scalings used to determine stability -------------------------------------!
    90)       call get_yscal(integration_buff(ibuff)%y, integration_buff(ibuff)%dydx,h,integration_buff(ibuff)%yscal    &
    91)                        ,cpatch)
    92) 
    93)       !----- Be sure not to overstep ------------------------------------------------------!
    94)       if((x+h-tend)*(x+h-tbeg) > 0.d0) h=tend-x
    95) 
    96)       !----- Take the step ----------------------------------------------------------------!
    97)       call rkqs(x,h,hdid,hnext,csite,ipa,isi)
    98) 
    99)       !----- If the integration reached the next step, make some final adjustments --------!
   100)       if((x-tend)*dtrk4 >= 0.d0)then
   101) 
   102)          ksn = integration_buff(ibuff)%y%nlev_sfcwater
   103) 
   104)          !---------------------------------------------------------------------------------!
   105)          !   Make temporary surface liquid water disappear.  This will not happen          !
   106)          ! immediately, but liquid water will decay with the time scale defined by         !
   107)          ! runoff_time scale. If the time scale is too tiny, then it will be forced to be  !
   108)          ! hdid (no reason to be faster than that).                                        !
   109)          !---------------------------------------------------------------------------------!
   110)          if (simplerunoff .and. ksn >= 1) then
   111)             if (integration_buff(ibuff)%y%sfcwater_mass(ksn)    > 0.d0   .and.                    &
   112)                 integration_buff(ibuff)%y%sfcwater_fracliq(ksn) > 1.d-1) then
   113) 
   114)                wfreeb = min(1.d0, dtrk4 * runoff_time_i)                                   &
   115)                       * integration_buff(ibuff)%y%sfcwater_mass(ksn)                              &
   116)                       * (integration_buff(ibuff)%y%sfcwater_fracliq(ksn) - 1.d-1) / 9.d-1
   117) 
   118)                qwfree = wfreeb * tl2uint8(integration_buff(ibuff)%y%sfcwater_tempk(ksn),1.d0)
   119) 
   120)                integration_buff(ibuff)%y%sfcwater_mass(ksn) =                                     &
   121)                                    integration_buff(ibuff)%y%sfcwater_mass(ksn)                   &
   122)                                  - wfreeb
   123) 
   124)                integration_buff(ibuff)%y%sfcwater_depth(ksn) =                                    &
   125)                                    integration_buff(ibuff)%y%sfcwater_depth(ksn)                  &
   126)                                  - wfreeb*wdnsi8
   127) 
   128)                !----- Remove internal energy lost due to runoff. --------------------------!
   129)                integration_buff(ibuff)%y%sfcwater_energy(ksn) =                                   &
   130)                                      integration_buff(ibuff)%y%sfcwater_energy(ksn) - qwfree
   131) 
   132)                call adjust_sfcw_properties(nzg,nzs,integration_buff(ibuff)%y,dtrk4,csite)
   133)                call update_diagnostic_vars(integration_buff(ibuff)%y,csite,ipa)
   134) 
   135)                !----- Compute runoff for output -------------------------------------------!
   136)                if (fast_diagnostics) then
   137)                   !------------------------------------------------------------------------!
   138)                   !      There is no need to divide wfreeb and qwfree  by time step, which !
   139)                   ! will be done in subroutine normalize_averaged_vars.                    !
   140)                   !------------------------------------------------------------------------!
   141)                   csite%runoff       (ipa) = csite%runoff(ipa)                             &
   142)                                            + sngloff(wfreeb,tiny_offset)
   143)                   csite%fmean_runoff (ipa) = csite%fmean_runoff(ipa)                       &
   144)                                            + sngloff(wfreeb,tiny_offset)
   145)                   csite%fmean_qrunoff(ipa) = csite%fmean_qrunoff(ipa)                      &
   146)                                            + sngloff(qwfree,tiny_offset)
   147)                end if
   148)                if (checkbudget) then
   149)                   !------------------------------------------------------------------------!
   150)                   !      To make sure that the previous values of wbudget_loss2runoff and  !
   151)                   ! ebudget_loss2runoff are accumulated to the next time step.             !
   152)                   !------------------------------------------------------------------------!
   153)                   integration_buff(ibuff)%y%wbudget_loss2runoff = wfreeb                          &
   154)                                     + integration_buff(ibuff)%y%wbudget_loss2runoff
   155)                   integration_buff(ibuff)%y%ebudget_loss2runoff = qwfree                          &
   156)                                     + integration_buff(ibuff)%y%ebudget_loss2runoff
   157)                   integration_buff(ibuff)%y%wbudget_storage =                                     &
   158)                                       integration_buff(ibuff)%y%wbudget_storage - wfreeb
   159)                   integration_buff(ibuff)%y%ebudget_storage =                                     &
   160)                                       integration_buff(ibuff)%y%ebudget_storage - qwfree
   161)                end if
   162)             end if
   163)          end if
   164) 
   165)          !------ Copy the temporary patch to the next intermediate step -------------------!
   166)          call copy_rk4_patch(integration_buff(ibuff)%y,integration_buff(ibuff)%initp, cpatch)
   167)          !------ Update the substep for next time and leave -------------------------------!
   168)          csite%htry(ipa) = sngl(hnext)
   169) 
   170)          !---------------------------------------------------------------------------------!
   171)          !     Update the average time step.  The square of DTLSM (tend-tbeg) is needed    !
   172)          ! because we will divide this by the time between t0 and t0+frqsum.               !
   173)          !---------------------------------------------------------------------------------!
   174)          csite%fmean_rk4step(ipa) = csite%fmean_rk4step(ipa)                               &
   175)                                   + sngl((tend-tbeg)*(tend-tbeg))/real(i)
   176)          nsteps = i
   177)          !---------------------------------------------------------------------------------!
   178)          return
   179)       end if
   180)       
   181)       !----- Use hnext as the next substep ------------------------------------------------!
   182)       h = hnext
   183)    end do timesteploop
   184) 
   185)    !----- If it reached this point, that is really bad news... ----------------------------!
   186)    write (unit=*,fmt='(a)') ' ==> Too many steps in routine odeint'
   187)    call print_rk4patch(integration_buff(ibuff)%y, csite,ipa)
   188) 
   189)    return
   190) end subroutine odeint

   313)    !---------------------------------------------------------------------------------------!

   353) !    This subroutines increment the derivative into the previous guess to create the new   !
   354) ! guess.                                                                                   !
   355) !------------------------------------------------------------------------------------------!
   356) subroutine inc_rk4_patch(rkp, inc, fac, cpatch)
   357)    use ed_state_vars , only : sitetype           & ! structure
   358)                             , patchtype          ! ! structure
   359)    use rk4_coms      , only : rk4patchtype       & ! structure
   360)                             , rk4site            & ! intent(in)
   361)                             , checkbudget        & ! intent(in)
   362)                             , print_detailed     ! ! intent(in)
   363)    use grid_coms     , only : nzg                ! ! intent(in)
   364)    use ed_misc_coms  , only : fast_diagnostics   ! ! intent(in)
   365)    implicit none
   366) 
   367)    !----- Arguments -----------------------------------------------------------------------!
   368)    type(rk4patchtype) , target     :: rkp    ! Temporary patch with previous state
   369)    type(rk4patchtype) , target     :: inc    ! Temporary patch with its derivatives
   370)    type(patchtype)    , target     :: cpatch ! Current patch (for characteristics)
   371)    real(kind=8)       , intent(in) :: fac    ! Increment factor
   372)    !----- Local variables -----------------------------------------------------------------!
   373)    integer                         :: ico    ! Cohort ID
   374)    integer                         :: k      ! Counter
   375)    !---------------------------------------------------------------------------------------!
   376) 
   377)    rkp%can_enthalpy = rkp%can_enthalpy + fac * inc%can_enthalpy
   378)    rkp%can_shv      = rkp%can_shv      + fac * inc%can_shv
   379)    rkp%can_co2      = rkp%can_co2      + fac * inc%can_co2
   380) 
   381)    do k=rk4site%lsl,nzg
   382)       rkp%soil_water(k)       = rkp%soil_water(k)  + fac * inc%soil_water(k)
   383)       rkp%soil_energy(k)      = rkp%soil_energy(k) + fac * inc%soil_energy(k)
   384)    end do
   385) 
   386)    do k=1,rkp%nlev_sfcwater
   387)       rkp%sfcwater_mass(k)   = rkp%sfcwater_mass(k)   + fac * inc%sfcwater_mass(k)
   388)       rkp%sfcwater_energy(k) = rkp%sfcwater_energy(k) + fac * inc%sfcwater_energy(k)
   389)       rkp%sfcwater_depth(k)  = rkp%sfcwater_depth(k)  + fac * inc%sfcwater_depth(k)
   390)    end do
   391) 
   392)    rkp%virtual_energy  = rkp%virtual_energy  + fac * inc%virtual_energy
   393)    rkp%virtual_water   = rkp%virtual_water   + fac * inc%virtual_water
   394)    rkp%virtual_depth   = rkp%virtual_depth   + fac * inc%virtual_depth
   395) 
   396)    rkp%water_deficit   = rkp%water_deficit   + fac * inc%water_deficit
   397) 
   398)    rkp%upwp = rkp%upwp + fac * inc%upwp
   399)    rkp%wpwp = rkp%wpwp + fac * inc%wpwp
   400)    rkp%tpwp = rkp%tpwp + fac * inc%tpwp
   401)    rkp%qpwp = rkp%qpwp + fac * inc%qpwp
   402)    rkp%cpwp = rkp%cpwp + fac * inc%cpwp
   403) 
   404)    do ico = 1,cpatch%ncohorts
   405)       rkp%leaf_water (ico) = rkp%leaf_water (ico) + fac * inc%leaf_water (ico)
   406)       rkp%leaf_energy(ico) = rkp%leaf_energy(ico) + fac * inc%leaf_energy(ico)
   407)       rkp%wood_water (ico) = rkp%wood_water (ico) + fac * inc%wood_water (ico)
   408)       rkp%wood_energy(ico) = rkp%wood_energy(ico) + fac * inc%wood_energy(ico)
   409)       rkp%veg_water (ico)  = rkp%veg_water  (ico) + fac * inc%veg_water  (ico)
   410)       rkp%veg_energy(ico)  = rkp%veg_energy (ico) + fac * inc%veg_energy (ico)
   411) 
   412)       rkp%psi_open  (ico) = rkp%psi_open  (ico) + fac * inc%psi_open  (ico)
   413)       rkp%psi_closed(ico) = rkp%psi_closed(ico) + fac * inc%psi_closed(ico)
   414)    end do
   415) 
   416)    if (checkbudget) then
   417) 
   418)       rkp%co2budget_storage      = rkp%co2budget_storage     + fac * inc%co2budget_storage
   419)       rkp%co2budget_loss2atm     = rkp%co2budget_loss2atm    + fac * inc%co2budget_loss2atm
   420) 
   421)       rkp%wbudget_storage       = rkp%wbudget_storage       + fac * inc%wbudget_storage
   422)       rkp%wbudget_loss2atm      = rkp%wbudget_loss2atm      + fac * inc%wbudget_loss2atm
   423)       rkp%wbudget_loss2drainage = rkp%wbudget_loss2drainage                                &
   424)                                 + fac * inc%wbudget_loss2drainage
   425) 
   426)       rkp%ebudget_storage       = rkp%ebudget_storage       + fac * inc%ebudget_storage
   427)       rkp%ebudget_netrad        = rkp%ebudget_netrad        + fac * inc%ebudget_netrad
   428)       rkp%ebudget_loss2atm      = rkp%ebudget_loss2atm      + fac * inc%ebudget_loss2atm
   429)       rkp%ebudget_loss2drainage = rkp%ebudget_loss2drainage                                &
   430)                                 + fac * inc%ebudget_loss2drainage
   431)    end if
   432)    if (fast_diagnostics) then
   433)       rkp%avg_ustar          = rkp%avg_ustar          + fac * inc%avg_ustar
   434)       rkp%avg_tstar          = rkp%avg_tstar          + fac * inc%avg_tstar
   435)       rkp%avg_qstar          = rkp%avg_qstar          + fac * inc%avg_qstar
   436)       rkp%avg_cstar          = rkp%avg_cstar          + fac * inc%avg_cstar
   437) 
   438) 
   439)       rkp%avg_carbon_ac      = rkp%avg_carbon_ac      + fac * inc%avg_carbon_ac
   440)       rkp%avg_carbon_st      = rkp%avg_carbon_st      + fac * inc%avg_carbon_st
   441) 
   442)       rkp%avg_throughfall    = rkp%avg_throughfall    + fac * inc%avg_throughfall
   443)       rkp%avg_vapor_ac       = rkp%avg_vapor_ac       + fac * inc%avg_vapor_ac
   444)       rkp%avg_vapor_gc       = rkp%avg_vapor_gc       + fac * inc%avg_vapor_gc
   445)       rkp%avg_drainage       = rkp%avg_drainage       + fac * inc%avg_drainage
   446)       rkp%avg_qdrainage      = rkp%avg_qdrainage      + fac * inc%avg_qdrainage
   447)       rkp%avg_qthroughfall   = rkp%avg_qthroughfall   + fac * inc%avg_qthroughfall
   448)       rkp%avg_sensible_gc    = rkp%avg_sensible_gc    + fac * inc%avg_sensible_gc
   449)       rkp%avg_sensible_ac    = rkp%avg_sensible_ac    + fac * inc%avg_sensible_ac
   450) 
   451)       do k=rk4site%lsl,nzg
   452)          rkp%avg_sensible_gg(k)  = rkp%avg_sensible_gg(k)  + fac * inc%avg_sensible_gg(k)
   453)          rkp%avg_smoist_gg(k)    = rkp%avg_smoist_gg(k)    + fac * inc%avg_smoist_gg(k)  
   454)          rkp%avg_transloss(k)    = rkp%avg_transloss(k)    + fac * inc%avg_transloss(k)  
   455)       end do
   456) 
   457) 
   458)       do k=1,cpatch%ncohorts
   459)          rkp%avg_sensible_lc   (k) =       rkp%avg_sensible_lc   (k)                       &
   460)                                    + fac * inc%avg_sensible_lc   (k)
   461)          rkp%avg_sensible_wc   (k) =       rkp%avg_sensible_wc   (k)                       &
   462)                                    + fac * inc%avg_sensible_wc   (k)
   463)          rkp%avg_vapor_lc      (k) =       rkp%avg_vapor_lc      (k)                       &
   464)                                    + fac * inc%avg_vapor_lc      (k)
   465)          rkp%avg_vapor_wc      (k) =       rkp%avg_vapor_wc      (k)                       &
   466)                                    + fac * inc%avg_vapor_wc      (k)
   467)          rkp%avg_transp        (k) =       rkp%avg_transp        (k)                       &
   468)                                    + fac * inc%avg_transp        (k)
   469)          rkp%avg_intercepted_al(k) =       rkp%avg_intercepted_al(k)                       &
   470)                                    + fac * inc%avg_intercepted_al(k)
   471)          rkp%avg_intercepted_aw(k) =       rkp%avg_intercepted_aw(k)                       &
   472)                                    + fac * inc%avg_intercepted_aw(k)
   473)          rkp%avg_wshed_lg      (k) =       rkp%avg_wshed_lg      (k)                       &
   474)                                    + fac * inc%avg_wshed_lg      (k)
   475)          rkp%avg_wshed_wg      (k) =       rkp%avg_wshed_wg      (k)                       &
   476)                                    + fac * inc%avg_wshed_wg      (k)
   477)       end do
   478) 
   479)    end if
   480) 
   481)    !---------------------------------------------------------------------------------------!
   482)    !    Increment the instantaneous fluxes.  The derivative term should be the same as the !
   483)    ! the full fluxes, the only difference is that these variables are normalised and       !
   484)    ! re-set after each time step.                                                          !
   485)    !---------------------------------------------------------------------------------------!
   486)    if (print_detailed) then
   487)       rkp%flx_carbon_ac      = rkp%flx_carbon_ac      + fac * inc%avg_carbon_ac
   488)       rkp%flx_carbon_st      = rkp%flx_carbon_st      + fac * inc%avg_carbon_st
   489) 
   490)       rkp%flx_vapor_gc       = rkp%flx_vapor_gc       + fac * inc%avg_vapor_gc
   491)       rkp%flx_throughfall    = rkp%flx_throughfall    + fac * inc%avg_throughfall
   492)       rkp%flx_vapor_ac       = rkp%flx_vapor_ac       + fac * inc%avg_vapor_ac
   493)       rkp%flx_drainage       = rkp%flx_drainage       + fac * inc%avg_drainage
   494)       rkp%flx_qdrainage      = rkp%flx_qdrainage      + fac * inc%avg_qdrainage
   495)       rkp%flx_qthroughfall   = rkp%flx_qthroughfall   + fac * inc%avg_qthroughfall
   496)       rkp%flx_sensible_gc    = rkp%flx_sensible_gc    + fac * inc%avg_sensible_gc
   497)       rkp%flx_sensible_ac    = rkp%flx_sensible_ac    + fac * inc%avg_sensible_ac
   498) 
   499)       do k=rk4site%lsl,nzg
   500)          rkp%flx_sensible_gg(k)  = rkp%flx_sensible_gg(k)  + fac * inc%avg_sensible_gg(k)
   501)          rkp%flx_smoist_gg(k)    = rkp%flx_smoist_gg(k)    + fac * inc%avg_smoist_gg(k)  
   502)          rkp%flx_transloss(k)    = rkp%flx_transloss(k)    + fac * inc%avg_transloss(k)  
   503)       end do
   504) 
   505)       do ico = 1,cpatch%ncohorts
   506)          rkp%flx_vapor_lc           =         rkp%flx_vapor_lc                             &
   507)                                     + fac *   inc%avg_vapor_lc       (ico)
   508)          rkp%flx_vapor_wc           =         rkp%flx_vapor_wc                             &
   509)                                     + fac *   inc%avg_vapor_wc       (ico)
   510)          rkp%flx_wshed_vg           =         rkp%flx_wshed_vg                             &
   511)                                     + fac * ( inc%avg_wshed_lg       (ico)                 &
   512)                                             + inc%avg_wshed_wg       (ico) )
   513)          rkp%flx_intercepted        =       rkp%flx_intercepted                            &
   514)                                     + fac * ( inc%avg_intercepted_al (ico)                 &
   515)                                             + inc%avg_intercepted_aw (ico) )
   516)          rkp%flx_sensible_lc        =         rkp%flx_sensible_lc                          &
   517)                                     + fac *   inc%avg_sensible_lc    (ico)
   518)          rkp%flx_sensible_wc        =         rkp%flx_sensible_wc                          &
   519)                                     + fac *   inc%avg_sensible_wc    (ico)
   520)          rkp%flx_qwshed_vg          =         rkp%flx_qwshed_vg                            &
   521)                                     + fac *   inc%cfx_qwshed         (ico)
   522)          rkp%flx_qintercepted       =         rkp%flx_qintercepted                         &
   523)                                     + fac *   inc%cfx_qintercepted   (ico)
   524)          rkp%cfx_hflxlc      (ico)  =         rkp%cfx_hflxlc         (ico)                 &
   525)                                     + fac *   inc%cfx_hflxlc         (ico)
   526)          rkp%cfx_hflxwc      (ico)  =         rkp%cfx_hflxwc         (ico)                 &
   527)                                     + fac *   inc%cfx_hflxwc         (ico)
   528)          rkp%cfx_qwflxlc     (ico)  =         rkp%cfx_qwflxlc        (ico)                 &
   529)                                     + fac *   inc%cfx_qwflxlc        (ico)
   530)          rkp%cfx_qwflxwc     (ico)  =         rkp%cfx_qwflxwc        (ico)                 &
   531)                                     + fac *   inc%cfx_qwflxwc        (ico)
   532)          rkp%cfx_qwshed      (ico)  =         rkp%cfx_qwshed         (ico)                 &
   533)                                     + fac *   inc%cfx_qwshed         (ico)
   534)          rkp%cfx_qtransp     (ico)  =         rkp%cfx_qtransp        (ico)                 &
   535)                                     + fac *   inc%cfx_qtransp        (ico)
   536)          rkp%cfx_qintercepted(ico)  =         rkp%cfx_qintercepted   (ico)                 &
   537)                                     + fac *   inc%cfx_qintercepted   (ico)
   538)       end do
   539) 
   540)    end if
   541)    !---------------------------------------------------------------------------------------!
   542) 
   543)    return

   553) !==========================================================================================!
   554) !==========================================================================================!
   555) !    This subroutine finds the error scale for the integrated variables, which will be     !
   556) ! later used to define the relative error.                                                 !
   557) !------------------------------------------------------------------------------------------!
   558) subroutine get_yscal(y,dy,htry,yscal,cpatch)
   559)    use ed_state_vars        , only : patchtype             ! ! structure
   560)    use rk4_coms             , only : rk4patchtype          & ! structure
   561)                                    , rk4site               & ! intent(in)
   562)                                    , ibranch_thermo        & ! intent(in)
   563)                                    , tiny_offset           & ! intent(in)
   564)                                    , huge_offset           & ! intent(in)
   565)                                    , rk4water_stab_thresh  & ! intent(in)
   566)                                    , rk4tiny_sfcw_mass     & ! intent(in)
   567)                                    , rk4leaf_drywhc        & ! intent(in)
   568)                                    , checkbudget           ! ! intent(in)
   569)    use grid_coms            , only : nzg                   & ! intent(in)
   570)                                    , nzs                   ! ! intent(in)
   571)    use consts_coms          , only : wdnsi8                ! ! intent(in)
   572)    use soil_coms            , only : isoilbc               & ! intent(in)
   573)                                    , dslzi8                ! ! intent(in)
   574)    implicit none
   575)    !----- Arguments -----------------------------------------------------------------------!
   576)    type(rk4patchtype), target     :: y                     ! Struct. with the guesses
   577)    type(rk4patchtype), target     :: dy                    ! Struct. with their derivatives
   578)    type(rk4patchtype), target     :: yscal                 ! Struct. with their scales
   579)    type(patchtype)   , target     :: cpatch                ! Current patch
   580)    real(kind=8)      , intent(in) :: htry                  ! Time-step we are trying
   581)    !----- Local variables -----------------------------------------------------------------!
   582)    real(kind=8)                   :: meanscale_sfcw_mass   ! Average Sfc. water mass scale
   583)    real(kind=8)                   :: meanscale_sfcw_energy ! Average Sfc. water en. scale
   584)    real(kind=8)                   :: meanscale_sfcw_depth  ! Average Sfc. water depth scale
   585)    integer                        :: k                     ! Counter
   586)    integer                        :: ico                   ! Current cohort ID
   587)    !---------------------------------------------------------------------------------------!
   588) 
   589)    yscal%can_enthalpy =  abs(y%can_enthalpy) + abs(dy%can_enthalpy * htry)
   590)    yscal%can_shv      =  abs(y%can_shv     ) + abs(dy%can_shv      * htry)
   591)    yscal%can_co2      =  abs(y%can_co2     ) + abs(dy%can_co2      * htry)
   592) 
   593)    !---------------------------------------------------------------------------------------!
   594)    !     We don't solve pressure prognostically, so the scale cannot be computed based on  !
   595)    ! the derivative.  Also, pressure is a variable with a large absolute variable and tiny !
   596)    ! variation, so the scale is given by the scale of the variability...                   !
   597)    !---------------------------------------------------------------------------------------!
   598)    yscal%can_prss    = abs(rk4site%atm_prss - y%can_prss)
   599)    !---------------------------------------------------------------------------------------!
   600) 
   601) 
   602)    !---------------------------------------------------------------------------------------!
   603)    !     We determine the scale for all layers but the top soil, which will be done        !
   604)    ! differently depending on the status of the temporary surface water.                   !
   605)    !---------------------------------------------------------------------------------------!
   606)    do k=1,rk4site%lsl-1
   607)       yscal%soil_water (k) = huge_offset
   608)       yscal%soil_energy(k) = huge_offset
   609)    end do
   610)    do k=rk4site%lsl,nzg-1
   611)       yscal%soil_water (k) = abs(y%soil_water(k) ) + abs(dy%soil_water(k)  * htry)
   612)       yscal%soil_energy(k) = abs(y%soil_energy(k)) + abs(dy%soil_energy(k) * htry)
   613)    end do
   614)    !---------------------------------------------------------------------------------------!
   615) 
   616) 
   617) 
   618)    !---------------------------------------------------------------------------------------!
   619)    !     Temporary surface layers require a special approach. The number of layers may     !
   620)    ! vary during the integration process, so we must make sure that all layers initially   !
   621)    ! have a scale.  Also, if the total mass is small, we must be more tolerant to avoid    !
   622)    ! overestimating the error because of the small size.                                   !
   623)    !---------------------------------------------------------------------------------------!
   624)    select case(y%flag_sfcwater)
   625)    case(0)
   626)       !------------------------------------------------------------------------------------!
   627)       !     No temporary surface water, we can skip the check as no layer will be created  !
   628)       ! until the upcoming step.                                                           !
   629)       !------------------------------------------------------------------------------------!
   630)       do k=1,nzs
   631)          yscal%sfcwater_mass(k)   = huge_offset
   632)          yscal%sfcwater_energy(k) = huge_offset
   633)          yscal%sfcwater_depth(k)  = huge_offset
   634)       end do
   635)       !------------------------------------------------------------------------------------!
   636) 
   637) 
   638)       !------ Soil scale won't be affected by the temporary surface water. ----------------!
   639)       yscal%soil_water (nzg) = abs(y%soil_water (nzg)) + abs(dy%soil_water (nzg) * htry)
   640)       yscal%soil_energy(nzg) = abs(y%soil_energy(nzg)) + abs(dy%soil_energy(nzg) * htry)
   641)       !------------------------------------------------------------------------------------!
   642) 
   643)    case(1)
   644)       !------------------------------------------------------------------------------------!
   645)       !    Low stability threshold, there can't be more than one layer, and the energy     !
   646)       ! will be solved together with the top soil layer.  Therefore, we skip energy check, !
   647)       ! but attribute the scale for mass and depth.                                        !
   648)       !------------------------------------------------------------------------------------!
   649)       yscal%sfcwater_mass  (1) = abs(y%sfcwater_mass   (1)       )                         &
   650)                                + abs(dy%sfcwater_mass  (1) * htry)
   651)       yscal%sfcwater_depth (1) = abs(y%sfcwater_depth  (1)       )                         &
   652)                                + abs(dy%sfcwater_depth (1) * htry)
   653)       yscal%sfcwater_energy(1) = huge_offset
   654)       do k=2,nzs
   655)          yscal%sfcwater_mass  (k) = huge_offset
   656)          yscal%sfcwater_energy(k) = huge_offset
   657)          yscal%sfcwater_depth (k) = huge_offset
   658)       end do
   659)       !------------------------------------------------------------------------------------!
   660) 
   661) 
   662) 
   663)       !------------------------------------------------------------------------------------!
   664)       !     Soil mass won't be affected by the temporary surface water, but soil energy    !
   665)       ! will include the energy associated with the temporary surface water energy.  In    !
   666)       ! reality the derivative of the temporary surface water will be zero, but we add for !
   667)       ! code consistency.                                                                  !
   668)       !------------------------------------------------------------------------------------!
   669)       yscal%soil_water (nzg) = abs(y%soil_water (nzg)) + abs(dy%soil_water (nzg) * htry)
   670)       yscal%soil_energy(nzg) = abs(y%soil_energy(nzg)) + abs(dy%soil_energy(nzg) * htry)   &
   671)                              + dslzi8(nzg) * ( abs(y%sfcwater_energy(1))                   &
   672)                                              + abs(dy%sfcwater_energy(1) * htry) )
   673)       !------------------------------------------------------------------------------------!
   674) 
   675)    case (2)
   676)       !----- Computationally stable layer. ------------------------------------------------!
   677)       meanscale_sfcw_mass   = 0.d0
   678)       meanscale_sfcw_energy = 0.d0
   679)       meanscale_sfcw_depth  = 0.d0
   680)       do k=1,y%nlev_sfcwater
   681)          yscal%sfcwater_mass  (k) = abs(y%sfcwater_mass   (k)       )                      &
   682)                                   + abs(dy%sfcwater_mass  (k) * htry)
   683)          yscal%sfcwater_energy(k) = abs(y%sfcwater_energy (k)       )                      &
   684)                                   + abs(dy%sfcwater_energy(k) * htry)
   685)          yscal%sfcwater_depth (k) = abs(y%sfcwater_depth  (k)       )                      &
   686)                                   + abs(dy%sfcwater_depth (k) * htry)
   687)       end do
   688)       do k=y%nlev_sfcwater+1,nzs
   689)          yscal%sfcwater_mass  (k) = huge_offset
   690)          yscal%sfcwater_energy(k) = huge_offset
   691)          yscal%sfcwater_depth (k) = huge_offset
   692)       end do
   693) 
   694) 
   695)       !------ Soil scale won't be affected by the temporary surface water. ----------------!
   696)       yscal%soil_water (nzg) = abs(y%soil_water (nzg)) + abs(dy%soil_water (nzg) * htry)
   697)       yscal%soil_energy(nzg) = abs(y%soil_energy(nzg)) + abs(dy%soil_energy(nzg) * htry)
   698)       !------------------------------------------------------------------------------------!
   699)    end select
   700)    !---------------------------------------------------------------------------------------!
   701) 
   702) 
   703) 
   704)    !----- Scale for the virtual water pools -----------------------------------------------!
   705)    if (abs(y%virtual_water) > 1.d-2*rk4water_stab_thresh) then
   706)       yscal%virtual_water   = abs(y%virtual_water)  + abs(dy%virtual_water*htry)
   707)       yscal%virtual_energy  = abs(y%virtual_energy) + abs(dy%virtual_energy*htry)
   708)    elseif (abs(y%virtual_water) > rk4tiny_sfcw_mass) then
   709)       yscal%virtual_water   = 1.d-2*rk4water_stab_thresh
   710)       yscal%virtual_energy  = (yscal%virtual_water / abs(y%virtual_water))                 &
   711)                             * (abs(y%virtual_energy) + abs(dy%virtual_energy*htry))
   712)    else
   713)       yscal%virtual_water   = huge_offset
   714)       yscal%virtual_energy  = huge_offset
   715)    end if
   716)    !---------------------------------------------------------------------------------------!
   717) 
   718) 
   719) 
   720)    !---------------------------------------------------------------------------------------!
   721)    !    Scale for leaf, wood, and vegetation water and energy.  In case the plants have    !
   722)    ! few or no leaves, or the plant is buried in snow, we assign huge values for typical   !
   723)    ! scale, thus preventing unecessary small steps.                                        !
   724)    !    Also, if the cohort has almost no water, make the scale less strict.               !
   725)    !---------------------------------------------------------------------------------------!
   726)    select case (ibranch_thermo)
   727)    case (1)
   728)       !----- Combined leaf+wood solution. -------------------------------------------------!
   729)       do ico=1,cpatch%ncohorts
   730)          !----- Copy the logical tests. ---------------------------------------------------!
   731)          yscal%leaf_resolvable(ico) = y%leaf_resolvable(ico)
   732)          yscal%wood_resolvable(ico) = y%wood_resolvable(ico)
   733)          yscal%veg_resolvable(ico)  = y%veg_resolvable(ico)
   734)          !---------------------------------------------------------------------------------!
   735) 
   736) 
   737) 
   738)          !----- Find the scale only if we must solve veg. ---------------------------------!
   739)          if (y%veg_resolvable(ico)) then
   740)             yscal%veg_energy(ico) = abs( y%veg_energy(ico))                                &
   741)                                        + abs(dy%veg_energy(ico) * htry)
   742)             yscal%veg_water(ico)  = max( abs(y%veg_water(ico))                             &
   743)                                        + abs(dy%veg_water(ico)  * htry)                    &
   744)                                        , rk4leaf_drywhc * y%tai(ico))  
   745)          else
   746)             yscal%veg_water(ico)  = huge_offset
   747)             yscal%veg_energy(ico) = huge_offset
   748)          end if
   749)          !---------------------------------------------------------------------------------!
   750) 
   751) 
   752) 
   753)          !----- No need to scale wood and leaf correctly, let's make it always acceptable. !
   754)          yscal%leaf_water(ico)  = huge_offset
   755)          yscal%leaf_energy(ico) = huge_offset
   756)          yscal%leaf_temp(ico)   = huge_offset
   757)          yscal%wood_water(ico)  = huge_offset
   758)          yscal%wood_energy(ico) = huge_offset
   759)          yscal%wood_temp(ico)   = huge_offset
   760)          !---------------------------------------------------------------------------------!
   761)       end do
   762)       !------------------------------------------------------------------------------------!
   763) 
   764)    case (0,2)
   765)       !------------------------------------------------------------------------------------!
   766)       !      Either we are solving leaves only, or leaves and branches are treated as      !
   767)       ! independent pools.                                                                 !
   768)       !------------------------------------------------------------------------------------!
   769)       do ico = 1,cpatch%ncohorts
   770)          !----- Copy the logical tests. ---------------------------------------------------!
   771)          yscal%leaf_resolvable(ico) = y%leaf_resolvable(ico)
   772)          yscal%wood_resolvable(ico) = y%wood_resolvable(ico)
   773)          yscal%veg_resolvable (ico) = y%veg_resolvable(ico)
   774)          !---------------------------------------------------------------------------------!
   775) 
   776) 
   777)          !----- Find the scale only if we must solve leaves. ------------------------------!
   778)          if (y%leaf_resolvable(ico)) then
   779)             yscal%leaf_energy(ico) = abs( y%leaf_energy(ico))                              &
   780)                                         + abs(dy%leaf_energy(ico) * htry)
   781)             yscal%leaf_temp(ico)   = abs( y%leaf_temp(ico))
   782)             yscal%leaf_water(ico)  = max( abs(y%leaf_water(ico))                           &
   783)                                         + abs(dy%leaf_water(ico)  * htry)                  &
   784)                                         , rk4leaf_drywhc * y%lai(ico))
   785)          else
   786)             yscal%leaf_water(ico)  = huge_offset
   787)             yscal%leaf_energy(ico) = huge_offset
   788)             yscal%leaf_temp(ico)   = huge_offset
   789)          end if
   790)          !---------------------------------------------------------------------------------!
   791) 
   792) 
   793)          !----- Find the scale only if we must solve wood. --------------------------------!
   794)          if (y%wood_resolvable(ico)) then
   795)             yscal%wood_energy(ico) = abs( y%wood_energy(ico))                              &
   796)                                         + abs(dy%wood_energy(ico) * htry)
   797)             yscal%wood_temp(ico)   = abs( y%wood_temp(ico))
   798)             yscal%wood_water(ico)  = max( abs(y%wood_water(ico))                           &
   799)                                         + abs(dy%wood_water(ico)  * htry)                  &
   800)                                         , rk4leaf_drywhc * y%wai(ico))
   801)          else
   802)             yscal%wood_water(ico)  = huge_offset
   803)             yscal%wood_energy(ico) = huge_offset
   804)             yscal%wood_temp(ico)   = huge_offset
   805)          end if
   806)          !---------------------------------------------------------------------------------!
   807) 
   808) 
   809) 
   810)          !----- No need to scale veg correctly, let's make it always acceptable. ----------!
   811)          yscal%veg_water(ico)   = huge_offset
   812)          yscal%veg_energy(ico)  = huge_offset
   813)          !---------------------------------------------------------------------------------!
   814)       end do
   815)    end select
   816)    !---------------------------------------------------------------------------------------!
   817) 
   818) 
   819) 
   820)    !---------------------------------------------------------------------------------------!
   821)    !     Here we just need to make sure the user is checking mass, otherwise  these vari-  !
   822)    ! ables will not be computed at all.  If this turns out to be essential, we will make   !
   823)    ! this permanent and not dependent on checkbudget.  The only one that is not checked is !
   824)    ! the runoff, because it is computed after a step was accepted.                         !
   825)    !---------------------------------------------------------------------------------------!
   826)    if (checkbudget) then
   827)       !------------------------------------------------------------------------------------!
   828)       !    If this is the very first time step, or if we are misfortuned, we may have a    !
   829)       ! situation in which the derivative is numerically zero, and making the check will   !
   830)       ! become impossible because the scale would be ridiculously tiny, so we skip the     !
   831)       ! check this time and hope that everything will be alright next step.                !
   832)       !------------------------------------------------------------------------------------!
   833)       if (abs(y%co2budget_loss2atm)  < tiny_offset .and.                                   &
   834)           abs(dy%co2budget_loss2atm) < tiny_offset) then
   835)          yscal%co2budget_loss2atm = 1.d-1
   836)       else 
   837)          yscal%co2budget_loss2atm = abs(y%co2budget_loss2atm)                              &
   838)                                   + abs(dy%co2budget_loss2atm*htry)
   839)          yscal%co2budget_loss2atm = max(yscal%co2budget_loss2atm,1.d-1)
   840)       end if
   841) 
   842)       if (abs(y%ebudget_netrad)  < tiny_offset .and.                                       &
   843)           abs(dy%ebudget_netrad) < tiny_offset) then
   844)          yscal%ebudget_netrad  = 1.d0
   845)       else 
   846)          yscal%ebudget_netrad  = abs(y%ebudget_netrad)                                     &
   847)                                + abs(dy%ebudget_netrad*htry)
   848)          yscal%ebudget_netrad = max(yscal%ebudget_netrad,1.d0)
   849)       end if
   850) 
   851)       if (abs(y%ebudget_loss2atm)  < tiny_offset .and.                                     &
   852)           abs(dy%ebudget_loss2atm) < tiny_offset) then
   853)          yscal%ebudget_loss2atm = 1.d0
   854)       else 
   855)          yscal%ebudget_loss2atm = abs(y%ebudget_loss2atm)                                  &
   856)                                 + abs(dy%ebudget_loss2atm*htry)
   857)          yscal%ebudget_loss2atm = max(yscal%ebudget_loss2atm,1.d0)
   858)       end if
   859) 
   860)       if (abs(y%wbudget_loss2atm)  < tiny_offset .and.                                     &
   861)           abs(dy%wbudget_loss2atm) < tiny_offset) then
   862)          yscal%wbudget_loss2atm      = 1.d-6
   863)       else 
   864)          yscal%wbudget_loss2atm = abs(y%wbudget_loss2atm)                                  &
   865)                                 + abs(dy%wbudget_loss2atm*htry)
   866)          yscal%wbudget_loss2atm = max(yscal%wbudget_loss2atm,1.d-6)
   867)       end if
   868) 
   869)       if (abs(y%ebudget_storage)  < tiny_offset .and.                                      &
   870)           abs(dy%ebudget_storage) < tiny_offset) then
   871)          yscal%ebudget_storage = huge_offset
   872)       else 
   873)          yscal%ebudget_storage = abs(y%ebudget_storage)                                    &
   874)                                + abs(dy%ebudget_storage*htry)
   875)       end if
   876) 
   877)       if (abs(y%co2budget_storage)  < tiny_offset .and.                                    &
   878)           abs(dy%co2budget_storage) < tiny_offset) then
   879)          yscal%co2budget_storage = huge_offset
   880)       else 
   881)          yscal%co2budget_storage = abs(y%co2budget_storage)                                &
   882)                                  + abs(dy%co2budget_storage*htry)
   883)       end if
   884) 
   885)       if (abs(y%wbudget_storage)  < tiny_offset .and.                                      &
   886)           abs(dy%wbudget_storage) < tiny_offset) then
   887)          yscal%wbudget_storage      = huge_offset
   888)       else 
   889)          yscal%wbudget_storage = abs(y%wbudget_storage)                                    &
   890)                                + abs(dy%wbudget_storage*htry)
   891)       end if
   892) 
   893)       !------------------------------------------------------------------------------------!
   894)       !     Drainage terms will be checked only if the boundary condition is free drain-   !
   895)       ! age.                                                                               !
   896)       !------------------------------------------------------------------------------------!
   897)       if (isoilbc == 0 .or. (abs(y%ebudget_loss2drainage)  < tiny_offset .and.             &
   898)                              abs(dy%ebudget_loss2drainage) < tiny_offset)      ) then
   899)          yscal%ebudget_loss2drainage = huge_offset
   900)       else 
   901)          yscal%ebudget_loss2drainage = abs(y%ebudget_loss2drainage)                        &
   902)                                      + abs(dy%ebudget_loss2drainage*htry)
   903)       end if
   904)       if (isoilbc == 0 .or. (abs(y%wbudget_loss2drainage)  < tiny_offset .and.             &
   905)                              abs(dy%wbudget_loss2drainage) < tiny_offset)      ) then
   906)          yscal%wbudget_loss2drainage = huge_offset
   907)       else 
   908)          yscal%wbudget_loss2drainage = abs(y%wbudget_loss2drainage)                        &
   909)                                      + abs(dy%wbudget_loss2drainage*htry)
   910)       end if
   911) 
   912)    else 
   913)       yscal%co2budget_storage       = huge_offset
   914)       yscal%co2budget_loss2atm      = huge_offset
   915)       yscal%ebudget_netrad          = huge_offset
   916)       yscal%ebudget_loss2atm        = huge_offset
   917)       yscal%wbudget_loss2atm        = huge_offset
   918)       yscal%ebudget_storage         = huge_offset
   919)       yscal%wbudget_storage         = huge_offset
   920)       yscal%ebudget_loss2drainage   = huge_offset
   921)       yscal%wbudget_loss2drainage   = huge_offset
   922)    end if
   923) 
   924)    return
   925) end subroutine get_yscal

   935) !==========================================================================================!
   936) !    This subroutine loops through the integrating variables, seeking for the largest      !
   937) ! error.                                                                                   !
   938) !------------------------------------------------------------------------------------------!
   939) subroutine get_errmax(errmax,yerr,yscal,cpatch,y,ytemp)
   940) 
   941)    use rk4_coms              , only : rk4patchtype       & ! structure
   942)                                     , ibranch_thermo     & ! intent(in)
   943)                                     , rk4eps             & ! intent(in)
   944)                                     , rk4site            & ! intent(in)
   945)                                     , checkbudget        & ! intent(in)
   946)                                     , integ_err          & ! intent(inout)
   947)                                     , record_err         & ! intent(in)
   948)                                     , osow               & ! intent(in)
   949)                                     , osoe               & ! intent(in)
   950)                                     , oswe               & ! intent(in)
   951)                                     , oswm               ! ! intent(in)
   952)    use ed_state_vars         , only : patchtype          ! ! structure
   953)    use grid_coms             , only : nzg                ! ! intent(in)
   954)    implicit none
   955)    !----- Arguments -----------------------------------------------------------------------!
   956)    type(rk4patchtype) , target      :: yerr             ! Error structure
   957)    type(rk4patchtype) , target      :: yscal            ! Scale structure
   958)    type(rk4patchtype) , target      :: y                ! Structure with previous value
   959)    type(rk4patchtype) , target      :: ytemp            ! Structure with attempted values
   960)    type(patchtype)    , target      :: cpatch           ! Current patch
   961)    real(kind=8)       , intent(out) :: errmax           ! Maximum error
   962)    !----- Local variables -----------------------------------------------------------------!
   963)    integer                          :: ico              ! Current cohort ID
   964)    real(kind=8)                     :: errh2o           ! Scratch error variable
   965)    real(kind=8)                     :: errene           ! Scratch error variable
   966)    real(kind=8)                     :: err              ! Scratch error variable
   967)    real(kind=8)                     :: errh2oMAX        ! Scratch error variable
   968)    real(kind=8)                     :: erreneMAX        ! Scratch error variable
   969)    integer                          :: k                ! Counter
   970)    !---------------------------------------------------------------------------------------!
   971) 
   972)    !----- Initialise the error with an optimistic number... -------------------------------!
   973)    errmax = 0.d0
   974) 
   975) 
   976) 
   977)    !---------------------------------------------------------------------------------------!
   978)    !    We now check each variable error, and keep track of the worst guess, which will    !
   979)    ! be our worst guess in the end.                                                        !
   980)    !---------------------------------------------------------------------------------------!
   981) 
   982) 
   983)    !---------------------------------------------------------------------------------------!
   984)    !     Get the canopy air space errors.  Only the ice-vapour equivalent potential        !
   985)    ! temperature, water vapour mixing ratio and carbon dioxide mixing ratio are accounted. !
   986)    ! Temperature and density will be also checked for sanity.                              !
   987)    !---------------------------------------------------------------------------------------!
   988)    err    = abs(yerr%can_enthalpy/yscal%can_enthalpy)
   989)    errmax = max(errmax,err)
   990)    if(record_err .and. err > rk4eps) integ_err(1,1) = integ_err(1,1) + 1_8 
   991) 
   992)    err    = abs(yerr%can_shv/yscal%can_shv)
   993)    errmax = max(errmax,err)
   994)    if(record_err .and. err > rk4eps) integ_err(2,1) = integ_err(2,1) + 1_8 
   995) 
   996)    err    = abs(yerr%can_co2/yscal%can_co2)
   997)    errmax = max(errmax,err)
   998)    if(record_err .and. err > rk4eps) integ_err(6,1) = integ_err(6,1) + 1_8 
   999) 
  1000)    !---------------------------------------------------------------------------------------!
  1001)    !      Pressure is only a semi-prognostic variable, which means that we don't really    !
  1002)    ! have a good error estimate, but we try to get a sense of the error.                   !
  1003)    !---------------------------------------------------------------------------------------!
  1004)    err    = abs(yerr%can_prss/yscal%can_prss)
  1005)    errmax = max(errmax,err)
  1006)    if(record_err .and. err > rk4eps) integ_err(5,1) = integ_err(5,1) + 1_8 
  1007)    !---------------------------------------------------------------------------------------!
  1008) 
  1009) 
  1010) 
  1011)    !---------------------------------------------------------------------------------------!
  1012)    !     Get the worst error only amongst the cohorts in which leaf or wood properties     !
  1013)    ! were computed.                                                                        !
  1014)    !---------------------------------------------------------------------------------------!
  1015)    select case (ibranch_thermo)
  1016)    case (1)
  1017)       !------------------------------------------------------------------------------------!
  1018)       !     The combined leaf+branch pool is being solved. Check the veg variables only,   !
  1019)       ! but add the error to both leaf and wood.                                           !
  1020)       !------------------------------------------------------------------------------------!
  1021)       errh2oMAX  = 0.d0
  1022)       erreneMAX  = 0.d0
  1023)       do ico = 1,cpatch%ncohorts
  1024)          if (yscal%veg_resolvable(ico)) then
  1025)             errh2o     = abs(yerr%veg_water (ico) / yscal%veg_water (ico))
  1026)             errene     = abs(yerr%veg_energy(ico) / yscal%veg_energy(ico))
  1027)             errmax     = max(errmax,errh2o,errene)
  1028)             errh2oMAX  = max(errh2oMAX ,errh2o )
  1029)             erreneMAX  = max(erreneMAX ,errene )
  1030)          end if
  1031)       end do
  1032)       if(cpatch%ncohorts > 0 .and. record_err) then
  1033)          if (errh2oMAX  > rk4eps) then
  1034)             integ_err( 7,1) = integ_err( 7,1) + 1_8
  1035)             integ_err( 9,1) = integ_err( 9,1) + 1_8
  1036)          end if
  1037)          if (erreneMAX  > rk4eps) then
  1038)             integ_err( 8,1) = integ_err( 8,1) + 1_8
  1039)             integ_err(10,1) = integ_err(10,1) + 1_8
  1040)          end if
  1041)       end if
  1042)       !------------------------------------------------------------------------------------!
  1043) 
  1044)    case default
  1045)       !------------------------------------------------------------------------------------!
  1046)       !     Either we are solving leaves only, or both leaf and branch pools are being     !
  1047)       ! solved independently. Check both leaf and wood variables.                          !
  1048)       !------------------------------------------------------------------------------------!
  1049)       !----- Leaves. ----------------------------------------------------------------------!
  1050)       errh2oMAX  = 0.d0
  1051)       erreneMAX  = 0.d0
  1052)       do ico = 1,cpatch%ncohorts
  1053)          if (yscal%leaf_resolvable(ico)) then
  1054)             errh2o     = abs(yerr%leaf_water (ico) / yscal%leaf_water (ico))
  1055)             errene     = abs(yerr%leaf_energy(ico) / yscal%leaf_energy(ico))
  1056)             errmax     = max(errmax,errh2o,errene)
  1057)             errh2oMAX  = max(errh2oMAX ,errh2o )
  1058)             erreneMAX  = max(erreneMAX ,errene )
  1059)          end if
  1060)       end do
  1061)       if(cpatch%ncohorts > 0 .and. record_err) then
  1062)          if (errh2oMAX  > rk4eps) integ_err(7,1) = integ_err(7,1) + 1_8
  1063)          if (erreneMAX  > rk4eps) integ_err(8,1) = integ_err(8,1) + 1_8
  1064)       end if
  1065)       !----- Wood. ------------------------------------------------------------------------!
  1066)       errh2oMAX  = 0.d0
  1067)       erreneMAX  = 0.d0
  1068)       do ico = 1,cpatch%ncohorts
  1069)          if (yscal%wood_resolvable(ico)) then
  1070)             errh2o     = abs(yerr%wood_water (ico) / yscal%wood_water (ico))
  1071)             errene     = abs(yerr%wood_energy(ico) / yscal%wood_energy(ico))
  1072)             errmax     = max(errmax,errh2o,errene)
  1073)             errh2oMAX  = max(errh2oMAX ,errh2o )
  1074)             erreneMAX  = max(erreneMAX ,errene )
  1075)          end if
  1076)       end do
  1077)       if(cpatch%ncohorts > 0 .and. record_err) then
  1078)          if (errh2oMAX  > rk4eps) integ_err( 9,1) = integ_err( 9,1) + 1_8
  1079)          if (erreneMAX  > rk4eps) integ_err(10,1) = integ_err(10,1) + 1_8
  1080)       end if
  1081)       !------------------------------------------------------------------------------------!
  1082)    end select
  1083)    !---------------------------------------------------------------------------------------!
  1084) 
  1085) 
  1086)    !---------------------------------------------------------------------------------------!
  1087)    !     Virtual pool.                                                                     !
  1088)    !---------------------------------------------------------------------------------------!
  1089)    err    = abs(yerr%virtual_energy/yscal%virtual_energy)
  1090)    errmax = max(errmax,err)
  1091)    if(record_err .and. err > rk4eps) integ_err(11,1) = integ_err(11,1) + 1_8
  1092) 
  1093)    err    = abs(yerr%virtual_water/yscal%virtual_water)
  1094)    errmax = max(errmax,err)
  1095)    if(record_err .and. err > rk4eps) integ_err(12,1) = integ_err(12,1) + 1_8      
  1096)    !---------------------------------------------------------------------------------------!
  1097) 
  1098) 
  1099) 
  1100)    !---------------------------------------------------------------------------------------!
  1101)    !     Here we just need to make sure the user is checking mass, otherwise these         !
  1102)    ! variables will not be computed at all.  The only one that is not checked is the       !
  1103)    ! runoff, because it is computed only after a step was accepted.                        !
  1104)    !---------------------------------------------------------------------------------------!
  1105)    if (checkbudget) then
  1106)       err    = abs(yerr%co2budget_storage/yscal%co2budget_storage)
  1107)       errmax = max(errmax,err)
  1108)       if(record_err .and. err > rk4eps) integ_err(13,1) = integ_err(13,1) + 1_8
  1109) 
  1110)       err    = abs(yerr%co2budget_loss2atm/yscal%co2budget_loss2atm)
  1111)       errmax = max(errmax,err)
  1112)       if(record_err .and. err > rk4eps) integ_err(14,1) = integ_err(14,1) + 1_8
  1113) 
  1114)       err    = abs(yerr%ebudget_netrad/yscal%ebudget_netrad)
  1115)       errmax = max(errmax,err)
  1116)       if(record_err .and. err > rk4eps) integ_err(15,1) = integ_err(15,1) + 1_8
  1117) 
  1118)       err    = abs(yerr%ebudget_loss2atm/yscal%ebudget_loss2atm)
  1119)       errmax = max(errmax,err)
  1120)       if(record_err .and. err > rk4eps) integ_err(16,1) = integ_err(17,1) + 1_8
  1121) 
  1122)       err    = abs(yerr%wbudget_loss2atm/yscal%wbudget_loss2atm)
  1123)       errmax = max(errmax,err)
  1124)       if(record_err .and. err > rk4eps) integ_err(17,1) = integ_err(18,1) + 1_8
  1125) 
  1126)       err    = abs(yerr%ebudget_loss2drainage/yscal%ebudget_loss2drainage)
  1127)       errmax = max(errmax,err)
  1128)       if(record_err .and. err > rk4eps) integ_err(18,1) = integ_err(19,1) + 1_8
  1129) 
  1130)       err    = abs(yerr%wbudget_loss2drainage/yscal%wbudget_loss2drainage)
  1131)       errmax = max(errmax,err)
  1132)       if(record_err .and. err > rk4eps) integ_err(19,1) = integ_err(20,1) + 1_8
  1133) 
  1134)       err    = abs(yerr%ebudget_storage/yscal%ebudget_storage)
  1135)       errmax = max(errmax,err)
  1136)       if(record_err .and. err > rk4eps) integ_err(20,1) = integ_err(21,1) + 1_8
  1137) 
  1138)       err    = abs(yerr%wbudget_storage/yscal%wbudget_storage)
  1139)       errmax = max(errmax,err)
  1140)       if(record_err .and. err > rk4eps) integ_err(21,1) = integ_err(22,1) + 1_8
  1141)    end if
  1142)    !---------------------------------------------------------------------------------------!
  1143) 
  1144) 
  1145) 
  1146)    !---------------------------------------------------------------------------------------!
  1147)    !     Soil properties.  Notice that the index depends on the number of layers.          !
  1148)    !---------------------------------------------------------------------------------------!
  1149)    do k=rk4site%lsl,nzg
  1150)       err    = abs(yerr%soil_water(k)/yscal%soil_water(k))
  1151)       errmax = max(errmax,err)
  1152)       if(record_err .and. err > rk4eps) integ_err(osow+k,1) = integ_err(osow+k,1) + 1_8 
  1153)    end do
  1154) 
  1155)    do k=rk4site%lsl,nzg
  1156)       err    = abs(yerr%soil_energy(k)/yscal%soil_energy(k))
  1157)       errmax = max(errmax,err)
  1158)       if(record_err .and. err > rk4eps) integ_err(osoe+k,1) = integ_err(osoe+k,1) + 1_8
  1159)    end do
  1160)    !---------------------------------------------------------------------------------------!
  1161) 
  1162)    !---------------------------------------------------------------------------------------!
  1163)    !     Surface water/snow properties.  Notice that the index also depends on the number  !
  1164)    ! of layers.                                                                            !
  1165)    !---------------------------------------------------------------------------------------!
  1166)    do k=1,ytemp%nlev_sfcwater
  1167)       err = abs(yerr%sfcwater_energy(k) / yscal%sfcwater_energy(k))
  1168)       errmax = max(errmax,err)
  1169)       if(record_err .and. err > rk4eps) integ_err(oswe+k,1) = integ_err(oswe+k,1) + 1_8
  1170)    end do
  1171) 
  1172)    do k=1,ytemp%nlev_sfcwater
  1173)       err    = abs(yerr%sfcwater_mass(k) / yscal%sfcwater_mass(k))
  1174)       errmax = max(errmax,err)
  1175)       if(record_err .and. err > rk4eps) integ_err(oswm+k,1) = integ_err(oswm+k,1) + 1_8
  1176)    end do
  1177)    !---------------------------------------------------------------------------------------!
  1178) 
  1179)    return
  1180) end subroutine get_errmax

  1692)    subroutine rkqs(x,htry,hdid,hnext,csite,ipa,isi)
  1693)       use rk4_copy_patch
  1694)       use rk4_misc
  1695)       use rk4_coms      , only : rk4patchtype        & ! structure
  1696)                                , integration_buff    & ! intent(inout)
  1697)                                , rk4site             & ! intent(in)
  1698)                                , hmin                & ! intent(in)
  1699)                                , rk4eps              & ! intent(in)
  1700)                                , rk4epsi             & ! intent(in)
  1701)                                , safety              & ! intent(in)
  1702)                                , pgrow               & ! intent(in)
  1703)                                , pshrnk              & ! intent(in)
  1704)                                , errcon              & ! intent(in)
  1705)                                , print_detailed      & ! intent(in)
  1706)                                , norm_rk4_fluxes     & ! intent(in)
  1707)                                , reset_rk4_fluxes    ! ! intent(in)
  1708)       use ed_state_vars , only : sitetype            & ! structure
  1709)                                , patchtype           ! ! structure
  1710)       use grid_coms     , only : nzg                 & ! intent(in)
  1711)                                , nzs                 ! ! intent(in)
  1712)       !$ use omp_lib
  1713) 
  1714)       implicit none
  1715)       !----- Arguments --------------------------------------------------------------------!
  1716)       integer                  , intent(in)    :: ipa
  1717)       integer                  , intent(in)    :: isi
  1718)       type(sitetype)           , target        :: csite
  1719)       real(kind=8)             , intent(in)    :: htry
  1720)       real(kind=8)             , intent(inout) :: x
  1721)       real(kind=8)             , intent(out)   :: hdid,hnext
  1722)       !----- Local variables --------------------------------------------------------------!
  1723)       real(kind=8)                             :: h,errmax,xnew,newh,oldh
  1724)       logical                                  :: reject_step,reject_result
  1725)       logical                                  :: minstep,stuck,test_reject
  1726)       integer                                  :: k
  1727)       integer                                  :: ibuff
  1728)       !------------------------------------------------------------------------------------!
  1729) 
  1730)       ibuff=1
  1731)       !$ ibuff = OMP_get_thread_num()+1
  1732) 
  1733)       h           =  htry
  1734)       reject_step =  .false.
  1735)       hstep:   do
  1736) 
  1737)          !---------------------------------------------------------------------------------!
  1738)          ! 1. Try a step of varying size.                                                  !
  1739)          !---------------------------------------------------------------------------------!
  1740)          call rkck(integration_buff(ibuff)%y,integration_buff(ibuff)%dydx                  &
  1741)                   ,integration_buff(ibuff)%ytemp,integration_buff(ibuff)%yerr              &
  1742)                   ,integration_buff(ibuff)%ak2,integration_buff(ibuff)%ak3                 &
  1743)                   ,integration_buff(ibuff)%ak4,integration_buff(ibuff)%ak5                 &
  1744)                   ,integration_buff(ibuff)%ak6,integration_buff(ibuff)%ak7,h,csite,ipa     &
  1745)                   ,reject_step,reject_result)
  1746) 
  1747)          !---------------------------------------------------------------------------------!
  1748)          ! 2. Check to see how accurate the step was.  Errors were calculated by integrat- !
  1749)          !    ing the derivative of that last step.                                        !
  1750)          !---------------------------------------------------------------------------------!
  1751)          if (reject_step .or. reject_result) then
  1752)             !------------------------------------------------------------------------------!
  1753)             !    If step was already rejected, that means the step had finished premature- !
  1754)             ! ly, so we assign a standard large error (10.0).                              !
  1755)             !------------------------------------------------------------------------------!
  1756)             errmax = 1.d1
  1757)          else
  1758)             call get_errmax(errmax,integration_buff(ibuff)%yerr                            &
  1759)                            ,integration_buff(ibuff)%yscal,csite%patch(ipa)                 &
  1760)                            ,integration_buff(ibuff)%y,integration_buff(ibuff)%ytemp)
  1761)             errmax = errmax * rk4epsi
  1762)          end if
  1763) 
  1764)          !---------------------------------------------------------------------------------!
  1765)          ! 3. If that error was large, then calculate a new step size to try.  There are   !
  1766)          !    two types of new tries.  If step failed to be minimally reasonable (reject-  !
  1767)          !    ed) we have assigned a standard large error (10.0).  Otherwise a new step is !
  1768)          !    calculated based on the size of that error.  Hopefully, those new steps      !
  1769)          !    should be less than the previous h.  If the error was small, i.e. less then  !
  1770)          !    rk4eps, then we are done with this step, and we can move forward             !
  1771)          !    time: x = x + h                                                              !
  1772)          !---------------------------------------------------------------------------------!
  1773)          if (errmax > 1.d0) then
  1774)             !----- Defining new step and checking if it can be. ---------------------------!
  1775)             oldh    = h
  1776)             newh    = safety * h * errmax**pshrnk
  1777)             minstep = (newh == h) .or. newh < hmin
  1778) 
  1779)             !----- Defining next time, and checking if it really added something. ---------!
  1780)             h       = max(1.d-1*h, newh)
  1781)             xnew    = x + h
  1782)             stuck   = xnew == x
  1783) 
  1784)             !------------------------------------------------------------------------------!
  1785)             ! 3a. Here is the moment of truth... If we reached a tiny step and yet the     !
  1786)             !     model didn't converge, then we print various values to inform the user   !
  1787)             !     and abort the run.  Please, don't hate the messenger.                    !
  1788)             !------------------------------------------------------------------------------!
  1789)             if (minstep .or. stuck) then
  1790) 
  1791)                write (unit=*,fmt='(80a)')         ('=',k=1,80)
  1792)                write (unit=*,fmt='(a)')           '   STEPSIZE UNDERFLOW IN RKQS'
  1793)                write (unit=*,fmt='(80a)')         ('-',k=1,80)
  1794)                write (unit=*,fmt='(a,1x,f9.4)')   ' + LONGITUDE:     ',rk4site%lon
  1795)                write (unit=*,fmt='(a,1x,f9.4)')   ' + LATITUDE:      ',rk4site%lat
  1796)                write (unit=*,fmt='(a)')           ' + PATCH INFO:    '
  1797)                write (unit=*,fmt='(a,1x,i6)')     '   - NUMBER:      ',ipa
  1798)                write (unit=*,fmt='(a,1x,es12.4)') '   - AGE:         ',csite%age(ipa)
  1799)                write (unit=*,fmt='(a,1x,i6)')     '   - DIST_TYPE:   ',csite%dist_type(ipa)
  1800)                write (unit=*,fmt='(a,1x,l1)')     ' + REJECT_STEP:   ',reject_step
  1801)                write (unit=*,fmt='(a,1x,l1)')     ' + REJECT_RESULT: ',reject_result
  1802)                write (unit=*,fmt='(a,1x,l1)')     ' + MINSTEP:       ',minstep
  1803)                write (unit=*,fmt='(a,1x,l1)')     ' + STUCK:         ',stuck

  1805)                write (unit=*,fmt='(a,1x,es12.4)') ' + X:             ',x

  1808)                write (unit=*,fmt='(a,1x,es12.4)') ' + NEWH:          ',newh
  1809)                write (unit=*,fmt='(a,1x,es12.4)') ' + SAFETY:        ',safety
  1810)                write (unit=*,fmt='(80a)') ('-',k=1,80)
  1811)                if (reject_step .or. reject_result) then

  1813)                   write (unit=*,fmt='(80a)') ('=',k=1,80)

  1817)                end if
  1818) 
  1819)                if (reject_result) then

  1822)                                        ,ipa,integration_buff(ibuff)%dydx,h,.true.)
  1823)                   call print_sanity_check(integration_buff(ibuff)%y,csite,ipa)

  1826)                                        ,integration_buff(ibuff)%dydx,h,.true.)

  1830)                                    ,integration_buff(ibuff)%yscal,csite%patch(ipa)         &

  1835)                   write (unit=*,fmt='(a,1x,es12.4)') ' - Epsilon:',rk4eps
  1836)                   write (unit=*,fmt='(80a)') ('=',k=1,80)

  1840)          
  1841)          else
  1842) 
  1843)             !------------------------------------------------------------------------------!
  1844)             ! 3b.  Great, it worked, so now we can advance to the next step.  We just need !
  1845)             !      to do some minor adjustments before...                                  !
  1846)             !------------------------------------------------------------------------------!
  1847)             !----- i.   Final update of leaf properties to avoid negative water. ----------!
  1848)             call adjust_veg_properties(integration_buff(ibuff)%ytemp,h,csite,ipa)
  1849)             !----- ii.  Final update of top soil properties to avoid off-bounds moisture. -!
  1850)             call adjust_topsoil_properties(integration_buff(ibuff)%ytemp,h,csite)
  1851)             !----- iii. Make temporary surface water stable and positively defined. -------!
  1852)             call adjust_sfcw_properties(nzg,nzs,integration_buff(ibuff)%ytemp, h, csite)
  1853)             !----- iv.  Update the diagnostic variables. ----------------------------------!
  1854)             call update_diagnostic_vars(integration_buff(ibuff)%ytemp, csite,ipa)
  1855)             !------------------------------------------------------------------------------!
  1856) 
  1857)             !------------------------------------------------------------------------------!
  1858)             ! 3c. Set up h for the next time.  And here we can relax h for the next step,  !
  1859)             !    and try something faster.                                                 !
  1860)             !------------------------------------------------------------------------------!
  1861)             if (errmax > errcon) then
  1862)                hnext = safety * h * errmax**pgrow
  1863)             else
  1864)                hnext = 5.d0 * h
  1865)             endif
  1866)             hnext = max(2.d0*hmin,hnext)
  1867) 
  1868)             !------ 3d. Normalise the fluxes if the user wants detailed debugging. --------!
  1869)             if (print_detailed) then
  1870)                call norm_rk4_fluxes(integration_buff(ibuff)%ytemp,h)
  1871)                call print_rk4_state(integration_buff(ibuff)%y                              &
  1872)                                    ,integration_buff(ibuff)%ytemp,csite,ipa,isi,x,h)
  1873)             end if
  1874) 
  1875)             !------------------------------------------------------------------------------!
  1876)             !    3e. Copy the temporary structure to the intermediate state.               !
  1877)             !------------------------------------------------------------------------------!
  1878)             call copy_rk4_patch(integration_buff(ibuff)%ytemp,integration_buff(ibuff)%y    &
  1879)                                ,csite%patch(ipa))
  1880) 
  1881)             !------------------------------------------------------------------------------!
  1882)             !    3f. Flush step-by-step fluxes to zero if the user wants detailed          !
  1883)             !        debugging.                                                            !
  1884)             !------------------------------------------------------------------------------!
  1885)             if (print_detailed) then
  1886)                call reset_rk4_fluxes(integration_buff(ibuff)%y)
  1887)             end if
  1888) 
  1889)             !----- 3g. Update time. -------------------------------------------------------!
  1890)             x    = x + h
  1891)             hdid = h
  1892) 
  1893)             exit hstep
  1894)          end if
  1895)       end do hstep
  1896) 
  1897)       return
  1898)    end subroutine rkqs
