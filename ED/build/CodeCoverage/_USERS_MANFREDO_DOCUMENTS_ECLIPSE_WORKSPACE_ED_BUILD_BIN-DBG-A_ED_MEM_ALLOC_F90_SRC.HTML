    30)                                      , ed_nullify_work         ! ! subroutine
    31)       use ed_node_coms        , only : mmxp                    & ! intent(in)

    48)       select case (proc_type)

    50)          !  This is the call for either a single processor run or
    51)          !    for the master process
    52)          nmxp => nnxp
    53)          nmyp => nnyp
    54)       case (2)
    55)          !  This is the call for a initial compute node process
    56)          nmxp => mmxp
    57)          nmyp => mmyp

    62)

    64)    ! Changed this part. The structure will not be allocated at the nodes at this moment,   !
    65)    ! and even at the master node we will have deallocated to reallocate with the node size !

    67)    ! grid, because it will be used to assign the polygon information.                      !
    68)    !---------------------------------------------------------------------------------------!
    69) 
    70)       write (unit=*,fmt='(a,i5,a)') ' + Work allocation, node ',mynum,';'
    71)       allocate(work_e(ngrids))

    73)          call ed_nullify_work(work_e(ng))

    78)    !   Allocate the top most hierachical memory structures for the ED2 LSM. We changed the !
    79)    ! the way the standalone deals with different regions ans sites of interest. Now it     !

    81)    ! the parallel code is implemented, to take full advantage of MPI. By doing this, we    !
    82)    ! can split the polygons among the different nodes for the regional run, and split the  !
    83)    ! patches and cohorts in different nodes in the POI grids --- the latter is yet to be   !
    84)    ! implemented.                                                                          !

    86)       write (unit=*,fmt='(a,i5,a)') ' + Polygon array allocation, node ',mynum,';'
    87) 
    88)       call allocate_edglobals(ngrids)
    89)       do ng=1,ngrids
    90)          call ed_newgrid(ng)
    91)          call allocate_edtype(edgrid_g(ng),gdpy(mynum,ng))
    92)       end do
    93) 
    94)       write (unit=*,fmt='(a,i5,a)') ' + Memory successfully allocated on none ',mynum,';'
    95)       return
