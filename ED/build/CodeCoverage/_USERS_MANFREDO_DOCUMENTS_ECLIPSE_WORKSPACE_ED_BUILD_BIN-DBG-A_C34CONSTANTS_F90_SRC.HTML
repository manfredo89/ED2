    39)    ! for the photosynthesis model.  The units inside the photosynthesis model are not the  !

    43)    type farq_consts
    44)       integer      :: photo_pathway ! The photosynthetic pathway                 [     ---]

    49)       real(kind=8) :: vm0           ! Ref. value for maximum rubisco activity    [mol/m²/s]
    50)       real(kind=8) :: vm_low_temp   ! Low temp. threshold for rapid decline      [       K]
    51)       real(kind=8) :: vm_high_temp  ! High temp. threshold for rapid decline     [       K]
    52)       real(kind=8) :: vm_hor        ! "Activation energy" for Vm                 [       K]
    53)       real(kind=8) :: vm_q10        ! Vm base for C91 physiology                 [      --]
    54)       real(kind=8) :: vm_decay_a    ! Decay factor for Vm correction as in C91   [   J/mol]

    60)       real(kind=8) :: rd_hor        ! "Activation energy" for Vm                 [       K]
