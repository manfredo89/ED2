    14)    !    This subroutine computes the total PFT-dependent mortality rate:                   !
    15)    !---------------------------------------------------------------------------------------!
    16)    subroutine mortality_rates(cpatch,ico,avg_daily_temp, patch_age)
    17)       use ed_state_vars , only : patchtype                  ! ! Structure
    18)       use pft_coms      , only : mort0                      & ! intent(in)
    19)                                , mort1                      & ! intent(in)
    20)                                , mort2                      & ! intent(in)
    21)                                , mort3                      & ! intent(in)
    22)                                , plant_min_temp             & ! intent(in)
    23)                                , frost_mort                 ! ! intent(in)
    24)       use disturb_coms  , only : treefall_disturbance_rate  & ! intent(in)
    25)                                , treefall_hite_threshold    & ! intent(in)
    26)                                , time2canopy                ! ! intent(in)
    27)       use ed_max_dims   , only : n_pft                      ! ! intent(in)
    28)       use consts_coms   , only : lnexp_min                  & ! intent(in)
    29)                                , lnexp_max                  ! ! intent(in)
    30)       implicit none
    31)       !----- Arguments --------------------------------------------------------------------!
    32)       type(patchtype), target     :: cpatch          ! Current patch
    33)       integer        , intent(in) :: ico             ! Current cohort ID
    34)       real           , intent(in) :: avg_daily_temp  ! Mean temperature yesterday
    35)       real           , intent(in) :: patch_age       ! Patch age
    36)       !----- Local variables --------------------------------------------------------------!
    37)       integer                     :: ipft            ! PFT 
    38)       real                        :: temp_dep        ! Temp. function  (frost mortality)
    39)       real                        :: expmort         ! Carbon-balance term
    40)       !------------------------------------------------------------------------------------!
    41) 
    42) 
    43)       !----- Assume happy end, all plants survive... --------------------------------------!
    44)       cpatch%mort_rate(1:4,ico) = 0.0
    45)       ipft = cpatch%pft(ico)
    46) 
    47)       !------------------------------------------------------------------------------------!
    48)       ! 1.  Ageing, PFT-dependent but otherwise constant.                                  !
    49)       !------------------------------------------------------------------------------------!
    50)       cpatch%mort_rate(1,ico) = mort3(ipft)
    51)       !------------------------------------------------------------------------------------!
    52) 
    53) 
    54)       !------------------------------------------------------------------------------------!
    55)       ! 2.  Mortality rates due to negative carbon balance.                                !
    56)       !------------------------------------------------------------------------------------!
    57)       expmort = max( lnexp_min, min( lnexp_max                                             &
    58)                                    , mort2(ipft) * ( cpatch%cbr_bar(ico) - mort0(ipft) ) ) )
    59)       cpatch%mort_rate(2,ico) = mort1(ipft) / (1. + exp(expmort))
    60)       !------------------------------------------------------------------------------------!
    61) 
    62) 
    63)       !------------------------------------------------------------------------------------!
    64)       ! 3.  Mortality due to treefall.                                                     !
    65)       !------------------------------------------------------------------------------------!
    66)       if (cpatch%hite(ico) <= treefall_hite_threshold .and. patch_age > time2canopy) then
    67)          cpatch%mort_rate(3,ico) = treefall_disturbance_rate
    68)       else
    69)          cpatch%mort_rate(3,ico) = 0.
    70)       end if
    71)       !------------------------------------------------------------------------------------!
    72) 
    73) 
    74) 
    75)       !------------------------------------------------------------------------------------!
    76)       ! 4.   Mortality due to cold, after:                                                 !
    77)       !      Albani, M.; D. Medvigy; G. C. Hurtt; P. R. Moorcroft, 2006: The contributions !
    78)       !           of land-use change, CO2 fertilization, and climate variability to the    !
    79)       !           Eastern US carbon sink.  Glob. Change Biol., 12, 2370-2390,              !
    80)       !           doi: 10.1111/j.1365-2486.2006.01254.x                                    !
    81)       !------------------------------------------------------------------------------------!
    82)       temp_dep = max( 0.0, min( 1.0, 1.0 - (avg_daily_temp - plant_min_temp(ipft)) / 5.0) )
    83)       cpatch%mort_rate(4,ico) = frost_mort(ipft) * temp_dep
    84)       !------------------------------------------------------------------------------------!
    85) 
    86) 
    87) 
    88)       !------------------------------------------------------------------------------------!
    89)       ! 5. Disturbance rate mortality.  This is not used by the cohort dynamics, instead   !
    90)       !    this is just to account for the lost density due to the patch creation.  This   !
    91)       !    mortality will be determined by the disturbance_mortality subroutine, not here. !
    92)       !------------------------------------------------------------------------------------!
    93)       !cpatch%mort_rate(5,ico) = TBD
    94)       !------------------------------------------------------------------------------------!
    95) 
    96)       return

   108)    !     This subroutine determines the mortality rates associated with the current        !
   109)    ! disturbance.                                                                          !
   110)    !---------------------------------------------------------------------------------------!
   111)    subroutine disturbance_mortality(csite,ipa,disturbance_rate,new_lu,dist_path            &
   112)                                    ,mindbh_harvest)
   113)       use ed_state_vars, only : sitetype  & ! structure
   114)                               , patchtype ! ! structure
   115)       use ed_max_dims  , only : n_pft     ! ! intent(in)
   116)       implicit none
   117)       !----- Arguments. -------------------------------------------------------------------!
   118)       type(sitetype)                   , target     :: csite
   119)       integer                          , intent(in) :: ipa
   120)       real                             , intent(in) :: disturbance_rate
   121)       integer                          , intent(in) :: new_lu
   122)       integer                          , intent(in) :: dist_path
   123)       real           , dimension(n_pft), intent(in) :: mindbh_harvest
   124)       !----- Local variables. -------------------------------------------------------------!
   125)       type(patchtype)                  , pointer    :: cpatch
   126)       integer                                       :: ico
   127)       real                                          :: f_survival
   128)       !------------------------------------------------------------------------------------!
   129) 
   130)       cpatch => csite%patch(ipa)
   131)       do ico=1,cpatch%ncohorts
   132)          f_survival = survivorship(new_lu,dist_path,mindbh_harvest,cpatch,ico)
   133)          cpatch%mort_rate(5,ico) = cpatch%mort_rate(5,ico)                                 &
   134)                                  - log( f_survival                                         &
   135)                                       + (1.0 - f_survival) * exp(- disturbance_rate) )
   136)       end do
   137)       return
   138)    end subroutine disturbance_mortality

   148)    !     This function computes the survivorship rate associated with a disturbance.       !
   149)    !  Input variables:                                                                     !
   150)    !  -- new_lu: the disturbance/land use type after disturbance:                          !
   151)    !     1. Clear cut (cropland and pasture).                                              !
   152)    !     2. Forest plantation.                                                             !
   153)    !     3. Tree fall.                                                                     !
   154)    !     4. Fire.                                                                          !
   155)    !     5. Forest regrowth.                                                               !
   156)    !     6. Logged forest.                                                                 !
   157)    !  -- dist_path: the pathway for the disturbance.  The flags depend on new_lu.  See     !
   158)    !        comments at the select case (new_lu) block for additional details.             !
   159)    !  -- mindbh_harvest: minimum DBH for harvesting (selective logging and forest          !
   160)    !        plantantions).  If the tree DBH is greater than mindbh_harvest, the tree may   !
   161)    !        be harvested, otherwise it may be damaged by logging but not harvested.        !
   162)    !  -- cpatch: current patch.                                                            !
   163)    !  -- ico: index for current cohort.                                                    !
   164)    !---------------------------------------------------------------------------------------!
   165)    real function survivorship(new_lu,dist_path,mindbh_harvest,cpatch,ico)
   166)       use ed_state_vars, only : patchtype                ! ! structure
   167)       use disturb_coms , only : treefall_hite_threshold  & ! intent(in)
   168)                               , fire_hite_threshold      ! ! intent(in)
   169)       use pft_coms     , only : treefall_s_ltht          & ! intent(in)
   170)                               , treefall_s_gtht          & ! intent(in)
   171)                               , fire_s_ltht              & ! intent(in)
   172)                               , fire_s_gtht              ! ! intent(in)
   173)       use ed_max_dims  , only : n_pft                    ! ! intent(in)
   174)       
   175)       implicit none
   176)       !----- Arguments. -------------------------------------------------------------------!
   177)       type(patchtype)                 , target     :: cpatch
   178)       real          , dimension(n_pft), intent(in) :: mindbh_harvest
   179)       integer                         , intent(in) :: ico
   180)       integer                         , intent(in) :: new_lu
   181)       integer                         , intent(in) :: dist_path
   182)       !----- Local variables. -------------------------------------------------------------!
   183)       integer                                      :: ipft
   184)       !------------------------------------------------------------------------------------!
   185) 
   186) 
   187)       !----- Alias for current PFT. -------------------------------------------------------!
   188)       ipft = cpatch%pft(ico)
   189)       !------------------------------------------------------------------------------------!
   190) 
   191)       !------------------------------------------------------------------------------------!
   192)       !      Survivorship depends on the disturbance type (new_lu), and within each        !
   193)       ! disturbance type, it may depend on the disturbance pathway (dist_path), the PFT,   !
   194)       ! and size.                                                                          !
   195)       !------------------------------------------------------------------------------------!
   196)       select case(new_lu)
   197)       case (1)
   198)          !---------------------------------------------------------------------------------!
   199)          !     Clear cut (cropland/pasture).  For now, nothing survives.                   !
   200)          !---------------------------------------------------------------------------------!
   201)          survivorship = 0.0
   202)          !---------------------------------------------------------------------------------!
   203) 
   204)       case (2)
   205)          !---------------------------------------------------------------------------------!
   206)          !     Forest plantation.  Two types of mortality may exist: treefall disturbance  !
   207)          ! rates (which will maintain a plantation set as a plantation as it is a managed  !
   208)          ! land), and harvesting.                                                          !
   209)          !---------------------------------------------------------------------------------!
   210)          select case (dist_path)
   211)          case (20)
   212)             !----- Harvesting, assumes that nothing survives. -----------------------------!
   213)             survivorship = 0.0
   214)             !------------------------------------------------------------------------------!
   215) 
   216)          case (21)
   217)             !----- Tree fall, assumes typical tree fall mortality. ------------------------!
   218)             if (cpatch%hite(ico) < treefall_hite_threshold) then
   219)                survivorship = treefall_s_ltht(ipft)
   220)             else
   221)                survivorship = treefall_s_gtht(ipft)
   222)             end if
   223)             !------------------------------------------------------------------------------!
   224)          end select
   225)          !---------------------------------------------------------------------------------!
   226) 
   227)       case (3)
   228)          !---------------------------------------------------------------------------------!
   229)          !     Tree fall.  Mortality depends on the cohort height and PFT.                 !
   230)          !---------------------------------------------------------------------------------!
   231)          if (cpatch%hite(ico) < treefall_hite_threshold) then
   232)             survivorship = treefall_s_ltht(ipft)
   233)          else
   234)             survivorship = treefall_s_gtht(ipft)
   235)          end if
   236)          !---------------------------------------------------------------------------------!
   237) 
   238)       case (4)
   239)          !---------------------------------------------------------------------------------!
   240)          !     Fire.  For now fire mortality depends on the cohort height and PFT.         !
   241)          !---------------------------------------------------------------------------------!
   242)          if (cpatch%hite(ico) < fire_hite_threshold) then
   243)             survivorship = fire_s_ltht(ipft)
   244)          else
   245)             survivorship = fire_s_gtht(ipft)
   246)          end if
   247)          !---------------------------------------------------------------------------------!
   248) 
   249)        case (5)
   250)          !---------------------------------------------------------------------------------!
   251)          !     Abandonment (secondary regrowth).  Two paths are possible: abandonment      !
   252)          ! occurs after one last harvest, or the field/plantation is left as is.           !
   253)          !---------------------------------------------------------------------------------!
   254)          select case (dist_path)
   255)          case (50)
   256)             !----- Agriculture field is left as is. ---------------------------------------!
   257)             survivorship = 1.0
   258)             !------------------------------------------------------------------------------!
   259)          case (51)
   260)             !----- Forest plantation abandoned following fire. ----------------------------!
   261)             if (cpatch%hite(ico) < fire_hite_threshold) then
   262)                survivorship = fire_s_ltht(ipft)
   263)             else
   264)                survivorship = fire_s_gtht(ipft)
   265)             end if
   266)             !------------------------------------------------------------------------------!
   267)          case (52)
   268)             !----- Harvest precedes abandonment.  Nothing survives. -----------------------!
   269)             survivorship = 0.0
   270)             !------------------------------------------------------------------------------!
   271)          end select
   272)          !---------------------------------------------------------------------------------!
   273) 
   274)        case (6)
   275)          !---------------------------------------------------------------------------------!
   276)          !     Logging.  At this point a single pathway exists: cohorts above threshold    !
   277)          ! are completely removed, and small cohorts have the same survivorship as small   !
   278)          ! cohorts at a treefall site.  Both could be re-visited in the future, and        !
   279)          ! different pathways for conventional and reduced-impact logging could be         !
   280)          ! applied.                                                                        !
   281)          !---------------------------------------------------------------------------------!
   282)          if (cpatch%dbh(ico) >= mindbh_harvest(ipft)) then
   283)             survivorship = 0.0
   284)          else
   285)             survivorship = treefall_s_ltht(ipft)
   286)          end if
   287)          !---------------------------------------------------------------------------------!
   288)       end select
   289)       !------------------------------------------------------------------------------------!
   290) 
   291)       return
   292)    end function survivorship
