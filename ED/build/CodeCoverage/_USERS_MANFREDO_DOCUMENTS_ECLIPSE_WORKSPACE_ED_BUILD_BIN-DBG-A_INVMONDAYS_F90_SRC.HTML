   150)    outday = inday - 1

   152)       !------------------------------------------------------------------------------------!

   155)       !------------------------------------------------------------------------------------!

   157)       if (outmonth == 0) then
   158)          outyear  = inyear - 1

   161)          outyear  = inyear
   162)       end if

   165)       if (isleap(outyear)) maxdays(2)=29

   167)       !------------------------------------------------------------------------------------!
   168)    else
   169)       !------------------------------------------------------------------------------------!
   170)       !    Middle of the month, copy the month and year for current time.                  !
   171)       !------------------------------------------------------------------------------------!
