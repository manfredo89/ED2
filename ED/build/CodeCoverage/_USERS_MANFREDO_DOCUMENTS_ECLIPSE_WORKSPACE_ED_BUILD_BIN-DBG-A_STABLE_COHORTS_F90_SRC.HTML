     6) !     This subroutine will go through all cohorts in a polygon, and assign them as either  !
     7) ! numerically safe or unsafe.  The idea of to use a unique test to decide which cohorts    !
     8) ! we will solve in photosynthesis, radiation, and the energy balance (RK4 or Euler).       !
     9) !------------------------------------------------------------------------------------------!
    10) subroutine flag_stable_cohorts(cgrid)
    11)    use ed_state_vars         , only : edtype          & ! structure
    12)                                     , polygontype     & ! structure
    13)                                     , sitetype        & ! structure
    14)                                     , patchtype       ! ! structure
    15)    implicit none
    16)    !----- Arguments -----------------------------------------------------------------------!
    17)    type(edtype)     , target   :: cgrid         ! Current grid
    18)    !----- Local variables. ----------------------------------------------------------------!
    19)    type(polygontype), pointer  :: cpoly         ! Current polygon
    20)    type(sitetype)   , pointer  :: csite         ! Current site
    21)    type(patchtype)  , pointer  :: cpatch        ! Current patch
    22)    integer                     :: ipy           ! Polygon index
    23)    integer                     :: isi           ! Site index
    24)    integer                     :: ipa           ! Patch index
    25)    integer                     :: ico           ! Cohort index
    26)    !---------------------------------------------------------------------------------------!
    27) 
    28)    polyloop: do ipy=1, cgrid%npolygons
    29) 
    30)       cpoly => cgrid%polygon(ipy)
    31)       siteloop: do isi=1,cpoly%nsites
    32) 
    33)          csite => cpoly%site(isi)
    34)          patchloop: do ipa=1,csite%npatches

    40)                call is_resolvable(csite,ipa,ico)
    41)                !---------------------------------------------------------------------------!
    42) 
    43)             end do cohortloop
    44)             !------------------------------------------------------------------------------!
    45)          end do patchloop

    50) end subroutine flag_stable_cohorts

    61) !     This sub-routine checks whether cohort leaves and wood thermodynamics can or cannot  !
    62) ! be solved.  This was taken out of flag_stable_cohorts so we can call it in other places  !
    63) ! and change the cohort status as soon as the conditions change.  We will skip the cohort  !
    64) ! if any of these conditions happens:                                                      !
    65) ! 1. The cohort leaf biomass is much less than what it would have when all                 !
    66) !    leaves are fully flushed (leaves only);                                               !
    67) ! 2. The cohort is buried in snow or under water                                           !
    68) ! 3. The cohort is extremely sparse.                                                       !
    69) ! 4. The user doesn't want to solve wood thermodynamics (wood only).                       !
    70) !------------------------------------------------------------------------------------------!
    71) subroutine is_resolvable(csite,ipa,ico)
    72)    use ed_state_vars , only : sitetype        & ! structure
    73)                             , patchtype       ! ! structure
    74)    use pft_coms      , only : veg_hcap_min    ! ! intent(in)
    75)    use ed_max_dims   , only : n_pft           ! ! intent(in)
    76) 
    77)    implicit none
    78)    !----- Arguments -----------------------------------------------------------------------!
    79)    type(sitetype)        , target     :: csite        ! Current site
    80)    integer               , intent(in) :: ipa          ! Patch index
    81)    integer               , intent(in) :: ico          ! Cohort index
    82)    !----- Local variables. ----------------------------------------------------------------!
    83)    type(patchtype)  , pointer         :: cpatch       ! Current patch
    84)    integer                            :: ipft         ! Cohort PFT
    85)    logical                            :: exposed      ! Cohort is above snow       [   T|F]
    86)    logical                            :: leaf_enough  ! Cohort have enough leaves  [   T|F]
    87)    logical                            :: wood_enough  ! Cohort have enough wood    [   T|F]
    88)    !---------------------------------------------------------------------------------------!
    89) 
    90) 
    91)    cpatch => csite%patch(ipa)
    92) 
    93)    ipft      = cpatch%pft(ico)
    94) 
    95)    !---------------------------------------------------------------------------------------!
    96)    ! 1.  Check for cohort height relative to snow/water depth.  If the cohort is buried in !
    97)    !     snow or has drowned in the standing water, we can't solve it.                     !
    98)    !---------------------------------------------------------------------------------------!
    99)    exposed      = cpatch%hite(ico)  > csite%total_sfcw_depth(ipa)
   100)    !---------------------------------------------------------------------------------------!
   101) 
   102) 
   103) 
   104)    !---------------------------------------------------------------------------------------!
   105)    ! 2.   Check whether this cohort is not extremely sparse.  Wood area heat capacity is   !
   106)    !      always set to zero when branch thermodynamics is turned off, so this will always !
   107)    !      be .false. in this case.                                                         !
   108)    !---------------------------------------------------------------------------------------!
   109)    leaf_enough  = cpatch%leaf_hcap(ico) > veg_hcap_min(ipft)
   110)    wood_enough  = cpatch%wood_hcap(ico) > veg_hcap_min(ipft)
   111)    !---------------------------------------------------------------------------------------!
   112) 
   113) 
   114) 
   115)    !---------------------------------------------------------------------------------------!
   116)    !     Save the tests in the cohort variable, so the checks are done consistently.       !
   117)    !---------------------------------------------------------------------------------------!
   118)    cpatch%leaf_resolvable(ico) = exposed .and. leaf_enough
   119)    cpatch%wood_resolvable(ico) = exposed .and. wood_enough
   120)    !---------------------------------------------------------------------------------------!
   121) 
   122) 
   123)    return
   124) end subroutine is_resolvable
