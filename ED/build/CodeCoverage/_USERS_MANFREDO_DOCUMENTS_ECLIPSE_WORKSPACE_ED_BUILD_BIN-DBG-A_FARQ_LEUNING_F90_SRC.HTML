   102)                                 , vm_q10                   & ! intent(in)
   103)                                 , vm_decay_a               & ! intent(in)
   104)                                 , vm_decay_b               & ! intent(in)
   105)                                 , vm_decay_e               & ! intent(in)

   108)                                 , rd_high_temp             & ! intent(in)

   110)                                 , rd_q10                   & ! intent(in)
   111)                                 , rd_decay_e               & ! intent(in)
   112)                                 , cuticular_cond           & ! intent(in)
   113)                                 , dark_respiration_factor  & ! intent(in)
   114)                                 , stomatal_slope           & ! intent(in)
   115)                                 , quantum_efficiency       ! ! intent(in)
   116)       use phenology_coms , only : vm0_tran                 & ! intent(in)

   119)                                 , vm0_min                  ! ! intent(in)

   121)                                 , o2_ref8                  ! ! intent(in)
   122)       use consts_coms    , only : mmh2oi8                  & ! intent(in)

   128)                                 , t008                     & ! intent(in)
   129)                                 , umol_2_mol8              & ! intent(in)

   131)                                 , Watts_2_Ein8             ! ! intent(in)

   136)       real(kind=4), intent(in)    :: can_rhos          ! Canopy air density     [    kg/m�]
   137)       real(kind=4), intent(in)    :: can_shv           ! Canopy air sp. hum.    [    kg/kg]
   138)       real(kind=4), intent(in)    :: can_co2           ! Canopy air CO2         [ �mol/mol]
   139)       integer     , intent(in)    :: ipft              ! Plant functional type  [      ---]
   140)       real(kind=4), intent(in)    :: leaf_par          ! Absorbed PAR           [     W/m�]
   141)       real(kind=4), intent(in)    :: leaf_temp         ! Leaf temperature       [        K]
   142)       real(kind=4), intent(in)    :: lint_shv          ! Leaf interc. sp. hum.  [    kg/kg]
   143)       real(kind=4), intent(in)    :: green_leaf_factor ! Frac. of on-allom. gr. [      ---]
   144)       real(kind=4), intent(in)    :: leaf_aging_factor ! Ageing parameter       [      ---]
   145)       real(kind=4), intent(in)    :: llspan            ! Leaf life span         [     mnth]
   146)       real(kind=4), intent(in)    :: vm_bar            ! Average Vm function    [�mol/m�/s]
   147)       real(kind=4), intent(in)    :: leaf_gbw          ! B.lyr. cnd. of H2O     [  kg/m�/s]
   148)       real(kind=4), intent(out)   :: A_open            ! Photosyn. rate (op.)   [�mol/m�/s]
   149)       real(kind=4), intent(out)   :: A_closed          ! Photosyn. rate (cl.)   [�mol/m�/s]
   150)       real(kind=4), intent(out)   :: A_light           ! Photosyn. rate (light) [�mol/m�/s]
   151)       real(kind=4), intent(out)   :: A_rubp            ! Photosyn. rate (RuBP)  [�mol/m�/s]
   152)       real(kind=4), intent(out)   :: A_co2             ! Photosyn. rate (CO2)   [�mol/m�/s]
   153)       real(kind=4), intent(out)   :: gsw_open          ! St. cnd. of H2O  (op.) [  kg/m�/s]
   154)       real(kind=4), intent(out)   :: gsw_closed        ! St. cnd. of H2O  (cl.) [  kg/m�/s]
   155)       real(kind=4), intent(out)   :: lsfc_shv_open     ! Leaf sfc. sp.hum.(op.) [    kg/kg]
   156)       real(kind=4), intent(out)   :: lsfc_shv_closed   ! Leaf sfc. sp.hum.(cl.) [    kg/kg]
   157)       real(kind=4), intent(out)   :: lsfc_co2_open     ! Leaf sfc. CO2    (op.) [ �mol/mol]
   158)       real(kind=4), intent(out)   :: lsfc_co2_closed   ! Leaf sfc. CO2    (cl.) [ �mol/mol]
   159)       real(kind=4), intent(out)   :: lint_co2_open     ! Intercell. CO2   (op.) [ �mol/mol]
   160)       real(kind=4), intent(out)   :: lint_co2_closed   ! Intercell. CO2   (cl.) [ �mol/mol]
   161)       real(kind=4), intent(out)   :: leaf_resp         ! Leaf respiration rate  [�mol/m�/s]
   162)       real(kind=4), intent(out)   :: vmout             ! Max. Rubisco capacity  [�mol/m�/s]
   163)       real(kind=4), intent(out)   :: comppout          ! GPP compensation point [ �mol/mol]

   166)       integer                     :: ib

   168)       real(kind=4)    , external  :: sngloff           ! Safe double -> single precision
   169)       !------------------------------------------------------------------------------------!

   171)       ib = 1
   172)       !$ ib = OMP_get_thread_num()+1
   173) 
   174)       !----- Initialise limit_flag to night time value. -----------------------------------!
   175)       limit_flag = 0

   180)       !------------------------------------------------------------------------------------!
   181)       !     Copy the meteorological forcing to the "met" structure.  Notice that some      !
   182)       ! variables go through unit conversions, and all variables are converted to double   !
   183)       ! precision.                                                                         !
   184)       !------------------------------------------------------------------------------------!

   186)       met(ib)%leaf_temp    = dble(leaf_temp)
   187)       met(ib)%can_rhos     = dble(can_rhos )
   188)       met(ib)%can_prss     = dble(can_prss )
   189)       met(ib)%can_o2       = o2_ref8
   190)       !----- 2. Convert specific humidity to mol/mol. -------------------------------------!
   191)       met(ib)%can_shv      = epi8 * dble(can_shv) 
   192)       !----- 3. Convert CO2 to mol/mol. ---------------------------------------------------!
   193)       met(ib)%can_co2      = dble(can_co2) * umol_2_mol8
   194)       !----- 4. Convert W/m2 to mol/m2/s. -------------------------------------------------!
   195)       met(ib)%par          = dble(leaf_par) * Watts_2_Ein8
   196)       !------------------------------------------------------------------------------------!
   197)       !  5. Intercellular specific humidity, which is assumed to be at saturation          !
   198)       !     given the leaf temperature.  We convert it to mol/mol.                         !
   199)       !------------------------------------------------------------------------------------!
   200)       met(ib)%lint_shv    = epi8 * dble(lint_shv)
   201)       !------------------------------------------------------------------------------------!
   202)       !  6. Find the conductivities for water and carbon.  The input for water is in       !
   203)       !     kg/m�/s, and here we convert to mol/m�/s.  The convertion coefficient from     !
   204)       !     water to carbon dioxide comes from M09's equation B14.  Here we multiply by    !
   205)       !     the effective area for transpiration, depending on whether the leaves of this  !
   206)       !     plant functional type are hypo-stomatous, symmetrical, or amphistomatous.      !
   207)       !------------------------------------------------------------------------------------!
   208)       met(ib)%blyr_cond_h2o = dble(leaf_gbw)  * mmdryi8 * effarea_transp(ipft)

   210)       !------------------------------------------------------------------------------------!
   211) 
   212) 
   213) 
   214)       !------------------------------------------------------------------------------------!

   216)       ! Convert all variables to mol and Kelvin, when needed.                              !
   217)       !------------------------------------------------------------------------------------!
   218)       thispft(ib)%photo_pathway = photosyn_pathway(ipft)
   219)       thispft(ib)%D0            = dble(D0(ipft))
   220)       thispft(ib)%b             = dble(cuticular_cond(ipft)) * umol_2_mol8
   221)       thispft(ib)%m             = dble(stomatal_slope(ipft))
   222)       thispft(ib)%vm_low_temp   = dble(vm_low_temp(ipft))  + t008
   223)       thispft(ib)%vm_high_temp  = dble(vm_high_temp(ipft)) + t008
   224)       thispft(ib)%vm_hor        = dble(vm_hor(ipft))
   225)       thispft(ib)%vm_q10        = dble(vm_q10(ipft))
   226)       thispft(ib)%vm_decay_a    = dble(vm_decay_a(ipft))
   227)       thispft(ib)%vm_decay_b    = dble(vm_decay_b(ipft))
   228)       thispft(ib)%vm_decay_e    = dble(vm_decay_e(ipft))
   229)       thispft(ib)%rd_low_temp   = dble(rd_low_temp(ipft))  + t008
   230)       thispft(ib)%rd_high_temp  = dble(rd_high_temp(ipft)) + t008
   231)       thispft(ib)%rd_hor        = dble(rd_hor(ipft))
   232)       thispft(ib)%rd_q10        = dble(rd_q10(ipft))
   233)       thispft(ib)%rd_decay_e    = dble(rd_decay_e(ipft))
   234)       thispft(ib)%alpha0        = dble(quantum_efficiency(ipft))
   235)       !------------------------------------------------------------------------------------!
   236) 
   237) 
   238) 
   239) 
   240)       !------------------------------------------------------------------------------------!
   241)       !     Find Vm0 for photosynthesis and respiration, depending on whether this PFT     !
   242)       ! has a light-controlled phenology or not.  Convert the resulting Vm into mol/m�/s.  !
   243)       !------------------------------------------------------------------------------------!
   244)       select case(phenology(ipft))
   245)       case (3)
   246)          !------ Light-controlled phenology. ----------------------------------------------!
   247)          thispft(ib)%vm0 = dble(vm0_amp / (1.0 + (llspan/vm0_tran)**vm0_slope) + vm0_min)  &
   248)                          * umol_2_mol8
   249)          thispft(ib)%rd0 = dble(vm_bar) * umol_2_mol8 * dble(dark_respiration_factor(ipft))
   250)       case default
   251)          !------ Other phenologies, no distinction on Vm0. --------------------------------!
   252)          thispft(ib)%vm0 = dble(vm0(ipft)) * umol_2_mol8
   253)          thispft(ib)%rd0 = dble(rd0(ipft)) * umol_2_mol8
   254)       end select
   255)       !------------------------------------------------------------------------------------!
   256) 
   257) 
   258) 
   259)       !------------------------------------------------------------------------------------!
   260)       !     Compute the photosynthesis and leaf respiration parameters that depend on      !
   261)       ! temperature, according to the parameters defined in the "thispft" structure and    !
   262)       ! the functional form chosen by the user.  The variables that are defined there are: !
   263)       ! - alpha     - the quantum yield, which may be a function of temperature.           !
   264)       ! - Vm        - the maximum capacity of Rubisco to perform the carboxylase function. !
   265)       ! - leaf_resp - the leaf respiration.                                                !
   266)       ! - compp     - the CO2 compensation point for gross photosynthesis (Gamma*)         !
   267)       ! - kco2      - Michaelis-Mentel coefficient for CO2                                 !
   268)       ! - ko2       - Michaelis-Mentel coefficient for O2.                                 !
   269)       !------------------------------------------------------------------------------------!
   270)       call comp_photo_tempfun(leaf_aging_factor,green_leaf_factor)
   271)       !------------------------------------------------------------------------------------!
   272) 
   273) 
   274) 
   275)       !------------------------------------------------------------------------------------!
   276)       !     Call the main solver.                                                          !
   277)       !------------------------------------------------------------------------------------!
   278)       call photosynthesis_exact_solver(limit_flag)
   279)       !------------------------------------------------------------------------------------!
   280) 
   281) 
   282) 
   283)       !------------------------------------------------------------------------------------!
   284)       !    Copy the solution to the standard output variables.  Here we convert the values !
   285)       ! back to the standard ED units                                                      !
   286)       !------------------------------------------------------------------------------------!
   287)       !----- Carbon demand, convert them to [�mol/m�/s]. ----------------------------------!
   288)       A_closed       = sngloff(stclosed(ib)%co2_demand    * mol_2_umol8 , tiny_offset)
   289)       A_open         = sngloff(stopen(ib)%co2_demand      * mol_2_umol8 , tiny_offset)
   290)       A_light        = sngloff(lightlim(ib)%co2_demand    * mol_2_umol8 , tiny_offset)
   291)       A_rubp         = sngloff(rubiscolim(ib)%co2_demand  * mol_2_umol8 , tiny_offset)
   292)       A_co2          = sngloff(co2lim(ib)%co2_demand      * mol_2_umol8 , tiny_offset)
   293)       !----- Stomatal resistance, convert the conductances to [kg/m�/s]. ------------------!
   294)       gsw_closed     = sngloff(stclosed(ib)%stom_cond_h2o * mmdry8 / effarea_transp(ipft)  &
   295)                               , tiny_offset)
   296)       gsw_open       = sngloff(stopen(ib)%stom_cond_h2o   * mmdry8 / effarea_transp(ipft)  &
   297)                               , tiny_offset)
   298)       !----- Leaf surface specific humidity, convert them to [kg/kg]. ---------------------!
   299)       lsfc_shv_closed = sngloff(stclosed(ib)%lsfc_shv     * ep8         , tiny_offset)
   300)       lsfc_shv_open   = sngloff(stopen(ib)%lsfc_shv       * ep8         , tiny_offset)
   301)       !----- Leaf surface CO2 concentration, convert them to [�mol/mol]. ------------------!
   302)       lsfc_co2_closed = sngloff(stclosed(ib)%lsfc_co2     * mol_2_umol8 , tiny_offset)
   303)       lsfc_co2_open   = sngloff(stopen(ib)%lsfc_co2       * mol_2_umol8 , tiny_offset)
   304)       !----- Intercellular carbon dioxide concentration, convert them to [�mol/mol]. ------!
   305)       lint_co2_closed = sngloff(stclosed(ib)%lint_co2     * mol_2_umol8 , tiny_offset)
   306)       lint_co2_open   = sngloff(stopen(ib)%lint_co2       * mol_2_umol8 , tiny_offset)
   307)       !----- Leaf respiration [�mol/m�/s]. ------------------------------------------------!
   308)       leaf_resp       = sngloff(aparms(ib)%leaf_resp      * mol_2_umol8 , tiny_offset)
   309)       !----- Maximum Rubisco capacity to perform the carboxylase function [�mol/m�/s]. ----!
   310)       vmout           = sngloff(aparms(ib)%vm             * mol_2_umol8 , tiny_offset)
   311)       !----- Gross photosynthesis compensation point, convert it to [�mol/mol]. -----------!
   312)       comppout        = sngloff(aparms(ib)%compp          * mol_2_umol8 , tiny_offset)
   313)       !------------------------------------------------------------------------------------!
   314)       return
   315)    end subroutine lphysiol_full

   326)    !     This subroutine computes the maximum capacity of Rubisco to perform the           !
   327)    ! carboxylase function at a certain temperature, given the PFT and phenology            !
   328)    ! properties, and, in case of light controlled phenology, the leaf life span and the    !
   329)    ! average reference value of the Vm function (Vm0).  The output variables are stored in !
   330)    ! the aparms structure, as they are going to be used in other sub-routines. Both Vm and !
   331)    ! the reference value Vm0 have units of �mol/m�/s                                       !
   332)    !     Compute the photosynthesis and leaf respiration parameters that depend on temper- !
   333)    ! ature, according to the parameters defined in the "thispft" structure and the         !
   334)    ! functional form chosen by the user.  The variables that are defined there are:        !
   335)    ! - alpha     - the quantum yield, which may be a function of temperature.              !
   336)    ! - Vm        - the maximum capacity of Rubisco to perform the carboxylase function.    !
   337)    ! - leaf_resp - the leaf respiration.                                                   !
   338)    ! - compp     - the CO2 compensation point for gross photosynthesis (Gamma*)            !
   339)    ! - kco2      - Michaelis-Mentel coefficient for CO2                                    !
   340)    ! - ko2       - Michaelis-Mentel coefficient for O2.                                    !
   341)    !---------------------------------------------------------------------------------------!
   342)    subroutine comp_photo_tempfun(leaf_aging_factor,green_leaf_factor)
   343)       use physiology_coms, only : iphysiol              & ! intent(in)
   344)                                 , quantum_efficiency_T  & ! intent(in)
   345)                                 , qyield08              & ! intent(in)
   346)                                 , qyield18              & ! intent(in)
   347)                                 , qyield28              & ! intent(in)
   348)                                 , ehleringer_alpha0c8   & ! intent(in)
   349)                                 , compp_refval8         & ! intent(out)
   350)                                 , compp_hor8            & ! intent(out)
   351)                                 , compp_q108            & ! intent(out)
   352)                                 , kco2_refval8          & ! intent(out)
   353)                                 , kco2_hor8             & ! intent(out)
   354)                                 , kco2_q108             & ! intent(out)
   355)                                 , ko2_refval8           & ! intent(out)
   356)                                 , ko2_hor8              & ! intent(out)
   357)                                 , ko2_q108              ! ! intent(out)
   358)       use c34constants   , only : met                   & ! intent(in) 
   359)                                 , aparms                & ! intent(in)
   360)                                 , thispft               ! ! intent(in)
   361)       use consts_coms    , only : lnexp_min8            & ! intent(in)
   362)                                 , lnexp_max8            & ! intent(in)
   363)                                 , rmol8                 & ! intent(in)
   364)                                 , t008                  ! ! intent(in)
   365)       !$ use omp_lib
   366)       implicit none
   367)       !------ Arguments. ------------------------------------------------------------------!
   368)       real(kind=4), intent(in) :: leaf_aging_factor ! Ageing factor             [      ---]
   369)       real(kind=4), intent(in) :: green_leaf_factor ! Greeness (prescr. phen.)  [      ---]
   370)       !------ Local variables. ------------------------------------------------------------!
   371)       real(kind=8)             :: vm_nocorr         ! Vm  with no correction    [ mol/m�/s]
   372)       real(kind=8)             :: rd_nocorr         ! Rd  with no correction    [ mol/m�/s]
   373)       real(kind=8)             :: lnexplow          ! Low temperature exponent  [      ---]
   374)       real(kind=8)             :: lnexphigh         ! High temperature exponent [      ---]
   375)       real(kind=8)             :: tlow_fun          ! Low temperature bound     [      ---]
   376)       real(kind=8)             :: thigh_fun         ! High temperature bound    [      ---]
   377)       real(kind=8)             :: greeness          ! Leaf "Greeness"           [   0 to 1]
   378)       integer                  :: ib
   379)       !------------------------------------------------------------------------------------!
   380)       
   381)       ib = 1
   382)       !$ ib = OMP_get_thread_num()+1
   383) 
   384)       !------------------------------------------------------------------------------------!
   385)       !     If this plant functional type has cold phenology associated with it, we must   !
   386)       ! determine the correction term for Vm.  Otherwise, we set the greeness to 1.        !
   387)       !------------------------------------------------------------------------------------!
   388)       if (leaf_aging_factor > 0.01 .and. green_leaf_factor > 0.0001) then
   389)          greeness = dble(leaf_aging_factor) / dble(green_leaf_factor)
   390)       else
   391)          greeness = 1.d0
   392)       end if
   393)       !------------------------------------------------------------------------------------!
   394) 
   395) 
   396) 
   397)       !------------------------------------------------------------------------------------!
   398)       !    Decide whether to make alpha dependent on temperature or not.  This is only     !
   399)       ! done for C3 plants, C4 plants always have constant alpha in our model.             !
   400)       !------------------------------------------------------------------------------------!
   401)       select case (thispft(ib)%photo_pathway)
   402)       case (3)
   403)          select case(quantum_efficiency_T)
   404)          case (0)
   405)             !----- The user wants constant alpha for C3 plants. ---------------------------!
   406)             aparms(ib)%alpha = thispft(ib)%alpha0
   407) 
   408)          case (1)
   409)             !------------------------------------------------------------------------------!
   410)             !    The user wants alpha as a function of temperature.  The equation below is !
   411)             ! equation 2 from E78, modified to take in temperature in Kelvin.              !
   412)             !------------------------------------------------------------------------------!
   413)             if (met(ib)%leaf_temp > t008) then
   414)                aparms(ib)%alpha = qyield08 + met(ib)%leaf_temp                             &
   415)                                            * (qyield18   + qyield28 * met(ib)%leaf_temp)
   416)             else
   417)                !---------------------------------------------------------------------------!
   418)                !     We don't apply the equation when the temperature is below freezing.   !
   419)                ! If this is the case, we use the value at 0 degC instead.                  !
   420)                !---------------------------------------------------------------------------!
   421)                aparms(ib)%alpha = ehleringer_alpha0c8
   422)             end if
   423)             !------------------------------------------------------------------------------!
   424)          end select
   425) 
   426)       case (4)
   427)          aparms(ib)%alpha = thispft(ib)%alpha0
   428) 
   429)       end select
   430)       !------------------------------------------------------------------------------------!
   431) 
   432) 
   433) 
   434)       !------------------------------------------------------------------------------------!
   435)       !     Here we decide which temperature-dependence to use, the "Arrhenius" form as in !
   436)       ! F96, or the power law as in C91.                                                   !
   437)       !------------------------------------------------------------------------------------!
   438)       select case (iphysiol)
   439)       case (0,1)
   440)          !---------------------------------------------------------------------------------!
   441)          !     We go for the "Arrhenius" form, like in F96 and M01.                        !
   442)          !---------------------------------------------------------------------------------!
   443) 
   444) 
   445)          !----- Find Vm using the Arrhenius equation, with no correction. -----------------!
   446)          vm_nocorr = greeness * arrhenius(met(ib)%leaf_temp,thispft(ib)%vm0                &
   447)                                          ,thispft(ib)%vm_hor)
   448)          !---------------------------------------------------------------------------------!
   449) 
   450) 
   451)          !---------------------------------------------------------------------------------!
   452)          !    Compute the functions that will control the Vm function for low and high     !
   453)          ! temperature.  In order to avoid floating point exceptions, we check whether the !
   454)          ! temperature will make the exponential too large or too small.                   !
   455)          !---------------------------------------------------------------------------------!
   456)          !----- Low temperature. ----------------------------------------------------------!
   457)          lnexplow  = thispft(ib)%vm_decay_e * (thispft(ib)%vm_low_temp  - met(ib)%leaf_temp)
   458)          lnexplow  = max(lnexp_min8,min(lnexp_max8,lnexplow))
   459)          tlow_fun  = 1.d0 +  exp(lnexplow)
   460)          !----- High temperature. ---------------------------------------------------------!
   461)          lnexphigh = thispft(ib)%vm_decay_e * (met(ib)%leaf_temp - thispft(ib)%vm_high_temp)
   462)          lnexphigh = max(lnexp_min8,min(lnexp_max8,lnexphigh))
   463)          thigh_fun = 1.d0 + exp(lnexphigh)
   464)          !---------------------------------------------------------------------------------!
   465) 
   466) 
   467)          !------ Correct Vm. --------------------------------------------------------------!
   468)          aparms(ib)%vm = vm_nocorr / (tlow_fun * thigh_fun)
   469)          !---------------------------------------------------------------------------------!
   470) 
   471) 
   472)          !----- Find Rd using the Arrhenius equation, with no correction. -----------------!
   473)          rd_nocorr = greeness * arrhenius(met(ib)%leaf_temp,thispft(ib)%rd0                &
   474)                                          ,thispft(ib)%rd_hor)
   475)          !---------------------------------------------------------------------------------!
   476) 
   477) 
   478)          !---------------------------------------------------------------------------------!
   479)          !    Compute the functions that will control the rd function for low and high     !
   480)          ! temperature.  In order to avoid floating point exceptions, we check whether the !
   481)          ! temperature will make the exponential too large or too small.                   !
   482)          !---------------------------------------------------------------------------------!
   483)          !----- Low temperature. ----------------------------------------------------------!
   484)          lnexplow  = thispft(ib)%rd_decay_e * (thispft(ib)%rd_low_temp - met(ib)%leaf_temp)
   485)          lnexplow  = max(lnexp_min8,min(lnexp_max8,lnexplow))
   486)          tlow_fun  = 1.d0 +  exp(lnexplow)
   487)          !----- High temperature. ---------------------------------------------------------!
   488)          lnexphigh = thispft(ib)%rd_decay_e * (met(ib)%leaf_temp - thispft(ib)%rd_high_temp)
   489)          lnexphigh = max(lnexp_min8,min(lnexp_max8,lnexphigh))
   490)          thigh_fun = 1.d0 + exp(lnexphigh)
   491)          !---------------------------------------------------------------------------------!
   492) 
   493) 
   494)          !------ Correct rd. --------------------------------------------------------------!
   495)          aparms(ib)%leaf_resp = rd_nocorr / (tlow_fun * thigh_fun)
   496)          !---------------------------------------------------------------------------------!
   497) 
   498) 
   499)          !---------------------------------------------------------------------------------!
   500)          !    If this is a C3 plant, find the compensation point, and the Michaelis-Mentel !
   501)          ! constants for CO2 and O2.  Otherwise, assign zeroes for compensation point and  !
   502)          ! the Michaelis-Mentel constant for CO2.  The oxygen one should have no impact,   !
   503)          ! but we always assign it to avoid divisions by zero.                             !
   504)          !---------------------------------------------------------------------------------!
   505)          select case (thispft(ib)%photo_pathway)
   506)          case (3)
   507)             aparms(ib)%compp = arrhenius(met(ib)%leaf_temp,compp_refval8,compp_hor8)
   508)             aparms(ib)%kco2  = arrhenius(met(ib)%leaf_temp,kco2_refval8 ,kco2_hor8 )
   509)          case (4)
   510)             aparms(ib)%compp = 0.d0
   511)             aparms(ib)%kco2  = 0.d0
   512)          end select
   513)          aparms(ib)%ko2   = arrhenius(met(ib)%leaf_temp,ko2_refval8,ko2_hor8)
   514)          !---------------------------------------------------------------------------------!
   515) 
   516) 
   517) 
   518) 
   519)       case (2,3)
   520)          !---------------------------------------------------------------------------------!
   521)          !     We go for the power law, as in C91/C92.                                     !
   522)          !---------------------------------------------------------------------------------!
   523) 
   524) 
   525)          !----- Find Vm using the Collatz equation, with no correction. -------------------!
   526)          vm_nocorr = greeness * collatz(met(ib)%leaf_temp,thispft(ib)%vm0                  &
   527)                                        ,thispft(ib)%vm_q10)
   528)          !---------------------------------------------------------------------------------!
   529) 
   530) 
   531)          !---------------------------------------------------------------------------------!
   532)          !    Compute the functions that will control the Vm function for high temper-     !
   533)          ! ature.  In order to avoid floating point exceptions, we check whether the       !
   534)          ! temperature will make the exponential too small or too large.                   !
   535)          !---------------------------------------------------------------------------------!
   536)          !----- Low temperature. ----------------------------------------------------------!
   537)          lnexplow  = thispft(ib)%vm_decay_e * (thispft(ib)%vm_low_temp - met(ib)%leaf_temp)
   538)          lnexplow  = max(lnexp_min8,min(lnexp_max8,lnexplow))
   539)          tlow_fun  = 1.d0 +  exp(lnexplow)
   540)          !----- High temperature. ---------------------------------------------------------!
   541)          !lnexphigh = ( - thispft(ib)%vm_decay_a + thispft(ib)%vm_decay_b * met(ib)%leaf_temp)             &
   542)          !          / (rmol8 * met(ib)%leaf_temp)
   543)          !lnexphigh = max(lnexp_min8,min(lnexp_max8,lnexphigh))
   544)          !thigh_fun = 1.d0 + exp(lnexphigh)
   545)          !----- High temperature. ---------------------------------------------------------!
   546)          lnexphigh = thispft(ib)%vm_decay_e * (met(ib)%leaf_temp - thispft(ib)%vm_high_temp)
   547)          lnexphigh = max(lnexp_min8,min(lnexp_max8,lnexphigh))
   548)          thigh_fun = 1.d0 + exp(lnexphigh)
   549)          !---------------------------------------------------------------------------------!
   550) 
   551) 
   552)          !------ Correct Vm. --------------------------------------------------------------!
   553)          aparms(ib)%vm = vm_nocorr / (tlow_fun * thigh_fun)
   554)          !---------------------------------------------------------------------------------!
   555) 
   556) 
   557)          !----- Find Rd using the Collatz equation, with no correction. -------------------!
   558)          rd_nocorr = greeness * collatz(met(ib)%leaf_temp,thispft(ib)%rd0                  &
   559)                                        ,thispft(ib)%rd_q10)
   560)          !---------------------------------------------------------------------------------!
   561) 
   562) 
   563)          !---------------------------------------------------------------------------------!
   564)          !    Compute the functions that will control the rd function for low and high     !
   565)          ! temperature.  In order to avoid floating point exceptions, we check whether the !
   566)          ! temperature will make the exponential too large or too small.                   !
   567)          !---------------------------------------------------------------------------------!
   568)          !----- Low temperature. ----------------------------------------------------------!
   569)          lnexplow  = thispft(ib)%rd_decay_e * (thispft(ib)%rd_low_temp  - met(ib)%leaf_temp)
   570)          lnexplow  = max(lnexp_min8,min(lnexp_max8,lnexplow))
   571)          tlow_fun  = 1.d0 +  exp(lnexplow)
   572)          !----- High temperature. ---------------------------------------------------------!
   573)          lnexphigh = thispft(ib)%rd_decay_e * (met(ib)%leaf_temp - thispft(ib)%rd_high_temp)
   574)          lnexphigh = max(lnexp_min8,min(lnexp_max8,lnexphigh))
   575)          thigh_fun = 1.d0 + exp(lnexphigh)
   576)          !---------------------------------------------------------------------------------!
   577) 
   578)          !------ Correct Rd. --------------------------------------------------------------!
   579)          aparms(ib)%leaf_resp = rd_nocorr / (tlow_fun * thigh_fun)
   580)          !---------------------------------------------------------------------------------!
   581) 
   582) 
   583)          !---------------------------------------------------------------------------------!
   584)          !    If this is a C3 plant, find the compensation point, and the Michaelis-Mentel !
   585)          ! constants for CO2 and O2.  Otherwise, assign zeroes for compensation point and  !
   586)          ! the Michaelis-Mentel constant for CO2.  The oxygen one should have no impact,   !
   587)          ! but we always assign it to avoid divisions by zero.                             !
   588)          !---------------------------------------------------------------------------------!
   589)          select case (thispft(ib)%photo_pathway)
   590)          case (3)
   591)             aparms(ib)%compp = collatz(met(ib)%leaf_temp,compp_refval8,compp_q108)
   592)             aparms(ib)%kco2  = collatz(met(ib)%leaf_temp,kco2_refval8 ,kco2_q108 )
   593)          case (4)
   594)             aparms(ib)%compp = 0.d0
   595)             aparms(ib)%kco2  = 0.d0
   596)          end select
   597)          aparms(ib)%ko2   = collatz(met(ib)%leaf_temp,ko2_refval8,ko2_q108)
   598)          !---------------------------------------------------------------------------------!
   599)       end select
   600)       !------------------------------------------------------------------------------------!
   601) 
   602) 
   603)       return
   604)    end subroutine comp_photo_tempfun

   614)    !=======================================================================================!
   615)    !     This subroutine is the main driver for the C3 photosynthesis.                     !
   616)    !---------------------------------------------------------------------------------------!
   617)    subroutine photosynthesis_exact_solver(limit_flag)
   618)       use c34constants   , only : met              & ! intent(in)
   619)                                 , thispft          & ! intent(in)
   620)                                 , stopen           & ! intent(inout)
   621)                                 , stclosed         & ! intent(inout)
   622)                                 , rubiscolim       & ! intent(inout)
   623)                                 , co2lim           & ! intent(inout)
   624)                                 , lightlim         & ! intent(inout)
   625)                                 , copy_solution    ! ! intent(in)
   626)       !$ use omp_lib
   627)       implicit none
   628)       !------ Arguments. ------------------------------------------------------------------!
   629)       integer     , intent(out) :: limit_flag       ! Flag with limiting case    [     ---]
   630)       !------ Local variables. ------------------------------------------------------------!
   631)       logical                   :: success          ! The solver succeeded.      [     T|F]
   632)       real(kind=8)              :: par_twilight_min ! Minimum daytime radiation  [mol/m2/s]
   633)       integer                   :: ib
   634)       !------------------------------------------------------------------------------------!
   635)       ib = 1
   636)       !$ ib = OMP_get_thread_num()+1
   637) 
   638)       !------------------------------------------------------------------------------------!
   639)       !      Initialise the limitation flag with some dummy value, so the debugger does    !
   640)       ! not complain.                                                                      !
   641)       !------------------------------------------------------------------------------------!
   642)       limit_flag = 99
   643)       !------------------------------------------------------------------------------------!
   644) 
   645) 
   646) 
   647)       !------------------------------------------------------------------------------------!
   648)       !      Initialise the parameters to compute the carbon demand for the case where the !
   649)       ! stomata are closed.  We call the solver for the specific case in which the stomata !
   650)       ! are closed, because in this case neither the carbon demand nor the stomatal water  !
   651)       ! conductance depend on the intercellular CO2 mixing ratio.                          !
   652)       !------------------------------------------------------------------------------------!
   653)       call set_co2_demand_params('CLOSED')
   654)       call solve_aofixed_case(stclosed(ib),success)
   655)       if (.not. success) then
   656)          call fatal_error ('Solution failed for closed case'                               &
   657)                           ,'photosynthesis_exact_solver','farq_leuning.f90')
   658)       end if
   659)       !------------------------------------------------------------------------------------!
   660) 
   661) 
   662) 
   663)       !------------------------------------------------------------------------------------!
   664)       !    There is enough light to be considered at least dawn or dusk, so we go with     !
   665)       ! photosynthesis.  The closed stomata is already solved, so now we must solve the    !
   666)       ! open stomata case only.                                                            !
   667)       !    During the day, conditions may lead to light, rubisco, or CO2 limitation on the !
   668)       ! photosynthesis.  Following M01, we solve all cases and determine which ones gives  !
   669)       ! the lowest carbon demand.  The actual solution is going to be simply the one with  !
   670)       ! the lowest carbon_demand.                                                          !
   671)       !------------------------------------------------------------------------------------!
   672) 
   673) 
   674) 
   675)       !------------------------------------------------------------------------------------!
   676)       !   1. The light-limited (aka PAR) case.                                             !
   677)       !------------------------------------------------------------------------------------!
   678)       !----- Update the CO2 demand function parameters for light limitation. --------------!
   679)       call set_co2_demand_params('LIGHT')
   680)       !------------------------------------------------------------------------------------!
   681)       !    First we check whether it is at least dawn or dusk.  In case it is not, no      !
   682)       ! photosynthesis should happen, so we copy the closed case stomata values to the     !
   683)       ! open case.  Limit_flag becomes 0, which is the flag for night time limitation.     !
   684)       !------------------------------------------------------------------------------------!
   685)       par_twilight_min = find_twilight_min()
   686)       if (met(ib)%par < par_twilight_min) then
   687)          call copy_solution(stclosed(ib),stopen  (ib))
   688)          call copy_solution(stclosed(ib),lightlim(ib))
   689)          limit_flag = 0
   690)       else
   691)          !---------------------------------------------------------------------------------!
   692)          !     Day time: choose the appropriate solver depending on the photosynthetic     !
   693)          ! pathway.                                                                        !
   694)          !---------------------------------------------------------------------------------!
   695)          select case(thispft(ib)%photo_pathway)
   696)          case (3)
   697)             call solve_iterative_case(lightlim(ib),success)
   698)          case (4)
   699)             call solve_aofixed_case(lightlim(ib),success)
   700)          end select
   701)          !---------------------------------------------------------------------------------!
   702)          !     In case success was returned as "false", this means that the light-limited  !
   703)          ! case didn't converge (there was no root).  If this is the case we give up and   !
   704)          ! close all stomata, as there was no viable state for stomata to remain opened.   !
   705)          !---------------------------------------------------------------------------------!
   706)          if (.not. success) then
   707)             call copy_solution(stclosed(ib),stopen  (ib))
   708)             call copy_solution(stclosed(ib),lightlim(ib))
   709)             limit_flag = -1
   710)          end if
   711)       end if
   712)       !------------------------------------------------------------------------------------!
   713) 
   714) 
   715) 
   716) 
   717)       !------------------------------------------------------------------------------------!
   718)       !    2. The Rubisco-limited (aka Vm) case.                                           !
   719)       !------------------------------------------------------------------------------------!
   720)       !----- Update the CO2 demand function parameters for Rubisco limitation. ------------!
   721)       call set_co2_demand_params('RUBISCO')
   722)       !----- Choose the appropriate solver depending on the kind of photosynthesis. -------!
   723)       select case(thispft(ib)%photo_pathway)
   724)       case (3)
   725)          call solve_iterative_case(rubiscolim(ib),success)
   726)       case (4)
   727)          call solve_aofixed_case(rubiscolim(ib),success)
   728)       end select
   729)       !------------------------------------------------------------------------------------!
   730)       !     In case success was returned as "false", this means that the Rubisco-limited   !
   731)       ! case didn't converge (there was no root).  If this is the case we give up and      !
   732)       ! close all stomata, as there was no viable state for stomata to remain opened.      !
   733)       !------------------------------------------------------------------------------------!
   734)       if (.not. success) then
   735)          call copy_solution(stclosed(ib),rubiscolim(ib))
   736)          select case (limit_flag)
   737)          case (99)
   738)             call copy_solution(stclosed(ib),stopen(ib))
   739)             limit_flag = -2
   740)          end select
   741)       end if
   742)       !------------------------------------------------------------------------------------!
   743) 
   744) 
   745) 
   746) 
   747)       !------------------------------------------------------------------------------------!
   748)       ! 3. The low CO2 concentration case (aka the CO2 case).                              !
   749)       !------------------------------------------------------------------------------------!
   750)       !----- Update the CO2 demand function parameters for CO2 limitation. ----------------!
   751)       call set_co2_demand_params('CO2')
   752)       !----- Choose the appropriate solver depending on the kind of photosynthesis. -------!
   753)       select case(thispft(ib)%photo_pathway)
   754)       case (3)
   755)          !---------------------------------------------------------------------------------!
   756)          !    C3, there is no CO2 limitation in this formulation.  Copy the Rubisco-       !
   757)          ! -limited case.                                                                  !
   758)          !---------------------------------------------------------------------------------!
   759)          call copy_solution(rubiscolim(ib),co2lim(ib))
   760)          success           = .true.
   761)          !---------------------------------------------------------------------------------!
   762)          !    C3, use the expression from C91, that Ao should not exceed 0.5 * Vm.         !
   763)          !---------------------------------------------------------------------------------!
   764)          ! call solve_aofixed_case(co2lim(ib),success)
   765)          !---------------------------------------------------------------------------------!
   766) 
   767)       case (4)
   768)          !---------------------------------------------------------------------------------!
   769)          !    C4, both carbon_demand and stomatal conductance of water depend on the       !
   770)          ! inter-cellular CO2 concentration.  We must find all these three variables       !
   771)          ! simultaneously, using an iterative method.                                      !
   772)          !---------------------------------------------------------------------------------!
   773)          call solve_iterative_case(co2lim(ib),success)
   774)       end select
   775)       !------------------------------------------------------------------------------------!
   776)       !     In case success was returned as "false", this means that the CO2-limited case  !
   777)       ! didn't converge (there was no root).  If this is the case we give up and close all !
   778)       ! stomata, as there was no viable state for stomata to remain opened.                !
   779)       !------------------------------------------------------------------------------------!
   780)       if (.not. success) then
   781)          call copy_solution(stclosed(ib),co2lim(ib))
   782)          select case (limit_flag)
   783)          case (99)
   784)             call copy_solution(stclosed(ib),stopen(ib))
   785)             limit_flag = -3
   786)          end select
   787)       end if
   788)       !------------------------------------------------------------------------------------!
   789) 
   790) 
   791) 
   792)       !------------------------------------------------------------------------------------!
   793)       !     The actual solution will be the one with the lowest carbon demand.  In case    !
   794)       ! it is night time or one of the solvers failed, we close all stomata and skip this  !
   795)       ! decision step.                                                                     !
   796)       !------------------------------------------------------------------------------------!
   797)       select case (limit_flag)
   798)       case (99)
   799)          if (lightlim(ib)%co2_demand <= rubiscolim(ib)%co2_demand .and.                    &
   800)              lightlim(ib)%co2_demand <= co2lim(ib)%co2_demand              )  then
   801)             !----- Light is the strongest limitation. -------------------------------------!
   802)             call copy_solution(lightlim(ib),stopen(ib))
   803)             limit_flag = 1
   804)          elseif (rubiscolim(ib)%co2_demand <  lightlim(ib)%co2_demand .and.                &
   805)                  rubiscolim(ib)%co2_demand <= co2lim(ib)%co2_demand        ) then
   806)             !----- Rubisco is the strongest limitation. -----------------------------------!
   807)             call copy_solution(rubiscolim(ib),stopen(ib))
   808)             limit_flag = 2
   809)          else
   810)             !----- CO2 is the strongest limitation. ---------------------------------------!
   811)             call copy_solution(co2lim(ib),stopen(ib))
   812)             limit_flag = 3
   813)          end if
   814)          !---------------------------------------------------------------------------------!
   815)       end select
   816)       !------------------------------------------------------------------------------------!
   817) 
   818) 
   819)       return
   820)    end subroutine photosynthesis_exact_solver

   831)    !     This subroutine solves the model for the case where the carbon demand doesn't     !
   832)    ! depend on the internal carbon.  This is simpler than the iterative case because we    !
   833)    ! can solve through a quadratic equation for the stomatal conductance for water vapour. !
   834)    !---------------------------------------------------------------------------------------!
   835)    subroutine solve_aofixed_case(answer,success)
   836)       use c34constants   , only : solution_vars     & ! structure
   837)                                 , met               & ! intent(in)
   838)                                 , thispft           & ! intent(in)
   839)                                 , aparms            ! ! intent(in)
   840)       use physiology_coms, only : gsw_2_gsc8        & ! intent(in)
   841)                                 , c34smax_gsw8      & ! intent(in)
   842)                                 , c34smin_lint_co28 & ! intent(in)
   843)                                 , c34smax_lint_co28 ! ! intent(in)
   844)       !$ use omp_lib
   845)       implicit none
   846)       !----- Arguments. -------------------------------------------------------------------!
   847)       type(solution_vars), intent(out) :: answer
   848)       logical            , intent(out) :: success
   849)       !----- Local variables. -------------------------------------------------------------!
   850)       real(kind=8)                     :: qterm1
   851)       real(kind=8)                     :: qterm2
   852)       real(kind=8)                     :: qterm3
   853)       real(kind=8)                     :: aquad
   854)       real(kind=8)                     :: bquad
   855)       real(kind=8)                     :: cquad
   856)       real(kind=8)                     :: discr
   857)       real(kind=8)                     :: gswroot1
   858)       real(kind=8)                     :: gswroot2
   859)       real(kind=8)                     :: ciroot1
   860)       real(kind=8)                     :: ciroot2
   861)       logical                          :: bounded1
   862)       logical                          :: bounded2
   863)       integer                          :: ib
   864)       !------------------------------------------------------------------------------------!
   865) 
   866)       ib = 1
   867)       !$ ib = OMP_get_thread_num()+1
   868) 
   869)       !------------------------------------------------------------------------------------!
   870)       !   1. Initialise the success flag as true.  In case we have trouble solving this    !
   871)       !      case, we switch the flag to false before we quit.                             !
   872)       !------------------------------------------------------------------------------------!
   873)       success = .true.
   874)       !------------------------------------------------------------------------------------!
   875) 
   876)       !------------------------------------------------------------------------------------!
   877)       !   2. Since the carbon demand doesn't depend on the intercellular CO2, compute it   !
   878)       !      using the first guess.                                                        !
   879)       !------------------------------------------------------------------------------------!
   880)       answer%co2_demand = calc_co2_demand(met(ib)%can_co2)
   881)       !------------------------------------------------------------------------------------!
   882) 
   883) 
   884) 
   885)       !----- 3. Compute the leaf surface CO2. ---------------------------------------------!
   886)       answer%lsfc_co2  = met(ib)%can_co2 - answer%co2_demand / met(ib)%blyr_cond_co2
   887)       !------------------------------------------------------------------------------------!
   888) 
   889) 
   890) 
   891)       !------------------------------------------------------------------------------------!
   892)       !   3. Here we check the sign of the carbon demand.                                  !
   893)       !------------------------------------------------------------------------------------!
   894)       if (answer%co2_demand <= 0.d0) then
   895)          !---------------------------------------------------------------------------------!
   896)          !     If carbon demand is zero or negative, this means that light is below the    !
   897)          ! light compensation point, so all stomata should remain closed.                  !
   898)          !---------------------------------------------------------------------------------!
   899)          answer%stom_cond_h2o = thispft(ib)%b
   900)          answer%stom_cond_co2 = gsw_2_gsc8 * answer%stom_cond_h2o
   901)          answer%lint_co2      = answer%lsfc_co2 - answer%co2_demand / answer%stom_cond_co2
   902)       else
   903)          !---------------------------------------------------------------------------------!
   904)          !     Carbon demand is positive, look for a solution.                             !
   905)          !---------------------------------------------------------------------------------!
   906)          !----- Find auxiliary coefficients to compute the quadratic terms. ---------------!
   907)          qterm1 = (met(ib)%can_co2 - aparms(ib)%compp) * met(ib)%blyr_cond_co2             &
   908)                 - answer%co2_demand
   909)          qterm2 = (thispft(ib)%d0 + met(ib)%lint_shv - met(ib)%can_shv)                    &
   910)                 * met(ib)%blyr_cond_h2o
   911)          qterm3 = thispft(ib)%m * answer%co2_demand * thispft(ib)%d0                       &
   912)                 * met(ib)%blyr_cond_co2
   913)          !----- Find the coefficients for the quadratic equation. -------------------------!
   914)          aquad = qterm1 * thispft(ib)%d0
   915)          bquad = qterm1 * qterm2 - aquad * thispft(ib)%b - qterm3
   916)          cquad = - qterm1 * qterm2 * thispft(ib)%b - qterm3 * met(ib)%blyr_cond_h2o
   917)          !----- Solve the quadratic equation for gsw. -------------------------------------!
   918)          if (aquad == 0.d0) then
   919)             !----- Not really a quadratic equation. ---------------------------------------!
   920)             gswroot1 = -cquad / bquad
   921)             gswroot2 = discard
   922)          else
   923)             !----- A quadratic equation, find the discriminant. ---------------------------!
   924)             discr = bquad * bquad - 4.d0 * aquad * cquad
   925)             !----- Decide what to do based on the discriminant. ---------------------------!
   926)             if (discr == 0.d0) then
   927)                !----- Double root. --------------------------------------------------------!
   928)                gswroot1 = - bquad / (2.d0 * aquad)
   929)                gswroot2 = discard
   930)             elseif (discr > 0.d0) then
   931)                !----- Two distinct roots. -------------------------------------------------!
   932)                gswroot1 = (- bquad - sqrt(discr)) / (2.d0 * aquad)
   933)                gswroot2 = (- bquad + sqrt(discr)) / (2.d0 * aquad)
   934)             else
   935)                !----- None of the roots are real, this solution failed. -------------------!
   936)                success = .false.
   937)                return
   938)             end if
   939)          end if
   940)          !---------------------------------------------------------------------------------!
   941) 
   942) 
   943) 
   944)          !---------------------------------------------------------------------------------!
   945)          !     Check both solutions, and decide which one makes sense.  Once the right     !
   946)          ! one is determined, compute the stomatal resistance for CO2 and the inter-       !
   947)          ! cellular CO2 concentration.  In case both make solutions make sense (unlikely), !
   948)          ! we decide the root based on the intercellular CO2.                              !
   949)          !---------------------------------------------------------------------------------!
   950)          bounded1 = gswroot1 >= thispft(ib)%b .and. gswroot1 <= c34smax_gsw8
   951)          bounded2 = gswroot2 >= thispft(ib)%b .and. gswroot2 <= c34smax_gsw8
   952)          if (bounded1 .and. bounded2) then
   953)             !----- Both solutions are valid, warn the user as this should never happen. ---!
   954)             ciroot1 = answer%lsfc_co2 - answer%co2_demand / (gsw_2_gsc8 * gswroot1)
   955)             ciroot2 = answer%lsfc_co2 - answer%co2_demand / (gsw_2_gsc8 * gswroot2)
   956) 
   957)             bounded1 = ciroot1 >= c34smin_lint_co28 .and.                                  &
   958)                        ciroot1 <= min(c34smax_lint_co28,met(ib)%can_co2)
   959)             bounded2 = ciroot2 >= c34smin_lint_co28 .and.                                  &
   960)                        ciroot2 <= min(c34smax_lint_co28,met(ib)%can_co2)
   961)              
   962)             if (bounded1 .and. bounded2) then
   963)                !----- Both intercellular CO2 work, pick the highest and warn the user. ----!
   964)                if (ciroot1 >= ciroot2) then
   965)                   answer%stom_cond_h2o = gswroot1
   966)                   answer%stom_cond_co2 = gsw_2_gsc8 * answer%stom_cond_h2o
   967)                   answer%lint_co2      = ciroot1
   968)                else
   969)                   answer%stom_cond_h2o = gswroot2
   970)                   answer%stom_cond_co2 = gsw_2_gsc8 * answer%stom_cond_h2o
   971)                   answer%lint_co2      = ciroot2
   972)                end if
   973)             elseif (bounded1) then
   974)                answer%stom_cond_h2o = gswroot1
   975)                answer%stom_cond_co2 = gsw_2_gsc8 * answer%stom_cond_h2o
   976)                answer%lint_co2      = ciroot1
   977)             elseif (bounded2) then
   978)                answer%stom_cond_h2o = gswroot2
   979)                answer%stom_cond_co2 = gsw_2_gsc8 * answer%stom_cond_h2o
   980)                answer%lint_co2      = ciroot2
   981)             else
   982)                success = .false.
   983)                return
   984)             end if
   985)          elseif (bounded1) then
   986)             !----- First root is the only one that makes sense. ---------------------------!
   987)             answer%stom_cond_h2o = gswroot1
   988)             answer%stom_cond_co2 = gsw_2_gsc8 * answer%stom_cond_h2o
   989)             answer%lint_co2      = answer%lsfc_co2                                         &
   990)                                  - answer%co2_demand / answer%stom_cond_co2
   991) 
   992)          elseif (bounded2) then
   993)             !----- Second root is the only one that makes sense. --------------------------!
   994)             answer%stom_cond_h2o = gswroot2
   995)             answer%stom_cond_co2 = gsw_2_gsc8 * answer%stom_cond_h2o
   996)             answer%lint_co2      = answer%lsfc_co2                                         &
   997)                                  - answer%co2_demand / answer%stom_cond_co2
   998)          else
   999)             !----- None of the solutions are bounded.  This solution failed. --------------!
  1000)             success = .false.
  1001)             return
  1002)          end if
  1003)          !---------------------------------------------------------------------------------!
  1004)       end if
  1005)       !------------------------------------------------------------------------------------!
  1006) 
  1007) 
  1008)       !------------------------------------------------------------------------------------!
  1009)       !   8. Lastly, find the surface water specific humidity.                             !
  1010)       !------------------------------------------------------------------------------------!
  1011)       answer%lsfc_shv = ( answer%stom_cond_h2o * met(ib)%lint_shv                          &
  1012)                         + met(ib)%blyr_cond_h2o    * met(ib)%can_shv  )                    &
  1013)                       / ( met(ib)%blyr_cond_h2o + answer%stom_cond_h2o)
  1014)       !------------------------------------------------------------------------------------!
  1015) 
  1016)       return
  1017)    end subroutine solve_aofixed_case

  1027)    !=======================================================================================!
  1028)    !     This sub-routine will solve the case in which both the carbon demand and the      !
  1029)    ! stomatal conductance of water are functions of the intercellular CO2.  This function  !
  1030)    ! can be used for the simpler cases too, but it's not advisable because this is based   !
  1031)    ! on iterative methods, which makes the solution slower.                                !
  1032)    !     The iterative method is designed to use Newton's method as the default, and this  !
  1033)    ! should take care of most cases.  In case Newton's method fails, it will fall back to  !
  1034)    ! the modified Regula Falsi method (Illinois) and look for guesses with opposite sign.  !
  1035)    ! If the method fails finding the pair, it means that there is no viable solution       !
  1036)    ! within this range, so the method quits and return the error message.                  !
  1037)    !---------------------------------------------------------------------------------------!
  1038)    subroutine solve_iterative_case(answer,converged)
  1039)       use c34constants   , only : solution_vars     & ! structure
  1040)                                 , met               ! ! intent(in)
  1041)       use physiology_coms, only : gsw_2_gsc8        ! ! intent(in)
  1042)       !$ use omp_lib
  1043)       implicit none
  1044)       !----- Arguments. -------------------------------------------------------------------!
  1045)       type(solution_vars), intent(out) :: answer    ! The strutcure with the answer
  1046)       logical            , intent(out) :: converged ! A solution was found     [      T|F]
  1047)       !----- Local variables. -------------------------------------------------------------!
  1048)       real(kind=8)                     :: ci        ! Intercellular CO2         [  mol/mol]
  1049)       real(kind=8)                     :: cia       ! Smallest/previous guess   [  mol/mol]
  1050)       real(kind=8)                     :: ciz       ! Largest/new guess         [  mol/mol]
  1051)       real(kind=8)                     :: deriv     ! Function derivative       [  mol/mol]
  1052)       real(kind=8)                     :: fun       ! Function evaluation       [mol�/mol�]
  1053)       real(kind=8)                     :: funa      ! Smallest  guess function  [mol�/mol�]
  1054)       real(kind=8)                     :: funz      ! Largest   guess function  [mol�/mol�]
  1055)       real(kind=8)                     :: delta     ! Aux. var to find 2nd guess[         ]
  1056)       real(kind=8)                     :: cimin     ! Minimum intercell. CO2    [  mol/mol]
  1057)       real(kind=8)                     :: cimax     ! Maximum intercell. CO2    [  mol/mol]
  1058)       integer                          :: itn       ! Iteration counter         [      ---]
  1059)       integer                          :: itb       ! Iteration counter         [      ---]
  1060)       logical                          :: zside     ! Check for 1-sided appr.   [      ---]
  1061)       logical                          :: hitmin    ! 2nd guess tried minimum   [      ---]
  1062)       logical                          :: hitmax    ! 2nd guess tried maximum   [      ---]
  1063)       logical                          :: bounded   ! Guess range is bounded.   [      T|F]
  1064)       integer                          :: ib
  1065)       !------------------------------------------------------------------------------------!
  1066) 
  1067)       ib = 1
  1068)       !$ ib = OMP_get_thread_num()+1
  1069) 
  1070)       !------------------------------------------------------------------------------------!
  1071)       !      Initialise the convergence flag.  Here we start realistic, ops, I mean,       !
  1072)       ! pessimistic, and assume that we are failing.  This will be switched to true only   !
  1073)       ! if a real answer is found, so in case we quit the sub-routine due to impossible    !
  1074)       ! solution, the result will be failure.                                              !
  1075)       !------------------------------------------------------------------------------------!
  1076)       converged = .false.
  1077)       !------------------------------------------------------------------------------------!
  1078) 
  1079) 
  1080) 
  1081)       !------------------------------------------------------------------------------------!
  1082)       !    Determine the minimum and maximum intercellular CO2 that will still produce  a  !
  1083)       ! positive and bounded conductance, which should be above the cuticular conductance  !
  1084)       ! (otherwise there is no reason to keep stomata opened.                              !
  1085)       !------------------------------------------------------------------------------------!
  1086)       call find_lint_co2_bounds(cimin,cimax,bounded)
  1087)       if (bounded) then
  1088)          !---------------------------------------------------------------------------------!
  1089)          !     We have found bounds, find put the first guess in the middle and find the   !
  1090)          ! function evaluation and derivative of this guess.                               !
  1091)          !---------------------------------------------------------------------------------!
  1092)          cia = sqrt(cimin*cimax)
  1093)          call iter_solver_step(.true.,cia,funa,deriv)
  1094)       else
  1095)          !----- No reasonable bound was found, we don't even bother solving this case. ----!
  1096)          return
  1097)       end if
  1098)       !------------------------------------------------------------------------------------!
  1099) 
  1100) 
  1101) 
  1102)       !----- Copy to the new guess just in case it fails at the first iteration -----------!
  1103)       ciz = cia
  1104)       fun = funa
  1105)       !------------------------------------------------------------------------------------!
  1106) 
  1107)       if (fun == 0.d0) then 
  1108)          !----- We have actually hit the jackpot, the answer is ciz. ----------------------!
  1109)          ci        = ciz
  1110)          converged = .true.
  1111)       end if
  1112) 
  1113) 
  1114)       !------------------------------------------------------------------------------------!
  1115)       !     Enter Newton's method loop in case we haven't found the answer.                !
  1116)       !------------------------------------------------------------------------------------!
  1117)       if (.not. converged) then
  1118)          newloop: do itn = 1,maxfpofl/6
  1119)             !------------------------------------------------------------------------------!
  1120)             !    In case the derivative is bad, we give up on Newton's and go with Regula  !
  1121)             ! Falsi.                                                                       !
  1122)             !------------------------------------------------------------------------------!
  1123)             if (abs(deriv) < tolerfl8) exit newloop
  1124)             !------------------------------------------------------------------------------!
  1125) 
  1126) 
  1127)             !----- Copy the previous guess. -----------------------------------------------!
  1128)             cia   = ciz
  1129)             funa  = fun
  1130)             !------------------------------------------------------------------------------!
  1131) 
  1132) 
  1133)             !----- Update guess. ----------------------------------------------------------!
  1134)             ciz      = cia - fun/deriv
  1135)             !------------------------------------------------------------------------------!
  1136) 
  1137) 
  1138)             !------------------------------------------------------------------------------!
  1139)             !     Check whether the method converged.                                      !
  1140)             !------------------------------------------------------------------------------!
  1141)             converged = 2.d0 * abs(cia-ciz) < tolerfl8 * (abs(cia)+abs(ciz))
  1142)             !------------------------------------------------------------------------------!
  1143) 
  1144)  
  1145)             !------------------------------------------------------------------------------!
  1146)             !    At this point we test the current status of the guess.                    !
  1147)             !------------------------------------------------------------------------------!
  1148)             if (ciz < cimin .or. ciz > cimax) then
  1149)                !----- This guess went off-bounds, we give up on Newton's. -----------------!
  1150)                exit newloop
  1151)             elseif (converged) then
  1152)                !----- Converged, find the root as the mid-point. --------------------------!
  1153)                ci  = 5.d-1 * (cia+ciz)
  1154)                exit newloop
  1155)             else
  1156)                !----- Not there yet, update the function evaluation and the derivative. ---!
  1157)                call iter_solver_step(.true.,ciz,fun,deriv)
  1158)                if (fun == 0.d0) then 
  1159)                   !----- We have actually hit the jackpot, the answer is ciz. -------------!
  1160)                   ci        = ciz
  1161)                   converged = .true.
  1162)                   exit newloop
  1163)                end if
  1164)             end if
  1165)             !------------------------------------------------------------------------------!
  1166)          end do newloop
  1167)       end if
  1168)       !------------------------------------------------------------------------------------!
  1169) 
  1170) 
  1171) 
  1172)       !------------------------------------------------------------------------------------!
  1173)       if (.not. converged) then 
  1174) 
  1175)          !---------------------------------------------------------------------------------!
  1176)          !     If we have reached this point it means that Newton's method failed.  We     !
  1177)          ! switch to the Regula Falsi instead.  The first step is to find two solutions of !
  1178)          ! the opposite side.                                                              !
  1179)          !---------------------------------------------------------------------------------!
  1180)          if (ciz < cimin .or. ciz > cimax) then
  1181)             !----- The guess is outside the range, discard it and start over. -------------!
  1182)             cia      = sqrt(cimin*cimax)
  1183)             call iter_solver_step(.false.,cia,funa,deriv)
  1184)             zside    = .false.
  1185)          else if (funa * fun < 0.d0) then
  1186)             funz     = fun
  1187)             zside    = .true.
  1188)          else
  1189)             zside    = .false.
  1190)          end if
  1191)          !---------------------------------------------------------------------------------!
  1192) 
  1193) 
  1194) 
  1195)          !---------------------------------------------------------------------------------!
  1196)          !     If we still don't have a good second guess, look for one.                   !
  1197)          !---------------------------------------------------------------------------------!
  1198)          if (.not. zside) then
  1199)             !----- Find the extrapolation term to try to hit the other side. --------------!
  1200)             delta = 1.d-2 * min(abs(cimax-cia),abs(cimin-cia),abs(cimax-cimin))
  1201)             !------------------------------------------------------------------------------!
  1202) 
  1203) 
  1204) 
  1205)             !----- First attempt. ---------------------------------------------------------!
  1206)             ciz   = cia + delta
  1207)             zside = .false.
  1208)             !------------------------------------------------------------------------------!
  1209) 
  1210) 
  1211)             !------------------------------------------------------------------------------!
  1212)             !     Second guess seeker loop.                                                !
  1213)             !------------------------------------------------------------------------------!
  1214)             hitmin = .false.
  1215)             hitmax = .false.
  1216)             itb    = 0
  1217)             zgssloop: do
  1218)                itb = itb + 1
  1219)                if (hitmin .and. hitmax) then
  1220)                   !------------------------------------------------------------------------!
  1221)                   !     We searched through the entire range of ci, and we couldn't find   !
  1222)                   ! any pair of roots of the opposite sign, it's likely that there is no   !
  1223)                   ! solution, so we give up.                                               !
  1224)                   !------------------------------------------------------------------------!
  1225)                   return
  1226)                end if
  1227) 
  1228)                ciz = cia + dble((-1)**itb * (itb+3)/2) * delta
  1229)                if (ciz < cimin) then
  1230)                    !-----------------------------------------------------------------------!
  1231)                    !    We have hit the minimum.  Force it to be the minimum, and make the !
  1232)                    ! hitmin flag true.                                                     !
  1233)                    !-----------------------------------------------------------------------!
  1234)                    ciz    = cimin
  1235)                    hitmin = .true.
  1236)                elseif (ciz > cimax) then
  1237)                    ciz    = cimax
  1238)                    hitmax = .true.
  1239)                end if
  1240)                !---------------------------------------------------------------------------!
  1241)                !     Compute the function evaluate and check signs.                        !
  1242)                !---------------------------------------------------------------------------!
  1243)                call iter_solver_step(.false.,ciz,funz,deriv)
  1244)                zside = funa*funz < 0.d0
  1245)                if (zside) exit zgssloop
  1246)             end do zgssloop
  1247)             !------------------------------------------------------------------------------!
  1248)          end if
  1249)          !---------------------------------------------------------------------------------!
  1250) 
  1251) 
  1252)          !---------------------------------------------------------------------------------!
  1253)          !     If we have reached this points, it means that there is a root that hasn't   !
  1254)          ! been found yet, but at least we know that it is between cia and ciz.  Use the   !
  1255)          ! modified Regula Falsi (Illinois) method to find it.                             !
  1256)          !---------------------------------------------------------------------------------!
  1257)          fpoloop: do itb=itn,maxfpofl
  1258)             ci = (funz * cia - funa * ciz) / (funz-funa)
  1259) 
  1260)             !------------------------------------------------------------------------------!
  1261)             !     Now that we updated the guess, check whether they are really close.      !
  1262)             ! In case they are, this means that it converged, we can use this as our root. !
  1263)             !------------------------------------------------------------------------------!
  1264)             converged = 2.d0 * abs(ci - cia) < tolerfl8 * (abs(cia)+abs(ciz))
  1265)             if (converged) then
  1266)                exit fpoloop
  1267)             end if
  1268)             !------------------------------------------------------------------------------!
  1269) 
  1270) 
  1271)             !----- Find the new function evaluation. --------------------------------------!
  1272)             call iter_solver_step(.false.,ci,fun,deriv)
  1273) 
  1274) 
  1275)             !------ Define the new interval based on the intermediate value theorem. ------!
  1276)             if (fun == 0.d0) then
  1277)                !----- We have actually hit the jackpot, the answer is ciz. ----------------!
  1278)                converged = .true.
  1279)                exit fpoloop
  1280)             elseif (fun*funa < 0.d0 ) then
  1281)                ciz = ci
  1282)                funz  = fun
  1283)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  1284)                if (zside) funa=funa * 5.d-1
  1285)                !----- We just updated zside, set zside to true. ---------------------------!
  1286)                zside = .true.
  1287)             else
  1288)                cia = ci
  1289)                funa   = fun
  1290)                !----- If we are updating aside again, modify aside (Illinois method) ------!
  1291)                if (.not. zside) funz=funz * 5.d-1
  1292)                !----- We just updated aside, set zside to false. --------------------------!
  1293)                zside = .false.
  1294)             end if
  1295)          end do fpoloop
  1296)          !---------------------------------------------------------------------------------!
  1297)       end if
  1298)       !------------------------------------------------------------------------------------!
  1299) 
  1300) 
  1301) 
  1302) 
  1303) 
  1304)       !------------------------------------------------------------------------------------!
  1305)       !     In case it converged, we can compute the answer.                               !
  1306)       !------------------------------------------------------------------------------------!
  1307)       if (converged) then
  1308)          !----- 1. Intercellular CO2, we just utilise the answer we've just got. ----------!
  1309)          answer%lint_co2      = ci
  1310)          !----- 3. Compute the CO2 demand. ------------------------------------------------!
  1311)          answer%co2_demand    = calc_co2_demand(answer%lint_co2)
  1312)          !----- 4. Compute the actual stomatal conductance of water and CO2. --------------!
  1313)          answer%stom_cond_h2o = calc_stom_cond_h2o(answer%lint_co2,answer%co2_demand)
  1314)          answer%stom_cond_co2 = gsw_2_gsc8 * answer%stom_cond_h2o
  1315)          !----- 5. Compute the leaf surface CO2. ------------------------------------------!
  1316)          answer%lsfc_co2  = met(ib)%can_co2 - answer%co2_demand / met(ib)%blyr_cond_co2
  1317)          !----- 6. Compute the leaf surface vapour mixing ratio. --------------------------!
  1318)          answer%lsfc_shv  = ( answer%stom_cond_h2o * met(ib)%lint_shv                      &
  1319)                             + met(ib)%blyr_cond_h2o    * met(ib)%can_shv  )                &
  1320)                           / ( met(ib)%blyr_cond_h2o + answer%stom_cond_h2o)
  1321)          !---------------------------------------------------------------------------------!
  1322)       end if
  1323)       !------------------------------------------------------------------------------------!
  1324) 
  1325)       return
  1326)    end subroutine solve_iterative_case

  1344)    subroutine set_co2_demand_params(whichlim)
  1345)       use c34constants   , only : aparms       & ! intent(inout)

  1400)

  1404)                aparms(ib)%sigma = 5.d-1 * aparms(ib)%vm
  1405)                aparms(ib)%xi    = 0.d0
  1406)                aparms(ib)%tau   = 1.d0
  1407)                aparms(ib)%nu    = -aparms(ib)%leaf_resp
  1408) 
  1409)             end select
  1410)          case (4)
  1411)             !------------------------------------------------------------------------------!
  1412)             !     C4 case.  There are three possibilities, the light-limited, the Rubisco- !

  1416)             case ('LIGHT')
  1417)                !----- Light-limited case. -------------------------------------------------!
  1418)                aparms(ib)%rho   = 0.d0
  1419)                aparms(ib)%sigma = aparms(ib)%alpha * met(ib)%par

  1427)                aparms(ib)%sigma = aparms(ib)%vm
  1428)                aparms(ib)%xi    = 0.d0

  1430)                aparms(ib)%nu    = -aparms(ib)%leaf_resp
  1431) 
  1432)             case ('CO2')
  1433)                !----- CO2-limited for low CO2 concentration case. -------------------------!
  1434)                aparms(ib)%rho   = klowco28 * aparms(ib)%vm
  1435)                aparms(ib)%sigma = 0.d0
  1436)                aparms(ib)%xi    = 0.d0
  1437)                aparms(ib)%tau   = 1.d0
  1438)                aparms(ib)%nu    = -aparms(ib)%leaf_resp

  1442)          end select

  1454)    !     This subroutine finds the updated function evaluation and derivative for the      !
  1455)    ! iterative step for Newton's (or Regula Falsi) method.  The "function" here is a       !
  1456)    ! combination of the definition of the water stomatal conductance for open stomata      !
  1457)    ! (F96's equation 13), after substituting Ds by a combination of M09's equations B13    !
  1458)    ! and B16 so water demand is eliminated(Psi_open), and incorporating F96's equation 14  !
  1459)    ! to eliminate the surface carbon.  This function has the property of having one root   !
  1460)    ! that corresponds to the intercellular CO2 concentration.                              !
  1461)    !  The logical variable is used to decide whether to compute the derivatives or not.    !
  1462)    ! They are necessary only when it is a Newton's method call.                            !
  1463)    !---------------------------------------------------------------------------------------!
  1464)    subroutine iter_solver_step(newton,lint_co2,fun,deriv)
  1465)       use c34constants, only : thispft & ! intent(in)
  1466)                              , aparms  & ! intent(in)
  1467)                              , met     ! ! intent(in)
  1468)       !$ use omp_lib
  1469)       
  1470)       implicit none
  1471)       !----- Arguments. -------------------------------------------------------------------!
  1472)       logical     , intent(in)  :: newton              ! Newton's method step  [       T|F]
  1473)       real(kind=8), intent(in)  :: lint_co2            ! Intercell. CO2 conc.  [   mol/mol]
  1474)       real(kind=8), intent(out) :: fun                 ! Function evaluation   [       ---]
  1475)       real(kind=8), intent(out) :: deriv               ! Derivative.           [   mol/mol]
  1476)       !----- Local variables. -------------------------------------------------------------!
  1477)       real(kind=8)              :: stom_cond_h2o       ! Water conductance     [  mol/m�/s]
  1478)       real(kind=8)              :: co2_demand          ! CO2 demand            [  mol/m�/s]
  1479)       real(kind=8)              :: co2_demand_prime    ! Deriv. of CO2 demand  [1/mol/m�/s]
  1480)       real(kind=8)              :: stom_cond_h2o_prime ! Deriv. of H2O cond.   [1/mol/m�/s]
  1481)       real(kind=8)              :: efun1               ! 1st term
  1482)       real(kind=8)              :: efun2               ! 2nd term
  1483)       real(kind=8)              :: efun3               ! 3rd term
  1484)       real(kind=8)              :: eprime1             ! Derivative of the 1st term
  1485)       real(kind=8)              :: eprime2             ! Derivative of the 2nd term
  1486)       real(kind=8)              :: eprime3             ! Derivative of the 3rd term
  1487)       integer                   :: ib
  1488)       !------------------------------------------------------------------------------------!
  1489)       
  1490)       ib = 1
  1491)       !$ ib = OMP_get_thread_num()+1
  1492) 
  1493)       !----- Find the CO2 demand. ---------------------------------------------------------!
  1494)       co2_demand       = calc_co2_demand(lint_co2)
  1495)       !----- Find the stomatal conductance of water. --------------------------------------!
  1496)       stom_cond_h2o    = calc_stom_cond_h2o(lint_co2,co2_demand)
  1497)       !------------------------------------------------------------------------------------!
  1498) 
  1499) 
  1500) 
  1501)       !------------------------------------------------------------------------------------!
  1502)       !     Find the function components, then the function evaluation.                    !
  1503)       !------------------------------------------------------------------------------------!
  1504)       efun1 = (stom_cond_h2o - thispft(ib)%b) / (thispft(ib)%m * co2_demand)
  1505)       efun2 = (met(ib)%can_co2 - aparms(ib)%compp - co2_demand/met(ib)%blyr_cond_co2)
  1506)       efun3 = 1.d0 + ( met(ib)%blyr_cond_h2o * (met(ib)%lint_shv - met(ib)%can_shv)        &
  1507)                      / (thispft(ib)%d0 * (met(ib)%blyr_cond_h2o + stom_cond_h2o)))
  1508)       fun   = efun1 * efun2 * efun3 - 1.d0
  1509)       !------------------------------------------------------------------------------------!
  1510) 
  1511) 
  1512) 
  1513)       !------------------------------------------------------------------------------------!
  1514)       !     In case this is a Newton's step, we must also compute the derivatives.  Other- !
  1515)       ! wise, we assign any number, but it should not be used.                             !
  1516)       !------------------------------------------------------------------------------------!
  1517)       if (newton) then
  1518)          !----- CO2 demand. ---------------------------------------------------------------!
  1519)          co2_demand_prime    = calc_co2_demand_prime(lint_co2,co2_demand)
  1520)          !----- stomatal conductance of water. --------------------------------------------!
  1521)          stom_cond_h2o_prime = calc_stom_cond_h2o_prime(lint_co2,stom_cond_h2o,co2_demand  &
  1522)                                                        ,co2_demand_prime)
  1523)          !----- Function components. ------------------------------------------------------!
  1524)          eprime1 = ( stom_cond_h2o_prime * co2_demand                                      &
  1525)                    - co2_demand_prime * (stom_cond_h2o - thispft(ib)%b) )                      &
  1526)                  / (thispft(ib)%m * co2_demand * co2_demand)
  1527) 
  1528)          eprime2 = - co2_demand_prime / met(ib)%blyr_cond_co2
  1529) 
  1530)          eprime3 = - (efun3 - 1.d0) *  stom_cond_h2o_prime                                 &
  1531)                  / ( met(ib)%blyr_cond_h2o + stom_cond_h2o )
  1532) 
  1533)          deriv   = eprime1 * efun2   * efun3                                               &
  1534)                  + efun1   * eprime2 * efun3                                               &
  1535)                  + efun1   * efun2   * eprime3
  1536)       else
  1537)          deriv = 0.d0
  1538)       end if
  1539)       !------------------------------------------------------------------------------------!
  1540) 
  1541)       return
  1542)    end subroutine iter_solver_step

  1685)    !     This sub-routine computes the minimum and maximum intercellular carbon dioxide    !
  1686)    ! concentration that we should seek the solution.  Both the function from which we seek !
  1687)    ! the root and the stomatal conductance function have singularities, so we don't want   !
  1688)    ! the intercellular carbon dioxide to cross these singularities because the root-find-  !
  1689)    ! ing method assumes continuity.                                                        !
  1690)    !---------------------------------------------------------------------------------------!
  1691)    subroutine find_lint_co2_bounds(cimin,cimax,bounded)
  1692)       use c34constants   , only : thispft           & ! intent(in)
  1693)                                 , aparms            & ! intent(in)
  1694)                                 , met               ! ! intent(in)
  1695)       use physiology_coms, only : gbw_2_gbc8        & ! intent(in)
  1696)                                 , c34smin_lint_co28 ! ! intent(in)
  1697)       !$ use omp_lib
  1698) 
  1699)       implicit none
  1700)       !----- Arguments. -------------------------------------------------------------------!
  1701)       real(kind=8), intent(out)  :: cimin   ! Minimum intercellular CO2         [  mol/mol]
  1702)       real(kind=8), intent(out)  :: cimax   ! Maximum intercellular CO2         [  mol/mol]
  1703)       logical     , intent(out)  :: bounded ! This problem is bounded           [      T|F]
  1704)       !----- Local variables. -------------------------------------------------------------!
  1705)       real(kind=8)               :: aquad   ! Quadratic coefficient             [      ---]
  1706)       real(kind=8)               :: bquad   ! Linear coefficient                [  mol/mol]
  1707)       real(kind=8)               :: cquad   ! Intercept                         [mol�/mol�]
  1708)       real(kind=8)               :: ciAo    ! Ci at singularity where Aopen=0   [  mol/mol]
  1709)       real(kind=8)               :: cigsw   ! Ci at sing. where gbc(ca-ci)=Ao   [  mol/mol]
  1710)       real(kind=8)               :: ciQ     ! Ci at singularity where qi-qs=-D0 [  mol/mol]
  1711)       real(kind=8)               :: xtmp    ! variable for ciQ equation
  1712)       real(kind=8)               :: ytmp    ! variable for ciQ
  1713)       real(kind=8)               :: ztmp    ! variable for ciQ
  1714)       real(kind=8)               :: wtmp    ! variable for ciQ
  1715)       real(kind=8)               :: discr   ! The discriminant of the quad. eq. [mol�/mol�]
  1716)       real(kind=8)               :: ciroot1 ! 1st root for the quadratic eqn.   [  mol/mol]
  1717)       real(kind=8)               :: ciroot2 ! 2nd root for the quadratic eqn.   [  mol/mol]
  1718)       integer                    :: ib
  1719)       !------------------------------------------------------------------------------------!
  1720)       ib = 1
  1721)       !$ ib = OMP_get_thread_num()+1
  1722) 
  1723) 
  1724)       !------------------------------------------------------------------------------------!
  1725)       ! First case: This check will find when Aopen goes to 0., which causes a singularity !
  1726)       ! in the function of which we are looking for a root.                                !
  1727)       !------------------------------------------------------------------------------------!
  1728)       ciAo = - (aparms(ib)%tau * aparms(ib)%nu + aparms(ib)%sigma)                         &
  1729)            / (aparms(ib)%xi  * aparms(ib)%nu + aparms(ib)%rho  )
  1730)       !------------------------------------------------------------------------------------!
  1731) 
  1732) 
  1733) 
  1734)       !------------------------------------------------------------------------------------!
  1735)       ! Second case: This check finds which ci causes the terms [gbc (ca -ci) - Aopen] to  !
  1736)       ! be 0.   This will cause a singularity in the gsw function.                         !
  1737)       !------------------------------------------------------------------------------------!
  1738)       !----- 1. Define the coefficients for the quadratic equation. -----------------------!
  1739)       aquad = met(ib)%blyr_cond_co2 * aparms(ib)%xi
  1740)       bquad = aparms(ib)%xi * (aparms(ib)%nu - met(ib)%blyr_cond_co2 * met(ib)%can_co2 )   &
  1741)             + met(ib)%blyr_cond_co2 * aparms(ib)%tau + aparms(ib)%rho
  1742)       cquad = aparms(ib)%tau * (aparms(ib)%nu - met(ib)%blyr_cond_co2 * met(ib)%can_co2 )  &
  1743)             + aparms(ib)%sigma
  1744)       !----- 2. Decide whether this is a true quadratic case or not. ----------------------!
  1745)       if (aquad /= 0.d0) then
  1746)          !---------------------------------------------------------------------------------!
  1747)          !     This is a true quadratic case, the first step is to find the discriminant.  !
  1748)          !---------------------------------------------------------------------------------!
  1749)          discr = bquad * bquad - 4.d0 * aquad * cquad
  1750)          if (discr == 0.d0) then
  1751)             !------------------------------------------------------------------------------!
  1752)             !      Discriminant is zero, both roots are the same.  We save only one, and   !
  1753)             ! make the other negative, which will make the guess discarded.                !
  1754)             !------------------------------------------------------------------------------!
  1755)             ciroot1      = - bquad / (2.d0 * aquad)
  1756)             ciroot2      = - discard
  1757)          elseif (discr > 0.d0) then
  1758)             ciroot1 = (- bquad + sqrt(discr)) / (2.d0 * aquad)
  1759)             ciroot2 = (- bquad - sqrt(discr)) / (2.d0 * aquad)
  1760)          else
  1761)             !----- Discriminant is negative.  Impossible to solve. ------------------------!
  1762)             ciroot1      = - discard
  1763)             ciroot2      = - discard
  1764)          end if
  1765)       else
  1766)          !---------------------------------------------------------------------------------!
  1767)          !    This is a linear case, the xi term is zero.  There is only one number that   !
  1768)          ! works for this case.                                                            !
  1769)          !---------------------------------------------------------------------------------!
  1770)          ciroot1      = - cquad / bquad
  1771)          ciroot2      = - discard
  1772)          !----- Not used, just for the debugging process. ---------------------------------!
  1773)          discr        = bquad * bquad
  1774)       end if
  1775)       !------------------------------------------------------------------------------------!
  1776)       !     Save the largest of the values.  In case both were discarded, we switch it to  !
  1777)       ! the positive discard so this will never be chosen.                                 !
  1778)       !------------------------------------------------------------------------------------!
  1779)       cigsw=max(ciroot1, ciroot2)
  1780)       if (cigsw == -discard) cigsw = discard
  1781)       !------------------------------------------------------------------------------------!
  1782)

  1787)       ! Third case: This will find the intercellular CO2 mixing ratio that causes the      !
  1788)       ! qi-qs term to be equal to -D0, which also creates a singularity in the stomatal    !
  1789)       ! conductance.                                                                       !
  1790)       !------------------------------------------------------------------------------------!
  1791)       !----- 1. Find some auxiliary variables. --------------------------------------------!
  1792)       xtmp = met(ib)%blyr_cond_h2o                                                         &

  1795)       ztmp = xtmp * gbw_2_gbc8 * met(ib)%blyr_cond_co2
  1796)       wtmp = ztmp * met(ib)%can_co2 - ytmp * aparms(ib)%nu
  1797) 
  1798)       !----- 2. Define the coefficients for the quadratic equation. -----------------------!
  1799)       aquad = ztmp * aparms(ib)%xi
  1800)       bquad = ytmp * aparms(ib)%rho - aparms(ib)%xi * wtmp + ztmp * aparms(ib)%tau
  1801)       cquad = - aparms(ib)%tau * wtmp + ytmp * aparms(ib)%sigma
  1802) 
  1803)       !----- 3. Decide whether this is a true quadratic case or not. ----------------------!
  1804)       if (aquad /= 0.d0) then

  1806)          !     This is a true quadratic case, the first step is to find the discriminant.  !
  1807)          !---------------------------------------------------------------------------------!

  1812)             ! make the other negative, which will make the guess discarded.                !
  1813)             !------------------------------------------------------------------------------!
  1814)             ciroot1 = - bquad / (2.d0 * aquad)
  1815)             ciroot2 = -discard

  1818)             ciroot2 = (- bquad - sqrt(discr)) / (2.d0 * aquad)
  1819)          else

  1821)             ciroot1      = -discard
  1822)             ciroot2      = -discard
  1823)          end if
  1824)       else

  1827)          ! that works for this case.                                                       !
  1828)          !---------------------------------------------------------------------------------!
  1829)          ciroot1 = - cquad / bquad
  1830)          ciroot2 = -discard
  1831)          !----- Not used, just for the debugging process. ---------------------------------!
  1832)          discr = bquad * bquad
  1833)       end if

  1836)       ! the positive discard so this will never be chosen.                                 !

  1841)

  1843)       !------------------------------------------------------------------------------------!

  1845)       ! tests, we know that cimin is the one associated with the case in which Aopen goes  !
  1846)       ! to zero, and the maximum is the minimum between gsw case, q case, and the canopy   !
  1847)       ! air CO2.                                                                           !
  1848)       !------------------------------------------------------------------------------------!
  1849)       ciroot1 = max(c34smin_lint_co28,ciAo)
  1850)       ciroot2 = min(ciQ, cigsw,met(ib)%can_co2)
  1851)       !------------------------------------------------------------------------------------!
  1852) 
  1853) 
  1854) 
  1855)       !------------------------------------------------------------------------------------!
  1856)       !     The actual bounds are slightly squeezed so the edges will not be at the the    !
  1857)       ! singularities.                                                                     !
  1858)       !------------------------------------------------------------------------------------!
  1859)       cimin = ciroot1 + 1.d-5 * (ciroot2 - ciroot1)
  1860)       cimax = ciroot2 - 1.d-5 * (ciroot2 - ciroot1)
  1861)       !------------------------------------------------------------------------------------!
  1862) 
  1863) 
  1864) 
  1865)       !------------------------------------------------------------------------------------!
  1866)       !    In the last part we make sure that the two guesses are positively defined and   !
  1867)       ! make sense.  If both bounds are non-positive, or if the minimum ci is greater than !
  1868)       ! the maximum cimax, the bounds make no sense, and the solution is not bounded.      !
  1869)       !------------------------------------------------------------------------------------!
  1870)       if (cimin > cimax) then
  1871)          cimin = c34smin_lint_co28
  1872)          cimax = c34smin_lint_co28
  1873)          bounded = .false.
  1874)       elseif (cimin <= 0.d0 .and. cimax <= c34smin_lint_co28) then
  1875)          cimin = c34smin_lint_co28
  1876)          cimax = c34smin_lint_co28
  1877)          bounded = .false.
  1878)       elseif (cimin <= 0.d0) then
  1879)          cimin = c34smin_lint_co28
  1880)          bounded = .true.
  1881)       else
  1882)          bounded = .true.
  1883)       end if
  1884) 
  1885)       return
