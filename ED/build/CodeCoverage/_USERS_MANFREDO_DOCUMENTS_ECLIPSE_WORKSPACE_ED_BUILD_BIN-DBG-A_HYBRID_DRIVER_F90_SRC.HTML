     1) module hybrid_driver
     2)   contains
     3) 
     4) !=============================================================================!
     5) !=============================================================================!
     6) !     This subroutine is the main driver for the Forward/Backward (FB)        !
     7) !     Euler integration scheme.                                               !
     8) !-----------------------------------------------------------------------------!
     9) subroutine hybrid_timestep(cgrid)
    10)   use rk4_integ_utils
    11)   use soil_respiration_module
    12)   use photosyn_driv
    13)   use rk4_misc
    14)   use update_derived_props_module
    15)   use rk4_coms              , only : integration_vars   & ! structure
    16)                                    , rk4patchtype       & ! structure
    17)                                    , zero_rk4_patch     & ! subroutine
    18)                                    , zero_rk4_cohort    & ! subroutine
    19)                                    , zero_bdf2_patch    &
    20)                                    , integration_buff   & ! intent(out)
    21)                                    , rk4site            & ! intent(out)
    22)                                    , bdf2patchtype      &
    23)                                    , tbeg               &
    24)                                    , tend               &
    25)                                    , dtrk4              &
    26)                                    , dtrk4i
    27)   use rk4_driver            , only : initp2modelp
    28)   use ed_state_vars         , only : edtype             & ! structure
    29)                                    , polygontype        & ! structure
    30)                                    , sitetype           & ! structure
    31)                                    , patchtype          ! ! structure
    32)   use met_driver_coms       , only : met_driv_state     ! ! structure
    33)   use grid_coms             , only : nzg                & ! intent(in)
    34)                                    , nzs                ! ! intent(in)
    35)   use ed_misc_coms          , only : current_time       & ! intent(in)
    36)                                    , dtlsm              ! ! intent(in)
    37)   use therm_lib             , only : tq2enthalpy        ! ! function
    38)   use budget_utils          , only : update_budget      & ! function
    39)                                    , compute_budget     ! ! function
    40) 
    41) !$  use omp_lib
    42) 
    43)   implicit none
    44)   !----- Arguments ----------------------------------------------------------!
    45)   type(edtype)             , target      :: cgrid
    46)   !----- Local variables ----------------------------------------------------!
    47)   type(polygontype)        , pointer     :: cpoly
    48)   type(sitetype)           , pointer     :: csite
    49)   type(patchtype)          , pointer     :: cpatch
    50)   type(met_driv_state)     , pointer     :: cmet
    51)   type(rk4patchtype)       , pointer     :: initp
    52)   type(rk4patchtype)       , pointer     :: dinitp
    53)   type(rk4patchtype)       , pointer     :: ytemp
    54)   type(bdf2patchtype)      , pointer     :: yprev
    55)   integer                                :: ipy
    56)   integer                                :: isi
    57)   integer                                :: ipa
    58)   integer                                :: ico
    59)   integer                                :: imon
    60)   integer                                :: nsteps
    61)   real                                   :: patch_vels
    62)   real                                   :: thetaatm
    63)   real                                   :: thetacan
    64)   real                                   :: rasveg
    65)   real                                   :: storage_decay
    66)   real                                   :: leaf_flux
    67)   real                                   :: veg_tai
    68)   real                                   :: wcurr_loss2atm
    69)   real                                   :: ecurr_loss2atm
    70)   real                                   :: co2curr_loss2atm
    71)   real                                   :: ecurr_netrad
    72)   real                                   :: wcurr_loss2drainage
    73)   real                                   :: ecurr_loss2drainage
    74)   real                                   :: wcurr_loss2runoff
    75)   real                                   :: ecurr_loss2runoff
    76)   real                                   :: old_can_theiv
    77)   real                                   :: old_can_shv
    78)   real                                   :: old_can_co2
    79)   real                                   :: old_can_rhos
    80)   real                                   :: old_can_temp
    81)   real                                   :: old_can_prss
    82)   real                                   :: old_can_enthalpy
    83)   real                                   :: fm
    84)   real                                   :: wtime0
    85)   real(kind=8)                           :: hbeg
    86)   integer                                :: ibuff
    87)   !----- Local constants. -----------------------------------------------!
    88)   logical                  , parameter   :: test_energy_sanity = .false.
    89)   !----- External functions. --------------------------------------------!
    90)   real, external                         :: walltime
    91)   !----------------------------------------------------------------------!
    92) 
    93) !- Assigning some constants which will remain the same throughout      !
    94) !      the run.                                                    ----!
    95) !!  if (first_time) then
    96) !!     first_time = .false.
    97) !!     tbeg   = 0.d0
    98) !!     tend   = dble(dtlsm)
    99) !!     dtrk4  = tend - tbeg
   100) !!     dtrk4i = 1.d0/dtrk4
   101) !!  end if
   102) 
   103)   polyloop: do ipy = 1,cgrid%npolygons
   104)      cpoly => cgrid%polygon(ipy)
   105) 
   106)      wtime0=walltime(0.)
   107) 
   108)      siteloop: do isi = 1,cpoly%nsites
   109)         csite => cpoly%site(isi)
   110)         cmet  => cpoly%met(isi)
   111) 
   112)         !---------------------------------------------------------------------!
   113)         !     Update the monthly rainfall.                                    !
   114)         !---------------------------------------------------------------------!
   115)         imon                             = current_time%month
   116)         cpoly%avg_monthly_pcpg(imon,isi) = cpoly%avg_monthly_pcpg(imon,isi)   &
   117)                                          + cmet%pcpg * dtlsm
   118)         !---------------------------------------------------------------------!
   119) 
   120)         call copy_met_2_rk4site(nzg,cmet%atm_ustar,cmet%atm_theiv,         &
   121)              cmet%atm_vpdef      &
   122)              ,cmet%atm_theta,cmet%atm_tmp,cmet%atm_shv   &
   123)              ,cmet%atm_co2,cmet%geoht,cmet%exner         &
   124)              ,cmet%pcpg,cmet%qpcpg,cmet%dpcpg,cmet%prss  &
   125)              ,cmet%rshort,cmet%rlong,cmet%par_beam       &
   126)              ,cmet%par_diffuse,cmet%nir_beam             &
   127)              ,cmet%nir_diffuse,cmet%geoht                &
   128)              ,cpoly%lsl(isi),cpoly%ntext_soil(:,isi)     &
   129)              ,cpoly%green_leaf_factor(:,isi)             &
   130)              ,cgrid%lon(ipy),cgrid%lat(ipy)              &
   131)              ,cgrid%cosz(ipy))
   132) 
   133) 
   134) 
   135)         !$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE( &
   136)         !$OMP initp,ytemp,dinitp,yprev,hbeg,nsteps,ibuff,&
   137)         !$OMP patch_vels,old_can_shv,                    &
   138)         !$OMP old_can_co2,old_can_rhos,old_can_temp,     &
   139)         !$OMP old_can_prss,old_can_enthalpy,             &
   140)         !$OMP wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm,&
   141)         !$OMP co2curr_loss2atm,wcurr_loss2drainage,      &
   142)         !$OMP ecurr_loss2drainage,wcurr_loss2runoff,     &
   143)         !$OMP ecurr_loss2runoff,cpatch)
   144) 
   145)         patchloop: do ipa = 1,csite%npatches
   146)            cpatch => csite%patch(ipa)
   147) 
   148)            ibuff = 1
   149)            !$ ibuff = OMP_get_thread_num()+1
   150) 
   151)            initp => integration_buff(ibuff)%initp
   152)            ytemp => integration_buff(ibuff)%ytemp
   153)            dinitp => integration_buff(ibuff)%dinitp
   154)            yprev  => integration_buff(ibuff)%yprev
   155) 
   156)            !----- Reset all buffers to zero, as a safety measure. ------------!
   157)            call zero_rk4_patch(initp)
   158)            call zero_rk4_patch(ytemp)
   159)            call zero_rk4_patch(dinitp)
   160)            call zero_bdf2_patch(yprev)
   161) 
   162)            call zero_rk4_cohort(initp)
   163)            call zero_rk4_cohort(ytemp)
   164)            call zero_rk4_cohort(dinitp)
   165) 
   166)            !----- Get velocity for aerodynamic resistance. -------------------!
   167)            if (csite%can_theta(ipa) < cmet%atm_theta) then
   168)               patch_vels = cmet%vels_stab
   169)               cmet%vels  = cmet%vels_stab
   170)            else
   171)               patch_vels = cmet%vels_unstab
   172)               cmet%vels  = cmet%vels_unstab
   173)            end if
   174) 
   175)            !------------------------------------------------------------------!
   176) 
   177)            !------------------------------------------------------------------!
   178)            !    Update roughness and canopy depth.                            !
   179)            !------------------------------------------------------------------!
   180)            call update_patch_thermo_props(csite,ipa,ipa,nzg,nzs,&
   181)                 cpoly%ntext_soil(:,isi))
   182)            call update_patch_derived_props(csite,ipa)
   183)            !------------------------------------------------------------------!
   184) 
   185)            !----- Save the previous thermodynamic state. ---------------------!
   186)            old_can_shv      = csite%can_shv(ipa)
   187)            old_can_co2      = csite%can_co2(ipa)
   188)            old_can_rhos     = csite%can_rhos(ipa)
   189)            old_can_temp     = csite%can_temp(ipa)
   190)            old_can_prss     = csite%can_prss (ipa)
   191)            old_can_enthalpy = tq2enthalpy(csite%can_temp(ipa)                 &
   192)                                          ,csite%can_shv(ipa),.true.)
   193)            !------------------------------------------------------------------!
   194) 
   195) 
   196)            !----- Compute current storage terms. -----------------------------!
   197)            call update_budget(csite,cpoly%lsl(isi),ipa)
   198)            !------------------------------------------------------------------!
   199) 
   200) 
   201) 
   202)            !------------------------------------------------------------------!
   203)            !      Test whether temperature and energy are reasonable.         !
   204)            !------------------------------------------------------------------!
   205)            if (test_energy_sanity) then
   206)               call sanity_check_veg_energy(csite,ipa)
   207)            end if
   208)            !------------------------------------------------------------------!
   209) 
   210) 
   211) 
   212)            !------------------------------------------------------------------!
   213)            !     Set up the integration patch.                                !
   214)            !------------------------------------------------------------------!
   215)            call copy_patch_init(csite,ipa,initp,patch_vels)
   216) 
   217)            !------------------------------------------------------------------!
   218)            !     Set up the buffer for the previous step's leaf temperature   !
   219)            !------------------------------------------------------------------!
   220)            call copy_bdf2_prev(csite,ipa,yprev)
   221) 
   222)            !----- Get photosynthesis, stomatal conductance,
   223)            !                                    and transpiration. -----------!
   224)            call canopy_photosynthesis(csite,cmet,nzg,ipa,                     &
   225)                 cpoly%ntext_soil(:,isi),cpoly%leaf_aging_factor(:,isi),       &
   226)                 cpoly%green_leaf_factor(:,isi))
   227) 
   228)             !----- Compute root and heterotrophic respiration. ----------------!
   229)            call soil_respiration(csite,ipa,nzg,cpoly%ntext_soil(:,isi))
   230) 
   231)            !------------------------------------------------------------------!
   232)            ! Set up the remaining, carbon-dependent variables to the buffer.  !
   233)            !------------------------------------------------------------------!
   234)            call copy_patch_init_carbon(csite,ipa,initp)
   235) 
   236)            !------------------------------------------------------------------!
   237)            !  Perform the forward and backward step.  It is possible this will!
   238)            !  be done over a series of sub-steps.  1)derivs,2)forward,3)back  !
   239)            !  4) check stability and error 5) repeat as shorter or continue   !
   240)            !------------------------------------------------------------------!
   241) !            call integrate_patch_hybrid(csite,                                 &
   242) !                 integration_buff(ibuff)%yprev,integration_buff(ibuff)%initp,                &
   243) !                 integration_buff(ibuff)%dinitp,integration_buff(ibuff)%ytemp,               &
   244) !                 ipa,wcurr_loss2atm,                                           &
   245) !                 ecurr_loss2atm,co2curr_loss2atm,wcurr_loss2drainage,          &
   246) !                 ecurr_loss2drainage,wcurr_loss2runoff,                        &
   247) !                 ecurr_loss2runoff,ecurr_netrad,nsteps)
   248) 
   249) 
   250) 
   251) 
   252)             !--------------------------------------------------------------------------!
   253)             ! Initial step size.  Experience has shown that giving this too large a    !
   254)             ! value causes the integrator to fail (e.g., soil layers become            !
   255)             ! supersaturated).                                                         !
   256)             !--------------------------------------------------------------------------!
   257)             hbeg = dble(csite%htry(ipa))
   258) 
   259)             !--------------------------------------------------------------------------!
   260)             ! Zero the canopy-atmosphere flux values.  These values are updated        !
   261)             ! every dtlsm, so they must be zeroed at each call.                        !
   262)             !--------------------------------------------------------------------------!
   263)             initp%upwp = 0.d0
   264)             initp%tpwp = 0.d0
   265)             initp%qpwp = 0.d0
   266)             initp%cpwp = 0.d0
   267)             initp%wpwp = 0.d0
   268) 
   269)             !----- Go into the ODE integrator using Euler. ----------------------------!
   270) 
   271)             call hybrid_integ(hbeg,csite,yprev,initp,dinitp,                         &
   272)                  ytemp,ipa,isi,nsteps)
   273) 
   274)             !--------------------------------------------------------------------------!
   275)             !  Normalize canopy-atmosphere flux values.  These values are updated ever !
   276)             ! dtlsm, so they must be normalized every time.                            !
   277)             !--------------------------------------------------------------------------!
   278)             initp%upwp = initp%can_rhos * initp%upwp * dtrk4i
   279)             initp%tpwp = initp%can_rhos * initp%tpwp * dtrk4i
   280)             initp%qpwp = initp%can_rhos * initp%qpwp * dtrk4i
   281)             initp%cpwp = initp%can_rhos * initp%cpwp * dtrk4i
   282)             initp%wpwp = initp%can_rhos * initp%wpwp * dtrk4i
   283) 
   284)             !--------------------------------------------------------------------------!
   285)             ! Move the state variables from the integrated patch to the model patch.   !
   286)             !--------------------------------------------------------------------------!
   287)             call initp2modelp(tend-tbeg,initp,csite,ipa,cpoly%nighttime(isi)           &
   288)                              ,wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm               &
   289)                              ,co2curr_loss2atm,wcurr_loss2drainage,ecurr_loss2drainage &
   290)                              ,wcurr_loss2runoff,ecurr_loss2runoff)
   291) 
   292) 
   293)             !----- Add the number of steps into the step counter. -------------!
   294)             cgrid%workload(13,ipy) = cgrid%workload(13,ipy) + real(nsteps)
   295) 
   296)             !------------------------------------------------------------------!
   297)             !    Update the minimum monthly temperature,                       !
   298)             !    based on canopy temperature.                                  !
   299)             !------------------------------------------------------------------!
   300)             if (cpoly%site(isi)%can_temp(ipa) < cpoly%min_monthly_temp(isi)) then
   301)                cpoly%min_monthly_temp(isi) = cpoly%site(isi)%can_temp(ipa)
   302)             end if
   303) 
   304)             !------------------------------------------------------------------!
   305)             !     Compute the residuals.                                       !
   306)             !------------------------------------------------------------------!
   307) 
   308)             call compute_budget(csite,cpoly%lsl(isi),cmet%pcpg,cmet%qpcpg,ipa       &
   309)                  ,wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm                 &
   310)                  ,co2curr_loss2atm,wcurr_loss2drainage,ecurr_loss2drainage   &
   311)                  ,wcurr_loss2runoff,ecurr_loss2runoff,cpoly%area(isi)        &
   312)                  ,cgrid%cbudget_nep(ipy),old_can_enthalpy,old_can_shv        &
   313)                  ,old_can_co2,old_can_rhos,old_can_prss)
   314) 
   315)          end do patchloop
   316) 
   317)          !$OMP END PARALLEL DO
   318) 
   319) 
   320)          !-------------------------------------------------------------------!
   321)       end do siteloop
   322)       !----------------------------------------------------------------------!
   323) 
   324)       cgrid%walltime_py(ipy) = cgrid%walltime_py(ipy)+walltime(wtime0)
   325) 
   326)    end do polyloop
   327) 
   328)    return
   329)  end subroutine hybrid_timestep

   336)  !  This subroutine will drive the integration of several ODEs that drive     !
   337)  !  the fast-scale state variables.                                           !
   338)  !----------------------------------------------------------------------------!
   339)  subroutine hybrid_integ(h1,csite,yprev,initp,dinitp,ytemp,ipa,isi,nsteps)
   340)    use rk4_copy_patch
   341)    use rk4_integ_utils
   342)    use rk4_misc
   343)    use ed_state_vars  , only : sitetype               & ! structure
   344)                              , patchtype                ! structure
   345)    use rk4_coms       , only : integration_vars       & ! structure
   346)                              , rk4patchtype           & ! structure
   347)                              , bdf2patchtype           & ! structure
   348)                              , rk4site                & ! intent(in)
   349)                              , print_diags            & ! intent(in)
   350)                              , maxstp                 & ! intent(in)
   351)                              , tbeg                   & ! intent(in)
   352)                              , tend                   & ! intent(in)
   353)                              , dtrk4                  & ! intent(in)
   354)                              , dtrk4i                 & ! intent(in)
   355)                              , tiny_offset            & ! intent(in)
   356)                              , checkbudget            & ! intent(in)
   357)                              , zero_rk4_patch         & ! subroutine
   358)                              , zero_rk4_cohort        & ! subroutine
   359)                              , hmin                   & ! intent(in)
   360)                              , rk4eps                 & ! intent(in)
   361)                              , rk4epsi                & ! intent(in)
   362)                              , safety                 & ! intent(in)
   363)                              , pgrow                  & ! intent(in)
   364)                              , pshrnk                 & ! intent(in)
   365)                              , errcon                 & ! intent(in)
   366)                              , print_detailed         & ! intent(in)
   367)                              , norm_rk4_fluxes        & ! sub-routine
   368)                              , reset_rk4_fluxes       ! ! sub-routine
   369)    use ed_misc_coms   , only : fast_diagnostics       ! ! intent(in)
   370)    use hydrology_coms , only : useRUNOFF              ! ! intent(in)
   371)    use grid_coms      , only : nzg                    & ! intent(in)
   372)                              , nzs                    & ! intent(in)
   373)                              , time                   ! ! intent(in)
   374)    use soil_coms      , only : dslz8                  & ! intent(in)
   375)                              , runoff_time            & ! intent(in)
   376)                              , runoff_time_i          & ! intent(in)
   377)                              , simplerunoff           ! ! intent(in)
   378)    use consts_coms    , only : cliq8                  & ! intent(in)
   379)                              , t3ple8                 & ! intent(in)
   380)                              , tsupercool_liq8            & ! intent(in)
   381)                              , wdnsi8                 ! ! intent(in)
   382)    !$ use omp_lib
   383)    implicit none
   384)    !----- Arguments ----------------------------------------------------------!
   385)    type(sitetype)            , target      :: csite   ! Current site
   386) 
   387)    type(rk4patchtype)        , target      :: initp   ! Current integ. patch
   388)    type(rk4patchtype)        , target      :: dinitp  ! Integration derivative
   389)    type(rk4patchtype)        , target      :: ytemp   ! Patch at n+1
   390)    type(bdf2patchtype)       , target      :: yprev   ! Patch at n-1
   391) 
   392)    integer                   , intent(in)  :: ipa     ! Current patch ID
   393)    integer                   , intent(in)  :: isi     ! Current site ID
   394)    real(kind=8)              , intent(in)  :: h1      ! First guess of delta-t
   395)    integer                   , intent(out) :: nsteps  ! Number of steps taken.
   396)    !----- Local variables ----------------------------------------------------!
   397)    type(patchtype)           , pointer     :: cpatch           ! Current patch
   398)    logical                                 :: restart_step
   399)    logical                                 :: reject_step
   400)    logical                                 :: minstep
   401)    logical                                 :: stuck
   402)    logical                                 :: test_reject
   403)    integer                                 :: i
   404)    integer                                 :: k            ! Format counter
   405)    integer                                 :: ksn          ! # of snow/water
   406)                                                            ! layers
   407)    real(kind=8)                            :: x            ! Elapsed time
   408)    real(kind=8)                            :: xnew         ! Elapsed time + h
   409)    real(kind=8)                            :: newh         ! New time step
   410)                                                            ! suggested
   411)    real(kind=8)                            :: oldh         ! Old time step
   412)    real(kind=8)                            :: h            ! Current delta-t
   413)                                                            ! attempt
   414)    real(kind=8)                            :: htrunc
   415)    real(kind=8)                            :: hnext        ! Next delta-t
   416)    real(kind=8)                            :: hdid         ! delta-t that
   417)                                                            ! worked (???)
   418)    real(kind=8)                            :: qwfree       ! Free water
   419)                                                            ! internal energy
   420)    real(kind=8)                            :: wfreeb       ! Free water
   421)    real(kind=8)                            :: errmax       ! Maximum error
   422)                                                            ! of this step
   423)    real(kind=8)                            :: elaptime     ! Absolute elapsed
   424)                                                            ! time.
   425)    integer                                 :: nsolve       ! Size of a badger
   426) 
   427)    integer                                 :: ibuff
   428)    !----- External function. -------------------------------------------------!
   429)    real                      , external    :: sngloff
   430)    !--------------------------------------------------------------------------!
   431) 
   432)    ibuff = 1
   433)    !$ ibuff = OMP_get_thread_num()+1
   434) 
   435)    !----- Use some aliases for simplicity. -----------------------------------!
   436)    cpatch => csite%patch(ipa)
   437) 
   438)    !--------------------------------------------------------------------------!
   439)    ! Set initial time and stepsize.                                           !
   440)    !--------------------------------------------------------------------------!
   441)    x = tbeg
   442)    h = h1
   443)    if (dtrk4 < 0.d0) h = -h1
   444) 
   445)    !----- Define total elapsed time. -----------------------------------------!
   446)    elaptime = time + x
   447) 
   448)    !--------------------------------------------------------------------------!
   449)    ! Begin timestep loop                                                      !
   450)    !--------------------------------------------------------------------------!
   451)    timesteploop: do i=1,maxstp
   452) 
   453)       !----- Be sure not to overstep -----------------------------------------!
   454)       if((x+h-tend)*(x+h-tbeg) > 0.d0) h=tend-x
   455) 
   456)       reject_step =  .false.
   457)       hstep:   do
   458) 
   459)          call leaf_derivs(initp,dinitp,csite,ipa,h,.true.)
   460) 
   461)          !---------------------------------------------------------------------!
   462)          ! Very simple analysis of derivative.  ie try to reduce drastic
   463)          ! changes in key state variables.
   464)          !---------------------------------------------------------------------!
   465)          call fb_dy_step_trunc(initp,restart_step,csite,ipa,dinitp,h,htrunc)
   466) 
   467)          if (restart_step) then
   468) 
   469)             oldh    = h
   470)             newh    = htrunc
   471)             minstep = (newh == h) .or. newh < hmin
   472) 
   473)             if(minstep)then
   474) 
   475)                call fail_whale()
   476)                write(*,*) "hybrid euler truncation converged"
   477)                print*,htrunc,h
   478)                stop
   479)             end if
   480) 
   481)             !----- Defining next time, and checking if it really added something. !
   482)             h       = max(1.d-1*h, newh)
   483)             xnew    = x + h
   484)             stuck   = xnew == x
   485) 
   486)             cycle
   487)          end if
   488) 
   489)          !--------------------------------------------------------------------!
   490)          !   Copy patch to the temporary structure                            !
   491)          !   Note that this routine also calculates the size of the matrix    !
   492)          !   used in the implicit step.                                       !
   493)          !   nsolve = 1 + n_leaf_cohorts + n_wood_cohorts                     !
   494)          !--------------------------------------------------------------------!
   495)          call copy_fb_patch(initp,ytemp,cpatch)
   496) 
   497)          !--------------------------------------------------------------------!
   498)          !   Integrate the forward step                                       !
   499)          !--------------------------------------------------------------------!
   500)          call inc_fwd_patch(ytemp,dinitp,h,cpatch)
   501) 
   502)          !--------------------------------------------------------------------!
   503)          !   Integrate the implicit/backwards step                            !
   504)          !--------------------------------------------------------------------!
   505)          call bdf2_solver(cpatch,yprev,initp,ytemp,dinitp,h, &
   506)               dble(csite%hprev(ipa)))
   507) 
   508)          !----- Perform a sanity check on canopy,leaf and wood stuff ---------!
   509)          call fb_sanity_check(ytemp,reject_step,csite,ipa,dinitp,h,print_diags)
   510) 
   511)          !---------------------------------------------------------------------------------!
   512)          !     Here we check the error of this step.  Three outcomes are possible:         !
   513)          ! 1.  The updated values make no sense.  Reject step, assign a large error and    !
   514)          !     try again with a smaller time step;                                         !
   515)          ! 2.  The updated values are reasonable, but the error is large.  Reject step and !
   516)          !     try again with a smaller time step;                                         !
   517)          ! 3.  The updated values are reasonable, and the error is small.  Accept step and !
   518)          !     try again with a larger time step.                                          !
   519)          !---------------------------------------------------------------------------------!
   520)          if (reject_step) then
   521)             !------------------------------------------------------------------------------!
   522)             !    If step was already rejected, that means the step had finished premature- !
   523)             ! ly, so we assign a standard large error (10.0).                              !
   524)             !------------------------------------------------------------------------------!
   525)             errmax = 1.d1
   526) 
   527)          else
   528) 
   529)             errmax = 1.d-1
   530) 
   531)          end if
   532) 
   533)          !---------------------------------------------------------------------------------!
   534)          ! 3. If the step failed, then calculate a new shorter step size to try.           !
   535)          !---------------------------------------------------------------------------------!
   536) 
   537)          if (reject_step) then
   538) 
   539)             !----- Defining new step and checking if it can be. ---------------------------!
   540)             oldh    = h
   541)             newh    = safety * h * errmax**pshrnk
   542)             minstep = (newh == h) .or. newh < hmin
   543) 
   544)             !----- Defining next time, and checking if it really added something. ---------!
   545)             h       = max(1.d-1*h, newh)
   546)             xnew    = x + h
   547)             stuck   = xnew == x
   548) 
   549)             !------------------------------------------------------------------------------!
   550)             ! 3a. Here is the moment of truth... If we reached a tiny step and yet the     !
   551)             !     model didn't converge, then we print various values to inform the user   !
   552)             !     and abort the run.  Please, don't hate the messenger.                    !
   553)             !------------------------------------------------------------------------------!
   554) 
   555)             if (minstep .or. stuck ) then
   556) 
   557)                write (unit=*,fmt='(80a)')         ('=',k=1,80)
   558)                write (unit=*,fmt='(a)')           '   STEPSIZE UNDERFLOW IN EULER_INT'
   559)                write (unit=*,fmt='(80a)')         ('-',k=1,80)
   560)                write (unit=*,fmt='(a,1x,f11.6)')   ' + LONGITUDE:     ',rk4site%lon
   561)                write (unit=*,fmt='(a,1x,f11.6)')   ' + LATITUDE:      ',rk4site%lat
   562)                write (unit=*,fmt='(a)')           ' + PATCH INFO:    '
   563)                write (unit=*,fmt='(a,1x,i6)')     '   - NUMBER:      ',ipa
   564)                write (unit=*,fmt='(a,1x,es12.4)') '   - AGE:         ',csite%age(ipa)
   565)                write (unit=*,fmt='(a,1x,i6)')     '   - DIST_TYPE:   ',csite%dist_type(ipa)
   566)                write (unit=*,fmt='(a,1x,l1)')     ' + MINSTEP:       ',minstep
   567)                write (unit=*,fmt='(a,1x,l1)')     ' + STUCK:         ',stuck
   568)                write (unit=*,fmt='(a,1x,es12.4)') ' + ERRMAX:        ',errmax
   569)                write (unit=*,fmt='(a,1x,es12.4)') ' + X:             ',x
   570)                write (unit=*,fmt='(a,1x,es12.4)') ' + H:             ',h
   571)                write (unit=*,fmt='(a,1x,es12.4)') ' + OLDH:          ',oldh
   572)                write (unit=*,fmt='(a,1x,es12.4)') ' + NEWH:          ',newh
   573)                write (unit=*,fmt='(a,1x,es12.4)') ' + SAFETY:        ',safety
   574)                write (unit=*,fmt='(80a)') ('-',k=1,80)
   575)                write (unit=*,fmt='(a)') '   Likely to be a rejected step problem.'
   576)                write (unit=*,fmt='(80a)') ('=',k=1,80)
   577) 
   578)                call fb_sanity_check(ytemp,test_reject,csite,ipa,dinitp,h,.true.)
   579)                call print_sanity_check(ytemp,csite,ipa)
   580)                call print_rk4patch(ytemp, csite,ipa)
   581)             end if
   582) 
   583)          else
   584)             !------------------------------------------------------------------------------!
   585)             ! 3b.  Great, it worked, so now we can advance to the next step.  We just need !
   586)             !      to do some minor adjustments before...                                  !
   587)             !------------------------------------------------------------------------------!
   588) 
   589)             call adjust_veg_properties(ytemp,h,csite,ipa)
   590) 
   591)             !----- ii.  Final update of top soil properties to avoid off-bounds moisture. -!
   592)             call adjust_topsoil_properties(ytemp,h,csite)
   593) 
   594)             !----- ii. Make temporary surface water stable and positively defined. --------!
   595)             call adjust_sfcw_properties(nzg,nzs,ytemp,h,csite)
   596) 
   597)             !----- iii.  Update the diagnostic variables. ---------------------------------!
   598)             call update_diagnostic_vars(ytemp, csite,ipa)
   599)             !------------------------------------------------------------------------------!
   600) 
   601)             !------------------------------------------------------------------------------!
   602)             ! 3c. Set up h for the next time.  And here we can relax h for the next step,  !
   603)             !    and try something faster.                                                 !
   604)             !------------------------------------------------------------------------------!
   605)             if (errmax > errcon) then
   606) 
   607)                hnext = min( (1.0+sqrt(2.0))*h,safety*h*errmax**pgrow)!,60.d0)
   608) 
   609)             else
   610) 
   611)                hnext = min( (1.d0+sqrt(2.d0))*h,5.d0*h)!,60.d0)
   612) 
   613)             endif
   614) 
   615)             hnext = max(2.d0*hmin,hnext)
   616) 
   617)             !------ 3d. Normalise the fluxes if the user wants detailed debugging. --------!
   618)             if (print_detailed) then
   619)                call norm_rk4_fluxes(ytemp,h)
   620)                call print_rk4_state(initp,ytemp,csite,ipa,isi,x,h)
   621)             end if
   622) 
   623)             !----- 3e. Copy the temporary structure to the intermediate state. ------------!
   624)             call copy_initp2prev(initp,yprev,csite%patch(ipa))
   625) 
   626)             call copy_rk4_patch(ytemp, initp,csite%patch(ipa))
   627) 
   628)             !------------------------------------------------------------------------------!
   629)             !    3f. Flush step-by-step fluxes to zero if the user wants detailed          !
   630)             !        debugging.                                                            !
   631)             !------------------------------------------------------------------------------!
   632)             if (print_detailed) then
   633)                call reset_rk4_fluxes(initp)
   634)             end if
   635) 
   636)             !----- 3g. Update time. -------------------------------------------------------!
   637)             csite%hprev(ipa) = sngl(h)
   638)             x        = x + h
   639)             h        = hnext
   640)             elaptime = elaptime + h
   641) 
   642)             exit hstep
   643)          end if
   644)       end do hstep
   645) 
   646) 
   647)       !----- If the integration reached the next step, make some final adjustments --------!
   648)       if((x-tend)*dtrk4 >= 0.d0)then
   649) 
   650)          ksn = initp%nlev_sfcwater
   651) 
   652)          !---------------------------------------------------------------------------------!
   653)          !   Make temporary surface liquid water disappear.  This will not happen          !
   654)          ! immediately, but liquid water will decay with the time scale defined by         !
   655)          ! runoff_time scale. If the time scale is too tiny, then it will be forced to be  !
   656)          ! hdid (no reason to be faster than that).                                        !
   657)          !---------------------------------------------------------------------------------!
   658)          if (simplerunoff .and. ksn >= 1) then
   659) 
   660)             if (initp%sfcwater_mass(ksn)    > 0.d0           .and.                         &
   661)                 initp%sfcwater_fracliq(ksn) > 1.d-1        ) then
   662) 
   663)                wfreeb = min(1.d0,dtrk4*runoff_time_i) * initp%sfcwater_mass(ksn)           &
   664)                       * (initp%sfcwater_fracliq(ksn) - 1.d-1) / 9.d-1
   665) 
   666)                qwfree = wfreeb * cliq8 * (initp%sfcwater_tempk(ksn) - tsupercool_liq8 )
   667) 
   668)                initp%sfcwater_mass(ksn) = initp%sfcwater_mass(ksn)   - wfreeb
   669) 
   670)                initp%sfcwater_depth(ksn) = initp%sfcwater_depth(ksn) - wfreeb * wdnsi8
   671) 
   672)                !----- Recompute the energy removing runoff --------------------------------!
   673)                initp%sfcwater_energy(ksn) = initp%sfcwater_energy(ksn) - qwfree
   674) 
   675)                call adjust_sfcw_properties(nzg,nzs,initp,dtrk4,csite)
   676)                call update_diagnostic_vars(initp,csite,ipa)
   677) 
   678)                !----- Compute runoff for output -------------------------------------------!
   679)                if (fast_diagnostics) then
   680)                   csite%runoff       (ipa) = csite%runoff(ipa)                             &
   681)                                            + sngloff(wfreeb * dtrk4i,tiny_offset)
   682)                   csite%fmean_runoff (ipa) = csite%fmean_runoff(ipa)                       &
   683)                                            + sngloff(wfreeb * dtrk4i,tiny_offset)
   684)                   csite%fmean_qrunoff(ipa) = csite%fmean_qrunoff(ipa)                      &
   685)                                            + sngloff(qwfree * dtrk4i,tiny_offset)
   686)                end if
   687)                if (checkbudget) then
   688)                   initp%wbudget_loss2runoff = initp%wbudget_loss2runoff + wfreeb
   689)                   initp%ebudget_loss2runoff = initp%ebudget_loss2runoff + qwfree
   690)                   initp%wbudget_storage     = initp%wbudget_storage - wfreeb
   691)                   initp%ebudget_storage     = initp%ebudget_storage - qwfree
   692)                end if
   693)             end if
   694)          end if
   695) 
   696)          !------ Update the substep for next time and leave -------------------------------!
   697) 
   698)          csite%htry(ipa)  = sngl(hnext)
   699) 
   700)          call copy_prev2patch(yprev,csite,ipa)
   701) 
   702)          !---------------------------------------------------------------------------------!
   703)          !     Update the average time step.  The square of DTLSM (tend-tbeg) is needed    !
   704)          ! because we will divide this by the time between t0 and t0+frqsum.               !
   705)          !---------------------------------------------------------------------------------!
   706)          csite%fmean_rk4step(ipa) = csite%fmean_rk4step(ipa)                               &
   707)               + sngl((tend-tbeg)*(tend-tbeg))/real(i)
   708)          nsteps = i
   709)          return
   710)       end if
   711) 
   712)       !----- Use hnext as the next substep ------------------------------------------------!
   713)       h = hnext
   714)    end do timesteploop
   715) 
   716)    !----- If it reached this point, that is really bad news... ----------------------------!
   717)    write (unit=*,fmt='(a)') ' ==> Too many steps in routine euler_integ'
   718)    call print_rk4patch(ytemp, csite,ipa)
   719) 
   720)    return
   721)  end subroutine hybrid_integ
