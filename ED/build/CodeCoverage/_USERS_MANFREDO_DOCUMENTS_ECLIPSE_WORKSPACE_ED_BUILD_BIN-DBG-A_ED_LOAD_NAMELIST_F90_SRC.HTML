     3) subroutine read_nl(namelist_name)
     4)    use ename_coms, only : nl              & ! intent(inout)
     5)                         , init_ename_vars ! ! subroutine
     6)   
     7)    implicit none
     8)    !----- Arguments. ----------------------------------------------------------------------!
     9)    character(len=*), intent(in) :: namelist_name
    10)    !----- Local variables. ----------------------------------------------------------------!
    11)    logical                      :: fexists
    12)    !----- Name lists. ---------------------------------------------------------------------!
    13)    namelist /ED_NL/ nl
    14)    !---------------------------------------------------------------------------------------!
    15) 
    16)    !----- Open the namelist file. ---------------------------------------------------------!
    17)    inquire (file=trim(namelist_name),exist=fexists)
    18)    if (.not. fexists) then
    19)       call fatal_error('The namelist file '//trim(namelist_name)//' is missing.'           &
    20)                       ,'read_nl','ed_load_namelist.f90')
    21)    end if
    22) 
    23)    !----- Initialise the name list with absurd, undefined values. -------------------------!
    24)    call init_ename_vars(nl) 
    25)   
    26)    !----- Read grid point and options information from the namelist. ----------------------!
    27)    open (unit=10, status='OLD', file=namelist_name)
    28)    read (unit=10, nml=ED_NL)
    29)    close(unit=10)
    30) 
    31)    return
    32) end subroutine read_nl
