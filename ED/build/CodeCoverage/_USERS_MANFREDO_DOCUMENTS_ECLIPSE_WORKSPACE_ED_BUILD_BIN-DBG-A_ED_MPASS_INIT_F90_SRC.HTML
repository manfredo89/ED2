     5) subroutine ed_masterput_processid(nproc,headnode_num,masterworks,par_run)
     6) 
     7)    use ed_para_coms, only: mainnum,nmachs,machsize,machnum
     8)    use ed_node_coms, only: mynum,nnodetot,sendnum,recvnum,master_num,machs
     9) 
    10)    implicit none
    11)    integer :: headnode_num,nproc
    12) #if defined(RAMS_MPI)
    13)    include 'mpif.h'
    14) #endif
    15)    integer :: nm
    16)    integer :: ierr
    17)    integer :: par_run
    18)    logical :: masterworks
    19) 
    20)    mainnum=headnode_num
    21)    master_num=headnode_num
    22)    nmachs=nproc
    23) 
    24)    if (masterworks) then
    25)      mynum=nmachs+1
    26)      nnodetot=machsize
    27)      sendnum=1
    28)      recvnum=nmachs
    29)    else
    30)      mynum=-1
    31)      nnodetot=nmachs
    32)      sendnum=-1
    33)      recvnum=-1
    34)    end if
    35) 
    36) 
    37)    if (par_run == 0) return
    38) 
    39)    do nm=1,nmachs
    40)       machnum(nm)=nm
    41)       machs(nm)=nm
    42)    enddo
    43) 
    44)    machs(machsize)=0  !Thats me!!
    45) 
    46) #if defined(RAMS_MPI)
    47)    do nm=1,nmachs
    48)      call MPI_Send(mainnum,1,MPI_INTEGER,machnum(nm),311,MPI_COMM_WORLD,ierr)
    49)      call MPI_Send(machnum(nm),1,MPI_INTEGER,machnum(nm),312,MPI_COMM_WORLD,ierr)
    50)      call MPI_Send(nm,1,MPI_INTEGER,machnum(nm),313,MPI_COMM_WORLD,ierr)
    51)      call MPI_Send(nmachs,1,MPI_INTEGER,machnum(nm),314,MPI_COMM_WORLD,ierr)
    52)      call MPI_Send(machnum,nmachs,MPI_INTEGER,machnum(nm),315,MPI_COMM_WORLD,ierr)
    53)      call MPI_Send(machsize,1,MPI_INTEGER,machnum(nm),316,MPI_COMM_WORLD,ierr)
    54)    enddo
    55) #endif
    56) 
    57) 
    58) 
    59)    return

   601)    use ed_para_coms   , only : mainnum           ! ! intent(in)

   623) #if defined(RAMS_MPI)

   627)    integer                      , intent(in)   :: par_run

   629)    integer                                     :: ierr

   632)    integer                                     :: v

   634)

   637)    if (par_run == 0) return

   639) 
   640) #if defined(RAMS_MPI)
   641)    nsize=nformats*max_met_vars
   642) 
   643) 
   644)    !----- First I send the scalars --------------------------------------------------------!
   645)    call MPI_Bcast (nformats,1,MPI_INTEGER,mainnum,MPI_COMM_WORLD,ierr)
   646)    !---------------------------------------------------------------------------------------!
   647) 
   648)    !---------------------------------------------------------------------------------------!
   649)    !    Here we need a MPI Barrier. The master has the variables already allocated, but we !
   650)    ! need the nodes with their structures already allocated before we proceed sending the  !

   694)    use ed_state_vars , only : gdpy         & ! intent(in)
   695)                             , py_off       ! ! intent(in)
   696)    use grid_coms     , only : nnxp         & ! intent(in)
   697)                             , nnyp         & ! intent(in)
   698)                             , ngrids       ! ! intent(in)
   699)    use ed_work_vars  , only : work_v       & ! intent(in)
   700)                             , npolys_run   ! ! intent(in)
   701)    use ed_para_coms  , only : mainnum      & ! intent(in)
   702)                             , nmachs       & ! intent(in)
   703)                             , loadmeth     ! ! intent(in)
   704)    use mem_polygons  , only : n_ed_region  & ! intent(in)
   705)                             , n_poi        ! ! intent(in)
   706)    implicit none
   707) #if defined(RAMS_MPI)
   708)    include 'mpif.h'
   709) #endif
   710)    !----- Local constants. ----------------------------------------------------------------!
   711)    integer                     , parameter   :: nmethods = 3
   712)    !----- Arguments. ----------------------------------------------------------------------!
   713)    integer                     , intent(in)  :: par_run
   714)    logical                     , intent(in)  :: masterworks
   715)    !----- Local variables. ----------------------------------------------------------------!
   716)    integer, dimension(:,:)     , allocatable :: machind
   717)    integer, dimension(:)       , allocatable :: mpolys
   718)    integer, dimension(:)       , allocatable :: moffset
   719)    integer                                   :: ierr
   720)    integer                                   :: npolys
   721)    integer                                   :: ifm
   722)    integer                                   :: ipy
   723)    integer                                   :: imach
   724)    integer                                   :: imeth
   725)    integer                                   :: ibest
   726)    integer                                   :: maxnmachs
   727)    integer                                   :: ntotmachs
   728)    real   , dimension(:,:)     , allocatable :: machload
   729)    real   , dimension(:)       , allocatable :: thiswork
   730)    real   , dimension(:)       , allocatable :: cumwork
   731)    real   , dimension(nmethods)              :: maxload
   732)    real                                      :: totalwork
   733)    !---------------------------------------------------------------------------------------!
   734) 
   735) 
   736)    !---------------------------------------------------------------------------------------!
   737)    !    This is a logical flag to test wheter the master should also simulate polygons.    !
   738)    ! This is true for offline runs, but false for coupled runs.                            !
   739)    !---------------------------------------------------------------------------------------!
   740)    if (masterworks) then
   741)       ntotmachs=nmachs+1
   742)    else
   743)       ntotmachs=nmachs
   744)    end if
   745) 
   746) 
   747)    !---------------------------------------------------------------------------------------!
   748)    !     We currently don't allow POI simulations to be run in parallel.                   !
   749)    !---------------------------------------------------------------------------------------!
   750)    if (par_run == 1 .and. n_poi > 0 ) then
   751)       write (unit=*,fmt='(a)') '--------------------------------------------------------------------'
   752)       write (unit=*,fmt='(a)') '                                                                    '
   753)       write (unit=*,fmt='(a)') '  Dear ED user,                                                     '
   754)       write (unit=*,fmt='(a)') '                                                                    '
   755)       write (unit=*,fmt='(a)') '  Thank you for choosing ED, the Ecosystem Demography Model!        '
   756)       write (unit=*,fmt='(a)') 'We know that choosing a model is more than looking for a            '
   757)       write (unit=*,fmt='(a)') 'numeric solver, it is an investment in your academic career.        '
   758)       write (unit=*,fmt='(a)') 'Therefore, it is our commitment to provide our valuable researchers '
   759)       write (unit=*,fmt='(a)') 'with the very best in ecosystem dynamics modelling and we want to   '
   760)       write (unit=*,fmt='(a)') 'reaffirm that your satisfaction is our number-one priority.         '
   761)       write (unit=*,fmt='(a)') '  Unfortunately, the option of using parallelism under POI is not   '
   762)       write (unit=*,fmt='(a)') 'available yet, and we would like to offer our most sincere apologies'
   763)       write (unit=*,fmt='(a)') 'for any inconvenience that this may have caused. Our team is        '
   764)       write (unit=*,fmt='(a)') 'currently working on implementing this capability, and we hope to   '
   765)       write (unit=*,fmt='(a)') 'achieve this goal in a very near future, thus meeting your needs of '
   766)       write (unit=*,fmt='(a)') 'a fast and reliable model. Meanwhile you may find useful to run the '
   767)       write (unit=*,fmt='(a)') 'SOI runs in serial mode instead.                                    '
   768)       write (unit=*,fmt='(a)') '                                                                    '
   769)       write (unit=*,fmt='(a)') '--------------------------------------------------------------------'
   770)       call fatal_error('Parallel version of POI runs not available.'                       &
   771)                       , 'ed_masterput_poly_dims','ed_mpass_init.F90')
   772)    end if
   773) 
   774) 
   775)    !---------------------------------------------------------------------------------------!
   776)    !     Now we decide how many polygons we send for each node.  If this is a serial       !
   777)    ! (single node) run, then all nodes go to the only node available.  Otherwise, we       !
   778)    ! try to split the nodes evenly.  This doesn't necessarily mean giving each node the    !
   779)    ! same number of polygons, because different environments may require very different    !
   780)    ! number of operations.  So if we have information on how the model has been behaving   !
   781)    ! (from a history start, for example), we use that information to split the work load   !
   782)    ! more evenly.  Otherwise, if this is a near bare ground or an ED-1/ED-2.0 restart run, !
   783)    ! we assume that all polygons are equally expensive.                                    !
   784)    !---------------------------------------------------------------------------------------!
   785)    if (par_run == 0) then
   786)      do ifm=1,n_ed_region
   787)         gdpy(1,ifm)   = npolys_run(ifm)
   788)         py_off(1,ifm) = 0
   789)      end do
   790)      do ifm=n_ed_region+1,ngrids
   791)         gdpy(1,ifm)   = 1
   792)         py_off(1,ifm) = 0
   793)      end do
   794)    else
   795)       !----- Parallel run, split the grids into the nodes. --------------------------------!
   796)       gridloop: do ifm=1,ngrids
   797)          npolys    = npolys_run(ifm)
   798) 
   799)          !---------------------------------------------------------------------------------!
   800)          !     We will now try three different methods to put the nodes together.  The     !
   801)          ! final choice will be the one with the lowest maximum load, because this is the  !
   802)          ! one with the smallest load disparity.  But first we allocate some scratch       !
   803)          ! arrays.                                                                         !
   804)          !---------------------------------------------------------------------------------!
   805)          allocate(thiswork(npolys),cumwork(npolys),machind(npolys,3))
   806)          allocate(machload(ntotmachs,3),mpolys(ntotmachs),moffset(ntotmachs))
   807)          !----- Copy the workload and also find the cumulative sum of workload. -----------!
   808)          call atob(npolys,work_v(ifm)%work,thiswork)
   809)          call atob(npolys,work_v(ifm)%work,cumwork)
   810)          call cumsum(npolys,cumwork)
   811)          totalwork = cumwork(npolys)
   812) 
   813)          !---------------------------------------------------------------------------------!
   814)          !     Worksum is the sum of the relative workload. This will be always less than  !
   815)          ! or equal to the number of polygons.  In case we have no idea on how the         !
   816)          ! polygons behave (if they are slow or fast), then all polygons received workload !
   817)          ! 1., and the total workload will be the same as the number of polygons.  If we   !
   818)          ! do know the distribution from the history file, and the workload is very skew-  !
   819)          ! ed (some polygons are extremely slow compared to the others), worksum will be   !
   820)          ! significantly less than the number of polygons.                                 !
   821)          !---------------------------------------------------------------------------------!
   822)          maxnmachs = ceiling(totalwork)
   823) 
   824)          !---------------------------------------------------------------------------------!
   825)          !     If nmachsmax is less than the number of machines, stop the run.  The MPI    !
   826)          ! process contains way too many nodes, and this would be a waste of resources.    !
   827)          ! At the run crash, give this information to the user...                          !
   828)          !---------------------------------------------------------------------------------!
   829)          if (maxnmachs < ntotmachs .and. loadmeth /= 2) then
   830)             write (unit=*,fmt='(a)') '----------------------------------------------------'
   831)             write (unit=*,fmt='(a)') '       The number of requested nodes exceeds the    '
   832)             write (unit=*,fmt='(a)') ' maximum number of nodes in which you would still   '
   833)             write (unit=*,fmt='(a)') ' take advantage of parallel processing.  This can   '
   834)             write (unit=*,fmt='(a)') ' be because you have requested a number of nodes    '
   835)             write (unit=*,fmt='(a)') ' that exceeds the number of polygons, or because    '
   836)             write (unit=*,fmt='(a)') ' in the past some of your polygons were so much     '
   837)             write (unit=*,fmt='(a)') ' slower than the others that having more nodes      '
   838)             write (unit=*,fmt='(a)') ' wouldn''t help.  Reduce the number of nodes and    '
   839)             write (unit=*,fmt='(a)') ' try again...                                       '
   840)             write (unit=*,fmt='(a)') '----------------------------------------------------'
   841)             write (unit=*,fmt='(a,1x,i6)') ' # of requested slave nodes :',ntotmachs
   842)             write (unit=*,fmt='(a,1x,i6)') ' # of polygons              :',npolys
   843)             write (unit=*,fmt='(a,1x,i6)') ' Max. # of nodes needed     :',maxnmachs
   844)             write (unit=*,fmt='(a)') '----------------------------------------------------'
   845)             call fatal_error('Requested number of nodes exceeds the maximum needed.'       &
   846)                             ,'ed_masterput_poly_dims','ed_mpass_init.F90')
   847)          end if
   848) 
   849) 
   850)          do ipy = 1, npolys
   851)             !------------------------------------------------------------------------------!
   852)             ! Method 1.  Load will be evenly distributed accross the nodes.  The node      !
   853)             !            index will be defined by the next integer of the relative         !
   854)             !            position of the cumulative sum split into ntotmachs categories.   !
   855)             !------------------------------------------------------------------------------!
   856)             machind(ipy,1) = ceiling(cumwork(ipy)*real(ntotmachs)/totalwork)
   857) 
   858)             !------------------------------------------------------------------------------!
   859)             ! Method 2.  We simply split the polygons evenly, without bothering about the  !
   860)             !            workload.                                                         !
   861)             !------------------------------------------------------------------------------!
   862)             machind(ipy,2) = ceiling(real(ipy)*real(ntotmachs)/real(npolys))
   863) 
   864)             !------------------------------------------------------------------------------!
   865)             ! Method 3.  Similar to 1, but here we round the index instead of finding the  !
   866)             !            next integer.  Because of that we use the number of machines      !
   867)             !            minus one.  This is theoretically less efficient, but sometimes   !
   868)             !            it can be less just because a large workload polygon is next to a !
   869)             !            small one such that it would cause an almost empty node next to   !
   870)             !            very busy one.                                                    !
   871)             !------------------------------------------------------------------------------!
   872)             machind(ipy,3) = 1 + nint(cumwork(ipy)*real(ntotmachs-1)/totalwork)
   873)          end do
   874) 
   875)          !---------------------------------------------------------------------------------!
   876)          !    We now play it safe and ensure all indices are within bounds.                !
   877)          !---------------------------------------------------------------------------------!
   878)          where (machind < 1)
   879)             machind = 1
   880)          end where
   881)          where (machind > ntotmachs)
   882)             machind = ntotmachs
   883)          end where
   884) 
   885)          !---------------------------------------------------------------------------------!
   886)          !     We now loop over the machine indices and find the workload that each        !
   887)          ! node will receive, as well as the maximum load that each method got.            !
   888)          !---------------------------------------------------------------------------------!
   889)          machload(:,:) = 0.0
   890)          do imeth = 1, nmethods
   891)             do imach = 1, ntotmachs
   892)                machload(imach,imeth) = sum(thiswork,machind(:,imeth) == imach)
   893)             end do
   894)             maxload(imeth) = maxval(machload(:,imeth))
   895)          end do
   896) 
   897)          !---------------------------------------------------------------------------------!
   898)          !     The best method is the one that gives the lowest maximum load.  The average !
   899)          ! load is actually the same for all the three methods, so the only difference is  !
   900)          ! the range, that's why the smallest maximum is the preferred one.                !
   901)          !---------------------------------------------------------------------------------!
   902)          ibest = minloc(maxload,dim=1)
   903) 
   904)          if(loadmeth>0) ibest = loadmeth
   905) 
   906)          !----- Count how many polygons go to each node. ----------------------------------!
   907)          do imach=1,ntotmachs
   908)             mpolys(imach) = count(machind(:,ibest) == imach)
   909)          end do
   910) 
   911)          !----- Now find the offset. ------------------------------------------------------!
   912)          moffset(1) = 0
   913)          do imach = 2, ntotmachs
   914)             moffset(imach) = moffset(imach-1) + mpolys(imach-1)
   915)          end do
   916) 
   917)          !---------------------------------------------------------------------------------!
   918)          !     We now print a banner so the user can have some idea of the workload        !
   919)          ! distribution.                                                                   !
   920)          !---------------------------------------------------------------------------------!
   921)          write(unit=*,fmt='(a)') '--------------------------------------------------------'
   922)          write(unit=*,fmt='(a)') '                WORKLOAD PER POLYGON'
   923)          write(unit=*,fmt='(a,1x,i6)')     ' - Maxnmachs   : ',maxnmachs
   924)          write(unit=*,fmt='(a,1x,i6)')     ' - NPolys      : ',npolys
   925)          write(unit=*,fmt='(a,1x,es13.6)') ' - Total Work  : ',totalwork
   926)          write(unit=*,fmt='(a)') '--------------------------------------------------------'
   927)          write(unit=*,fmt='(7(a,1x))') '   IPY','     WORKLOAD','       CUMSUM'            &
   928)                                              ,'METH1','METH2','METH3'
   929)          write(unit=*,fmt='(a)') '--------------------------------------------------------'
   930)          do ipy=1,npolys
   931)             write(unit=*,fmt='(i6,1x,2(es13.6,1x),3(i5,1x))')                              &
   932)                                               ipy,thiswork(ipy),cumwork(ipy)               &
   933)                                             , machind(ipy,1),machind(ipy,2),machind(ipy,3)
   934)          end do
   935)          write(unit=*,fmt='(a)') '--------------------------------------------------------'
   936)          !---------------------------------------------------------------------------------!
   937) 
   938)          !---------------------------------------------------------------------------------!
   939)          !     We now print a banner so the user can have some idea of the workload        !
   940)          ! distribution.                                                                   !
   941)          !---------------------------------------------------------------------------------!
   942)          write(unit=*,fmt='(a)') '--------------------------------------------------------'
   943)          write(unit=*,fmt='(a)') '                Node distribution'
   944)          write(unit=*,fmt='(a,1x,i6)') ' - Best method :',ibest
   945)          write(unit=*,fmt='(a)') '--------------------------------------------------------'
   946)          write(unit=*,fmt='(6(a,1x))') ' IMACH','MPOLYS','OFFSET','    MACHLOAD1'          &
   947)                                       ,'    MACHLOAD2','    MACHLOAD3'
   948)          write(unit=*,fmt='(a)') '--------------------------------------------------------'
   949)          do imach=1,ntotmachs
   950)             write(unit=*,fmt='(3(i6,1x),3(es13.6,1x))') imach,mpolys(imach),moffset(imach) &
   951)                                                        ,machload(imach,1)                  &
   952)                                                        ,machload(imach,2)                  &
   953)                                                        ,machload(imach,3)
   954)          end do
   955)          write(unit=*,fmt='(a)') '--------------------------------------------------------'
   956)          !---------------------------------------------------------------------------------!
   957) 
   958) 
   959)          !---------------------------------------------------------------------------------!
   960)          !     Send the number of polygons and offset to the other nodes.                  !
   961)          !---------------------------------------------------------------------------------!
   962)          do imach=1,nmachs
   963)             gdpy  (imach,ifm) = mpolys (imach)
   964)             py_off(imach,ifm) = moffset(imach)
   965) #if defined(RAMS_MPI)
   966)             call MPI_Bcast(mpolys (imach),1,MPI_INTEGER,mainnum,MPI_COMM_WORLD,ierr)
   967)             call MPI_Bcast(moffset(imach),1,MPI_INTEGER,mainnum,MPI_COMM_WORLD,ierr)
   968) #endif
   969)          end do
   970) 
   971)          !----- Getting the message about this node itself. -------------------------------!
   972)          gdpy  (ntotmachs,ifm) = mpolys(ntotmachs)
   973)          py_off(ntotmachs,ifm) = moffset(ntotmachs)
   974) 
   975)          !---------------------------------------------------------------------------------!
   976)          !     Free memory for re-allocation in the next grid (or to quit the subroutine). !
   977)          !---------------------------------------------------------------------------------!
   978)          deallocate(thiswork,cumwork,machind)
   979)          deallocate(machload,mpolys,moffset)
   980)       end do gridloop
   981)    end if
   982) 
   983)    return
   984) end subroutine ed_masterput_poly_dims

  1094)                call MPI_Send(rscratch,npoly,MPI_REAL,machnum(nm),mpiid,MPI_COMM_WORLD      &
  1095)                             ,ierr)
  1096)                mpiid = mpiid + 1
  1097)             end do
  1098) 
  1099)             !----- Deallocate the scratch arrays, as the polygon sizes may change. --------!
  1100)             deallocate(rscratch)

  1102) 
  1103)          end do
  1104)       end do
  1105)    end if
  1106) #endif

  1108)

  1112)    !     Deallocate each of the global vectors, and reallocate (and fill) as local         !
  1113)    ! vectors.                                                                              !

  1117)    do ifm=1,ngrids
  1118)

  1120)       offset = py_off(nm,ifm)
  1121)       ipya   = offset + 1

  1123)

  1126) 
  1127)       sc_work(ifm)%glon              (1:npoly) = work_v(ifm)%glon              (ipya:ipyz)
  1128)       sc_work(ifm)%glat              (1:npoly) = work_v(ifm)%glat              (ipya:ipyz)
