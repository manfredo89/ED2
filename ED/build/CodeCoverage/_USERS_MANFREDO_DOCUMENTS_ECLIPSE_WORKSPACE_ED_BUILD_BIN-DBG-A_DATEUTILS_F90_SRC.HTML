     6)    use consts_coms, only : day_sec,hr_sec,min_sec

    11)

    24)       do iy=firstyear,year1-1
    25)          if (isleap(iy)) then

    27)          else
    28)            elapdays=elapdays + 365
    29)          end if
    30)       end do
    31)    elseif (firstyear > year1) then
    32)       do iy=firstyear-1,year1
    33)          if (isleap(iy)) then
    34)            elapdays=elapdays - 366
    35)          else
    36)            elapdays=elapdays - 365
    37)          end if
    38)       end do
    39)    end if
    40)    
    41)    ndays = elapdays + julday(month1,date1,year1)
    42)    s1= dble(ndays) * dble(day_sec)
    43)    s2= dble(hour1/10000) * dble(hr_sec)
    44)    s3= dble(mod(hour1,10000)/100)*dble(min_sec)
    45)    s4= dble(mod(hour1,100))

    96) subroutine date_secs_ymdt (seconds,iyear1,imonth1,idate1,ihour1)
    97)    use consts_coms, only : day_sec,hr_sec,min_sec
    98)    implicit none
    99)    real(kind=8) :: seconds,s1
   100)    integer :: iyear1,imonth1,idate1,ihour1
   101) 
   102)    integer,parameter :: firstyear=1000
   103)    logical, external :: isleap
   104) 
   105)    ! compute real time given number of seconds past 1 January 1000 12:00 am  
   106) 
   107)    integer :: ny,nyr,ileap,nm,ihr,imn,isc
   108) 
   109)    integer :: mondays(12)
   110)    data mondays/31,28,31,30,31,30,31,31,30,31,30,31/
   111) 
   112)    ! Get what year it is
   113)    s1=seconds
   114)    do ny=0,10000
   115)       ileap=0
   116)       if(isleap(firstyear+ny)) ileap=1
   117)       s1=s1-(365.d0 + dble(ileap))* dble(day_sec)
   118)       if(s1 < 0.d0) then
   119)          nyr=ny
   120)          s1=s1+(365.d0 +dble(ileap))* dble(day_sec)
   121)          exit
   122)       endif
   123)    enddo
   124)    iyear1=firstyear+nyr
   125) 
   126)    ! s1 is now number of secs into the year
   127)    !   Get month
   128)    do nm=1,12
   129)       ileap=0
   130)       if(isleap(firstyear+ny) .and. nm == 2) ileap=1
   131)       s1=s1-dble(mondays(nm)+ileap)* dble(day_sec)
   132)       if(s1 < 0.d0) then
   133)          s1=s1+(dble(mondays(nm)+ileap))* dble(day_sec)
   134)          exit
   135)       endif
   136)    enddo
   137)    imonth1=nm
   138) 
   139)    ! s1 is now number of secs into the month
   140)    !   Get date and time
   141) 
   142)    idate1=int(s1/dble(day_sec))
   143)    s1=s1-dble(idate1)*dble(day_sec)
   144)    !---- Don't know if this is right. idate1 should not be +1. Although days start
   145)    ! at idate1, idate =0 means that it is still running the last day of previous month,
   146)    ! just that it is past midnight so the previous test gave a negative. 
   147)    ! idate1=idate1+1 ! Since date starts at 1
   148)    if (idate1 == 0) then
   149)      imonth1=imonth1-1
   150)      if (imonth1 == 0) then
   151)        imonth1 = 12
   152)        iyear1  = iyear1 - 1
   153)      end if
   154)      ileap =0
   155)      if (isleap(iyear1) .and. imonth1 == 2) ileap=1
   156)      idate1=mondays(imonth1)+ileap
   157)    end if
   158) 
   159)    ihr=int(s1/dble(hr_sec))
   160)    s1=s1-dble(ihr)*dble(hr_sec)
   161)    imn=int(s1/dble(min_sec))
   162)    s1=s1-dble(imn)*dble(min_sec)
   163)    isc=int(s1)
   164)    ihour1=ihr*10000+imn*100+isc
   165) 
   166)    return
   167) end subroutine date_secs_ymdt

   286) integer function julday1000 (imonth,iday,iyear)
   287)    implicit none
   288)    integer :: imonth,iday,iyear
   289) 
   290)    integer :: i,imm,idd,jd
   291)    
   292)    integer           :: febdays
   293)    logical, external :: isleap
   294) 
   295)    ! compute the julian day (from 1000) from a normal date w/4 digit yr
   296)    ! 1583 is the first full year with Gregorian calendar, but we may want to do historical
   297)    ! runs that predate the calandar so 1000 was chosen instead
   298) 
   299)    julday1000=0
   300)    do i=1000,iyear
   301) 
   302)       imm=12
   303)       idd=31
   304)       if(i==iyear)then
   305)          imm=imonth
   306)          idd=iday
   307)       endif
   308)       
   309)       if (isleap(i)) then
   310)          febdays=29
   311)       else
   312)          febdays=28
   313)       end if
   314) 
   315)       jd= idd  &
   316)          + min(1,max(0,imm-1))*31  &
   317)          + min(1,max(0,imm-2))*febdays  &
   318)          + min(1,max(0,imm-3))*31  &
   319)          + min(1,max(0,imm-4))*30  &
   320)          + min(1,max(0,imm-5))*31  &
   321)          + min(1,max(0,imm-6))*30  &
   322)          + min(1,max(0,imm-7))*31  &
   323)          + min(1,max(0,imm-8))*31  &
   324)          + min(1,max(0,imm-9))*30  &
   325)          + min(1,max(0,imm-10))*31  &
   326)          + min(1,max(0,imm-11))*30  &
   327)          + min(1,max(0,imm-12))*31
   328) 
   329)       julday1000=julday1000+jd

   331)    enddo    
   332)
