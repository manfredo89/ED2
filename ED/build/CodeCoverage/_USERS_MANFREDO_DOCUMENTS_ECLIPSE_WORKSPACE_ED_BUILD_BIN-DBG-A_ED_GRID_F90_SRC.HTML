     3) subroutine ed_gridset(ngra)
     4) 
     5) 
     6)   ! ed_gridset: grid initialization or moving a nested grid.
     7)   !   initialization iff ngra = 1
     8)   !   moving grid "ngra" ow
     9) 
    10)   use grid_coms, only: &
    11)        jdim,           &
    12)        ngrids,         &
    13)        polelat,        &
    14)        polelon,        &
    15)        platn,          &
    16)        plonn,          &
    17)        centlat,        &
    18)        centlon,        &
    19)        deltax,         &
    20)        deltaxn,        &
    21)        deltay,         &
    22)        deltayn,        &
    23)        nstratx,        &
    24)        nstraty,        &
    25)        ninest,         &
    26)        xmn,            &
    27)        xtn,            &
    28)        njnest,         &
    29)        ymn,            &
    30)        ytn,            &
    31)        nxtnest,        &
    32)        nnxp,           &
    33)        nnyp
    34) 
    35) 
    36)   use ed_node_coms,only : mxp,myp
    37) 
    38)   use ed_max_dims, only: &
    39)        maxgrds,        &
    40)        nzpmax
    41) 
    42)   use consts_coms, only: erad
    43) 
    44)   implicit none
    45) 
    46)   integer, intent(in) :: ngra
    47) 
    48)   integer :: ifm
    49)   integer :: icm
    50)   integer :: i
    51)   integer :: j
    52)   integer :: ngrb
    53)   real :: centx1
    54)   real :: centy1
    55)   real :: centx
    56)   real :: centy
    57) 
    58)   ! Grid initialization:
    59) 
    60)   if (ngra == 1) then
    61) 
    62)      ! Set platn plonn arrays, the values of polelat and polelon for each grid
    63) 
    64)      do ifm = 1,ngrids
    65)         icm = nxtnest(ifm)
    66)         if (ifm == 1) then
    67)            platn(ifm) = polelat
    68)            plonn(ifm) = polelon
    69)         else
    70)            platn(ifm) = platn(icm)
    71)            plonn(ifm) = plonn(icm)
    72)         endif
    73)      enddo
    74) 
    75)      ! just for the coarser grid:
    76)      ! set deltax and deltay (deltaxn and deltayn)
    77)      ! find grid center point coordinates at polar stereographic projection
    78)      ! from these, find coordinates of the lower left coarser grid cell
    79)      ! (xmn, ymn) are coordinates of the "higher" cell boundary
    80)      ! these coordinates are required for computing ninest, njnest of nested grids
    81) 
    82)      deltaxn(1) = deltax
    83)      deltayn(1) = deltay
    84)      call ed_ll_xy(centlat(1),centlon(1),platn(1),plonn(1),centx1,centy1)
    85)      xmn(1,1) = centx1 - 0.5 * float(nnxp(1)-2) * deltaxn(1)
    86)      ymn(1,1) = centy1 - 0.5 * float(nnyp(1)-2) * deltayn(1)
    87) 
    88) 
    89)      if (nnyp(1) == 1) ymn(1,1) = centy1
    90) 
    91)      ! for all nested grids:
    92)      ! set deltaxn and deltayn of the nested grid as multiples of deltaxn and
    93)      ! deltayn of the coarser grid.
    94)      ! convert (centlat, centlon) into (ninest, njnest) or use given (ninest, njnest);
    95)      ! with (ninest, njnest), place all interior cells of the nested grid 
    96)      ! fully covering inner cells of the coarser grid. This is guaranteed by
    97)      ! placing the first cell of the nested grid just outside one cell of the
    98)      ! coarser grid and by the fact that the number of inner cells of the nested
    99)      ! grid is a multiple of nstartx and nstraty (as enforced by opspec2)
   100) 
   101)      do ifm = 1,ngrids       ! fine grid number
   102)         icm = nxtnest(ifm)   ! coarse grid number
   103)         if (icm >= 1) then
   104)            deltaxn(ifm) = deltaxn(icm) / float(nstratx(ifm))
   105)            deltayn(ifm) = deltayn(icm) / float(nstraty(ifm))
   106)            if (ninest(ifm) <= 0 .or. njnest(ifm) <= 0)  &
   107)                 call ed_ll_xy(centlat(ifm),centlon(ifm),platn(ifm)  &
   108)                 ,plonn(ifm),centx,centy)
   109)            if (ninest(ifm) <= 0) then
   110)               xmn(1,ifm) = centx - 0.5 * float(nnxp(ifm)-2) * deltaxn(ifm)
   111)               ninest(ifm) = int((xmn(1,ifm) - xmn(1,icm)) / deltaxn(icm) + 1.5)
   112)            endif
   113)            if (njnest(ifm) <= 0) then
   114)               ymn(1,ifm) = centy - 0.5 * float(nnyp(ifm)-2) * deltayn(ifm)
   115)               njnest(ifm) = int((ymn(1,ifm) - ymn(1,icm)) / deltayn(icm) + 1.5)
   116)            endif
   117)            xmn(1,ifm) = xmn(1,icm) + deltaxn(icm) * float(ninest(ifm)-1)
   118)            ymn(1,ifm) = ymn(1,icm) + deltayn(icm) * float(njnest(ifm)-1)
   119)         endif
   120)      enddo
   121)   endif
   122) 
   123)   ! Given the first point of xmn and ymn for the outer
   124)   ! grid, compute remaining outer grid cell locations
   125) 
   126) 
   127)   if (ngra == 1) then
   128)      call ed_newgrid(1)
   129) 
   130)      do i = 2,mxp
   131)         xmn(i,1) = xmn(i-1,1) + deltaxn(1)
   132)      enddo
   133) 
   134)      do j = 2,myp
   135)         ymn(j,1) = ymn(j-1,1) + deltayn(1)
   136)      end do
   137) 
   138)   endif
   139) 
   140)   ! compute xmn and ymn for any required nested grids, and xtn and ytn for
   141)   ! any required grids.
   142) 
   143)   ngrb = ngrids
   144)   if (ngra > 1) ngrb = ngra
   145) 
   146)   do ifm = ngra,ngrb
   147)      call ed_newgrid(ifm)
   148)      icm = nxtnest(ifm)
   149)      if (icm >= 1) then
   150)         xmn(1,ifm) = xmn(ninest(ifm),icm)
   151)         ymn(1,ifm) = ymn(njnest(ifm),icm)
   152)      endif
   153) 
   154)      do i = 2,mxp
   155)         xmn(i,ifm) = xmn(i-1,ifm) + deltaxn(ifm)
   156)         xtn(i,ifm) = .5 * (xmn(i,ifm) + xmn(i-1,ifm))
   157)      enddo
   158)      xtn(1,ifm) = 1.5 * xmn(1,ifm) - .5 * xmn(2,ifm)
   159) 
   160)      if (jdim == 1) then
   161)         do j = 2,myp
   162)            ymn(j,ifm) = ymn(j-1,ifm) + deltayn(ifm)
   163)            ytn(j,ifm) = .5 * (ymn(j,ifm) + ymn(j-1,ifm))
   164)         enddo
   165)         ytn(1,ifm) = 1.5 * ymn(1,ifm) - .5 * ymn(2,ifm)
   166)      else
   167)         ytn(1,ifm) = ymn(1,ifm)
   168)      endif
   169)   enddo
   170) 
   171)   return
   172) end subroutine ed_gridset

   183) subroutine ed_polarst(n2,n3,glat,glon)
   184) 
   185)    use grid_coms, only: xt,yt,platn,plonn,ngrid
   186) 
   187)    implicit none
   188) 
   189)    integer , intent(in)                    :: n2,n3
   190)    real    , intent(out), dimension(n2,n3) :: glat,glon
   191) 
   192)    integer :: i,j
   193) 
   194)    !  Calculates map factors and inverse map factors at u,v,t,m-points and
   195)    !  geographical lat/lon at t-points for a given polar stereographic grid
   196) 
   197)    do j = 1,n3
   198)       do i = 1,n2
   199)          call ed_xy_ll(glat(i,j),glon(i,j),platn(ngrid),plonn(ngrid),xt(i),yt(j))
   200)       enddo
   201)    enddo
   202)

   214) subroutine ed_newgrid(ngr)
   215) 
   216)    use grid_coms, only : ngrid,nxp,nnxp,nyp,nnyp,deltax,deltaxn,xt,xtn,xm,xmn              &
   217)                         ,deltay,deltayn,yt,ytn,ym,ymn
   218)    use ed_node_coms , only : mxp,myp,mmxp,mmyp,ia,mia,iz,miz,ja,mja,jz,mjz,i0,mi0,j0,mj0   &
   219)                             ,ibcon,mibcon
   220) 
   221)    implicit none
   222)    integer :: ngr
   223) 
   224)    integer :: i,j
   225) 
   226)    !     +----------------------------------------------------------------
   227)    !     !    Fill the single and 1D variables that the rest of the model
   228)    !     !      uses from the nest arrays and change grid level in the I/O.
   229)    !     +----------------------------------------------------------------
   230) 
   231)    ngrid = ngr
   232) 
   233)    !         grid point references
   234)    !         x - direction
   235) 
   236)    nxp=nnxp(ngr)
   237)    nyp=nnyp(ngr)
   238)    !
   239)    !          grid spacings
   240) 
   241)    deltax=deltaxn(ngr)
   242)    do i=1,nxp
   243)       xt(i)=xtn(i,ngr)
   244)       xm(i)=xmn(i,ngr)
   245)    enddo
   246)    !
   247)    deltay=deltayn(ngr)
   248)    do j=1,nyp
   249)       yt(j)=ytn(j,ngr)
   250)       ym(j)=ymn(j,ngr)
   251)    enddo
   252)    !
   253)    !        node gridpoint info
   254) 
   255)    mxp=mmxp(ngr)
   256)    myp=mmyp(ngr)
   257) 
   258)    ia=mia(ngr)
   259)    iz=miz(ngr)
   260) 
   261)    ja=mja(ngr)
   262)    jz=mjz(ngr)
   263)   
   264)    i0=mi0(ngr)
   265)    j0=mj0(ngr)
   266)    ibcon=mibcon(ngr)
   267) 
   268)    return
   269) end subroutine ed_newgrid

   280) subroutine ed_xy_ll (qlat,qlon,polelat,polelon,x,y)
   281)   use consts_coms, only: erad,erad2,pio180
   282)   implicit none
   283)   real, intent(out) :: qlat
   284)   real, intent(out) :: qlon
   285)   real, intent(in ) :: polelat
   286)   real, intent(in ) :: polelon
   287)   real, intent(in ) :: x
   288)   real, intent(in ) :: y
   289) 
   290)   real :: sinplat
   291)   real :: cosplat
   292)   real :: sinplon
   293)   real :: cosplon
   294)   real :: x3p
   295)   real :: y3p
   296)   real :: z3p
   297)   real :: z3q
   298)   real :: x3q
   299)   real :: y3q
   300)   real :: xq
   301)   real :: yq
   302)   real :: zq
   303)   real :: t
   304)   real :: d
   305)   real :: alpha
   306)   real :: r3q
   307) 
   308) 
   309)   ! Evaluate sine and cosine of latitude and longitude of pole point p.
   310) 
   311)   sinplat = sin(polelat * pio180)
   312)   cosplat = cos(polelat * pio180)
   313)   sinplon = sin(polelon * pio180)
   314)   cosplon = cos(polelon * pio180)
   315) 
   316)   ! Compute (x3,y3,z3) coordinates of the pole point where the origin is the
   317)   ! center of the earth, the z axis is the north pole, the x axis is the
   318)   ! equator and prime meridian, and the y axis is the equator and 90 E.
   319) 
   320)   x3p = erad * cosplat * cosplon
   321)   y3p = erad * cosplat * sinplon
   322)   z3p = erad * sinplat
   323) 
   324)   ! Compute distance d from given point R on the polar stereographic plane
   325)   ! to the pole point P:
   326) 
   327)   d = sqrt (x ** 2 + y ** 2)
   328) 
   329)   ! Compute angle QCP where C is the center of the Earth.  This is twice
   330)   ! angle QAP where A is the antipodal point.  Angle QAP is the same as
   331)   ! angle RAP:
   332) 
   333)   alpha = 2. * atan2(d,erad2)
   334) 
   335)   ! Compute zq, the height of Q relative to the polar stereographic plane:
   336) 
   337)   zq = erad * (cos(alpha) - 1.)
   338) 
   339)   ! Compute the parameter t which is the the distance ratio AQ:AR
   340) 
   341)   t = (erad2 + zq) / erad2
   342) 
   343)   ! Compute xq and yq, the x and y coordinates of Q in polar stereographic space:
   344) 
   345)   xq = t * x
   346)   yq = t * y
   347) 
   348)   ! Transform location of Q from (x,y,z) coordinates to (x3,y3,z3):
   349) 
   350)   x3q = x3p - xq * sinplon - yq * cosplon * sinplat  &
   351)        + zq * cosplat * cosplon
   352)   y3q = y3p + xq * cosplon - yq * sinplon * sinplat  &
   353)        + zq * cosplat * sinplon
   354)   z3q = z3p + yq * cosplat + zq * sinplat
   355) 
   356)   ! Compute the latitude and longitude of Q:
   357) 
   358)   qlon = atan2(y3q,x3q) / pio180
   359)   r3q = sqrt(x3q ** 2 + y3q ** 2)
   360)   qlat = atan2(z3q,r3q) / pio180
   361) 
   362) end subroutine ed_xy_ll

   373) subroutine ed_ll_xy (qlat,qlon,polelat,polelon,x,y)
   374)   use consts_coms, only : erad,erad2,pio180
   375)   ! ll_xy: projects earth surface point (given by latitude,longitude in degrees)
   376)   !        into a plane tangent to earths surface at a pole point (given by
   377)   !        latitude, longitude in degrees) using a polar stereographic projection;
   378)   !        returns point position in the projected cartesian system (in meters).
   379) 
   380)   implicit none
   381)   real, intent(in ) :: qlat
   382)   real, intent(in ) :: qlon
   383)   real, intent(in ) :: polelat
   384)   real, intent(in ) :: polelon
   385)   real, intent(out) :: x
   386)   real, intent(out) :: y
   387) 
   388)   real :: sinplat
   389)   real :: cosplat
   390)   real :: sinplon
   391)   real :: cosplon
   392)   real :: sinqlat
   393)   real :: cosqlat
   394)   real :: sinqlon
   395)   real :: cosqlon
   396)   real :: x3p
   397)   real :: y3p
   398)   real :: z3p
   399)   real :: z3q
   400)   real :: x3q
   401)   real :: y3q
   402)   real :: xq
   403)   real :: yq
   404)   real :: zq
   405)   real :: t
   406) 
   407) 
   408)   ! Evaluate sine and cosine of latitude and longitude of pole point p and
   409)   ! input point q.
   410) 
   411)   sinplat = sin(polelat * pio180)
   412)   cosplat = cos(polelat * pio180)
   413)   sinplon = sin(polelon * pio180)
   414)   cosplon = cos(polelon * pio180)
   415) 
   416)   sinqlat = sin(qlat * pio180)
   417)   cosqlat = cos(qlat * pio180)
   418)   sinqlon = sin(qlon * pio180)
   419)   cosqlon = cos(qlon * pio180)
   420) 
   421)   ! Compute (x3,y3,z3) coordinates where the origin is the center of the earth,
   422)   ! the z axis is the north pole, the x axis is the equator and prime
   423)   ! meridian, and the y axis is the equator and 90 E.
   424) 
   425)   ! For the pole point, these are:
   426) 
   427)   x3p = erad * cosplat * cosplon
   428)   y3p = erad * cosplat * sinplon
   429)   z3p = erad * sinplat
   430) 
   431)   ! For the given lat,lon point, these are:
   432) 
   433)   z3q = erad * sinqlat
   434)   x3q = erad * cosqlat * cosqlon
   435)   y3q = erad * cosqlat * sinqlon
   436) 
   437)   ! Transform q point from (x3,y3,z3) coordinates in the above system to
   438)   ! polar stereographic coordinates (x,y,z):
   439) 
   440)   xq = - sinplon * (x3q-x3p) + cosplon * (y3q-y3p)
   441)   yq =   cosplat * (z3q-z3p)  &
   442)        - sinplat * ( cosplon * (x3q-x3p) + sinplon * (y3q-y3p) )
   443)   zq =   sinplat * (z3q-z3p)  &
   444)        + cosplat * ( cosplon * (x3q-x3p) + sinplon * (y3q-y3p) )
   445) 
   446)   ! Parametric equation for line from antipodal point at (0,0,-2 erad) to
   447)   ! point q has the following parameter (t) value on the polar stereographic
   448)   ! plane:
   449) 
   450)   t = erad2 / (erad2 + zq)
   451) 
   452)   ! This gives the following x and y coordinates for the projection of point q
   453)   ! onto the polar stereographic plane:
   454) 
   455)   x = xq * t
   456)   y = yq * t
   457) end subroutine ed_ll_xy
