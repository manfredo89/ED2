     1) !==========================================================================================!

     3) !    This file contains the main driver for ED2.                                           !
     4) !------------------------------------------------------------------------------------------!
     5) ! SUBROUTINE: ED_MODEL
     6) !
     7) !> \brief   Begins, updates, and outputs results from ecosystem simulation.
     8) !> \details Coordinates meteorological forcing data, time-stepping, radiation, vegetation
     9) !>          dynamics, hydraulogy, and the HDF5 output subsystem.
    10) !> \author  Translated from ED1 by Ryan Knox and Marcos Longo
    11) !------------------------------------------------------------------------------------------!
    12) subroutine ed_model()
    13)    use ed_met_driver
    14)    use heun_driver
    15)    use euler_driver
    16)    use update_derived_props_module
    17)    use ism_hyd
    18)    use stable_cohorts
    19)    use rk4_integ_utils
    20)    use radiate_driver_module
    21)    use hybrid_driver
    22)    use ed_misc_coms  , only : ivegt_dynamics              & ! intent(in)
    23)                             , integration_scheme          & ! intent(in)
    24)                             , current_time                & ! intent(in)
    25)                             , frqfast                     & ! intent(in)
    26)                             , frqstate                    & ! intent(in)
    27)                             , out_time_fast               & ! intent(in)
    28)                             , dtlsm                       & ! intent(in)
    29)                             , ifoutput                    & ! intent(in)
    30)                             , isoutput                    & ! intent(in)
    31)                             , iqoutput                    & ! intent(in)
    32)                             , itoutput                    & ! intent(in)
    33)                             , frqsum                      & ! intent(in)
    34)                             , unitfast                    & ! intent(in)
    35)                             , unitstate                   & ! intent(in)
    36)                             , imontha                     & ! intent(in)
    37)                             , iyeara                      & ! intent(in)
    38)                             , outstate                    & ! intent(in)
    39)                             , outfast                     & ! intent(in)
    40)                             , nrec_fast                   & ! intent(in)
    41)                             , nrec_state                  & ! intent(in)
    42)                             , runtype                     ! ! intent(in)
    43)    use ed_misc_coms  , only : outputMonth                 & ! intent(in)
    44)                             , writing_dail                & ! intent(in)
    45)                             , writing_mont                & ! intent(in)
    46)                             , writing_dcyc                & ! intent(in)
    47)                             , writing_eorq                & ! intent(in)
    48)                             , writing_long                & ! intent(in)
    49)                             , writing_year                ! ! intent(in)
    50)    use grid_coms     , only : ngrids                      & ! intent(in)
    51)                             , istp                        & ! intent(in)
    52)                             , time                        & ! intent(in)
    53)                             , timmax                      ! ! intent(in)
    54)    use ed_state_vars , only : edgrid_g                    & ! intent(in)
    55)                             , edtype                      & ! intent(in)
    56)                             , patchtype                   & ! intent(in)
    57)                             , filltab_alltypes            & ! intent(in)
    58)                             , filltables                  ! ! intent(in)
    59)    use rk4_driver    , only : rk4_timestep                ! ! intent(in)
    60)    use rk4_coms      , only : integ_err                   & ! intent(in)
    61)                             , integ_lab                   & ! intent(in)
    62)                             , record_err                  & ! intent(inout)
    63)                             , print_detailed              & ! intent(inout)
    64)                             , nerr                        & ! intent(in)
    65)                             , errmax_fout                 & ! intent(in)
    66)                             , sanity_fout                 & ! intent(in)
    67)                             , alloc_integ_err             & ! subroutine
    68)                             , assign_err_label            & ! subroutine
    69)                             , reset_integ_err             ! ! subroutine
    70)    use ed_node_coms  , only : mynum                       & ! intent(in)
    71)                             , nnodetot                    ! ! intent(in)
    72)    use mem_polygons  , only : n_ed_region                 & ! intent(in)
    73)                             , n_poi                       ! ! intent(in)
    74)    use consts_coms   , only : day_sec                     ! ! intent(in)
    75)    use average_utils , only : update_ed_yearly_vars       & ! sub-routine
    76)                             , zero_ed_dmean_vars          & ! sub-routine
    77)                             , zero_ed_mmean_vars          & ! sub-routine
    78)                             , zero_ed_qmean_vars          & ! sub-routine
    79)                             , zero_ed_fmean_vars          & ! sub-routine
    80)                             , integrate_ed_fmean_met_vars & ! sub-routine
    81)                             , zero_ed_yearly_vars         ! ! sub-routine
    82)    implicit none
    83)    !----- Common blocks. ------------------------------------------------------------------!
    84) #if defined(RAMS_MPI)
    85)    include 'mpif.h'
    86) #endif
    87)    !----- Local variables. ----------------------------------------------------------------!
    88)    character(len=28)  :: fmthead
    89)    character(len=32)  :: fmtcntr
    90)    integer            :: ifm
    91)    integer            :: nn
    92)    integer            :: ndays
    93)    logical            :: analysis_time
    94)    logical            :: new_day
    95)    logical            :: new_month
    96)    logical            :: new_year
    97)    logical            :: history_time
    98)    logical            :: dcycle_time
    99)    logical            :: annual_time
   100)    logical            :: mont_analy_time
   101)    logical            :: dail_analy_time
   102)    logical            :: dcyc_analy_time
   103)    logical            :: reset_time
   104)    logical            :: past_one_day
   105)    logical            :: past_one_month
   106)    logical            :: printbanner
   107)    real               :: wtime_start
   108)    real               :: t1
   109)    real               :: wtime1
   110)    real               :: wtime2
   111)    real               :: t2
   112)    real               :: wtime_tot
   113)    !----- Local constants. ----------------------------------------------------------------!
   114)    logical         , parameter :: whos_slow=.false. ! This will print out node numbers
   115)                                                     !    during synchronization, so you
   116)                                                     !    can find out which node is the
   117)                                                     !    slow one
   118)    !----- External functions. -------------------------------------------------------------!
   119)    real    , external :: walltime ! Wall time
   120)    integer , external :: num_days ! Number of days in the current month
   121)    !---------------------------------------------------------------------------------------!
   122) 
   123)    past_one_day   = .false.
   124)    past_one_month = .false.
   125)    filltables     = .false.
   126) 
   127)    !----- Print the hour banner only for regional runs that aren't massively parallel. ----!
   128)    printbanner = n_ed_region > 0 .and. edgrid_g(1)%npolygons > 50 .and. mynum == 1
   129) 
   130) 
   131)    wtime_start=walltime(0.)
   132)    istp = 0
   133) 
   134)    !---------------------------------------------------------------------------------------!
   135)    !     If we are going to record the integrator errors, here is the time to open it for  !
   136)    ! the first time and write the header.  But just before we do it, we check whether this !
   137)    ! is a single POI run, the only case where we will allow this recording.                !
   138)    !---------------------------------------------------------------------------------------!
   139)    record_err     = record_err     .and. n_ed_region == 0 .and. n_poi == 1
   140)    print_detailed = print_detailed .and. n_ed_region == 0 .and. n_poi == 1
   141)    if(record_err) then
   142)       !----- Initialise the error structures. ---------------------------------------------!
   143)       call alloc_integ_err()
   144)       call reset_integ_err()
   145)       call assign_err_label()
   146) 
   147)       !----- Define the formats for both the header and the actual output. ----------------!
   148)       write(fmthead,fmt='(a,i3.3,a)')  '(a4,1x,2(a3,1x),',nerr,'(a13,1x))'
   149)       write(fmtcntr,fmt='(a,i3.3,a)')  '(i4.4,1x,2(i3.2,1x),',nerr,'(i13,1x))'
   150) 
   151)       open  (unit=77,file=trim(errmax_fout),form='formatted',status='replace')
   152)       write (unit=77,fmt=fmthead) 'YEAR','MON','DAY',(integ_lab(nn),nn=1,nerr)
   153)       close (unit=77,status='keep')
   154) 
   155)       open  (unit=78,file=trim(sanity_fout),form='formatted',status='replace')
   156)       write (unit=78,fmt=fmthead) 'YEAR','MON','DAY',(integ_lab(nn),nn=1,nerr)
   157)       close (unit=78,status='keep')
   158)    end if
   159) 
   160)    out_time_fast     = current_time
   161)    out_time_fast%month = -1
   162) 
   163)    !---------------------------------------------------------------------------------------!
   164)    !      If this is not a history restart, then zero out the long term diagnostics.       !
   165)    !---------------------------------------------------------------------------------------!
   166)    if (trim(runtype) /= 'HISTORY') then
   167) 
   168)       do ifm=1,ngrids
   169)          if (writing_long) call zero_ed_dmean_vars(edgrid_g(ifm))
   170)          if (writing_eorq) call zero_ed_mmean_vars(edgrid_g(ifm))
   171)          if (writing_dcyc) call zero_ed_qmean_vars(edgrid_g(ifm))
   172)       end do
   173) 
   174)       !----- Output initial state. --------------------------------------------------------!
   175)       do ifm=1,ngrids
   176)          call update_ed_yearly_vars(edgrid_g(ifm))
   177)       end do
   178)    end if
   179)    !---------------------------------------------------------------------------------------!
   180) 
   181) 
   182)    !---------------------------------------------------------------------------------------!
   183)    !      The fast analysis is always reset, including history runs.                       !
   184)    !---------------------------------------------------------------------------------------!
   185)    do ifm=1,ngrids
   186)       call zero_ed_fmean_vars(edgrid_g(ifm))
   187)    end do
   188)    !---------------------------------------------------------------------------------------!
   189) 
   190) 
   191)    !---------------------------------------------------------------------------------------!
   192)    !    Allocate memory to the integration patch, Euler now utilises the RK4 buffers too.  !
   193)    !---------------------------------------------------------------------------------------!
   194)    call initialize_rk4patches(.true.)
   195)    !---------------------------------------------------------------------------------------!
   196) 
   197)    !---------------------------------------------------------------------------------------!
   198)    !    Initialize some stepping variables.                                                !
   199)    !---------------------------------------------------------------------------------------!
   200)    call initialize_misc_stepvars()
   201)    !---------------------------------------------------------------------------------------!
   202) 
   203) 
   204)    if (ifoutput /= 0) call h5_output('INST')
   205) 
   206)    if (isoutput /= 0) then
   207)       select case (trim(runtype))
   208)       case ('INITIAL')
   209)          call h5_output('HIST')
   210)       case ('HISTORY')
   211)          call h5_output('CONT')
   212)       end select
   213)    end if
   214) 
   215)    if (writing_year ) call h5_output('YEAR')
   216) 
   217)    !----- Start the timesteps. ------------------------------------------------------------!
   218)    if (mynum == 1) write(unit=*,fmt='(a)') ' === Time integration starts (model) ==='
   219) 
   220) 
   221)    timestep: do while (time < timmax)
   222) 
   223)       istp = istp + 1
   224) 
   225)       !------------------------------------------------------------------------------------!
   226)       !   CPU timing information & model timing information.                               !
   227)       !------------------------------------------------------------------------------------!
   228)       call timing(1,t1)
   229)       wtime1=walltime(wtime_start)
   230)       !------------------------------------------------------------------------------------!
   231) 
   232)       if (current_time%time < dtlsm .and. mynum == 1) then
   233)            write (unit=*,fmt='(a,3x,2(i2.2,a),i4.4,a,3(i2.2,a))')                          &
   234)               ' - Simulating:',current_time%month,'/',current_time%date,'/'                &
   235)                               ,current_time%year,' ',current_time%hour,':'                 &
   236)                               ,current_time%min,':',current_time%sec,' UTC'
   237)       end if
   238) 
   239) 
   240)       !----- Define which cohorts are to be solved prognostically. ------------------------!
   241)       do ifm=1,ngrids
   242)          call flag_stable_cohorts(edgrid_g(ifm))
   243)       end do
   244)       !------------------------------------------------------------------------------------!
   245) 
   246) 
   247)       !----- Solve the radiation profile. -------------------------------------------------!
   248)       do ifm=1,ngrids
   249)           call radiate_driver(edgrid_g(ifm))
   250)       end do
   251)       !------------------------------------------------------------------------------------!
   252) 
   253) 
   254) 
   255)       !------------------------------------------------------------------------------------!
   256)       !     At this point, all meteorologic driver data for the land surface model has     !
   257)       ! been updated for the current timestep.  Perform the time average for the output    !
   258)       ! diagnostic.                                                                        !
   259)       !------------------------------------------------------------------------------------!
   260)       do ifm=1,ngrids
   261)          call integrate_ed_fmean_met_vars(edgrid_g(ifm))
   262)       end do
   263)       !------------------------------------------------------------------------------------!
   264) 
   265) 
   266)       !----- Solve the photosynthesis and biophysics. -------------------------------------!
   267)       select case (integration_scheme)
   268)       case (0)
   269)          do ifm=1,ngrids
   270)             call euler_timestep(edgrid_g(ifm))
   271)          end do
   272)       case (1)
   273)          do ifm=1,ngrids
   274)             call rk4_timestep(edgrid_g(ifm))
   275)          end do
   276)       case (2)
   277)          do ifm=1,ngrids
   278)             call heun_timestep(edgrid_g(ifm))
   279)          end do
   280)       case (3)
   281)          do ifm=1,ngrids
   282)             call hybrid_timestep(edgrid_g(ifm))
   283)          end do
   284)       end select
   285)       !------------------------------------------------------------------------------------!
   286) 
   287) 
   288) 
   289)       !------------------------------------------------------------------------------------!
   290)       !     Update the model time.                                                         !
   291)       !------------------------------------------------------------------------------------!
   292)       time=time+dble(dtlsm)
   293)       call update_model_time_dm(current_time, dtlsm)
   294)       !------------------------------------------------------------------------------------!
   295) 
   296) 
   297) 
   298)       !----- Check whether it is some special time... -------------------------------------!
   299)       new_day         = current_time%time < dtlsm
   300)       if (.not. past_one_day .and. new_day) past_one_day=.true.
   301) 
   302)       new_month       = current_time%date == 1  .and. new_day
   303)       if (.not. past_one_month .and. new_month) past_one_month=.true.
   304) 
   305)       new_year        = current_time%month == 1 .and. new_month
   306)       mont_analy_time = new_month .and. writing_mont
   307)       dail_analy_time = new_day   .and. writing_dail
   308)       dcyc_analy_time = new_month .and. writing_dcyc
   309)       reset_time      = mod(time,dble(frqsum)) < dble(dtlsm)
   310)       annual_time     = new_month .and. writing_year .and.                                 &
   311)                         current_time%month == outputMonth
   312) 
   313)       !----- Check whether this is time to write fast analysis output or not. -------------!
   314)       select case (unitfast)
   315)       case (0,1) !----- Now both are in seconds -------------------------------------------!
   316)          analysis_time   = mod(current_time%time, frqfast) < dtlsm .and.                   &
   317)                            (ifoutput /= 0 .or. itoutput /=0)
   318)          dcycle_time     = mod(current_time%time, frqfast) < dtlsm .and. iqoutput /= 0
   319)       case (2)   !----- Months, analysis time is at the new month -------------------------!
   320)          analysis_time   = new_month .and. (ifoutput /= 0 .or. itoutput /=0) .and.         &
   321)                            mod(real(12+current_time%month-imontha),frqfast) == 0.
   322)          dcycle_time     = mod(current_time%time, frqfast) < dtlsm .and. iqoutput /= 0
   323)       case (3) !----- Year, analysis time is at the same month as initial time ------------!
   324)          analysis_time   = new_month .and. (ifoutput /= 0 .or. itoutput /= 0) .and.        &
   325)                            current_time%month == imontha .and.                             &
   326)                            mod(real(current_time%year-iyeara),frqfast) == 0.
   327)          dcycle_time     = mod(current_time%time, frqfast) < dtlsm .and. iqoutput /= 0
   328)       end select
   329) 
   330)       !----- Check whether this is time to write restart output or not. -------------------!
   331)       select case(unitstate)
   332)       case (0,1) !----- Now both are in seconds -------------------------------------------!
   333)          history_time   = mod(current_time%time, frqstate) < dtlsm .and. isoutput /= 0
   334)       case (2)   !----- Months, history time is at the new month --------------------------!
   335)          history_time   = new_month .and. isoutput /= 0 .and.                              &
   336)                           mod(real(12+current_time%month-imontha),frqstate) == 0.
   337)       case (3) !----- Year, history time is at the same month as initial time -------------!
   338)          history_time   = new_month .and. isoutput /= 0 .and.                              &
   339)                           current_time%month == imontha .and.                              &
   340)                           mod(real(current_time%year-iyeara),frqstate) == 0.
   341)       end select
   342) 
   343)       !------------------------------------------------------------------------------------!
   344)       !    Update nrec_fast and nrec_state if it is a new month and outfast/outstate are   !
   345)       ! monthly and frqfast/frqstate are daily or by seconds.                              !
   346)       !------------------------------------------------------------------------------------!
   347)       if (new_month) then
   348)          ndays=num_days(current_time%month,current_time%year)
   349)          if (outfast  == -2.) nrec_fast  = ndays*ceiling(day_sec/frqfast)
   350)          if (outstate == -2.) nrec_state = ndays*ceiling(day_sec/frqstate)
   351)       end if
   352) 
   353)       !----- Check if this is the beginning of a new simulated day. -----------------------!
   354)       if (new_day) then
   355) 
   356)          if (record_err) then
   357) 
   358)             open (unit=77,file=trim(errmax_fout),form='formatted',access='append'          &
   359)                  ,status='old')
   360)             write (unit=77,fmt=fmtcntr) current_time%year,current_time%month               &
   361)                                        ,current_time%date,(integ_err(nn,1),nn=1,nerr)
   362)             close(unit=77,status='keep')
   363) 
   364)             open (unit=78,file=trim(sanity_fout),form='formatted',access='append'          &
   365)                  ,status='old')
   366)             write (unit=78,fmt=fmtcntr) current_time%year,current_time%month               &
   367)                                        ,current_time%date,(integ_err(nn,2),nn=1,nerr)
   368)             close(unit=78,status='keep')
   369) 
   370)             call reset_integ_err()
   371)          end if
   372) 
   373) 
   374)          !---------------------------------------------------------------------------------!
   375)          !     Compute phenology, growth, mortality, recruitment, disturbance, and check   !
   376)          ! whether we will apply them to the ecosystem or not.                             !
   377)          !---------------------------------------------------------------------------------!
   378)          select case (ivegt_dynamics)
   379)          case (0)
   380)             !------------------------------------------------------------------------------!
   381)             !     Dummy vegetation dynamics, we compute the tendencies but we don't really !
   382)             ! apply to the vegetation, so they will remain constant throughout the entire  !
   383)             ! simulation.                                                                  !
   384)             !------------------------------------------------------------------------------!
   385)             call vegetation_dynamics_eq_0(new_month,new_year)
   386)             !------------------------------------------------------------------------------!
   387) 
   388)          case (1)
   389)             !------------------------------------------------------------------------------!
   390)             !     Actual vegetation dynamics, we compute the tendencies and apply to the   !
   391)             ! vegetation.                                                                  !
   392)             !------------------------------------------------------------------------------!
   393)             call vegetation_dynamics(new_month,new_year)
   394)             !------------------------------------------------------------------------------!
   395) 
   396)          end select
   397)          !---------------------------------------------------------------------------------!
   398) 
   399)          !----- First day of a month. -----------------------------------------------------!
   400)          if (new_month) then
   401) 
   402)             !------------------------------------------------------------------------------!
   403)             !      On the monthly timestep we have performed various fusion/fission calls. !
   404)             ! Therefore the var-table's pointer vectors must be updated, and the global    !
   405)             ! definitions of the total numbers must be exported to all nodes.              !
   406)             !      Also, if we do not need to fill the tables until we do I/O, so instead  !
   407)             ! of running this routine every time the demographics change, we set this flag !
   408)             ! and run the routine when the next IO occurs.                                 !
   409)             !------------------------------------------------------------------------------!
   410)             if (nnodetot > 1) then
   411)                if (mynum == 1) write(unit=*,fmt='(a)')                                     &
   412)                                                '-- Monthly node synchronization - waiting'
   413)                if (whos_slow ) then
   414)                   write(unit=*,fmt='(a,1x,i5,1x,a,1x,f7.1)') 'Node',mynum                  &
   415)                                                             ,'time', walltime(wtime_start)
   416)                end if
   417) #if defined(RAMS_MPI)
   418)                call MPI_Barrier(MPI_COMM_WORLD,ierr)
   419) #endif
   420)                if (mynum == 1) write(unit=*,fmt='(a)') '-- Synchronized.'
   421)             end if
   422) 
   423)             filltables=.true.   ! call filltab_alltypes
   424) 
   425)             !----- Read new met driver files only if this is the first timestep. ----------!
   426)             call read_met_drivers()
   427)             !----- Re-allocate integration buffer. ----------------------------------------!
   428)             call initialize_rk4patches(.false.)
   429)          end if
   430)       end if
   431)       !------------------------------------------------------------------------------------!
   432) 
   433) 
   434)       !------------------------------------------------------------------------------------!
   435)       !      Update the yearly variables.                                                  !
   436)       !------------------------------------------------------------------------------------!
   437)       if (analysis_time .and. new_month .and. new_day .and. current_time%month == 6) then
   438)          do ifm = 1,ngrids
   439)             call update_ed_yearly_vars(edgrid_g(ifm))
   440)          end do
   441)       end if
   442)       !------------------------------------------------------------------------------------!
   443) 
   444) 
   445)       !------------------------------------------------------------------------------------!
   446)       !     Call the model output driver.                                                  !
   447)       !------------------------------------------------------------------------------------!
   448)       call ed_output(analysis_time,new_day,dail_analy_time,mont_analy_time,dcyc_analy_time &
   449)                     ,annual_time,history_time,dcycle_time)
   450)       !------------------------------------------------------------------------------------!
   451) 
   452) 
   453) 
   454)       !------------------------------------------------------------------------------------!
   455)       !      Reset time happens every frqsum.  This is to avoid variables to build up when !
   456)       ! history and analysis are off.  This should be done outside ed_output so I have a   !
   457)       ! chance to copy some of these to BRAMS structures.                                  !
   458)       !------------------------------------------------------------------------------------!
   459)       if (reset_time) then
   460)          do ifm=1,ngrids
   461)             call zero_ed_fmean_vars(edgrid_g(ifm))
   462)          end do
   463)       end if
   464)       !------------------------------------------------------------------------------------!
   465) 
   466) 
   467) 
   468)       !------------------------------------------------------------------------------------!
   469)       !      Update the meteorological driver, and the hydrology parameters.               !
   470)       !------------------------------------------------------------------------------------!
   471)       do ifm=1,ngrids
   472)          call update_met_drivers(edgrid_g(ifm))
   473)       end do
   474)       if (new_day .and. new_month) then
   475)          do ifm = 1,ngrids
   476)             call updateHydroParms(edgrid_g(ifm))
   477)          end do
   478)       end if
   479)       !------------------------------------------------------------------------------------!
   480) 
   481) 
   482) 
   483)       !------------------------------------------------------------------------------------!
   484)       !      Update the yearly variables.                                                  !
   485)       !------------------------------------------------------------------------------------!
   486)       !if (analysis_time .and. new_month .and. new_day .and. current_time%month == 6) then
   487)       !   do ifm = 1,ngrids
   488)       !      call update_ed_yearly_vars(edgrid_g(ifm))
   489)       !   end do
   490)       !end if
   491)       !------------------------------------------------------------------------------------!
   492) 
   493) 
   494) 
   495)       !------------------------------------------------------------------------------------!
   496)       !      Update lateral hydrology.                                                     !
   497)       !------------------------------------------------------------------------------------!
   498)       call calcHydroSubsurface()
   499)       call calcHydroSurface()
   500)       call writeHydro()
   501)       !------------------------------------------------------------------------------------!
   502) 
   503) 
   504) 
   505)       !------------------------------------------------------------------------------------!
   506)       !      Update wall time.                                                             !
   507)       !------------------------------------------------------------------------------------!
   508)       wtime2=walltime(wtime_start)
   509)       call timing(2,t2)
   510)       if (printbanner) then
   511)          write (unit=*,fmt='(a,i10,a,i2.2,a,i2.2,a,i4.4,a,f6.0,2(a,f7.3),a)')              &
   512)              ' Timestep ',istp,'; Sim time  '                                              &
   513)             ,current_time%month,'-',current_time%date,'-',current_time%year,' '            &
