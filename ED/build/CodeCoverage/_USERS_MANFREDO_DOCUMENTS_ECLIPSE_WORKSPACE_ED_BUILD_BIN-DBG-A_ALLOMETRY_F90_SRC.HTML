    12)    real function h2dbh(h,ipft)
    13) 
    14)       use pft_coms    , only : is_tropical & ! intent(in)
    15)          , b1Ht        & ! intent(in), lookup table
    16)          , b2Ht        & ! intent(in), lookup table
    17)          , hgt_ref     ! ! intent(in)
    18)       use ed_misc_coms, only : iallom      ! ! intent(in)
    19) 
    20)       !----- Arguments --------------------------------------------------------------------!
    21)       real   , intent(in) :: h
    22)       integer, intent(in) :: ipft
    23)       !------------------------------------------------------------------------------------!
    24) 
    25)       !------------------------------------------------------------------------------------!
    26)       !      Select which type of model we are running.                                    !
    27)       !------------------------------------------------------------------------------------!
    28)       !----- Size- and age-structure (typical ED model). ----------------------------------!
    29)       if (is_tropical(ipft)) then
    30)          select case (iallom)
    31)             case (0,1)
    32)                !----- Default ED-2.1 allometry. -------------------------------------------!
    33)                h2dbh = exp((log(h)-b1Ht(ipft))/b2Ht(ipft))
    34)             case default
    35)                !----- Poorter et al. (2006) allometry. ------------------------------------!
    36)                h2dbh =  ( log(hgt_ref(ipft) / ( hgt_ref(ipft) - h ) ) / b1Ht(ipft) )       &
    37)                   ** ( 1.0 / b2Ht(ipft) )
    38)          end select
    39)       else ! Temperate
    40)          h2dbh = log(1.0-(h-hgt_ref(ipft))/b1Ht(ipft))/b2Ht(ipft)
    41)       end if
    42)       !------------------------------------------------------------------------------------!
    43) 
    44)       return
    45)    end function h2dbh

    56)    real function dbh2h(ipft, dbh)
    57)       use pft_coms    , only : is_tropical & ! intent(in)
    58)          , dbh_crit    & ! intent(in)
    59)          , b1Ht        & ! intent(in)
    60)          , b2Ht        & ! intent(in)
    61)          , hgt_ref     & ! intent(in)
    62)          , hgt_max     ! ! intent(in)
    63)       use ed_misc_coms, only : iallom      & ! intent(in)
    64)          , ibigleaf    ! ! intent(in)
    65)       use ed_state_vars, only:  patchtype  ! ! structure
    66) 
    67)       !----- Arguments --------------------------------------------------------------------!
    68)       integer       , intent(in) :: ipft
    69)       real          , intent(in) :: dbh
    70)       !----- Local variables --------------------------------------------------------------!
    71)       real                :: mdbh
    72)       !------------------------------------------------------------------------------------!
    73) 
    74)       !------------------------------------------------------------------------------------!
    75)       !      Select which type of model we are running.                                    !
    76)       !------------------------------------------------------------------------------------!
    77)       select case (ibigleaf)
    78)          case (0)
    79)             !----- Size- and age-structure (typical ED model). ----------------------------!
    80)             if (is_tropical(ipft)) then
    81)                mdbh = min(dbh,dbh_crit(ipft))
    82)                select case (iallom)
    83)                   case (0,1)
    84)                      !----- Default ED-2.1 allometry. -------------------------------------!
    85)                      dbh2h = exp (b1Ht(ipft) + b2Ht(ipft) * log(mdbh) )
    86)                   case default
    87)                      !----- Poorter et al. (2006) allometry. ------------------------------!
    88)                      dbh2h = hgt_ref(ipft) * (1. - exp(-b1Ht(ipft) * mdbh ** b2Ht(ipft)))
    89)                end select
    90)             else !----- Temperate PFT allometry. ------------------------------------------!
    91)                dbh2h = hgt_ref(ipft) + b1Ht(ipft) * (1.0 - exp(b2Ht(ipft) * dbh))
    92)             end if
    93) 
    94)          case (1)
    95)             !------------------------------------------------------------------------------!
    96)             !     Big-leaf version of ED. DBH is not really meaningful, but in the big-leaf!
    97)             ! model the typical allometry doesn't really make sense so we impose maximum   !
    98)             ! height.                                                                      !
    99)             !------------------------------------------------------------------------------!
   100)             dbh2h = hgt_max(ipft)
   101)          !---------------------------------------------------------------------------------!
   102)       end select
   103)       !------------------------------------------------------------------------------------!
   104) 
   105)       return
   106)    end function dbh2h

   114) 
   115)       use pft_coms    , only : C2B         & ! intent(in)
   116)          , b1Bs_small  & ! intent(in), lookup table
   117)          , b2Bs_small  & ! intent(in), lookup table
   118)          , b1Bs_large  & ! intent(in), lookup table
   119)          , b2Bs_large  & ! intent(in), lookup table
   120)          , is_grass    & ! intent(in)
   121)          , dbh_crit    ! ! intent(in), lookup table
   122)       use ed_misc_coms, only : igrass      ! ! intent(in)
   123) 
   124)       !----- Arguments --------------------------------------------------------------------!
   125)       real   , intent(in) :: dbh
   126)       integer, intent(in) :: ipft
   127)       !------------------------------------------------------------------------------------!

   130)             dbh2bd = 0.0

   144)    !     This function computes the diameter at the breast height given the structural     !
   145)    ! (dead) biomass and the PFT type.  In addition to these inputs, it is also required to !
   146)    ! give a first guess for DBH, which can be the previous value, for example.   This is   !
   147)    ! only used for the new allometry, because we must solve an iterative root-finding      !
   148)    ! method.                                                                               !
   149)    !---------------------------------------------------------------------------------------!
   150)    real function bd2dbh(ipft, bdead)
   151)       use pft_coms    , only : b1Bs_small  & ! intent(in), lookup table
   152)          , b2Bs_small  & ! intent(in), lookup table
   153)          , b1Bs_large  & ! intent(in), lookup table
   154)          , b2Bs_large  & ! intent(in), lookup table
   155)          , bdead_crit  & ! intent(in), lookup table
   156)          , C2B         ! ! intent(in)
   157) 
   158)       !----- Arguments --------------------------------------------------------------------!
   159)       integer, intent(in) :: ipft      ! PFT type                            [         ---]
   160)       real   , intent(in) :: bdead     ! Structural (dead) biomass           [   kgC/plant]
   161)       !------------------------------------------------------------------------------------!
   162) 
   163) 
   164) 
   165)       !------------------------------------------------------------------------------------!
   166)       !    Decide which coefficients to use based on the critical bdead.                   !
   167)       !------------------------------------------------------------------------------------!
   168)          if (bdead <= bdead_crit(ipft)) then
   169)             bd2dbh = (bdead / b1Bs_small(ipft) * C2B)**(1.0/b2Bs_small(ipft))
   170)          else
   171)             bd2dbh = (bdead / b1Bs_large(ipft) * C2B)**(1.0/b2Bs_large(ipft))
   172)       end if
   173)       !------------------------------------------------------------------------------------!
   174) 
   175)       return
   176)    end function bd2dbh

   187)    !      This function computes the maximum leaf biomass [kgC/m2] given the size (either  !
   188)    ! height or DBH).  Trees would always use DBH as the starting point, but grasses may    !
   189)    ! use DBH (old style) or height (new style).  This replaces dbh2bl and h2bl with a      !
   190)    ! single generic function that should be used by all plants.                            !
   191)    !---------------------------------------------------------------------------------------!
   192)    real function size2bl(dbh,hite,ipft)
   193)       use pft_coms    , only : dbh_adult   & ! intent(in), lookup table
   194)          , dbh_crit    & ! intent(in), lookup table
   195)          , C2B         & ! intent(in), lookup table
   196)          , b1Bl_small  & ! intent(in), lookup table
   197)          , b2Bl_small  & ! intent(in), lookup table
   198)          , b1Bl_large  & ! intent(in), lookup table
   199)          , b2Bl_large  & ! intent(in), lookup table
   200)          , is_liana    & ! intent(in)
   201)          , is_grass    ! ! intent(in)
   202)       use ed_misc_coms, only : igrass      ! ! intent(in)
   203)       use ed_state_vars, only: patchtype  ! ! structure
   204) 
   205)       !----- Arguments --------------------------------------------------------------------!
   206)       real          , intent(in) :: dbh
   207)       real          , intent(in) :: hite
   208)       integer       , intent(in) :: ipft
   209)       !----- Local variables --------------------------------------------------------------!
   210)       real                :: mdbh
   211)       real                :: ldbh
   212)       !------------------------------------------------------------------------------------!
   213) 
   214)       !------------------------------------------------------------------------------------!
   215)       ! lianas dbh_crit would be too small for the leaf biomass. We choose ldbh to be the  !
   216)       ! DBH that give lianas the same (~) Bl_max as late successionals                     !
   217)       !------------------------------------------------------------------------------------!
   218)       if (is_liana(ipft)) then
   219)          ldbh = 26.0
   220)       else
   221)          ldbh = dbh_crit(ipft)
   222)       end if
   223)       !------------------------------------------------------------------------------------!
   224) 
   225)       !------------------------------------------------------------------------------------!
   226)       !     Get the DBH, or potential DBH in case of grasses.                              !
   227)       !------------------------------------------------------------------------------------!
   228)       if (is_grass(ipft) .and. igrass == 1) then 
   229)          !---- Use height for new grasses. ------------------------------------------------!
   230)          mdbh   = min(h2dbh(hite,ipft),ldbh)
   231)       else
   232)          !---- Use dbh for trees. ---------------------------------------------------------!
   233)          mdbh   = min(dbh,ldbh)
   234)       end if
   235)       !------------------------------------------------------------------------------------!
   236) 
   237)       !------------------------------------------------------------------------------------!
   238)       !     Find leaf biomass depending on the tree size.                                  !
   239)       !------------------------------------------------------------------------------------!
   240)          if (mdbh < dbh_adult(ipft)) then
   241)             size2bl = b1Bl_small(ipft) / C2B * mdbh ** b2Bl_small(ipft)
   242)          else
   243)             size2bl = b1Bl_large(ipft) / C2B * mdbh ** b2Bl_large(ipft)
   244)       end if
   245)       !------------------------------------------------------------------------------------!
   246) 
   247) 
   248)       return
   249)    end function size2bl

   256)    !      This function determines an effective DBH for grasses given their leaf biomass.  !
   257)    ! DBH has no real meaning for grasses with the new allometry.                           !
   258)    !---------------------------------------------------------------------------------------!
   259)    real function bl2dbh(bleaf,ipft)
   260)       use pft_coms,     only : dbh_crit    & ! intent(in), lookup table
   261)          , hgt_max     & ! intent(in), lookup table
   262)          , is_grass    & ! intent(in)
   263)          , C2B         & ! intent(in)
   264)          , b1Bl_small  & ! intent(in), lookup table
   265)          , b2Bl_small  & ! intent(in), lookup table
   266)          , b1Bl_large  & ! intent(in), lookup table
   267)          , b2Bl_large  & ! intent(in), lookup table
   268)          , bleaf_adult ! ! intent(in), lookup table
   269)       use ed_misc_coms, only : igrass      ! ! intent(in)
   270) 
   271)       !----- Arguments --------------------------------------------------------------------!
   272)       real   , intent(in)      :: bleaf
   273)       integer, intent(in)      :: ipft
   274)       !----- Local variables --------------------------------------------------------------!
   275)       real                :: mdbh
   276)       !------------------------------------------------------------------------------------!
   277) 
   278) 
   279) 
   280)          !----- Find out whether this is an adult tree or a sapling/grass. ----------------!
   281)          if (bleaf < bleaf_adult(ipft)) then
   282)             mdbh = (bleaf * C2B / b1Bl_small(ipft) ) ** (1./b2Bl_small(ipft))
   283)          else
   284)             mdbh = (bleaf * C2B / b1Bl_large(ipft) ) ** (1./b2Bl_large(ipft))
   285)          end if
   286)          !---------------------------------------------------------------------------------!
   287) 
   288) 
   289)          !---------------------------------------------------------------------------------!
   290)          !     For grasses, limit maximum effective dbh by maximum height.                 !
   291)          !---------------------------------------------------------------------------------!
   292)          if (is_grass(ipft) .and. igrass == 1) then
   293)             bl2dbh = min(mdbh, h2dbh(hgt_max(ipft),ipft))
   294)          else
   295)             bl2dbh = min(mdbh, dbh_crit(ipft))
   296)       end if
   297)       !------------------------------------------------------------------------------------!
   298) 
   299) 
   300)       return
   301)    end function bl2dbh

   333)    !    Canopy Area allometry from Dietze and Clark (2008).                                !
   334)    !---------------------------------------------------------------------------------------!
   335)    real function dbh2ca(dbh,hite,sla,ipft)
   336)       use ed_misc_coms, only : iallom      ! ! intent(in)
   337)       use pft_coms    , only : dbh_crit    & ! intent(in)
   338)          , hgt_max     & ! intent(in)
   339)          , is_grass    & ! intent(in)
   340)          , is_liana    & ! intent(in)
   341)          , b1Ca        & ! intent(in)
   342)          , b2Ca        ! ! intent(in)
   343)       use ed_misc_coms, only : igrass      ! ! intent(in)
   344)       use ed_state_vars, only: patchtype  ! ! structure
   345) 
   346)       !----- Arguments --------------------------------------------------------------------!
   347)       real          , intent(in) :: dbh
   348)       real          , intent(in) :: hite
   349)       real          , intent(in) :: sla
   350)       integer       , intent(in) :: ipft
   351)       !----- Internal variables -----------------------------------------------------------!
   352)       real                :: loclai ! The maximum local LAI for a given DBH
   353)       real                :: ldbh
   354)       !------------------------------------------------------------------------------------!
   355) 
   356)       !------------------------------------------------------------------------------------!
   357)       ! lianas dbh_crit would be too small for the leaf biomass. We choose ldbh to be the  !
   358)       ! DBH that give lianas the same (~) Bl_max as late successionals                     !
   359)       !------------------------------------------------------------------------------------!
   360)       if (is_liana(ipft)) then
   361)          ldbh = 26.0
   362)       else
   363)          ldbh = dbh_crit(ipft)
   364)       end if
   365)       !------------------------------------------------------------------------------------!
   366) 
   367)       if (dbh < tiny(1.0)) then
   368)          loclai = 0.0
   369)          dbh2ca = 0.0
   370)       else
   371) 
   372)          !----- make this function generic to size, not just dbh. -------------------------!
   373) 
   374)          loclai = sla * size2bl(dbh,hite,ipft)
   375) 
   376)          select case (iallom)
   377)             case (0)
   378)                !----- No upper bound in the allometry. ------------------------------------!
   379)                dbh2ca = b1Ca(ipft) * dbh ** b2Ca(ipft)
   380) 
   381)             case default
   382)                !----- Impose a maximum crown area. ----------------------------------------!
   383)                if (is_grass(ipft) .and. igrass==1) then
   384)                   dbh2ca = b1Ca(ipft) * min(dbh,h2dbh(hgt_max(ipft),ipft) ) ** b2Ca(ipft)
   385)                else
   386)                   dbh2ca = b1Ca(ipft) * min(dbh, ldbh                     ) ** b2Ca(ipft)
   387)                end if
   388)          end select
   389)       end if
   390) 
   391)       !----- Local LAI / Crown area should never be less than one. ------------------------!
   392)       dbh2ca = min (loclai, dbh2ca)
   393) 
   394)       return
   395)    end function dbh2ca

   431)    integer function dbh2krdepth(hite,dbh,ipft,lsl)
   432)       use ed_misc_coms, only : iallom   ! ! intent(in)
   433)       use grid_coms   , only : nzg      ! ! intent(in)
   434)       use soil_coms   , only : slz      ! ! intent(in)
   435)       use pft_coms    , only : b1Rd     & ! intent(in)
   436)          , b2Rd     ! ! intent(in)
   437) 
   438)       !----- Arguments --------------------------------------------------------------------!
   439)       real   , intent(in) :: hite
   440)       real   , intent(in) :: dbh
   441)       integer, intent(in) :: ipft
   442)       integer, intent(in) :: lsl
   443)       !----- Local variables --------------------------------------------------------------!
   444)       real                :: volume
   445)       real                :: root_depth
   446)       integer             :: k
   447)       !------------------------------------------------------------------------------------!
   448) 
   449) 
   450)       !----- Grasses get a fixed rooting depth of 70 cm. ----------------------------------!
   451)       select case (iallom)
   452)          case (0)
   453)             !------------------------------------------------------------------------------!
   454)             !    Original ED-2.1 (I don't know the source for this equation, though).      !
   455)             !------------------------------------------------------------------------------!
   456)             volume     = dbh2vol(hite,dbh,ipft)
   457)             root_depth = b1Rd(ipft)  * volume ** b2Rd(ipft)
   458) 
   459)          case default
   460)             !------------------------------------------------------------------------------!
   461)             !    This is just a test allometry, that imposes root depth to be 0.5 m for    !
   462)             ! plants that are 0.15-m tall, and 5.0 m for plants that are 35-m tall.        !
   463)             !------------------------------------------------------------------------------!
   464)             root_depth = b1Rd(ipft) * hite ** b2Rd(ipft)
   465)       end select
   466) 
   467) 
   468)       !------------------------------------------------------------------------------------!
   469)       !     Root depth is the maximum root depth if the soil is that deep.  Find what is   !
   470)       ! the deepest soil layer this root can go.                                           !
   471)       !------------------------------------------------------------------------------------!
   472)       dbh2krdepth = nzg
   473)       do k=nzg,lsl+1,-1
   474)          if (root_depth < slz(k)) dbh2krdepth = k-1
   475)       end do
   476)       !------------------------------------------------------------------------------------!
   477) 
   478)       return
   479)    end function dbh2krdepth

   490)    !    This function finds the trunk height.  Currently this is based on the following    !
   491)    ! reference, which is for a site in Bolivia:                                            !
   492)    !                                                                                       !
   493)    ! Poorter L., L. Bongers, F. Bongers, 2006: Architecture of 54 moist-forest tree        !
   494)    !     species: traits, trade-offs, and functional groups. Ecology, 87, 1289-1301.       !
   495)    !                                                                                       !
   496)    ! Crown length is defined as the height of the tree minus the height of the first leaf  !
   497)    !---------------------------------------------------------------------------------------!
   498)    real function h2crownbh(height,ipft)
   499)       use pft_coms, only : b1Cl & ! intent(in)
   500)          , b2Cl ! ! intent(in)
   501) 
   502)       !----- Arguments --------------------------------------------------------------------!
   503)       real   , intent(in) :: height
   504)       integer, intent(in) :: ipft
   505)       !----- Local variables. -------------------------------------------------------------!
   506)       real                :: crown_length
   507)       !------------------------------------------------------------------------------------!
   508)       
   509)       crown_length = b1Cl(ipft) * height ** b2Cl(ipft)
   510)       h2crownbh    = max(0.05,height - crown_length)
   511) 
   512)       return
   513)    end function h2crownbh

   545)    !=======================================================================================!
   546)    !     This subroutine estimates the tree area indices, namely leaf, branch(plus twigs), !
   547)    ! and stem.  For the leaf area index (LAI), we use the specific leaf area (SLA), a      !
   548)    ! constant.  The wood area index WAI is found using the model proposed by Jrvel      !
   549)    ! (2004) to find the specific projected area.                                           !
   550)    !                                                                                       !
   551)    ! G. Hormann, S. Irrgan, H. Jochheim, M. Lukes, H. Meesenburg, J. Muller, B. Scheler,   !
   552)    !    J. Scherzer, G. Schuler, B. Schultze, B. Strohbach, F. Suckow, M. Wegehenkel, and  !
   553)    !    G. Wessolek.   Wasserhaushalt von waldokosystemen: methodenleitfaden zur           !
   554)    !    bestimmung der wasserhaushaltskomponenten auf level II-flachen. Technical note,    !
   555)    !    Bundesministerium fur Verbraucherschutz, Ernahrung und Landwirtschaft (BMVEL),     !
