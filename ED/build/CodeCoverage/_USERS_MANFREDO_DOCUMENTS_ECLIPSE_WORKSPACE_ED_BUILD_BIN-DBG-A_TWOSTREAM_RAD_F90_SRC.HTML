     3) !     This sub-routine solves the long-wave radiation method using the two-stream model.   !
     4) ! We consider the finite crown area when computing the layer transmittance.                !
     5) ! Contrary to short-wave radiation, we don't normalise long-wave radiation.                !
     6) !                                                                                          !
     7) ! References:                                                                              !
     8) !                                                                                          !
     9) ! Liou, K. N., 2002: An introduction to atmospheric radiation. Academic Press, Intl.       !
    10) !    Geophys. Series, vol. 84.  Chapter 6., p. 257-347. (L02)                              !
    11) !                                                                                          !
    12) ! Oleson, K. W., and co-authors, 2004: Technical description of version 4.0 of the         !
    13) !    community land model (CLM). NCAR Technical note NCAR/TN-478+STR. 257pp. (CLM10)       !
    14) !                                                                                          !
    15) ! Sellers, P. J., 1985: Canopy reflectance, photosynthesis and transpiration. Intl. J.     !
    16) !    of remote sensing, 6, 1335-1372. (S85)                                                !
    17) !                                                                                          !
    18) !------------------------------------------------------------------------------------------!
    19) subroutine lw_two_stream(grnd_emiss4,grnd_temp4,rlong_top4,ncoh,pft,lai,wai,cai,leaf_temp  &
    20)                         ,wood_temp,radprof_flip,tir_flip,dw_tirlo,uw_tirlo,uw_tirhi)
    21)    use ed_max_dims          , only : n_pft                   & ! intent(in)
    22)                                    , n_radprof               ! ! intent(in)
    23)    use rk4_coms             , only : tiny_offset             ! ! intent(in)
    24)    use canopy_radiation_coms, only : clumping_factor         & ! intent(in)
    25)                                    , leaf_emiss_tir          & ! intent(in)
    26)                                    , wood_emiss_tir          & ! intent(in)
    27)                                    , mu_bar                  & ! intent(in)
    28)                                    , leaf_backscatter_tir    & ! intent(in)
    29)                                    , wood_backscatter_tir    ! ! intent(in)
    30)    use consts_coms          , only : stefan8                 ! ! intent(in)
    31)    implicit none
    32) 
    33)    !----- Arguments. ----------------------------------------------------------------------!
    34)    real(kind=4)                              , intent(in)    :: grnd_emiss4
    35)    real(kind=4)                              , intent(in)    :: grnd_temp4
    36)    real(kind=4)                              , intent(in)    :: rlong_top4
    37)    integer                                   , intent(in)    :: ncoh
    38)    integer     , dimension(ncoh)             , intent(in)    :: pft
    39)    real(kind=8), dimension(ncoh)             , intent(in)    :: lai
    40)    real(kind=8), dimension(ncoh)             , intent(in)    :: wai
    41)    real(kind=8), dimension(ncoh)             , intent(in)    :: cai
    42)    real(kind=8), dimension(ncoh)             , intent(in)    :: leaf_temp
    43)    real(kind=8), dimension(ncoh)             , intent(in)    :: wood_temp
    44)    real(kind=4), dimension(n_radprof,ncoh)   , intent(inout) :: radprof_flip
    45)    real(kind=4), dimension(ncoh)             , intent(out)   :: tir_flip
    46)    real(kind=4)                              , intent(out)   :: dw_tirlo
    47)    real(kind=4)                              , intent(out)   :: uw_tirlo
    48)    real(kind=4)                              , intent(out)   :: uw_tirhi
    49)    !----- Local variables. ----------------------------------------------------------------!
    50)    integer                                                   :: ipft
    51)    integer                                                   :: nsiz
    52)    integer                                                   :: i
    53)    integer                                                   :: ip1
    54)    integer                                                   :: i2
    55)    integer                                                   :: i2p1
    56)    integer                                                   :: i2m1
    57)    integer                                                   :: i2p2
    58)    logical                                                   :: sing
    59)    real(kind=8), dimension(ncoh+1)                           :: black
    60)    real(kind=8), dimension(ncoh+1)                           :: expl_plus
    61)    real(kind=8), dimension(ncoh+1)                           :: expl_minus
    62)    real(kind=8), dimension(ncoh+1)                           :: gamm_plus
    63)    real(kind=8), dimension(ncoh+1)                           :: gamm_minus
    64)    real(kind=8), dimension(ncoh+1)                           :: down
    65)    real(kind=8), dimension(ncoh+1)                           :: up
    66)    real(kind=8), dimension(2*ncoh+2,2*ncoh+2)                :: mmat
    67)    real(kind=8), dimension(2*ncoh+2)                         :: yvec
    68)    real(kind=8), dimension(2*ncoh+2)                         :: xvec
    69)    real(kind=8), dimension(ncoh+1)                           :: elai
    70)    real(kind=8), dimension(ncoh+1)                           :: etai
    71)    real(kind=8), dimension(ncoh+1)                           :: mu
    72)    real(kind=8), dimension(ncoh+1)                           :: leaf_weight
    73)    real(kind=8), dimension(ncoh+1)                           :: wood_weight
    74)    real(kind=8), dimension(ncoh+1)                           :: beta
    75)    real(kind=8), dimension(ncoh+1)                           :: epsil
    76)    real(kind=8), dimension(ncoh+1)                           :: iota
    77)    real(kind=8), dimension(ncoh+1)                           :: lambda
    78)    real(kind=8)                                              :: temiss_four
    79)    real(kind=8)                                              :: iota_g
    80)    real(kind=8)                                              :: black_g
    81)    real(kind=8)                                              :: down_sky
    82)    !----- External functions. -------------------------------------------------------------!
    83)    real(kind=8)                              , external      :: eifun8
    84)    real(kind=4)                              , external      :: sngloff
    85)    !---------------------------------------------------------------------------------------!
    86) 
    87) 
    88) 
    89)    !---------------------------------------------------------------------------------------!
    90)    !     Find the size of the solution matrix and array.                                   !
    91)    !---------------------------------------------------------------------------------------!
    92)    nsiz = 2*ncoh + 2
    93)    !---------------------------------------------------------------------------------------!
    94) 
    95) 
    96) 
    97)    !---------------------------------------------------------------------------------------!
    98)    !     Convert some properties to double precision.                                      !
    99)    !---------------------------------------------------------------------------------------!
   100)    iota_g   = 1.d0 - dble(grnd_emiss4)
   101)    black_g  = stefan8 * dble(grnd_temp4) ** 4
   102)    down_sky = dble(rlong_top4)
   103)    !---------------------------------------------------------------------------------------!
   104) 
   105) 
   106) 
   107)    !---------------------------------------------------------------------------------------!
   108)    !     Convert some leaf variables to double precision, then find some general           !
   109)    ! properties of leaves/branches (i.e., properties that do not depend on which band we   !
   110)    ! are solving):  the extinction coefficient lambda, the layer transmittance             !
   111)    ! coefficients tau_beam and tau_diff, and the weight of LAI and WAI relative to the     !
   112)    ! total area.                                                                           !
   113)    !---------------------------------------------------------------------------------------!
   114)    do i=1,ncoh
   115)       ipft = pft(i)
   116)       !----- Area indices, correct LAI by clumping factor. --------------------------------!
   117)       elai(i) = clumping_factor(ipft) * lai(i)
   118)       etai(i) = elai(i) + wai(i)
   119)       !------------------------------------------------------------------------------------!
   120) 
   121) 
   122) 
   123)       !------------------------------------------------------------------------------------!
   124)       !    The weighting factors are defined by the relative LAI and WAI areas.            !
   125)       !------------------------------------------------------------------------------------!
   126)       leaf_weight(i)  =  elai(i) / etai(i)
   127)       wood_weight(i)  = 1.d0 -  leaf_weight(i)
   128)       !------------------------------------------------------------------------------------!
   129) 
   130) 
   131) 
   132)       !------------------------------------------------------------------------------------!
   133)       !     mu is the corrected mu_bar to account for finite crown area, if it is          !
   134)       ! provided, otherwise, it is just the same as mu_bar.                                !
   135)       !------------------------------------------------------------------------------------!
   136)       mu(i) = - etai(i) / log( 1.d0 - cai(i)                                               &
   137)                              + cai(i) * exp ( - etai(i) / ( cai(i) * mu_bar(ipft) ) ) )
   138)       !------------------------------------------------------------------------------------!
   139) 
   140)       !------------------------------------------------------------------------------------!
   141)       !      Black is the emission of a black body at the same temperature of the layer.   !
   142)       !  The temperature is the weighted average where the weights are the product between !
   143)       !  the area and the emissivity.                                                      !
   144)       !------------------------------------------------------------------------------------!
   145)       temiss_four = ( leaf_weight(i) * leaf_emiss_tir(ipft) * leaf_temp(i) ** 4            &
   146)                     + wood_weight(i) * wood_emiss_tir(ipft) * wood_temp(i) ** 4 )          &
   147)                   / ( leaf_weight(i) * leaf_emiss_tir(ipft)                                &
   148)                     + wood_weight(i) * wood_emiss_tir(ipft) )
   149)       black(i) = cai(i) * stefan8 * temiss_four
   150)       !------------------------------------------------------------------------------------!
   151) 
   152) 
   153) 
   154)       !------------------------------------------------------------------------------------!
   155)       !      iota is the single scattering albedo.  Because we assume absorptivity to be   !
   156)       ! the same as emissivity, we use the latter to define it.                            !
   157)       !------------------------------------------------------------------------------------!
   158)       iota(i) = 1.d0 - ( leaf_weight(i) * leaf_emiss_tir(ipft)                             &
   159)                        + wood_weight(i) * wood_emiss_tir(ipft) )
   160)       !------------------------------------------------------------------------------------!
   161) 
   162) 
   163) 
   164)       !------------------------------------------------------------------------------------!
   165)       !     Beta is the backward scattering of diffuse radiation. Epsil = 1 - 2*beta.      !
   166)       !------------------------------------------------------------------------------------!
   167)       beta (i) = leaf_weight(i) * leaf_backscatter_tir(ipft)                               &
   168)                + wood_weight(i) * wood_backscatter_tir(ipft)
   169)       epsil(i) = 1.d0 - 2.d0 * beta(i)
   170)       !------------------------------------------------------------------------------------!
   171) 
   172) 
   173) 
   174)       !----- lambda is the coefficient associated with the optical depth. -----------------!
   175)       lambda(i) = sqrt( ( 1.d0 - epsil(i) * iota(i) ) * ( 1.d0 - iota(i) ) ) / mu(i)
   176)       !------------------------------------------------------------------------------------!
   177) 
   178) 
   179) 
   180)       !------------------------------------------------------------------------------------!
   181)       !    gamm_plus and gamm_minus are the coefficients that relate upwelling and down-   !
   182)       ! welling radiation.                                                                 !
   183)       !------------------------------------------------------------------------------------!
   184)       gamm_plus (i) = 5.d-1 * ( 1.d0 + sqrt( ( 1.d0 -            iota(i) )                 &
   185)                                            / ( 1.d0 - epsil(i) * iota(i) ) ) )
   186)       gamm_minus(i) = 5.d-1 * ( 1.d0 - sqrt( ( 1.d0 -            iota(i) )                 &
   187)                                            / ( 1.d0 - epsil(i) * iota(i) ) ) )
   188)       !------------------------------------------------------------------------------------!
   189) 
   190) 
   191) 
   192)       !------------------------------------------------------------------------------------!
   193)       !    expl_plus and expl_minus are the transmitivity of diffuse light.                !
   194)       !------------------------------------------------------------------------------------!
   195)       expl_plus (i) = exp(   lambda(i) * etai(i) )
   196)       expl_minus(i) = exp( - lambda(i) * etai(i) )
   197)       !------------------------------------------------------------------------------------!
   198)    end do
   199)    !---------------------------------------------------------------------------------------!
   200) 
   201) 
   202) 
   203)    !---------------------------------------------------------------------------------------!
   204)    !     Define some boundary conditions for the vector properties above.                  !
   205)    !---------------------------------------------------------------------------------------!
   206)    i              = ncoh+1
   207)    elai       (i) = 0.d0
   208)    etai       (i) = 0.d0
   209)    leaf_weight(i) = 5.d-1
   210)    wood_weight(i) = 5.d-1
   211)    mu         (i) = 1.d0
   212)    black      (i) = 0.d0
   213)    iota       (i) = 1.d0
   214)    beta       (i) = 0.d0
   215)    epsil      (i) = 1.d0 - 2.d0 * beta(i)
   216)    lambda     (i) = 0.d0
   217)    gamm_plus  (i) = 1.d0
   218)    gamm_minus (i) = 0.d0
   219)    expl_plus  (i) = 1.d0
   220)    expl_minus (i) = 1.d0
   221)    !---------------------------------------------------------------------------------------!
   222) 
   223) 
   224) 
   225) 
   226)    !---------------------------------------------------------------------------------------!
   227)    !     Fill in the right hand side vector (Y) and the matrix (X)                         !
   228)    !---------------------------------------------------------------------------------------!
   229)    !------ Initialise the vector and the matrix.  The matrix is a sparse one... -----------!
   230)    mmat(:,:)         = 0.d0
   231)    yvec(:)           = 0.d0
   232)    !------ Add the bottom and top boundary conditions. ------------------------------------!
   233)    mmat(1,1)         = (gamm_minus(1) - iota_g * gamm_plus (1)) * expl_minus(1)
   234)    mmat(1,2)         = (gamm_plus (1) - iota_g * gamm_minus(1)) * expl_plus (1)
   235)    mmat(nsiz,nsiz-1) = gamm_plus (ncoh+1)
   236)    mmat(nsiz,nsiz  ) = gamm_minus(ncoh+1)
   237)    yvec(1)           = (1.d0 - iota_g) * black_g - (1.d0 - iota_g) * black(1)
   238)    yvec(nsiz)        = down_sky - black(ncoh+1)
   239)    do i=1,ncoh
   240)       !----- Find auxiliary indices. ------------------------------------------------------!
   241)       ip1  =  i + 1
   242)       i2   =  2 * i
   243)       i2m1 = i2 - 1
   244)       i2p1 = i2 + 1
   245)       i2p2 = i2 + 2
   246)       !------------------------------------------------------------------------------------!
   247) 
   248)       !----- Make elements of C vector. ---------------------------------------------------!
   249)       yvec(i2  ) = black(ip1) - black(i)
   250)       yvec(i2p1) = black(ip1) - black(i)
   251)       !------------------------------------------------------------------------------------!
   252) 
   253) 
   254) 
   255)       !------------------------------------------------------------------------------------!
   256)       !     Make elements of the tri-diagonal A array.                                     !
   257)       !------------------------------------------------------------------------------------!
   258)       mmat(i2  ,i2m1) =   gamm_plus   (i)
   259)       mmat(i2  ,i2  ) =   gamm_minus  (i)
   260)       mmat(i2  ,i2p1) = - gamm_plus (ip1) * expl_minus(ip1)
   261)       mmat(i2  ,i2p2) = - gamm_minus(ip1) * expl_plus (ip1)
   262)       mmat(i2p1,i2m1) =   gamm_minus  (i)
   263)       mmat(i2p1,i2  ) =   gamm_plus   (i)
   264)       mmat(i2p1,i2p1) = - gamm_minus(ip1) * expl_minus(ip1)
   265)       mmat(i2p1,i2p2) = - gamm_plus (ip1) * expl_plus (ip1)
   266)    end do
   267)    !---------------------------------------------------------------------------------------!
   268) 
   269) 
   270) 
   271)    !---------------------------------------------------------------------------------------!
   272)    !       Solve the linear system.  In the future we could use a band diagonal solver,    !
   273)    ! which is a lot cheaper than the regular Gauss elimination, but for the time being, we !
   274)    ! go with a tested method.                                                              !
   275)    !---------------------------------------------------------------------------------------!
   276)    call lisys_solver8(nsiz,mmat,yvec,xvec,sing)
   277)    if (sing) then
   278)       call fatal_error('LW radiation failed... The matrix is singular!'                    &
   279)                       ,'lw_two_stream','twostream_rad.f90')
   280)    end if
   281)    !---------------------------------------------------------------------------------------!
   282) 
   283) 
   284) 
   285)    !---------------------------------------------------------------------------------------!
   286)    !     Copy the solution to the vectors, using the properties:                           !
   287)    !  Un = Un(P)                                                                           !
   288)    !  Dn = Dn(P)                                                                           !
   289)    !---------------------------------------------------------------------------------------!
   290)    do i=1,ncoh+1
   291) 
   292)       !----- Auxiliary indices. -----------------------------------------------------------!
   293)       i2   =  2 * i
   294)       i2m1 = i2 - 1
   295)       !------------------------------------------------------------------------------------!
   296) 
   297) 
   298) 
   299)       !------------------------------------------------------------------------------------!
   300)       !      Retrieve the downward and upward diffuse (hemispheric) radiation, by using    !
   301)       ! the solutions for full TAI                                                         !
   302)       !------------------------------------------------------------------------------------!
   303)       down(i) = xvec(i2m1) * gamm_plus (i) * expl_minus(i)                                 &
   304)               + xvec  (i2) * gamm_minus(i) * expl_plus (i)                                 &
   305)               + black  (i)
   306)       up  (i) = xvec(i2m1) * gamm_minus(i) * expl_minus(i)                                 &
   307)               + xvec  (i2) * gamm_plus (i) * expl_plus (i)                                 &
   308)               + black  (i)
   309)       !------------------------------------------------------------------------------------!
   310)    end do
   311)    !---------------------------------------------------------------------------------------!
   312) 
   313) 
   314) 
   315)    !------ Save the fluxes reaching the surface and leaving the top. ----------------------!
   316)    dw_tirlo = sngloff(down     (1), tiny_offset)
   317)    uw_tirlo = sngloff(up       (1), tiny_offset)
   318)    uw_tirhi = sngloff(up  (ncoh+1), tiny_offset)
   319)    !---------------------------------------------------------------------------------------!
   320) 
   321) 
   322)    !---------------------------------------------------------------------------------------!
   323)    !     Save the radiation fluxes to the output variable.                                 !
   324)    !---------------------------------------------------------------------------------------!
   325)    do i=1,ncoh
   326)       ip1 = i + 1
   327)       radprof_flip ( 9,i) = sngloff(down(i), tiny_offset)
   328)       radprof_flip (10,i) = sngloff(up  (i), tiny_offset)
   329)       tir_flip        (i) = sngloff(down(ip1) - down(i) + up(i) - up(ip1), tiny_offset)
   330)    end do
   331)    !---------------------------------------------------------------------------------------!
   332) 
   333)    return
   334) end subroutine lw_two_stream

   345) !     This sub-routine solves the short-wave radiation method using the two-stream model.  !
   346) ! We consider the finite crown area when computing the transmittance of both direct and    !
   347) ! diffuse radiation.  References:                                                          !
   348) !                                                                                          !
   349) ! Liou, K. N., 2002: An introduction to atmospheric radiation. Academic Press, Intl.       !
   350) !    Geophys. Series, vol. 84.  Chapter 6., p. 257-347. (L02)                              !
   351) !                                                                                          !
   352) ! Oleson, K. W., and co-authors, 2004: Technical description of the community land model   !
   353) !   (CLM). NCAR Technical note NCAR/TN-461+STR. 186pp. (CLM04)                             !
   354) !                                                                                          !
   355) ! Sellers, P. J., 1985: Canopy reflectance, photosynthesis and transpiration. Intl. J.     !
   356) !    of remote sensing, 6, 1335-1372. (S85)                                                !
   357) !                                                                                          !
   358) !------------------------------------------------------------------------------------------!
   359) subroutine sw_two_stream(grnd_alb_par4,grnd_alb_nir4,cosaoi4,ncoh,pft,lai,wai,cai          &
   360)                         ,radprof_flip,par_beam_flip,par_diff_flip,sw_abs_beam_flip         &
   361)                         ,sw_abs_diff_flip,dw_parlo_beam,dw_parlo_diff                      &
   362)                         ,uw_parhi_diff,dw_nirlo_beam,dw_nirlo_diff,uw_nirhi_diff           &
   363)                         ,par_level_beam,par_level_diffd,par_level_diffu                   &
   364)                         ,light_level,light_beam_level        &
   365)                         ,light_diff_level)
   366)    use ed_max_dims          , only : n_pft                   & ! intent(in)
   367)                                    , n_radprof               ! ! intent(in)
   368)    use rk4_coms             , only : tiny_offset             ! ! intent(in)
   369)    use canopy_radiation_coms, only : clumping_factor         & ! intent(in)
   370)                                    , phi1                    & ! intent(in)
   371)                                    , phi2                    & ! intent(in)
   372)                                    , mu_bar                  & ! intent(in)
   373)                                    , leaf_scatter_vis        & ! intent(in)
   374)                                    , leaf_scatter_nir        & ! intent(in)
   375)                                    , leaf_backscatter_nir    & ! intent(in)
   376)                                    , leaf_backscatter_vis    & ! intent(in)
   377)                                    , wood_scatter_nir        & ! intent(in)
   378)                                    , wood_scatter_vis        & ! intent(in)
   379)                                    , wood_backscatter_nir    & ! intent(in)
   380)                                    , wood_backscatter_vis    & ! intent(in)
   381)                                    , par_beam_norm           & ! intent(in)
   382)                                    , par_diff_norm           & ! intent(in)
   383)                                    , nir_beam_norm           & ! intent(in)
   384)                                    , nir_diff_norm           & ! intent(in)
   385)                                    , cosz_min8               ! ! intent(in)
   386) 
   387)    implicit none
   388) 
   389)    !----- Arguments. ----------------------------------------------------------------------!
   390)    real(kind=4)                              , intent(in)     :: grnd_alb_par4
   391)    real(kind=4)                              , intent(in)     :: grnd_alb_nir4
   392)    real(kind=4)                              , intent(in)     :: cosaoi4
   393)    integer                                   , intent(in)     :: ncoh
   394)    integer     , dimension(ncoh)             , intent(in)     :: pft
   395)    real(kind=8), dimension(ncoh)             , intent(in)     :: lai
   396)    real(kind=8), dimension(ncoh)             , intent(in)     :: wai
   397)    real(kind=8), dimension(ncoh)             , intent(in)     :: cai
   398)    real(kind=4), dimension(n_radprof,ncoh)   , intent(inout)  :: radprof_flip
   399)    real(kind=4), dimension(ncoh)             , intent(out)    :: par_beam_flip
   400)    real(kind=4), dimension(ncoh)             , intent(out)    :: par_diff_flip
   401)    real(kind=4), dimension(ncoh)             , intent(out)    :: sw_abs_beam_flip
   402)    real(kind=4), dimension(ncoh)             , intent(out)    :: sw_abs_diff_flip
   403)    real(kind=4)                              , intent(out)    :: uw_parhi_diff
   404)    real(kind=4)                              , intent(out)    :: uw_nirhi_diff
   405)    real(kind=4)                              , intent(out)    :: dw_parlo_beam
   406)    real(kind=4)                              , intent(out)    :: dw_parlo_diff
   407)    real(kind=4)                              , intent(out)    :: dw_nirlo_beam
   408)    real(kind=4)                              , intent(out)    :: dw_nirlo_diff
   409)    real(kind=8), dimension(ncoh)             , intent(out)    :: par_level_beam
   410)    real(kind=8), dimension(ncoh)             , intent(out)    :: par_level_diffd
   411)    real(kind=8), dimension(ncoh)             , intent(out)    :: par_level_diffu
   412)    real(kind=8), dimension(ncoh)             , intent(out)    :: light_level
   413)    real(kind=8), dimension(ncoh)             , intent(out)    :: light_beam_level
   414)    real(kind=8), dimension(ncoh)             , intent(out)    :: light_diff_level
   415)    !----- Local variables. ----------------------------------------------------------------!
   416)    integer                                                    :: ipft
   417)    integer                                                    :: iband
   418)    integer                                                    :: nsiz
   419)    integer                                                    :: i
   420)    integer                                                    :: ip1
   421)    integer                                                    :: i2
   422)    integer                                                    :: i2p1
   423)    integer                                                    :: i2m1
   424)    integer                                                    :: i2p2
   425)    logical                                                    :: sing
   426)    real(kind=4), dimension(ncoh)                              :: nir_beam_flip
   427)    real(kind=4), dimension(ncoh)                              :: nir_diff_flip
   428)    real(kind=8), dimension(ncoh+1)                            :: expl_plus
   429)    real(kind=8), dimension(ncoh+1)                            :: expl_minus
   430)    real(kind=8), dimension(ncoh+1)                            :: expm0_minus
   431)    real(kind=8), dimension(ncoh+1)                            :: gamm_plus
   432)    real(kind=8), dimension(ncoh+1)                            :: gamm_minus
   433)    real(kind=8), dimension(ncoh+1)                            :: delta
   434)    real(kind=8), dimension(ncoh+1)                            :: upsilon
   435)    real(kind=8), dimension(ncoh+1)                            :: down0
   436)    real(kind=8), dimension(ncoh+1)                            :: down
   437)    real(kind=8), dimension(ncoh+1)                            :: up
   438)    real(kind=8), dimension(2*ncoh+2,2*ncoh+2)                 :: mmat
   439)    real(kind=8), dimension(2*ncoh+2)                          :: yvec
   440)    real(kind=8), dimension(2*ncoh+2)                          :: xvec
   441)    real(kind=8), dimension(n_pft)                             :: leaf_scatter
   442)    real(kind=8), dimension(n_pft)                             :: wood_scatter
   443)    real(kind=8), dimension(n_pft)                             :: leaf_backscatter
   444)    real(kind=8), dimension(n_pft)                             :: wood_backscatter
   445)    real(kind=8), dimension(ncoh+1)                            :: elai
   446)    real(kind=8), dimension(ncoh+1)                            :: etai
   447)    real(kind=8), dimension(ncoh+1)                            :: mu
   448)    real(kind=8), dimension(ncoh+1)                            :: mu0
   449)    real(kind=8), dimension(ncoh+1)                            :: leaf_weight
   450)    real(kind=8), dimension(ncoh+1)                            :: wood_weight
   451)    real(kind=8), dimension(ncoh+1)                            :: beta0
   452)    real(kind=8), dimension(ncoh+1)                            :: beta
   453)    real(kind=8), dimension(ncoh+1)                            :: epsil
   454)    real(kind=8), dimension(ncoh+1)                            :: epsil0
   455)    real(kind=8), dimension(ncoh+1)                            :: iota
   456)    real(kind=8), dimension(ncoh+1)                            :: lambda
   457)    real(kind=8), dimension(ncoh+1)                            :: iota_ratio
   458)    real(kind=8), dimension(ncoh+1)                            :: proj_area
   459)    real(kind=8), dimension(ncoh+1)                            :: a_aux
   460)    real(kind=8), dimension(ncoh+1)                            :: s_aux
   461)    real(kind=8)                                               :: iota_g
   462)    real(kind=8)                                               :: iota_g_par
   463)    real(kind=8)                                               :: iota_g_nir
   464)    real(kind=8)                                               :: down_sky
   465)    real(kind=8)                                               :: down0_sky
   466)    real(kind=8)                                               :: czen
   467)    !----- External functions. -------------------------------------------------------------!
   468)    real(kind=4)                              , external       :: sngloff
   469)    !---------------------------------------------------------------------------------------!
   470) 
   471) 
   472)    !---------------------------------------------------------------------------------------!
   473)    !     Find the size of the solution matrix and array.                                   !
   474)    !---------------------------------------------------------------------------------------!
   475)    nsiz = 2*ncoh + 2
   476)    !---------------------------------------------------------------------------------------!
   477) 
   478) 
   479) 
   480)    !---------------------------------------------------------------------------------------!
   481)    !     Convert cosine of angle of incidence to double precision.                         !
   482)    !---------------------------------------------------------------------------------------!
   483)    czen       = max(cosz_min8,dble(cosaoi4))
   484)    iota_g_par = dble(grnd_alb_par4)
   485)    iota_g_nir = dble(grnd_alb_nir4)
   486)    !---------------------------------------------------------------------------------------!
   487) 
   488)    !---------------------------------------------------------------------------------------!
   489)    !     Initialise the light level variables.                                             !
   490)    !---------------------------------------------------------------------------------------!
   491)    light_level     (:) = 0.d0
   492)    light_beam_level(:) = 0.d0
   493)    light_diff_level(:) = 0.d0
   494) !   par_level_beam (:) = 0.d0
   495) !   par_level_diffd(:) = 0.d0
   496) !   par_level_diffu(:) = 0.d0
   497)    !---------------------------------------------------------------------------------------!
   498) 
   499) 
   500) 
   501)    !---------------------------------------------------------------------------------------!
   502)    !     Here we loop over the two bands (band 1 is PAR and band 2 is NIR), and solve the  !
   503)    ! same set of equations.                                                                !
   504)    !---------------------------------------------------------------------------------------!
   505)    bandloop: do iband = 1,2
   506)       select case (iband)
   507)       case (1)
   508)          !---------------------------------------------------------------------------------!
   509)          !     Visible (PAR).                                                              !
   510)          !---------------------------------------------------------------------------------!
   511)          leaf_scatter    (:) = leaf_scatter_vis    (:)
   512)          wood_scatter    (:) = wood_scatter_vis    (:)
   513)          leaf_backscatter(:) = leaf_backscatter_vis(:)
   514)          wood_backscatter(:) = wood_backscatter_vis(:)
   515)          down_sky            = par_diff_norm
   516)          down0_sky           = par_beam_norm
   517)          iota_g              = iota_g_par
   518)          !---------------------------------------------------------------------------------!
   519)       case (2)
   520)          !---------------------------------------------------------------------------------!
   521)          !     Near-Infrared (NIR).                                                        !
   522)          !---------------------------------------------------------------------------------!
   523)          leaf_scatter    (:) = leaf_scatter_nir    (:)
   524)          wood_scatter    (:) = wood_scatter_nir    (:)
   525)          leaf_backscatter(:) = leaf_backscatter_nir(:)
   526)          wood_backscatter(:) = wood_backscatter_nir(:)
   527)          down_sky            = nir_diff_norm
   528)          down0_sky           = nir_beam_norm
   529)          iota_g              = iota_g_nir
   530)          !---------------------------------------------------------------------------------!
   531)       end select
   532)       !------------------------------------------------------------------------------------!
   533) 
   534) 
   535) 
   536)       !------------------------------------------------------------------------------------!
   537)       !     Find some general properties of leaves/branches and also direct radiation      !
   538)       ! properties.                                                                        !
   539)       !------------------------------------------------------------------------------------!
   540)       directloop: do i=1,ncoh
   541)          ipft      = pft(i)
   542) 
   543)          !----- Area indices, correct LAI by clumping factor. -----------------------------!
   544)          elai(i) = clumping_factor(ipft) * lai(i)
   545)          etai(i) = elai(i) + wai(i)
   546)          !---------------------------------------------------------------------------------!
   547) 
   548) 
   549) 
   550)          !----- The weighting factors are defined by the relative LAI and WAI areas. ------!
   551)          leaf_weight(i) = elai(i) / etai(i)
   552)          wood_weight(i) = 1.d0 - leaf_weight(i)
   553)          !---------------------------------------------------------------------------------!
   554) 
   555) 
   556) 
   557)          !---------------------------------------------------------------------------------!
   558)          !     We find the inverse optical depth of the direct radiation (mu0), following  !
   559)          ! CLM10 (equation 3.3 and text after equation 3.3).                               !
   560)          !---------------------------------------------------------------------------------!
   561)          proj_area(i) = phi1(ipft) + phi2(ipft) * czen
   562)          mu0      (i) = - etai(i)                                                          &
   563)                         / log( ( 1.d0 - cai(i) )                                           &
   564)                              + cai(i) * exp( - proj_area(i) * etai(i)                      &
   565)                                              / ( cai(i) * czen ) ) )
   566)          !---------------------------------------------------------------------------------!
   567) 
   568) 
   569) 
   570) 
   571)          !---------------------------------------------------------------------------------!
   572)          !     The inverse optical depth of the diffuse radiation (mu), same as in long-   !
   573)          ! wave radiation.                                                                 !
   574)          !---------------------------------------------------------------------------------!
   575)          mu (i) = - etai(i) / log( ( 1.d0 - cai(i) )                                       &
   576)                                  + cai(i) * exp( - etai(i) / mu_bar(ipft) ) )
   577)          !---------------------------------------------------------------------------------!
   578) 
   579) 
   580) 
   581)          !---------------------------------------------------------------------------------!
   582)          !      Find the backscatter for direct beam radiation, following CLM10, equations !
   583)          ! (3.14) and (3.15).  The forward scattering was omitted on both equations        !
   584)          ! because they are different for PAR and NIR, but in both cases they would cancel !
   585)          ! out.  epsil0 = 1 - 2*beta0, no special definition...                            !
   586)          !---------------------------------------------------------------------------------!
   587)          iota_ratio(i) = 1.d0 / ( 2.d0 * ( 1.d0 + phi2(ipft) * mu0(i) ) )                  &
   588)                        * ( 1.d0 - phi1(ipft) * mu0(i) / ( 1.d0 + phi2(ipft) * mu0(i) )     &
   589)                                 * log ( ( 1.d0  + ( phi1(ipft) + phi2(ipft) ) * mu0(i) )   &
   590)                                       / ( phi1(ipft) * mu0(i) ) ) )
   591)          beta0     (i) = iota_ratio(i) * ( 1.d0 + mu0(i) / mu(i) )
   592)          epsil0    (i) = 1.d0 - 2.d0 * beta0(i)
   593)          !---------------------------------------------------------------------------------!
   594) 
   595) 
   596)          !----- Expm0_minus is the transmissivity of direct radiation. --------------------!
   597)          expm0_minus(i) = exp( - etai  (i) / mu0 (i) )
   598)          !---------------------------------------------------------------------------------!
   599)       end do directloop
   600)       !------------------------------------------------------------------------------------!
   601) 
   602) 
   603) 
   604)       !------------------------------------------------------------------------------------!
   605)       !     Define some boundary conditions for the vector properties above.               !
   606)       !------------------------------------------------------------------------------------!
   607)       i              = ncoh+1
   608)       elai       (i) = 0.d0
   609)       etai       (i) = 0.d0
   610)       leaf_weight(i) = 5.d-1
   611)       wood_weight(i) = 5.d-1
   612)       proj_area  (i) = 5.d-1
   613)       mu0        (i) = czen / proj_area(i)
   614)       mu         (i) = 1.d0
   615)       iota_ratio (i) = 5.d-1 * ( 1.d0 - 5.d-1 * mu0(i)                                     &
   616)                                       * log ( 1.d0  / ( 5.d-1 * mu0(i) ) + 1.d0 ) )
   617)       beta0      (i) = iota_ratio(i) * ( mu0(i) + mu(i) ) / mu(i)
   618)       epsil0     (i) = 1.d0 - 2.d0 * beta0(i)
   619)       expm0_minus(i) = 1.d0
   620)       !------------------------------------------------------------------------------------!
   621) 
   622)       !------------------------------------------------------------------------------------!
   623)       !     Find the direct radiation profile using the exponential attenuation curve.     !
   624)       !------------------------------------------------------------------------------------!
   625)       down0(ncoh+1) = down0_sky
   626)       do i = ncoh,1,-1
   627)          ip1      = i + 1
   628)          down0(i) = down0(ip1) * expm0_minus(i) 
   629)       end do
   630)       !------------------------------------------------------------------------------------!
   631) 
   632) 
   633) 
   634)       !------------------------------------------------------------------------------------!
   635)       !     Find the diffuse radiation properties.                                         !
   636)       !------------------------------------------------------------------------------------!
   637)       diffuseloop: do i=1,ncoh
   638)          !----- Scattering coefficient. ---------------------------------------------------!
   639)          iota      (i) = leaf_weight(i) * leaf_scatter(ipft)                               &
   640)                        + wood_weight(i) * wood_scatter(ipft)
   641)          !---------------------------------------------------------------------------------!
   642) 
   643) 
   644) 
   645)          !---------------------------------------------------------------------------------!
   646)          !     Beta is the backward scattering of diffuse radiation. Epsil = 1 - 2*beta.   !
   647)          !---------------------------------------------------------------------------------!
   648)          beta  (i) = leaf_weight(i) * leaf_backscatter(ipft)                               &
   649)                    + wood_weight(i) * wood_backscatter(ipft)
   650)          epsil (i) = 1.d0 - 2.d0 * beta(i)
   651)          !---------------------------------------------------------------------------------!
   652) 
   653) 
   654) 
   655)          !----- lambda is the coefficient associated with the optical depth. --------------!
   656)          lambda(i) = sqrt( ( 1.d0 - epsil(i) * iota(i) ) * ( 1.d0 - iota(i) ) ) / mu(i)
   657)          !---------------------------------------------------------------------------------!
   658) 
   659) 
   660) 
   661)          !---------------------------------------------------------------------------------!
   662)          !     Find some auxiliary variables to determine the right hand side.             !
   663)          !---------------------------------------------------------------------------------!
   664)          a_aux  (i) = - ( ( 1.d0 - epsil(i) * iota(i) ) * iota(i) / mu(i)                  &
   665)                         + epsil0(i) * iota(i) / mu0(i) ) * down0(i+1) / mu0(i)
   666)          s_aux  (i) = - ( ( 1.d0 - iota (i) ) * epsil0(i) * iota(i) / mu(i)                &
   667)                         + iota(i) / mu0(i) ) * down0(i+1) / mu0(i)
   668)          delta  (i) = ( a_aux(i) + s_aux(i) ) * mu0(i) * mu0(i)                            &
   669)                     / ( 2.d0 * ( 1.d0 - lambda(i) * lambda(i) * mu0(i) * mu0(i) ) )
   670)          upsilon(i) = ( a_aux(i) - s_aux(i) ) * mu0(i) * mu0(i)                            &
   671)                     / ( 2.d0 * ( 1.d0 - lambda(i) * lambda(i) * mu0(i) * mu0(i) ) )
   672)          !---------------------------------------------------------------------------------!
   673) 
   674) 
   675)          !---------------------------------------------------------------------------------!
   676)          !    gamm_plus and gamm_minus are the coefficients that relate upwelling and      !
   677)          ! downwelling radiation.                                                          !
   678)          !---------------------------------------------------------------------------------!
   679)          gamm_plus (i) = 5.d-1 * ( 1.d0 + sqrt( ( 1.d0 -            iota(i) )              &
   680)                                               / ( 1.d0 - epsil(i) * iota(i) ) ) )
   681)          gamm_minus(i) = 5.d-1 * ( 1.d0 - sqrt( ( 1.d0 -            iota(i) )              &
   682)                                               / ( 1.d0 - epsil(i) * iota(i) ) ) )
   683)          !---------------------------------------------------------------------------------!
   684) 
   685) 
   686) 
   687)          !---------------------------------------------------------------------------------!
   688)          !    expl_plus and expl_minus are the transmitivity of diffuse light.             !
   689)          !---------------------------------------------------------------------------------!
   690)          expl_plus  (i) = exp(   lambda(i) * etai(i) )
   691)          expl_minus (i) = exp( - lambda(i) * etai(i) )
   692)          !---------------------------------------------------------------------------------!
   693)       end do diffuseloop
   694)       !------------------------------------------------------------------------------------!
   695) 
   696) 
   697) 
   698)       !------------------------------------------------------------------------------------!
   699)       !     Define some boundary conditions for the vector properties above.               !
   700)       !------------------------------------------------------------------------------------!
   701)       i              = ncoh+1
   702)       iota       (i) = 1.d0
   703)       beta       (i) = 0.d0
   704)       epsil      (i) = 1.d0 - 2.d0 * beta(i)
   705)       lambda     (i) = 0.d0
   706)       a_aux      (i) = 0.d0
   707)       s_aux      (i) = 0.d0
   708)       delta      (i) = 0.d0
   709)       upsilon    (i) = 0.d0
   710)       a_aux      (i) = -  epsil0(i) * down0_sky / ( mu0(i) * mu0(i) )
   711)       s_aux      (i) = -  iota  (i) * down0_sky / ( mu0(i) * mu0(i) )
   712)       delta      (i) = 5.d-1 * ( a_aux(i) + s_aux(i) ) * mu0(i) * mu0(i)
   713)       upsilon    (i) = 5.d-1 * ( a_aux(i) - s_aux(i) ) * mu0(i) * mu0(i)
   714)       gamm_plus  (i) = 1.d0
   715)       gamm_minus (i) = 0.d0
   716)       expl_plus  (i) = 1.d0
   717)       expl_minus (i) = 1.d0
   718)       !------------------------------------------------------------------------------------!
   719) 
   720) 
   721) 
   722) 
   723)       !------------------------------------------------------------------------------------!
   724)       !     Fill in the right hand side vector (Y) and the matrix (X)                      !
   725)       !------------------------------------------------------------------------------------!
   726)       !------ Initialise the vector and the matrix.  The matrix is a sparse one... --------!
   727)       mmat(:,:)         = 0.d0
   728)       yvec(:)           = 0.d0
   729)       !------ Add the bottom and top boundary conditions. ---------------------------------!
   730)       mmat(1,1)         = (gamm_minus(1) - iota_g * gamm_plus (1)) * expl_minus(1)
   731)       mmat(1,2)         = (gamm_plus (1) - iota_g * gamm_minus(1)) * expl_plus (1)
   732)       mmat(nsiz,nsiz-1) = gamm_plus (ncoh+1)
   733)       mmat(nsiz,nsiz  ) = gamm_minus(ncoh+1)
   734)       yvec(1)           = iota_g * down0(1)                                                &
   735)                         - ( upsilon(1) - iota_g * delta(1) ) * expm0_minus(1)
   736)       yvec(nsiz)        = down_sky - delta(ncoh+1)
   737)       do i=1,ncoh
   738)          !----- Find auxiliary indices. ---------------------------------------------------!
   739)          ip1  =  i + 1
   740)          i2   =  2 * i
   741)          i2m1 = i2 - 1
   742)          i2p1 = i2 + 1
   743)          i2p2 = i2 + 2
   744)          !---------------------------------------------------------------------------------!
   745) 
   746)          !----- Make elements of C vector. ------------------------------------------------!
   747)          yvec(i2  ) = delta  (ip1) * expm0_minus(ip1) - delta  (i)
   748)          yvec(i2p1) = upsilon(ip1) * expm0_minus(ip1) - upsilon(i)
   749)          !---------------------------------------------------------------------------------!
   750) 
   751) 
   752) 
   753)          !---------------------------------------------------------------------------------!
   754)          !     Make elements of the tri-diagonal A array.                                  !
   755)          !---------------------------------------------------------------------------------!
   756)          mmat(i2  ,i2m1) =   gamm_plus   (i)
   757)          mmat(i2  ,i2  ) =   gamm_minus  (i)
   758)          mmat(i2  ,i2p1) = - gamm_plus (ip1) * expl_minus(ip1)
   759)          mmat(i2  ,i2p2) = - gamm_minus(ip1) * expl_plus (ip1)
   760)          mmat(i2p1,i2m1) =   gamm_minus  (i)
   761)          mmat(i2p1,i2  ) =   gamm_plus   (i)
   762)          mmat(i2p1,i2p1) = - gamm_minus(ip1) * expl_minus(ip1)
   763)          mmat(i2p1,i2p2) = - gamm_plus (ip1) * expl_plus (ip1)
   764)          !---------------------------------------------------------------------------------!
   765)       end do
   766)       !------------------------------------------------------------------------------------!
   767) 
   768) 
   769) 
   770)       !------------------------------------------------------------------------------------!
   771)       !       Solve the linear system.  In the future we could use a tridiagonal solver,   !
   772)       ! which is a lot cheaper than the regular Gauss elimination, but for the time being, !
   773)       ! we go with a tested method.                                                        !
   774)       !------------------------------------------------------------------------------------!
   775)       call lisys_solver8(nsiz,mmat,yvec,xvec,sing)
   776)       if (sing) then
   777)          call fatal_error('SW radiation failed... The matrix is singular!'                 &
   778)                          ,'sw_two_stream','twostream_rad.f90')
   779)       end if
   780)       !------------------------------------------------------------------------------------!
   781) 
   782) 
   783) 
   784)       !------------------------------------------------------------------------------------!
   785)       !     Copy the solution to the vectors, using the properties:                        !
   786)       !  Un = Un(P)                                                                        !
   787)       !  Dn = Dn(P)                                                                        !
   788)       !------------------------------------------------------------------------------------!
   789)       do i=1,ncoh+1
   790) 
   791)          !----- Auxiliary indices. --------------------------------------------------------!
   792)          i2   =  2 * i
   793)          i2m1 = i2 - 1
   794)          !---------------------------------------------------------------------------------!
   795) 
   796) 
   797) 
   798)          !---------------------------------------------------------------------------------!
   799)          !      Retrieve the downward and upward diffuse (hemispheric) radiation, by using !
   800)          ! the solutions for full TAI                                                      !
   801)          !---------------------------------------------------------------------------------!
   802)          down(i) = xvec(i2m1) * gamm_plus (i) * expl_minus (i)                             &
   803)                  + xvec  (i2) * gamm_minus(i) * expl_plus  (i)                             &
   804)                               + delta     (i) * expm0_minus(i)
   805)          up  (i) = xvec(i2m1) * gamm_minus(i) * expl_minus (i)                             &
   806)                  + xvec  (i2) * gamm_plus (i) * expl_plus  (i)                             &
   807)                               + upsilon   (i) * expm0_minus(i)
   808)          !---------------------------------------------------------------------------------!
   809)       end do
   810)       !------------------------------------------------------------------------------------!
   811) 
   812) 
   813)       !------------------------------------------------------------------------------------!
   814)       !     Integrate the light levels.                                                    !
   815)       !------------------------------------------------------------------------------------!
   816)       do i=1,ncoh
   817)          ip1 = i + 1
   818)          light_level     (i) = light_level     (i) + 5.d-1 * ( down (i) + down (ip1)       &
   819)                                                              + down0(i) + down0(ip1) )
   820)          light_beam_level(i) = light_beam_level(i) + 5.d-1 * ( down0(i) + down0(ip1) )
   821)          light_diff_level(i) = light_diff_level(i) + 5.d-1 * ( down (i) + down (ip1) )
   822)       end do
   823)       !------------------------------------------------------------------------------------!
   824) 
   825) 
   826) 
   827)       !------------------------------------------------------------------------------------!
   828)       !     Save the fluxes that we will use outside this sub-routine.                     !
   829)       !------------------------------------------------------------------------------------!
   830)       select case (iband)
   831)       case (1)
   832)          !---------------------------------------------------------------------------------!
   833)          !     Visible (PAR).                                                              !
   834)          !---------------------------------------------------------------------------------!
   835) 
   836)          !------ Mean PAR centered on the different levels beam, up and down diffuse ------!
   837)          !------ these are fluxes, not absorbed, as a fraction of total PAR
   838)          !------ The current units of up and down are fraction of total rshort
   839)          !------ so we have to divide by the total PAR fraction of rshort, the
   840)          !------ units are fraction of total par
   841)          do i=1,ncoh
   842)             ip1 = i + 1
   843)             par_level_diffu(i) = 5.d-1*(up(i)    + up(ip1)) /(par_beam_norm+par_diff_norm)
   844)             par_level_diffd(i) = 5.d-1*(down(i)  + down(ip1)) /(par_beam_norm+par_diff_norm)
   845)             par_level_beam(i)  = 5.d-1*(down0(i) + down0(ip1)) /(par_beam_norm+par_diff_norm)
   846)          end do
   847)          !---------------------------------------------------------------------------------!
   848) 
   849)          ! par_level_beam
   850)          ! par_level_diffu
   851)          ! par_level_diffd
   852) 
   853)          !---------------------------------------------------------------------------------!
   854)          !     Save the radiation fluxes to the output variable.                           !
   855)          !---------------------------------------------------------------------------------!
   856)          do i=1,ncoh
   857)             ip1 = i + 1
   858)             par_beam_flip   (i) = sngloff( down0(ip1) - down0  (i), tiny_offset)
   859)             par_diff_flip   (i) = sngloff( down (ip1) - down   (i)                         &
   860)                                          + up     (i) - up   (ip1), tiny_offset)
   861)             radprof_flip ( 1,i) = sngloff(down0(i), tiny_offset)
   862)             radprof_flip ( 2,i) = 0.0
   863)             radprof_flip ( 3,i) = sngloff(down(i), tiny_offset)
   864)             radprof_flip ( 4,i) = sngloff(up  (i), tiny_offset)
   865)          end do
   866)          !---------------------------------------------------------------------------------!
   867) 
   868) 
   869) 
   870)          !------ Save the fluxes reaching the surface and leaving the top. ----------------!
   871)          dw_parlo_beam = sngloff(down0          (1), tiny_offset)
   872)          dw_parlo_diff = sngloff(down           (1), tiny_offset)
   873)          uw_parhi_diff = sngloff(up        (ncoh+1), tiny_offset)
   874)          !---------------------------------------------------------------------------------!
   875) 
   876)       case (2)
   877)          !---------------------------------------------------------------------------------!
   878)          !     Near-Infrared (NIR).                                                        !
   879)          !---------------------------------------------------------------------------------!
   880) 
   881) 
   882)          !---------------------------------------------------------------------------------!
   883)          !     Save the radiation fluxes to the output variable.                           !
   884)          !---------------------------------------------------------------------------------!
   885)          do i=1,ncoh
   886)             ip1 = i + 1
   887)             nir_beam_flip   (i) = sngloff( down0(ip1) - down0  (i), tiny_offset)
   888)             nir_diff_flip   (i) = sngloff( down (ip1) - down   (i)                         &
   889)                                          + up     (i) - up   (ip1), tiny_offset)
   890)             radprof_flip ( 5,i) = sngloff( down0  (i), tiny_offset)
   891)             radprof_flip ( 6,i) = 0.0
   892)             radprof_flip ( 7,i) = sngloff( down   (i), tiny_offset)
   893)             radprof_flip ( 8,i) = sngloff( up     (i), tiny_offset)
   894)          end do
   895)          !---------------------------------------------------------------------------------!
   896) 
   897) 
   898) 
   899)          !------ Save the fluxes reaching the surface and leaving the top. ----------------!
   900)          dw_nirlo_beam = sngloff(down0          (1), tiny_offset)
   901)          dw_nirlo_diff = sngloff(down           (1), tiny_offset)
