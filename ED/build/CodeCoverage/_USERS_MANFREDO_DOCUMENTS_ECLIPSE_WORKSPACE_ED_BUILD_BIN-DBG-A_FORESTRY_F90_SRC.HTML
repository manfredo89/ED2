     4) !==========================================================================================!
     5) ! Forestry.f90. These subroutines calculate the area to be taken from each patch to meet   !
     6) !               the biomass demand, in case this site is to be harvest using biomass       !
     7) !               targets.                                                                   !                                                                                          !
     8) !                                                                                          !
     9) !               The site is harvested up to a target biomass.  Patches are harvested first !
    10) !               from those above the harvest_age, with equal rates.  If the biomass target !
    11) !               is not met, the remaining biomass is harvested from the patches below the  !
    12) !               minimum age, starting with the oldest.  Harvest targets are taken from the !
    13) !               land use transition matrix dataset, which must be formatted as George      !
    14) !               Hurtt's GLM output.                                                        !
    15) !                                                                                          !
    16) ! NOTICE:  These subroutines have not been thoroughly tested. Please                       !
    17) !          report problems to David Medvigy, medvigy@post.harvard.edu.                     !
    18) !==========================================================================================!
    19) subroutine find_harvest_area(cpoly,isi,onsp,harvestable_agb,pot_area_harv)
    20)    use ed_state_vars        , only : polygontype                & ! structure
    21)                                    , sitetype                   & ! structure
    22)                                    , patchtype                  & ! structure
    23)                                    , allocate_sitetype          & ! subroutine
    24)                                    , deallocate_sitetype        & ! subroutine
    25)                                    , copy_sitetype              ! ! subroutine
    26)    use disturb_coms         , only : ianth_disturb              & ! intent(in)
    27)                                    , lutime                     & ! intent(in)
    28)                                    , min_patch_area             & ! intent(in)
    29)                                    , min_harvest_biomass        ! ! intent(in)
    30)    use fuse_fiss_utils      , only : terminate_patches          ! ! subroutine
    31)    use ed_max_dims          , only : n_pft                      & ! intent(in)
    32)                                    , n_dbh                      ! ! intent(in)
    33)    implicit none
    34)    !----- Arguments -----------------------------------------------------------------------!
    35)    type(polygontype)             , target        :: cpoly
    36)    integer                       , intent(in)    :: isi
    37)    integer                       , intent(in)    :: onsp
    38)    real, dimension(onsp)         , intent(inout) :: harvestable_agb
    39)    real, dimension(onsp)         , intent(inout) :: pot_area_harv
    40)    !----- Local variables -----------------------------------------------------------------!
    41)    type(sitetype)                , pointer       :: csite
    42)    type(patchtype)               , pointer       :: cpatch
    43)    integer                                       :: ipa
    44)    integer                                       :: ico
    45)    integer                                       :: ipft
    46)    integer                                       :: ilu
    47)    real, dimension(n_pft)                        :: mindbh_harvest
    48)    real                                          :: primary_harvest_target
    49)    real                                          :: secondary_harvest_target
    50)    real                                          :: site_harvest_target
    51)    real                                          :: site_harvestable_agb
    52)    real                                          :: area_mature_primary
    53)    real                                          :: hvagb_mature_primary
    54)    real                                          :: area_mature_secondary
    55)    real                                          :: hvagb_mature_secondary
    56)    real                                          :: area_mature_plantation
    57)    real                                          :: hvagb_mature_plantation
    58)    real                                          :: lambda_mature_primary
    59)    real                                          :: lambda_mature_secondary
    60)    real                                          :: lambda_mature_plantation
    61)    real                                          :: harvest_deficit
    62)    !----- Local constants. ----------------------------------------------------------------!
    63)    logical                       , parameter     :: print_detail = .false.
    64)    !---------------------------------------------------------------------------------------!
    65) 
    66) 
    67)    !----- Nothing to do here in case anthropogenic disturbance is turned off. -------------!
    68)    if (ianth_disturb == 0) return
    69)    !---------------------------------------------------------------------------------------!
    70) 
    71) 
    72)    !----- Link to the current site. -------------------------------------------------------!
    73)    csite => cpoly%site(isi)
    74)    !---------------------------------------------------------------------------------------!
    75) 
    76) 
    77) 
    78)    !---------------------------------------------------------------------------------------!
    79)    !      Set biomass targets based on current rates and unapplied harvest from previous   !
    80)    ! years (memory).  These are in kgC/m2.  In case DBH-based logging is applied, then     !
    81)    ! harvest target is set to zero.                                                        !
    82)    !---------------------------------------------------------------------------------------!
    83)    primary_harvest_target   = cpoly%primary_harvest_target  (isi)                          &
    84)                             + cpoly%primary_harvest_memory  (isi)
    85)    secondary_harvest_target = cpoly%secondary_harvest_target(isi)                          &
    86)                             + cpoly%primary_harvest_memory  (isi)
    87)    site_harvest_target      = primary_harvest_target + secondary_harvest_target
    88)    !---------------------------------------------------------------------------------------!
    89) 
    90) 
    91)    !---------------------------------------------------------------------------------------!
    92)    !      Find total harvestable biomass density in kgC/m2.  The harvestable biomass is    !
    93)    ! not necessarily the same as the total site biomass, because it is possible that the   !
    94)    ! demand is for trees above a minimum size (which is very common practice in the        !
    95)    ! tropics).                                                                             !
    96)    !---------------------------------------------------------------------------------------!
    97)    site_harvestable_agb = 0.
    98)    hpat_loop: do ipa=1,onsp
    99)       cpatch => csite%patch(ipa)
   100)       ilu = csite%dist_type(ipa)
   101) 
   102)       !----- Select the minimum DBH depending on the forest category. ---------------------!
   103)       select case(ilu)
   104)       case (1)
   105)          !----- Agriculture.  No timber harvesting here. ----------------------------------!
   106)          mindbh_harvest(:) = huge(1.)
   107)          !---------------------------------------------------------------------------------!
   108)       case (2)
   109)          !----- Forest plantation.  Usually all biomass is cleared. -----------------------!
   110)          mindbh_harvest(:) = 0.
   111)          !---------------------------------------------------------------------------------!
   112)       case (4,5,6)
   113)          !----- Secondary vegetation. -----------------------------------------------------!
   114)          mindbh_harvest(:) = cpoly%mindbh_secondary(:,isi)
   115)          !---------------------------------------------------------------------------------!
   116)       case (3)
   117)          !----- Primary vegetation. -------------------------------------------------------!
   118)          mindbh_harvest(:) = cpoly%mindbh_primary  (:,isi)
   119)          !---------------------------------------------------------------------------------!
   120)       end select
   121)       !------------------------------------------------------------------------------------!
   122) 
   123)       !------------------------------------------------------------------------------------!
   124)       !     Go over each cohort, seek harvestable biomass.                                 !
   125)       !------------------------------------------------------------------------------------!
   126)       hcoh_loop: do ico=1,cpatch%ncohorts
   127)          ipft = cpatch%pft(ico)
   128)          if (cpatch%dbh(ico) >= mindbh_harvest(ipft)) then
   129)             !----- Cohort is harvestable. -------------------------------------------------!
   130)             harvestable_agb(ipa) = harvestable_agb(ipa)                                    &
   131)                                  + cpatch%nplant(ico) * cpatch%agb(ico)
   132)             !------------------------------------------------------------------------------!
   133)          end if
   134)          !---------------------------------------------------------------------------------!
   135)       end do hcoh_loop
   136)       !------------------------------------------------------------------------------------!
   137) 
   138)       !----- Update site biomass only when the patch has sufficient biomass. --------------!
   139)       if (harvestable_agb(ipa) >= min_harvest_biomass) then
   140)          site_harvestable_agb = site_harvestable_agb                                       &
   141)                               + harvestable_agb(ipa) * csite%area(ipa)
   142)       end if
   143)       !------------------------------------------------------------------------------------!
   144)    end do hpat_loop
   145)    !---------------------------------------------------------------------------------------!
   146) 
   147) 
   148) 
   149)    !=======================================================================================!
   150)    !=======================================================================================!
   151)    !     Find out whether any harvest can occur at this site.  For harvest to occur, the   !
   152)    ! site must meet two criteria:                                                          !
   153)    ! a. Harvestable biomass must be greater than min_harvest_biomass;                      !
   154)    ! b. Target biomass must be greater than min_harvest_biomass.                           !
   155)    !                                                                                       !
   156)    !    In case one or both criteria are not met, we add the harvest target to memory.     !
   157)    !---------------------------------------------------------------------------------------!
   158)    if ( site_harvestable_agb <  min_harvest_biomass                 .or.                   &
   159)         site_harvest_target  <= site_harvestable_agb * min_patch_area      ) then
   160)       cpoly%primary_harvest_target  (isi) = 0.0
   161)       cpoly%secondary_harvest_target(isi) = 0.0
   162)       cpoly%primary_harvest_memory  (isi) = primary_harvest_target
   163)       cpoly%secondary_harvest_memory(isi) = secondary_harvest_target
   164)       !------------------------------------------------------------------------------------!
   165) 
   166) 
   167)       !------------------------------------------------------------------------------------!
   168)       !       Print the inventory and the target.                                          !
   169)       !------------------------------------------------------------------------------------!
   170)       if (print_detail) then
   171)          write (unit=*,fmt='(a)'      )     ' '
   172)          write (unit=*,fmt='(a)'      )     '---------------------------------------------'
   173)          write (unit=*,fmt='(a)'      )     ' FORESTRY.  HARVEST WON''T OCCUR THIS YEAR'
   174)          write (unit=*,fmt='(a)'      )     ' '
   175)          write (unit=*,fmt='(a,1x,i5)')     ' ISI                   = ',isi
   176)          write (unit=*,fmt='(a,1x,es12.5)') ' PRIMARY TARGET AGB    = '                    &
   177)                                                          , primary_harvest_target
   178)          write (unit=*,fmt='(a,1x,es12.5)') ' SECONDARY TARGET AGB  = '                    &
   179)                                                          , secondary_harvest_target
   180)          write (unit=*,fmt='(a,1x,es12.5)') ' TOTAL TARGET AGB      = '                    &
   181)                                                          , site_harvest_target
   182)          write (unit=*,fmt='(a,1x,es12.5)') ' TOTAL HARVESTABLE AGB = '                    &
   183)                                                          , site_harvestable_agb
   184)          write (unit=*,fmt='(a)'      )     '---------------------------------------------'
   185)          write (unit=*,fmt='(a)'      )     ' '
   186)       end if
   187)       !------------------------------------------------------------------------------------!
   188)       return
   189)    end if
   190)    !---------------------------------------------------------------------------------------!
   191) 
   192) 
   193) 
   194)    !------ Compute current stocks of agb in mature forests. -------------------------------!
   195)    call inventory_mat_forests(cpoly,isi,onsp,harvestable_agb                               &
   196)                              ,area_mature_primary   ,hvagb_mature_primary                  &
   197)                              ,area_mature_secondary ,hvagb_mature_secondary                &
   198)                              ,area_mature_plantation,hvagb_mature_plantation)
   199)    !---------------------------------------------------------------------------------------!
   200) 
   201) 
   202) 
   203)    !------ Compute the mature-forest harvest rates. ---------------------------------------!
   204)    call mat_forest_harv_rates(hvagb_mature_primary,hvagb_mature_secondary                  &
   205)                              ,hvagb_mature_plantation,primary_harvest_target               &
   206)                              ,secondary_harvest_target,lambda_mature_primary               &
   207)                              ,lambda_mature_secondary,lambda_mature_plantation             &
   208)                              ,harvest_deficit)                                    
   209)    !---------------------------------------------------------------------------------------!
   210) 
   211) 
   212) 
   213)    !---------------------------------------------------------------------------------------!
   214)    !      Compute the area lost by mature patches through harvest.                         !
   215)    !---------------------------------------------------------------------------------------!
   216)    call area_harvest_mature(cpoly,isi,onsp,harvestable_agb,pot_area_harv                   &
   217)                            ,lambda_mature_primary,lambda_mature_secondary                  &
   218)                            ,lambda_mature_plantation)
   219)    !---------------------------------------------------------------------------------------!
   220) 
   221) 
   222) 
   223)    !---------------------------------------------------------------------------------------!
   224)    !      Compute the area lost by immature patches to meet the biomass demand.            !
   225)    !---------------------------------------------------------------------------------------!
   226)    call area_harvest_immature(cpoly,isi,onsp,harvestable_agb,pot_area_harv,harvest_deficit)
   227)    !---------------------------------------------------------------------------------------!
   228) 
   229) 
   230) 
   231)    !---------------------------------------------------------------------------------------!
   232)    !       Print the inventory and the target.                                             !
   233)    !---------------------------------------------------------------------------------------!
   234)    if (print_detail) then
   235)       write (unit=*,fmt='(a)'      )     ' '
   236)       write (unit=*,fmt='(a)'      )     '------------------------------------------------'
   237)       write (unit=*,fmt='(a)'      )     ' FORESTRY.  HARVEST RATES'
   238)       write (unit=*,fmt='(a)'      )     ' '
   239)       write (unit=*,fmt='(a,1x,i5)')     ' ISI                    = ',isi
   240)       write (unit=*,fmt='(a,1x,es12.5)') ' PRIMARY TARGET AGB     = '                      &
   241)                                                       , primary_harvest_target
   242)       write (unit=*,fmt='(a,1x,es12.5)') ' SECONDARY TARGET AGB   = '                      &
   243)                                                       , secondary_harvest_target
   244)       write (unit=*,fmt='(a,1x,es12.5)') ' TOTAL TARGET AGB       = '                      &
   245)                                                       , site_harvest_target
   246)       write (unit=*,fmt='(a,1x,es12.5)') ' TOTAL HARVESTABLE AGB  = '                      &
   247)                                                       , site_harvestable_agb
   248)       write (unit=*,fmt='(a,1x,es12.5)') ' HV AGB (PRIMARY)       = '                      &
   249)                                                       , hvagb_mature_primary
   250)       write (unit=*,fmt='(a,1x,es12.5)') ' HV AGB (SECONDARY)     = '                      &
   251)                                                       , hvagb_mature_secondary
   252)       write (unit=*,fmt='(a,1x,es12.5)') ' HV AGB (PLANTATION)    = '                      &
   253)                                                       , hvagb_mature_plantation
   254)       write (unit=*,fmt='(a,1x,es12.5)') ' HV AREA (PRIMARY)      = '                      &
   255)                                                       , area_mature_primary
   256)       write (unit=*,fmt='(a,1x,es12.5)') ' HV AREA (SECONDARY)    = '                      &
   257)                                                       , area_mature_secondary
   258)       write (unit=*,fmt='(a,1x,es12.5)') ' HV AREA (PLANTATION)   = '                      &
   259)                                                       , area_mature_plantation
   260)       write (unit=*,fmt='(a,1x,es12.5)') ' HV LAMBDA (PRIMARY)    = '                      &
   261)                                                       , lambda_mature_primary
   262)       write (unit=*,fmt='(a,1x,es12.5)') ' HV LAMBDA (SECONDARY)  = '                      &
   263)                                                       , lambda_mature_secondary
   264)       write (unit=*,fmt='(a,1x,es12.5)') ' HV LAMBDA (PLANTATION) = '                      &
   265)                                                       , lambda_mature_plantation
   266)       write (unit=*,fmt='(a)'          ) ' '
   267)       write (unit=*,fmt='(a,1x,es12.5)') ' HARVEST DEFICIT = ', harvest_deficit
   268)       write (unit=*,fmt='(a)'          ) ' '
   269)       write (unit=*,fmt='(a)'          ) '------------------------------------------------'
   270)       write (unit=*,fmt='(5(a,1x))'    ) '  IPA','   LU','         AGE','        AREA'     &
   271)                                                         ,'     HV_AREA'
   272)       write (unit=*,fmt='(a)'      )     '------------------------------------------------'
   273)       do ipa=1,onsp
   274)          write (unit=*,fmt='(2(i5,1x),3(f12.7,1x))') ipa,csite%dist_type(ipa)              &
   275)                                                    ,csite%age(ipa),csite%area(ipa)         &
   276)                                                    ,pot_area_harv(ipa)
   277)       end do
   278)       write (unit=*,fmt='(a)'      )     '------------------------------------------------'
   279)       write (unit=*,fmt='(a)'      )     ' '
   280)    end if
   281)    !---------------------------------------------------------------------------------------!
   282) 
   283) 
   284)    !---------------------------------------------------------------------------------------!
   285)    !      Reset the primary forest memory, and assign any remaining deficit to the         !
   286)    ! secondary forest memory.                                                              !
   287)    !---------------------------------------------------------------------------------------!
   288)    cpoly%primary_harvest_memory  (isi) = 0.0
   289)    cpoly%secondary_harvest_memory(isi) = harvest_deficit
   290)    !---------------------------------------------------------------------------------------!
   291) 
   292)    return
   293) end subroutine find_harvest_area

   304) !    This sub-routine calculates the area and total biomass associated with mature primary !
   305) ! forests, mature secondary forests, and mature forest plantations.                        !
   306) !------------------------------------------------------------------------------------------!
   307) subroutine inventory_mat_forests(cpoly,isi,onsp,harvestable_agb                            &
   308)                                 ,area_mature_primary   , hvagb_mature_primary              &
   309)                                 ,area_mature_secondary , hvagb_mature_secondary            &
   310)                                 ,area_mature_plantation, hvagb_mature_plantation )
   311)    use ed_state_vars , only : polygontype         & ! structure
   312)                             , sitetype            ! ! structure
   313)    use disturb_coms  , only : plantation_rotation & ! intent(in)
   314)                             , mature_harvest_age  & ! intent(in)
   315)                             , min_harvest_biomass ! ! intent(in)
   316)    implicit none
   317)    !----- Arguments -----------------------------------------------------------------------!
   318)    type(polygontype)    , target      :: cpoly
   319)    integer              , intent(in)  :: isi
   320)    integer              , intent(in)  :: onsp
   321)    real, dimension(onsp), intent(in)  :: harvestable_agb
   322)    real                 , intent(out) :: area_mature_primary
   323)    real                 , intent(out) :: hvagb_mature_primary
   324)    real                 , intent(out) :: area_mature_secondary
   325)    real                 , intent(out) :: hvagb_mature_secondary
   326)    real                 , intent(out) :: area_mature_plantation
   327)    real                 , intent(out) :: hvagb_mature_plantation
   328)    !----- Local variables -----------------------------------------------------------------!
   329)    type(sitetype)       , pointer     :: csite
   330)    integer                            :: ipa
   331)    !---------------------------------------------------------------------------------------!
   332) 
   333) 
   334) 
   335)    !----- Initialize inventory. -----------------------------------------------------------!
   336)    area_mature_primary     = 0.0
   337)    area_mature_secondary   = 0.0
   338)    area_mature_plantation  = 0.0
   339)    hvagb_mature_primary    = 0.0
   340)    hvagb_mature_secondary  = 0.0
   341)    hvagb_mature_plantation = 0.0
   342)    !---------------------------------------------------------------------------------------!
   343) 
   344)    csite => cpoly%site(isi)
   345)    patchloop: do ipa=1,csite%npatches
   346) 
   347)       !----- Skip the patch if the biomass is low. ----------------------------------------!
   348)       if (harvestable_agb(ipa) < min_harvest_biomass) cycle patchloop
   349)       !------------------------------------------------------------------------------------!
   350) 
   351) 
   352)       !------------------------------------------------------------------------------------!
   353)       !      Run some checks to determine the type of forest and whether it has reached    !
   354)       ! maturity.  TO DO: check whether George Hurtt's data consider burnt patches primary !
   355)       ! or secondary.                                                                      !
   356)       !------------------------------------------------------------------------------------!
   357)       select case (csite%dist_type(ipa))
   358)       case (2)
   359)          !---- Forest plantation. ---------------------------------------------------------!
   360)          if (csite%age(ipa) > plantation_rotation) then
   361)             area_mature_plantation   = area_mature_plantation + csite%area(ipa)
   362)             hvagb_mature_plantation  = hvagb_mature_plantation                             &
   363)                                      + harvestable_agb(ipa)   * csite%area(ipa)
   364)          end if
   365)          !---------------------------------------------------------------------------------!
   366) 
   367)       case (3)
   368)          !---- Primary forest. ------------------------------------------------------------!
   369)          if (csite%age(ipa) > mature_harvest_age) then
   370)             area_mature_primary      = area_mature_primary + csite%area(ipa)
   371)             hvagb_mature_primary     = hvagb_mature_primary                                &
   372)                                      + harvestable_agb(ipa)   * csite%area(ipa)
   373)          end if
   374)          !---------------------------------------------------------------------------------!
   375) 
   376)       case (4,5,6)
   377)          !---- Secondary forest. ----------------------------------------------------------!
   378)          if (csite%age(ipa) > mature_harvest_age) then
   379)             area_mature_secondary   = area_mature_secondary   + csite%area(ipa)
   380)             hvagb_mature_secondary  = hvagb_mature_secondary                               &
   381)                                     + harvestable_agb(ipa)    * csite%area(ipa)
   382)          end if
   383)          !---------------------------------------------------------------------------------!
   384)       case default
   385)          !----- Probably pasture/croplands, skip the patch. -------------------------------!
   386)          continue
   387)          !---------------------------------------------------------------------------------!
   388)       end select 
   389)       !------------------------------------------------------------------------------------!
   390)    end do patchloop
   391)    !---------------------------------------------------------------------------------------!
   392) 
   393)    return
   394) end subroutine inventory_mat_forests

   403) !==========================================================================================!
   404) !==========================================================================================!
   405) !     This sub-routine adjusts the mature forest disturbance rates associated with         !
   406) ! harvest.  In case the target biomass for primary forests is not met, it first tries to   !
   407) ! take the remaining biomass from old secondary forests before going to immature forests.  !
   408) !------------------------------------------------------------------------------------------!
   409) subroutine mat_forest_harv_rates(hvagb_mature_primary,hvagb_mature_secondary               &
   410)                                 ,hvagb_mature_plantation,primary_harvest_target            &
   411)                                 ,secondary_harvest_target,lambda_mature_primary            &
   412)                                 ,lambda_mature_secondary,lambda_mature_plantation          &
   413)                                 ,harvest_deficit)
   414)    implicit none
   415)    !----- Arguments -----------------------------------------------------------------------!
   416)    real, intent(in)    :: hvagb_mature_primary
   417)    real, intent(in)    :: hvagb_mature_secondary
   418)    real, intent(in)    :: hvagb_mature_plantation
   419)    real, intent(in)    :: primary_harvest_target
   420)    real, intent(inout) :: secondary_harvest_target
   421)    real, intent(out)   :: lambda_mature_primary
   422)    real, intent(out)   :: lambda_mature_plantation
   423)    real, intent(out)   :: lambda_mature_secondary
   424)    real, intent(out)   :: harvest_deficit
   425)    !---------------------------------------------------------------------------------------!
   426) 
   427) 
   428)    !---------------------------------------------------------------------------------------!
   429)    !    Find harvesting rate in mature primary vegetation.  In case there is not enough    !
   430)    ! biomass harvest, harvest all primary vegetation then add the unmet biomass to the     !
   431)    ! target for secondary vegetation.                                                      !
   432)    !---------------------------------------------------------------------------------------!
   433)    if (hvagb_mature_primary > primary_harvest_target) then
   434)       lambda_mature_primary = primary_harvest_target / hvagb_mature_primary
   435)    else
   436)       lambda_mature_primary    = 1.0
   437)       harvest_deficit          = primary_harvest_target   - hvagb_mature_primary
   438)       secondary_harvest_target = secondary_harvest_target + harvest_deficit
   439)    end if
   440)    !---------------------------------------------------------------------------------------!
   441) 
   442) 
   443) 
   444)    !---------------------------------------------------------------------------------------!
   445)    !      Find harvesting rate for mature plantations and mature secondary forests.        !
   446)    ! First try to remove all biomass from plantations.  In case there isn't sufficient     !
   447)    ! biomass, harvest all plantations then remove the unmet biomass from mature secondary  !
   448)    ! vegetation.  In case there isn't enough biomass, leave the remaining target in        !
   449)    ! harvest_deficit, which will be used to determine harvesting from immature forests.    !
   450)    !---------------------------------------------------------------------------------------!
   451)    if (hvagb_mature_plantation > secondary_harvest_target) then
   452)       lambda_mature_plantation = secondary_harvest_target / hvagb_mature_plantation
   453)       lambda_mature_secondary  = 0.0
   454)       harvest_deficit          = 0.0
   455)    else
   456)       lambda_mature_plantation = 1.0
   457)       harvest_deficit          = secondary_harvest_target - hvagb_mature_plantation
   458) 
   459)       if(hvagb_mature_secondary > harvest_deficit) then
   460)          lambda_mature_secondary = harvest_deficit / hvagb_mature_secondary
   461)          harvest_deficit         = 0.0
   462)       else
   463)          lambda_mature_secondary = 1.0
   464)          harvest_deficit         = harvest_deficit - hvagb_mature_secondary
   465)       end if
   466)    end if
   467)    !---------------------------------------------------------------------------------------!
   468) 
   469)    return
   470) end subroutine mat_forest_harv_rates

   481) !    This subroutine finds the area of mature patches that is going to be harvested.       !
   482) !------------------------------------------------------------------------------------------!
   483) subroutine area_harvest_mature(cpoly,isi,onsp,harvestable_agb,pot_area_harv                &
   484)                               ,lambda_mature_primary,lambda_mature_secondary               &
   485)                               ,lambda_mature_plantation)
   486)    use ed_state_vars    , only : polygontype          & ! structure
