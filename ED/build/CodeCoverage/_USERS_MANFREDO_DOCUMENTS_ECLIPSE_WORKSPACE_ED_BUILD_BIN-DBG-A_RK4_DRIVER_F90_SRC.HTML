     9)    !=======================================================================================!

    11)    !      Main driver of short-time scale dynamics of the Runge-Kutta integrator           !
    12)    !      for the land surface model.                                                      !
    13)    !---------------------------------------------------------------------------------------!
    14)    subroutine rk4_timestep(cgrid)
    15)       use rk4_integ_utils
    16)       use soil_respiration_module
    17)       use photosyn_driv
    18)       use rk4_misc
    19)       use update_derived_props_module
    20)       use rk4_coms               , only : integration_vars     & ! structure
    21)                                         , rk4patchtype         & ! structure
    22)                                         , zero_rk4_patch       & ! subroutine
    23)                                         , zero_rk4_cohort      & ! subroutine
    24)                                         , integration_buff     ! ! intent(out)
    25)       use ed_state_vars          , only : edtype               & ! structure
    26)                                         , polygontype          & ! structure
    27)                                         , sitetype             & ! structure
    28)                                         , patchtype            ! ! structure
    29)       use met_driver_coms        , only : met_driv_state       ! ! structure
    30)       use grid_coms              , only : nzg                  & ! intent(in)
    31)                                         , nzs                  ! ! intent(in)
    32)       use ed_misc_coms           , only : current_time         & ! intent(in)
    33)                                         , dtlsm                ! ! intent(in)
    34)       use therm_lib              , only : tq2enthalpy          ! ! function
    35)       use budget_utils           , only : update_budget        & ! function
    36)                                         , compute_budget       ! ! function
    37)       !$ use omp_lib
    38)       implicit none
    39) 
    40)       !----- Arguments --------------------------------------------------------------------!
    41)       type(edtype)              , target      :: cgrid
    42)       !----- Local variables --------------------------------------------------------------!
    43)       type(polygontype)         , pointer     :: cpoly
    44)       type(sitetype)            , pointer     :: csite
    45)       type(patchtype)           , pointer     :: cpatch
    46)       type(met_driv_state)      , pointer     :: cmet
    47) 
    48)       type(rk4patchtype)       , pointer      :: initp
    49)       type(rk4patchtype)       , pointer      :: yscal
    50)       type(rk4patchtype)       , pointer      :: y
    51)       type(rk4patchtype)       , pointer      :: dydx
    52)       type(rk4patchtype)       , pointer      :: yerr
    53)       type(rk4patchtype)       , pointer      :: ytemp
    54)       type(rk4patchtype)       , pointer      :: ak2
    55)       type(rk4patchtype)       , pointer      :: ak3
    56)       type(rk4patchtype)       , pointer      :: ak4
    57)       type(rk4patchtype)       , pointer      :: ak5
    58)       type(rk4patchtype)       , pointer      :: ak6
    59)       type(rk4patchtype)       , pointer      :: ak7
    60)       integer                                 :: ipy
    61)       integer                                 :: isi
    62)       integer                                 :: ipa
    63)       integer                                 :: nsteps
    64)       integer                                 :: imon
    65)       real                                    :: wcurr_loss2atm
    66)       real                                    :: ecurr_netrad
    67)       real                                    :: ecurr_loss2atm
    68)       real                                    :: co2curr_loss2atm
    69)       real                                    :: wcurr_loss2drainage
    70)       real                                    :: ecurr_loss2drainage
    71)       real                                    :: wcurr_loss2runoff
    72)       real                                    :: ecurr_loss2runoff
    73)       real                                    :: old_can_enthalpy
    74)       real                                    :: old_can_shv
    75)       real                                    :: old_can_co2
    76)       real                                    :: old_can_rhos
    77)       real                                    :: old_can_temp
    78)       real                                    :: old_can_prss
    79)       real                                    :: patch_vels
    80)       integer                                 :: ibuff
    81)       !----- Local constants. -------------------------------------------------------------!
    82)       logical                   , parameter   :: test_energy_sanity = .false.
    83)       !----- Functions --------------------------------------------------------------------!
    84)       real                      , external    :: walltime
    85)       !------------------------------------------------------------------------------------!
    86) 
    87) 
    88) 
    89) 
    90)       polygonloop: do ipy = 1,cgrid%npolygons
    91)          cpoly => cgrid%polygon(ipy)
    92) 
    93)          siteloop: do isi = 1,cpoly%nsites
    94)             csite => cpoly%site(isi)
    95)             cmet  => cpoly%met(isi)
    96) 
    97)             !------------------------------------------------------------------------------!
    98)             !     Update the monthly rainfall.                                             !
    99)             !------------------------------------------------------------------------------!
   100)             imon                             = current_time%month
   101)             cpoly%avg_monthly_pcpg(imon,isi) = cpoly%avg_monthly_pcpg(imon,isi)            &
   102)                                              + cmet%pcpg * dtlsm
   103)             !------------------------------------------------------------------------------!
   104) 
   105) 
   106) 
   107)             !------------------------------------------------------------------------------!
   108)             !    Copy the meteorological variables to the rk4site structure.               !
   109)             !------------------------------------------------------------------------------!
   110)             call copy_met_2_rk4site(nzg,cmet%atm_ustar,cmet%atm_theiv,cmet%atm_vpdef       &
   111)                                    ,cmet%atm_theta,cmet%atm_tmp,cmet%atm_shv,cmet%atm_co2  &
   112)                                    ,cmet%geoht,cmet%exner,cmet%pcpg,cmet%qpcpg,cmet%dpcpg  &
   113)                                    ,cmet%prss,cmet%rshort,cmet%rlong,cmet%par_beam         &
   114)                                    ,cmet%par_diffuse,cmet%nir_beam,cmet%nir_diffuse        &
   115)                                    ,cmet%geoht,cpoly%lsl(isi),cpoly%ntext_soil(:,isi)      &
   116)                                    ,cpoly%green_leaf_factor(:,isi),cgrid%lon(ipy)          &
   117)                                    ,cgrid%lat(ipy),cgrid%cosz(ipy))
   118)             !------------------------------------------------------------------------------!
   119) 
   120)             !$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(      &
   121)             !$OMP initp,yscal,y,dydx,yerr,ytemp,ak2,ak3,    &
   122)             !$OMP ak4,ak5,ak6,ak7,cpatch,patch_vels,        &
   123)             !$OMP old_can_co2,old_can_rhos,old_can_temp,    &
   124)             !$OMP old_can_prss,old_can_enthalpy,            &
   125)             !$OMP old_can_shv,ecurr_netrad,                 &
   126)             !$OMP wcurr_loss2atm,ecurr_loss2atm,            &
   127)             !$OMP co2curr_loss2atm,                         &
   128)             !$OMP wcurr_loss2drainage,ecurr_loss2drainage,  &
   129)             !$OMP wcurr_loss2runoff,ecurr_loss2runoff,nsteps )
   130) 
   131)             patchloop: do ipa = 1,csite%npatches
   132)                cpatch => csite%patch(ipa)
   133) 
   134)                ibuff = 1
   135)                !$ ibuff = OMP_get_thread_num()+1
   136) 
   137)                initp => integration_buff(ibuff)%initp
   138)                yscal => integration_buff(ibuff)%yscal
   139)                y     => integration_buff(ibuff)%y
   140)                dydx  => integration_buff(ibuff)%dydx
   141)                yerr  => integration_buff(ibuff)%yerr
   142)                ytemp => integration_buff(ibuff)%ytemp
   143)                ak2   => integration_buff(ibuff)%ak2
   144)                ak3   => integration_buff(ibuff)%ak3
   145)                ak4   => integration_buff(ibuff)%ak4
   146)                ak5   => integration_buff(ibuff)%ak5
   147)                ak6   => integration_buff(ibuff)%ak6
   148)                ak7   => integration_buff(ibuff)%ak7
   149) 
   150) 
   151) 
   152)                !----- Reset all buffers to zero, as a safety measure. ---------------------!
   153)                call zero_rk4_patch(initp)
   154)                call zero_rk4_patch(yscal)
   155)                call zero_rk4_patch(y)
   156)                call zero_rk4_patch(dydx)
   157)                call zero_rk4_patch(yerr)
   158)                call zero_rk4_patch(ytemp)
   159)                call zero_rk4_patch(ak2)
   160)                call zero_rk4_patch(ak3)
   161)                call zero_rk4_patch(ak4)
   162)                call zero_rk4_patch(ak5)
   163)                call zero_rk4_patch(ak6)
   164)                call zero_rk4_patch(ak7)
   165)                call zero_rk4_cohort(initp)
   166)                call zero_rk4_cohort(yscal)
   167)                call zero_rk4_cohort(y)
   168)                call zero_rk4_cohort(dydx)
   169)                call zero_rk4_cohort(yerr)
   170)                call zero_rk4_cohort(ytemp)
   171)                call zero_rk4_cohort(ak2)
   172)                call zero_rk4_cohort(ak3)
   173)                call zero_rk4_cohort(ak4)
   174)                call zero_rk4_cohort(ak5)
   175)                call zero_rk4_cohort(ak6)
   176)                call zero_rk4_cohort(ak7)
   177) 
   178)                !----- Get velocity for aerodynamic resistance. ----------------------------!
   179)                if (csite%can_theta(ipa) < cmet%atm_theta) then
   180)                   patch_vels = cmet%vels_stab
   181)                else
   182)                   patch_vels = cmet%vels_unstab
   183)                end if
   184)                !---------------------------------------------------------------------------!
   185) 
   186)                !---------------------------------------------------------------------------!
   187)                !    Update roughness and canopy depth.                                     !
   188)                !---------------------------------------------------------------------------!
   189)                call update_patch_thermo_props(csite,ipa,ipa,nzg,nzs                        &
   190)                                              ,cpoly%ntext_soil(:,isi))
   191)                call update_patch_derived_props(csite,ipa)
   192)                !---------------------------------------------------------------------------!
   193) 
   194) 
   195)                !----- Save the previous thermodynamic state. ------------------------------!
   196)                old_can_shv      = csite%can_shv  (ipa)
   197)                old_can_co2      = csite%can_co2  (ipa)
   198)                old_can_rhos     = csite%can_rhos (ipa)
   199)                old_can_temp     = csite%can_temp (ipa)
   200)                old_can_prss     = csite%can_prss (ipa)
   201)                old_can_enthalpy = tq2enthalpy(csite%can_temp(ipa),csite%can_shv(ipa),.true.)
   202)                !---------------------------------------------------------------------------!
   203) 
   204)                !----- Compute current storage terms. --------------------------------------!
   205)                call update_budget(csite,cpoly%lsl(isi),ipa)
   206)                !---------------------------------------------------------------------------!
   207) 
   208) 
   209) 
   210)                !---------------------------------------------------------------------------!
   211)                !      Test whether temperature and energy are reasonable.                  !
   212)                !---------------------------------------------------------------------------!
   213)                if (test_energy_sanity) then
   214)                   call sanity_check_veg_energy(csite,ipa)
   215)                end if
   216)                !---------------------------------------------------------------------------!
   217) 
   218)                !---------------------------------------------------------------------------!
   219)                !     Set up the integration patch.                                         !
   220)                !---------------------------------------------------------------------------!
   221)                call copy_patch_init(csite,ipa,initp,patch_vels)
   222)                !---------------------------------------------------------------------------!
   223) 
   224) 
   225)                !----- Get photosynthesis, stomatal conductance, and transpiration. --------!
   226)                call canopy_photosynthesis(csite,cmet,nzg,ipa,cpoly%ntext_soil(:,isi)       &
   227)                                          ,cpoly%leaf_aging_factor(:,isi)                   &
   228)                                          ,cpoly%green_leaf_factor(:,isi))
   229)                !---------------------------------------------------------------------------!
   230) 
   231) 
   232)                !----- Compute root and heterotrophic respiration. -------------------------!
   233)                call soil_respiration(csite,ipa,nzg,cpoly%ntext_soil(:,isi))
   234)                !---------------------------------------------------------------------------!
   235) 
   236) 
   237)                !---------------------------------------------------------------------------!
   238)                !     Set up the integration patch.                                         !
   239)                !---------------------------------------------------------------------------!
   240)                call copy_patch_init_carbon(csite,ipa,initp)
   241)                !---------------------------------------------------------------------------!
   242) 
   243) 
   244)                !---------------------------------------------------------------------------!
   245)                !    This is the driver for the integration process...                      !
   246)                !---------------------------------------------------------------------------!
   247)                call integrate_patch_rk4(csite,initp,ipa,isi                                &
   248)                                        ,cpoly%nighttime(isi),wcurr_loss2atm                &
   249)                                        ,ecurr_netrad,ecurr_loss2atm,co2curr_loss2atm       &
   250)                                        ,wcurr_loss2drainage,ecurr_loss2drainage            &
   251)                                        ,wcurr_loss2runoff,ecurr_loss2runoff,nsteps)
   252)                !---------------------------------------------------------------------------!
   253) 
   254) 
   255)                !----- Add the number of steps into the step counter. ----------------------!
   256)                !----- workload accumulation is order-independent, so this can stay shared
   257)                cgrid%workload(13,ipy) = cgrid%workload(13,ipy) + real(nsteps)
   258)                !---------------------------------------------------------------------------!
   259) 
   260) 
   261)                !---------------------------------------------------------------------------!
   262)                !    Update the minimum monthly temperature, based on canopy temperature.   !
   263)                !---------------------------------------------------------------------------!
   264)                if (cpoly%site(isi)%can_temp(ipa) < cpoly%min_monthly_temp(isi)) then
   265)                   cpoly%min_monthly_temp(isi) = cpoly%site(isi)%can_temp(ipa)
   266)                end if
   267)                !---------------------------------------------------------------------------!
   268) 
   269) 
   270)                !---------------------------------------------------------------------------!
   271)                !     Compute the residuals.                                                !
   272)                !---------------------------------------------------------------------------!
   273)                call compute_budget(csite,cpoly%lsl(isi),cmet%pcpg,cmet%qpcpg,ipa           &
   274)                                   ,wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm              &
   275)                                   ,co2curr_loss2atm,wcurr_loss2drainage                    &
   276)                                   ,ecurr_loss2drainage,wcurr_loss2runoff,ecurr_loss2runoff &
   277)                                   ,cpoly%area(isi),cgrid%cbudget_nep(ipy),old_can_enthalpy &
   278)                                   ,old_can_shv,old_can_co2,old_can_rhos,old_can_prss)
   279)                !---------------------------------------------------------------------------!
   280)             end do patchloop
   281)             !$OMP END PARALLEL DO
   282) 
   283)             !------------------------------------------------------------------------------!
   284)          end do siteloop
   285)          !---------------------------------------------------------------------------------!
   286)       end do polygonloop
   287)       !------------------------------------------------------------------------------------!
   288) 
   289)       return
   290)    end subroutine rk4_timestep

   300)    !=======================================================================================!
   301)    !     This subroutine will drive the integration process.                               !
   302)    !---------------------------------------------------------------------------------------!
   303)    subroutine integrate_patch_rk4(csite,initp,ipa,isi,nighttime,wcurr_loss2atm             &
   304)                                  ,ecurr_netrad,ecurr_loss2atm,co2curr_loss2atm             &
   305)                                  ,wcurr_loss2drainage,ecurr_loss2drainage                  &
   306)                                  ,wcurr_loss2runoff,ecurr_loss2runoff,nsteps)
   307)       use rk4_integ_utils
   308)       use ed_state_vars   , only : sitetype             & ! structure
   309)                                  , patchtype            ! ! structure
   310)       use rk4_coms        , only : integration_vars     & ! structure
   311)                                  , rk4patchtype         & ! structure
   312)                                  , zero_rk4_patch       & ! subroutine
   313)                                  , zero_rk4_cohort      & ! subroutine
   314)                                  , tbeg                 & ! intent(inout)
   315)                                  , tend                 & ! intent(inout)
   316)                                  , dtrk4i               ! ! intent(inout)
   317)       implicit none
   318)       !----- Arguments --------------------------------------------------------------------!
   319)       type(sitetype)        , target      :: csite
   320)       type(rk4patchtype)    , target      :: initp
   321)       integer               , intent(in)  :: ipa
   322)       integer               , intent(in)  :: isi
   323)       logical               , intent(in)  :: nighttime
   324)       real                  , intent(out) :: wcurr_loss2atm
   325)       real                  , intent(out) :: ecurr_netrad
   326)       real                  , intent(out) :: ecurr_loss2atm
   327)       real                  , intent(out) :: co2curr_loss2atm
   328)       real                  , intent(out) :: wcurr_loss2drainage
   329)       real                  , intent(out) :: ecurr_loss2drainage
   330)       real                  , intent(out) :: wcurr_loss2runoff
   331)       real                  , intent(out) :: ecurr_loss2runoff
   332)       integer               , intent(out) :: nsteps
   333)       !----- Local variables --------------------------------------------------------------!
   334)       real(kind=8)                        :: hbeg
   335)       !------------------------------------------------------------------------------------!
   336) 
   337)       !------------------------------------------------------------------------------------!
   338)       !      Initial step size.  Experience has shown that giving this too large a value   !
   339)       ! causes the integrator to fail (e.g., soil layers become supersaturated).           !
   340)       !------------------------------------------------------------------------------------!
   341)       hbeg = dble(csite%htry(ipa))
   342) 
   343) 
   344)       !------------------------------------------------------------------------------------!
   345)       !     Zero the canopy-atmosphere flux values.  These values are updated every dtlsm, !
   346)       ! so they must be zeroed at each call.                                               !
   347)       !------------------------------------------------------------------------------------!
   348)       initp%upwp = 0.d0
   349)       initp%tpwp = 0.d0
   350)       initp%qpwp = 0.d0
   351)       initp%cpwp = 0.d0
   352)       initp%wpwp = 0.d0
   353) 
   354)       !----- Go into the ODE integrator. --------------------------------------------------!
   355) 
   356)       call odeint(hbeg,csite,ipa,isi,nsteps)
   357) 
   358)       !------------------------------------------------------------------------------------!
   359)       !      Normalize canopy-atmosphere flux values.  These values are updated every      !
   360)       ! dtlsm, so they must be normalized every time.                                      !
   361)       !------------------------------------------------------------------------------------!
   362)       initp%upwp = initp%can_rhos * initp%upwp * dtrk4i
   363)       initp%tpwp = initp%can_rhos * initp%tpwp * dtrk4i
   364)       initp%qpwp = initp%can_rhos * initp%qpwp * dtrk4i
   365)       initp%cpwp = initp%can_rhos * initp%cpwp * dtrk4i
   366)       initp%wpwp = initp%can_rhos * initp%wpwp * dtrk4i
   367) 
   368) 
   369)       !------------------------------------------------------------------------------------!
   370)       ! Move the state variables from the integrated patch to the model patch.             !
   371)       !------------------------------------------------------------------------------------!
   372)       call initp2modelp(tend-tbeg,initp,csite,ipa,nighttime,wcurr_loss2atm,ecurr_netrad    &
   373)                        ,ecurr_loss2atm,co2curr_loss2atm,wcurr_loss2drainage                &
   374)                        ,ecurr_loss2drainage,wcurr_loss2runoff,ecurr_loss2runoff)
   375) 
   376)       return
   377)    end subroutine integrate_patch_rk4

   413)                                       , dslz8                & ! intent(in)

   419)                                       , vpdefil              & ! subroutine

   421)                                       , cmtl2uext            & ! subroutine

   426)                                       , k_fire_first         ! ! intent(in)

   429)       type(rk4patchtype), target      :: initp
   430)       type(sitetype)    , target      :: csite
   431)       real(kind=8)      , intent(in)  :: hdid
