     1) !==========================================================================================!
     2) !==========================================================================================!
     3) !      This sub-routine creates the output files based on the variable table specification !
     4) ! and the type of output file to be created.  ED's default format is HDF5.                 !
     5) !------------------------------------------------------------------------------------------!
     6) subroutine h5_output(vtype)
     7) #if USE_HDF5
     8)    use hdf5
     9) #endif
    10)    use an_header
    11)   
    12)    use ed_var_tables, only : vt_info               & ! intent(in)
    13)                            , var_table             & ! intent(in)
    14)                            , var_table_vector      & ! intent(in)
    15)                            , num_var               ! ! intent(in)
    16)    use ed_misc_coms , only : ffilout               & ! intent(in)
    17)                            , sfilout               & ! intent(in)
    18)                            , itimea                & ! intent(in)
    19)                            , idatea                & ! intent(in)
    20)                            , imontha               & ! intent(in)
    21)                            , iyeara                & ! intent(in)
    22)                            , iclobber              & ! intent(in)
    23)                            , nrec_fast             & ! intent(in)
    24)                            , irec_fast             & ! intent(in)
    25)                            , out_time_fast         & ! intent(in)
    26)                            , outfast               & ! intent(in)
    27)                            , frqfast               ! ! intent(in)
    28)    use ed_misc_coms , only : attach_metadata       ! ! intent(in)
    29)    use grid_coms    , only : ngrids                & ! intent(in)
    30)                            , time                  ! ! intent(in)
    31)    use hdf5_coms    , only : chnkdims              & ! intent(in)
    32)                            , chnkoffs              & ! intent(in)
    33)                            , cnt                   & ! intent(in)
    34)                            , stride                & ! intent(in)
    35)                            , globdims              ! ! intent(in)
    36)    use ed_node_coms , only : mynum                 & ! intent(in)
    37)                            , nnodetot              ! ! intent(in)
    38)    use ed_max_dims  , only : n_pft                 & ! intent(in)
    39)                            , n_dist_types          & ! intent(in)
    40)                            , n_dbh                 & ! intent(in)
    41)                            , maxgrds               & ! intent(in)
    42)                            , str_len               ! ! intent(in)
    43)    use ed_state_vars, only : edtype                & ! structure
    44)                            , polygontype           & ! structure
    45)                            , sitetype              & ! structure
    46)                            , patchtype             & ! structure
    47)                            , gdpy                  ! ! intent(in)
    48)    implicit none
    49) 
    50)    !------ Include standard common blocks. ------------------------------------------------!
    51) #if defined(RAMS_MPI)
    52)    include 'mpif.h'
    53) #endif
    54)    !------ Arguments. ---------------------------------------------------------------------!
    55)    character(len=*)                                 , intent(in) :: vtype
    56)    !------ Local variables. ---------------------------------------------------------------!
    57)    character(len=str_len)                                        :: anamel
    58)    character(len=3)                                              :: cgrid
    59)    character(len=40)                                             :: subaname
    60)    character(len=64)                                             :: varn
    61)    character(len=1)                                              :: vnam
    62)    character(len=64)                                             :: c0
    63)    character(len=64)     , dimension(3)                          :: metadata
    64) #if defined(RAMS_MPI)
    65)    integer               ,dimension(MPI_STATUS_SIZE)             :: status
    66) #endif
    67)    integer                                                       :: ngr
    68)    integer                                                       :: nv
    69)    integer                                                       :: nvcnt
    70)    integer                                                       :: iptr
    71)    integer                                                       :: outyear
    72)    integer                                                       :: outmonth
    73)    integer                                                       :: outdate
    74)    integer                                                       :: outhour
    75)    integer                                                       :: irec
    76)    integer                                                       :: nrec
    77)    integer                                                       :: comm
    78)    integer                                                       :: info
    79)    logical                                                       :: exans
    80)    logical                                                       :: new_file
    81)    real(kind=8)                                                  :: dsec
    82)    !------ HDF specific data types. -------------------------------------------------------!
    83)    integer                                                       :: hdferr
    84)    integer                                                       :: dsetrank
    85)    integer(HID_T)                                                :: file_id
    86)    integer(HID_T)                                                :: dset_id
    87)    integer(HID_T)                                                :: memspace
    88)    integer(HID_T)                                                :: filespace
    89)    integer(HID_T)                                                :: attr_id
    90)    !------ Attribute types. ---------------------------------------------------------------!
    91)    integer(HID_T)                                                :: aspace_id
    92)    integer(HID_T)                                                :: atype_id
    93)    integer(HSIZE_T)      , dimension(1)                          :: adims
    94)    integer                                                       :: arank
    95)    integer(SIZE_T)                                               :: attrlen
    96)    !----- Locally saved variables. --------------------------------------------------------!
    97)    integer                                          , save       :: irec_opt       = 0
    98)    !----- Local constants. ----------------------------------------------------------------!
    99)    logical                                          , parameter  :: collective_mpi = .false.
   100)    logical                                          , parameter  :: verbose        = .false.
   101)    real(kind=8)                                     , parameter  :: zero           = 0.0d0
   102)    !----- External functions. -------------------------------------------------------------!
   103)    logical                                          , external   :: isleap
   104)    !---------------------------------------------------------------------------------------!
   105) 
   106) 
   107) 
   108)    !---------------------------------------------------------------------------------------!
   109)    !   Start with some banner just to make sure we got in here.                            !
   110)    !---------------------------------------------------------------------------------------!
   111)    if (verbose) then
   112)       write (unit=*,fmt='(a,1x,a,a,1x,i6,a)')                                              &
   113)                                  '+ HDF5.  Analysis:',trim(vtype),'.  Node:',mynum,'...'
   114)    end if
   115)    !---------------------------------------------------------------------------------------!
   116) 
   117) 
   118) 
   119)    !----- Initialise some variables. ------------------------------------------------------!
   120) #if defined(RAMS_MPI)
   121)    comm    = MPI_COMM_WORLD
   122)    info    = MPI_INFO_NULL
   123) #else
   124)    comm    = 0
   125)    info    = 0
   126) #endif
   127)    !---------------------------------------------------------------------------------------!
   128) 
   129) 
   130) 
   131)    !------ Find which letter we should use to denote this type of analysis. ---------------!
   132)    select case (trim(vtype))
   133)    case ('INST')
   134)       vnam='I'   ! Instantaneous analysis.
   135)    case ('DAIL') 
   136)       vnam='D'   ! Daily averages.
   137)    case ('MONT') 
   138)       vnam='E'   ! Monthly averages.
   139)    case ('DCYC')
   140)       vnam='Q'   ! Mean diurnal cycle
   141)    case ('YEAR') 
   142)       vnam='Y'   ! Yearly averages.
   143)    case ('OPTI') 
   144)       vnam='T'   ! Ouput for optimisation.
   145)    case ('HIST') 
   146)       vnam='S'   ! S for reStart, don't want confusion with RAMS' H or R files
   147)    case ('CONT') 
   148)       vnam='Z'   ! The first time with history start, so we don't replace the history
   149)    end select  
   150)    !---------------------------------------------------------------------------------------!
   151) 
   152) 
   153)    nvcnt = 0
   154)    nrec  = 1
   155)    irec  = 1  
   156) 
   157)    !---------------------------------------------------------------------------------------!
   158)    !    Loop over the grids.                                                               !
   159)    !---------------------------------------------------------------------------------------!
   160)    gridloop: do ngr=1,ngrids
   161) 
   162)       if (verbose) write (unit=*,fmt='(2(a,1x,i6),a)') '  * Grid:',ngr,'. Node:',mynum,'...'
   163) 
   164)       !----- I guess this cleans out anything that didn't finish correctly. ---------------!
   165)       if (verbose) write (unit=*,fmt='(a)') '  * Collect garbage...'
   166)       call h5garbage_collect_f(hdferr) 
   167)       !------------------------------------------------------------------------------------!
   168) 
   169) 
   170)       !----- Wait until the previous node has finished writing. ---------------------------!
   171)       new_file=.true.
   172) #if defined(RAMS_MPI)
   173)       if (mynum /= 1) then
   174)          call MPI_Recv(new_file,1,MPI_LOGICAL,recvnum,3510+ngr,MPI_COMM_WORLD,status,ierr)
   175)       end if
   176) #endif
   177)       !------------------------------------------------------------------------------------!
   178) 
   179) 
   180) 
   181)       !------------------------------------------------------------------------------------!
   182)       !     Print a check about the file status.                                           !
   183)       !------------------------------------------------------------------------------------!
   184) !      if (verbose) then
   185) !         write (unit=*,fmt='(a,1x,a,2(a,1x,i6),a,1x,l,a)')                                 &
   186) !               '  * HDF5.  Type:',trim(vtype),'. Node:',mynum,'. Grid: ',ngr               &
   187) !              ,'. New File:',new_file,'...'
   188) !      end if
   189)       !------------------------------------------------------------------------------------!
   190) 
   191) 
   192) 
   193)       !------------------------------------------------------------------------------------!
   194)       !    If there are no polygons on this node, then the node shouldn't do anything to   !
   195)       ! this file.  This is something that happens only in coupled model runs.             !
   196)       !------------------------------------------------------------------------------------!
   197)       if (gdpy(mynum,ngr) > 0) then
   198)          if (verbose) then
   199)             write (unit=*,fmt='(a,1x,i6,1x,a)') '    > Writing ',gdpy(mynum,ngr)           &
   200)                                                ,'polygons...'
   201)          end if
   202) 
   203)          !----- Make the grid flag. -------------------------------------------------------!
   204)          write(cgrid,fmt='(a1,i2.2)') 'g',ngr
   205)          !---------------------------------------------------------------------------------!
   206) 
   207) 
   208)          !---------------------------------------------------------------------------------!
   209)          !     Create the file name.                                                       !
   210)          !---------------------------------------------------------------------------------!
   211)          select case (trim(vtype))
   212)          case ('DAIL')
   213)             !----- Return the year, month, and day of the previous day. -------------------!
   214)             call date_add_to(iyeara,imontha,idatea,itimea*100,time-21600.0d0,'s',outyear   &
   215)                             ,outmonth,outdate,outhour)
   216)             !------------------------------------------------------------------------------!
   217)             !     Set outhour to be zero, it was pushed backwards just to get the previous !
   218)             ! day.                                                                         !
   219)             !------------------------------------------------------------------------------!
   220)             outhour = 0
   221)             !----- Make the file name. ----------------------------------------------------!
   222)             call makefnam(anamel,ffilout,zero,outyear,outmonth,outdate,outhour,vnam,cgrid  &
   223)                          ,'h5 ')
   224) 
   225)          case ('MONT','DCYC')
   226)             !----- Return the year, month, and day of the previous day. -------------------!
   227)             call date_add_to(iyeara,imontha,idatea,itimea*100,time-21600.0d0,'s',outyear   &
   228)                             ,outmonth,outdate,outhour)
   229)             !------------------------------------------------------------------------------!
   230)             !     Set outhour and outdate to zero.                                         !
   231)             !------------------------------------------------------------------------------!
   232)             outhour = 0
   233)             outdate = 0
   234)             !----- Make the file name. ----------------------------------------------------!
   235)             call makefnam(anamel,ffilout,zero,outyear,outmonth,outdate,outhour,vnam,cgrid   &
   236)                          ,'h5 ')
   237) 
   238)          case ('YEAR')
   239)             !----- Return the year, month, and day of the previous day. -------------------!
   240)             call date_add_to(iyeara,imontha,idatea,itimea*100,time-21600.0d0,'s',outyear   &
   241)                             ,outmonth,outdate,outhour)
   242)             !------------------------------------------------------------------------------!
   243)             !     Set outhour, outdate, and outmonth to zero.                              !
   244)             !------------------------------------------------------------------------------!
   245)             outhour  = 0
   246)             outdate  = 0
   247)             outmonth = 0
   248)             !----- Make the file name. ----------------------------------------------------!
   249)             call makefnam(anamel,ffilout,zero,outyear,outmonth,outdate,outhour,vnam,cgrid  &
   250)                          ,'h5 ')
   251) 
   252)          case ('OPTI')
   253)             !------ Make the optimisation file. -------------------------------------------!
   254)             new_file = .false.
   255)             irec_opt = irec_opt + 1
   256)             call date_add_to(iyeara,imontha,idatea,itimea*100,time-21600.0d0,'s',outyear   &
   257)                             ,outmonth,outdate,outhour)
   258) 
   259)             if (isleap(outyear)) then
   260)                nrec = int(366*86400/frqfast)
   261)             else
   262)                nrec = int(365*86400/frqfast)
   263)             end if
   264) 
   265)             call date_add_to(iyeara,imontha,idatea,itimea*100,time-21600.0d0,'s',outyear   &
   266)                             ,outmonth,outdate,outhour)
   267)            
   268)             call makefnam(anamel,ffilout,zero,outyear,0,0,0,vnam,cgrid,'h5 ')
   269) 
   270)             if (outmonth == 1 .and. outdate == 1 .and. outhour == 0) then
   271)                new_file = .true.
   272)                irec_opt = 1
   273)             end if
   274)             irec  = irec_opt
   275)             if (irec == 1) new_file = .true.
   276) 
   277)          case ('HIST','CONT')
   278)          
   279)             call makefnam(anamel,sfilout,time,iyeara,imontha,idatea,itimea*100,vnam,cgrid  &
   280)                          ,'h5 ')
   281)             outyear  = iyeara
   282)             outmonth = imontha
   283)             outdate  = idatea
   284)             outhour  = itimea*100
   285) 
   286)          case default
   287)             if (nrec_fast == 1 .or. nrec_fast == 0 ) then
   288) 
   289)                !----- Single file per output. ---------------------------------------------!
   290)                call makefnam(anamel,ffilout,time,iyeara,imontha,idatea,itimea*100,vnam     &
   291)                             ,cgrid,'h5 ')
   292)             else   
   293)                !----- Multiple times per file. --------------------------------------------!
   294)                new_file = .false.
   295)                !----- Determine whether to advance out_time_fast. -------------------------!
   296)                call date_add_to(iyeara,imontha,idatea,itimea*100,time,'s',outyear,outmonth &
   297)                                ,outdate,outhour)
   298)                call date_2_seconds(out_time_fast%year,out_time_fast%month                  &
   299)                                   ,out_time_fast%date,int(out_time_fast%time)              &
   300)                                   ,iyeara,imontha,idatea,itimea*100,dsec)
   301) 
   302)                if (time >= (dsec+dble(outfast)) .or. outmonth /= out_time_fast%month) then
   303)                   out_time_fast%year  = outyear
   304)                   out_time_fast%month = outmonth
   305)                   out_time_fast%date  = outdate
   306)                   out_time_fast%time  = real(outhour)
   307)                   dsec                = time
   308)                   new_file            = .true.
   309)                end if
   310) 
   311)                irec_fast = int(((time-dsec)/dble(frqfast))) + 1
   312)                nrec      = nrec_fast
   313)                irec      = irec_fast
   314) 
   315)                call makefnam(anamel,ffilout,zero,out_time_fast%year,out_time_fast%month    &
   316)                             ,out_time_fast%date,int(out_time_fast%time),vnam,cgrid,'h5 ')
   317)             end if           
   318)          end select
   319)          !---------------------------------------------------------------------------------!
   320) 
   321)          !---------------------------------------------------------------------------------!
   322)          !     Check whether the file exists, and stop the run if the user doesn't want    !
   323)          ! the files to be over-written.                                                   !
   324)          !---------------------------------------------------------------------------------!
   325)          inquire (file=trim(anamel),exist=exans)
   326)          if (exans .and. iclobber == 0) then
   327)             write(unit=*,fmt='(a)'      )  '-----------------------------------------------'
   328)             write(unit=*,fmt='(3(a,1x))') ' - File:',trim(anamel),'already exists.'
   329)             write(unit=*,fmt='(a)'      ) ' - Based on your ICLOBBER, you don''t want'
   330)             write(unit=*,fmt='(a)'      ) '   files to be over-written.  Stopping the run.'
   331)             write(unit=*,fmt='(a)'      ) '-----------------------------------------------'
   332)             call fatal_error('Attempted to over-write a file that already exists.'         &
   333)                             ,'h5_output','h5_output.F90')
   334)          end if
   335)          !---------------------------------------------------------------------------------!
   336) 
   337) 
   338) 
   339)          !---------------------------------------------------------------------------------!
   340)          !     Initialise the HDF5 environment.                                            !
   341)          !---------------------------------------------------------------------------------!
   342)          if (verbose) write (unit=*,fmt='(a)') '    > Opening HDF5 environment...'
   343)          call h5open_f(hdferr)
   344)          if (hdferr /= 0) then
   345)             write(unit=*,fmt='(a,1x,i8)') ' - HDF5 Open error #:',hdferr
   346)             call fatal_error('Could not initialize the hdf environment'                    &
   347)                             ,'h5_output','h5_output.F90')
   348)          end if
   349)          !---------------------------------------------------------------------------------!
   350) 
   351) 
   352) 
   353)          !---------------------------------------------------------------------------------!
   354)          !     Either open or create the output file.                                      !
   355)          !---------------------------------------------------------------------------------!
   356)          if (new_file) then
   357)             if (verbose) write (unit=*,fmt='(a)') '    > Creating new file...'
   358)             
   359)             call h5fcreate_f(trim(anamel)//char(0), H5F_ACC_TRUNC_F, file_id, hdferr)
   360)             if (hdferr /= 0) then
   361)                write(unit=*,fmt='(a)'      ) '--------------------------------------------'
   362)                write(unit=*,fmt='(a)'      ) ' Could not create the HDF5 file.'
   363)                write(unit=*,fmt='(a,1x,a)' ) ' - File   : ',trim(anamel)
   364)                write(unit=*,fmt='(a,1x,i8)') ' - file_id: ',file_id
   365)                write(unit=*,fmt='(a,1x,i8)') ' - hdferr : ',hdferr
   366)                write(unit=*,fmt='(a)'      ) '--------------------------------------------'
   367)                call fatal_error('Failed creating the HDF file','h5_output','h5_output.F90')
   368)             end if
   369)          else
   370)             if (verbose) write (unit=*,fmt='(a)') '    > Opening new file...'
   371)             call h5fopen_f(trim(anamel)//char(0), H5F_ACC_RDWR_F, file_id, hdferr)
   372)             if (hdferr /= 0) then
   373)                write(unit=*,fmt='(a)'      ) '--------------------------------------------'
   374)                write(unit=*,fmt='(a)'      ) ' Could not open the HDF5 file.'
   375)                write(unit=*,fmt='(a,1x,a)' ) ' - File   : ',trim(anamel)
   376)                write(unit=*,fmt='(a,1x,i8)') ' - file_id: ',file_id
   377)                write(unit=*,fmt='(a,1x,i8)') ' - hdferr : ',hdferr
   378)                write(unit=*,fmt='(a)'      ) '--------------------------------------------'
   379)                call fatal_error('Failed opening the HDF file','h5_output','h5_output.F90')
   380)             end if
   381)          end if
   382)          !---------------------------------------------------------------------------------!
   383) 
   384) 
   385) 
   386)          !---------------------------------------------------------------------------------!
   387)          !   Create HDF5 datasets and then put them in the file; we must loop over all     !
   388)          ! variables.                                                                      !
   389)          !---------------------------------------------------------------------------------!
   390)          if (verbose) then
   391)             write (unit=*,fmt='(a,1x,a,1x,a)') '    > Loop over ',num_var(ngr)             &
   392)                                               ,'variables...'
   393)          end if
   394)          varloop: do nv = 1,num_var(ngr)
   395)             !----- Check whether the variable goes to the output. -------------------------!
   396)             if ((vtype == 'INST' .and. vt_info(nv,ngr)%ianal == 1) .or.                    &
   397)                 (vtype == 'OPTI' .and. vt_info(nv,ngr)%iopti == 1) .or.                    &
   398)                 (vtype == 'LITE' .and. vt_info(nv,ngr)%ilite == 1) .or.                    &
   399)                 (vtype == 'DAIL' .and. vt_info(nv,ngr)%idail == 1) .or.                    &
   400)                 (vtype == 'MONT' .and. vt_info(nv,ngr)%imont == 1) .or.                    &
   401)                 (vtype == 'DCYC' .and. vt_info(nv,ngr)%idcyc == 1) .or.                    &
   402)                 (vtype == 'YEAR' .and. vt_info(nv,ngr)%iyear == 1) .or.                    &
   403)                 (vtype == 'HIST' .and. vt_info(nv,ngr)%ihist == 1) .or.                    &
   404)                 (vtype == 'CONT' .and. vt_info(nv,ngr)%ihist == 1) ) then
   405) 
   406)                !----- Variable name. ------------------------------------------------------!
   407)                varn= vt_info(nv,ngr)%name
   408)                !---------------------------------------------------------------------------!
   409) 
   410)                !----- Initialize global dimensions of the hyperslab. ----------------------!
   411)                call geth5dims(vt_info(nv,ngr)%idim_type,0,0,vt_info(nv,ngr)%var_len_global &
   412)                              ,dsetrank,varn,nrec,irec)
   413)                !---------------------------------------------------------------------------!
   414) 
   415)                if (verbose) then
   416)                   write (unit=*,fmt='(a,1x,a,5(a,1x,i12))')                                & 
   417)                       '      # Variable:',trim(varn),'. Type:',vt_info(nv,ngr)%idim_type   &
   418)                      ,'. Size:',vt_info(nv,ngr)%var_len_global,'. Rank:',dsetrank          &
   419)                      ,'. Nrec:',nrec,'. Irec:',irec
   420)                end if
   421) 
   422) 
   423)                !---------------------------------------------------------------------------!
   424)                !     Create the data set.                                                  !
   425)                !---------------------------------------------------------------------------!
   426)                if (verbose) write (unit=*,fmt='(a)') '      # Creating data set...'
   427)                call h5screate_simple_f(dsetrank, globdims, filespace, hdferr)
   428)                if (hdferr /= 0 .or. globdims(1) < 1 ) then
   429)                   write (unit=*,fmt='(a,1x,a)')  ' VTYPE:    ',trim(vtype)
   430)                   write (unit=*,fmt='(a,1x,a)')  ' VAR NAME: ',trim(varn)
   431)                   write (unit=*,fmt='(a,1x,i8)') ' IDIM_TYPE:',vt_info(nv,ngr)%idim_type
   432)                   write (unit=*,fmt='(a,1x,i8)') ' VLEN_GLOB:'                             &
   433)                                                             ,vt_info(nv,ngr)%var_len_global
   434)                   write (unit=*,fmt=*)           ' DSETRANK: ',dsetrank
   435)                   write (unit=*,fmt=*)           ' GLOBDIMS: ',globdims
   436)                   call fatal_error('Could not create the first filespace'                  &
   437)                                   ,'h5_output','h5_output.f90')
   438)                end if
   439)                !---------------------------------------------------------------------------!
   440) 
   441) 
   442)                !---------------------------------------------------------------------------!
   443)                !      Determine whether the dataset exists.                                !
   444)                !---------------------------------------------------------------------------!
   445)                if (verbose) write (unit=*,fmt='(a)') '      # Opening file...'
   446)                call h5eset_auto_f(0,hdferr)
   447)                call h5dopen_f(file_id,varn,dset_id,hdferr)
   448) 
   449)                if (hdferr < 0) then
   450)                   if (verbose) write (unit=*,fmt='(a)') '      # Eset auto (HDFERR < 0)...'
   451)                   call h5eset_auto_f(1,hdferr)
   452) 
   453)                   select case (vt_info(nv,ngr)%dtype)
   454)                   case ('R','r') !----- Real variable. ------------------------------------!
   455)                      call h5dcreate_f(file_id,varn,H5T_NATIVE_REAL, filespace,dset_id      &
   456)                                      ,hdferr)
   457) 
   458)                   case ('D','d') !----- Double precision variable. ------------------------!
   459)                      call h5dcreate_f(file_id,varn,H5T_NATIVE_DOUBLE, filespace,dset_id    &
   460)                                      ,hdferr)
   461) 
   462)                   case ('I','i') !----- Integer variable. ---------------------------------!
   463)                      call h5dcreate_f(file_id,varn,H5T_NATIVE_INTEGER, filespace,dset_id   &
   464)                                      ,hdferr)
   465) 
   466)                   case ('C','c') !----- Character variable. -------------------------------!
   467)                      call h5dcreate_f(file_id,varn,H5T_NATIVE_CHARACTER, filespace,dset_id &
   468)                                      ,hdferr)
   469) 
   470)                   case default
   471)                      write(unit=*,fmt='(a)')      '---------------------------------------'
   472)                      write(unit=*,fmt='(a,1x,a)') ' Variable: ',trim(varn)
   473)                      write(unit=*,fmt='(a,1x,a)') ' Type:     ',trim(vt_info(nv,ngr)%dtype)
   474)                      write(unit=*,fmt='(a)')      '---------------------------------------'
   475)                      call fatal_error('Attempted to write an undefined datatype'           &
   476)                                      ,'h5_output','h5_output.F90')
   477) 
   478)                   end select
   479) 
   480) 
   481) 
   482)                   !------------------------------------------------------------------------!
   483)                   !      Attached metadata if the user wants it.                           !
   484)                   !------------------------------------------------------------------------!
   485)                   if (attach_metadata == 1) then
   486)                      if (verbose) write (unit=*,fmt='(a)') '      # Attaching metadata...'
   487)                      arank       = 1
   488)                      adims       = 3_8
   489)                      attrlen     = 64_8
   490)                      metadata(1) = trim('Long Name: '//trim(vt_info(nv,ngr)%lname))
   491)                      metadata(2) = trim('Units: '//trim(vt_info(nv,ngr)%units))
   492)                      metadata(3) = trim('Dimensions: '//trim(vt_info(nv,ngr)%dimlab))
   493) 
   494)                      call h5screate_simple_f(arank,adims,aspace_id,hdferr)
   495)                      if (hdferr /= 0) then
   496)                         call fatal_error('Error calling h5screate_simple_f'                &
   497)                                         ,'h5_output','h5_output.F90')
   498)                      end if
   499) 
   500)                      call h5tcopy_f(H5T_NATIVE_CHARACTER, atype_id,hdferr)
   501)                      if (hdferr /= 0) then
   502)                         call fatal_error('Error calling h5tcopy_f'                         &
   503)                                         ,'h5_output','h5_output.F90')
   504)                      end if
   505) 
   506)                      call h5tset_size_f(atype_id,attrlen,hdferr)
   507)                      if (hdferr /= 0) then
   508)                         call fatal_error('Error calling h5tset_size_f'                     &
   509)                                         ,'h5_output','h5_output.F90')
   510)                      end if
   511) 
   512)                      call h5acreate_f(dset_id,'Metadata',atype_id,aspace_id,attr_id,hdferr)
   513)                      if (hdferr /= 0) then
   514)                         call fatal_error('Error calling h5acreate_f'                       &
   515)                                         ,'h5_output','h5_output.F90')
   516)                      end if
   517) 
   518)                      call h5awrite_f(attr_id,atype_id,metadata,adims,hdferr)
   519)                      if (hdferr /= 0) then
   520)                         call fatal_error('Error calling h5awrite_f'                        &
   521)                              ,'h5_output','h5_output.F90')
   522)                      end if
   523) 
   524)                      call h5aclose_f(attr_id,hdferr)
   525)                      call h5sclose_f(aspace_id,hdferr)
   526)                      call h5tclose_f(atype_id,hdferr)
   527) 
   528)                   elseif (verbose) then
   529)                      write (unit=*,fmt='(a)') '      # Skipping metadata...'
   530)                   end if
   531) 
   532)                   if (hdferr /= 0) then
   533)                      write (unit=*,fmt=*) 'File name:           ',trim(anamel)
   534)                      write (unit=*,fmt=*) 'Variable name:       ',trim(varn)
   535)                      write (unit=*,fmt=*) 'File ID:             ',file_id
   536)                      write (unit=*,fmt=*) 'Dataset ID:          ',dset_id
   537)                      write (unit=*,fmt=*) 'Dataset rank:        ',dsetrank
   538)                      write (unit=*,fmt=*) 'Global dimension:    ',globdims
   539)                      write (unit=*,fmt=*) 'Chunk size           ',chnkdims
   540)                      write (unit=*,fmt=*) 'Chunk offset         ',chnkoffs
   541)                      write (unit=*,fmt=*) 'Vars written so far: ',nv-1
   542)                      call fatal_error('Could not create the dataset'                       &
   543)                                      ,'h5_output','h5_output.F90')
   544)                   end if
   545)                else
   546)                   if (verbose) write (unit=*,fmt='(a)') '      # Eset auto (HDFERR >=0)...'
   547)                   call h5eset_auto_f(1,hdferr)
   548)                end if
   549) 
   550)                if (verbose) write (unit=*,fmt='(a)') '      # Closing filespace...'
   551)                call h5sclose_f(filespace,hdferr)
   552)                if (hdferr /= 0) then
   553)                   call fatal_error('Could not close the first filespace'                   &
   554)                                   ,'h5_output','h5_output.F90')
   555)                end if
   556)                !---------------------------------------------------------------------------!
   557) 
   558)                !---------------------------------------------------------------------------!
   559)                !      Loop over all the pointers.                                          !
   560)                !---------------------------------------------------------------------------!
   561)                if (verbose) then
   562)                   write (unit=*,fmt='(a,1x,i12,1x,a)')                                     &
   563)                                '      # Looping over ',vt_info(nv,ngr)%nptrs,'pointers ...'
   564)                end if
   565)                pointerloop: do iptr = 1,vt_info(nv,ngr)%nptrs
   566)                   if (verbose) then
   567)                      write (unit=*,fmt='(a,1x,i12,1x,a)') '        ~ Pointer ',iptr,'...'
   568)                   end if
   569) 
   570) 
   571)                   !------------------------------------------------------------------------!
   572)                   !      Set the size of the chunk and it's offset in the global dataset.  !
   573)                   !------------------------------------------------------------------------!
   574)                   if (vt_info(nv,ngr)%vt_vector(iptr)%varlen > 0 ) then
   575)                      if (verbose) then
   576)                         write (unit=*,fmt='(a,1x,i12,1x,a)')                               &
   577)                           '          + Length',vt_info(nv,ngr)%vt_vector(iptr)%varlen,'...'
   578)                      end if
   579)                      !---------------------------------------------------------------------!
   580)                      !     Evaluate the variable output type.  Resolve the dimensioning    !
   581)                      ! and the meta-data tags accordingly.  See ed_state_vars.f90 for a    !
   582)                      ! description of the various datatype.                                !
   583)                      !---------------------------------------------------------------------!
   584)                      !----- Initialize hyperslab indices. ---------------------------------!
   585)                      call geth5dims(vt_info(nv,ngr)%idim_type                              &
   586)                                    ,vt_info(nv,ngr)%vt_vector(iptr)%varlen                 &
   587)                                    ,vt_info(nv,ngr)%vt_vector(iptr)%globid                 &
   588)                                    ,vt_info(nv,ngr)%var_len_global,dsetrank,varn,nrec,irec)
   589) 
   590) 
   591) 
   592)                      if (verbose) then
   593)                         write (unit=*,fmt='(5(a,1x,i12))')                                 &
   594)                             '          + Type:',vt_info(nv,ngr)%idim_type                  &
   595)                            ,'. Size:',vt_info(nv,ngr)%var_len_global,'. Rank:',dsetrank    &
   596)                            ,'. Nrec:',nrec,'. Irec:',irec
   597)                      end if
   598) 
   599)                      !----- Create the data space for the dataset. ------------------------!
   600)                      if (verbose) then
   601)                         write (unit=*,fmt='(a)') '          + Create data space...'
   602)                      end if
   603) 
   604)                      call h5screate_simple_f(dsetrank, chnkdims, memspace, hdferr)
   605)                      if (hdferr /= 0) then
   606)                         write (unit=*,fmt=*) 'Chunk dimension:  ',chnkdims
   607)                         write (unit=*,fmt=*) 'Chunk offset:     ',chnkoffs
   608)                         write (unit=*,fmt=*) 'Global dimension: ',globdims
   609)                         write (unit=*,fmt=*) 'Dataset rank:     ',dsetrank
   610)                         call fatal_error('Could not create the hyperslabs memspace'        &
   611)                                         ,'h5_output','h5_output.F90')
   612)                      end if
   613) 
   614)                      !----- Get the hyperslab in the file. --------------------------------!
   615)                      if (verbose) then
   616)                         write (unit=*,fmt='(a)') '          + Get hyperslab in the file...'
   617)                      end if
   618)                      call h5dget_space_f(dset_id,filespace,hdferr)
   619)                      if (hdferr /= 0) then
   620)                         call fatal_error('Could not get the hyperslabs filespace'          &
   621)                                         ,'h5_output','h5_output.F90')
   622)                      end if
   623) 
   624)                      if (verbose) then
   625)                         write (unit=*,fmt='(a)') '          + Select hyperslab...'
   626)                      end if
   627)                      call h5sselect_hyperslab_f(filespace,H5S_SELECT_SET_F,chnkoffs,cnt    &
   628)                                                ,hdferr,stride,chnkdims)
   629)                      if (hdferr /= 0) then
   630)                         call fatal_error('Could not assign the hyperslabs filespace'       &
   631)                                         ,'h5_output','h5_output.F90')
   632)                      end if
   633) 
   634) 
   635)                      !---------------------------------------------------------------------!
   636)                      !     Choose the right pointer when writing the variable.             !
   637)                      !---------------------------------------------------------------------!
   638)                      if (verbose) then
   639)                         write (unit=*,fmt='(a,1x,a,a)')                                    &
   640)                                                 '          + Write variable of type '      &
   641)                                                ,vt_info(nv,ngr)%dtype,'...'
   642)                      end if
   643)                      select case (vt_info(nv,ngr)%dtype)
   644)                      case ('R') !----- Real variable (vector). ----------------------------!
   645)                         call h5dwrite_f(dset_id,H5T_NATIVE_REAL                            &
   646)                                        ,vt_info(nv,ngr)%vt_vector(iptr)%var_rp             &
   647)                                        ,globdims,hdferr,file_space_id=filespace            &
   648)                                        ,mem_space_id=memspace)
   649) 
   650)                      case ('D') !----- Double precision variable (vector). ----------------!
   651)                         call h5dwrite_f(dset_id,H5T_NATIVE_DOUBLE                          &
   652)                                        ,vt_info(nv,ngr)%vt_vector(iptr)%var_dp             &
   653)                                        ,globdims,hdferr,file_space_id=filespace            &
   654)                                        ,mem_space_id=memspace)
   655) 
   656)                      case ('I') !----- Integer variable (vector). -------------------------!
   657)                         call h5dwrite_f(dset_id,H5T_NATIVE_INTEGER                         &
   658)                                        ,vt_info(nv,ngr)%vt_vector(iptr)%var_ip             &
   659)                                        ,globdims,hdferr,file_space_id=filespace            &
   660)                                        ,mem_space_id=memspace)
   661) 
   662)                      case ('C') !----- Character variable (vector). -----------------------!
   663)                         call h5dwrite_f(dset_id,H5T_NATIVE_CHARACTER                       &
   664)                                        ,vt_info(nv,ngr)%vt_vector(iptr)%var_cp,globdims    &
   665)                                        ,hdferr,file_space_id=filespace                     &
   666)                                        ,mem_space_id = memspace)
   667) 
   668)                      case ('r') !----- Real variable (scalar). ----------------------------!
   669)                         call h5dwrite_f(dset_id,H5T_NATIVE_REAL                            &
   670)                                        ,vt_info(nv,ngr)%vt_vector(iptr)%sca_rp             &
   671)                                        ,globdims,hdferr,file_space_id=filespace            &
   672)                                        ,mem_space_id=memspace)
   673) 
   674)                      case ('d') !----- Double precision variable (scalar). ----------------!
   675)                         call h5dwrite_f(dset_id,H5T_NATIVE_DOUBLE                          &
   676)                                        ,vt_info(nv,ngr)%vt_vector(iptr)%sca_dp             &
   677)                                        ,globdims,hdferr,file_space_id=filespace            &
   678)                                        ,mem_space_id=memspace)
   679) 
   680)                      case ('i') !----- Integer variable (scalar). -------------------------!
   681)                         call h5dwrite_f(dset_id,H5T_NATIVE_INTEGER                         &
   682)                                        ,vt_info(nv,ngr)%vt_vector(iptr)%sca_ip             &
   683)                                        ,globdims,hdferr,file_space_id=filespace            &
   684)                                        ,mem_space_id=memspace)
   685) 
   686)                      case ('c') !----- Character variable (scalar). -----------------------!
   687)                         call h5dwrite_f(dset_id,H5T_NATIVE_CHARACTER                       &
   688)                                        ,vt_info(nv,ngr)%vt_vector(iptr)%sca_cp             &
   689)                                        ,globdims,hdferr,file_space_id=filespace            &
   690)                                        ,mem_space_id=memspace)
   691) 
   692)                      end select
   693)                      if (hdferr /= 0) then
   694)                         write (unit=*,fmt=*) 'Variable name:    ',varn
   695)                         write (unit=*,fmt=*) 'Global dimension: ',globdims
   696)                         write (unit=*,fmt=*) 'Chunk dimension:  ',chnkdims
   697)                         write (unit=*,fmt=*) 'Chunk offset:     ',chnkoffs
   698)                         write (unit=*,fmt=*) 'Count:            ',cnt
   699)                         write (unit=*,fmt=*) 'Stride:           ',stride
   700)                         call fatal_error('Could not write hyperslab into the dataset'      &
   701)                                         ,'h5_output','h5_output.F90')
   702)                      end if
   703) 
   704)                      if (verbose) then
   705)                         write (unit=*,fmt='(a)') '          + Close hyperslab filespace...'
   706)                      end if
   707)                      call h5sclose_f(filespace,hdferr)
   708)                      if (hdferr /= 0) then
   709)                         call fatal_error('Could not close the hyperslabs filespace'        &
   710)                                         ,'h5_output','h5_output.F90')
   711)                      end if
   712) 
   713)                      if (verbose) then
   714)                         write (unit=*,fmt='(a)') '          + Close hyperslab memspace...'
   715)                      end if
   716)                      call h5sclose_f(memspace,hdferr)
   717)                      if (hdferr /= 0) then
   718)                         call fatal_error('Could not close the hyperslabs memspace'         &
   719)                                         ,'h5_output','h5_output.F90')
   720)                      end if
   721)                   elseif (verbose) then
   722)                      write (unit=*,fmt='(a)') '          + Length 0, skipping...'
   723)                   end if
   724)                end do pointerloop
   725)                !---------------------------------------------------------------------------!
   726) 
   727)                if (verbose) write (unit=*,fmt='(a)') '      #  Close dataset...'
   728)                call h5dclose_f(dset_id,hdferr)
   729)                if (hdferr /= 0) then
   730)                   call fatal_error('Could not close dataset','h5_output','h5_output.F90')
   731)                end if
   732)             elseif (verbose) then
   733)                write (unit=*,fmt='(a,1x,a)')                                               &
   734)                       '      # Skipping variable:',trim(varn)                              &
   735)                              ,', as it doesn''t belong to this file...'
   736)             end if
   737)          end do varloop
   738) 
   739)          !---------------------------------------------------------------------------------!
   740) 
   741)          if (verbose) write (unit=*,fmt='(a)') '    > Closing file...'
   742)          call h5fclose_f(file_id,hdferr)
   743)          if (hdferr /= 0) then
   744)             call fatal_error('Could not close the file','h5_output','h5_output.F90')
   745)          end if
   746)         
   747)          if (verbose) write (unit=*,fmt='(a)') '    > Closing HDF5 environment...'
   748)          call h5close_f(hdferr)
   749)          if (hdferr /= 0) then
   750)             call fatal_error('Could not close the hdf environment'                         &
   751)                             ,'h5_output','h5_output.F90')
   752)          end if
   753)          if (verbose) write (unit=*,fmt='(a)') '    > Success!'
   754) 
   755)          new_file = .false.
   756)       elseif (verbose) then
   757)          write (unit=*,fmt='(a)') '    > No polygons in this node... '
   758)       end if
   759)       !------------------------------------------------------------------------------------!
   760) 
   761) 
   762)       !----- Send message for the next node that it is time for it to write the output. ---!
   763) #if defined(RAMS_MPI)
   764)       if (mynum < nnodetot) then
   765)          call MPI_Send(new_file,1,MPI_LOGICAL,sendnum,3510+ngr,MPI_COMM_WORLD,ierr)
   766)       end if
   767) #endif
   768)    end do gridloop
   769) 
   770)    select case (trim(vtype))
   771)    case ('DAIL')
   772)       subaname = '  Daily average analysis HDF write    '
   773) 
   774)    case ('MONT')
   775)       subaname = '  Monthly average analysis HDF write   '
   776) 
   777)    case ('DCYC')
   778)       subaname = '  Mean diurnal cycle analysis HDF write   '
   779) 
   780)    case ('YEAR')
   781)       subaname = '  Annual average analysis HDF write   '
   782) 
   783)    case ('OPTI')
   784)       subaname = '  Opt update   '
   785) 
   786)    case ('HIST','CONT')
   787)       !------------------------------------------------------------------------------------!
   788)       !      The following is helpful to those running very massive simulations.  File     !
   789)       ! write times may be significantly long in these cases, and with automation of       !
   790)       ! simulation queuing, the last thing you want to it try restarting partially written !
   791)       ! files.  The generation of the CPM file ensures that its sibling HDF5 file was      !
   792)       ! completely written.  For simplicity, this will remain commented out by default.    !
   793)       !------------------------------------------------------------------------------------!
   794)       if (nnodetot /= 1 .and. mynum == nnodetot) then
   795)          !----- Write a dummy file that signals we are done. ------------------------------!
   796)          call makefnam(anamel,sfilout,time,iyeara,imontha,idatea,itimea*100,vnam,'g00'     &
   797)                       ,'cmp')
   798)          open (unit=79,file=trim(anamel),form='formatted',status='replace')
   799)          write(unit=79,fmt='(a)') 'history write completed'
   800)          write(unit=* ,fmt='(2a)') ' - Completed History Write: ',trim(anamel)
   801)          close(unit=79,status='keep')
   802)       end if
   803) 
   804)       select case(trim(vtype))
   805)       case ('HIST')
   806)          subaname = '  History HDF write   '
   807)       case ('CONT')
   808)          subaname = '  Resume file HDF write   '
   809)       end select
   810)    case default
   811)       subaname = '  Analysis HDF write         '
   812)    end select
   813)      
   814)    if (mynum == nnodetot .and. new_file) then
   815)       write(c0,fmt='(f15.0)') sngl(time)
   816)       write(unit=*,fmt='(a)') ' === '//trim(adjustl(subaname))//' at Sim time '            &
   817)                                      //trim(adjustl(c0))//' ==='
   818)    end if
   819) 
   820)    return
   821) end subroutine h5_output

   832) subroutine geth5dims(idim_type,varlen,globid,var_len_global,dsetrank,varn,nrec,irec)
   833)    
   834)    use grid_coms          , only : nzg            & ! intent(in)
   835)                                  , nzs            ! ! intent(in)
   836)    use ed_max_dims        , only : n_pft          & ! intent(in)
   837)                                  , n_dist_types   & ! intent(in)
   838)                                  , n_dbh          & ! intent(in)
   839)                                  , n_age          & ! intent(in)
   840)                                  , n_mort         & ! intent(in)
   841)                                  , n_radprof      ! ! intent(in)
   842)    use hdf5_coms          , only : chnkdims       & ! intent(in)
   843)                                  , chnkoffs       & ! intent(in)
   844)                                  , cnt            & ! intent(in)
   845)                                  , stride         & ! intent(in)
   846)                                  , globdims       ! ! intent(in)
   847)    use fusion_fission_coms, only : ff_nhgt        ! ! intent(in)
   848)    use ed_misc_coms       , only : ndcycle        ! ! intent(in)
   849) 
   850)    implicit none
   851)    !----- Arguments. ----------------------------------------------------------------------!
   852)    character(len=*), intent(in)  :: varn
   853)    integer         , intent(in)  :: idim_type
   854)    integer         , intent(in)  :: varlen
   855)    integer         , intent(in)  :: globid
   856)    integer         , intent(in)  :: var_len_global
   857)    integer         , intent(out) :: dsetrank
   858)    integer         , intent(in)  :: nrec
   859)    integer         , intent(in)  :: irec
   860)    !---------------------------------------------------------------------------------------!
   861) 
   862) 
   863)    !----- Initialize the size of the memory and file-space dimensioning. ------------------!
   864)    globdims = 0_8
   865)    chnkdims = 0_8
   866)    chnkoffs = 0_8
   867)    cnt      = 0_8
   868)    stride   = 0_8
   869) 
   870)    !---------------------------------------------------------------------------------------!
   871)    !     Determine the array sizes based on idim_type.                                     !
   872)    !---------------------------------------------------------------------------------------!
   873) 
   874)    select case (idim_type) 
   875)    case(90,91,92,96) ! No polygon-site-patch or cohort dimension, or single-dim. vector
   876)       
   877)       dsetrank = 1
   878)       chnkdims(1) = int(varlen,8)
   879)       chnkoffs(1) = int(globid,8)
   880)       globdims(1) = int(var_len_global,8)
   881)       cnt(1)      = 1_8
   882)       stride(1)   = 1_8
   883)       
   884)    case (10,11) ! (npolygons) 
   885)       
   886)       ! Vector type,dimensions set
   887)       
   888)       dsetrank = 1
   889)       chnkdims(1) = int(varlen,8)
   890)       chnkoffs(1) = int(globid,8)
   891)       globdims(1) = int(var_len_global,8)
   892)       cnt(1)      = 1_8
   893)       stride(1)   = 1_8
   894)       
   895)    case (-11) ! (ndcycle,npolygons)  
   896)       
   897)       ! Soil column type
   898)       dsetrank = 2
   899)       globdims(1) = int(ndcycle,8)
   900)       chnkdims(1) = int(ndcycle,8)
   901)       chnkoffs(1) = 0_8
   902)       globdims(2) = int(var_len_global,8)
   903)       chnkdims(2) = int(varlen,8)
   904)       chnkoffs(2) = int(globid,8)
   905)       cnt(1:2)    = 1_8
   906)       stride(1:2) = 1_8
   907)       
   908)    case (12,120) ! (nzg,npolygons)  
   909)       
   910)       ! Soil column type
   911)       dsetrank = 2
   912)       globdims(1) = int(nzg,8)
   913)       chnkdims(1) = int(nzg,8)
   914)       chnkoffs(1) = 0_8
   915)       globdims(2) = int(var_len_global,8)
   916)       chnkdims(2) = int(varlen,8)
   917)       chnkoffs(2) = int(globid,8)
   918)       cnt(1:2)    = 1_8
   919)       stride(1:2) = 1_8
   920) 
   921)    case (-12) ! (nzg,ndcycle,npolygons)
   922)       
   923)       dsetrank = 3
   924)       
   925)       globdims(1) = int(nzg,8)
   926)       chnkdims(1) = int(nzg,8)
   927)       chnkoffs(1) = 0_8
   928)       globdims(2) = int(ndcycle,8)
   929)       chnkdims(2) = int(ndcycle,8)
   930)       chnkoffs(2) = 0_8
   931)       globdims(3) = int(var_len_global,8)
   932)       chnkdims(3) = int(varlen,8)
   933)       chnkoffs(3) = int(globid,8)
   934)       cnt(1:3)      = 1_8
   935)       stride(1:3)   = 1_8
   936)       
   937)    case (14) ! (n_pft,npolygons)  
   938)       
   939)       dsetrank = 2
   940)       globdims(1) = int(n_pft,8)
   941)       chnkdims(1) = int(n_pft,8)
   942)       chnkoffs(1) = 0_8
   943)       globdims(2) = int(var_len_global,8)
   944)       chnkdims(2) = int(varlen,8)
