     6) !      This subroutine initializes a near-bare ground polygon.                             !
     7) !------------------------------------------------------------------------------------------!
     8) subroutine near_bare_ground_init(cgrid)
     9)    use ed_state_vars  , only : edtype            & ! structure
    10)                              , polygontype       & ! structure
    11)                              , sitetype          & ! structure
    12)                              , allocate_sitetype ! ! subroutine
    13)    use ed_misc_coms   , only : ied_init_mode     ! ! intent(in)
    14)    use physiology_coms, only : n_plant_lim       ! ! intent(in)
    15) 
    16)    implicit none
    17) 
    18)    !----- Arguments. ----------------------------------------------------------------------!
    19)    type(edtype)      , target  :: cgrid
    20)    !----- Local variables. ----------------------------------------------------------------!
    21)    type(polygontype) , pointer :: cpoly
    22)    type(sitetype)    , pointer :: csite
    23)    integer                     :: ipy
    24)    integer                     :: isi
    25)    !---------------------------------------------------------------------------------------!
    26) 
    27) 
    28)    !----- Big loop ------------------------------------------------------------------------!
    29)    do ipy=1,cgrid%npolygons
    30)       cpoly => cgrid%polygon(ipy)
    31)       do isi=1,cpoly%nsites
    32)          csite => cpoly%site(isi)
    33)          !----- We start with a single patch per site, always with primary vegetation. ----!
    34)          csite%npatches = 1
    35)          call allocate_sitetype(csite,1)
    36)          csite%dist_type          (1) = 3
    37)          csite%age                (1) = 0.0
    38)          csite%area               (1) = 1.0
    39) 
    40) 
    41)          !---------------------------------------------------------------------------------!
    42)          !     Someone that uses the nitrogen model should check whether this is necessary !
    43)          ! or not.  If nitrogen limitation is off, then we start all carbon and nitrogen   !
    44)          ! pools with zeroes, otherwise we initialise with the former default values.      !
    45)          !---------------------------------------------------------------------------------!
    46)          select case (n_plant_lim)
    47)          case (0)
    48)             csite%fast_soil_C        (1) = 0.0
    49)             csite%slow_soil_C        (1) = 0.0
    50)             csite%structural_soil_C  (1) = 0.0
    51)             csite%structural_soil_L  (1) = 0.0
    52)             csite%mineralized_soil_N (1) = 0.0
    53)             csite%fast_soil_N        (1) = 0.0
    54) 
    55)          case (1)
    56)             csite%fast_soil_C        (1) = 0.2
    57)             csite%slow_soil_C        (1) = 0.01
    58)             csite%structural_soil_C  (1) = 10.0
    59)             csite%structural_soil_L  (1) = csite%structural_soil_C (1)
    60)             csite%mineralized_soil_N (1) = 1.0
    61)             csite%fast_soil_N        (1) = 1.0
    62) 
    63)          end select
    64)          !---------------------------------------------------------------------------------!
    65) 
    66)          csite%sum_dgd            (1) = 0.0
    67)          csite%sum_chd            (1) = 0.0
    68)          csite%plant_ag_biomass   (1) = 0.
    69) 
    70)          !----- We now populate the cohorts with near bare ground condition. --------------!
    71)          select case (ied_init_mode)
    72)          case (-8)
    73)             call init_cohorts_by_layers(csite,cpoly%lsl(isi),1,csite%npatches)
    74)          case (-1,0)
    75)             call init_nbg_cohorts(csite,cpoly%lsl(isi),1,csite%npatches)
    76)          end select
    77) 
    78)          !----- Initialise the patches now that cohorts are there. ------------------------!
    79)          call init_ed_patch_vars(csite,1,csite%npatches,cpoly%lsl(isi))
    80)       end do
    81)       !----- Initialise some site-level variables. ----------------------------------------!
    82)       call init_ed_site_vars(cpoly)
    83)    end do
    84) 
    85)    !----- Last, but not the least, the polygons. ------------------------------------------!
    86)    call init_ed_poly_vars(cgrid)
    87) 
    88)    return
    89) end subroutine near_bare_ground_init

   100) !      This subroutine assigns a near-bare ground (NBG) state for some patches.            !
   101) !------------------------------------------------------------------------------------------!
   102) subroutine init_nbg_cohorts(csite,lsl,ipa_a,ipa_z)
   103)    use ed_state_vars      , only : edtype             & ! structure
   104)                                  , polygontype        & ! structure
   105)                                  , sitetype           & ! structure
   106)                                  , patchtype          & ! structure
   107)                                  , allocate_sitetype  & ! subroutine
   108)                                  , allocate_patchtype ! ! subroutine
   109)    use ed_max_dims        , only : n_pft              ! ! intent(in)
   110)    use ed_misc_coms       , only : ied_init_mode      ! ! intent(in)
   111)    use pft_coms           , only : q                  & ! intent(in)
   112)                                  , qsw                & ! intent(in)
   113)                                  , sla                & ! intent(in)
   114)                                  , hgt_min            & ! intent(in)
   115)                                  , include_pft        & ! intent(in)
   116)                                  , include_pft_ag     & ! intent(in)
   117)                                  , include_pft_fp     & ! intent(in)
   118)                                  , init_density       & ! intent(in)
   119)                                  , agf_bs             ! ! intent(in)
   120)    use consts_coms        , only : t3ple              & ! intent(in)
   121)                                  , pio4               & ! intent(in)
   122)                                  , kgom2_2_tonoha     & ! intent(in)
   123)                                  , tonoha_2_kgom2     ! ! intent(in)
   124)    use allometry          , only : h2dbh              & ! function
   125)                                  , dbh2bd             & ! function
   126)                                  , size2bl            & ! function
   127)                                  , ed_biomass         & ! function
   128)                                  , area_indices       ! ! subroutine
   129)    use fuse_fiss_utils    , only : sort_cohorts    ! ! subroutine
   130) 
   131)    implicit none
   132)    !----- Arguments -----------------------------------------------------------------------!
   133)    type(sitetype)        , target     :: csite             ! Current site
   134)    integer               , intent(in) :: lsl               ! Lowest soil level
   135)    integer               , intent(in) :: ipa_a             ! 1st patch
   136)    integer               , intent(in) :: ipa_z             ! Last patch
   137)    !----- Local variables -----------------------------------------------------------------!
   138)    type(patchtype)       , pointer    :: cpatch            ! Current patch
   139)    integer                            :: ipa               ! Patch number
   140)    integer                            :: ico               ! Cohort counter
   141)    integer                            :: mypfts            ! Number of included PFTs
   142)    integer                            :: ipft              ! PFT counter
   143)    real                               :: salloc            ! balive/bleaf when on allom.
   144)    real                               :: salloci           ! 1./salloc
   145)    !---------------------------------------------------------------------------------------!
   146) 
   147)    !----- Patch loop. ---------------------------------------------------------------------!
   148)    patchloop: do ipa=ipa_a,ipa_z
   149)       cpatch => csite%patch(ipa)
   150) 
   151)       !----- Check which initialisation we are going to use. ------------------------------!
   152)       select case (ied_init_mode)
   153)       case (-1) !------ True bare ground simulation (absolute desert). --------------------!
   154)          mypfts = 0
   155)          return
   156)       case ( 0) !------ Nearly bare ground simulation (start with a few seedlings). -------!
   157) 
   158)          !---------------------------------------------------------------------------------!
   159)          !     Decide how many cohorts to allocate.                                        !
   160)          !---------------------------------------------------------------------------------!
   161)          select case (csite%dist_type(ipa))
   162)          case (1)     !---- Agriculture. --------------------------------------------------!
   163)             mypfts = count(include_pft_ag)
   164)          case (2)     !---- Forest plantation. --------------------------------------------!
   165)             mypfts = count(include_pft_fp)
   166)          case default !---- Secondary or primary forest. ----------------------------------!
   167)             mypfts = count(include_pft)
   168)          end select
   169)          !---------------------------------------------------------------------------------!
   170)       end select
   171) 
   172)       !----- Perform cohort allocation. ---------------------------------------------------!
   173)       call allocate_patchtype(cpatch,mypfts)
   174) 
   175)       !------------------------------------------------------------------------------------!
   176)       !    Here we loop over PFTs rather than assigning the cohorts, so we ensure to only  !
   177)       ! include the PFTs that should be  included (i.e., patches that should not happen in !
   178)       ! agricultural patches will be skipped, and agricultural PFTs will not be included   !
   179)       ! in the forest sites.                                                               !
   180)       !------------------------------------------------------------------------------------!
   181)       ico = 0
   182)       pftloop: do ipft = 1,n_pft
   183)          select case (csite%dist_type(ipa))
   184)          case (1)
   185)             !----- Agriculture, only the ones allowed are included. -----------------------!
   186)             if (include_pft_ag(ipft)) then
   187)                ico = ico + 1
   188)             else
   189)                cycle pftloop
   190)             end if
   191)          case (2)
   192)             !----- Forest plantation, only the ones allowed are included. -----------------!
   193)             if (include_pft_fp(ipft)) then
   194)                ico = ico + 1
   195)             else
   196)                cycle pftloop
   197)             end if
   198)          case default
   199)             !----- Primary or secondary vegetation, only the ones allowed are included. ---!
   200)             if (include_pft(ipft)) then
   201)                ico = ico + 1
   202)             else
   203)                cycle pftloop
   204)             end if
   205)          end select
   206) 
   207)          !----- The PFT is the plant functional type. -------------------------------------!
   208)          cpatch%pft(ico)              = ipft
   209)          !---------------------------------------------------------------------------------!
   210)          !     Define the near-bare ground state using the standard minimum height and     !
   211)          ! minimum plant density.  We assume all NBG PFTs to have leaves fully flushed,    !
   212)          ! but with no storage biomass.  We then compute the other biomass quantities      !
   213)          ! using the standard allometry for this PFT.                                      !
   214)          !---------------------------------------------------------------------------------!
   215)          cpatch%nplant(ico)           = init_density(ipft)
   216)          cpatch%hite(ico)             = hgt_min(ipft)
   217)          cpatch%phenology_status(ico) = 0
   218)          cpatch%dbh(ico)              = h2dbh(cpatch%hite(ico),ipft)
   219)          cpatch%bdead(ico)            = dbh2bd(cpatch%dbh(ico),ipft)
   220)          cpatch%bleaf(ico)            = size2bl(cpatch%dbh(ico),cpatch%hite(ico),ipft)
   221)          cpatch%sla(ico)              = sla(ipft)
   222) 
   223) 
   224)          salloc                       = 1.0 + q(ipft) + qsw(ipft) * cpatch%hite(ico)
   225)          salloci                      = 1. / salloc
   226) 
   227)          cpatch%balive(ico)           = cpatch%bleaf(ico) * salloc
   228)          cpatch%broot(ico)            = q(ipft) * cpatch%balive(ico) * salloci
   229)          cpatch%bsapwooda(ico)        = qsw(ipft) * cpatch%hite(ico) * cpatch%balive(ico)  &
   230)                                       * salloci * agf_bs(ipft)
   231)          cpatch%bsapwoodb(ico)        = qsw(ipft) * cpatch%hite(ico) * cpatch%balive(ico)  &
   232)                                       * salloci * (1.-agf_bs(ipft))
   233)          cpatch%bstorage(ico)         = 0.5 * ( cpatch%bleaf(ico)                          &
   234)                                               + cpatch%broot(ico)                          &
   235)                                               + cpatch%bsapwooda(ico)                      &
   236)                                               + cpatch%bsapwoodb(ico))
   237) 
   238)          !----- Find the initial area indices (LAI, WAI, CAI). ----------------------------!
   239)          call area_indices(cpatch, ico)
   240) 
   241)          !----- Find the above-ground biomass and basal area. -----------------------------!
   242)          cpatch%agb(ico) = ed_biomass(cpatch, ico)
   243)          cpatch%basarea(ico) = pio4 * cpatch%dbh(ico)*cpatch%dbh(ico)
   244) 
   245)          !----- Initialize other cohort-level variables. ----------------------------------!
   246)          call init_ed_cohort_vars(cpatch,ico,lsl)
   247) 
   248)          !----- Update total patch-level above-ground biomass -----------------------------!
   249)          csite%plant_ag_biomass(ipa) = csite%plant_ag_biomass(ipa)                         &
   250)                                      + cpatch%nplant(ico) * cpatch%agb(ico)
   251)       end do pftloop
   252) 
   253)       !------------------------------------------------------------------------------------!
   254)       !     Since initial heights may not be constant, we must sort the cohorts.           !
   255)       !------------------------------------------------------------------------------------!
   256)       call sort_cohorts(cpatch)
   257)    end do patchloop
   258) 
   259)    return
   260) end subroutine init_nbg_cohorts

   271) !      This subroutine assigns a near-bare ground (NBG) state for some patches.            !
   272) !------------------------------------------------------------------------------------------!
   273) subroutine init_cohorts_by_layers(csite,lsl,ipa_a,ipa_z)
   274)    use ed_state_vars      , only : edtype             & ! structure
   275)                                  , polygontype        & ! structure
   276)                                  , sitetype           & ! structure
   277)                                  , patchtype          & ! structure
   278)                                  , allocate_sitetype  & ! subroutine
   279)                                  , allocate_patchtype ! ! subroutine
   280)    use ed_max_dims        , only : n_pft              ! ! intent(in)
   281)    use pft_coms           , only : q                  & ! intent(in)
   282)                                  , qsw                & ! intent(in)
   283)                                  , sla                & ! intent(in)
   284)                                  , include_pft        & ! intent(in)
   285)                                  , include_these_pft  & ! intent(in)
   286)                                  , agf_bs             ! ! intent(in)
   287)    use consts_coms        , only : t3ple              & ! intent(in)
   288)                                  , pio4               & ! intent(in)
   289)                                  , kgom2_2_tonoha     & ! intent(in)
   290)                                  , tonoha_2_kgom2     ! ! intent(in)
   291)    use allometry          , only : h2dbh              & ! function
   292)                                  , dbh2bd             & ! function
   293)                                  , size2bl            & ! function
   294)                                  , ed_biomass         & ! function
   295)                                  , area_indices       ! ! subroutine
   296)    use fuse_fiss_utils    , only : sort_cohorts       ! ! subroutine
   297) 
   298)    implicit none
   299)    !----- Arguments -----------------------------------------------------------------------!
   300)    type(sitetype)        , target     :: csite   ! Current site
   301)    integer               , intent(in) :: lsl     ! Lowest soil level
   302)    integer               , intent(in) :: ipa_a   ! 1st patch to be assigned with NBG state
   303)    integer               , intent(in) :: ipa_z   ! Last patch to be assigned with NBG state
   304)    !----- Local variables -----------------------------------------------------------------!
   305)    type(patchtype)       , pointer    :: cpatch  ! Current patch
   306)    integer                            :: ipa     ! Patch number
   307)    integer                            :: ico     ! Cohort counter
   308)    integer                            :: ipft    ! PFT counter
   309)    real                               :: height  ! Cohort initial height
   310)    real                               :: salloc  ! Factor to find balive, broot, bsapwood
   311)    real                               :: salloci ! 1./salloc
   312)    !----- Local constants. ----------------------------------------------------------------!
   313)    integer               , parameter  :: nlayers = 8   ! # of cohort layers to be included.
   314)    real                  , parameter  :: dheight = 1.5 ! height interval.
   315)    real                  , parameter  :: lai0    = 1.  ! Initial LAI for each layer.
   316)    real                  , parameter  :: h0      = 1.5 ! Height of the lowest cohort.
   317)    !---------------------------------------------------------------------------------------!
   318) 
   319)    !----- Patch loop. ---------------------------------------------------------------------!
   320)    patchloop: do ipa=ipa_a,ipa_z
   321)       cpatch => csite%patch(ipa)
   322) 
   323)       if (count (include_pft) /= 1) then
   324)          call fatal_error('Multi-layer run cannot be run with more than 1 PFT...'          &
   325)                          ,'init_cohorts_by_layers','ed_nbg_init.f90')
   326)       end if
   327) 
   328)       !----- Assigning the PFT. -----------------------------------------------------------!
   329)       ipft = include_these_pft(1)
   330) 
   331)       !----- Perform cohort allocation. ---------------------------------------------------!
   332)       call allocate_patchtype(cpatch,nlayers)
   333) 
   334)       !------------------------------------------------------------------------------------!
   335)       !    Here we loop over layers and assign the new cohort.                             !
   336)       !------------------------------------------------------------------------------------!
   337)       height = h0 - dheight
   338)       layerloop: do ico = 1,nlayers
   339)          height = height + dheight
   340) 
   341)          !----- The PFT is the plant functional type. -------------------------------------!
   342)          cpatch%pft(ico)              = ipft
   343) 
   344)          !---------------------------------------------------------------------------------!
   345)          !     Define the initial state in such a way that the LAI is always the initial   !
   346)          ! LAI.  We then compute the other biomass quantities using the standard allometry !
   347)          ! for this PFT.                                                                   !
   348)          !---------------------------------------------------------------------------------!
   349)          cpatch%hite(ico)             = height
   350)          cpatch%phenology_status(ico) = 0
   351)          cpatch%bstorage(ico)         = 0.0
   352)          cpatch%dbh(ico)              = h2dbh(cpatch%hite(ico),ipft)
   353)          cpatch%bdead(ico)            = dbh2bd(cpatch%dbh(ico),ipft)
   354)          cpatch%bleaf(ico)            = size2bl(cpatch%dbh(ico),cpatch%hite(ico),ipft)
   355)          cpatch%sla(ico)              = sla(ipft)
   356) 
   357) 
   358)          salloc                       = 1.0 + q(ipft) + qsw(ipft) * cpatch%hite(ico)
   359)          salloci                      = 1. / salloc
   360) 
   361)          cpatch%balive(ico)           = cpatch%bleaf(ico) * salloc
   362)          cpatch%broot(ico)            = q(ipft) * cpatch%balive(ico) * salloci
   363)          cpatch%bsapwooda(ico)        = qsw(ipft) * cpatch%hite(ico) * cpatch%balive(ico)  &
   364)                                       * salloci * agf_bs(ipft)
   365)          cpatch%bsapwoodb(ico)        = qsw(ipft) * cpatch%hite(ico) * cpatch%balive(ico)  &
   366)                                       * salloci * (1.-agf_bs(ipft))
   367) 
   368)          !----- NPlant is defined such that the cohort LAI is equal to LAI0
   369)          cpatch%nplant(ico)           = lai0 / (cpatch%bleaf(ico) * cpatch%sla(ico))
   370) 
   371)          !----- Find the initial area indices (LAI, WAI, CAI). ----------------------------!
   372)          call area_indices(cpatch, ico)
   373) 
   374)          !----- Find the above-ground biomass and basal area. -----------------------------!
   375)          cpatch%agb(ico) = ed_biomass(cpatch, ico)
   376)          cpatch%basarea(ico) = pio4 * cpatch%dbh(ico) * cpatch%dbh(ico)
   377) 
   378)          !----- Initialize other cohort-level variables. ----------------------------------!
   379)          call init_ed_cohort_vars(cpatch,ico,lsl)
   380) 
   381)          !----- Update total patch-level above-ground biomass -----------------------------!
   382)          csite%plant_ag_biomass(ipa) = csite%plant_ag_biomass(ipa)                         &
   383)                                      + cpatch%nplant(ico) * cpatch%agb(ico)
   384)       end do layerloop
   385) 
   386)       !------------------------------------------------------------------------------------!
   387)       !     Since initial heights may not be constant, we must sort the cohorts.           !
   388)       !------------------------------------------------------------------------------------!
   389)       call sort_cohorts(cpatch)
   390)    end do patchloop
   391) 
   392)    return
   393) end subroutine init_cohorts_by_layers

   404) !      This subroutine initializes a near-bare ground 'big-leaf' ed run.                   !
   405) !------------------------------------------------------------------------------------------!
   406) subroutine near_bare_ground_big_leaf_init(cgrid)
   407)    use ed_state_vars      , only : edtype             & ! structure
   408)                                  , polygontype        & ! structure
   409)                                  , sitetype           & ! structure
   410)                                  , patchtype          & ! structure
   411)                                  , allocate_sitetype  & ! subroutine
   412)                                  , allocate_patchtype ! ! subroutine
   413)    use ed_misc_coms       , only : ied_init_mode      ! ! intent(in)
   414)    use physiology_coms    , only : n_plant_lim        ! ! intent(in)
   415)    use pft_coms           , only : q                  & ! intent(in)
   416)                                  , qsw                & ! intent(in)
   417)                                  , sla                & ! intent(in)
   418)                                  , agf_bs             & ! intent(in)
   419)                                  , dbh_bigleaf        & ! intent(in)
   420)                                  , hgt_max            & ! intent(in)
   421)                                  , include_pft        & ! intent(in)
   422)                                  , include_these_pft  & ! intent(in)
   423)                                  , init_density       ! ! intent(in)
   424)    use consts_coms        , only : t3ple              & ! intent(in)
   425)                                  , pio4               & ! intent(in)
   426)                                  , kgom2_2_tonoha     & ! intent(in)
   427)                                  , tonoha_2_kgom2     ! ! intent(in)
   428)    use allometry          , only : dbh2bd             & ! function
   429)                                  , size2bl            & ! function
   430)                                  , ed_biomass         & ! function
   431)                                  , area_indices       ! ! subroutine
   432)    implicit none
   433) 
   434)    !----- Arguments. ----------------------------------------------------------------------!
   435)    type(edtype)      , target  :: cgrid
   436)    !----- Local variables. ----------------------------------------------------------------!
   437)    type(polygontype) , pointer :: cpoly
   438)    type(sitetype)    , pointer :: csite
   439)    type(patchtype)   , pointer :: cpatch            ! Current patch
   440)    integer                     :: ipy               ! Patch counter
   441)    integer                     :: ico               ! Cohort counter
   442)    integer                     :: isi               ! Site counter
   443)    integer                     :: ipa               ! Patch counter
   444)    integer                     :: mypfts            ! Number of included PFTs
   445)    integer                     :: ipft              ! PFT counter
   446)    real                        :: salloc            ! balive/bleaf when on allom.
   447)    real                        :: salloci           ! 1./salloc
   448)    !---------------------------------------------------------------------------------------!
   449) 
   450) 
   451)    !----- Big loop ------------------------------------------------------------------------!
   452)    polyloop: do ipy=1,cgrid%npolygons
   453)       cpoly => cgrid%polygon(ipy)
   454)       siteloop: do isi=1,cpoly%nsites
   455)          csite => cpoly%site(isi)
   456) 
   457)          !-- Figure out how many patches (1 patch per pft), always primary vegetation. ----!
   458)          select case (ied_init_mode)
   459)          case (-1) !------ True bare ground simulation (absolute desert). -----------------!
   460)             mypfts = 1
   461)          case ( 0) !------ Nearly bare ground simulation (start with a few seedlings). ----!
   462)             mypfts = count(include_pft)
   463)          end select
   464)          csite%npatches = mypfts
   465) 
   466)          call allocate_sitetype(csite,mypfts)
   467) 
   468)          !----- Patch loop  ---------------------------------------------------------------!
   469)          patchloop: do ipa=1, csite%npatches
   470)             cpatch => csite%patch(ipa)
   471)             ipft = include_these_pft(ipa)
   472) 
   473)             csite%dist_type          (ipa) = 3
   474)             csite%age                (ipa) = 0.0
   475)             csite%area               (ipa) = 1.0 / mypfts
   476) 
   477)             select case (n_plant_lim)
   478)             case (0)
   479)                csite%fast_soil_C        (ipa) = 0.0
   480)                csite%slow_soil_C        (ipa) = 0.0
   481)                csite%structural_soil_C  (ipa) = 0.0
   482)                csite%structural_soil_L  (ipa) = 0.0
   483)                csite%mineralized_soil_N (ipa) = 0.0
   484)                csite%fast_soil_N        (ipa) = 0.0
   485) 
   486)             case (1)
   487)                csite%fast_soil_C        (ipa) = 0.2
   488)                csite%slow_soil_C        (ipa) = 0.01
   489)                csite%structural_soil_C  (ipa) = 10.0
   490)                csite%structural_soil_L  (ipa) = csite%structural_soil_C (1)
