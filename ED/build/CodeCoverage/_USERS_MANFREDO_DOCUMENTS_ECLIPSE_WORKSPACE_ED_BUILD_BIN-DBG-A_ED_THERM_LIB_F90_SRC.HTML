    16)    ! This function calculates the total heat capacity in (J/K/m2) of the cohort            !
    17)    ! leaf biomass.  This function is primarily used to calculate leaf temperatures         !
    18)    ! based on leaf energy.                                                                 !
    19)    !                                                                                       !
    20)    ! Inputs:                                                                               !
    21)    ! + BLEAF        - the leaf biomass of the cohort in kgC/plant.  NOTE: it will be       !
    22)    !                  converted to kg/m2, by the product of leaf_carbon in units kg/plant  !
    23)    !                  and number of plants per square meter, and the conversion of carbon  !
    24)    !                  to total biomass.  The right hand side of the main equation accounts !
    25)    !                  for the mass of insterstitial water and its ability to hold energy.  !
    26)    ! + BDEAD        - the structural wood biomass of the cohort in kgC/plant.              !
    27)    ! + BSAPWOODA    - the above ground sapwood biomass of the cohort, in kgC/plant.        !
    28)    ! + NPLANTS      - the number of plants per m2.                                         !
    29)    ! + PFT          - the plant functional type of the current cohort, which may serve     !
    30)    !                  for defining different parameterizations of specific heat capacity   !
    31)    !                                                                                       !
    32)    ! Ouputs:                                                                               !
    33)    ! + LEAF_HCAP    - the leaf heat capacity, in J/m2/K.                                   !
    34)    ! + WOOD_HCAP    - the wood heat capacity, in J/m2/K.                                   !
    35)    !                                                                                       !
    36)    ! These methods follow the ways of Gu et al. 2007, with the only difference that for    !
    37)    ! non-green biomass we dropped the temperature dependence and assumed T=T3ple, just to  !
    38)    ! make it simpler.  See the module pft_coms.f90 for a description of the parameters,    !
    39)    ! and see ed_params.f90 for the setting of these parameters.                            !
    40)    !                                                                                       !
    41)    ! Reference:                                                                            !
    42)    !                                                                                       !
    43)    ! Gu, L., T. Meyers, S. G. Pallardy, 2007: Influences of biomass heat and biochemical   !
    44)    !      energy storages on the land surface fluxes and radiative temperature.            !
    45)    !      J. Geophys. Res., v. 112, doi: 10.1029/2006JD007425.                             !
    46)    !---------------------------------------------------------------------------------------!
    47)    subroutine calc_veg_hcap(bleaf,bdead,bsapwooda,nplant,pft,leaf_hcap,wood_hcap)
    48)       use consts_coms          , only : cliq                ! ! intent(in)
    49)       use pft_coms             , only : c_grn_leaf_dry      & ! intent(in)
    50)                                       , wat_dry_ratio_grn   & ! intent(in)
    51)                                       , c_ngrn_biom_dry     & ! intent(in)
    52)                                       , wat_dry_ratio_ngrn  & ! intent(in)
    53)                                       , delta_c             & ! intent(in)
    54)                                       , agf_bs              & ! intent(in)
    55)                                       , C2B                 & ! intent(in)
    56)                                       , brf_wd              ! ! intent(in)
    57) 
    58)       use rk4_coms             , only : ibranch_thermo      ! ! intent(in)
    59)       implicit none
    60)       !----- Arguments --------------------------------------------------------------------!
    61)       real    , intent(in)    :: bleaf         ! Biomass of leaves              [kgC/plant]
    62)       real    , intent(in)    :: bdead         ! Biomass of structural wood     [kgC/plant]
    63)       real    , intent(in)    :: bsapwooda     ! Biomass of above ground sapwood[kgC/plant]
    64)       real    , intent(in)    :: nplant        ! Number of plants               [ plant/m2]
    65)       integer , intent(in)    :: pft           ! Plant functional type          [     ----]
    66)       real    , intent(out)   :: leaf_hcap     ! Leaf heat capacity             [   J/m2/K]
    67)       real    , intent(out)   :: wood_hcap     ! Wood heat capacity             [   J/m2/K]
    68)       !----- Local variables --------------------------------------------------------------!
    69)       real                    :: bwood         ! Wood biomass                   [kgC/plant]
    70)       real                    :: spheat_leaf   ! Leaf specific heat             [   J/kg/K]
    71)       real                    :: spheat_wood   ! Wood specific heat             [   J/kg/K]
    72)       !------------------------------------------------------------------------------------!
    73) 
    74)       !------------------------------------------------------------------------------------!
    75)       !    Here we decide whether we compute the branch heat capacity or not.              !
    76)       !------------------------------------------------------------------------------------!
    77)       select case (ibranch_thermo)
    78)       case (0)
    79)          !----- Skip it, the user doesn't want to solve for branches. ---------------------!
    80)          spheat_wood = 0.
    81)          bwood       = 0.
    82)       case default
    83)          !----- Find branch/twig specific heat and biomass. -------------------------------!
    84)          spheat_wood = (c_ngrn_biom_dry(pft) + wat_dry_ratio_ngrn(pft) * cliq)             &
    85)                      / (1. + wat_dry_ratio_ngrn(pft)) + delta_c(pft)
    86)          bwood       = brf_wd(pft) * (bsapwooda + bdead*agf_bs(pft))
    87)       end select
    88) 
    89)       !----- Find the leaf specific heat. -------------------------------------------------!
    90)       spheat_leaf = (c_grn_leaf_dry(pft) + wat_dry_ratio_grn(pft) * cliq)                  &
    91)                   / (1. + wat_dry_ratio_grn(pft))
    92) 
    93)       !------------------------------------------------------------------------------------!
    94)       !     The heat capacity is specific heat times the plant density times the leaf/wood !
    95)       ! biomass.                                                                           !
    96)       !------------------------------------------------------------------------------------!
    97)       leaf_hcap = nplant * C2B * bleaf * spheat_leaf * (1. + wat_dry_ratio_grn (pft))
    98)       wood_hcap = nplant * C2B * bwood * spheat_wood * (1. + wat_dry_ratio_ngrn(pft))
    99)       !------------------------------------------------------------------------------------!
   100) 
   101)       return
   102)    end subroutine calc_veg_hcap

   113)    !    This subroutine updates the vegetation energy when the plant heat capacity has     !
   114)    ! changed. This routine should be used only when leaf or structural biomass has         !
   115)    ! changed, it should never be used in fast time steps.                                  !
   116)    !                                                                                       !
   117)    !     We look at leaf and wood separately, but the idea is the same.  When heat         !
   118)    ! capacity is zero (i.e., no leaves or not solving branchwood thermodynamics), we       !
   119)    ! cannot find the temperature using uextcm2tl because it is a singularity.  Notice that !
   120)    ! this is different than skipping when cohorts are not resolvable...  If the cohort is  !
   121)    ! not resolvable but still has some heat capacity, we should update internal energy     !
   122)    ! using the traditional method, and NEVER force the heat capacity to be zero, otherwise !
   123)    ! we violate the fact that heat capacity is a linear function of mass and this will     !
   124)    ! cause problems during the fusion/splitting process.                                   !
   125)    !                                                                                       !
   126)    !    The "cweh" mean "consistent water&energy&hcap" assumption                          !
   127)    !---------------------------------------------------------------------------------------!
   128)    subroutine update_veg_energy_cweh(csite,ipa,ico,old_leaf_hcap,old_wood_hcap)
   129)       use ed_state_vars, only : sitetype   & ! structure
   130)                               , patchtype  ! ! structure
   131)       use therm_lib    , only : uextcm2tl  & ! subroutine
   132)                               , cmtl2uext  ! ! function
   133)       implicit none
   134)       !----- Arguments --------------------------------------------------------------------!
   135)       type(sitetype) , target     :: csite
   136)       integer        , intent(in) :: ipa
   137)       integer        , intent(in) :: ico
   138)       real           , intent(in) :: old_leaf_hcap
   139)       real           , intent(in) :: old_wood_hcap
   140)       !----- Local variables --------------------------------------------------------------!
   141)       type(patchtype), pointer    :: cpatch
   142)       real                        :: new_temp
   143)       real                        :: new_fliq
   144)       integer                     :: kclosest
   145)       integer                     :: k
   146)       !------------------------------------------------------------------------------------!
   147) 
   148)       cpatch => csite%patch(ipa)
   149) 
   150) 
   151)       !------------------------------------------------------------------------------------!
   152)       !     Leaves.  Check whether heat capacity is zero or not.                           !
   153)       !------------------------------------------------------------------------------------!
   154)       if (cpatch%leaf_hcap(ico) == 0. ) then
   155)          cpatch%leaf_energy(ico) = 0.
   156)          cpatch%leaf_water(ico)  = 0.
   157)          cpatch%leaf_fliq(ico)   = 0.
   158)          if (cpatch%hite(ico) > csite%total_sfcw_depth(ipa)) then
   159)             !----- Plant is exposed, set temperature to the canopy temperature. -----------!
   160)             cpatch%leaf_temp(ico) = csite%can_temp(ipa)
   161)          else
   162)             !----- Find the snow layer that is the closest to where the leaves would be. --!
   163)             do k = csite%nlev_sfcwater(ipa), 1, -1
   164)                if (sum(csite%sfcwater_depth(1:k,ipa)) >= cpatch%hite(ico)) then
   165)                   kclosest = k
   166)                end if
   167)             end do
   168)             cpatch%leaf_temp(ico) = csite%sfcwater_tempk(kclosest,ipa)
   169)          end if
   170)          !---------------------------------------------------------------------------------!
   171) 
   172)       else
   173)          !---------------------------------------------------------------------------------!
   174)          !     Heat capacity is not zero.  Since we track leaf temperature and liquid      !
   175)          ! fraction of water held by leaves, we can recalculate the internal energy by     !
   176)          ! just switching the old heat capacity by the new one.                            !
   177)          !---------------------------------------------------------------------------------!
   178)          cpatch%leaf_energy(ico) = cmtl2uext(cpatch%leaf_hcap(ico),cpatch%leaf_water(ico)  &
   179)                                             ,cpatch%leaf_temp(ico),cpatch%leaf_fliq(ico) )
   180)          !---------------------------------------------------------------------------------!
   181) 
   182) 
   183) 
   184)          !----- This is a sanity check, it can be removed if it doesn't crash. ------------!
   185)          call uextcm2tl(cpatch%leaf_energy(ico),cpatch%leaf_water(ico)                     &
   186)                        ,cpatch%leaf_hcap(ico),new_temp,new_fliq)
   187)          !---------------------------------------------------------------------------------!
   188) 
   189) 
   190) 
   191)          !---------------------------------------------------------------------------------!
   192)          !     In case the temperature is different, give the user the bad news...         !
   193)          !---------------------------------------------------------------------------------!
   194)          if (abs(new_temp - cpatch%leaf_temp(ico)) > 0.1) then
   195)             write(unit=*,fmt='(a)') '-----------------------------------------------------'
   196)             write(unit=*,fmt='(a)') ' LEAF ENERGY CONSERVATION FAILED!:'
   197)             write(unit=*,fmt='(a)') '-----------------------------------------------------'
   198)             write(unit=*,fmt='(a,1x,es12.5)') ' Old temperature:  ',cpatch%leaf_temp(ico)
   199)             write(unit=*,fmt='(a,1x,es12.5)') ' New temperature:  ',new_temp
   200)             write(unit=*,fmt='(a,1x,es12.5)') ' Old heat capacity:',old_leaf_hcap
   201)             write(unit=*,fmt='(a,1x,es12.5)') ' New heat capacity:',cpatch%leaf_hcap(ico)
   202)             write(unit=*,fmt='(a,1x,es12.5)') ' Leaf energy:      ',cpatch%leaf_energy(ico)
   203)             write(unit=*,fmt='(a,1x,es12.5)') ' Leaf water:       ',cpatch%leaf_water(ico)
   204)             write(unit=*,fmt='(a)') '-----------------------------------------------------'
   205)             call fatal_error('Leaf energy is leaking!!!','update_veg_energy_cweh'          &
   206)                             &,'ed_therm_lib.f90')
   207)          end if
   208)          !---------------------------------------------------------------------------------!
   209)       end if
   210)       !------------------------------------------------------------------------------------!
   211) 
   212) 
   213) 
   214) 
   215) 
   216)       !------------------------------------------------------------------------------------!
   217)       !     Wood.  Check whether heat capacity is zero or not.                             !
   218)       !------------------------------------------------------------------------------------!
   219)       if (cpatch%wood_hcap(ico) == 0. ) then
   220)          cpatch%wood_energy(ico) = 0.
   221)          cpatch%wood_water(ico)  = 0.
   222)          cpatch%wood_fliq(ico)   = 0.
   223)          if (cpatch%hite(ico) > csite%total_sfcw_depth(ipa)) then
   224)             !----- Plant is exposed, set temperature to the canopy temperature. -----------!
   225)             cpatch%wood_temp(ico) = csite%can_temp(ipa)
   226)          else
   227)             !----- Find the snow layer that is the closest to where the leaves would be. --!
   228)             do k = csite%nlev_sfcwater(ipa), 1, -1
   229)                if (sum(csite%sfcwater_depth(1:k,ipa)) >= cpatch%hite(ico)) then
   230)                   kclosest = k
   231)                end if
   232)             end do
   233)             cpatch%wood_temp(ico) = csite%sfcwater_tempk(kclosest,ipa)
   234)          end if
   235)          !---------------------------------------------------------------------------------!
   236) 
   237)       else
   238)          !---------------------------------------------------------------------------------!
   239)          !     Heat capacity is not zero.  Since we track leaf temperature and liquid      !
   240)          ! fraction of water held by leaves, we can recalculate the internal energy by     !
   241)          ! just switching the old heat capacity by the new one.                            !
   242)          !---------------------------------------------------------------------------------!
   243)          cpatch%wood_energy(ico) = cmtl2uext(cpatch%wood_hcap(ico),cpatch%wood_water(ico)  &
   244)                                             ,cpatch%wood_temp(ico),cpatch%wood_fliq (ico) )
   245)          !---------------------------------------------------------------------------------!
   246) 
   247) 
   248) 
   249)          !----- This is a sanity check, it can be removed if it doesn't crash. ------------!
   250)          call uextcm2tl(cpatch%wood_energy(ico),cpatch%wood_water(ico)                     &
   251)                        ,cpatch%wood_hcap(ico),new_temp,new_fliq)
   252)          !---------------------------------------------------------------------------------!
   253) 
   254) 
   255) 
   256) 
   257)          !---------------------------------------------------------------------------------!
   258)          !     In case the temperature is different, give the user the bad news...         !
   259)          !---------------------------------------------------------------------------------!
   260)          if (abs(new_temp - cpatch%wood_temp(ico)) > 0.1) then
   261)             write(unit=*,fmt='(a)') '-----------------------------------------------------'
   262)             write(unit=*,fmt='(a)') ' WOOD ENERGY CONSERVATION FAILED!:'
   263)             write(unit=*,fmt='(a)') '-----------------------------------------------------'
   264)             write(unit=*,fmt='(a,1x,es12.5)') ' Old temperature:  ',cpatch%wood_temp(ico)
   265)             write(unit=*,fmt='(a,1x,es12.5)') ' New temperature:  ',new_temp
   266)             write(unit=*,fmt='(a,1x,es12.5)') ' Old heat capacity:',old_wood_hcap
   267)             write(unit=*,fmt='(a,1x,es12.5)') ' New heat capacity:',cpatch%wood_hcap(ico)
   268)             write(unit=*,fmt='(a,1x,es12.5)') ' Wood energy:      ',cpatch%wood_energy(ico)
   269)             write(unit=*,fmt='(a,1x,es12.5)') ' Wood water:       ',cpatch%wood_water(ico)
   270)             write(unit=*,fmt='(a)') '-----------------------------------------------------'
   271)             call fatal_error('Wood energy is leaking!!!','update_veg_energy_cweh'          &
   272)                             &,'ed_therm_lib.f90')
   273)          end if
   274)          !---------------------------------------------------------------------------------!
   275)       end if
   276)       !------------------------------------------------------------------------------------!
   277) 
   278) 
   279)       return
   280)    end subroutine update_veg_energy_cweh

   291)    !      This routine computes ground_shv, which is is the saturation mixing ratio of     !
   292)    ! the top soil or snow surface and is used for dew formation and snow evaporation.      !
   293)    ! References:                                                                           !
   294)    !                                                                                       !
   295)    ! P86  - Passerat de Silans, A., 1986: Transferts de masse et de chaleur dans un sol    !
   296)    !        stratifi� soumis � une excitation amtosph�rique naturelle. Comparaison:        !
   297)    !        Mod�les-exp�rience. Thesis, Institut National Polytechnique de Grenoble.       !
   298)    !                                                                                       !
   299)    ! NP89 - Noilhan, J., S. Planton, 1989: A simple parameterization of land surface       !
   300)    !        processes for meteorological models. Mon. Wea. Rev., 117, 536-549.             !
   301)    !                                                                                       !
   302)    ! MN91 - Mahfouf, J. F., J. Noilhan, 1991: Comparative study of various formulations of !
   303)    !        evaporation from bare soil using in situ data. J. Appl. Meteorol., 30,         !
   304)    !        1354-1365.                                                                     !
   305)    !                                                                                       !
   306)    ! LP92 - Lee, T. J., R. A. Pielke, 1992: Estimating the soil surface specific humidity  !
   307)    !        J. Appl. Meteorol., 31, 480-484.                                               !
   308)    !                                                                                       !
   309)    ! LP93 - Lee, T. J., R. A. Pielke, 1993: CORRIGENDUM, J. Appl. Meteorol., 32, 580.      !
   310)    !                                                                                       !
   311)    !---------------------------------------------------------------------------------------!
   312)    subroutine ed_grndvap(ksn,nsoil,topsoil_water,topsoil_temp,topsoil_fliq,sfcwater_temp   &
   313)                         ,sfcwater_frac,can_prss,can_shv,ground_shv                         &
   314)                         ,ground_ssh,ground_temp,ground_fliq,ggsoil)
   315) 
   316)       use canopy_air_coms, only : ied_grndvap       & ! intent(in)
   317)                                 , ggsoil0           & ! intent(in)
   318)                                 , kksoil            ! ! intent(in)
   319)       use soil_coms      , only : soil              & ! intent(in)
   320)                                 , matric_potential  ! ! function
   321)       use consts_coms    , only : pi1               & ! intent(in)
   322)                                 , wdns              & ! intent(in)
   323)                                 , gorh2o            & ! intent(in)
   324)                                 , lnexp_min         & ! intent(in)
   325)                                 , tiny_num          & ! intent(in)
   326)                                 , huge_num          ! ! intent(in)
   327)       use therm_lib      , only : qslif             ! ! function
   328)       use ed_max_dims    , only : n_pft             ! ! intent(in)
   329)       implicit none
   330)       !----- Arguments --------------------------------------------------------------------!
   331)       integer     , intent(in)  :: ksn           ! # of surface water layers    [     ----]
   332)       integer     , intent(in)  :: nsoil         ! Soil type                    [     ----]
   333)       real(kind=4), intent(in)  :: topsoil_water ! Top soil water               [m�_h2o/m�]
   334)       real(kind=4), intent(in)  :: topsoil_temp  ! Top soil temperature         [        K]
   335)       real(kind=4), intent(in)  :: topsoil_fliq  ! Top soil liquid water frac.  [       --]
   336)       real(kind=4), intent(in)  :: sfcwater_temp ! Snow/water temperature       [        K]
   337)       real(kind=4), intent(in)  :: sfcwater_frac ! Snow/water liq. water frac.  [       --]
   338)       real(kind=4), intent(in)  :: can_prss      ! canopy pressure              [       Pa]
   339)       real(kind=4), intent(in)  :: can_shv       ! canopy vapour spec humidity  [kg_vap/kg]
   340)       real(kind=4), intent(out) :: ground_shv    ! ground equilibrium spec hum  [kg_vap/kg]
   341)       real(kind=4), intent(out) :: ground_ssh    ! sfc. saturation spec. hum.   [kg_vap/kg]
   342)       real(kind=4), intent(out) :: ground_temp   ! Surface temperature          [        K]
   343)       real(kind=4), intent(out) :: ground_fliq   ! Surface liquid water frac.   [       --]
   344)       real(kind=4), intent(out) :: ggsoil        ! Soil conductance for evap.   [      m/s]
   345)       !----- Local variables --------------------------------------------------------------!
   346)       real(kind=4)              :: slpotvn       ! soil water potential         [        m]
   347)       real(kind=4)              :: alpha         ! alpha term (Lee-Pielke,1992) [     ----]
   348)       real(kind=4)              :: beta          ! beta term  (Lee-Pielke,1992) [     ----]
   349)       real(kind=4)              :: lnalpha       ! ln(alpha)                    [     ----]
   350)       real(kind=4)              :: smterm        ! soil moisture term           [     ----]
   351)       real(kind=4)              :: topsoil_shv   ! ground equilibrium spec hum  [kg_vap/kg]
   352)       real(kind=4)              :: topsoil_ssh   ! sfc. saturation spec. hum.   [kg_vap/kg]
   353)       real(kind=4)              :: sfcwater_shv  ! ground equilibrium spec hum  [kg_vap/kg]
   354)       real(kind=4)              :: sfcwater_ssh  ! sfc. saturation spec. hum.   [kg_vap/kg]
   355)       !------------------------------------------------------------------------------------!
   356) 
   357) 
   358) 
   359)       !------------------------------------------------------------------------------------!
   360)       !      Topsoil_shv is the effective specific humidity of soil.  This value is a      !
   361)       ! combination of the canopy air specific humidity, the saturation specific humidity  !
   362)       ! at the soil temperature.  When the soil tends to dry air soil moisture,            !
   363)       ! topsoil_shv tends to the canopy air space specific humidity, whereas topsoil_shv   !
   364)       ! tends to the saturation value when the soil moisture is near or above field        !
   365)       ! capacity.  These tendencies are determined by the alpha and beta parameters.       !
   366)       !------------------------------------------------------------------------------------!
   367)       !----- Compute the saturation specific humidity at top soil temperature. ------------!
   368)       topsoil_ssh  = qslif(can_prss,topsoil_temp)
   369)       !----- Determine alpha. -------------------------------------------------------------!
   370)       slpotvn      = matric_potential(nsoil,topsoil_water)
   371)       lnalpha      = gorh2o * slpotvn / topsoil_temp
   372)       if (lnalpha > lnexp_min) then
   373)          alpha   = exp(lnalpha)
   374)       else
   375)          alpha   = 0.0
   376)       end if
   377)       !------------------------------------------------------------------------------------!
   378) 
   379) 
   380) 
   381) 
   382) 
   383)       !------------------------------------------------------------------------------------!
   384)       !     Determine Beta, following NP89.  However, because we want evaporation to be    !
   385)       ! shut down when the soil approaches the dry air soil moisture, we offset both       !
   386)       ! the soil moisture and field capacity to the soil moisture above dry air soil.      !
   387)       ! This is necessary to avoid evaporation to be large just slightly above the dry air !
   388)       ! soil, which would otherwise happen, especially for those soil types rich in clay.  !
   389)       !------------------------------------------------------------------------------------!
   390)       smterm = min(1.0, max(0.0, (topsoil_water       - soil(nsoil)%soilcp)                &
   391)                                / (soil(nsoil)%sfldcap - soil(nsoil)%soilcp) ))
   392)       beta   = 0.5 * (1.0 - cos (smterm * pi1))
   393)       !------------------------------------------------------------------------------------!
   394) 
   395) 
   396) 
   397) 
   398) 
   399)       !------------------------------------------------------------------------------------!
   400)       !     Decide which method to use to find the ground water vapour mixing ratio.       !
   401)       !------------------------------------------------------------------------------------!
   402)       select case (ied_grndvap)
   403)       case (0)
   404)          !----- LP92 method. --------------------------------------------------------------!
   405)          topsoil_shv = max(can_shv, topsoil_ssh * alpha * beta + (1.0 - beta) * can_shv)
   406)          ggsoil      = huge_num
   407)          !---------------------------------------------------------------------------------!
   408) 
   409)       case (1)
   410)          !----- MH91, test 1. -------------------------------------------------------------!
   411)          topsoil_shv = max(can_shv, topsoil_ssh * beta)
   412)          ggsoil      = huge_num
   413)          !---------------------------------------------------------------------------------!
   414) 
   415)       case (2)
   416)          !----- MH91, test 2. -------------------------------------------------------------!
   417)          topsoil_shv = topsoil_ssh
   418)          ggsoil      = ggsoil0 * exp(kksoil * smterm)
   419)          !---------------------------------------------------------------------------------!
   420) 
   421)       case (3)
   422)          !----- MH91, test 3. -------------------------------------------------------------!
   423)          topsoil_shv = max(can_shv, topsoil_ssh * beta + (1.0 - beta) * can_shv)
   424)          ggsoil      = huge_num
   425)          !---------------------------------------------------------------------------------!
   426) 
   427)       case (4)
   428)          !---------------------------------------------------------------------------------!
   429)          !     MH91, test 4.                                                               !
   430)          !---------------------------------------------------------------------------------!
   431)          topsoil_shv = max(can_shv, topsoil_ssh * alpha)
   432)          ggsoil      = ggsoil0 * exp(kksoil * smterm)
   433)          !---------------------------------------------------------------------------------!
   434) 
   435)       case (5)
   436)          !---------------------------------------------------------------------------------!
   437)          !     Combination of NP89 and P86.                                                !
   438)          !---------------------------------------------------------------------------------!
   439)          topsoil_shv = max(can_shv, topsoil_ssh * beta)
   440)          ggsoil      = ggsoil0 * exp(kksoil * smterm)
   441)          !---------------------------------------------------------------------------------!
   442)       end select
   443)       !------------------------------------------------------------------------------------!
   444) 
   445) 
   446)       !----- Compute the saturation specific humidity at ground temperature. --------------!
   447)       sfcwater_ssh = qslif(can_prss,sfcwater_temp)
   448)       !----- The ground specific humidity in this case is just the saturation value. ------!
   449)       sfcwater_shv = sfcwater_ssh
   450)       !------------------------------------------------------------------------------------!
   451) 
   452) 
   453)       !------------------------------------------------------------------------------------!
   454)       !      The properties are weighted averages with area being the weight.              !
   455)       !------------------------------------------------------------------------------------!
   456)       ground_temp = topsoil_temp
   457)       ground_fliq = topsoil_fliq
   458)       ground_ssh  = topsoil_ssh
   459)       ground_shv  = topsoil_shv
   460)       !------------------------------------------------------------------------------------!
   461) 
   462) 
   463) 
   464)       !------------------------------------------------------------------------------------!
   465)       !       Ggsoil is corrected as the weighted averages of the resistances (assumed to  !
   466)       ! be zero for temporary surface water).                                              !
   467)       !------------------------------------------------------------------------------------!
   468)       if (ggsoil /= huge_num .and. ksn > 0) then
   469)          ggsoil      = min(huge_num, ggsoil / max(tiny_num,(1.0 - sfcwater_frac)))
   470)       end if
   471)       !------------------------------------------------------------------------------------!
   472) 
   473) 
   474)       return
   475)    end subroutine ed_grndvap

   498)    !        evaporation from bare soil using in situ data. J. Appl. Meteorol., 30,         !
   499)    !        1354-1365.                                                                     !

   503)    !                                                                                       !

   505)    !                                                                                       !

   507)    subroutine ed_grndvap8(ksn,topsoil_water,topsoil_temp,topsoil_fliq,sfcwater_temp        &
   508)                          ,sfcwater_frac,can_prss,can_shv,ground_shv                        &
   509)                          ,ground_ssh,ground_temp,ground_fliq,ggsoil)
   510)       use canopy_air_coms, only : ied_grndvap       & ! intent(in)
   511)                                 , ggsoil08          & ! intent(in)
   512)                                 , kksoil8           ! ! intent(in)
   513)       use soil_coms      , only : soil8             & ! intent(in)
   514)                                 , matric_potential8 ! ! function
   515)       use consts_coms    , only : pi18              & ! intent(in)
   516)                                 , wdns8             & ! intent(in)
   517)                                 , gorh2o8           & ! intent(in)
   518)                                 , lnexp_min8        & ! intent(in)
   519)                                 , tiny_num8         & ! intent(in)
   520)                                 , huge_num8         ! ! intent(in)
   521)       use therm_lib8     , only : qslif8            ! ! function
   522)       use rk4_coms       , only : rk4site           ! ! intent(in)
   523)       use grid_coms      , only : nzg               ! ! intent(in)
   524)       use ed_max_dims    , only : n_pft             ! ! intent(in)
   525)       implicit none
   526)       !----- Arguments --------------------------------------------------------------------!
   527)       integer     , intent(in)  :: ksn           ! # of surface water layers    [     ----]
   528)       real(kind=8), intent(in)  :: topsoil_water ! Top soil water               [m�_h2o/m�]
   529)       real(kind=8), intent(in)  :: topsoil_temp  ! Top soil temperature         [        K]
   530)       real(kind=8), intent(in)  :: topsoil_fliq  ! Top soil liquid water frac.  [       --]
   531)       real(kind=8), intent(in)  :: sfcwater_temp ! Snow/water temperature       [        K]
   532)       real(kind=8), intent(in)  :: sfcwater_frac ! Snow/water liq. water frac.  [       --]
   533)       real(kind=8), intent(in)  :: can_prss      ! canopy pressure              [       Pa]
   534)       real(kind=8), intent(in)  :: can_shv       ! canopy vapour spec humidity  [kg_vap/kg]
   535)       real(kind=8), intent(out) :: ground_shv    ! ground equilibrium spec hum  [kg_vap/kg]
   536)       real(kind=8), intent(out) :: ground_ssh    ! sfc. saturation spec. hum.   [kg_vap/kg]
   537)       real(kind=8), intent(out) :: ground_temp   ! Surface temperature          [        K]
   538)       real(kind=8), intent(out) :: ground_fliq   ! Surface liquid water frac.   [       --]
   539)       real(kind=8), intent(out) :: ggsoil        ! Soil conductance for evap.   [      m/s]
   540)       !----- Local variables --------------------------------------------------------------!
   541)       integer                   :: nsoil         ! Soil type                    [     ----]
   542)       real(kind=8)              :: slpotvn       ! soil water potential         [        m]
   543)       real(kind=8)              :: alpha         ! alpha term (Lee-Pielke,1992) [     ----]
   544)       real(kind=8)              :: beta          ! beta term  (Lee-Pielke,1992) [     ----]
   545)       real(kind=8)              :: lnalpha       ! ln(alpha)                    [     ----]
   546)       real(kind=8)              :: smterm        ! soil moisture term           [     ----]
   547)       real(kind=8)              :: topsoil_shv   ! ground equilibrium spec hum  [kg_vap/kg]
   548)       real(kind=8)              :: topsoil_ssh   ! sfc. saturation spec. hum.   [kg_vap/kg]
   549)       real(kind=8)              :: sfcwater_shv  ! ground equilibrium spec hum  [kg_vap/kg]
   550)       real(kind=8)              :: sfcwater_ssh  ! sfc. saturation spec. hum.   [kg_vap/kg]
   551)       !------------------------------------------------------------------------------------!
   552) 
   553) 
   554)       !------ Soil type at the top layer. -------------------------------------------------!
   555)       nsoil = rk4site%ntext_soil(nzg)
   556)       !------------------------------------------------------------------------------------!
   557) 
   558) 
   559)       !------------------------------------------------------------------------------------!
   560)       !      Topsoil_shv is the effective specific humidity of soil.  This value is a      !
   561)       ! combination of the canopy air specific humidity, the saturation specific humidity  !
   562)       ! at the soil temperature.  When the soil tends to dry air soil moisture,            !
   563)       ! topsoil_shv tends to the canopy air space specific humidity, whereas topsoil_shv   !
   564)       ! tends to the saturation value when the soil moisture is near or above field        !
   565)       ! capacity.  These tendencies are determined by the alpha and beta parameters.       !
   566)       !------------------------------------------------------------------------------------!
   567)       !----- Compute the saturation specific humidity at top soil temperature. ------------!
   568)       topsoil_ssh  = qslif8(can_prss,topsoil_temp)
   569)       !----- Determine alpha. -------------------------------------------------------------!
   570)       slpotvn      = matric_potential8(nsoil,topsoil_water)
   571)       lnalpha      = gorh2o8 * slpotvn / topsoil_temp
   572)       if (lnalpha > lnexp_min8) then
   573)          alpha   = exp(lnalpha)
   574)       else
   575)          alpha   = 0.d0
   576)       end if
   577)       !------------------------------------------------------------------------------------!
   578) 
   579) 
   580) 
   581) 
   582) 
   583)       !------------------------------------------------------------------------------------!
   584)       !     Determine Beta, following NP89.  However, because we want evaporation to be    !
   585)       ! shut down when the soil approaches the dry air soil moisture, we offset both       !
   586)       ! the soil moisture and field capacity to the soil moisture above dry air soil.      !
   587)       ! This is necessary to avoid evaporation to be large just slightly above the dry air !
   588)       ! soil, which would otherwise happen, especially for those soil types rich in clay.  !
   589)       !------------------------------------------------------------------------------------!
   590)       smterm = min(1.d0, max(0.d0, (topsoil_water        - soil8(nsoil)%soilcp)            &
   591)                                  / (soil8(nsoil)%sfldcap - soil8(nsoil)%soilcp) ))
   592)       beta   = 5.d-1 * (1.d0 - cos (smterm * pi18))
   593)       !------------------------------------------------------------------------------------!
   594) 
   595) 
   596) 
   597) 
   598) 
   599)       !------------------------------------------------------------------------------------!
   600)       !     Decide which method to use to find the ground water vapour mixing ratio.       !
   601)       !------------------------------------------------------------------------------------!
   602)       select case (ied_grndvap)
   603)       case (0)
   604)          !----- LP92 method. --------------------------------------------------------------!
   605)          topsoil_shv = max(can_shv, topsoil_ssh * alpha * beta + (1.d0 - beta) * can_shv)
   606)          ggsoil      = huge_num8
   607)          !---------------------------------------------------------------------------------!
   608) 
   609)       case (1)
   610)          !----- MH91, test 1. -------------------------------------------------------------!
   611)          topsoil_shv = max(can_shv, topsoil_ssh * beta)
   612)          ggsoil      = huge_num8
   613)          !---------------------------------------------------------------------------------!
   614) 
   615)       case (2)
   616)          !----- MH91, test 2. -------------------------------------------------------------!
   617)          topsoil_shv = topsoil_ssh
   618)          ggsoil      = ggsoil08 * exp(kksoil8 * smterm)
   619)          !---------------------------------------------------------------------------------!
   620) 
   621)       case (3)
   622)          !----- MH91, test 3. -------------------------------------------------------------!
   623)          topsoil_shv = max(can_shv, topsoil_ssh * beta + (1.d0 - beta) * can_shv)
   624)          ggsoil      = huge_num8
   625)          !---------------------------------------------------------------------------------!
   626) 
   627)       case (4)
   628)          !---------------------------------------------------------------------------------!
   629)          !     MH91, test 4.                                                               !
   630)          !---------------------------------------------------------------------------------!
   631)          topsoil_shv = max(can_shv, topsoil_ssh * alpha)
   632)          ggsoil      = ggsoil08 * exp(kksoil8 * smterm)
   633)          !---------------------------------------------------------------------------------!
   634) 
   635)       case (5)
   636)          !---------------------------------------------------------------------------------!
   637)          !     Combination of NP89 and P86.                                                !
   638)          !---------------------------------------------------------------------------------!
   639)          topsoil_shv = max(can_shv, topsoil_ssh * beta)
   640)          ggsoil      = ggsoil08 * exp(kksoil8 * smterm)
   641)          !---------------------------------------------------------------------------------!
   642)       end select
   643)       !------------------------------------------------------------------------------------!
   644) 
   645) 
   646)       !----- Compute the saturation specific humidity at ground temperature. --------------!
   647)       sfcwater_ssh = qslif8(can_prss,sfcwater_temp)
   648)       !----- The ground specific humidity in this case is just the saturation value. ------!
   649)       sfcwater_shv = sfcwater_ssh
   650)       !------------------------------------------------------------------------------------!
   651) 
   652) 
   653)       !------------------------------------------------------------------------------------!
   654)       !      The properties are weighted averages with area being the weight.              !
   655)       !------------------------------------------------------------------------------------!
   656)       ground_temp = topsoil_temp
   657)       ground_fliq = topsoil_fliq
   658)       ground_ssh  = topsoil_ssh
   659)       ground_shv  = topsoil_shv
   660)       !------------------------------------------------------------------------------------!
   661) 
   662) 
   663) 
   664)       !------------------------------------------------------------------------------------!
   665)       !       Ggsoil is corrected as the weighted averages of the resistances (assumed to  !
   666)       ! be zero for temporary surface water).                                              !
   667)       !------------------------------------------------------------------------------------!
   668)       if (ggsoil /= huge_num8 .and. ksn > 0) then
   669)          ggsoil      = min(huge_num8, ggsoil / max(tiny_num8,(1.d0 - sfcwater_frac)))
   670)       end if

   672)

   674)       return
