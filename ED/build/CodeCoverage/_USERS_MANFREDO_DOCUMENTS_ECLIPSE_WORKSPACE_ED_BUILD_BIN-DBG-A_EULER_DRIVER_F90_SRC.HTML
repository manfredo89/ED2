     1) module euler_driver
     2)   contains
     3) 
     4) !==========================================================================================!
     5) !==========================================================================================!
     6) !     This subroutine is the main driver for the Euler integration scheme.                 !
     7) !------------------------------------------------------------------------------------------!
     8) subroutine euler_timestep(cgrid)
     9)    use rk4_integ_utils
    10)    use soil_respiration_module
    11)    use photosyn_driv
    12)    use rk4_misc
    13)    use update_derived_props_module
    14)    use rk4_coms              , only : integration_vars   & ! structure
    15)                                     , rk4patchtype       & ! structure
    16)                                     , zero_rk4_patch     & ! subroutine
    17)                                     , zero_rk4_cohort    & ! subroutine
    18)                                     , integration_buff   & ! intent(out)
    19)                                     , rk4site            ! ! intent(out)
    20)    use ed_state_vars         , only : edtype             & ! structure
    21)                                     , polygontype        & ! structure
    22)                                     , sitetype           & ! structure
    23)                                     , patchtype          ! ! structure
    24)    use met_driver_coms       , only : met_driv_state     ! ! structure
    25)    use grid_coms             , only : nzg                & ! intent(in)
    26)                                     , nzs                ! ! intent(in)
    27)    use ed_misc_coms          , only : current_time       & ! intent(in)
    28)                                     , dtlsm              ! ! intent(in)
    29)    use ed_max_dims           , only : n_dbh              ! ! intent(in)
    30)    use soil_coms             , only : soil_rough         & ! intent(in)
    31)                                     , snow_rough         ! ! intent(in)
    32)    use therm_lib             , only : tq2enthalpy        ! ! function
    33)    use budget_utils          , only : update_budget      & ! function
    34)                                     , compute_budget     ! ! function
    35)    ! OMP use omp_lib
    36) 
    37)    implicit none
    38)    !----- Arguments -----------------------------------------------------------------------!
    39)    type(edtype)             , target      :: cgrid
    40)    !----- Local variables -----------------------------------------------------------------!
    41)    type(polygontype)        , pointer     :: cpoly
    42)    type(sitetype)           , pointer     :: csite
    43)    type(patchtype)          , pointer     :: cpatch
    44)    type(met_driv_state)     , pointer     :: cmet
    45)    integer                                :: ipy
    46)    integer                                :: isi
    47)    integer                                :: ipa
    48)    integer                                :: nsteps
    49)    integer                                :: imon
    50)    real                                   :: patch_vels
    51)    real                                   :: wcurr_loss2atm
    52)    real                                   :: ecurr_netrad
    53)    real                                   :: ecurr_loss2atm
    54)    real                                   :: co2curr_loss2atm
    55)    real                                   :: wcurr_loss2drainage
    56)    real                                   :: ecurr_loss2drainage
    57)    real                                   :: wcurr_loss2runoff
    58)    real                                   :: ecurr_loss2runoff
    59)    real                                   :: old_can_enthalpy
    60)    real                                   :: old_can_shv
    61)    real                                   :: old_can_co2
    62)    real                                   :: old_can_rhos
    63)    real                                   :: old_can_temp
    64)    real                                   :: old_can_prss
    65)    integer                                :: ibuff
    66)    ! OMP  integer(kind=OMP_integer_kind)   :: omp_ibuff
    67)    !----- Local constants. ----------------------------------------------------------------!
    68)    logical                  , parameter   :: test_energy_sanity = .false.
    69)    !---------------------------------------------------------------------------------------!
    70) 
    71)    ibuff = 1
    72) 
    73)    polyloop: do ipy = 1,cgrid%npolygons
    74)       cpoly => cgrid%polygon(ipy)
    75) 
    76)       siteloop: do isi = 1,cpoly%nsites
    77)          csite => cpoly%site(isi)
    78)          cmet  => cpoly%met(isi)
    79) 
    80)          !---------------------------------------------------------------------------------!
    81)          !     Update the monthly rainfall.                                                !
    82)          !---------------------------------------------------------------------------------!
    83)          imon                             = current_time%month
    84)          cpoly%avg_monthly_pcpg(imon,isi) = cpoly%avg_monthly_pcpg(imon,isi)               &
    85)                                           + cmet%pcpg * dtlsm
    86)          !---------------------------------------------------------------------------------!
    87) 
    88)          !------------------------------------------------------------------------------!
    89)          !    Copy the meteorological variables to the rk4site structure.               !
    90)          !------------------------------------------------------------------------------!
    91)          call copy_met_2_rk4site(nzg,cmet%atm_ustar                                     &
    92)                                    ,cmet%atm_theiv,cmet%atm_vpdef,cmet%atm_theta           &
    93)                                    ,cmet%atm_tmp,cmet%atm_shv,cmet%atm_co2,cmet%geoht      &
    94)                                    ,cmet%exner,cmet%pcpg,cmet%qpcpg,cmet%dpcpg,cmet%prss   &
    95)                                    ,cmet%rshort,cmet%rlong,cmet%par_beam,cmet%par_diffuse  &
    96)                                    ,cmet%nir_beam,cmet%nir_diffuse,cmet%geoht              &
    97)                                    ,cpoly%lsl(isi),cpoly%ntext_soil(:,isi)                 &
    98)                                    ,cpoly%green_leaf_factor(:,isi),cgrid%lon(ipy)          &
    99)                                    ,cgrid%lat(ipy),cgrid%cosz(ipy))
   100)             !------------------------------------------------------------------------------!
   101) 
   102)          patchloop: do ipa = 1,csite%npatches
   103)             cpatch => csite%patch(ipa)
   104) 
   105)             ibuff=1
   106) 
   107)             !----- Reset all buffers to zero, as a safety measure. ------------------------!
   108)             call zero_rk4_patch(integration_buff(ibuff)%initp)
   109)             call zero_rk4_patch(integration_buff(ibuff)%dinitp)
   110)             call zero_rk4_patch(integration_buff(ibuff)%ytemp)
   111)             call zero_rk4_patch(integration_buff(ibuff)%yerr)
   112)             call zero_rk4_patch(integration_buff(ibuff)%yscal)
   113)             call zero_rk4_patch(integration_buff(ibuff)%dydx)
   114)             call zero_rk4_cohort(integration_buff(ibuff)%initp)
   115)             call zero_rk4_cohort(integration_buff(ibuff)%dinitp)
   116)             call zero_rk4_cohort(integration_buff(ibuff)%ytemp)
   117)             call zero_rk4_cohort(integration_buff(ibuff)%yerr)
   118)             call zero_rk4_cohort(integration_buff(ibuff)%yscal)
   119)             call zero_rk4_cohort(integration_buff(ibuff)%dydx)
   120) 
   121)             !----- Get velocity for aerodynamic resistance. -------------------------------!
   122)             if (csite%can_theta(ipa) < cmet%atm_theta) then
   123) !               cmet%vels = cmet%vels_stab
   124)                patch_vels = cmet%vels_stab
   125)             else
   126) !               cmet%vels = cmet%vels_unstab
   127)                patch_vels = cmet%vels_unstab
   128)             end if
   129)             !------------------------------------------------------------------------------!
   130) 
   131) 
   132) 
   133)             !------------------------------------------------------------------------------!
   134)             !    Update roughness and canopy depth.                                        !
   135)             !------------------------------------------------------------------------------!
   136)             call update_patch_thermo_props(csite,ipa,ipa,nzg,nzs,cpoly%ntext_soil(:,isi))
   137)             call update_patch_derived_props(csite,ipa)
   138)             !------------------------------------------------------------------------------!
   139) 
   140) 
   141) 
   142)             !----- Save the previous thermodynamic state. ---------------------------------!
   143)             old_can_shv      = csite%can_shv(ipa)
   144)             old_can_co2      = csite%can_co2(ipa)
   145)             old_can_rhos     = csite%can_rhos(ipa)
   146)             old_can_temp     = csite%can_temp(ipa)
   147)             old_can_prss     = csite%can_prss(ipa)
   148)             old_can_enthalpy = tq2enthalpy(csite%can_temp(ipa),csite%can_shv(ipa),.true.)
   149)             !------------------------------------------------------------------------------!
   150) 
   151)             !----- Compute current storage terms. -----------------------------------------!
   152)             call update_budget(csite,cpoly%lsl(isi),ipa)
   153)             !------------------------------------------------------------------------------!
   154) 
   155) 
   156) 
   157)             !------------------------------------------------------------------------------!
   158)             !      Test whether temperature and energy are reasonable.                     !
   159)             !------------------------------------------------------------------------------!
   160)             if (test_energy_sanity) then
   161)                call sanity_check_veg_energy(csite,ipa)
   162)             end if
   163)             !------------------------------------------------------------------------------!
   164) 
   165) 
   166) 
   167)             !------------------------------------------------------------------------------!
   168)             !     Set up the integration patch.                                            !
   169)             !------------------------------------------------------------------------------!
   170)             call copy_patch_init(csite,ipa,integration_buff(ibuff)%initp,patch_vels)
   171)             !------------------------------------------------------------------------------!
   172) 
   173) 
   174) 
   175)             !----- Get photosynthesis, stomatal conductance, and transpiration. -----------!
   176)             call canopy_photosynthesis(csite,cmet,nzg,ipa                                  &
   177)                                       ,cpoly%ntext_soil(:,isi)                             &
   178)                                       ,cpoly%leaf_aging_factor(:,isi)                      &
   179)                                       ,cpoly%green_leaf_factor(:,isi))
   180)             !------------------------------------------------------------------------------!
   181) 
   182) 
   183) 
   184)             !----- Compute root and heterotrophic respiration. ----------------------------!
   185)             call soil_respiration(csite,ipa,nzg,cpoly%ntext_soil(:,isi))
   186)             !------------------------------------------------------------------------------!
   187) 
   188) 
   189)             !------------------------------------------------------------------------------!
   190)             !     Set up the remaining, carbon-dependent variables to the buffer.          !
   191)             !------------------------------------------------------------------------------!
   192)             call copy_patch_init_carbon(csite,ipa,integration_buff(ibuff)%initp)
   193)             !------------------------------------------------------------------------------!
   194) 
   195) 
   196)             !------------------------------------------------------------------------------!
   197)             !     This is the step in which the derivatives are computed, we a structure   !
   198)             ! that is very similar to the Runge-Kutta, though a simpler one.               !
   199)             !------------------------------------------------------------------------------!
   200)             call integrate_patch_euler(csite,integration_buff(ibuff)%initp                 &
   201)                                       ,integration_buff(ibuff)%dinitp                      &
   202)                                       ,integration_buff(ibuff)%ytemp                       &
   203)                                       ,integration_buff(ibuff)%yscal                       &
   204)                                       ,integration_buff(ibuff)%yerr                        &
   205)                                       ,integration_buff(ibuff)%dydx                        &
   206)                                       ,ipa,isi,cpoly%nighttime(isi)                        &
   207)                                       ,wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm          &
   208)                                       ,co2curr_loss2atm,wcurr_loss2drainage                &
   209)                                       ,ecurr_loss2drainage,wcurr_loss2runoff               &
   210)                                       ,ecurr_loss2runoff,nsteps)
   211)             !------------------------------------------------------------------------------!
   212) 
   213) 
   214) 
   215)             !----- Add the number of steps into the step counter. -------------------------!
   216)             cgrid%workload(13,ipy) = cgrid%workload(13,ipy) + real(nsteps)
   217)             !------------------------------------------------------------------------------!
   218) 
   219) 
   220) 
   221)             !------------------------------------------------------------------------------!
   222)             !    Update the minimum monthly temperature, based on canopy temperature.      !
   223)             !------------------------------------------------------------------------------!
   224)             if (cpoly%site(isi)%can_temp(ipa) < cpoly%min_monthly_temp(isi)) then
   225)                cpoly%min_monthly_temp(isi) = cpoly%site(isi)%can_temp(ipa)
   226)             end if
   227)             !------------------------------------------------------------------------------!
   228) 
   229) 
   230) 
   231)             !------------------------------------------------------------------------------!
   232)             !     Compute the residuals.                                                   !
   233)             !------------------------------------------------------------------------------!
   234)             call compute_budget(csite,cpoly%lsl(isi),cmet%pcpg,cmet%qpcpg,ipa              &
   235)                                ,wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm                 &
   236)                                ,co2curr_loss2atm,wcurr_loss2drainage,ecurr_loss2drainage   &
   237)                                ,wcurr_loss2runoff,ecurr_loss2runoff,cpoly%area(isi)        &
   238)                                ,cgrid%cbudget_nep(ipy),old_can_enthalpy,old_can_shv        &
   239)                                ,old_can_co2,old_can_rhos,old_can_prss)
   240)             !------------------------------------------------------------------------------!
   241)          end do patchloop
   242)          !---------------------------------------------------------------------------------!
   243)       end do siteloop
   244)       !------------------------------------------------------------------------------------!
   245)    end do polyloop
   246)    !---------------------------------------------------------------------------------------!
   247) 
   248)    return
   249) end subroutine euler_timestep

   258) !==========================================================================================!
   259) !==========================================================================================!
   260) !     This subroutine will drive the integration process using the Euler method.  Notice   !
   261) ! that most of the Euler method utilises the subroutines from Runge-Kutta.                 !
   262) !------------------------------------------------------------------------------------------!
   263) subroutine integrate_patch_euler(csite,initp,dinitp,ytemp,yscal,yerr,dydx,ipa,isi          &
   264)                                 ,nighttime,wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm      &
   265)                                 ,co2curr_loss2atm,wcurr_loss2drainage,ecurr_loss2drainage  &
   266)                                 ,wcurr_loss2runoff,ecurr_loss2runoff,nsteps)
   267)    use ed_state_vars   , only : sitetype             & ! structure
   268)                               , patchtype            ! ! structure
   269)    use ed_misc_coms    , only : dtlsm                ! ! intent(in)
   270)    use soil_coms       , only : soil_rough           & ! intent(in)
   271)                               , snow_rough           ! ! intent(in)
   272)    use canopy_air_coms , only : exar8                ! ! intent(in)
   273)    use rk4_coms        , only : integration_vars     & ! structure
   274)                               , rk4patchtype         & ! structure
   275)                               , rk4site              & ! intent(inout)
   276)                               , zero_rk4_patch       & ! subroutine
   277)                               , zero_rk4_cohort      & ! subroutine
   278)                               , tbeg                 & ! intent(inout)
   279)                               , tend                 & ! intent(inout)
   280)                               , dtrk4                & ! intent(inout)
   281)                               , dtrk4i               ! ! intent(inout)
   282)    use rk4_driver      , only : initp2modelp         ! ! subroutine
   283) 
   284)    implicit none
   285)    !----- Arguments -----------------------------------------------------------------------!
   286)    type(sitetype)        , target      :: csite
   287)    type(rk4patchtype)    , target      :: initp
   288)    type(rk4patchtype)    , target      :: dinitp
   289)    type(rk4patchtype)    , target      :: ytemp
   290)    type(rk4patchtype)    , target      :: yscal
   291)    type(rk4patchtype)    , target      :: yerr
   292)    type(rk4patchtype)    , target      :: dydx
   293)    integer               , intent(in)  :: ipa
   294)    integer               , intent(in)  :: isi
   295)    logical               , intent(in)  :: nighttime
   296)    real                  , intent(out) :: wcurr_loss2atm
   297)    real                  , intent(out) :: ecurr_netrad
   298)    real                  , intent(out) :: ecurr_loss2atm
   299)    real                  , intent(out) :: co2curr_loss2atm
   300)    real                  , intent(out) :: wcurr_loss2drainage
   301)    real                  , intent(out) :: ecurr_loss2drainage
   302)    real                  , intent(out) :: wcurr_loss2runoff
   303)    real                  , intent(out) :: ecurr_loss2runoff
   304)    integer               , intent(out) :: nsteps
   305)    !----- Local variables -----------------------------------------------------------------!
   306)    real(kind=8)                        :: hbeg
   307)    !----- Locally saved variable ----------------------------------------------------------!
   308)    logical                  , save     :: first_time=.true.
   309)    !---------------------------------------------------------------------------------------!
   310) 
   311)    !----- Assigning some constants which will remain the same throughout the run. ---------!
   312)    if (first_time) then
   313)       first_time = .false.
   314)       tbeg   = 0.d0
   315)       tend   = dble(dtlsm)
   316)       dtrk4  = tend - tbeg
   317)       dtrk4i = 1.d0/dtrk4
   318)    end if
   319) 
   320)    !---------------------------------------------------------------------------------------!
   321)    !      Initial step size.  Experience has shown that giving this too large a value      !
   322)    ! causes the integrator to fail (e.g., soil layers become supersaturated).              !
   323)    !---------------------------------------------------------------------------------------!
   324)    hbeg = dble(csite%htry(ipa))
   325) 
   326)    !---------------------------------------------------------------------------------------!
   327)    !     Zero the canopy-atmosphere flux values.  These values are updated every dtlsm,    !
   328)    ! so they must be zeroed at each call.                                                  !
   329)    !---------------------------------------------------------------------------------------!
   330)    initp%upwp = 0.d0
   331)    initp%tpwp = 0.d0
   332)    initp%qpwp = 0.d0
   333)    initp%cpwp = 0.d0
   334)    initp%wpwp = 0.d0
   335) 
   336)    !----- Go into the ODE integrator using Euler. -----------------------------------------!
   337)    call euler_integ(hbeg,csite,initp,dinitp,ytemp,yscal,yerr,dydx,ipa,isi,nsteps)
   338) 
   339)    !---------------------------------------------------------------------------------------!
   340)    !      Normalize canopy-atmosphere flux values.  These values are updated every         !
   341)    ! dtlsm, so they must be normalized every time.                                         !
   342)    !---------------------------------------------------------------------------------------!
   343)    initp%upwp = initp%can_rhos * initp%upwp * dtrk4i
   344)    initp%tpwp = initp%can_rhos * initp%tpwp * dtrk4i
   345)    initp%qpwp = initp%can_rhos * initp%qpwp * dtrk4i
   346)    initp%cpwp = initp%can_rhos * initp%cpwp * dtrk4i
   347)    initp%wpwp = initp%can_rhos * initp%wpwp * dtrk4i
   348) 
   349)    !---------------------------------------------------------------------------------------!
   350)    ! Move the state variables from the integrated patch to the model patch.                !
   351)    !---------------------------------------------------------------------------------------!
   352)    call initp2modelp(tend-tbeg,initp,csite,ipa,nighttime,wcurr_loss2atm,ecurr_netrad       &
   353)                     ,ecurr_loss2atm,co2curr_loss2atm,wcurr_loss2drainage                   &
   354)                     ,ecurr_loss2drainage,wcurr_loss2runoff,ecurr_loss2runoff)
   355) 
   356)    return
   357) end subroutine integrate_patch_euler

   368) ! Subroutine odeint                                                                        !
   369) !                                                                                          !
   370) !     This subroutine will drive the integration of several ODEs that drive the fast-scale !
   371) ! state variables.                                                                         !
   372) !------------------------------------------------------------------------------------------!
   373) subroutine euler_integ(h1,csite,initp,dinitp,ytemp,yscal,yerr,dydx,ipa,isi,nsteps)
   374)    use rk4_copy_patch
   375)    use rk4_integ_utils
   376)    use rk4_misc
   377)    use ed_state_vars  , only : sitetype               & ! structure
   378)                              , patchtype              & ! structure
   379)                              , polygontype            ! ! structure
   380)    use rk4_coms       , only : integration_vars       & ! structure
   381)                              , rk4patchtype           & ! structure
   382)                              , rk4site                & ! intent(in)
   383)                              , print_diags            & ! intent(in)
   384)                              , maxstp                 & ! intent(in)
   385)                              , tbeg                   & ! intent(in)
   386)                              , tend                   & ! intent(in)
   387)                              , dtrk4                  & ! intent(in)
   388)                              , dtrk4i                 & ! intent(in)
   389)                              , tiny_offset            & ! intent(in)
   390)                              , checkbudget            & ! intent(in)
   391)                              , zero_rk4_patch         & ! subroutine
   392)                              , zero_rk4_cohort        & ! subroutine
   393)                              , hmin                   & ! intent(in)
   394)                              , rk4eps                 & ! intent(in)
   395)                              , rk4epsi                & ! intent(in)
   396)                              , safety                 & ! intent(in)
   397)                              , pgrow                  & ! intent(in)
   398)                              , pshrnk                 & ! intent(in)
   399)                              , errcon                 & ! intent(in)
   400)                              , print_detailed         & ! intent(in)
   401)                              , norm_rk4_fluxes        & ! sub-routine
   402)                              , reset_rk4_fluxes       ! ! sub-routine
   403)    use ed_misc_coms   , only : fast_diagnostics       ! ! intent(in)
   404)    use hydrology_coms , only : useRUNOFF              ! ! intent(in)
   405)    use grid_coms      , only : nzg                    & ! intent(in)
   406)                              , nzs                    & ! intent(in)
   407)                              , time                   ! ! intent(in)
   408)    use soil_coms      , only : dslz8                  & ! intent(in)
   409)                              , runoff_time            & ! intent(in)
   410)                              , runoff_time_i          & ! intent(in)
   411)                              , simplerunoff           ! ! intent(in)
   412)    use consts_coms    , only : t3ple8                 & ! intent(in)
   413)                              , wdnsi8                 ! ! intent(in)
   414)    use therm_lib8     , only : tl2uint8               ! ! intent(in)
   415)    implicit none
   416)    !----- Arguments -----------------------------------------------------------------------!
   417)    type(sitetype)            , target      :: csite            ! Current site
   418)    type(rk4patchtype)        , target      :: initp            ! Current integ. patch
   419)    type(rk4patchtype)        , target      :: dinitp           ! Integration derivative
   420)    type(rk4patchtype)        , target      :: ytemp            ! Temporary integ. patch
   421)    type(rk4patchtype)        , target      :: yscal            ! Scale for error analysis
   422)    type(rk4patchtype)        , target      :: yerr             ! Patch integration error
   423)    type(rk4patchtype)        , target      :: dydx             ! Patch integration error
   424)    integer                   , intent(in)  :: ipa              ! Current patch ID
   425)    integer                   , intent(in)  :: isi              ! Current patch ID
   426)    real(kind=8)              , intent(in)  :: h1               ! First guess of delta-t
   427)    integer                   , intent(out) :: nsteps           ! Number of steps taken.
   428)    !----- Local variables -----------------------------------------------------------------!
   429)    type(patchtype)           , pointer     :: cpatch           ! Current patch
   430)    logical                                 :: reject_step      ! Should I reject the step?
   431)    logical                                 :: minstep          ! Minimum time step reached
   432)    logical                                 :: stuck            ! Tiny step, it won't advance
   433)    logical                                 :: test_reject      ! Reject the test
   434)    integer                                 :: i                ! Step counter
   435)    integer                                 :: k                ! Format counter
   436)    integer                                 :: ksn              ! # of snow/water layers
   437)    real(kind=8)                            :: x                ! Elapsed time
   438)    real(kind=8)                            :: xnew             ! Elapsed time + h
   439)    real(kind=8)                            :: newh             ! New time step suggested
   440)    real(kind=8)                            :: oldh             ! Old time step
   441)    real(kind=8)                            :: h                ! Current delta-t attempt
   442)    real(kind=8)                            :: hnext            ! Next delta-t
   443)    real(kind=8)                            :: qwfree           ! Free water internal energy
   444)    real(kind=8)                            :: wfreeb           ! Free water
   445)    real(kind=8)                            :: errmax           ! Maximum error of this step
   446)    real(kind=8)                            :: elaptime         ! Absolute elapsed time.
   447)    !----- External function. --------------------------------------------------------------!
   448)    real                      , external    :: sngloff
   449)    !---------------------------------------------------------------------------------------!
   450) 
   451)    !----- Use some aliases for simplicity. ------------------------------------------------!
   452)    cpatch => csite%patch(ipa)
   453) 
   454)    !---------------------------------------------------------------------------------------!
   455)    ! Set initial time and stepsize.                                                        !
   456)    !---------------------------------------------------------------------------------------!
   457)    x = tbeg
   458)    h = h1
   459)    if (dtrk4 < 0.d0) h = -h1
   460) 
   461)    !----- Define total elapsed time. ------------------------------------------------------!
   462)    elaptime = time + x
   463) 
   464)    !---------------------------------------------------------------------------------------!
   465)    ! Begin timestep loop                                                                   !
   466)    !---------------------------------------------------------------------------------------!
   467)    timesteploop: do i=1,maxstp
   468) 
   469) 
   470)       !----- Get initial derivatives ------------------------------------------------------!
   471)       call leaf_derivs(initp,dinitp,csite,ipa,h,.false.)
   472) 
   473)       !----- Get scalings used to determine stability -------------------------------------!
   474)       call get_yscal(initp,dinitp,h,yscal,cpatch)
   475) 
   476)       !----- Be sure not to overstep ------------------------------------------------------!
   477)       if((x+h-tend)*(x+h-tbeg) > 0.d0) h=tend-x
   478) 
   479)       !------------------------------------------------------------------------------------!
   480)       !     Here we will perform the Euler integration using the time step.  As in Runge-  !
   481)       ! Kutta, we also check whether the integration is going well and if needed we shrink !
   482)       ! the intermediate time steps.  However, we cannot estimate the errors as in Runge-  !
   483)       ! Kutta or Euler, so steps are shrunk only when it fails the sanity check.           !
   484)       !------------------------------------------------------------------------------------!
   485)       reject_step =  .false.
   486)       hstep:   do
   487) 
   488)          !----- Copy patch to the temporary structure. ------------------------------------!
   489)          call copy_rk4_patch(initp, ytemp, cpatch)
   490) 
   491)          !---------------------------------------------------------------------------------!
   492)          !    Integrate, then update and correct diagnostic variables to avoid overshoot-  !
   493)          ! ing, provided that the overshooting is small.                                   !
   494)          !---------------------------------------------------------------------------------!
   495)          call inc_rk4_patch(ytemp,dinitp,h,cpatch)
   496)          call update_diagnostic_vars(ytemp,csite,ipa)
   497) 
   498)          !----- Perform a sanity check. ---------------------------------------------------!
   499)          call rk4_sanity_check(ytemp,reject_step,csite,ipa,dinitp,h,print_diags)
   500) 
   501)          !---------------------------------------------------------------------------------!
   502)          !     Here we check the error of this step.  Three outcomes are possible:         !
   503)          ! 1.  The updated values make no sense.  Reject step, assign a large error and    !
   504)          !     try again with a smaller time step;                                         !
   505)          ! 2.  The updated values are reasonable, but the error is large.  Reject step and !
   506)          !     try again with a smaller time step;                                         !
   507)          ! 3.  The updated values are reasonable, and the error is small.  Accept step and !
   508)          !     try again with a larger time step.                                          !
   509)          !---------------------------------------------------------------------------------!
   510)          if (reject_step) then
   511)             !------------------------------------------------------------------------------!
   512)             !    If step was already rejected, that means the step had finished premature- !
   513)             ! ly, so we assign a standard large error (10.0).                              !
   514)             !------------------------------------------------------------------------------!
   515)             errmax = 1.d1
   516)          else
   517) 
   518)             errmax = 1.d-1
   519)          end if
   520) 
   521) 
   522)          !---------------------------------------------------------------------------------!
   523)          ! 3. If the step failed, then calculate a new shorter step size to try.           !
   524)          !---------------------------------------------------------------------------------!
   525)          if (reject_step) then
   526)             !----- Defining new step and checking if it can be. ---------------------------!
   527)             oldh    = h
   528)             newh    = safety * h * errmax**pshrnk
   529)             minstep = (newh == h) .or. newh < hmin
   530) 
   531)             !----- Defining next time, and checking if it really added something. ---------!
   532)             h       = max(1.d-1*h, newh)
   533)             xnew    = x + h
   534)             stuck   = xnew == x
   535) 
   536)             !------------------------------------------------------------------------------!
   537)             ! 3a. Here is the moment of truth... If we reached a tiny step and yet the     !
   538)             !     model didn't converge, then we print various values to inform the user   !
   539)             !     and abort the run.  Please, don't hate the messenger.                    !
   540)             !------------------------------------------------------------------------------!
   541)             if (minstep .or. stuck) then
   542) 
   543)                write (unit=*,fmt='(80a)')         ('=',k=1,80)
   544)                write (unit=*,fmt='(a)')           '   STEPSIZE UNDERFLOW IN EULER_INT'
   545)                write (unit=*,fmt='(80a)')         ('-',k=1,80)
   546)                write (unit=*,fmt='(a,1x,f9.4)')   ' + LONGITUDE:     ',rk4site%lon
   547)                write (unit=*,fmt='(a,1x,f9.4)')   ' + LATITUDE:      ',rk4site%lat
   548)                write (unit=*,fmt='(a)')           ' + PATCH INFO:    '
   549)                write (unit=*,fmt='(a,1x,i6)')     '   - NUMBER:      ',ipa
   550)                write (unit=*,fmt='(a,1x,es12.4)') '   - AGE:         ',csite%age(ipa)
   551)                write (unit=*,fmt='(a,1x,i6)')     '   - DIST_TYPE:   ',csite%dist_type(ipa)
   552)                write (unit=*,fmt='(a,1x,l1)')     ' + MINSTEP:       ',minstep
   553)                write (unit=*,fmt='(a,1x,l1)')     ' + STUCK:         ',stuck
   554)                write (unit=*,fmt='(a,1x,es12.4)') ' + ERRMAX:        ',errmax
   555)                write (unit=*,fmt='(a,1x,es12.4)') ' + X:             ',x
   556)                write (unit=*,fmt='(a,1x,es12.4)') ' + H:             ',h
   557)                write (unit=*,fmt='(a,1x,es12.4)') ' + OLDH:          ',oldh
   558)                write (unit=*,fmt='(a,1x,es12.4)') ' + NEWH:          ',newh
   559)                write (unit=*,fmt='(a,1x,es12.4)') ' + SAFETY:        ',safety
   560)                write (unit=*,fmt='(80a)') ('-',k=1,80)
   561)                write (unit=*,fmt='(a)') '   Likely to be a rejected step problem.'
   562)                write (unit=*,fmt='(80a)') ('=',k=1,80)
   563) 
   564)                call rk4_sanity_check(ytemp,test_reject,csite,ipa,dinitp,h,.true.)
   565)                call print_sanity_check(ytemp,csite,ipa)
   566)                call print_rk4patch(ytemp, csite,ipa)
   567)             end if
   568) 
   569)          else
   570)             !------------------------------------------------------------------------------!
   571)             ! 3b.  Great, it worked, so now we can advance to the next step.  We just need !
   572)             !      to do some minor adjustments before...                                  !
   573)             !------------------------------------------------------------------------------!
   574)             !----- i.   Final update of leaf properties to avoid negative water. ----------!
   575)             call adjust_veg_properties(ytemp,h,csite,ipa)
   576)             !----- ii.  Final update of top soil properties to avoid off-bounds moisture. -!
   577)             call adjust_topsoil_properties(ytemp,h,csite)
   578)             !----- ii. Make temporary surface water stable and positively defined. --------!
   579)             call adjust_sfcw_properties(nzg,nzs,ytemp,h,csite)
   580)             !----- iii.  Update the diagnostic variables. ---------------------------------!
   581)             call update_diagnostic_vars(ytemp, csite,ipa)
   582)             !------------------------------------------------------------------------------!
   583) 
   584)             !------------------------------------------------------------------------------!
   585)             ! 3c. Set up h for the next time.  And here we can relax h for the next step,  !
   586)             !    and try something faster.                                                 !
   587)             !------------------------------------------------------------------------------!
   588)             if (errmax > errcon) then
   589)                hnext = safety * h * errmax**pgrow
   590)             else
   591)                hnext = 5.d0 * h
   592)             endif
   593)             hnext = max(2.d0*hmin,hnext)
   594) 
   595)             call leaf_derivs(ytemp,dydx,csite,ipa,hnext,.false.)
   596) 
   597) 
   598)             !------ 3d. Normalise the fluxes if the user wants detailed debugging. --------!
   599)             if (print_detailed) then
   600)                call norm_rk4_fluxes(ytemp,h)
   601)                call print_rk4_state(initp,ytemp,csite,ipa,isi,x,h)
   602)             end if
   603) 
   604)             !----- 3e. Copy the temporary structure to the intermediate state. ------------!
   605)             call copy_rk4_patch(ytemp, initp,csite%patch(ipa))
   606)             call copy_rk4_patch(dydx ,dinitp,csite%patch(ipa))
   607) 
   608)             !------------------------------------------------------------------------------!
   609)             !    3f. Flush step-by-step fluxes to zero if the user wants detailed          !
   610)             !        debugging.                                                            !
   611)             !------------------------------------------------------------------------------!
   612)             if (print_detailed) then
   613)                call reset_rk4_fluxes(initp)
   614)             end if
   615) 
   616)             !----- 3g. Update time. -------------------------------------------------------!
   617)             x        = x + h
   618)             h        = hnext
   619)             elaptime = elaptime + h
   620) 
   621)             exit hstep
   622)          end if
   623)       end do hstep
   624) 
   625)       !----- If the integration reached the next step, make some final adjustments --------!
   626)       if((x-tend)*dtrk4 >= 0.d0)then
   627) 
   628)          ksn = initp%nlev_sfcwater
   629) 
   630)          !---------------------------------------------------------------------------------!
   631)          !   Make temporary surface liquid water disappear.  This will not happen          !
   632)          ! immediately, but liquid water will decay with the time scale defined by         !
   633)          ! runoff_time scale.                                                              !
   634)          !---------------------------------------------------------------------------------!
   635)          if (simplerunoff .and. ksn >= 1) then
   636) 
   637)             if (initp%sfcwater_mass(ksn)    > 0.d0   .and.                                 &
   638)                 initp%sfcwater_fracliq(ksn) > 1.d-1        ) then
   639) 
   640)                wfreeb = min(1.d0,dtrk4*runoff_time_i) * initp%sfcwater_mass(ksn)           &
   641)                       * (initp%sfcwater_fracliq(ksn) - 1.d-1) / 9.d-1
   642)                qwfree = wfreeb * tl2uint8(initp%sfcwater_tempk(ksn),1.d0)
   643) 
   644)                initp%sfcwater_mass(ksn) = initp%sfcwater_mass(ksn)   - wfreeb
   645)                initp%sfcwater_depth(ksn) = initp%sfcwater_depth(ksn) - wfreeb * wdnsi8
   646)                !----- Recompute the energy removing runoff --------------------------------!
   647)                initp%sfcwater_energy(ksn) = initp%sfcwater_energy(ksn) - qwfree
   648) 
   649)                call adjust_sfcw_properties(nzg,nzs,initp,dtrk4,csite)
   650)                call update_diagnostic_vars(initp,csite,ipa)
   651) 
   652)                !----- Compute runoff for output -------------------------------------------!
   653)                if (fast_diagnostics) then
   654)                   csite%runoff       (ipa) = csite%runoff(ipa)                             &
   655)                                            + sngloff(wfreeb * dtrk4i,tiny_offset)
   656)                   csite%fmean_runoff (ipa) = csite%fmean_runoff(ipa)                       &
   657)                                            + sngloff(wfreeb * dtrk4i,tiny_offset)
   658)                   csite%fmean_qrunoff(ipa) = csite%fmean_qrunoff(ipa)                      &
   659)                                            + sngloff(qwfree * dtrk4i,tiny_offset)
   660)                end if
   661)                if (checkbudget) then
   662)                   initp%wbudget_loss2runoff = initp%wbudget_loss2runoff + wfreeb
   663)                   initp%ebudget_loss2runoff = initp%ebudget_loss2runoff + qwfree
   664)                   initp%wbudget_storage     = initp%wbudget_storage - wfreeb
   665)                   initp%ebudget_storage     = initp%ebudget_storage - qwfree
   666)                end if
   667)             end if
   668)          end if
   669)          !------ Update the substep for next time and leave -------------------------------!
   670)          csite%htry(ipa) = sngl(hnext)
   671) 
   672)          !---------------------------------------------------------------------------------!
   673)          !     Update the average time step.  The square of DTLSM (tend-tbeg) is needed    !
   674)          ! because we will divide this by the time between t0 and t0+frqsum.               !
   675)          !---------------------------------------------------------------------------------!
   676)          csite%fmean_rk4step(ipa) = csite%fmean_rk4step(ipa)                               &
   677)                                   + sngl((tend-tbeg)*(tend-tbeg))/real(i)
   678)          nsteps = i
   679)          return
   680)       end if
   681) 
   682)       !----- Use hnext as the next substep ------------------------------------------------!
   683)       h = hnext
   684)    end do timesteploop
   685) 
   686)    !----- If it reached this point, that is really bad news... ----------------------------!
   687)    write (unit=*,fmt='(a)') ' ==> Too many steps in routine euler_integ'
   688)    call print_rk4patch(ytemp, csite,ipa)
   689) 
   690)    return
   691) end subroutine euler_integ
