    13)    !    This subroutine calculates phenology factors for prescribed phenology schemes.     !
    14)    !---------------------------------------------------------------------------------------!
    15)    subroutine prescribed_leaf_state(lat,imonth,iyear,doy,green_leaf_factor                 &
    16)                                    ,leaf_aging_factor,phen_pars)
    17) 
    18)       use phenology_coms , only : iphenys1         & ! intent(in)
    19)                                 , iphenysf         & ! intent(in)
    20)                                 , iphenyf1         & ! intent(in)
    21)                                 , iphenyff         & ! intent(in)
    22)                                 , prescribed_phen  & ! intent(in)
    23)                                 , elongf_min       ! ! intent(in)
    24)       use ed_max_dims    , only : n_pft            ! ! intent(in)
    25)       use pft_coms       , only : phenology        ! ! intent(in)
    26)       implicit none
    27)       !----- Arguments --------------------------------------------------------------------!
    28)       type(prescribed_phen) , intent(in)  :: phen_pars
    29)       integer               , intent(in)  :: iyear
    30)       integer               , intent(in)  :: doy
    31)       real                  , intent(in)  :: lat
    32)       integer               , intent(in)  :: imonth
    33)       real, dimension(n_pft), intent(out) :: green_leaf_factor
    34)       real, dimension(n_pft), intent(out) :: leaf_aging_factor
    35)       !----- Local variables --------------------------------------------------------------!
    36)       integer                             :: n_recycle_years
    37)       integer                             :: my_year
    38)       real                                :: elongf
    39)       real                                :: delay
    40)       real(kind=8)                        :: elonDen
    41)       integer                             :: pft
    42)       !------------------------------------------------------------------------------------!
    43) 
    44)       !------------------------------------------------------------------------------------!
    45)       !     This assumes dropping/flushing based on the day of year and hemisphere.        !
    46)       ! + Northern Hemisphere: dropping between August 1 and December 31;                  !
    47)       !                        flushing between January 1 and July 31.                     !
    48)       ! + Southern Hemisphere: dropping between February 1 and July 31;                    !
    49)       !                        flushing between August 1 and January 31.                   !
    50)       !------------------------------------------------------------------------------------!
    51)       if( (lat >= 0.0 .and. imonth <= 7) .or.                                              &
    52)           (lat < 0.0  .and. (imonth > 7 .or. imonth == 1)) )then
    53) 
    54)          !----- Get the year. -------------------------------------------------------------!
    55)          n_recycle_years = iphenysf - iphenys1 + 1
    56) 
    57)          if (iyear > iphenysf) then
    58)             my_year = mod(iyear-iphenys1,n_recycle_years) + 1
    59)          elseif (iyear < iphenys1) then
    60)             my_year = n_recycle_years - mod(iphenysf-iyear,n_recycle_years)
    61)          else
    62)             my_year = iyear - iphenys1 + 1
    63)          end if
    64) 
    65)          !---------------------------------------------------------------------------------!
    66)          !      Calculate the factors.  Precalc denominator and limit rate in order to     !
    67)          ! increase numerical stability (MCD 10/23/08).                                    !
    68)          !---------------------------------------------------------------------------------!
    69)          elonDen = real((phen_pars%flush_a(my_year) * real(doy)),kind=8)                   &
    70)                  ** dble(max(phen_pars%flush_b(my_year),-100.))
    71)          elonDen = 1.0d0 / (1.0d0 + elonDen)
    72)          if(elonDen < 0.0001d0) then
    73)             elongf = 0.0
    74)          else
    75)             elongf = sngl(elonDen)
    76)          end if
    77)          delay = elongf
    78)       else
    79)          !---------------------------------------------------------------------------------!
    80)          !      Leaves turning color.  Get the year.                                       !
    81)          !---------------------------------------------------------------------------------!
    82)          n_recycle_years = iphenyff - iphenyf1 + 1
    83)          if (iyear > iphenyff) then
    84)             my_year = mod(iyear-iphenyf1,n_recycle_years) + 1
    85)          elseif (iyear < iphenyf1) then
    86)             my_year = n_recycle_years - mod(iphenyff-iyear,n_recycle_years)
    87)          else
    88)             my_year = iyear - iphenyf1 + 1
    89)          end if
    90) 
    91)          !----- Calculate the factors. ----------------------------------------------------!
    92)          elongf =  1.0                                                                     &
    93)                 /  (1.0 + (phen_pars%color_a(my_year) * real(doy))                         &
    94)                 ** phen_pars%color_b(my_year))
    95)          delay  =  1.0                                                                     &
    96)                 /  (1.0 + (phen_pars%color_a(my_year) * real(doy) * 1.095)                 &
    97)                 ** phen_pars%color_b(my_year))
    98)       end if
    99) 
   100)       if(elongf < elongf_min) elongf = 0.0
   101) 
   102)       !----- Load the values for each PFT. ------------------------------------------------!
   103)       do pft = 1, n_pft
   104)          select case (phenology(pft))
   105)          case (2)
   106)             green_leaf_factor(pft) = elongf
   107)             leaf_aging_factor(pft) = delay
   108)          case default
   109)             green_leaf_factor(pft) = 1.0
   110)             leaf_aging_factor(pft) = 1.0
   111)          end select
   112)       end do
   113) 
   114)       return
   115)    end subroutine prescribed_leaf_state

   126)    !     This subroutine computes the number of chill and warming days in a month.         !
   127)    !  + Chill days  - number of days with average temperatures below 278.15 K;             !
   128)    !  + Degree days - sum of daily average temperatures above 278.15 K.                    !
   129)    !---------------------------------------------------------------------------------------!
   130)    subroutine update_thermal_sums(month, cpoly, isi, lat)
   131) 
   132)       use ed_state_vars ,only : polygontype & ! structure
   133)                               , sitetype    ! ! structure
   134)       implicit none
   135)       !----- Arguments --------------------------------------------------------------------!
   136)       type(polygontype) , target     :: cpoly
   137)       integer           , intent(in) :: isi
   138)       integer           , intent(in) :: month
   139)       real              , intent(in) :: lat
   140)       !----- Local variables --------------------------------------------------------------!
   141)       type(sitetype)    , pointer    :: csite
   142)       integer                        :: ipa
   143)       !------------------------------------------------------------------------------------!
   144) 
   145) 
   146) 
   147)       !----- Loop over patches. -----------------------------------------------------------!
   148)       csite => cpoly%site(isi)
   149) 
   150)       do ipa = 1,csite%npatches
   151) 
   152)          !----- Minimum monthly temperature of the site. ----------------------------------!
   153)          cpoly%min_monthly_temp(isi) = min(cpoly%min_monthly_temp(isi)                     &
   154)                                           ,csite%avg_daily_temp(ipa))
   155) 
   156)          !---------------------------------------------------------------------------------!
   157)          !    Warm day, so check whether it is growing season and update the degree day... !
   158)          !---------------------------------------------------------------------------------!
   159)          if (csite%avg_daily_temp(ipa) > 278.15) then
   160)             !----- Update dgd only for growing season. ------------------------------------!
   161)             if ((lat >= 0.0 .and. month <= 8) .or.                                         &
   162)                 (lat <  0.0 .and. (month <= 2 .or. month >= 7))) then
   163)                csite%sum_dgd(ipa) = csite%sum_dgd(ipa) + (csite%avg_daily_temp(ipa)-278.15)
   164)             !----- Warm day during dropping season, set degree sum to zero... -------------!
   165)             else
   166)                csite%sum_dgd(ipa) = 0.0
   167)             end if
   168)          !---- Cold day, check whether it is dropping season and update chilling days... --!
   169)          elseif ((lat >= 0.0 .and. (month >= 11 .or. month <= 6)) .or.                     &
   170)                  (lat <  0.0 .and.  month >= 5)                 ) then
   171)             csite%sum_chd(ipa) = csite%sum_chd(ipa) + 1.0
   172)          !---- Cold day, but not during dropping season, set chilling days to zero... -----!
   173)          else
   174)             csite%sum_chd(ipa) = 0.0
   175)          end if
   176)       end do
   177) 
   178)       return
   179)    end subroutine update_thermal_sums

   190)    !     This subroutine updates the turnover ratio and specific leaf area, taking into    !
   191)    ! account the available radiation.                                                      !
   192)    !---------------------------------------------------------------------------------------!
   193)    subroutine update_turnover(cpoly, isi)
   194)       use ed_state_vars  , only : polygontype        & ! structure
   195)                                 , sitetype           & ! structure
   196)                                 , patchtype          ! ! structure
   197)       use pft_coms       , only : phenology          & ! intent(in)
   198)                                 , sla_scale          & ! intent(in)
   199)                                 , sla_inter          & ! intent(in)
   200)                                 , sla_slope          & ! intent(in)
   201)                                 , leaf_turnover_rate ! ! intent(in)
   202)       use phenology_coms , only : radint             & ! intent(in)
   203)                                 , radslp             & ! intent(in)
   204)                                 , turnamp_wgt        & ! intent(out)
   205)                                 , turnamp_min        & ! intent(out)
   206)                                 , turnamp_max        & ! intent(out)
   207)                                 , radto_min          & ! intent(out)
   208)                                 , radto_max          & ! intent(out)
   209)                                 , llspan_wgt         & ! intent(out)
   210)                                 , llspan_min         & ! intent(out)
   211)                                 , llspan_max         & ! intent(out)
   212)                                 , llspan_inf         & ! intent(out)
   213)                                 , vm0_wgt            & ! intent(out)
   214)                                 , vm0_tran           & ! intent(out)
   215)                                 , vm0_slope          & ! intent(out)
   216)                                 , vm0_amp            & ! intent(out)
   217)                                 , vm0_min            ! ! intent(out)
   218)       use consts_coms    , only : day_sec            ! ! intent(in)
   219) 
   220)       implicit none
   221)       !----- Arguments --------------------------------------------------------------------!
   222)       type(polygontype) , target     :: cpoly
   223)       integer           , intent(in) :: isi
   224)       !----- Local variables --------------------------------------------------------------!
   225)       type(sitetype)    , pointer    :: csite
   226)       type(patchtype)   , pointer    :: cpatch
   227)       integer                        :: ipa
   228)       integer                        :: ico
   229)       integer                        :: ipft
   230)       real                           :: turnover_now
   231)       real                           :: turnamp_now
   232)       real                           :: llspan_now
   233)       real                           :: vm0_now
   234)       !------------------------------------------------------------------------------------!
   235) 
   236) 
   237) 
   238)       !----- Loop over patches. -----------------------------------------------------------!
   239)       csite => cpoly%site(isi)
   240)       patchloop: do ipa = 1,csite%npatches
   241) 
   242)          cpatch => csite%patch(ipa)
   243)          cohortloop: do ico = 1,cpatch%ncohorts
   244) 
   245)             ipft = cpatch%pft(ico)
   246) 
   247) 
   248)             !------------------------------------------------------------------------------!
   249)             !     We must check whether the light phenology is to be applied for this PFT. !
   250)             !------------------------------------------------------------------------------!
   251)             select case (phenology(ipft))
   252)             case (3)
   253)                !---------------------------------------------------------------------------!
   254)                !      Find the target turnover rate amplitude (turnamp_now).               !
   255)                !---------------------------------------------------------------------------!
   256)                if (cpoly%rad_avg(isi) <= radto_min) then
   257)                   turnamp_now = turnamp_min
   258)                elseif (cpoly%rad_avg(isi) >= radto_max) then
   259)                   turnamp_now = turnamp_max
   260)                else
   261)                   turnamp_now = radint + radslp * cpoly%rad_avg(isi)
   262)                end if
   263)                !---------------------------------------------------------------------------!
   264) 
   265) 
   266)                !------ The actual turnover amplitude is based on a running average. -------!
   267)                cpatch%turnover_amp(ico) = (1.0 - turnamp_wgt) * cpatch%turnover_amp(ico)   &
   268)                                         +        turnamp_wgt  * turnamp_now
   269)                !---------------------------------------------------------------------------!
   270) 
   271) 
   272) 
   273)                !---------------------------------------------------------------------------!
   274)                !     Update target leaf lifespan.                                          !
   275)                !---------------------------------------------------------------------------!
   276)                if (leaf_turnover_rate(ipft) > 0.) then
   277)                   llspan_now = 12.0 / (cpatch%turnover_amp(ico) * leaf_turnover_rate(ipft))
   278)                   !----- Make sure the life span is bounded. ------------------------------!
   279)                   if ( llspan_now < llspan_min) then
   280)                       llspan_now = llspan_min
   281)                   elseif (llspan_now > llspan_max) then
   282)                       llspan_now = llspan_max
   283)                   end if
   284)                   !------------------------------------------------------------------------!
   285)                else
   286)                   !---- Nothing lasts forever, so impose a maximum life span. -------------!
   287)                   llspan_now = llspan_inf
   288)                   !------------------------------------------------------------------------!
   289)                end if
   290)                !---------------------------------------------------------------------------!
   291) 
   292) 
   293) 
   294)                !---------------------------------------------------------------------------!
   295)                !     The actual leaf lifespan is the weighted average.                     !
   296)                !---------------------------------------------------------------------------!
   297)                cpatch%llspan(ico) = (1.0 - llspan_wgt) * cpatch%llspan(ico)                &
   298)                                   +        llspan_wgt  * llspan_now
   299)                !---------------------------------------------------------------------------!
   300) 
   301) 
   302) 
   303)                !----- Update the running average of the photosythetic capacity (Vm0). -----!
   304)                vm0_now = vm0_amp / (1.0 + (cpatch%llspan(ico) / vm0_tran)**vm0_slope)      &
   305)                        + vm0_min
   306)                cpatch%vm_bar(ico) = (1.0 - vm0_wgt) * cpatch%vm_bar(ico)                   &
   307)                                   + vm0_wgt * vm0_now
   308)                !---------------------------------------------------------------------------!
   309) 
   310) 
   311) 
   312)                !----- Update the specific leaf area (SLA). --------------------------------!
   313)                turnover_now    = cpatch%turnover_amp(ico) * leaf_turnover_rate(ipft)
   314)                cpatch%sla(ico) = 10.0 ** ( sla_inter                                       &
   315)                                          + sla_slope * log10( 12.0 / turnover_now ) )      &
   316)                                * sla_scale
   317)             case default
   318)                !----- The default is to keep these variables the same. --------------------!
   319)                continue
   320)                !---------------------------------------------------------------------------!
   321)             end select
   322)          end do cohortloop
   323) 
   324)       end do patchloop
   325) 
   326)       return
   327)    end subroutine update_turnover

   338)    !      This sub-routine will assign the initial potential available water and the       !
   339)    ! phenology that has been assigned.  This sub-routine should be called whenever a new   !
   340)    ! cohort is created, or at the initial run (except history).  The initial running aver- !
   341)    ! age is simply the the instantaneous soil moisture variable.  For plants other than    !
   342)    ! the drought-deciduous, the potential available water is found but it doesn't control  !
   343)    ! the phenology, so we assign fully flushed leaves.                                     !
   344)    !---------------------------------------------------------------------------------------!
   345)    subroutine first_phenology(cgrid)
   346)       use stable_cohorts
   347)       use ed_state_vars , only : edtype           & ! structure
   348)                                , polygontype      & ! structure
   349)                                , sitetype         & ! structure
   350)                                , patchtype        ! ! structure
   351)       use ed_therm_lib  , only : calc_veg_hcap    ! ! function
   352)       use ed_max_dims   , only : n_pft            ! ! intent(in)
   353)       use allometry     , only : area_indices     ! ! subroutine
   354)       use grid_coms     , only : nzg              ! ! intent(in)
   355)       use therm_lib     , only : cmtl2uext        ! ! function
   356)       implicit none
   357)       !----- Arguments --------------------------------------------------------------------!
   358)       type(edtype)                   , target      :: cgrid       ! Current grid
   359)       !----- Local variables --------------------------------------------------------------!
   360)       type(polygontype)              , pointer     :: cpoly      ! Current polygon
   361)       type(sitetype)                 , pointer     :: csite      ! Current site
   362)       type(patchtype)                , pointer     :: cpatch     ! Current patch
   363)       integer                                      :: ipy        ! Polygon counter
   364)       integer                                      :: isi        ! Site counter
   365)       integer                                      :: ipa        ! Patch counter
   366)       integer                                      :: ico        ! Cohort counter
   367)       !------------------------------------------------------------------------------------!
   368) 
   369) 
   370) 
   371)       !------------------------------------------------------------------------------------!
   372)       !     Loop over all sites in this grid.                                              !
   373)       !------------------------------------------------------------------------------------!
   374)       polyloop: do ipy=1,cgrid%npolygons
   375)          nullify(cpoly)
   376)          cpoly => cgrid%polygon(ipy)
   377)          siteloop: do isi=1,cpoly%nsites
   378)             csite => cpoly%site(isi)
   379) 
   380) 
   381)             !------------------------------------------------------------------------------!
   382)             !     Loop over all patches and cohorts.                                       !
   383)             !------------------------------------------------------------------------------!
   384)             patchloop: do ipa=1,csite%npatches
   385)                cpatch => csite%patch(ipa)
   386)                cohortloop: do ico=1,cpatch%ncohorts
   387) 
   388)                   !------------------------------------------------------------------------!
   389)                   !    Find the initial guess for potential available water and elongation !
   390)                   ! factor, then compute the equilibrium biomass of active tissues and     !
   391)                   ! storage.                                                               !
   392)                   !------------------------------------------------------------------------!
   393)                   call pheninit_balive_bstorage(nzg,cpatch%pft(ico),cpatch%krdepth(ico)    &
   394)                                                ,cpatch%hite(ico),cpatch%dbh(ico)           &
   395)                                                ,csite%soil_water(:,ipa)                    &
   396)                                                ,cpoly%ntext_soil(:,isi)                    &
   397)                                                ,cpatch%paw_avg(ico),cpatch%elongf(ico)     &
   398)                                                ,cpatch%phenology_status(ico)               &
   399)                                                ,cpatch%bleaf(ico),cpatch%broot(ico)        &
   400)                                                ,cpatch%bsapwooda(ico)                      &
   401)                                                ,cpatch%bsapwoodb(ico)                      &
   402)                                                ,cpatch%balive(ico),cpatch%bstorage(ico))
   403)                   !------------------------------------------------------------------------!
   404) 
   405) 
   406)                   !----- Find LAI, WAI, and CAI. ------------------------------------------!
   407)                   call area_indices(cpatch, ico)
   408)                   !------------------------------------------------------------------------!
   409) 
   410) 
   411)                   !----- Find heat capacity and vegetation internal energy. ---------------!
   412)                   call calc_veg_hcap(cpatch%bleaf(ico),cpatch%bdead(ico)                   &
   413)                                     ,cpatch%bsapwooda(ico),cpatch%nplant(ico)              &
   414)                                     ,cpatch%pft(ico)                                       &
   415)                                     ,cpatch%leaf_hcap(ico),cpatch%wood_hcap(ico) )
   416)                   cpatch%leaf_energy(ico) = cmtl2uext(cpatch%leaf_hcap (ico)               &
   417)                                                      ,cpatch%leaf_water(ico)               &
   418)                                                      ,cpatch%leaf_temp (ico)               &
   419)                                                      ,cpatch%leaf_fliq (ico))
   420)                   cpatch%wood_energy(ico) = cmtl2uext(cpatch%wood_hcap (ico)               &
   421)                                                      ,cpatch%wood_water(ico)               &
   422)                                                      ,cpatch%wood_temp (ico)               &
   423)                                                      ,cpatch%wood_fliq (ico))
   424)                   call is_resolvable(csite,ipa,ico)
   425)                   !------------------------------------------------------------------------!
   426)                end do cohortloop
   427)                !---------------------------------------------------------------------------!
   428)             end do patchloop
   429)             !------------------------------------------------------------------------------!
   430)          end do siteloop
   431)          !---------------------------------------------------------------------------------!
   432)       end do polyloop
   433)       !------------------------------------------------------------------------------------!
   434) 
   435)       return
   436)    end subroutine first_phenology

   445)    !=======================================================================================!
   446)    !=======================================================================================!
   447)    !      This sub-routine will assign the initial potential available water and the       !
   448)    ! phenology that has been assigned, then find the biomass of active tissues and storage !
   449)    ! that is in equilibrium with the initial soil moisture.  This sub-routine should be    !
   450)    ! called whenever a new cohort is planted or recruited, or at the initial run (except   !
   451)    ! history).  The initial running average is simply the the instantaneous soil moisture  !
   452)    ! variable.  For plants other than the drought-deciduous, the potential available water !
   453)    ! is found but it doesn't control the phenology, so we assign the biomass that matches  !
   454)    ! the fully flushed leaves.                                                             !
   455)    !---------------------------------------------------------------------------------------!
   456)    subroutine pheninit_balive_bstorage(mzg,ipft,kroot,height,dbh,soil_water,ntext_soil     &
   457)                                       ,paw_avg,elongf,phenology_status                     &
   458)                                       ,bleaf,broot,bsapwooda,bsapwoodb,balive,bstorage)
   459)       use soil_coms     , only : soil                & ! intent(in), look-up table
   460)                                , slz                 & ! intent(in)
   461)                                , slzt                & ! intent(in)
   462)                                , dslz                ! ! intent(in)
   463)       use phenology_coms, only : spot_phen           & ! intent(in)
   464)                                , elongf_min          ! ! intent(in)
   465)       use pft_coms      , only : phenology           & ! intent(in)
   466)                                , q                   & ! intent(in)
   467)                                , qsw                 & ! intent(in)
   468)                                , agf_bs              ! ! intent(in)
   469)       use ed_max_dims   , only : n_pft               ! ! intent(in)
   470)       use allometry     , only : size2bl             & ! function
   471)                                , h2crownbh           ! ! function
   472)       implicit none
   473)       !----- Arguments --------------------------------------------------------------------!
   474)       integer                  , intent(in)  :: mzg               ! # of soil layers
   475)       integer                  , intent(in)  :: ipft              ! PFT type
   476)       integer                  , intent(in)  :: kroot             ! Level of rooting depth
   477)       real                     , intent(in)  :: height            ! Height
   478)       real                     , intent(in)  :: dbh               ! DBH
   479)       integer, dimension(mzg)  , intent(in)  :: ntext_soil        ! Soil texture
   480)       real   , dimension(mzg)  , intent(in)  :: soil_water        ! Soil water
   481)       real                     , intent(out) :: paw_avg           ! Pot. available water
   482)       real                     , intent(out) :: elongf            ! Elongation factor
   483)       integer                  , intent(out) :: phenology_status  ! phenology Flag
   484)       real                     , intent(out) :: bleaf             ! Leaf biomass
   485)       real                     , intent(out) :: broot             ! Root biomass
   486)       real                     , intent(out) :: bsapwooda         ! AG Sapwood biomass
   487)       real                     , intent(out) :: bsapwoodb         ! BG Sapwood biomass
   488)       real                     , intent(out) :: balive            ! Living tissue biomass
   489)       real                     , intent(out) :: bstorage          ! Storage biomass
   490)       !----- Local variables --------------------------------------------------------------!
   491)       integer                                :: k                 ! Layer counter
   492)       integer                                :: nsoil             ! Soil texture class
   493)       real                                   :: salloc            ! balive:bleaf ratio
   494)       real                                   :: salloci           ! bleaf:balive ratio
   495)       real                                   :: bleaf_max         ! maximum bleaf
   496)       real                                   :: balive_max        ! balive if on-allometry
   497)       real                                   :: psi_layer         ! Water pot. of this layer
   498)       real                                   :: mcheight          ! Mid-crown height
   499)       !------------------------------------------------------------------------------------!
   500) 
   501) 
   502) 
   503) 
   504)       !------------------------------------------------------------------------------------!
   505)       !     Here we decide how to compute the mean available water fraction.               !
   506)       !------------------------------------------------------------------------------------!
   507)       if (spot_phen) then
   508)          !----- Use soil potential to determine phenology. --------------------------------!
   509)          paw_avg = 0.0
   510)          do k=kroot,mzg
   511)             nsoil     = ntext_soil(k)
   512)             mcheight  = 0.5 * ( height + h2crownbh(height,ipft) )
   513) 
   514)             psi_layer = slzt(k) - mcheight                                                 &
   515)                       + soil(nsoil)%slpots                                                 &
   516)                       / (soil_water(k)      / soil(nsoil)%slmsts) ** soil(nsoil)%slbs
   517)             paw_avg   = paw_avg + max(0.0, (psi_layer - soil(nsoil)%slpotwp)) * dslz(k)               &
   518)                                 / (soil(nsoil)%slpotld  - soil(nsoil)%slpotwp)
   519)          end do
   520)          paw_avg = paw_avg / abs(slz(kroot))
   521)       else
   522)          !----- Use soil moisture (mass) to determine phenology. --------------------------!
   523)          paw_avg = 0.0
   524)          do k = kroot, mzg
   525)             nsoil   = ntext_soil(k)
   526)             paw_avg = paw_avg + max(0.0, (soil_water(k) - soil(nsoil)%soilwp)) * dslz(k)   &
   527)                               / (soil(nsoil)%soilld - soil(nsoil)%soilwp)
   528)          end do
   529)          paw_avg = paw_avg / abs(slz(kroot))
   530)       end if
   531) 
   532)       !------------------------------------------------------------------------------------!
   533)       !    Find the elongation factor according to the phenology of this PFT.              !
   534)       !------------------------------------------------------------------------------------!
   535)       select case (phenology(ipft))
   536)       case (1)
   537)          if (paw_avg < 1.0) then
   538)             elongf = 0.0
   539)          else
   540)             elongf = 1.0
   541)          end if
   542)       case (3,4)
   543)          elongf = max(0.0,min(1.0,paw_avg))
   544)       case default
   545)          elongf = 1.0
   546)       end select
   547)       !------------------------------------------------------------------------------------!
   548) 
   549)       !----- Set phenology status according to the elongation factor. ---------------------!
   550)       if (elongf >= 1.0) then
   551)          phenology_status = 0
   552)       elseif (elongf > elongf_min) then
   553)          phenology_status = 1
   554)       else
   555)          phenology_status = -2
   556)          elongf           = 0.
   557)       end if
   558)       !------------------------------------------------------------------------------------!
   559) 
   560) 
   561) 
   562)       !----- Compute the biomass of living tissues. ---------------------------------------!
   563)       salloc     = 1.0 + q(ipft) + qsw(ipft) * height
   564)       salloci    = 1.0 / salloc
   565)       bleaf_max  = size2bl(dbh,height,ipft)
   566)       balive_max = bleaf_max * salloc
   567)       bleaf      = bleaf_max * elongf
   568)       broot      = balive_max * q(ipft)   * salloci
   569)       bsapwooda  = balive_max * qsw(ipft) * height * salloci * agf_bs(ipft)
   570)       bsapwoodb  = balive_max * qsw(ipft) * height * salloci * (1.0 - agf_bs(ipft))
   571)       balive     = bleaf + broot + bsapwooda + bsapwoodb
   572)       !------------------------------------------------------------------------------------!
   573) 
   574) 
   575)       !------------------------------------------------------------------------------------!
   576)       !    Here we account for part of the carbon that didn't go to the leaves.  At this   !
   577)       ! point  we will be nice to the plants and leave all the carbon that didn't go to    !
   578)       ! leaves in the storage.  This gives some extra chance for the plant whilst it       !
   579)       ! conserves the total carbon.                                                        !
   580)       !------------------------------------------------------------------------------------!
   581)       bstorage = max(0.0, bleaf_max - bleaf)
   582)       !------------------------------------------------------------------------------------!
   583) 
   584)       return
   585)    end subroutine pheninit_balive_bstorage

   599)    real function daylength(lat,doy)
   600) 
   601)       use consts_coms , only : pio180 & ! intent(in)
   602)                              , twopi  ! ! intent(in)
   603)       implicit none
   604)       !----- Arguments --------------------------------------------------------------------!
   605)       real    , intent(in) :: lat
   606)       integer , intent(in) :: doy

   608)       real                 :: arg

   611)       arg = -tan(lat * pio180) * tan(-23.5 * pio180 * cos(twopi/365.0 * (float(doy)+9.0)))
