    31) subroutine prescribed_event(year,doy)
    32)   use ed_misc_coms, only: event_file
    33)   implicit none
    34) 
    35)   integer, intent(in) :: year
    36)   integer, intent(in) :: doy !! day of year
    37)   logical,save :: first = .true.
    38)   integer,save :: next_year = 100000
    39)   integer,save :: next_doy = 367
    40)   integer :: curr_year, curr_doy
    41)   integer(kind=4) :: i,j,nevent,nrep
    42)   integer(kind=4),dimension(20) ::pft
    43)   real(kind=8),dimension(20):: rval
    44)   logical(kind=4) :: texist = .false.
    45) 
    46) 
    47) 
    48)   if (event_file /= '') then
    49) 
    50)      if(first) then  !! is first pass, init vars
    51) 
    52)         call read_events_xml(trim(event_file),first)
    53) 
    54)         if(first) then !! if read was successful
    55) 
    56)            !! access list and find next event
    57)            call libxml2f90__ll_selectlist('EVENTS')
    58)            call libxml2f90__ll_selecttag('ACT','eventlist',1) !select upper level tag
    59)            call libxml2f90__ll_exist('DOWN','event',nevent)
    60)            print*,"NUMBER OF EVENTS FOUND", nevent
    61)            if(nevent .ge. 1) then
    62)               do i=1,nevent
    63)                  call libxml2f90__ll_selecttag('DOWN','event',i) !select event
    64)                  call getConfigINT   ('year','event',i,curr_year,texist)
    65)                  call getConfigINT   ('doy','event',i,curr_doy,texist)
    66)                  print*,curr_year,curr_doy,texist,next_year,next_doy
    67)                  !! see if the event bumps the closest event
    68)                  if((curr_year == year .and. curr_doy .ge. doy) .or. (curr_year .gt. year)) then
    69)                     if((curr_year .lt. next_year) .or. (curr_year == next_year .and. curr_doy .lt. next_doy)) then
    70)                        next_year = curr_year
    71)                        next_doy  = curr_doy
    72)                     end if
    73)                  end if
    74)                  call libxml2f90__ll_selecttag('UP','eventlist',1) !move back up to top level
    75)               enddo
    76)               print*,"FIRST EVENT",next_doy,next_year
    77)            end if
    78) 
    79)         endif
    80) 
    81)         first = .false.
    82)      end if
    83) 
    84)      if(next_year == year .and. next_doy == doy) then
    85) 
    86)         !! access list and find todays events
    87)         call libxml2f90__ll_selectlist('EVENTS')
    88)         call libxml2f90__ll_selecttag('ACT','eventlist',1) !select upper level tag
    89)         call libxml2f90__ll_exist('DOWN','event',nevent)
    90) 
    91) !        print*,"EVENT",nevent
    92)         findevent: do i=1,nevent !! Find Event in list
    93)            call libxml2f90__ll_selecttag('DOWN','event',i) !select event
    94)            call getConfigINT   ('year','event',i,curr_year,texist)
    95)            call getConfigINT   ('doy','event',i,curr_doy,texist)
    96)            !print*,curr_year,curr_doy,texist,next_year,next_doy
    97)            !! see if the event bumps the closest event
    98)            if((curr_year == year .and. curr_doy == doy)) then
    99)               print*,"EVENT",i,"of",nevent
   100) 
   101)               !!! PROCESS AGRICULTURAL EVENTS
   102) 
   103)               !!! HARVESTING
   104)               call libxml2f90__ll_exist('DOWN','harvest',nrep)
   105)               if(nrep .gt. 0) then
   106)                  print*,"HARVEST FESTIVAL"
   107)                  ! agb_frac
   108)                  ! bgb_frac
   109)                  do j = 1,nrep
   110)                     call libxml2f90__ll_selecttag('DOWN','harvest',j)
   111)                     !! Above-ground coarse biomass fraction removed
   112)                     call getConfigREAL  ('agb_frac','harvest',j,rval(1),texist)
   113)                     if(.not.texist) rval(1) = 1.0d+0
   114)                     !! Below-ground biomass fraction removed
   115)                     call getConfigREAL   ('bgb_frac','harvest',j,rval(2),texist)
   116)                     if(.not.texist) rval(2) = 1.0d+0
   117)                     !! Foliage fraction removed
   118)                     call getConfigREAL   ('fol_frac','harvest',j,rval(3),texist)
   119)                     if(.not.texist) rval(3) = 1.0d+0
   120)                     !! storage fraction removed
   121)                     call getConfigREAL   ('stor_frac','harvest',j,rval(3),texist)
   122)                     if(.not.texist) rval(4) = 1.0d+0
   123) 
   124)                     call event_harvest(rval(1),rval(2),rval(3),rval(4))
   125) 
   126)                     call libxml2f90__ll_selecttag('UP','event',i)
   127)                  end do
   128)               end if
   129) 
   130)               call libxml2f90__ll_exist('DOWN','fertilize',nrep)
   131)               if(nrep .gt. 0) then
   132)                  print*,"I say HABER, you say PROCESS"
   133)                  do j = 1,nrep
   134) 
   135)                     call libxml2f90__ll_selecttag('DOWN','fertilize',j)
   136)                     rval = 0.0d+0
   137)                     ! nitrogen
   138)                     call getConfigREAL  ('NH4','fertilize',j,rval(1),texist)
   139)                     if(.not.texist) rval(1) = 0.0d+0
   140)                     call getConfigREAL  ('N03','fertilize',j,rval(1),texist)
   141)                     if(.not.texist) rval(2) = 0.0d+0
   142)                     ! Phosphorus
   143)                     call getConfigREAL  ('P','fertilize',j,rval(1),texist)
   144)                     if(.not.texist) rval(3) = 0.0d+0
   145)                     ! Potassium
   146)                     call getConfigREAL  ('K','fertilize',j,rval(1),texist)
   147)                     if(.not.texist) rval(4) = 0.0d+0
   148)                     ! Calcium
   149)                     call getConfigREAL  ('Ca','fertilize',j,rval(1),texist)
   150)                     if(.not.texist) rval(5) = 0.0d+0
   151) 
   152)                     call event_fertilize(rval(1:5))
   153) 
   154)                     call libxml2f90__ll_selecttag('UP','event',i)
   155)                  end do
   156)               endif
   157) 
   158)               call libxml2f90__ll_exist('DOWN','till',nrep)
   159)               if(nrep .gt. 0) then
   160)                  print*,"TILL"
   161)                  do j = 1,nrep
   162)                     ! depth (m)
   163)                     call libxml2f90__ll_selecttag('DOWN','till',j)
   164) 
   165)                     call getConfigREAL  ('depth','till',j,rval(1),texist)
   166)                     if(.not.texist) rval(1) = 0.15d+0 !assume a default of 15cm
   167) 
   168)                     call event_till(rval(1))
   169) 
   170)                     call libxml2f90__ll_selecttag('UP','event',i)
   171) 
   172)                  enddo
   173)               end if
   174) 
   175)               call libxml2f90__ll_exist('DOWN','irrigate',nrep)
   176)               if(nrep .gt. 0) then
   177)                  print*,"IRRIGATE"
   178)                  do j = 1,nrep
   179) 
   180)                     call getConfigREAL ('irrigate','event',j,rval(1),texist)
   181)                     if(.not.texist) rval(1) = 0.0d+0
   182) 
   183)                     call event_irrigate(rval(1))
   184) 
   185)                  enddo
   186)               endif
   187) 
   188)               call libxml2f90__ll_exist('DOWN','pesticide',nrep)
   189)               if(nrep .gt. 0) print*,"PESTICIDE"
   190) 
   191)               call libxml2f90__ll_exist('DOWN','herbicide',nrep)
   192)               if(nrep .gt. 0) print*,"HERBICIDE"
   193)               !pft !! spp removed
   194) 
   195)               call libxml2f90__ll_exist('DOWN','grazer',nrep)
   196)               if(nrep .gt. 0) print*,"grazer"
   197)               !pft !! spp removed
   198)               !fol_frac
   199) 
   200) 
   201) 
   202)               !!! PROCESS FORESTRY MANAGEMENT
   203)               call libxml2f90__ll_exist('DOWN','thin',nrep)
   204)               if(nrep .gt. 0) print*,"There's too many trees around here"
   205)               !pft, default = all
   206) 
   207)               !!! PROCESS NATURAL DISTURBANCES
   208)               call libxml2f90__ll_exist('DOWN','wind',nrep)
   209)               if(nrep .gt. 0) print*,"If the wind blows, your gone"
   210)               !fol_frac
   211)               !agb_frac
   212) 
   213)               call libxml2f90__ll_exist('DOWN','fire',nrep)
   214)               if(nrep .gt. 0) then
   215)                  print*,"FIRE, FIRE, FIRE"
   216)                  do j = 1,nrep
   217)                     call libxml2f90__ll_selecttag('DOWN','fire',j)
   218) 
   219)                     call event_fire()
   220) 
   221)                     call libxml2f90__ll_selecttag('UP','event',i)
   222)                  end do
   223)               end if
   224) 
   225)               call libxml2f90__ll_exist('DOWN','pathogen',nrep)
   226)               if(nrep .gt. 0) print*,"How can you not love the Beatles"
   227) 
   228)               call libxml2f90__ll_exist('DOWN','invasive',nrep)
   229)               if(nrep .gt. 0) print*,"introduced species"
   230) 
   231)               !! PLANTING (assumed to occur AFTER the other events)
   232)               call libxml2f90__ll_exist('DOWN','plant',nrep)
   233)               if(nrep .gt. 0) then
   234)                  do j = 1,nrep
   235)                     call libxml2f90__ll_selecttag('DOWN','plant',j)
   236) 
   237)                     call getConfigINT   ('pft','plant',j,pft(1),texist)
   238)                     if(.not.texist) then
   239)                        PRINT*,"Planting prescribed but PFT not specified"
   240)                        stop
   241)                     endif
   242)                     call getConfigREAL   ('density','plant',j,rval(1),texist)
   243)                     if(.not.texist) rval(1) = 1.0d+0
   244)                     call event_planting(pft(1),rval(1))
   245) 
   246)                     call libxml2f90__ll_selecttag('UP','event',i)
   247)                  end do
   248)               end if
   249) 
   250) 
   251) 
   252)               call libxml2f90__ll_selecttag('UP','eventlist',1) !move back up to top level
   253)               exit findevent
   254)            endif
   255)            call libxml2f90__ll_selecttag('UP','eventlist',1) !move back up to top level
   256)         enddo findevent
   257) 
   258) 
   259)         !! find next event
   260)         next_year = 100000000
   261)         next_doy  = 365
   262)         do i=1,nevent
   263)            call libxml2f90__ll_selecttag('DOWN','event',i) !select event
   264)            call getConfigINT   ('year','event',i,curr_year,texist)
   265)            call getConfigINT   ('doy','event',i,curr_doy,texist)
   266) 
   267)            !! see if the event bumps the closest event
   268)            if((curr_year == year .and. curr_doy .gt. doy) .or. (curr_year .gt. year)) then
   269)               if((curr_year .lt. next_year) .or. (curr_year == next_year .and. curr_doy .lt. next_doy)) then
   270)                  next_year = curr_year
   271)                  next_doy  = curr_doy
   272)               end if
   273)            end if
   274)            call libxml2f90__ll_selecttag('UP','eventlist',1) !move back up to top level
   275)         enddo
   276)         print*,"NEXT EVENT",next_doy,next_year
   277) 
   278) 
   279)      endif
   280)   end if
   281) 
   282) end subroutine prescribed_event

   285) subroutine event_harvest(agb_frac8,bgb_frac8,fol_frac8,stor_frac8)
   286)   use stable_cohorts
   287)   use update_derived_props_module
   288)   use grid_coms, only : ngrids
   289)   use ed_state_vars,only: edgrid_g, &
   290)        edtype,polygontype,sitetype, &
   291)        patchtype,allocate_patchtype,copy_patchtype,deallocate_patchtype
   292)   use pft_coms, only: qsw,q,hgt_min, agf_bs, is_grass
   293)   use ed_therm_lib, only: calc_veg_hcap,update_veg_energy_cweh
   294)   use fuse_fiss_utils, only: terminate_cohorts
   295)   use allometry, only : bd2dbh, dbh2h, bl2dbh, bl2h, h2dbh, area_indices, ed_biomass
   296)   use consts_coms, only : pio4
   297)   use ed_misc_coms     , only : igrass               ! ! intent(in)
   298)   use budget_utils     , only : update_budget
   299)   implicit none
   300)   real(kind=8),intent(in) :: agb_frac8
   301)   real(kind=8),intent(in) :: bgb_frac8
   302)   real(kind=8),intent(in) :: fol_frac8
   303)   real(kind=8),intent(in) :: stor_frac8
   304)   real :: ialloc,bdead_new,bswa_new,bswb_new,bleaf_new,bfr_new,bstore_new
   305)   real :: agb_frac,bgb_frac,fol_frac,stor_frac
   306)   real :: old_leaf_hcap
   307)   real :: old_wood_hcap
   308)   real :: elim_nplant
   309)   real :: elim_lai
   310)   integer :: ifm,ipy,isi,ipa,ico,pft
   311)   type(edtype), pointer :: cgrid
   312)   type(polygontype), pointer :: cpoly
   313)   type(sitetype),pointer :: csite
   314)   type(patchtype),pointer :: cpatch
   315) 
   316)   agb_frac  = real(agb_frac8)
   317)   bgb_frac  = real(bgb_frac8)
   318)   fol_frac  = real(fol_frac8)
   319)   stor_frac = real(stor_frac8)
   320) 
   321)   print*,"----------------------------"
   322)   print*,"----------------------------"
   323)   print*,"----------------------------"
   324)   print*,"----------------------------"
   325)   print*,"         HARVESTING "
   326)   print*,"----------------------------"
   327)   print*,"----------------------------"
   328)   print*,"----------------------------"
   329)   print*,"----------------------------"
   330)   print*,""
   331)   print*,agb_frac,bgb_frac,fol_frac
   332) 
   333)   do ifm = 1,ngrids
   334)      cgrid => edgrid_g(ifm)
   335)      do ipy = 1,cgrid%npolygons
   336) 
   337)         cpoly => cgrid%polygon(ipy)
   338) 
   339)         do isi = 1,cpoly%nsites
   340) 
   341)            csite => cpoly%site(isi)
   342) 
   343)            do ipa=1,csite%npatches
   344) 
   345)               cpatch => csite%patch(ipa)
   346) 
   347)               if(cpatch%ncohorts > 0) then
   348) 
   349)               do ico=1,cpatch%ncohorts
   350) 
   351)                  pft = cpatch%pft(ico)
   352)                  !! calc new pool sizes
   353) 
   354)                  ialloc     =  1.0 / (1.0 + q(pft) + qsw(pft) * cpatch%hite(ico))
   355)                  bdead_new  = cpatch%bdead(ico) *                                        &
   356)                               (1.0-agb_frac * agf_bs(pft) - bgb_frac*(1.0-agf_bs(pft)))
   357)                  bswa_new   = cpatch%balive(ico) * qsw(pft) * cpatch%hite(ico)           &
   358)                               * agf_bs(pft) * ialloc * (1.0-agb_frac*agf_bs(pft)         &
   359)                               - bgb_frac*(1.0-agf_bs(pft)))
   360)                  bswb_new   = cpatch%balive(ico) * qsw(pft) * cpatch%hite(ico)           &
   361)                               * (1.-agf_bs(pft)) * ialloc                                &
   362)                               * (1.0-agb_frac*agf_bs(pft) - bgb_frac*(1.0-agf_bs(pft)))
   363) 
   364)                  bstore_new = cpatch%bstorage(ico) * (1.0-stor_frac)
   365)                  bleaf_new  = cpatch%balive(ico)   * ialloc *(1.0-fol_frac)
   366)                  bfr_new    = cpatch%balive(ico)   * q(pft) * ialloc * (1.0-bgb_frac)
   367) 
   368)                  !! move residual frac to debris/litter pools
   369)                    !! For now assume 100% removal [[needs to be updated]]
   370) 
   371) 
   372)                  !! update biomass pools
   373)                  !! [[this needs to be more sophisticated]]
   374) 
   375)                  cpatch%balive(ico)    = max(0.0,bleaf_new + bfr_new + bswa_new + bswb_new)
   376)                  cpatch%broot(ico)     = max(0.0,bfr_new)
   377)                  cpatch%bsapwooda(ico) = max(0.0,bswa_new)
   378)                  cpatch%bsapwoodb(ico) = max(0.0,bswb_new)
   379)                  cpatch%bdead(ico)     = max(0.0,bdead_new)
   380)                  cpatch%bstorage(ico)  = max(0.0,bstore_new)
   381) 
   382)                  if(bleaf_new .le. tiny(1.0)) then
   383)                     cpatch%phenology_status(ico) = -2
   384)                     cpatch%elongf(ico)           = 0.0
   385)                     !----- No leaves, then set it to zero. --------------------------------!
   386)                     cpatch%bleaf(ico)      = 0.0
   387)                  else
   388)                     cpatch%phenology_status(ico) = 1
   389)                     cpatch%bleaf(ico)   = bleaf_new
   390)                  end if
   391) 
   392)                  if(cpatch%bdead(ico) .gt. tiny(1.0)) then
   393)                     if(is_grass(cpatch%pft(ico)) .and. igrass==1) then
   394)                        cpatch%hite(ico) = max( hgt_min(pft), bl2h(cpatch%bleaf(ico),pft))
   395)                        cpatch%dbh (ico) = h2dbh(cpatch%hite(ico),pft)
   396)                     else
   397)                        cpatch%dbh (ico) = bd2dbh(cpatch%pft(ico), cpatch%bdead(ico))
   398)                        cpatch%hite(ico) = dbh2h (cpatch%pft(ico), cpatch%dbh(ico))
   399)                     end if
   400)                  else
   401)                     cpatch%dbh(ico)  = 0.0
   402)                     cpatch%hite(ico) = 0.0
   403)                  end if
   404) 
   405)                  !----- Update LAI, WAI, and CAI ------------------------------------------!
   406)                  call area_indices(cpatch, ico)
   407) 
   408)                  !----- Update basal area and above-ground biomass. -----------------------!
   409)                  cpatch%basarea(ico) = pio4 * cpatch%dbh(ico) * cpatch%dbh(ico)
   410)                  cpatch%agb(ico)     = ed_biomass(cpatch, ico)
   411)                  !-------------------------------------------------------------------------!
   412)                  !    Here we are leaving all water in the branches and twigs... Do not    !
   413)                  ! worry, if there is any, it will go down through shedding the next       !
   414)                  ! step.                                                                   !
   415)                  !-------------------------------------------------------------------------!
   416)                  old_leaf_hcap = cpatch%leaf_hcap(ico)
   417)                  old_wood_hcap = cpatch%wood_hcap(ico)
   418)                  call calc_veg_hcap(cpatch%bleaf(ico),cpatch%bdead(ico)                    &
   419)                                    ,cpatch%bsapwooda(ico),cpatch%nplant(ico)               &
   420)                                    ,cpatch%pft(ico)                                        &
   421)                                    ,cpatch%leaf_hcap(ico),cpatch%wood_hcap(ico))
   422)                  call update_veg_energy_cweh(csite,ipa,ico,old_leaf_hcap,old_wood_hcap)
   423) 
   424)                  !----- Update flags telling whether leaves and branches can be solved. ---!
   425)                  call is_resolvable(csite,ipa,ico)
   426) 
   427)               enddo
   428) 
   429)               !! remove small cohorts
   430)               call terminate_cohorts(csite,ipa,elim_nplant,elim_lai)
   431) 
   432)               call update_patch_derived_props(csite,ipa)
   433)               call update_budget(csite, cpoly%lsl(isi),ipa)
   434)            end if  !! check to make sure there ARE cohorts
   435) 
   436)            enddo
   437)            ! Update site properties. ## THINK ABOUT WHAT TO SET FLAG##########
   438)            call update_site_derived_props(cpoly,0,isi)
   439)         end do
   440)      end do
   441) 
   442)   end do
   443) 
   444)   print*,"done HARVEST"
   445) 
   446) end subroutine event_harvest

   449) subroutine event_planting(pft,density8)
   450)   use rk4_integ_utils
   451)   use update_derived_props_module
   452)   use grid_coms, only : ngrids,nzg,nzs
   453)   use ed_state_vars,only: edgrid_g, &
   454)        edtype,polygontype,sitetype, &
   455)        patchtype,allocate_patchtype,copy_patchtype,deallocate_patchtype, &
   456)        filltab_alltypes
   457)   use disturbance_utils,only: plant_patch
   458)   use budget_utils     , only : update_budget
   459)   implicit none
   460)   integer(kind=4),intent(in) :: pft
   461)   real(kind=8),intent(in) :: density8
   462)   real :: density
   463)   integer :: ifm,ipy,isi,ipa
   464)   type(edtype), pointer :: cgrid
   465)   type(polygontype), pointer :: cpoly
   466)   type(sitetype),pointer :: csite
   467)   real, parameter :: planting_ht = 1.0 !multiple of hgt_min
   468) 
   469)   density = real(density8)
   470) 
   471)   print*,"----------------------------"
   472)   print*,"----------------------------"
   473)   print*,"----------------------------"
   474)   print*,"----------------------------"
   475)   print*,"         PLANTING "
   476)   print*,"----------------------------"
   477)   print*,"----------------------------"
   478)   print*,"----------------------------"
   479)   print*,"----------------------------"
   480)   print*,""
   481)   print*,pft,density
   482) 
   483)   do ifm = 1,ngrids
   484)      cgrid => edgrid_g(ifm)
   485)      do ipy = 1,cgrid%npolygons
   486) 
   487)         cpoly => cgrid%polygon(ipy)
   488) 
   489)         do isi = 1,cpoly%nsites
   490) 
   491)            csite => cpoly%site(isi)
   492) 
   493)            do ipa=1,csite%npatches
   494)               call update_patch_thermo_props(csite,ipa,ipa,nzg,nzs,cpoly%ntext_soil(:,isi))
   495)               call plant_patch(csite,ipa,nzg,pft,density,cpoly%ntext_soil(:,isi) &
   496)                               ,planting_ht,cpoly%lsl(isi))
