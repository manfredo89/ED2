     1) module heun_driver
     2)   contains
     3) 
     4) !==========================================================================================!
     5) !==========================================================================================!
     6) !     This subroutine is the main driver for the Heun's (RK2) integration scheme.          !
     7) !------------------------------------------------------------------------------------------!
     8) subroutine heun_timestep(cgrid)
     9)    use rk4_integ_utils
    10)    use soil_respiration_module
    11)    use photosyn_driv
    12)    use rk4_misc
    13)    use update_derived_props_module
    14)    use rk4_coms              , only : integration_vars   & ! structure
    15)                                     , rk4patchtype       & ! structure
    16)                                     , zero_rk4_patch     & ! subroutine
    17)                                     , zero_rk4_cohort    & ! subroutine
    18)                                     , integration_buff   & ! intent(out)
    19)                                     , rk4site            ! ! intent(out)
    20)    use ed_state_vars         , only : edtype             & ! structure
    21)                                     , polygontype        & ! structure
    22)                                     , sitetype           & ! structure
    23)                                     , patchtype          ! ! structure
    24)    use met_driver_coms       , only : met_driv_state     ! ! structure
    25)    use grid_coms             , only : nzg                & ! intent(in)
    26)                                     , nzs                ! ! intent(in)
    27)    use ed_misc_coms          , only : current_time       & ! intent(in)
    28)                                     , dtlsm              ! ! intent(in)
    29)    use ed_max_dims           , only : n_dbh              ! ! intent(in)
    30)    use soil_coms             , only : soil_rough         & ! intent(in)
    31)                                     , snow_rough         ! ! intent(in)
    32)    use therm_lib             , only : tq2enthalpy        ! ! function
    33)    use budget_utils          , only : update_budget      & ! function
    34)                                     , compute_budget     ! ! function
    35)    ! OMP use omp_lib
    36) 
    37)    implicit none
    38)    !----- Arguments -----------------------------------------------------------------------!
    39)    type(edtype)             , target      :: cgrid
    40)    !----- Local variables -----------------------------------------------------------------!
    41)    type(polygontype)        , pointer     :: cpoly
    42)    type(sitetype)           , pointer     :: csite
    43)    type(patchtype)          , pointer     :: cpatch
    44)    type(met_driv_state)     , pointer     :: cmet
    45)    integer                                :: ipy
    46)    integer                                :: isi
    47)    integer                                :: ipa
    48)    integer                                :: imon
    49)    integer                                :: nsteps
    50)    real                                   :: wcurr_loss2atm
    51)    real                                   :: ecurr_netrad
    52)    real                                   :: ecurr_loss2atm
    53)    real                                   :: co2curr_loss2atm
    54)    real                                   :: wcurr_loss2drainage
    55)    real                                   :: ecurr_loss2drainage
    56)    real                                   :: wcurr_loss2runoff
    57)    real                                   :: ecurr_loss2runoff
    58)    real                                   :: old_can_enthalpy
    59)    real                                   :: old_can_shv
    60)    real                                   :: old_can_co2
    61)    real                                   :: old_can_rhos
    62)    real                                   :: old_can_temp
    63)    real                                   :: old_can_prss
    64)    real                                   :: patch_vels
    65)    integer                                :: ibuff
    66)    !----- Local constants. ----------------------------------------------------------------!
    67)    logical                   , parameter  :: test_energy_sanity = .false.
    68)    !---------------------------------------------------------------------------------------!
    69) 
    70)    ibuff = 1
    71) 
    72)    polyloop: do ipy = 1,cgrid%npolygons
    73)       cpoly => cgrid%polygon(ipy)
    74) 
    75)       siteloop: do isi = 1,cpoly%nsites
    76)          csite => cpoly%site(isi)
    77)          cmet  => cpoly%met(isi)
    78) 
    79)          !---------------------------------------------------------------------------------!
    80)          !     Update the monthly rainfall.                                                !
    81)          !---------------------------------------------------------------------------------!
    82)          imon                             = current_time%month
    83)          cpoly%avg_monthly_pcpg(imon,isi) = cpoly%avg_monthly_pcpg(imon,isi)               &
    84)                                           + cmet%pcpg * dtlsm
    85)          !---------------------------------------------------------------------------------!
    86) 
    87)          !---------------------------------------------------------------------------------!
    88)          !    Copy the meteorological variables to the rk4site structure.                  !
    89)          !---------------------------------------------------------------------------------!
    90) 
    91)          call copy_met_2_rk4site(nzg,cmet%atm_ustar,cmet%atm_theiv,cmet%atm_vpdef          &
    92)                                 ,cmet%atm_theta,cmet%atm_tmp,cmet%atm_shv,cmet%atm_co2     &
    93)                                 ,cmet%geoht,cmet%exner,cmet%pcpg,cmet%qpcpg,cmet%dpcpg     &
    94)                                 ,cmet%prss,cmet%rshort,cmet%rlong,cmet%par_beam            &
    95)                                 ,cmet%par_diffuse,cmet%nir_beam,cmet%nir_diffuse           &
    96)                                 ,cmet%geoht,cpoly%lsl(isi),cpoly%ntext_soil(:,isi)         &
    97)                                 ,cpoly%green_leaf_factor(:,isi),cgrid%lon(ipy)             &
    98)                                 ,cgrid%lat(ipy),cgrid%cosz(ipy))
    99)          !---------------------------------------------------------------------------------!
   100) 
   101) 
   102)          patchloop: do ipa = 1,csite%npatches
   103)             cpatch => csite%patch(ipa)
   104) 
   105) 
   106)             !----- Reset all buffers to zero, as a safety measure. ------------------------!
   107)             call zero_rk4_patch(integration_buff(ibuff)%initp)
   108)             call zero_rk4_patch(integration_buff(ibuff)%ytemp)
   109)             call zero_rk4_patch(integration_buff(ibuff)%yerr)
   110)             call zero_rk4_patch(integration_buff(ibuff)%yscal)
   111)             call zero_rk4_patch(integration_buff(ibuff)%dydx)
   112)             call zero_rk4_cohort(integration_buff(ibuff)%initp)
   113)             call zero_rk4_cohort(integration_buff(ibuff)%ytemp)
   114)             call zero_rk4_cohort(integration_buff(ibuff)%yerr)
   115)             call zero_rk4_cohort(integration_buff(ibuff)%yscal)
   116)             call zero_rk4_cohort(integration_buff(ibuff)%dydx)
   117) 
   118)             !----- Get velocity for aerodynamic resistance. -------------------------------!
   119)             if (csite%can_theta(ipa) < cmet%atm_theta) then
   120) !               cmet%vels = cmet%vels_stab
   121)                patch_vels = cmet%vels_stab
   122)             else
   123) !               cmet%vels = cmet%vels_unstab
   124)                patch_vels = cmet%vels_stab
   125)             end if
   126)             !------------------------------------------------------------------------------!
   127) 
   128) 
   129) 
   130)             !------------------------------------------------------------------------------!
   131)             !    Update roughness and canopy depth.                                        !
   132)             !------------------------------------------------------------------------------!
   133)             call update_patch_thermo_props(csite,ipa,ipa,nzg,nzs,cpoly%ntext_soil(:,isi))
   134)             call update_patch_derived_props(csite,ipa)
   135)             !------------------------------------------------------------------------------!
   136) 
   137) 
   138) 
   139)             !----- Save the previous thermodynamic state. ---------------------------------!
   140)             old_can_shv      = csite%can_shv  (ipa)
   141)             old_can_co2      = csite%can_co2  (ipa)
   142)             old_can_rhos     = csite%can_rhos (ipa)
   143)             old_can_temp     = csite%can_temp (ipa)
   144)             old_can_prss     = csite%can_prss (ipa)
   145)             old_can_enthalpy = tq2enthalpy(csite%can_temp(ipa),csite%can_shv(ipa),.true.)
   146)             !------------------------------------------------------------------------------!
   147) 
   148)             !----- Compute current storage terms. -----------------------------------------!
   149)             call update_budget(csite,cpoly%lsl(isi),ipa)
   150)             !------------------------------------------------------------------------------!
   151)             !------------------------------------------------------------------------------!
   152)             !      Test whether temperature and energy are reasonable.                     !
   153)             !------------------------------------------------------------------------------!
   154)             if (test_energy_sanity) then
   155)                call sanity_check_veg_energy(csite,ipa)
   156)             end if
   157)             !------------------------------------------------------------------------------!
   158) 
   159) 
   160) 
   161)             !------------------------------------------------------------------------------!
   162)             !     Set up the integration patch.                                            !
   163)             !------------------------------------------------------------------------------!
   164)             call copy_patch_init(csite,ipa,integration_buff(ibuff)%initp,patch_vels)
   165)             !------------------------------------------------------------------------------!
   166) 
   167)             !----- Get photosynthesis, stomatal conductance, and transpiration. -----------!
   168)             call canopy_photosynthesis(csite,cmet,nzg,ipa                                  &
   169)                                       ,cpoly%ntext_soil(:,isi)                             &
   170)                                       ,cpoly%leaf_aging_factor(:,isi)                      &
   171)                                       ,cpoly%green_leaf_factor(:,isi))
   172)             !------------------------------------------------------------------------------!
   173) 
   174) 
   175) 
   176)             !----- Compute root and heterotrophic respiration. ----------------------------!
   177)             call soil_respiration(csite,ipa,nzg,cpoly%ntext_soil(:,isi))
   178)             !------------------------------------------------------------------------------!
   179) 
   180) 
   181) 
   182)             !------------------------------------------------------------------------------!
   183)             !     Set up the remaining, carbon-dependent variables to the buffer.          !
   184)             !------------------------------------------------------------------------------!
   185)             call copy_patch_init_carbon(csite,ipa,integration_buff(ibuff)%initp)
   186)             !------------------------------------------------------------------------------!
   187) 
   188) 
   189)             !------------------------------------------------------------------------------!
   190)             !     This is the step in which the derivatives are computed, we use a         !
   191)             ! structure that is very similar to the Runge-Kutta, though a simpler one.     !
   192)             !------------------------------------------------------------------------------!
   193)             call integrate_patch_heun(csite,ipa,isi,cpoly%nighttime(isi),wcurr_loss2atm    &
   194)                                      ,ecurr_netrad,ecurr_loss2atm,co2curr_loss2atm         &
   195)                                      ,wcurr_loss2drainage,ecurr_loss2drainage              &
   196)                                      ,wcurr_loss2runoff,ecurr_loss2runoff,nsteps)
   197)             !------------------------------------------------------------------------------!
   198) 
   199) 
   200) 
   201)             !----- Add the number of steps into the step counter. -------------------------!
   202)             cgrid%workload(13,ipy) = cgrid%workload(13,ipy) + real(nsteps)
   203)             !------------------------------------------------------------------------------!
   204) 
   205) 
   206) 
   207)             !------------------------------------------------------------------------------!
   208)             !    Update the minimum monthly temperature, based on canopy temperature.      !
   209)             !------------------------------------------------------------------------------!
   210)             if (cpoly%site(isi)%can_temp(ipa) < cpoly%min_monthly_temp(isi)) then
   211)                cpoly%min_monthly_temp(isi) = cpoly%site(isi)%can_temp(ipa)
   212)             end if
   213)             !------------------------------------------------------------------------------!
   214) 
   215) 
   216) 
   217) 
   218)             !------------------------------------------------------------------------------!
   219)             !     Compute the residuals.                                                   !
   220)             !------------------------------------------------------------------------------!
   221)             call compute_budget(csite,cpoly%lsl(isi),cmet%pcpg,cmet%qpcpg,ipa              &
   222)                                ,wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm                 &
   223)                                ,co2curr_loss2atm,wcurr_loss2drainage,ecurr_loss2drainage   &
   224)                                ,wcurr_loss2runoff,ecurr_loss2runoff,cpoly%area(isi)        &
   225)                                ,cgrid%cbudget_nep(ipy),old_can_enthalpy,old_can_shv        &
   226)                                ,old_can_co2,old_can_rhos,old_can_prss)
   227)             !------------------------------------------------------------------------------!
   228)          end do patchloop
   229)          !---------------------------------------------------------------------------------!
   230)       end do siteloop
   231)       !------------------------------------------------------------------------------------!
   232)    end do polyloop
   233)    !---------------------------------------------------------------------------------------!
   234) 
   235)    return
   236) end subroutine heun_timestep

   245) !==========================================================================================!
   246) !==========================================================================================!
   247) !     This subroutine will drive the integration process using the Heun method.  Notice    !
   248) ! that most of the Heun method utilises the subroutines from Runge-Kutta.                  !
   249) !------------------------------------------------------------------------------------------!
   250) subroutine integrate_patch_heun(csite,ipa,isi,nighttime,wcurr_loss2atm,ecurr_netrad        &
   251)                                ,ecurr_loss2atm,co2curr_loss2atm,wcurr_loss2drainage        &
   252)                                ,ecurr_loss2drainage,wcurr_loss2runoff,ecurr_loss2runoff    &
   253)                                ,nsteps)
   254)    use ed_state_vars   , only : sitetype             & ! structure
   255)                               , patchtype            ! ! structure
   256)    use ed_misc_coms    , only : dtlsm                ! ! intent(in)
   257)    use soil_coms       , only : soil_rough           & ! intent(in)
   258)                               , snow_rough           ! ! intent(in)
   259)    use canopy_air_coms , only : exar8                ! ! intent(in)
   260)    use rk4_coms        , only : integration_vars     & ! structure
   261)                               , integration_buff     & ! structure
   262)                               , rk4site              & ! intent(inout)
   263)                               , zero_rk4_patch       & ! subroutine
   264)                               , zero_rk4_cohort      & ! subroutine
   265)                               , tbeg                 & ! intent(inout)
   266)                               , tend                 & ! intent(inout)
   267)                               , dtrk4                & ! intent(inout)
   268)                               , dtrk4i               ! ! intent(inout)
   269)    use rk4_driver      , only : initp2modelp         ! ! subroutine
   270)    ! OMP use omp_lib
   271) 
   272)    implicit none
   273)    !----- Arguments -----------------------------------------------------------------------!
   274)    type(sitetype)        , target      :: csite
   275)    integer               , intent(in)  :: ipa
   276)    integer               , intent(in)  :: isi
   277)    logical               , intent(in)  :: nighttime
   278)    real                  , intent(out) :: wcurr_loss2atm
   279)    real                  , intent(out) :: ecurr_netrad
   280)    real                  , intent(out) :: ecurr_loss2atm
   281)    real                  , intent(out) :: co2curr_loss2atm
   282)    real                  , intent(out) :: wcurr_loss2drainage
   283)    real                  , intent(out) :: ecurr_loss2drainage
   284)    real                  , intent(out) :: wcurr_loss2runoff
   285)    real                  , intent(out) :: ecurr_loss2runoff
   286)    integer               , intent(out) :: nsteps
   287)    integer                             :: ibuff
   288)    !----- Local variables -----------------------------------------------------------------!
   289)    real(kind=8)                        :: hbeg
   290)    !---------------------------------------------------------------------------------------!
   291) 
   292)    ibuff = 1
   293) 
   294)    !---------------------------------------------------------------------------------------!
   295)    !      Initial step size.  Experience has shown that giving this too large a value      !
   296)    ! causes the integrator to fail (e.g., soil layers become supersaturated).              !
   297)    !---------------------------------------------------------------------------------------!
   298)    hbeg = dble(csite%htry(ipa))
   299) 
   300)    !---------------------------------------------------------------------------------------!
   301)    !     Zero the canopy-atmosphere flux values.  These values are updated every dtlsm,    !
   302)    ! so they must be zeroed at each call.                                                  !
   303)    !---------------------------------------------------------------------------------------!
   304)    integration_buff(ibuff)%initp%upwp = 0.d0
   305)    integration_buff(ibuff)%initp%tpwp = 0.d0
   306)    integration_buff(ibuff)%initp%qpwp = 0.d0
   307)    integration_buff(ibuff)%initp%cpwp = 0.d0
   308)    integration_buff(ibuff)%initp%wpwp = 0.d0
   309) 
   310)    !----- Go into the ODE integrator using Euler. -----------------------------------------!
   311)    call heun_integ(hbeg,csite,ipa,isi,nsteps)
   312) 
   313)    !---------------------------------------------------------------------------------------!
   314)    !      Normalize canopy-atmosphere flux values.  These values are updated every         !
   315)    ! dtlsm, so they must be normalized every time.                                         !
   316)    !---------------------------------------------------------------------------------------!
   317)    integration_buff(ibuff)%initp%upwp = integration_buff(ibuff)%initp%can_rhos             &
   318)                                       * integration_buff(ibuff)%initp%upwp * dtrk4i
   319)    integration_buff(ibuff)%initp%tpwp = integration_buff(ibuff)%initp%can_rhos             &
   320)                                       * integration_buff(ibuff)%initp%tpwp * dtrk4i
   321)    integration_buff(ibuff)%initp%qpwp = integration_buff(ibuff)%initp%can_rhos             &
   322)                                       * integration_buff(ibuff)%initp%qpwp * dtrk4i
   323)    integration_buff(ibuff)%initp%cpwp = integration_buff(ibuff)%initp%can_rhos             &
   324)                                       * integration_buff(ibuff)%initp%cpwp * dtrk4i
   325)    integration_buff(ibuff)%initp%wpwp = integration_buff(ibuff)%initp%can_rhos             &
   326)                                       * integration_buff(ibuff)%initp%wpwp * dtrk4i
   327) 
   328)    !---------------------------------------------------------------------------------------!
   329)    ! Move the state variables from the integrated patch to the model patch.                !
   330)    !---------------------------------------------------------------------------------------!
   331)    call initp2modelp(tend-tbeg,integration_buff(ibuff)%initp,csite,ipa,nighttime           &
   332)                     ,wcurr_loss2atm,ecurr_netrad,ecurr_loss2atm,co2curr_loss2atm           &
   333)                     ,wcurr_loss2drainage,ecurr_loss2drainage,wcurr_loss2runoff             &
   334)                     ,ecurr_loss2runoff)
   335) 
   336)    return
   337) end subroutine integrate_patch_heun

   348) ! Subroutine heun_integ                                                                    !
   349) !                                                                                          !
   350) !     This subroutine will drive the integration of several ODEs that drive the fast-scale !
   351) ! state variables using the Heun's method (a 2nd order Runge-Kutta).                       !
   352) !------------------------------------------------------------------------------------------!
   353) subroutine heun_integ(h1,csite,ipa,isi,nsteps)
   354)    use rk4_integ_utils
   355)    use rk4_misc
   356)    use rk4_copy_patch
   357)    use ed_state_vars  , only : sitetype               & ! structure
   358)                              , patchtype              & ! structure
   359)                              , polygontype            ! ! structure
   360)    use rk4_coms       , only : integration_vars       & ! structure
   361)                              , integration_buff       & ! structure
   362)                              , rk4site                & ! intent(in)
   363)                              , print_diags            & ! intent(in)
   364)                              , print_detailed         & ! intent(in)
   365)                              , maxstp                 & ! intent(in)
   366)                              , tbeg                   & ! intent(in)
   367)                              , tend                   & ! intent(in)
   368)                              , dtrk4                  & ! intent(in)
   369)                              , dtrk4i                 & ! intent(in)
   370)                              , tiny_offset            & ! intent(in)
   371)                              , checkbudget            & ! intent(in)
   372)                              , zero_rk4_patch         & ! subroutine
   373)                              , zero_rk4_cohort        & ! subroutine
   374)                              , hmin                   & ! intent(in)
   375)                              , rk4eps                 & ! intent(in)
   376)                              , rk4epsi                & ! intent(in)
   377)                              , safety                 & ! intent(in)
   378)                              , pgrow                  & ! intent(in)
   379)                              , pshrnk                 & ! intent(in)
   380)                              , errcon                 & ! intent(in)
   381)                              , norm_rk4_fluxes        & ! sub-routine
   382)                              , reset_rk4_fluxes       ! ! sub-routine
   383)    use ed_misc_coms   , only : fast_diagnostics       ! ! intent(in)
   384)    use hydrology_coms , only : useRUNOFF              ! ! intent(in)
   385)    use grid_coms      , only : nzg                    & ! intent(in)
   386)                              , nzs                    & ! intent(in)
   387)                              , time                   ! ! intent(in)
   388)    use soil_coms      , only : dslz8                  & ! intent(in)
   389)                              , runoff_time            & ! intent(in)
   390)                              , runoff_time_i          & ! intent(in)
   391)                              , simplerunoff           ! ! intent(in)
   392)    use consts_coms    , only : t3ple8                 & ! intent(in)
   393)                              , wdnsi8                 ! ! intent(in)
   394)    use therm_lib8     , only : tl2uint8               ! ! intent(in)
   395)    ! OMP use omp_lib
   396) 
   397)    implicit none
   398)    !----- Arguments -----------------------------------------------------------------------!
   399)    type(sitetype)            , target      :: csite            ! Current site
   400)    integer                   , intent(in)  :: ipa              ! Current patch ID
   401)    integer                   , intent(in)  :: isi              ! Current site ID
   402)    real(kind=8)              , intent(in)  :: h1               ! First guess of delta-t
   403)    integer                   , intent(out) :: nsteps           ! Number of steps taken.
   404)    !----- Local variables -----------------------------------------------------------------!
   405)    type(patchtype)           , pointer     :: cpatch           ! Current patch
   406)    logical                                 :: reject_step      ! Should I reject the step?
   407)    logical                                 :: reject_result    ! Should I reject the result?
   408)    logical                                 :: minstep          ! Minimum time step reached
   409)    logical                                 :: stuck            ! Tiny step, it won't advance
   410)    logical                                 :: test_reject      ! Reject the test
   411)    integer                                 :: i                ! Step counter
   412)    integer                                 :: k                ! Format counter
   413)    integer                                 :: ksn              ! # of snow/water layers
   414)    real(kind=8)                            :: x                ! Elapsed time
   415)    real(kind=8)                            :: xnew             ! Elapsed time + h
   416)    real(kind=8)                            :: newh             ! New time step suggested
   417)    real(kind=8)                            :: oldh             ! Old time step
   418)    real(kind=8)                            :: h                ! Current delta-t attempt
   419)    real(kind=8)                            :: hnext            ! Next delta-t
   420)    real(kind=8)                            :: hdid             ! delta-t that worked (???)
   421)    real(kind=8)                            :: qwfree           ! Free water internal energy
   422)    real(kind=8)                            :: wfreeb           ! Free water
   423)    real(kind=8)                            :: errmax           ! Maximum error of this step
   424)    real(kind=8)                            :: elaptime         ! Absolute elapsed time.
   425)    integer                                 :: ibuff
   426)    !----- External function. --------------------------------------------------------------!
   427)    real                      , external    :: sngloff
   428)    !---------------------------------------------------------------------------------------!
   429) 
   430)    ibuff = 1
   431) 
   432)    !----- Use some aliases for simplicity. ------------------------------------------------!
   433)    cpatch => csite%patch(ipa)
   434) 
   435)    !---------------------------------------------------------------------------------------!
   436)    !     Create temporary patches.                                                         !
   437)    !---------------------------------------------------------------------------------------!
   438)    call copy_rk4_patch(integration_buff(ibuff)%initp,integration_buff(ibuff)%y,cpatch)
   439) 
   440)    !---------------------------------------------------------------------------------------!
   441)    ! Set initial time and stepsize.                                                        !
   442)    !---------------------------------------------------------------------------------------!
   443)    x = tbeg
   444)    h = h1
   445)    if (dtrk4 < 0.d0) h = -h1
   446) 
   447)    !----- Define total elapsed time. ------------------------------------------------------!
   448)    elaptime = time + x
   449) 
   450)    !---------------------------------------------------------------------------------------!
   451)    ! Begin timestep loop                                                                   !
   452)    !---------------------------------------------------------------------------------------!
   453)    timesteploop: do i=1,maxstp
   454) 
   455) 
   456)       !----- Get initial derivatives ------------------------------------------------------!
   457)       call leaf_derivs(integration_buff(ibuff)%y,integration_buff(ibuff)%dydx,csite,ipa,h  &
   458)                       ,.false.)
   459) 
   460)       !----- Get scalings used to determine stability -------------------------------------!
   461)       call get_yscal(integration_buff(ibuff)%y,integration_buff(ibuff)%dydx,h              &
   462)                     ,integration_buff(ibuff)%yscal,cpatch)
   463) 
   464)       !----- Be sure not to overstep ------------------------------------------------------!
   465)       if((x+h-tend)*(x+h-tbeg) > 0.d0) h=tend-x
   466) 
   467)       !------------------------------------------------------------------------------------!
   468)       !     Here we will perform the Heun's integration using the time step.  As in Runge- !
   469)       ! Kutta, we also check whether the integration is going well and if needed we shrink !
   470)       ! the intermediate time steps.                                                       !
   471)       !------------------------------------------------------------------------------------!
   472)       reject_step =  .false.
   473)       hstep:   do
   474) 
   475)          !---------------------------------------------------------------------------------!
   476)          ! 1. Try a step of varying size.                                                  !
   477)          !---------------------------------------------------------------------------------!
   478)          call heun_stepper(x,h,csite,ipa,reject_step,reject_result)
   479) 
   480)          !---------------------------------------------------------------------------------!
   481)          !     Here we check the error of this step.  Three outcomes are possible:         !
   482)          ! 1.  The updated values make no sense.  Reject step, assign a large error and    !
   483)          !     try again with a smaller time step;                                         !
   484)          ! 2.  The updated values are reasonable, but the error is large.  Reject step and !
   485)          !     try again with a smaller time step;                                         !
   486)          ! 3.  The updated values are reasonable, and the error is small.  Accept step and !
   487)          !     try again with a larger time step.                                          !
   488)          !---------------------------------------------------------------------------------!
   489)          if (reject_step .or. reject_result) then
   490)             !------------------------------------------------------------------------------!
   491)             !    If step was already rejected, that means the step had finished premature- !
   492)             ! ly, so we assign a standard large error (10.0).                              !
   493)             !------------------------------------------------------------------------------!
   494)             errmax = 1.d1
   495)          else
   496)             call get_errmax(errmax,integration_buff(ibuff)%yerr                            &
   497)                            ,integration_buff(ibuff)%yscal,csite%patch(ipa)                 &
   498)                            ,integration_buff(ibuff)%initp,integration_buff(ibuff)%ytemp)
   499)             !----- Scale the error based on the prescribed tolerance. ---------------------!
   500)             errmax = errmax * rk4epsi
   501)          end if
   502) 
   503) 
   504)          !---------------------------------------------------------------------------------!
   505)          ! 3. If that error was large, then calculate a new step size to try.  There are   !
   506)          !    two types of new tries.  If step failed to be minimally reasonable (reject-  !
   507)          !    ed) we have assigned a standard large error (10.0).  Otherwise a new step is !
   508)          !    calculated based on the size of that error.  Hopefully, those new steps      !
   509)          !    should be less than the previous h.  If the error was small, i.e. less than  !
   510)          !    rk4eps, then we are done with this step, and we can move forward             !
   511)          !    time: x = x + h                                                              !
   512)          !---------------------------------------------------------------------------------!
   513)          if (errmax > 1.d0) then
   514)             !----- Defining new step and checking if it can be. ---------------------------!
   515)             oldh    = h
   516)             newh    = safety * h * errmax**pshrnk
   517)             minstep = (newh == h) .or. newh < hmin
   518) 
   519)             !----- Defining next time, and checking if it really added something. ---------!
   520)             h       = max(1.d-1*h, newh)
   521)             xnew    = x + h
   522)             stuck   = xnew == x
   523) 
   524)             !------------------------------------------------------------------------------!
   525)             ! 3a. Here is the moment of truth... If we reached a tiny step and yet the     !
   526)             !     model didn't converge, then we print various values to inform the user   !
   527)             !     and abort the run.  Please, don't hate the messenger.                    !
   528)             !------------------------------------------------------------------------------!
   529)             if (minstep .or. stuck) then
   530) 
   531)                write (unit=*,fmt='(80a)')         ('=',k=1,80)
   532)                write (unit=*,fmt='(a)')           '   STEPSIZE UNDERFLOW IN HEUN_INTEG'
   533)                write (unit=*,fmt='(80a)')         ('-',k=1,80)
   534)                write (unit=*,fmt='(a,1x,f9.4)')   ' + LONGITUDE:     ',rk4site%lon
   535)                write (unit=*,fmt='(a,1x,f9.4)')   ' + LATITUDE:      ',rk4site%lat
   536)                write (unit=*,fmt='(a)')           ' + PATCH INFO:    '
   537)                write (unit=*,fmt='(a,1x,i6)')     '   - NUMBER:      ',ipa
   538)                write (unit=*,fmt='(a,1x,es12.4)') '   - AGE:         ',csite%age(ipa)
   539)                write (unit=*,fmt='(a,1x,i6)')     '   - DIST_TYPE:   ',csite%dist_type(ipa)
   540)                write (unit=*,fmt='(a,1x,l1)')     ' + REJECT_STEP:   ',reject_step
   541)                write (unit=*,fmt='(a,1x,l1)')     ' + REJECT_RESULT: ',reject_result
   542)                write (unit=*,fmt='(a,1x,l1)')     ' + MINSTEP:       ',minstep
   543)                write (unit=*,fmt='(a,1x,l1)')     ' + STUCK:         ',stuck
   544)                write (unit=*,fmt='(a,1x,es12.4)') ' + ERRMAX:        ',errmax
   545)                write (unit=*,fmt='(a,1x,es12.4)') ' + X:             ',x
   546)                write (unit=*,fmt='(a,1x,es12.4)') ' + H:             ',h
   547)                write (unit=*,fmt='(a,1x,es12.4)') ' + OLDH:          ',oldh
   548)                write (unit=*,fmt='(a,1x,es12.4)') ' + NEWH:          ',newh
   549)                write (unit=*,fmt='(a,1x,es12.4)') ' + SAFETY:        ',safety
   550)                write (unit=*,fmt='(80a)') ('-',k=1,80)
   551)                if (reject_step) then
   552)                   write (unit=*,fmt='(a)') '   Likely to be a rejected step problem.'
   553)                   write (unit=*,fmt='(80a)') ('=',k=1,80)
   554)                else
   555)                   write (unit=*,fmt='(a)') '   Likely to be an errmax problem.'
   556)                   write (unit=*,fmt='(80a)') ('=',k=1,80)
   557)                end if
   558) 
   559)                if (reject_result) then
   560)                   !----- Run the LSM sanity check but this time we force the print. -------!
   561)                   call rk4_sanity_check(integration_buff(ibuff)%ytemp,test_reject,csite    &
   562)                                        ,ipa,integration_buff(ibuff)%dydx,h,.true.)
   563)                   call print_sanity_check(integration_buff(ibuff)%y,csite,ipa)
   564)                elseif (reject_step) then
   565)                   call rk4_sanity_check(integration_buff(ibuff)%ak3,test_reject,csite,ipa  &
   566)                                        ,integration_buff(ibuff)%dydx,h,.true.)
   567)                   call print_sanity_check(integration_buff(ibuff)%y,csite,ipa)
   568)                else
   569)                   call print_errmax(errmax,integration_buff(ibuff)%yerr                    &
   570)                                    ,integration_buff(ibuff)%yscal,csite%patch(ipa)         &
   571)                                    ,integration_buff(ibuff)%y,integration_buff(ibuff)%ytemp)
   572)                   write (unit=*,fmt='(80a)') ('=',k=1,80)
   573)                   write (unit=*,fmt='(a,1x,es12.4)') ' - Rel. errmax:',errmax
   574)                   write (unit=*,fmt='(a,1x,es12.4)') ' - Raw errmax: ',errmax*rk4eps
   575)                   write (unit=*,fmt='(a,1x,es12.4)') ' - Epsilon:',rk4eps
   576)                   write (unit=*,fmt='(80a)') ('=',k=1,80)
   577)                end if
   578)                call print_rk4patch(integration_buff(ibuff)%y, csite,ipa)
   579)             end if
   580) 
   581)          else
   582)             !------------------------------------------------------------------------------!
   583)             ! 3b.  Great, it worked, so now we can advance to the next step.  We just need !
   584)             !      to do some minor adjustments before...                                  !
   585)             !------------------------------------------------------------------------------!
   586)             !----- i.   Final update of leaf properties to avoid negative water. ----------!
   587)             call adjust_veg_properties(integration_buff(ibuff)%ytemp,h,csite,ipa)
   588)             !----- ii.  Final update of top soil properties to avoid off-bounds moisture. -!
   589)             call adjust_topsoil_properties(integration_buff(ibuff)%ytemp,h,csite)
   590)             !----- iii.  Make snow layers stable and positively defined. ------------------!
   591)             call adjust_sfcw_properties(nzg,nzs,integration_buff(ibuff)%ytemp,h,csite)
   592)             !----- iv. Update the diagnostic variables. -----------------------------------!
   593)             call update_diagnostic_vars(integration_buff(ibuff)%ytemp,csite,ipa)
   594)             !------------------------------------------------------------------------------!
   595) 
   596) 
   597) 
   598)             !------------------------------------------------------------------------------!
   599)             ! 3c. Set up h for the next time.  And here we can relax h for the next step,  !
   600)             !    and try something faster.                                                 !
   601)             !------------------------------------------------------------------------------!
   602)             if (errmax > errcon) then
   603)                hnext = safety * h * errmax**pgrow
   604)             else
   605)                hnext = 5.d0 * h
   606)             endif
   607)             hnext = max(2.d0*hmin,hnext)
   608) 
   609)             !------ 3d. Normalise the fluxes if the user wants detailed debugging. --------!
   610)             if (print_detailed) then
   611)                call norm_rk4_fluxes(integration_buff(ibuff)%ytemp,h)
   612)                call print_rk4_state(integration_buff(ibuff)%y                              &
   613)                                    ,integration_buff(ibuff)%ytemp,csite,ipa,isi,x,h)
   614)             end if
   615) 
   616)             !----- 3e. Copy the temporary structure to the intermediate state. ------------!
   617)             call copy_rk4_patch(integration_buff(ibuff)%ytemp,integration_buff(ibuff)%y    &
   618)                                ,csite%patch(ipa))
   619) 
   620)             !------------------------------------------------------------------------------!
   621)             !    3f. Flush step-by-step fluxes to zero if the user wants detailed          !
   622)             !        debugging.                                                            !
   623)             !------------------------------------------------------------------------------!
   624)             if (print_detailed) then
   625)                call reset_rk4_fluxes(integration_buff(ibuff)%y)
   626)             end if
   627) 
   628)             !----- 3g. Update time. -------------------------------------------------------!
   629)             x        = x + h
   630)             hdid     = h
   631)             h        = hnext
   632)             elaptime = elaptime + h
   633) 
   634)             exit hstep
   635)          end if
   636)       end do hstep
   637) 
   638)       !----- If the integration reached the next step, make some final adjustments --------!
   639)       if((x-tend)*dtrk4 >= 0.d0)then
   640) 
   641)          ksn = integration_buff(ibuff)%y%nlev_sfcwater
   642) 
   643)          !---------------------------------------------------------------------------------!
   644)          !   Make temporary surface liquid water disappear.  This will not happen          !
   645)          ! immediately, but liquid water will decay with the time scale defined by         !
   646)          ! runoff_time scale. If the time scale is too tiny, then it will be forced to be  !
   647)          ! hdid (no reason to be faster than that).                                        !
   648)          !---------------------------------------------------------------------------------!
   649)          if (simplerunoff .and. ksn >= 1) then
   650) 
   651)             if (integration_buff(ibuff)%y%sfcwater_mass(ksn)    > 0.d0  .and.              &
   652)                 integration_buff(ibuff)%y%sfcwater_fracliq(ksn) > 1.d-1 ) then
   653) 
   654)                wfreeb = min(1.d0,dtrk4*runoff_time_i)                                      &
   655)                       * integration_buff(ibuff)%y%sfcwater_mass(ksn)                       &
   656)                       * (integration_buff(ibuff)%y%sfcwater_fracliq(ksn) - 1.d-1) / 9.d-1
   657)                qwfree = wfreeb                                                             &
   658)                       * tl2uint8(integration_buff(ibuff)%y%sfcwater_tempk(ksn),1.d0)
   659) 
   660)                integration_buff(ibuff)%y%sfcwater_mass(ksn) =                              &
   661)                                   integration_buff(ibuff)%y%sfcwater_mass(ksn)  - wfreeb
   662)                integration_buff(ibuff)%y%sfcwater_depth(ksn) =                             &
   663)                                   integration_buff(ibuff)%y%sfcwater_depth(ksn)            &
   664)                                 - wfreeb * wdnsi8
   665)                !----- Recompute the energy removing runoff --------------------------------!
   666)                integration_buff(ibuff)%y%sfcwater_energy(ksn) =                            &
   667)                                   integration_buff(ibuff)%y%sfcwater_energy(ksn) - qwfree
   668) 
   669)                call adjust_sfcw_properties(nzg,nzs,integration_buff(ibuff)%y,dtrk4,csite)
   670)                call update_diagnostic_vars(integration_buff(ibuff)%y,csite,ipa)
   671) 
   672)                !----- Compute runoff for output -------------------------------------------!
   673)                if (fast_diagnostics) then
   674)                   csite%runoff       (ipa) = csite%runoff(ipa)                             &
   675)                                            + sngloff(wfreeb * dtrk4i,tiny_offset)
   676)                   csite%fmean_runoff (ipa) = csite%fmean_runoff(ipa)                       &
   677)                                            + sngloff(wfreeb * dtrk4i,tiny_offset)
   678)                   csite%fmean_qrunoff(ipa) = csite%fmean_qrunoff(ipa)                      &
   679)                                            + sngloff(qwfree * dtrk4i,tiny_offset)
   680)                end if
   681)                if (checkbudget) then
   682)                   integration_buff(ibuff)%y%wbudget_loss2runoff = wfreeb                   &
   683)                         + integration_buff(ibuff)%y%wbudget_loss2runoff
   684)                   integration_buff(ibuff)%y%ebudget_loss2runoff = qwfree                   &
   685)                         + integration_buff(ibuff)%y%ebudget_loss2runoff
   686)                   integration_buff(ibuff)%y%wbudget_storage     =                          &
   687)                                      integration_buff(ibuff)%y%wbudget_storage - wfreeb
   688)                   integration_buff(ibuff)%y%ebudget_storage    =                           &
   689)                                      integration_buff(ibuff)%y%ebudget_storage - qwfree
   690)                end if
   691)             end if
   692)          end if
   693) 
   694)          !------ Copy the temporary patch to the next intermediate step -------------------!
   695)          call copy_rk4_patch(integration_buff(ibuff)%y,integration_buff(ibuff)%initp,cpatch)
   696) 
   697)          !------ Update the substep for next time and leave -------------------------------!
   698)          csite%htry(ipa) = sngl(hnext)
   699) 
   700)          !---------------------------------------------------------------------------------!
   701)          !     Update the average time step.  The square of DTLSM (tend-tbeg) is needed    !
   702)          ! because we will divide this by the time between t0 and t0+frqsum.               !
   703)          !---------------------------------------------------------------------------------!
   704)          csite%fmean_rk4step(ipa) = csite%fmean_rk4step(ipa)                               &
   705)                                   + sngl((tend-tbeg)*(tend-tbeg))/real(i)
   706)          nsteps = i
   707)          return
   708)       end if
   709) 
   710)       !----- Use hnext as the next substep ------------------------------------------------!
   711)       h = hnext
   712)    end do timesteploop
   713) 
   714)    !----- If it reached this point, that is really bad news... ----------------------------!
   715)    write (unit=*,fmt='(a)') ' ==> Too many steps in routine heun_integ'
   716)    call print_rk4patch(integration_buff(ibuff)%ytemp, csite,ipa)
   717) 
   718)    return
   719) end subroutine heun_integ

   730) !    This subroutine computes the integration step of the Heun's method.                   !
   731) ! Here we use several buffers, so a quick guide of what each one means...                  !
   732) !                                                                                          !
   733) ! 1. integration_buff(ibuff)%y      => This is the state y(t)                              !
   734) ! 2. integration_buff(ibuff)%ytemp  => This is the state y(t+h)                            !
   735) ! 3. integration_buff(ibuff)%yerr   => This is the error e(t+h)                            !
   736) ! 4. integration_buff(ibuff)%dydx   => This is the Runge-Kutta term K1                     !
   737) ! 5. integration_buff(ibuff)%ak2    => This is the Runge-Kutta term K2                     !
   738) ! 6. integration_buff(ibuff)%ak3    => This is the next step using Euler:                  !
   739) !                                      ye(t+h) = y(t) + K1 * h                             !
   740) !                                                                                          !
   741) !------------------------------------------------------------------------------------------!
   742) subroutine heun_stepper(x,h,csite,ipa,reject_step,reject_result)
   743)    use rk4_copy_patch
   744)    use rk4_misc
   745)    use rk4_integ_utils
   746)    use rk4_coms      , only : integration_buff    & ! structure
   747)                             , integration_vars    & ! structure
   748)                             , zero_rk4_patch      & ! subroutine
   749)                             , zero_rk4_cohort     & ! subroutine
   750)                             , rk4site             & ! intent(in)
   751)                             , print_diags         & ! intent(in)
   752)                             , heun_a2             & ! intent(in)
   753)                             , heun_b21            & ! intent(in)
   754)                             , heun_c1             & ! intent(in)
   755)                             , heun_c2             & ! intent(in)
   756)                             , heun_dc1            & ! intent(in)
   757)                             , heun_dc2            ! ! intent(in)
   758)    use ed_state_vars , only : sitetype            & ! structure
   759)                             , patchtype           ! ! structure
   760)    use grid_coms     , only : nzg                 & ! intent(in)
   761)                             , nzs                 ! ! structure
   762)    ! OMP use omp_lib
   763) 
   764)    implicit none
   765) 
   766)    !----- Arguments -----------------------------------------------------------------------!
   767)    type(sitetype)    , target      :: csite
   768)    integer           , intent(in)  :: ipa
   769)    logical           , intent(out) :: reject_step
   770)    logical           , intent(out) :: reject_result
   771)    real(kind=8)      , intent(in)  :: x
   772)    real(kind=8)      , intent(in)  :: h
   773)    !----- Local variables -----------------------------------------------------------------!
   774)    type(patchtype)   , pointer     :: cpatch
   775)    integer                         :: ibuff
   776)    !---------------------------------------------------------------------------------------!
   777) 
   778)    ibuff = 1
   779) 
   780)    !---------------------------------------------------------------------------------------!
   781)    !     Start and assume that nothing went wrong up to this point... If we find any       !
   782)    ! seriously bad step, quit and reduce the time step without even bothering to try       !
   783)    ! further.                                                                              !
   784)    !---------------------------------------------------------------------------------------!
   785)    reject_step   = .false.
   786)    reject_result = .false.
   787)    cpatch => csite%patch(ipa)
   788)    !---------------------------------------------------------------------------------------!
   789) 
   790) 
   791) 
   792)    !----- yeuler is the temporary array with the Euler step with no correction. -----------!
   793)    call copy_rk4_patch(integration_buff(ibuff)%y,integration_buff(ibuff)%ak3,cpatch)
   794)    call inc_rk4_patch (integration_buff(ibuff)%ak3,integration_buff(ibuff)%dydx            &
   795)                       ,heun_b21*h, cpatch)
   796)    call update_diagnostic_vars(integration_buff(ibuff)%ak3      ,csite,ipa)
   797)    !---------------------------------------------------------------------------------------!
   798) 
   799) 
   800) 
   801) 
   802)    !---------------------------------------------------------------------------------------!
   803)    !    Check to see if the Euler result makes sense.  Since we will compute the           !
   804)    ! derivative correction using it, the Euler step must be bounded.  If not, reject the   !
   805)    ! step and try a smaller step size.                                                     !
   806)    !---------------------------------------------------------------------------------------!
   807)    call rk4_sanity_check(integration_buff(ibuff)%ak3,reject_step,csite,ipa                 &
   808)                         ,integration_buff(ibuff)%dydx,h,print_diags)
   809)    if (reject_step) return
   810)    !---------------------------------------------------------------------------------------!
   811) 
   812) 
   813) 
   814)    !---------------------------------------------------------------------------------------!
   815)    !     Compute the second term (correction) of the derivative, using the Euler's         !
   816)    ! predicted state.                                                                      !
   817)    !---------------------------------------------------------------------------------------!
   818)    call leaf_derivs(integration_buff(ibuff)%ak3,integration_buff(ibuff)%ak2, csite,ipa,h   &
   819)                    ,.false.)
   820)    !---------------------------------------------------------------------------------------!
   821) 
   822) 
   823) 
   824)    !----- We now combine both derivatives and find the potential next step. ---------------!
   825)    call copy_rk4_patch(integration_buff(ibuff)%y,integration_buff(ibuff)%ytemp, cpatch)
   826)    call inc_rk4_patch(integration_buff(ibuff)%ytemp,integration_buff(ibuff)%dydx           &
   827)                      ,heun_c1*h, cpatch)
   828)    call inc_rk4_patch(integration_buff(ibuff)%ytemp,integration_buff(ibuff)%ak2            &
   829)                      ,heun_c2*h, cpatch)
   830)    !---------------------------------------------------------------------------------------!
   831) 
   832) 
   833) 
   834)    !---------------------------------------------------------------------------------------!
   835)    !      Update the diagnostic properties and make final adjustments.  This time we will  !
   836)    ! run the full adjustment, to make sure that the step will be rejected especially if    !
   837)    ! there are issues with the top soil properties.                                        !
   838)    !---------------------------------------------------------------------------------------!
   839)    call update_diagnostic_vars   (integration_buff(ibuff)%ytemp        ,csite,ipa)
   840)    !---------------------------------------------------------------------------------------!
   841) 
   842) 
   843) 
   844)    !---------------------------------------------------------------------------------------!
   845)    !    Check to see if this attempt of advancing one time step makes sense.  If not,      !
   846)    ! reject the result and try a smaller step size.                                        !
   847)    !---------------------------------------------------------------------------------------!
   848)    call rk4_sanity_check(integration_buff(ibuff)%ytemp,reject_result,csite,ipa             &
   849)                         ,integration_buff(ibuff)%ak2,h,print_diags)
   850)    if(reject_result)return
   851)    !---------------------------------------------------------------------------------------!
   852) 
   853) 
   854) 
   855)    !---------------------------------------------------------------------------------------!
   856)    !     Compute the estimate of the error associated with the step.                       !
   857)    !---------------------------------------------------------------------------------------!
   858)    call zero_rk4_patch (integration_buff(ibuff)%yerr)
   859)    call zero_rk4_cohort(integration_buff(ibuff)%yerr)
   860)    call inc_rk4_patch(integration_buff(ibuff)%yerr,integration_buff(ibuff)%dydx            &
   861)                      ,heun_dc1*h,cpatch)
   862)    call inc_rk4_patch(integration_buff(ibuff)%yerr,integration_buff(ibuff)%ak2             &
   863)                      ,heun_dc2*h,cpatch)
   864)    !---------------------------------------------------------------------------------------!
   865) 
   866)    return
   867) end subroutine heun_stepper
