    38)    logical, intent(in)  :: history_time

    49)    !      If there is any IO, then we need to check whether the pointer tables are up to   !

    51)    ! cohorts or patches (e.g, a cohort or a patch has been terminated or created).         !

    53)    if (analysis_time   .or. history_time    .or.                                           &

    72)    if ( analysis_time .or.   history_time .or. dcycle_time  .or.                           &

    81)             call integrate_ed_dmean_vars(edgrid_g(ifm))

    85)    end if

    94)       if (itoutput == 3) call h5_output('OPTI')

   111)          call normalize_ed_dmean_vars(edgrid_g(ifm))
   112)          if (writing_mont .or. writing_dcyc) then
   113)             call integrate_ed_mmean_vars(edgrid_g(ifm))
   114)          end if
   115)       end do
   116) 
   117)       if (dail_analy_time) call h5_output('DAIL')
   118) 
   119)       do ifm=1,ngrids
   120)          call zero_ed_dmean_vars(edgrid_g(ifm))
   121)       end do
   122)    end if
   123)    !---------------------------------------------------------------------------------------!
   124) 
   125) 
   126) 
   127) 
   128)    !----- Monthly analysis and monthly mean diurnal cycle output. -------------------------!
   129)    if (mont_analy_time .or. dcyc_analy_time) then
   130)       do ifm=1,ngrids
   131)          call normalize_ed_mmean_vars(edgrid_g(ifm))
   132)          if (writing_dcyc) call normalize_ed_qmean_vars(edgrid_g(ifm))
   133)       end do

   135)       if (dcyc_analy_time) call h5_output('DCYC')

   137)          call zero_ed_mmean_vars(edgrid_g(ifm))
   138)          if (writing_dcyc) call zero_ed_qmean_vars(edgrid_g(ifm))
   139)       end do
   140)    end if
   141)    !---------------------------------------------------------------------------------------!
   142) 
   143) 
   144) 
   145)    !----- Yearly analysis output. ---------------------------------------------------------!
   146)    if (annual_time) then
   147)       call h5_output('YEAR')
   148)       do ifm=1,ngrids
   149)          call zero_ed_yearly_vars(edgrid_g(ifm))
   150)       end do
   151)    end if
   152)    !---------------------------------------------------------------------------------------!
   153) 
   154) 
   155) 
   156) 
   157)    !---------------------------------------------------------------------------------------!
   158)    !      History files should only be output at a frequency which divides by frqanl, thus !
   159)    ! the integrated fast-time variables are valid, but representative of the last frqanl   !
   160)    ! period, not the last frqhist period.                                                  !
   161)    !---------------------------------------------------------------------------------------!
   162)    if (history_time .and. isoutput /= 0) then
