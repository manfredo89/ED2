    73) subroutine shdf5_open_f(locfn,access,idelete)
    74) 
    75)   use hdf5_coms
    76)   implicit none
    77)   
    78)   character(len=*) :: locfn     ! file name
    79)   character(len=*) :: access    ! File access ('R','W','RW')
    80)   integer, optional :: idelete  ! If W, delete/overwrite file if exists? 1=yes, 0=no
    81)   ! Only needed when access='W'
    82) #if USE_HDF5
    83) 
    84)   integer :: hdferr ! Error flag for HDF5
    85)   character(len=2) :: caccess ! File access ('R ','W ','RW')
    86)   
    87)   logical :: exists ! File existence
    88)   caccess = access
    89)   
    90)   ! Check for existence of RAMS file.
    91) 
    92)   ! Open the HDF environment
    93)   
    94)   call h5open_f(hdferr)
    95) 
    96)   ! Collect garbage
    97)   call h5garbage_collect_f(hdferr)
    98)   
    99)   inquire(file=trim(locfn),exist=exists)
   100)   
   101)   ! Create a new file or open an existing RAMS file.
   102)   if (access(1:1) == 'R') then
   103)      if (.not.exists) then
   104)         print*,'shdf5_open:'
   105)         print*,'   Attempt to open a file for reading that does not exist.'
   106)         print*,'   Filename: ',trim(locfn)
   107)         call fatal_error('No file'//trim(locfn)//'.','shdf5_open','hdf5_utils.f90')
   108)      else
   109)         if (caccess == 'R ') then
   110)            call h5fopen_f(trim(locfn)//char(0), H5F_ACC_RDONLY_F, fileid_f, hdferr)
   111)         else if (caccess == 'RW') then
   112)            call h5fopen_f(trim(locfn)//char(0), H5F_ACC_RDWR_F, fileid_f, hdferr)
   113)         else
   114)            call fatal_error('INVALID FILE READ TYPE ACCESS. SHOULD BE READ OR READ-WRITE' &
   115)                            ,'shdf5_open','hdf5_utils.f90')
   116)            stop
   117)         endif
   118)         
   119)         if (hdferr < 0) then
   120)            print*,'   Error opening hdf5 file - error -',hdferr
   121)            call fatal_error('Error opening file '//trim(locfn)  &
   122)                            ,'shdf5_open','hdf5_utils.f90')
   123)         endif
   124)      endif
   125) 
   126)   elseif (access(1:1) == 'W') then
   127)      
   128)      if (.not.exists) then
   129) 
   130)         call h5fcreate_f(trim(locfn)//char(0), H5F_ACC_EXCL_F, fileid_f, hdferr)
   131)         
   132)      else
   133)         if(.not.present(idelete) ) then
   134)            call fatal_error('idelete not specified when access=W' &
   135)                            ,'shdf5_open','hdf5_utils.f90')
   136) 
   137)         endif
   138)         
   139)         if(idelete == 0) then
   140)            print*,'In shdf5_open:'
   141)            print*,'   Attempt to open an existing file for writing, '
   142)            print*,'      but overwrite is disabled. idelete=',idelete
   143)            call fatal_error('Open existing file for writing with no overwriting.' &
   144)                            ,'shdf5_open','hdf5_utils.f90')
   145)         else
   146)            ! Avoiding system calls...
   147)            open (unit=99,file=trim(locfn))
   148)            close (unit=99,status='delete')
   149) !           call system('rm -f '//trim(locfn)//char(0))
   150)            
   151)            call h5fcreate_f(trim(locfn)//char(0), H5F_ACC_TRUNC_F, fileid_f, hdferr)
   152)            
   153)         endif
   154)      endif
   155)      if(hdferr < 0) then
   156)         print*,'file name:',trim(locfn),' ',trim(access), idelete,hdferr
   157)         call fatal_error('HDF5 file '//trim(locfn)//' create failed:'   &
   158)                         ,'shdf5_open','hdf5_utils.f90')
   159)      endif
   160)   endif
   161)   
   162)   return
   163) #endif
   164) end  subroutine shdf5_open_f

   168) subroutine shdf5_orec_f(ndims,dims,dsetname,ivara,rvara,cvara,dvara,lvara  &
   169)                                          ,ivars,rvars,cvars,dvars,lvars)
   170)   
   171)   use hdf5_coms
   172)   implicit none
   173)   
   174)   character(len=*) :: dsetname ! Variable label
   175)   integer :: ndims             ! Number of dimensions or rank
   176)   integer, dimension(*) :: dims ! Dataset dimensions.
   177)   
   178)   ! Array and scalar arguments for different types. Only specify one in each call.
   179)   integer,          optional :: ivara(*),ivars
   180)   real,             optional :: rvara(*),rvars
   181)   character(len=*), optional :: cvara(*),cvars
   182)   real(kind=8),     optional :: dvara(*),dvars
   183)   logical,          optional :: lvara(*),lvars
   184) #if USE_HDF5
   185)   
   186)   integer:: h5_type   ! Local type designator
   187)   
   188)   integer, dimension(4) :: dimsh ! Dataset dimensions.
   189)   
   190)   integer(HSIZE_T),dimension(4) :: dimshf,chunksize
   191)   
   192)   character(len=2) :: ctype    ! Variable type: int, real, char
   193)   integer :: hdferr ! Error flag
   194)   
   195)   ! Find which data type is input
   196)   if(present(ivars)) then ; ctype='is'
   197)   elseif(present(rvars)) then ; ctype='rs'
   198)   elseif(present(cvars)) then ; ctype='cs'
   199)   elseif(present(dvars)) then ; ctype='ds'
   200)   elseif(present(lvars)) then ; ctype='ls'
   201)   elseif(present(ivara)) then ; ctype='ia'
   202)   elseif(present(rvara)) then ; ctype='ra'
   203)   elseif(present(cvara)) then ; ctype='ca'
   204)   elseif(present(dvara)) then ; ctype='da'
   205)   elseif(present(lvara)) then ; ctype='la'
   206)   else
   207)      print*,'Incorrect or missing data field argument in shdf5_orec'
   208)      stop 'shdf5_orec: bad data field'
   209)   endif
   210)   
   211)   ! Check dimensions and set compression chunk size
   212)   
   213)   if (ndims <=0 .or. minval(dims(1:ndims)) <=0) then
   214)      print*,'Dimension error in shdf5_orec:',ndims,dims(1:ndims)
   215)      stop 'shdf5_orec: bad dims'
   216)   endif
   217)   
   218)   dimsh(1:ndims) = dims(1:ndims)
   219)   dimshf = 0_8
   220)   chunksize = 0_8
   221)   dimshf(1:ndims) = int(dimsh(1:ndims),8)
   222)   chunksize(1:ndims) = int(dimsh(1:ndims),8)
   223)   
   224)   ! Prepare memory and options for the write
   225)   
   226)   call h5screate_simple_f(ndims,dimshf,dspaceid_f,hdferr)
   227)   if (hdferr.ne.0) then
   228)      print*,"COULD NOT CREATE DATASPACE"
   229)      print*,ndims,dimshf,dspaceid_f,hdferr
   230)      stop
   231)   endif
   232)   
   233)   call h5pcreate_f(H5P_DATASET_CREATE_F, prpid_f, hdferr)
   234)   if (hdferr.ne.0) then
   235)      print*,"COULD NOT CREATE PROPERTY"
   236)      print*,prpid_f,hdferr
   237)      stop
   238)   endif
   239)   
   240)   call h5pset_chunk_f(prpid_f, ndims, chunksize, hdferr) 
   241)   if (hdferr.ne.0) then
   242)      print*,"COULD NOT SET CHUNK"
   243)      print*,ndims,chunksize,hdferr
   244)      stop
   245)   endif
   246)   
   247)   call h5pset_shuffle_f(prpid_f, hdferr)
   248)   if (hdferr.ne.0) then
   249)      print*,"COULD NOT SET SHUFFLE"
   250)      print*,hdferr
   251)      stop
   252)   endif
   253)   
   254)   call h5pset_deflate_f(prpid_f, 9, hdferr)
   255)   if (hdferr.ne.0) then
   256)      print*,"COULD NOT SET DEFLATE"
   257)      print*,hdferr
   258)      stop
   259)   endif
   260)   
   261)   if (ctype(1:1) == 'i') h5_type=1
   262)   if (ctype(1:1) == 'r') h5_type=2
   263)   if (ctype(1:1) == 'c') h5_type=3
   264)   if (ctype(1:1) == 'd') h5_type=4  ! If native precision is 8 bytes, do h5_type=2
   265)   if (ctype(1:1) == 'l') h5_type=5
   266)   
   267)   ! Write the dataset.
   268)   if (ctype == 'is') then
   269)      
   270)      call h5dcreate_f(fileid_f,trim(dsetname)//char(0), H5T_NATIVE_INTEGER, dspaceid_f,dsetid_f, hdferr)
   271)      call h5dwrite_f(dsetid_f,H5T_NATIVE_INTEGER,ivars,dimshf,hdferr)
   272)      
   273)   elseif (ctype == 'rs') then
   274)      
   275)      call h5dcreate_f(fileid_f,trim(dsetname)//char(0), H5T_NATIVE_REAL, dspaceid_f,dsetid_f, hdferr)
   276)      call h5dwrite_f(dsetid_f,H5T_NATIVE_REAL,rvars,dimshf,hdferr)
   277)      
   278)   elseif (ctype == 'cs') then
   279)      
   280)      call h5dcreate_f(fileid_f,trim(dsetname)//char(0), H5T_NATIVE_CHARACTER, dspaceid_f,dsetid_f, hdferr)
   281)      call h5dwrite_f(dsetid_f,H5T_NATIVE_CHARACTER,cvars,dimshf,hdferr)
   282)      
   283)   elseif (ctype == 'ds') then
   284)      
   285)      call h5dcreate_f(fileid_f,trim(dsetname)//char(0), H5T_NATIVE_DOUBLE, dspaceid_f,dsetid_f, hdferr)
   286)      call h5dwrite_f(dsetid_f,H5T_NATIVE_DOUBLE,dvars,dimshf,hdferr)
   287)      
   288)   elseif (ctype == 'ls') then
   289)      print*,"THERE IS NO HDF5 FORTRAN API DATATYPE FOR BOOLEAN"
   290)      print*,"YOU MUST CHANGE BACK TO C IO FOR THIS"
   291)      print*,"STOPPING"
   292)      stop
   293)      
   294)   elseif (ctype == 'ia') then
   295)      
   296)      call h5dcreate_f(fileid_f,trim(dsetname)//char(0), H5T_NATIVE_INTEGER, dspaceid_f,dsetid_f, hdferr)
   297)      call h5dwrite_f(dsetid_f,H5T_NATIVE_INTEGER,ivara,dimshf,hdferr)
   298)      
   299)   elseif (ctype == 'ra') then
   300)      
   301)      call h5dcreate_f(fileid_f,trim(dsetname)//char(0), H5T_NATIVE_REAL, dspaceid_f,dsetid_f, hdferr)
   302)      call h5dwrite_f(dsetid_f,H5T_NATIVE_REAL,rvara,dimshf,hdferr)
   303)      
   304)   elseif (ctype == 'ca') then
   305)      
   306)      call h5dcreate_f(fileid_f,trim(dsetname)//char(0), H5T_NATIVE_CHARACTER, dspaceid_f,dsetid_f, hdferr)
   307)      call h5dwrite_f(dsetid_f,H5T_NATIVE_CHARACTER,cvara,dimshf,hdferr)
   308)      
   309)   elseif (ctype == 'da') then
   310)      
   311)      call h5dcreate_f(fileid_f,trim(dsetname)//char(0), H5T_NATIVE_DOUBLE, dspaceid_f,dsetid_f, hdferr)
   312)      call h5dwrite_f(dsetid_f,H5T_NATIVE_DOUBLE,dvara,dimshf,hdferr)
   313)      
   314)   elseif (ctype == 'la') then
   315)      print*,"THERE IS NO HDF5 FORTRAN API DATATYPE FOR BOOLEAN"
   316)      print*,"YOU MUST CHANGE BACK TO C IO FOR THIS"
   317)      print*,"STOPPING"
   318)      stop
   319)      
   320)   endif
   321)   if (hdferr /= 0) then
   322)      print*,'In shdf5_orec: hdf5 write error =',hdferr
   323)      stop 'shdf5_orec: hdf5 write error'
   324)   endif
   325)   
   326)   ! Close the dataset, the dataspace for the dataset, and the dataspace properties.
   327)   
   328)   !   Close out the dataspace
   329)   call h5sclose_f(dspaceid_f, hdferr)
   330)   if (hdferr.ne.0) then
   331)      print*,"COULD NOT CLOSE DATASPACE"
   332)      stop
   333)   endif
   334)   
   335)   !   Close the dataset
   336)   call h5dclose_f(dsetid_f, hdferr)
   337)   if (hdferr.ne.0) then
   338)      print*,"COULD NOT CLOSE THE DATASET"
   339)      stop
   340)   endif
   341)   
   342)   !   Close the property
   343)   call h5pclose_f(prpid_f, hdferr)
   344)   if (hdferr.ne.0) then
   345)      print*,"COULD NOT CLOSE THE PROPERTY"
   346)      stop
   347)   endif
   348)     
   349)   return
   350) #endif
   351) end subroutine shdf5_orec_f

   355) subroutine shdf5_irec_f(ndims,dims,dsetname,ivara,rvara,cvara,dvara,lvara  &
   356)                                          ,ivars,rvars,cvars,dvars,lvars)
   357)   use hdf5_coms
   358)   implicit none
   359) 
   360)   character(len=*) :: dsetname ! Dataset name
   361)   integer :: ndims             ! Number of dimensions or rank
   362)   integer, dimension(*) :: dims ! Dataset dimensions.
   363)   
   364)   ! Array and scalar arguments for different types. Only specify one in each call.
   365)   integer,          optional :: ivara(*),ivars
   366)   real,             optional :: rvara(*),rvars
   367)   character(len=*), optional :: cvara(*),cvars
   368)   real(kind=8),     optional :: dvara(*),dvars
   369)   logical,          optional :: lvara(*),lvars
   370)   
   371) #if USE_HDF5
   372)   integer, dimension(4) :: dimsh ! Dataset dimensions.
   373)   
   374)   integer(HSIZE_T),dimension(4) :: dimshf
   375)   
   376)   integer :: hdferr ! Error flag
   377)   
   378)   character(len=2) :: ctype
   379)   
   380)   logical :: convert = .false.
   381)   integer(HID_T) :: type_id
   382)   real(kind=8), allocatable, dimension(:) :: dvaraTEMP
   383)   
   384)   ! Find which data type will be read
   385)   if(present(ivars)) then ; ctype='is'
   386)   elseif(present(rvars)) then ; ctype='rs'
   387)   elseif(present(cvars)) then ; ctype='cs'
   388)   elseif(present(dvars)) then ; ctype='ds'
   389)   elseif(present(lvars)) then ; ctype='ls'
   390)   elseif(present(ivara)) then ; ctype='ia'
   391)   elseif(present(rvara)) then ; ctype='ra'
   392)   elseif(present(cvara)) then ; ctype='ca'
   393)   elseif(present(dvara)) then ; ctype='da'
   394)   elseif(present(lvara)) then ; ctype='la'
   395)   else
   396)      print*,'Incorrect or missing data field argument in shdf5_irec'
   397)      stop 'shdf5_irec: bad data field'
   398)   endif
   399)   
   400)   ! Check dimensions and set compression chunk size
   401)   
   402)   if (ndims <=0 .or. minval(dims(1:ndims)) <=0) then
   403)      print*,'Dimension error in shdf5_irec:',ndims,dims(1:ndims)
   404)      stop 'shdf5_irec: bad dims'
   405)   endif
   406)   
   407)   dimsh(1:ndims) = dims(1:ndims)
   408)   dimshf = 0_8
   409)   dimshf(1:ndims) = int(dimsh(1:ndims),8)
   410)   
   411)   call h5dopen_f(fileid_f,trim(dsetname)//char(0), dsetid_f, hdferr)
   412)   call h5dget_space_f(dsetid_f, dspaceid_f, hdferr)
   413) 
   414)   !! check that data field arguement matches data type, convert if possible
   415)   call h5dget_type_f(dsetid_f,type_id,hdferr)
   416) 
   417)   if(ctype == 'ra' .and. (type_id == H5T_NATIVE_DOUBLE .or. type_id == H5T_IEEE_F64LE) .and. ndims == 1) then
   418)      print*,"TYPEID",type_id,H5T_NATIVE_DOUBLE,H5T_IEEE_F64LE
   419)      print*,"shdf5_irec_f: Trying to convert double to real may result in loss of precision" 
   420)      allocate(dvaraTEMP(1:dims(1)))
   421)      call h5dread_f(dsetid_f,H5T_NATIVE_DOUBLE,dvaraTEMP, dimshf, hdferr )
   422)      rvara(1:dims(1)) = sngl(dvaraTEMP(1:dims(1)))
   423)      stop
   424)   endif
   425) 
   426)

   428)      select case (ctype)
   429)      case ('is')
   430)         call h5dread_f(dsetid_f, H5T_NATIVE_INTEGER, ivars, dimshf, hdferr)
   431)      case ('rs') 
   432)         call h5dread_f(dsetid_f, H5T_NATIVE_REAL,rvars, dimshf, hdferr )
   433)      case ('cs') 
   434)         call h5dread_f(dsetid_f,H5T_NATIVE_CHARACTER,cvars, dimshf, hdferr )
   435)      case ('ds') 
   436)         call h5dread_f(dsetid_f,H5T_NATIVE_DOUBLE,dvars, dimshf, hdferr )
   437)      case ('ls') 
   438)         !      call h5dread_f(dsetid_f,H5T_NATIVE_HBOOL,lvars, dimsh, hdferr )
   439)         print*,"THERE IS NO HDF5 FORTRAN API DATATYPE FOR BOOLEAN"
   440)         print*,"YOU MUST CHANGE BACK TO C IO FOR THIS"
   441)         call fatal_error ('Attempt to convert logical variables'&
   442)                          &,'shdf5_irec_f','hdf5_utils.f90')
   443)      case ('ia') 
   444)         call h5dread_f(dsetid_f,H5T_NATIVE_INTEGER,ivara, dimshf, hdferr )
   445)      case ('ra') 
   446)         call h5dread_f(dsetid_f,H5T_NATIVE_REAL,rvara, dimshf, hdferr )
   447)      case ('ca') 
   448)         call h5dread_f(dsetid_f,H5T_NATIVE_CHARACTER,cvara, dimshf, hdferr )
   449)      case ('da') 
   450)         call h5dread_f(dsetid_f,H5T_NATIVE_DOUBLE,dvara, dimshf, hdferr )
   451)      case ('la') 
   452)         print*,"THERE IS NO HDF5 FORTRAN API DATATYPE FOR BOOLEAN"
   453)         print*,"YOU MUST CHANGE BACK TO C IO FOR THIS"
   454)         call fatal_error ('Attempt to convert logical variables'&
   455)                          &,'shdf5_irec_f','hdf5_utils.f90')
   456)      case default
   457)         call fatal_error ('Invalid ctype '//trim(ctype)//'!'&
   458)                          &,'shdf5_irec_f','hdf5_utils.f90')
   459)      end select
   460)   endif
   461)   
   462)   
   463)   call h5sclose_f(dspaceid_f, hdferr)
   464)   call h5dclose_f(dsetid_f, hdferr)
   465)   
   466) #endif
   467)  
   468)   return
   469) end subroutine shdf5_irec_f
