     1) module landuse_init_module
     2)   contains
     3) 
     4) !==========================================================================================!
     5) !==========================================================================================!
     6) !     This file will read the landuse files and assign the anthropogenic disturbance       !
     7) ! matrices.                                                                                !
     8) !------------------------------------------------------------------------------------------!
     9) subroutine landuse_init
    10) 
    11)    use ed_state_vars , only : edtype          & ! structure
    12)                             , polygontype     & ! structure
    13)                             , sitetype        & ! structure
    14)                             , edgrid_g        ! ! structure
    15)    use consts_coms   , only : erad            & ! intent(in)
    16)                             , pio180          ! ! intent(in)
    17)    use disturb_coms  , only : lutime          & ! intent(in)
    18)                             , max_lu_years    & ! intent(in)
    19)                             , num_lu_trans    & ! intent(in)
    20)                             , ianth_disturb   & ! intent(in)
    21)                             , lu_database     ! ! intent(in)
    22)    use ed_misc_coms  , only : iyeara          & ! intent(in)
    23)                             , iyearz          ! ! intent(in)
    24)    use grid_coms     , only : ngrids          ! ! intent(in)
    25)    use ed_max_dims   , only : str_len         & ! intent(in)
    26)                             , huge_lu         & ! intent(in)
    27)                             , n_pft           & ! intent(in)
    28)                             , maxlist         ! ! intent(in)
    29) 
    30)    implicit none
    31)    !----- Local variables -----------------------------------------------------------------!
    32)    type(edtype)          , pointer            :: cgrid
    33)    type(polygontype)     , pointer            :: cpoly
    34)    type(sitetype)        , pointer            :: csite
    35)    type(lutime)          , pointer            :: clutime
    36)    type(lutime)          , pointer            :: onelutime
    37)    character(len=str_len), dimension(maxlist) :: full_list
    38)    character(len=str_len), dimension(maxlist) :: lu_list
    39)    real                  , dimension(maxlist) :: llon_list
    40)    real                  , dimension(maxlist) :: llat_list
    41)    real                  , dimension(maxlist) :: file_ldist
    42)    character(len=6)                           :: hform
    43)    character(len=str_len)                     :: lu_name
    44)    character(len=str_len)                     :: cdum
    45)    integer                                    :: nharvest
    46)    integer               , dimension(n_pft)   :: harvest_pft
    47)    integer                                    :: nf
    48)    integer                                    :: nflist
    49)    integer                                    :: nfllu
    50)    integer                                    :: ncl
    51)    integer                                    :: iyear
    52)    integer                                    :: igr
    53)    integer                                    :: ipy
    54)    integer                                    :: isi
    55)    integer                                    :: ipft
    56)    integer                                    :: h
    57)    integer                                    :: sim_years
    58)    integer                                    :: yd_1st
    59)    integer                                    :: yd_this
    60)    integer                                    :: yd_last
    61)    logical                                    :: inside
    62)    real                  , dimension(n_pft)   :: mindbh_1ary
    63)    real                  , dimension(n_pft)   :: harvprob_1ary
    64)    real                  , dimension(n_pft)   :: mindbh_2ary
    65)    real                  , dimension(n_pft)   :: harvprob_2ary
    66)    real                                       :: lu_area
    67)    real                                       :: lu_area_i
    68)    real                                       :: wlon
    69)    real                                       :: elon
    70)    real                                       :: slat
    71)    real                                       :: nlat
    72)    !----- Local constants. ----------------------------------------------------------------!
    73)    character(len=12)    , parameter           :: fffmt    = '(a,1x,f12.5)'
    74)    character(len=13)    , parameter           :: esfmt    = '(a,1x,es12.5)'
    75)    integer              , parameter           :: hoff     = 18
    76)    real                 , parameter           :: huge_dbh = huge(1.)
    77)    !----- External function. --------------------------------------------------------------!
    78)    real                 , external            :: dist_gc
    79)    !---------------------------------------------------------------------------------------!
    80) 
    81) 
    82)    !----- Finding number of simulation years ----------------------------------------------!
    83)    sim_years = iyearz-iyeara+1
    84) 
    85)    !----- Crashing the run if the user set up a very long run... --------------------------!
    86)    if (ianth_disturb == 1 .and. sim_years > max_lu_years) then
    87)       write (unit=*,fmt='(a,1x,i5)') 'IYEARA       (From namelist)        :',iyeara
    88)       write (unit=*,fmt='(a,1x,i5)') 'IYEARZ       (From namelist)        : ',iyearz
    89)       write (unit=*,fmt='(a,1x,i5)') 'MAX_LU_YEARS (From disturb_coms.f90): ',max_lu_years
    90)       write (unit=*,fmt='(a)') ' Your run is too long.  Try increasing max_lu_years,'
    91)       write (unit=*,fmt='(a)') ' so MAX_LU_YEARS >= IYEARZ-IYEARA+1, then recompile'
    92)       write (unit=*,fmt='(a)') ' your model.'
    93)       call fatal_error ('Simulation is too long for anthropogenic disturbance.'            &
    94)                        ,'landuse_init','landuse_init.f90')
    95)    end if
    96) 
    97) 
    98)    gridloop: do igr = 1,ngrids
    99) 
   100)       !------------------------------------------------------------------------------------!
   101)       !     Find the list of disturbance rate files.                                       !
   102)       !------------------------------------------------------------------------------------!
   103)       if (ianth_disturb == 1) then
   104)          call ed_filelist(full_list,lu_database(igr),nflist)
   105)          call ed1_fileinfo('.lu',nflist,full_list,nfllu,lu_list,llon_list,llat_list)
   106)       end if
   107)       !------------------------------------------------------------------------------------!
   108) 
   109)       cgrid=>edgrid_g(igr)
   110) 
   111)       polyloop: do ipy = 1,cgrid%npolygons
   112)          cpoly => cgrid%polygon(ipy)
   113) 
   114)          select case (ianth_disturb)
   115)          case (1)
   116) 
   117)             !------------------------------------------------------------------------------!
   118)             !     Comput the distance between the current polygon and all the files.       !
   119)             !------------------------------------------------------------------------------!
   120)             do nf=1,nfllu
   121)                file_ldist(nf) = dist_gc(cgrid%lon(ipy),llon_list(nf)                       &
   122)                                        ,cgrid%lat(ipy),llat_list(nf) )
   123)             end do
   124) 
   125)             !------------------------------------------------------------------------------!
   126)             !    Pick the closest file.  This is not a guarantee that it will be used      !
   127)             ! because the closest polygon area must contain the point associated to the    !
   128)             ! current polygon.                                                             !
   129)             !------------------------------------------------------------------------------!
   130)             ncl     = minloc(file_ldist(1:nfllu),dim=1)
   131)             lu_name = lu_list(ncl)
   132)             write (unit=*,fmt='(2a)') 'Using land use file: ',trim(lu_name)
   133) 
   134)             !------------------------------------------------------------------------------!
   135)             !    Open the patch file and read in all patches.                              !
   136)             !------------------------------------------------------------------------------!
   137)             open(unit=12,file=trim(lu_name),form='formatted',status='old',action='read')
   138) 
   139)             !------------------------------------------------------------------------------!
   140)             !     Initialise the temporary arrays with data that will not cause any        !
   141)             ! harvesting.  The actual variables will be read in the following block.       !
   142)             !------------------------------------------------------------------------------!
   143)             harvest_pft(1:n_pft)   = -1
   144)             mindbh_1ary(1:n_pft)   = huge_dbh
   145)             harvprob_1ary(1:n_pft) = 0.
   146)             mindbh_2ary(1:n_pft)   = huge_dbh
   147)             harvprob_2ary(1:n_pft) = 0.
   148) 
   149)             !----- Define the format for the header. --------------------------------------!
   150)             write(hform,fmt='(a,i3.3,a)') '(a',str_len,')'
   151) 
   152)             !----- Read the header. -------------------------------------------------------!
   153)             read (unit=12,fmt=hform) cdum
   154)             cdum = cdum(hoff:)
   155)             read (cdum, fmt=*) wlon
   156) 
   157)             read (unit=12,fmt=hform) cdum
   158)             cdum = cdum(hoff:)
   159)             read (cdum, fmt=*) elon
   160) 
   161)             read (unit=12,fmt=hform) cdum
   162)             cdum = cdum(hoff:)
   163)             read (cdum, fmt=*) slat
   164) 
   165)             read (unit=12,fmt=hform) cdum
   166)             cdum = cdum(hoff:)
   167)             read (cdum, fmt=*) nlat
   168) 
   169)             read (unit=12,fmt=hform) cdum
   170)             cdum = cdum(hoff:)
   171)             read (cdum, fmt=*) lu_area
   172) 
   173)             read (unit=12,fmt=hform) cdum
   174)             cdum = cdum(hoff:)
   175)             read (cdum, fmt=*) yd_1st
   176) 
   177)             read (unit=12,fmt=hform) cdum
   178)             cdum = cdum(hoff:)
   179)             read (cdum, fmt=*) yd_last
   180) 
   181)             read (unit=12,fmt=hform) cdum
   182)             cdum = cdum(hoff:)
   183)             read (cdum, fmt=*) nharvest
   184) 
   185)             if (nharvest > 0 ) then
   186)                !---------------------------------------------------------------------------!
   187)                !     If nharvest is not 0, then the harvesting is not PFT-blind, read the  !
   188)                ! PFT information in the file.                                              !
   189)                !---------------------------------------------------------------------------!
   190)                read (unit=12,fmt=hform)  cdum
   191)                cdum = cdum(hoff:)
   192)                read (cdum, fmt=*) (harvest_pft(h)  ,h=1,nharvest)
   193) 
   194)                read (unit=12,fmt=hform)  cdum
   195)                cdum = cdum(hoff:)
   196)                read (cdum, fmt=*) (mindbh_1ary(h)  ,h=1,nharvest)
   197) 
   198)                read (unit=12,fmt=hform)  cdum
   199)                cdum = cdum(hoff:)
   200)                read (cdum, fmt=*) (harvprob_1ary(h)  ,h=1,nharvest)
   201) 
   202)                read (unit=12,fmt=hform)  cdum
   203)                cdum = cdum(hoff:)
   204)                read (cdum, fmt=*) (mindbh_2ary(h)  ,h=1,nharvest)
   205) 
   206)                read (unit=12,fmt=hform)  cdum
   207)                cdum = cdum(hoff:)
   208)                read (cdum, fmt=*) (harvprob_2ary(h)  ,h=1,nharvest)
   209)             else
   210)                !---------------------------------------------------------------------------!
   211)                !     No specific PFT information was given, this is likely to be a case in !
   212)                ! which the logging is based on absolute target biomass (PFT- and           !
   213)                ! DBH-blind).                                                               !
   214)                !---------------------------------------------------------------------------!
   215)                h = 0
   216)                nopftloop: do ipft=1,n_pft
   217)                   h=h+1
   218)                   harvest_pft(h)       = ipft
   219)                   mindbh_1ary(1:n_pft) = 0.
   220)                   mindbh_2ary(1:n_pft) = 0.
   221)                end do nopftloop
   222)             end if
   223)             read (unit=12,fmt=*) 
   224) 
   225)             !----- Use file_lat to compute the physical area sampled by the file. ---------!
   226)             if (lu_area == 0.) then
   227)                write (unit=*,fmt='(a)')           '---------------------------------------'
   228)                write (unit=*,fmt='(2(a,1x))')     ' - File:    ',trim(lu_name)
   229)                write (unit=*,fmt='(a,1x,es12.5)') ' - Wlon:    ',wlon
   230)                write (unit=*,fmt='(a,1x,es12.5)') ' - Elon:    ',elon
   231)                write (unit=*,fmt='(a,1x,es12.5)') ' - Slat:    ',slat
   232)                write (unit=*,fmt='(a,1x,es12.5)') ' - Nlat:    ',nlat
   233)                write (unit=*,fmt='(a,1x,es12.5)') ' - Lu_area: ',lu_area
   234)                write (unit=*,fmt='(a,1x,i6)')     ' - Yd_1st:  ',yd_1st
   235)                write (unit=*,fmt='(a,1x,i6)')     ' - Yd_last: ',yd_last
   236)                write (unit=*,fmt='(a,1x,i6)')     ' - Nharvest:',nharvest
   237)                write (unit=*,fmt='(a)')           '---------------------------------------'
   238)                call fatal_error('Land use area is zero, it doesn''t make any sense!'       &
   239)                                ,'landuse_init','landuse_init.f90')
   240)             else
   241)                lu_area_i = 1. / lu_area
   242)             end if
   243) 
   244)             !----- Determine whether this block contains the current polygon. -------------!
   245)             inside = cgrid%lon(ipy) >= wlon .and. cgrid%lon(ipy) <= elon .and.             &
   246)                      cgrid%lat(ipy) >= slat .and. cgrid%lat(ipy) <= nlat
   247) 
   248) 
   249)             !------------------------------------------------------------------------------!
   250)             !     Here we will only use the land information if the polygon centre is      !
   251)             ! inside the block of land use disturbance we are about to read.               !
   252)             !------------------------------------------------------------------------------!
   253)             if (inside) then
   254)                !----- File exists, allocate the maximum number of years. ------------------!
   255)                allocate(cpoly%clutimes(max_lu_years,cpoly%nsites))
   256) 
   257) 
   258)                !----- Copy the file information to the first site. ------------------------!
   259)                isi = 1
   260)                csite => cpoly%site(isi)
   261) 
   262) 
   263)                !----- Determine the number of disturbance years. --------------------------!
   264)                cpoly%num_landuse_years(isi) = max(yd_last,iyearz)-min(yd_1st,iyeara) + 1
   265) 
   266) 
   267)                !----- Initialise the PFT-dependent arrays. --------------------------------!
   268)                cpoly%mindbh_primary    (1:n_pft,isi) = huge_dbh
   269)                cpoly%probharv_primary  (1:n_pft,isi) = 0.
   270)                cpoly%mindbh_secondary  (1:n_pft,isi) = huge_dbh
   271)                cpoly%probharv_secondary(1:n_pft,isi) = 0.
   272) 
   273)                !----- Fill the arrays with the appropriate PFT. ---------------------------!
   274)                do ipft=1,n_pft
   275)                   harvloop: do h=1,nharvest
   276)                      if (harvest_pft(h) == ipft) then
   277)                         cpoly%mindbh_primary    (ipft,isi) = mindbh_1ary  (h)
   278)                         cpoly%probharv_primary  (ipft,isi) = harvprob_1ary(h)
   279)                         cpoly%mindbh_secondary  (ipft,isi) = mindbh_2ary  (h)
   280)                         cpoly%probharv_secondary(ipft,isi) = harvprob_2ary(h)
   281)                         exit harvloop
   282)                      end if
   283)                   end do harvloop
   284)                end do
   285) 
   286) 
   287)                !----- Padding disturbances with zero before first available lu year. ------!
   288)                iyear = 0
   289)                do yd_this = iyeara,(yd_1st-1)
   290)                   iyear = iyear + 1
   291)                   clutime => cpoly%clutimes(iyear,isi)
   292) 
   293)                   clutime%landuse_year            = yd_this
   294)                   clutime%landuse(1:num_lu_trans) = 0.0
   295)                end do
   296) 
   297)                !---- Reading the years that have data -------------------------------------!
   298)                do yd_this = yd_1st,yd_last
   299)                   iyear = iyear + 1
   300)                   clutime => cpoly%clutimes(iyear,isi)
   301) 
   302)                   read(unit=12,fmt=*) clutime%landuse_year, clutime%landuse(1:num_lu_trans)
   303)                   
   304)                   !------------------------------------------------------------------------!
   305)                   !    Here we normalise by the area, except when landuse(12) and/or       !
   306)                   ! landuse(14) are negative (a special flag to use the selective logging. !
   307)                   !------------------------------------------------------------------------!
   308)                   if ( clutime%landuse(12) > 0. ) then
   309)                      clutime%landuse(12) = lu_area_i * clutime%landuse(12)
   310)                   end if
   311)                   if ( clutime%landuse(14) > 0. ) then
   312)                      clutime%landuse(14) = lu_area_i * clutime%landuse(14) 
   313)                   end if
   314)                   clutime%landuse(16) = lu_area_i * clutime%landuse(16)
   315)                   clutime%landuse(18) = lu_area_i * clutime%landuse(18)
   316)                end do
   317) 
   318)                !----- Padding disturbances with zero after last available lu year. --------!
   319)                do yd_this = (yd_last+1),iyearz
   320)                   iyear = iyear + 1
   321)                   clutime => cpoly%clutimes(iyear,isi)
   322)                   clutime%landuse_year            = yd_this
   323)                   clutime%landuse(1:num_lu_trans) = 0.0
   324)                end do
   325) 
   326) 
   327)                !---------------------------------------------------------------------------!
   328)                !      Copy the information from the first site to the other, if they       !
   329)                ! exist.                                                                    !
   330)                !---------------------------------------------------------------------------!
   331)                siteloop: do isi = 2,cpoly%nsites
   332)                   csite => cpoly%site(isi)
   333) 
   334)                   !----- Determine the number of disturbance years. -----------------------!
   335)                   cpoly%num_landuse_years(isi) = cpoly%num_landuse_years(1)
   336) 
   337) 
   338)                   !----- Disturbances. ----------------------------------------------------!
   339)                   do iyear = 1,cpoly%num_landuse_years(isi)
   340)                      clutime   => cpoly%clutimes(iyear,isi)
   341)                      onelutime => cpoly%clutimes(iyear,1)
   342)                      clutime%landuse_year            = onelutime%landuse_year 
   343)                      clutime%landuse(1:num_lu_trans) = onelutime%landuse(1:num_lu_trans)
   344)                   end do
   345) 
   346)                end do siteloop
   347)             else
   348)                !---------------------------------------------------------------------------!
   349)                !      No GLU data for this site.  Probably water.                          !
   350)                !---------------------------------------------------------------------------!
   351)                write (unit=*,fmt='(a)') '-------------------------------------------------'
   352)                write (unit=*,fmt='(a)') ' The closest land use point is too far away...'
   353)                write (unit=*,fmt='(a)') ' - File:                        ',trim(lu_name)
   354)                write (unit=*,fmt=fffmt) ' - Polygon longitude:           ',cgrid%lon(ipy)
   355)                write (unit=*,fmt=fffmt) ' - Polygon latitude:            ',cgrid%lat(ipy)
   356)                write (unit=*,fmt=fffmt) ' - Closest LU central longitude:',llon_list(ncl)
   357)                write (unit=*,fmt=fffmt) ' - Closest LU central latitude: ',llat_list(ncl)
   358)                write (unit=*,fmt=fffmt) ' - Closest LU western edge:     ',wlon
   359)                write (unit=*,fmt=fffmt) ' - Closest LU eastern edge:     ',elon
   360)                write (unit=*,fmt=fffmt) ' - Closest LU southern edge:    ',slat
   361)                write (unit=*,fmt=fffmt) ' - Closest LU northern edge:    ',nlat
   362)                write (unit=*,fmt=esfmt) ' - Distance:                    ',file_ldist(ncl)
   363)                write (unit=*,fmt=*)     ' '
   364)                write (unit=*,fmt='(a)') ' We will assign no land use disturbance rate.'
   365)                write (unit=*,fmt='(a)') '-------------------------------------------------'
   366) 
   367)                !----- Allocate just 1 landuse year. ---------------------------------------!
   368)                allocate(cpoly%clutimes(1,cpoly%nsites))
   369) 
   370)                !----- Set the parameters in a way that no logging/ploughing will happen. --!
   371)                do isi = 1,cpoly%nsites
   372)                   cpoly%num_landuse_years(isi)                  = 1
   373)                   cpoly%mindbh_primary    (1:n_pft,isi)         = huge_dbh
   374)                   cpoly%probharv_primary  (1:n_pft,isi)         = 0.
   375)                   cpoly%mindbh_secondary  (1:n_pft,isi)         = huge_dbh
   376)                   cpoly%probharv_secondary(1:n_pft,isi)         = 0.
   377)                   cpoly%clutimes(1,isi)%landuse_year            = iyeara
   378)                   cpoly%clutimes(1,isi)%landuse(1:num_lu_trans) = 0.0
   379)                end do
   380)             end if
   381)             !------------------------------------------------------------------------------!
   382) 
   383) 
   384)             !----- Close the land use file, outside the if statement. ---------------------!
   385)             close(unit=12,status='keep')
   386)             !------------------------------------------------------------------------------!
   387) 
   388)             !----- Initialise plantation patches if plantation information is available. --!
   389)             cpoly%plantation(:) = 0
   390)             call read_plantation_fractions(cpoly,cgrid%lon(ipy),cgrid%lat(ipy),igr)
   391)          case (0)
   392)             !------------------------------------------------------------------------------!
   393)             !      Anthropogenic disturbance is not used this time, allocate only a single !
   394)             ! landuse year.                                                                !
   395)             !------------------------------------------------------------------------------!
   396)             allocate(cpoly%clutimes(1,cpoly%nsites))
   397)             !------------------------------------------------------------------------------!
   398) 
   399)             !----- Set the parameters in a way that no logging/ploughing will happen. -----!
   400)             do isi = 1,cpoly%nsites
   401)                cpoly%num_landuse_years(isi)                  = 1
   402)                cpoly%mindbh_primary    (1:n_pft,isi)         = huge_dbh
   403)                cpoly%probharv_primary  (1:n_pft,isi)         = 0.
   404)                cpoly%mindbh_secondary  (1:n_pft,isi)         = huge_dbh
   405)                cpoly%probharv_secondary(1:n_pft,isi)         = 0.
   406)                cpoly%clutimes(1,isi)%landuse_year            = iyeara
   407)                cpoly%clutimes(1,isi)%landuse(1:num_lu_trans) = 0.0
   408)             end do
   409)             !------------------------------------------------------------------------------!
   410) 
   411) 
   412)             !----- No plantations. --------------------------------------------------------!
   413)             cpoly%plantation(:) = 0
   414)             !------------------------------------------------------------------------------!
   415)          end select
   416)          
   417)       end do polyloop
   418)    end do gridloop
   419) 
   420)    return
   421) end subroutine landuse_init

   430) !==========================================================================================!
   431) !==========================================================================================!
   432) !     This subroutine reads the plantation fraction.                                       !
   433) !------------------------------------------------------------------------------------------!
   434) subroutine read_plantation_fractions(cpoly,polylon,polylat,igr)
   435)    use ed_state_vars , only : polygontype         ! ! structure
   436)    use ed_max_dims   , only : str_len             ! ! intent(in)
   437)    use disturb_coms  , only : plantation_file     & ! intent(in)
   438)                             , max_plantation_dist & ! intent(in)
   439)                             , min_plantation_frac ! ! intent(in)
   440) 
   441)    implicit none
   442)    !----- Arguments -----------------------------------------------------------------------!
   443)    type(polygontype)     , target      :: cpoly
   444)    real                  , intent(in)  :: polylon
   445)    real                  , intent(in)  :: polylat
   446)    integer               , intent(in)  :: igr
   447)    !----- Local variables -----------------------------------------------------------------!
   448)    logical                             :: exans
   449)    integer                             :: ierr
   450)    integer                             :: isi
   451)    real, dimension(:)    , allocatable :: plantlon
   452)    real, dimension(:)    , allocatable :: plantlat
   453)    real, dimension(:)    , allocatable :: plantdist
   454)    real, dimension(:)    , allocatable :: fracplant
   455)    real                                :: rdum
   456)    integer                             :: ndat
   457)    integer                             :: n
   458)    !----- Local constants. ----------------------------------------------------------------!
   459)    character(len=12)    , parameter    :: fffmt='(a,1x,f12.5)'
   460)    character(len=13)    , parameter    :: esfmt='(a,1x,es12.5)'
   461)    !----- External functions. -------------------------------------------------------------!
   462)    real                  , external    :: dist_gc
   463)    !---------------------------------------------------------------------------------------!
   464) 
   465) 
   466)    !---------------------------------------------------------------------------------------!
   467)    !     If the user left the plantation file empty, it means that they don't want to      !
   468)    ! use plantation files, skip the subroutine and don't print any warnings.               !
   469)    !---------------------------------------------------------------------------------------!
   470)    if (len_trim(plantation_file(igr)) == 0) then
   471)       return
   472)    else
   473)       !----- Check whether plantation file exists. ----------------------------------------!
   474)       inquire(file=trim(plantation_file(igr)),exist=exans)
   475)       if (.not.exans)then
   476)          write (unit=*,fmt='(a)') '-------------------------------------------------------'
   477)          write (unit=*,fmt='(a)') 'File :'//trim(plantation_file(igr))//' not found...'
   478)          write (unit=*,fmt='(a)') 'Assuming that there are no plantations'
