   916) !=========================================================
   917) subroutine libxml2f90_readin_file(nfil,ll_id)
   918)   !=======================================================
   919)   !===== READS THE FILE INTO 
   920)   !===== READINSTRING,a character(1) array
   921)   !===== INDEPENDENT FROM THE ACTUAL LINELENGTH!
   922)   !===== CUTS TABS AND UNNECESSARY BLANKS ETC.
   923)   !===== SETS THE LL_ID AND CALLS THE PARSER
   924)   !=======================================================
   925)   use libxml2f90_module
   926)   use libxml2f90_strings_module
   927)   implicit none
   928)   integer,intent(in)            :: nfil
   929)   character(*),intent(in)       :: ll_id
   930)   integer(4)                    :: eos,is,nwrite,i
   931)   character(256)                :: string !read in 256 chunks
   932)   logical(4)                    :: tblank,tnewline
   933)   integer(4),allocatable        :: tempia(:)
   934)   !.................................
   935)   
   936)   is = 0
   937) 
   938)   !===========read file into stringarry
   939)   filelines=0
   940)   if(allocated(readstring)) deallocate(readstring)
   941)   allocate(readstring(arraystep))
   942)   deallocate(readstring)
   943)   allocate(readstring(arraystep))
   944)   if(allocated(lineposa)) deallocate(lineposa)
   945)   allocate(lineposa(arraystep))
   946)   lineposa(:)=0
   947) 
   948)   nwrite=1
   949)   tblank=.false.
   950) 
   951)   do !over the lines
   952)      tnewline=.true.
   953)      do !the chunks per line
   954)         READ (nfil, FMT="(A)" , ADVANCE="NO" ,SIZE=is,IOSTAT= eos) string !read a chunk
   955)         !=======================================
   956)         !process that chunk (is=0 @ end of file!)
   957)         !=======================================
   958) 
   959)         if(eos.eq.-1) then
   960)            !the end of file leads to double reading of the last line
   961)            !-->if eos=-1 we have end of file (-2 end of line)
   962)            exit !the readin do-loop
   963)         end if
   964) 
   965)         
   966)         !test if the line starts with '<#' -> we use the line as comment
   967)         ! we also skip lines containing <?xml and <!DOCTYPE
   968)         if(tnewline) then
   969)            tnewline=.false.
   970)            !find first nonblank 
   971)            i=verify(string,' ')
   972)            if(i.gt.0) then
   973)               if(string(i:i).eq.'<'.and.(string(i+1:i+1).eq.'#'&
   974)                    &.or.+string(i+1:i+4).eq.'?XML'&
   975)                    &.or.+string(i+1:i+8).eq.'!DOCTYPE')) then
   976)                  do
   977)                     if(eos.eq.-1.or.eos.eq.-2) exit !end of line (-2) or file (-1) reached 
   978)                     READ (nfil, FMT="(A)" , ADVANCE="NO" ,SIZE=is,IOSTAT= eos) string !read a chunk
   979)                  end do
   980)                  tnewline=.true. !we skip the whole line --> the next read uses a new line
   981)                  !remember the lineposition (for the comment lines also!)
   982)                  !----------------
   983)                  filelines=filelines+1
   984)                  !keep the information about how many character(1) 
   985)                  !entries come from which line. this allows us to 
   986)                  !inform the user in which line syntax errors occur
   987)                  
   988)                  !check the size of the array
   989)                  if(size(lineposa).lt.filelines) then
   990)                     allocate(tempia(filelines-1))
   991)                     tempia(:)=lineposa(:)
   992)                     deallocate(lineposa)
   993)                     allocate(lineposa(filelines+arraystep))
   994)                     lineposa(1:filelines-1)=tempia(1:filelines-1)
   995)                     deallocate(tempia)
   996)                  end if
   997)                  lineposa(filelines)=nwrite
   998)                  !--------------
   999) 
  1000)                  cycle
  1001)               end if
  1002)            end if
  1003)         end if
  1004) 
  1005) 
  1006)         !ensure that readstring is large enough
  1007)         if(nwrite+is.gt.ubound(readstring,1)) then
  1008)             allocate(tempstringa(ubound(readstring,1)))
  1009)            tempstringa(:)=readstring(:)
  1010)            deallocate(readstring)
  1011)            allocate(readstring(ubound(tempstringa,1)+arraystep))
  1012)            readstring(1:ubound(tempstringa,1))=tempstringa(1:ubound(tempstringa,1))
  1013)            deallocate(tempstringa)
  1014)         end if
  1015) 
  1016)         !now transfer the chunk=> readstring and cut unneeded characters
  1017)         do i=1,is !if is=0 we skip
  1018)            if(iachar(string(i:i)).lt.33.or.&
  1019)                 &(trmquotes.and.iachar(string(i:i)).eq.34).or.&
  1020)                 &(trmquotes.and.iachar(string(i:i)).eq.39).or.&
  1021)                 &(trmcomma.and.iachar(string(i:i)).eq.44)) then 
  1022)               !special characters including blank or "/' if should be removed or 
  1023)               if(xmlformat.eq.1.or.xmlformat.eq.3) then !blank removal
  1024)                  if(.not.tblank) then
  1025)                     !write a blank instead if we do not have a blank at the last position 
  1026)                     readstring(nwrite)=' '
  1027)                     nwrite=nwrite+1
  1028)                     tblank=.true.
  1029)                  end if !else we drop the character
  1030)               else !no blank removal,just replace special characters
  1031)                  !write a blank instead
  1032)                  readstring(nwrite)=' '
  1033)                  nwrite=nwrite+1
  1034)               end if
  1035)            else if(string(i:i).eq.'=') then != corresponds to a blank on the left side
  1036)               !we do not need blanks left of a = if we use blankremoval:
  1037)               if(readstring(nwrite-1).eq.' '&
  1038)                    &.and.(xmlformat.eq.1.or.xmlformat.eq.3)) then 
  1039)                  !we can remove a blank to the left of a =
  1040)                  readstring(nwrite-1)=string(i:i)
  1041)                  !nwrite has to stay the same!
  1042)               else
  1043)                  !no blanks on the left side or no blank removal
  1044)                  !-> just copy the =
  1045)                  readstring(nwrite)=string(i:i)
  1046)                  tblank=.true.
  1047)                  nwrite=nwrite+1
  1048)               end if
  1049)            else if(string(i:i).eq.'>') then !> corresponds to a blank on the left side
  1050)               readstring(nwrite)=string(i:i)
  1051)               tblank=.true.
  1052)               nwrite=nwrite+1
  1053)            else if(nwrite.gt.1) then
  1054)               if(string(i:i).eq.'<'.and.readstring(nwrite-1).eq.' '&
  1055)                    &.and.(xmlformat.eq.1.or.xmlformat.eq.3)) then 
  1056)                  !we can remove a blank to the left of a <
  1057)                  readstring(nwrite-1)=string(i:i)
  1058)                  !nwrite has to stay the same!
  1059)               else    !no blank, no special character, no '='
  1060)                  readstring(nwrite)=string(i:i)
  1061)                  tblank=.false.
  1062)                  nwrite=nwrite+1
  1063)               end if
  1064)            else    !no blank, no special character, no '='
  1065)               readstring(nwrite)=string(i:i)
  1066)               tblank=.false.
  1067)               nwrite=nwrite+1
  1068)            end if
  1069)         end do
  1070)         if(eos.eq.-1) exit !end of file (-1) reached
  1071)         if(eos.eq.-2) then !end of line (-2) or file (-1) reached
  1072)            !add a blank for the linebreak if there is no blank at the end
  1073)            if(readstring(nwrite-1).ne.' ') then
  1074)               readstring(nwrite)=' '
  1075)               tblank=.true.
  1076)               nwrite=nwrite+1
  1077)            end if
  1078)            exit
  1079)         end if
  1080)      end do
  1081)      filelines=filelines+1
  1082)      !keep the information about how many character(1) 
  1083)      !entries come from which line. this allows us to 
  1084)      !inform the user in which line syntax errors occur
  1085)      
  1086)      !check the size of the array
  1087)      if(size(lineposa).lt.filelines) then
  1088)         allocate(tempia(filelines-1))
  1089)         tempia(:)=lineposa(:)
  1090)         deallocate(lineposa)
  1091)         allocate(lineposa(filelines+arraystep))
  1092)         lineposa(1:filelines-1)=tempia(1:filelines-1)
  1093)         deallocate(tempia)
  1094)      end if
  1095) 
  1096)      lineposa(filelines)=nwrite
  1097)      
  1098)      if(eos.eq.-1) exit 
  1099)   end do
  1100) 
  1101)   !cut the unneeded rest
  1102)   allocate(tempstringa(nwrite-2))!-2 because we have a blank at the end
  1103) !  tempstringa(:)=readstring(:)
  1104)   tempstringa(1:(nwrite-2))=readstring(1:(nwrite-2))
  1105)   deallocate(readstring)
  1106)   allocate(readstring(ubound(tempstringa,1)))
  1107)   readstring(:)=tempstringa(:)
  1108)   deallocate(tempstringa)
  1109)   
  1110) 
  1111)   if(ttransform_paw) call libxml2f90_transform_paw()
  1112)   call libxml2f90__set_default_ll_id(ll_id)
  1113)   call libxml2f90_parse_file()
  1114) 
  1115) end subroutine libxml2f90_readin_file
