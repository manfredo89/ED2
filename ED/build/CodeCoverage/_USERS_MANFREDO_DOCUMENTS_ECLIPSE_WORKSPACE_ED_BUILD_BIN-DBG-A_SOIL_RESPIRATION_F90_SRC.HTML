     6) !     This subroutine computes the soil respiration terms (root and heterotrophic).        !
     7) !------------------------------------------------------------------------------------------!
     8) subroutine soil_respiration(csite,ipa,mzg,ntext_soil)
     9) 
    10)    use ed_state_vars, only : sitetype                 & ! structure
    11)                            , patchtype                ! ! structure
    12)    use soil_coms    , only : soil                     & ! intent(in)
    13)                            , dslz                     & ! intent(in)
    14)                            , slz                      ! ! intent(in)
    15)    use decomp_coms  , only : k_rh_active              ! ! intent(in)
    16)    use consts_coms  , only : wdns                     & ! intent(in)
    17)                            , umols_2_kgCyr            ! ! intent(in)
    18)    use therm_lib    , only : uextcm2tl                ! ! function
    19)    use ed_misc_coms , only : dtlsm                    & ! intent(in)
    20)                            , frqsum                   ! ! intent(in)
    21)    implicit none
    22)    !----- Arguments. ----------------------------------------------------------------------!
    23)    type(sitetype)                , target     :: csite
    24)    integer                       , intent(in) :: ipa
    25)    integer                       , intent(in) :: mzg
    26)    integer       , dimension(mzg), intent(in) :: ntext_soil
    27)    !----- Local variables. ----------------------------------------------------------------!
    28)    type(patchtype)               , pointer    :: cpatch
    29)    integer                                    :: ico
    30)    integer                                    :: ipft
    31)    integer                                    :: k
    32)    integer                                    :: kroot
    33)    integer                                    :: nsoil
    34)    real                                       :: Lc
    35)    real                                       :: rel_soil_moist
    36)    real                                       :: sum_soil_energy
    37)    real                                       :: sum_soil_water
    38)    real                                       :: sum_soil_hcap
    39)    real                                       :: sum_soil_slmsts
    40)    real                                       :: sum_soil_soilcp
    41)    real                                       :: avg_soil_temp
    42)    real                                       :: avg_soil_fliq
    43)    !----- External functions. -------------------------------------------------------------!
    44)    real                          , external   :: het_resp_weight
    45)    real                          , external   :: root_resp_norm
    46)    !----- Locally saved variables. --------------------------------------------------------!
    47)    real                          , save       :: dtlsm_o_frqsum
    48)    logical                       , save       :: first_time = .true.
    49)    !---------------------------------------------------------------------------------------!
    50) 
    51) 
    52)    !----- Assign the constant scaling factor. ---------------------------------------------!
    53)    if (first_time) then
    54)       first_time     = .false.
    55)       dtlsm_o_frqsum = dtlsm / frqsum
    56)    end if
    57)    !---------------------------------------------------------------------------------------!
    58) 
    59) 
    60) 
    61)    !---------------------------------------------------------------------------------------!
    62)    !      Find the current root respiration.  This is done cohort by cohort because the    !
    63)    ! parameters may be different depending on the PFT, and also because each layer has a   !
    64)    ! different temperature.                                                                !
    65)    !---------------------------------------------------------------------------------------!
    66)    cpatch => csite%patch(ipa)
    67)    do ico = 1,cpatch%ncohorts
    68)       ipft  = cpatch%pft(ico)
    69)       kroot = cpatch%krdepth(ico)
    70) 
    71)       !------------------------------------------------------------------------------------!
    72)       !    Add "intensive" contribution of each layer, assuming that the roots are equally !
    73)       ! spread throughout the entire depth.                                                !
    74)       !------------------------------------------------------------------------------------!
    75)       cpatch%root_respiration(ico) = 0.0
    76)       do k = kroot,mzg
    77)          cpatch%root_respiration(ico) = cpatch%root_respiration(ico)                       &
    78)                                       + root_resp_norm(ipft,csite%soil_tempk(k,ipa))       &
    79)                                       * dslz(k)
    80)       end do
    81)       !------------------------------------------------------------------------------------!
    82) 
    83) 
    84)       !------------------------------------------------------------------------------------!
    85)       !      Now we make the value in umol/m2/s, by dividing by the total depth and        !
    86)       ! multiplying by the total root biomass.  The minus sign is because slz is negative. !
    87)       !------------------------------------------------------------------------------------!
    88)       cpatch%root_respiration(ico) = - cpatch%root_respiration(ico) * cpatch%broot(ico)    &
    89)                                      * cpatch%nplant(ico) / slz(kroot)
    90)       !------------------------------------------------------------------------------------!
    91) 
    92) 
    93)       !----- Add this time step to the daily mean root respiration. -----------------------!
    94)       cpatch%today_root_resp(ico)  = cpatch%today_root_resp(ico)                           &
    95)                                    + cpatch%root_respiration(ico)
    96)       !------------------------------------------------------------------------------------!
    97) 
    98) 
    99) 
   100)       !------------------------------------------------------------------------------------!
   101)       !     The following is for output only, we switch the units to kgC/plant/yr.         !
   102)       !------------------------------------------------------------------------------------!
   103)       cpatch%fmean_root_resp(ico)   = cpatch%fmean_root_resp (ico)                         &
   104)                                     + cpatch%root_respiration(ico)                         &
   105)                                     * umols_2_kgCyr * dtlsm_o_frqsum                       &
   106)                                     / cpatch%nplant          (ico)
   107)       !------------------------------------------------------------------------------------!
   108)    end do
   109)    !---------------------------------------------------------------------------------------!
   110) 
   111) 
   112) 
   113)    !---------------------------------------------------------------------------------------!
   114)    !     Integrate the soil extensive properties, plus the minimum and maximum possible    !
   115)    ! soil water content of the active layer.                                               !
   116)    !---------------------------------------------------------------------------------------!
   117)    sum_soil_energy = 0.0
   118)    sum_soil_hcap   = 0.0
   119)    sum_soil_water  = 0.0
   120)    sum_soil_slmsts = 0.0
   121)    sum_soil_soilcp = 0.0
   122)    do k = k_rh_active,mzg
   123)       nsoil = ntext_soil(k)
   124) 
   125)       !------------------------------------------------------------------------------------!
   126)       !    Convert the units so energy is in J/m2, heat capacity in J/m2/K, and water in   !
   127)       ! kg/m2.                                                                             !
   128)       !------------------------------------------------------------------------------------!
   129)       sum_soil_energy = sum_soil_energy + csite%soil_energy(k,ipa)        * dslz(k)
   130)       sum_soil_hcap   = sum_soil_hcap   + soil(nsoil)%slcpd               * dslz(k)
   131)       sum_soil_water  = sum_soil_water  + csite%soil_water (k,ipa) * wdns * dslz(k)
   132)       sum_soil_slmsts = sum_soil_slmsts + soil(nsoil)%slmsts       * wdns * dslz(k)
   133)       sum_soil_soilcp = sum_soil_soilcp + soil(nsoil)%soilcp       * wdns * dslz(k)
   134)       !------------------------------------------------------------------------------------!
   135)    end do
   136)    !---------------------------------------------------------------------------------------!
   137) 
   138) 
   139) 
   140)    !----- Find the average temperature and the relative soil moisture. --------------------!
   141)    call uextcm2tl(sum_soil_energy,sum_soil_water,sum_soil_hcap,avg_soil_temp,avg_soil_fliq)
   142)    rel_soil_moist = min( 1.0, max(0.0, ( sum_soil_water  - sum_soil_soilcp )               &
   143)                                      / ( sum_soil_slmsts - sum_soil_soilcp ) ) )
   144)    !---------------------------------------------------------------------------------------!
   145) 
   146) 
   147) 
   148)    !----- Compute soil/temperature modulation of heterotrophic respiration. ---------------!
   149)    csite%A_decomp(ipa) = het_resp_weight(avg_soil_temp,rel_soil_moist)
   150)    !---------------------------------------------------------------------------------------!
   151) 
   152) 
   153) 
   154)    !----- Compute nitrogen immobilization factor. -----------------------------------------!
   155)    call resp_f_decomp(csite,ipa, Lc)
   156)    !---------------------------------------------------------------------------------------!
   157) 
   158) 
   159) 
   160)    !----- Compute heterotrophic respiration. ----------------------------------------------!
   161)    call resp_rh(csite,ipa, Lc)
   162)    !---------------------------------------------------------------------------------------!
   163) 
   164) 
   165) 
   166)    !----- Update averaged variables. ------------------------------------------------------!
   167)    csite%today_A_decomp (ipa) = csite%today_A_decomp (ipa) + csite%A_decomp(ipa)
   168)    csite%today_Af_decomp(ipa) = csite%today_Af_decomp(ipa)                                 &
   169)                               + csite%A_decomp       (ipa) * csite%f_decomp(ipa)
   170)    !---------------------------------------------------------------------------------------!
   171) 
   172) 
   173)    !----- The output is converted to kgC/m2/yr. -------------------------------------------!
   174)    csite%fmean_rh    (ipa) = csite%fmean_rh    (ipa)                                       &
   175)                            + csite%rh          (ipa) * umols_2_kgCyr * dtlsm_o_frqsum
   176)    csite%fmean_cwd_rh(ipa) = csite%fmean_cwd_rh(ipa)                                       &
   177)                            + csite%cwd_rh      (ipa) * umols_2_kgCyr * dtlsm_o_frqsum
   178)    !---------------------------------------------------------------------------------------!
   179) 
   180)    return
   181) end subroutine soil_respiration

   193) !     This subroutine computes the Nitrogen immobilization factor.                         !
   194) !------------------------------------------------------------------------------------------!
   195) subroutine resp_f_decomp(csite,ipa,Lc)
   196) 
   197)    use ed_state_vars, only : sitetype               ! ! structure
   198)    use decomp_coms  , only : r_stsc                 & ! intent(in)
   199)                            , N_immobil_supply_scale & ! intent(in)
   200)                            , decay_rate_stsc        & ! intent(in)
   201)                            , n_decomp_lim           ! ! intent(in)
   202)    use pft_coms     , only : c2n_structural         & ! intent(in)
   203)                            , c2n_slow               ! ! intent(in)
   204) 
   205)    implicit none
   206)    !----- Arguments. ----------------------------------------------------------------------!
   207)    type(sitetype), target      :: csite
   208)    integer       , intent(in)  :: ipa
   209)    real          , intent(out) :: Lc
   210)    !----- Local variables. ----------------------------------------------------------------!
   211)    real                        :: N_immobilization_demand
   212)    !---------------------------------------------------------------------------------------!
   213) 
   214)  
   215)    if (csite%structural_soil_C(ipa) > 0.0) then
   216)       if (csite%structural_soil_L(ipa) == csite%structural_soil_C(ipa)) then
   217)          Lc = 0.049787 ! = exp(-3.0)
   218)       else
   219)          Lc = exp(-3.0 * csite%structural_soil_L(ipa)/csite%structural_soil_C(ipa))
   220)       end if
   221)    else
   222)       Lc=0.0
   223)    end if
   224)    
   225)    if (n_decomp_lim == 1) then
   226)       N_immobilization_demand = csite%A_decomp(ipa) * Lc * decay_rate_stsc                 &
   227)                               * csite%structural_soil_C(ipa)                               &
   228)                               * ((1.0 - r_stsc) / c2n_slow - 1.0 / c2n_structural)
   229)       
   230)       csite%f_decomp(ipa)     = N_immobil_supply_scale * csite%mineralized_soil_N(ipa)     &
   231)                               / ( N_immobilization_demand                                  &
   232)                                 + N_immobil_supply_scale  * csite%mineralized_soil_N(ipa))
   233)    else
   234)       !----- Option for no plant N limitation. --------------------------------------------!
   235)       csite%f_decomp(ipa)     = 1.0
   236)    end if
   237) 
   238)    return
   239) end subroutine resp_f_decomp

   249) !==========================================================================================!
   250) !     This subroutine computes the heterotrophic respiration.                              !
   251) !------------------------------------------------------------------------------------------!
   252) subroutine resp_rh(csite,ipa,Lc)
   253) 
   254)    use ed_state_vars, only : sitetype        ! ! structure
   255)    use consts_coms  , only : kgCday_2_umols  ! ! intent(in)
   256)    use decomp_coms  , only : decay_rate_stsc & ! intent(in)
   257)                            , decay_rate_fsc  & ! intent(in)
   258)                            , decay_rate_ssc  & ! intent(in)
   259)                            , r_fsc           & ! intent(in)
   260)                            , r_ssc           & ! intent(in)
   261)                            , r_stsc          & ! intent(in)
   262)                            , cwd_frac        ! ! intent(in)
   263) 
   264)    implicit none
   265)    !----- Arguments. ----------------------------------------------------------------------!
   266)    type(sitetype), target       :: csite
   267)    integer       , intent(in)   :: ipa
   268)    real          , intent(in)   :: Lc
   269)    !----- Local variables. ----------------------------------------------------------------!
   270)    real                         :: fast_C_loss
   271)    real                         :: structural_C_loss
   272)    real                         :: slow_C_loss
   273)    !---------------------------------------------------------------------------------------!
   274) 
   275) 
   276) 
   277)    !----- The following variables have units of [umol_CO2/m2/s]. --------------------------!
   278)    fast_C_loss       = kgCday_2_umols * csite%A_decomp(ipa)                                &
   279)                      * decay_rate_fsc * csite%fast_soil_C(ipa)
   280)    structural_C_loss = kgCday_2_umols * csite%A_decomp(ipa) * Lc * decay_rate_stsc         &
   281)                      * csite%structural_soil_C(ipa)* csite%f_decomp(ipa)
   282)    slow_C_loss       = kgCday_2_umols * csite%A_decomp(ipa)                                &
   283)                      * decay_rate_ssc * csite%slow_soil_C(ipa)
   284)    !---------------------------------------------------------------------------------------!
   285) 
   286)    !----- Find the heterotrophic respiration and the fraction due to CWD. -----------------!
   287)    csite%rh(ipa)     = r_fsc * fast_C_loss + r_stsc * structural_C_loss                    &
   288)                      + r_ssc * slow_C_loss
   289)    csite%cwd_rh(ipa) = cwd_frac * (r_stsc * structural_C_loss + r_ssc * slow_C_loss)
   290)    ! csite%cwd_rh(ipa) = r_stsc * structural_C_loss
   291)    !---------------------------------------------------------------------------------------!
   292) 
   293)    return
   294) end subroutine resp_rh

   304) !==========================================================================================!
   305) !     This subroutine will update the soil carbon and nitrogen pools.                      !
   306) !------------------------------------------------------------------------------------------!
   307) subroutine update_C_and_N_pools(cgrid)
   308)    
   309)    use ed_state_vars, only : edtype          & ! structure
   310)                            , polygontype     & ! structure
   311)                            , sitetype        ! ! structure
   312)    use decomp_coms  , only : decay_rate_fsc  & ! intent(in)
   313)                            , decay_rate_stsc & ! intent(in)
   314)                            , decay_rate_ssc  & ! intent(in)
   315)                            , r_stsc          ! ! intent(in)
   316)    use pft_coms     , only : c2n_slow        & ! intent(in)
   317)                            , c2n_structural  ! ! intent(in)
   318)    implicit none
   319)    !----- Arguments. ----------------------------------------------------------------------!
   320)    type(edtype)     , target   :: cgrid
   321)    !----- Local variables. ----------------------------------------------------------------!
   322)    type(polygontype), pointer  :: cpoly
   323)    type(sitetype)   , pointer  :: csite
   324)    integer                     :: ipy
   325)    integer                     :: isi
   326)    integer                     :: ipa
   327)    real                        :: Lc
   328)    real                        :: fast_C_loss
   329)    real                        :: fast_N_loss
   330)    real                        :: structural_C_loss
   331)    real                        :: structural_L_loss
   332)    real                        :: slow_C_input
   333)    real                        :: slow_C_loss
   334)    !---------------------------------------------------------------------------------------!
   335) 
   336)    polygonloop: do ipy = 1,cgrid%npolygons
   337) 
   338)       cpoly => cgrid%polygon(ipy)
   339) 
   340)       siteloop: do isi = 1,cpoly%nsites
   341)          
   342)          csite => cpoly%site(isi)
   343) 
   344)          patchloop: do ipa = 1,csite%npatches
   345) 
   346)             if (csite%structural_soil_C(ipa) > 0.0) then
   347)                if (csite%structural_soil_L(ipa) == csite%structural_soil_C(ipa)) then
   348)                   Lc = 0.049787 ! = exp(-3.0)
   349)                else
   350)                   Lc = exp( -3.0 * csite%structural_soil_L(ipa)                            &
   351)                           /  csite%structural_soil_C(ipa))
   352)                end if
   353)             else
   354)                Lc=0.0
   355)             end if
   356)       
   357)             !----- Fast pools. ------------------------------------------------------------!
   358)             fast_C_loss = csite%today_A_decomp(ipa) * decay_rate_fsc                       &
   359)                         * csite%fast_soil_C(ipa)
   360)             fast_N_loss = csite%today_A_decomp(ipa) * decay_rate_fsc                       &
   361)                         * csite%fast_soil_N(ipa)
   362) 
   363)             !----- Structural pools. ------------------------------------------------------!
   364)             structural_C_loss = csite%today_Af_decomp(ipa) * Lc * decay_rate_stsc          &
   365)                               * csite%structural_soil_C(ipa)
   366)             structural_L_loss = csite%today_Af_decomp(ipa) * Lc * decay_rate_stsc          &
   367)                               * csite%structural_soil_L(ipa)
   368) 
   369)             !----- Slow pools. ------------------------------------------------------------!
   370)             slow_C_input = (1.0 - r_stsc) * structural_C_loss
   371)             slow_C_loss  = csite%today_A_decomp(ipa) * decay_rate_ssc                      &
   372)                          * csite%slow_soil_C(ipa)
   373)             
   374)             !----- Mineralized pool. ------------------------------------------------------!
   375)             csite%mineralized_N_input = fast_N_loss + slow_C_loss / c2n_slow
   376)             csite%mineralized_N_loss  = csite%total_plant_nitrogen_uptake(ipa)             &
   377)                                       + csite%today_Af_decomp(ipa) * Lc * decay_rate_stsc  &
   378)                                       * csite%structural_soil_C(ipa)                       &
   379)                                       * ( (1.0 - r_stsc) / c2n_slow - 1.0 / c2n_structural)
   380) 
   381) 
   382)             !------------------------------------------------------------------------------!
   383)             !      All carbon fluxes have units kgC/m2/day, and we are updating on the     !
   384)             ! daily time step.                                                             !
   385)             !------------------------------------------------------------------------------!
   386)             csite%fast_soil_C(ipa)       = csite%fast_soil_C(ipa) + csite%fsc_in(ipa)      &
   387)                                          - fast_C_loss
   388)             csite%structural_soil_C(ipa) = csite%structural_soil_C(ipa)                    &
   389)                                          + csite%ssc_in(ipa) - structural_C_loss
   390)             csite%structural_soil_L(ipa) = csite%structural_soil_L(ipa)                    &
   391)                                          + csite%ssl_in(ipa) - structural_L_loss
   392)             csite%slow_soil_C(ipa)       = csite%slow_soil_C(ipa) + slow_C_input           &
   393)                                          - slow_C_loss
   394)             
   395)             !------------------------------------------------------------------------------!
   396)             !      All nitrogen fluxes have units kgN/m2/day, and we are updating on the   !
   397)             ! daily time step.                                                             !
   398)             !------------------------------------------------------------------------------!
   399)             csite%fast_soil_N(ipa)        = csite%fast_soil_N(ipa) + csite%fsn_in(ipa)     &
   400)                                           - fast_N_loss
   401)             csite%mineralized_soil_N(ipa) = csite%mineralized_soil_N(ipa)                  &
   402)                                           + csite%mineralized_N_input(ipa)                 &
   403)                                           - csite%mineralized_N_loss(ipa)
   404) 
   405)             !------------------------------------------------------------------------------!
   406)             !      Force all pools to be either zero or positive.                          !
   407)             !------------------------------------------------------------------------------!
   408)             csite%fast_soil_C(ipa)        = max(0.0,csite%fast_soil_C(ipa))
   409)             csite%structural_soil_C(ipa)  = max(0.0,csite%structural_soil_C(ipa))
   410)             csite%structural_soil_L(ipa)  = max(0.0,csite%structural_soil_L(ipa))
   411)             csite%slow_soil_C(ipa)        = max(0.0,csite%slow_soil_C(ipa))
   412)             csite%fast_soil_N(ipa)        = max(0.0,csite%fast_soil_N(ipa))
   413)             csite%mineralized_soil_N(ipa) = max(0.0,csite%mineralized_soil_N(ipa))
   414)             
   415)          end do patchloop
   416)       end do siteloop
   417)    end do polygonloop
   418)    
   419)    return
   420) end subroutine update_C_and_N_pools

   433) !==========================================================================================!
   434) !     This function determines the normalised root respiration (umol/kgC_fine_root/s) at a !
   435) ! given soil layer.                                                                        !
   436) !------------------------------------------------------------------------------------------!
   437) real function root_resp_norm(ipft,soil_temp)
   438)    use pft_coms       , only : root_respiration_factor  & ! intent(in)
   439)                              , rrf_low_temp             & ! intent(in)
   440)                              , rrf_high_temp            & ! intent(in)
   441)                              , rrf_decay_e              & ! intent(in)
   442)                              , rrf_hor                  & ! intent(in)
   443)                              , rrf_q10                  ! ! intent(in)
   444)    use farq_leuning   , only : arrhenius                & ! function
   445)                              , collatz                  ! ! function
   446)    use rk4_coms       , only : tiny_offset              ! ! intent(in)
   447)    use physiology_coms, only : iphysiol                 ! ! intent(in)
   448)    use consts_coms    , only : lnexp_min8               & ! intent(in)
   449)                              , lnexp_max8               & ! intent(in)
   450)                              , t008                     ! ! intent(in)
   451)    implicit none
   452)    !----- Arguments. ----------------------------------------------------------------------!
   453)    integer     , intent(in) :: ipft
   454)    real(kind=4), intent(in) :: soil_temp
   455)    !----- Local variables. ----------------------------------------------------------------!
   456)    real(kind=8)             :: soil_temp8
   457)    real(kind=8)             :: rrf08
   458)    real(kind=8)             :: rrf_low_temp8
   459)    real(kind=8)             :: rrf_high_temp8
   460)    real(kind=8)             :: rrf_decay_e8
   461)    real(kind=8)             :: rrf_hor8
   462)    real(kind=8)             :: rrf_q108
   463)    real(kind=8)             :: lnexplow
   464)    real(kind=8)             :: lnexphigh
   465)    real(kind=8)             :: tlow_fun
   466)    real(kind=8)             :: thigh_fun
   467)    real(kind=8)             :: rrf8
   468)    !----- External functions. -------------------------------------------------------------!
   469)    real(kind=4)             :: sngloff
   470)    !---------------------------------------------------------------------------------------!
   471) 
   472)    !----- Copy some variables to double precision temporaries. ----------------------------!
   473)    soil_temp8      = dble(soil_temp                    )
   474)    rrf08           = dble(root_respiration_factor(ipft))
   475)    rrf_low_temp8   = dble(rrf_low_temp           (ipft)) + t008
   476)    rrf_high_temp8  = dble(rrf_high_temp          (ipft)) + t008
   477)    rrf_decay_e8    = dble(rrf_decay_e            (ipft))
   478)    !---------------------------------------------------------------------------------------!
   479) 
   480) 
   481) 
   482)    !---------------------------------------------------------------------------------------!
   483)    !    Compute the functions that will control the Rrf function for low and high temper-  !
   484)    ! ature.  In order to avoid floating point exceptions, we check whether the temperature !
   485)    ! will make the exponential too large or too small.                                     !
   486)    !---------------------------------------------------------------------------------------!
   487)    !----- Low temperature. ----------------------------------------------------------------!
   488)    lnexplow  = rrf_decay_e8 * (rrf_low_temp8  - soil_temp8)
   489)    lnexplow  = max(lnexp_min8,min(lnexp_max8,lnexplow))
   490)    tlow_fun  = 1.d0 +  exp(lnexplow)
   491)    !----- High temperature. ---------------------------------------------------------------!
   492)    lnexphigh = rrf_decay_e8 * (soil_temp8 - rrf_high_temp8)
   493)    lnexphigh = max(lnexp_min8,min(lnexp_max8,lnexphigh))
   494)    thigh_fun = 1.d0 + exp(lnexphigh)
   495)    !---------------------------------------------------------------------------------------!
   496) 
   497) 
   498) 
   499)    !---------------------------------------------------------------------------------------!
   500)    !    Decide which functional form to use based on the physiology.  This is just to make !
   501)    ! it look similar to the leaf respiration respiration.                                  !
   502)    !---------------------------------------------------------------------------------------!
   503)    select case (iphysiol)
   504)    case (0,1)
   505)       rrf_hor8 = dble(rrf_hor(ipft))
   506)       rrf8     = arrhenius(soil_temp8,rrf08,rrf_hor8) / (tlow_fun * thigh_fun)
   507)    case (2,3)
   508)       rrf_q108 = dble(rrf_q10(ipft))
   509)       rrf8     = collatz(soil_temp8,rrf08,rrf_q108)   / (tlow_fun * thigh_fun)
   510)    end select
   511)    !---------------------------------------------------------------------------------------!
   512) 
   513) 
   514) 
   515)    !---------------------------------------------------------------------------------------!
   516)    !     Convert result to single precision.                                               !
   517)    !---------------------------------------------------------------------------------------!
   518)    root_resp_norm = sngloff(rrf8,tiny_offset)
   519)    !---------------------------------------------------------------------------------------!
   520) 
   521) 
   522)    return
   523) end function root_resp_norm

   534) !     This function computes the heterotrophic respiration limitation factor, which        !
   535) ! includes limitations due to temperature and soil moisture.                               !
   536) !------------------------------------------------------------------------------------------!
   537) real function het_resp_weight(soil_tempk,rel_soil_moist)
   538) 
   539)    use decomp_coms, only : resp_temperature_increase  & ! intent(in)
   540)                          , resp_opt_water             & ! intent(in)
   541)                          , resp_water_below_opt       & ! intent(in)
   542)                          , resp_water_above_opt       & ! intent(in)
   543)                          , decomp_scheme              & ! intent(in)
   544)                          , rh_lloyd_1                 & ! intent(in)
   545)                          , rh_lloyd_2                 & ! intent(in)
   546)                          , rh_lloyd_3                 & ! intent(in)
   547)                          , rh_decay_low               & ! intent(in)
   548)                          , rh_decay_high              & ! intent(in)
   549)                          , rh_low_temp                & ! intent(in)
   550)                          , rh_high_temp               & ! intent(in)
   551)                          , rh_decay_dry               & ! intent(in)
   552)                          , rh_decay_wet               & ! intent(in)
   553)                          , rh_dry_smoist              & ! intent(in)
   554)                          , rh_wet_smoist              ! ! intent(in)
   555)    use consts_coms, only : lnexp_min                  & ! intent(in)
   556)                          , lnexp_max                  ! ! intent(in)
   557) 
   558)    implicit none
   559)    !----- Arguments. ----------------------------------------------------------------------!
   560)    real(kind=4), intent(in) :: soil_tempk
   561)    real(kind=4), intent(in) :: rel_soil_moist
   562)    !----- Local variables. ----------------------------------------------------------------!
   563)    real(kind=4)             :: temperature_limitation
   564)    real(kind=4)             :: water_limitation
   565)    real(kind=4)             :: lnexplloyd
   566)    real(kind=4)             :: lnexplow
   567)    real(kind=4)             :: lnexphigh
   568)    real(kind=4)             :: tlow_fun
   569)    real(kind=4)             :: thigh_fun
   570)    real(kind=4)             :: lnexpdry
   571)    real(kind=4)             :: lnexpwet
   572)    real(kind=4)             :: smdry_fun
   573)    real(kind=4)             :: smwet_fun
   574)    !---------------------------------------------------------------------------------------!
   575) 
   576) 
   577)    !---------------------------------------------------------------------------------------!
   578)    !     Find the temperature dependence.                                                  !
   579)    !---------------------------------------------------------------------------------------!
   580)    select case(decomp_scheme)
   581)    case (0)
   582)       !----- Use original ED-2.1 exponential temperature dependence. ----------------------!
   583)       temperature_limitation = min( 1.0                                                    &
   584)                                   , exp( resp_temperature_increase * (soil_tempk-318.15)))
   585)       !------------------------------------------------------------------------------------!
   586)    case (1) 
   587)       !----- Use Lloyd and Taylor (1994) temperature dependence. --------------------------!
   588)       lnexplloyd             = rh_lloyd_1 * ( rh_lloyd_2 - 1. / (soil_tempk - rh_lloyd_3))
   589)       lnexplloyd             = max(lnexp_min,min(lnexp_max,lnexplloyd))
   590)       temperature_limitation = min( 1.0, resp_temperature_increase * exp(lnexplloyd) )
   591)       !------------------------------------------------------------------------------------!
   592)    case (2)
   593)       !------------------------------------------------------------------------------------!
   594)       !      Similar to the original ED-1.0 formulation, which is based on the CENTURY     !
   595)       ! model.  The change in the functional form is to avoid power of negative numbers,   !
   596)       ! but the coefficients were tuned to give a similar curve.                           !
   597)       !------------------------------------------------------------------------------------!
   598)       !----- Low temperature limitation. --------------------------------------------------!
   599)       lnexplow               = rh_decay_low * (rh_low_temp - soil_tempk)
   600)       lnexplow               = max(lnexp_min,min(lnexp_max,lnexplow))
   601)       tlow_fun               = 1.0 + exp(lnexplow)
   602)       !----- High temperature limitation. -------------------------------------------------!
   603)       lnexphigh              = rh_decay_high * (soil_tempk - rh_high_temp)
   604)       lnexphigh              = max(lnexp_min,min(lnexp_max,lnexphigh))
   605)       thigh_fun              = 1.0 + exp(lnexphigh)
   606)       !----- Temperature limitation is a combination of both. -----------------------------!
   607)       temperature_limitation = 1.0 / (tlow_fun * thigh_fun )
   608)       !------------------------------------------------------------------------------------!
   609)    end select
   610)    !---------------------------------------------------------------------------------------!
   611) 
   612) 
   613)    !---------------------------------------------------------------------------------------!
   614)    !     Find the limitation due to soil moisture.                                         !
   615)    !---------------------------------------------------------------------------------------!
   616)    select case (decomp_scheme)
   617)    case (0,1)
   618)       !----- ED-2.1 default, also used when decomp_scheme is 1. ---------------------------!
   619)       if (rel_soil_moist <= resp_opt_water)then
   620)          water_limitation = exp( (rel_soil_moist - resp_opt_water) * resp_water_below_opt)
   621)       else
   622)          water_limitation = exp( (resp_opt_water - rel_soil_moist) * resp_water_above_opt)
   623)       end if
   624)       !------------------------------------------------------------------------------------!
   625)    case (2)
   626)       !----- Dry soil limitation. ---------------------------------------------------------!
   627)       lnexpdry         = rh_decay_dry * (rh_dry_smoist - rel_soil_moist)
   628)       lnexpdry         = max(lnexp_min,min(lnexp_max,lnexpdry))
   629)       smdry_fun        = 1.0 + exp(lnexpdry)
   630)       !----- Wet soil limitation. ---------------------------------------------------------!
   631)       lnexpwet         = rh_decay_wet * (rel_soil_moist - rh_wet_smoist)
   632)       lnexpwet         = max(lnexp_min,min(lnexp_max,lnexpwet))
   633)       smwet_fun        = 1.0 + exp(lnexpwet)
   634)       !----- Soil moisture limitation is a combination of both. ---------------------------!
   635)       water_limitation = 1.0 / (smdry_fun * smwet_fun)
   636)       !------------------------------------------------------------------------------------!
   637)    end select
   638)    !---------------------------------------------------------------------------------------!
   639) 
   640) 
   641) 
   642)    !----- Compute the weight, which is just the combination of both. ----------------------!
   643)    het_resp_weight = temperature_limitation * water_limitation
   644)    !---------------------------------------------------------------------------------------!
   645) 
   646)    return
   647) end function het_resp_weight
