   124)    !     This function calculates the liquid saturation vapour pressure as a function of   !
   125)    ! Kelvin temperature. This expression came from MK05, equation (10).                    !
   126)    !---------------------------------------------------------------------------------------!
   127)    real(kind=4) function eslf(temp,l1funout,l2funout,ttfunout)
   128)       use consts_coms, only : t00 ! ! intent(in)
   129)       implicit none
   130)       !----- Required arguments. ----------------------------------------------------------!
   131)       real(kind=4), intent(in)            :: temp     ! Temperature                [     K]
   132)       !----- Optional arguments. ----------------------------------------------------------!
   133)       real(kind=4), intent(out), optional :: l1funout ! Function for high temperatures
   134)       real(kind=4), intent(out), optional :: ttfunout ! Interpolation function
   135)       real(kind=4), intent(out), optional :: l2funout ! Function for low temperatures
   136)       !----- Local variables. -------------------------------------------------------------!
   137)       real(kind=4)                        :: l1fun    !
   138)       real(kind=4)                        :: ttfun    !
   139)       real(kind=4)                        :: l2fun    !
   140)       real(kind=4)                        :: x        !
   141)       !------------------------------------------------------------------------------------!
   142) 
   143) 
   144) 
   145)       !------------------------------------------------------------------------------------!
   146)       !     Choose between the old and the new thermodynamics.                             !
   147)       !------------------------------------------------------------------------------------!
   148)       if (newthermo) then
   149)          !----- Updated method, using MK05 ------------------------------------------------!
   150)          l1fun = l01_10(0) + l01_10(1)/temp + l01_10(2)*log(temp) + l01_10(3) * temp
   151)          l2fun = l02_10(0) + l02_10(1)/temp + l02_10(2)*log(temp) + l02_10(3) * temp
   152)          ttfun = tanh(ttt_10(1) * (temp - ttt_10(2)))
   153)          eslf  = exp(l1fun + ttfun*l2fun)
   154)          !---------------------------------------------------------------------------------!
   155) 
   156)          if (present(l1funout)) l1funout = l1fun
   157)          if (present(l2funout)) l2funout = l2fun
   158)          if (present(ttfunout)) ttfunout = ttfun
   159)       else
   160)          !----- Original method, using polynomial fit (FWC92) -----------------------------!
   161)          x    = max(-80.,temp-t00)
   162)          eslf = cll(0) + x * (cll(1) + x * (cll(2) + x * (cll(3) + x * (cll(4)             &
   163)                        + x * (cll(5) + x * (cll(6) + x * (cll(7) + x * cll(8)) ) ) ) ) ) )
   164)          !---------------------------------------------------------------------------------!
   165) 
   166)          if (present(l1funout)) l1funout = eslf
   167)          if (present(l2funout)) l2funout = eslf
   168)          if (present(ttfunout)) ttfunout = eslf
   169)       end if
   170) 
   171)       return
   172)    end function eslf

  1135)    !     This function calculates the temperature from the liquid sat. vapour pressure.    !
  1136)    ! This is truly the inverse of eslf, which is done iteratively since it's not a simple  !
  1137)    ! function to invert. It uses Newton's method, which should take care of most cases. In !
  1138)    ! the unlikely case in which Newton's method fails, switch back to modified Regula      !
  1139)    ! Falsi method (Illinois).                                                              !
  1140)    !---------------------------------------------------------------------------------------!
  1141)    real(kind=4) function tslf(pvap)
  1142) 
  1143)       implicit none
  1144)       !----- Arguments. -------------------------------------------------------------------!
  1145)       real(kind=4), intent(in) :: pvap      ! Saturation vapour pressure           [    Pa]
  1146)       !----- Local variables for iterative method. ----------------------------------------!
  1147)       real(kind=4)             :: deriv     ! Function derivative                  [    Pa]
  1148)       real(kind=4)             :: fun       ! Function for which we seek a root.   [    Pa]
  1149)       real(kind=4)             :: funa      ! Smallest  guess function             [    Pa]
  1150)       real(kind=4)             :: funz      ! Largest   guess function             [    Pa]
  1151)       real(kind=4)             :: tempa     ! Smallest guess (or previous guess)   [    Pa]
  1152)       real(kind=4)             :: tempz     ! Largest guess (new guess in Newton)  [    Pa]
  1153)       real(kind=4)             :: delta     ! Aux. var --- 2nd guess for bisection [      ]
  1154)       integer                  :: itn       ! Iteration counter                    [   ---]
  1155)       integer                  :: itb       ! Iteration counter                    [   ---]
  1156)       logical                  :: converged ! Convergence handle                   [   ---]
  1157)       logical                  :: zside     ! Flag to check for one-sided approach [   ---]
  1158)       !------------------------------------------------------------------------------------!
  1159) 
  1160)       !----- First Guess, use Bolton (1980) equation 11, giving es in Pa and T in K -------!
  1161)       tempa = (29.65 * log(pvap) - 5016.78)/(log(pvap)-24.0854)
  1162)       funa  = eslf(tempa) - pvap
  1163)       deriv = eslfp(tempa)
  1164)       !------------------------------------------------------------------------------------!
  1165) 
  1166) 
  1167)       !----- Copy just in case it fails at the first iteration ----------------------------!
  1168)       tempz = tempa
  1169)       fun   = funa
  1170)       !------------------------------------------------------------------------------------!
  1171) 
  1172) 
  1173)       !----- Enter Newton's method loop: --------------------------------------------------!
  1174)       converged = .false.
  1175)       newloop: do itn = 1,maxfpo/6
  1176)          if (abs(deriv) < toler) exit newloop !----- Too dangerous, go with bisection -----!
  1177) 
  1178)          !----- Copy the previous guess. --------------------------------------------------!
  1179)          tempa = tempz
  1180)          funa  = fun
  1181)          !---------------------------------------------------------------------------------!
  1182) 
  1183) 
  1184)          !----- New guess, its function, and derivative evaluation. -----------------------!
  1185)          tempz = tempa - fun/deriv
  1186)          fun   = eslf(tempz) - pvap
  1187)          deriv = eslfp(tempz)
  1188)          !---------------------------------------------------------------------------------!
  1189) 
  1190) 
  1191)          !----- Check convergence. --------------------------------------------------------!
  1192)          converged = abs(tempa-tempz) < toler * tempz
  1193)          if (converged) then
  1194)             tslf = 0.5 * (tempa+tempz)
  1195)             return
  1196)          elseif (fun == 0.0) then
  1197)             !----- Converged by luck. -----------------------------------------------------!
  1198)             tslf = tempz
  1199)             return
  1200)          end if
  1201)          !---------------------------------------------------------------------------------!
  1202)       end do newloop
  1203)       !------------------------------------------------------------------------------------!
  1204) 
  1205) 
  1206) 
  1207)       !------------------------------------------------------------------------------------!
  1208)       !     If we have reached this point, then Newton's method has failed.  Use bisection !
  1209)       ! instead.  For bisection, we need two guesses whose function has opposite signs.    !
  1210)       !------------------------------------------------------------------------------------!
  1211)       if (funa * fun < 0.) then
  1212)          !----- We already have two guesses with opposite signs. --------------------------!
  1213)          funz  = fun
  1214)          zside = .true.
  1215)          !---------------------------------------------------------------------------------!
  1216)       else
  1217)          !----- Need to find the guesses with opposite signs. -----------------------------!
  1218)          if (abs(fun-funa) < 100.*toler*tempa) then
  1219)             delta = 100.*toler*tempa
  1220)          else
  1221)             delta = max(abs(funa * (tempz-tempa)/(fun-funa)),100.*toler*tempa)
  1222)          end if
  1223)          !---------------------------------------------------------------------------------!
  1224) 
  1225) 
  1226)          !---------------------------------------------------------------------------------!
  1227)          !    Try guesses on both sides of the first guess, increasingly further away      !
  1228)          ! until we spot a good guess.                                                     !
  1229)          !---------------------------------------------------------------------------------!
  1230)          tempz = tempa + delta
  1231)          zside = .false.
  1232)          zgssloop: do itb=1,maxfpo
  1233)             tempz = tempa + real((-1)**itb * (itb+3)/2) * delta
  1234)             funz  = eslf(tempz) - pvap
  1235)             zside = funa*funz < 0.
  1236)             if (zside) exit zgssloop
  1237)          end do zgssloop
  1238)          if (.not. zside) then
  1239)             write (unit=*,fmt='(a)')           '------------------------------------------'
  1240)             write (unit=*,fmt='(a)')           ' Failed finding the second guess:'
  1241)             write (unit=*,fmt='(a)')           '------------------------------------------'
  1242)             write (unit=*,fmt='(a)')           ' '
  1243)             write (unit=*,fmt='(a)')           ' Input:   '
  1244)             write (unit=*,fmt='(a,1x,es14.7)') ' + pvap  =',pvap
  1245)             write (unit=*,fmt='(a)')           ' '
  1246)             write (unit=*,fmt='(a)')           ' Output:  '
  1247)             write (unit=*,fmt='(a,1x,es14.7)') ' + tempa =',tempa
  1248)             write (unit=*,fmt='(a,1x,es14.7)') ' + funa  =',funa

  1250)             write (unit=*,fmt='(a,1x,es14.7)') ' + func  =',funz
  1251)             write (unit=*,fmt='(a)')           ' '
  1252)             write (unit=*,fmt='(a)')           '------------------------------------------'
  1253)             call fatal_error('Failed finding the second guess for regula falsi'            &
  1254)                             ,'tslf','therm_lib.f90')
  1255)          end if
  1256)       end if
  1257) 
  1258) 
  1259)       bisloop: do itb=itn,maxfpo
  1260)          tslf =  (funz*tempa-funa*tempz)/(funz-funa)
  1261) 
  1262)          !---------------------------------------------------------------------------------!
  1263)          !     Now that we updated the guess, check whether they are really close.  If so, !
  1264)          ! it converged, I can use this as my guess.                                       !
  1265)          !---------------------------------------------------------------------------------!
  1266)          converged = abs(tslf-tempa) < toler * tslf
  1267)          if (converged) exit bisloop
  1268) 
  1269)          !------ Find the new function evaluation. ----------------------------------------!
  1270)          fun       =  eslf(tslf) - pvap
  1271)          !---------------------------------------------------------------------------------!
  1272) 
  1273) 
  1274)          !------ Define the new interval based on the intermediate value theorem. ---------!
  1275)          if (fun*funa < 0. ) then
  1276)             tempz = tslf
  1277)             funz  = fun
  1278)             !----- If we are updating zside again, modify aside (Illinois method). --------!
  1279)             if (zside) funa=funa * 0.5
  1280)             !----- We have just updated zside, so we set zside to true. -------------------!
  1281)             zside = .true.
  1282)          else
  1283)             tempa = tslf
  1284)             funa   = fun
  1285)             !----- If we are updating aside again, modify zside (Illinois method). --------!
  1286)             if (.not. zside) funz=funz * 0.5
  1287)             !----- We have just updated aside, so we set zside to false. ------------------!
  1288)             zside = .false.
  1289)          end if
  1290)       end do bisloop
  1291) 
  1292)       if (.not.converged) then
  1293)          write (unit=*,fmt='(a)')           '------------------------------------------'
  1294)          write (unit=*,fmt='(a)')           ' Failed finding the solution:'
  1295)          write (unit=*,fmt='(a)')           '------------------------------------------'
  1296)          write (unit=*,fmt='(a)')           ' '
  1297)          write (unit=*,fmt='(a)')           ' Input:   '
  1298)          write (unit=*,fmt='(a,1x,es14.7)') ' + pvap  =',pvap
  1299)          write (unit=*,fmt='(a)')           ' '
  1300)          write (unit=*,fmt='(a)')           ' Output:  '
  1301)          write (unit=*,fmt='(a,1x,es14.7)') ' + tempa =',tempa
  1302)          write (unit=*,fmt='(a,1x,es14.7)') ' + funa  =',funa
  1303)          write (unit=*,fmt='(a,1x,es14.7)') ' + tempz =',tempz
  1304)          write (unit=*,fmt='(a,1x,es14.7)') ' + func  =',funz
  1305)          write (unit=*,fmt='(a)')           ' '
  1306)          write (unit=*,fmt='(a)')           '------------------------------------------'
  1307)          call fatal_error('Temperature didn''t converge, we give up!!!'                    &
  1308)                          ,'tslf','therm_lib.f90')

  2903)       real(kind=4), intent(in) :: temp  ! Temperature                             [      K]
  2904)       real(kind=4), intent(in) :: rliq  ! Liquid mixing ratio                     [  kg/kg]
  2905)       real(kind=4), intent(in) :: rice  ! Ice mixing ratio                        [  kg/kg]
  2906)       !----- Local variables --------------------------------------------------------------!
  2907)       real(kind=4)             :: hh    ! Enthalpy associated with sensible heat  [   J/kg]
  2908)       real(kind=4)             :: qq    ! Enthalpy associated with latent heat    [   J/kg]
  2909)       !------------------------------------------------------------------------------------!
  2910) 
  2911) 
  2912)       !----- Find the sensible heat enthalpy (assuming dry air). --------------------------!
  2913)       hh = cpdry * temp
  2914)       !------------------------------------------------------------------------------------!
  2915) 
  2916) 
  2917)       !------------------------------------------------------------------------------------!
  2918)       !      Find the latent heat enthalpy.  If using the old thermodynamics, we use the   !
  2919)       ! latent heat at T = T3ple, otherwise we use the temperature-dependent one.          !
  2920)       !------------------------------------------------------------------------------------!
  2921)       if (newthermo) then
  2922)          qq = alvl(temp) * rliq + alvi(temp) * rice
  2923)       else
  2924)          qq = alvl3 * rliq + alvi3 * rice
  2925)       end if

  2927) 
  2928) 
  2929)       !------------------------------------------------------------------------------------!

  2931)       ! the exponential to a linear function, nor do we impose temperature above the thre- !
  2932)       ! shold from Tripoli and Cotton (1981).                                              !
  2933)       !------------------------------------------------------------------------------------!
  2934)       if (newthermo) then
  2935)          !----- Decide how to compute, based on temperature. ------------------------------!
  2936)          theta_iceliq = hh * exp(-qq / hh) / exner
  2937)          !---------------------------------------------------------------------------------!
  2938)       else
  2939)          !----- Decide how to compute, based on temperature. ------------------------------!

  2941)             theta_iceliq = hh * hh / (exner * ( hh + qq))
  2942)          else
  2943)             theta_iceliq = hh * htripoli / (exner * ( htripoli + qq))
  2944)          end if
  2945)          !---------------------------------------------------------------------------------!
  2946)       end if

  2972)                             , htripolii  & ! intent(in)

  2978)       real(kind=4), intent(in)           :: thil       ! Ice liquid pot. temp.     [     K]

  2980)       real(kind=4), intent(in)           :: temp       ! Temperature               [     K]

  2982)       real(kind=4), intent(in), optional :: ricein     ! Ice mixing ratio          [ kg/kg]
  2983)       !----- Local variables --------------------------------------------------------------!
  2984)       real(kind=4)                       :: rice       ! Ice mixing ratio or 0.    [ kg/kg]
  2985)       real(kind=4)                       :: ldrst      ! L  d(rs)/dT  T          [  J/kg]
  2986)       real(kind=4)                       :: rdlt       ! r  d(L)/dT  T           [  J/kg]
  2987)       real(kind=4)                       :: hh         ! Sensible heat enthalpy    [  J/kg]
  2988)       real(kind=4)                       :: qq         ! Latent heat enthalpy      [  J/kg]
  2989)       logical                            :: thereisice ! Is ice present            [   ---]
  2990)       !------------------------------------------------------------------------------------!
  2991) 
  2992) 
  2993)       !------------------------------------------------------------------------------------!
  2994)       !   Check whether we should consider ice thermodynamics or not.                      !
  2995)       !------------------------------------------------------------------------------------!
  2996)       thereisice = present(ricein)
  2997)       if (thereisice) then
  2998)          rice = ricein
  2999)       else
  3000)          rice = 0.
  3001)       end if
  3002)       !------------------------------------------------------------------------------------!
  3003) 
  3004) 
  3005)       !------------------------------------------------------------------------------------!
  3006)       !     Check whether the current state has condensed water.                           !
  3007)       !------------------------------------------------------------------------------------!
  3008)       if (rliq+rice == 0.) then
  3009)          !----- No condensation, so dthetail_dt is a constant. ----------------------------!
  3010)          dthetail_dt = thil/temp
  3011)          return
  3012)          !---------------------------------------------------------------------------------!
  3013)       else
  3014)          !---------------------------------------------------------------------------------!
  3015)          !     Condensation exists.  Compute some auxiliary variables.                     !
  3016)          !---------------------------------------------------------------------------------!
  3017) 
  3018) 
  3019)          !---- Sensible heat enthalpy. ----------------------------------------------------!
  3020)          hh = cpdry * temp
  3021)          !---------------------------------------------------------------------------------!
  3022) 
  3023) 
  3024)          !---------------------------------------------------------------------------------!
  3025)          !      Find the latent heat enthalpy.  If using the old thermodynamics, we use    !
  3026)          ! the latent heat at T = T3ple, otherwise we use the temperature-dependent one.   !
  3027)          ! The term r  d(L)/dT  T is computed only when we use the new thermodynamics.   !
  3028)          !---------------------------------------------------------------------------------!
  3029)          if (newthermo) then
  3030)             qq   = alvl(temp) * rliq + alvi(temp) * rice
  3031)             rdlt = (dcpvl * rliq + dcpvi * rice ) * temp
  3032)          else
  3033)             qq   = alvl3 * rliq + alvi3 * rice
  3034)             rdlt = 0.0
  3035)          end if

  3037) 
  3038) 
  3039) 
  3040)          !---------------------------------------------------------------------------------!

  3042)          ! sublimation latent heat, depending on the temperature and whether we are consi- !
  3043)          ! dering ice or not.  We still need to check whether latent heat is a function of !
  3044)          ! temperature or not.  Also, if condensation mixing ratio is constant, then this  !
  3045)          ! term will be always zero.                                                       !
  3046)          !---------------------------------------------------------------------------------!
  3047)          if (condconst) then

  3049)          elseif (thereisice .and. temp < t3ple) then

  3051)                ldrst = alvi3 * rsifp(pres,temp) * temp
  3052)             else
  3053)                ldrst = alvi(temp) * rsifp(pres,temp) * temp
  3054)             end if
  3055)          else
  3056)             if (newthermo) then
  3057)                ldrst = alvl3 * rslfp(pres,temp) * temp
  3058)             else
  3059)                ldrst = alvl(temp) * rslfp(pres,temp) * temp

  3061)          end if
  3062)          !---------------------------------------------------------------------------------!
  3063)       end if
  3064)       !------------------------------------------------------------------------------------!
  3065)

  3070)       !------------------------------------------------------------------------------------!
  3071)       if (newthermo) then

  3073)       else
  3074)          !----- Decide how to compute, based on temperature. ------------------------------!
  3075)          if (temp > ttripoli) then
  3076)             dthetail_dt = thil * ( 1. + (ldrst + qq) / (hh+qq) ) / temp
  3077)          else
  3078)             dthetail_dt = thil * ( 1. + ldrst / (htripoli + alvl3 * rliq) ) / temp
  3079)          end if
  3080)          !---------------------------------------------------------------------------------!
  3081)       end if

  3083) 
  3084)       return

  3204)          tempz = tempa - fun/deriv
  3205)          fun   = theta_iceliq(exner,tempz,rliq,rice)
  3206)          deriv = dthetail_dt(.true.,fun,pres,tempz,rliq,rice)
  3207)          fun   = fun - thil
  3208)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3209)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3210)          !write(unit=89,fmt='(a,1x,i5,1x,a,1x,l1,1x,a,1x,f11.4,2(1x,a,1x,es12.5))')         &
  3211)          !   'itn=',itn,'bisection=',.false.,'tempz=',tempz-t00                             &
  3212)          !           ,'fun=',fun,'deriv=',deriv
  3213)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3214)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3215) 
  3216)          converged = abs(tempa-tempz) < toler*tempz
  3217)          !----- Converged, happy with that, return the average b/w the 2 previous guesses -!
  3218)          if (fun == 0.) then
  3219)             thil2temp = tempz
  3220)             converged = .true.
  3221)             return
  3222)          elseif(converged) then
  3223)             thil2temp = 0.5 * (tempa+tempz)
  3224)             return
  3225)          end if
  3226)       end do newloop
  3227)       !------------------------------------------------------------------------------------!
  3228) 
  3229) 
  3230)       !------------------------------------------------------------------------------------!
  3231)       !     If we have reached this point then Newton's method failed.  Use bisection      !
  3232)       ! instead.  For bisection, We need two guesses whose function evaluations have       !
  3233)       ! opposite sign.                                                                     !
  3234)       !------------------------------------------------------------------------------------!
  3235)       if (funa * fun < 0.) then
  3236)          !----- Guesses have opposite sign. -----------------------------------------------!
  3237)          funz  = fun
  3238)          zside = .true.
  3239)       else
  3240)          if (abs(fun-funa) < toler*tempa) then
  3241)             delta = 100.*toler*tempa
  3242)          else
  3243)             delta = max(abs(funa * (tempz-tempa)/(fun-funa)),100.*toler*tempa)
  3244)          end if
  3245)          tempz = tempa + delta
  3246)          zside = .false.
  3247)          zgssloop: do itb=1,maxfpo
  3248)             tempz = tempa + real((-1)**itb * (itb+3)/2) * delta
  3249)             funz  = theta_iceliq(exner,tempz,rliq,rice) - thil
  3250)             zside = funa*funz < 0.0
  3251)             if (zside) exit zgssloop
  3252)          end do zgssloop
  3253)          if (.not. zside) then
  3254)             write (unit=*,fmt='(a)') ' No second guess for you...'
  3255)             write (unit=*,fmt='(2(a,1x,i14,1x))')    'itn  =',itn  ,'itb =',itb
  3256)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'thil =',thil ,'t1st=',t1stguess
  3257)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'exner=',exner,'pres=',pres
  3258)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'rliq =',rliq ,'rice=',rice
  3259)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'tempa=',tempa,'funa=',funa
  3260)             write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'tempz=',tempz,'funz=',funz
  3261)             write (unit=*,fmt='(1(a,1x,es14.7,1x))') 'delta=',delta
  3262)             call fatal_error('Failed finding the second guess for regula falsi'            &
  3263)                           ,'thil2temp','therm_lib.f90')
  3264)          end if
  3265)       end if
  3266) 
  3267) 
  3268)       bisloop: do itb=itn,maxfpo
  3269)          thil2temp =  (funz*tempa-funa*tempz)/(funz-funa)
  3270) 
  3271)          !---------------------------------------------------------------------------------!
  3272)          !     Now that we updated the guess, check whether they are really close. If so,  !
  3273)          ! it converged, I can use this as my guess.                                       !
  3274)          !---------------------------------------------------------------------------------!
  3275)          converged = abs(thil2temp-tempa)< toler*thil2temp
  3276)          if (converged) exit bisloop
  3277) 
  3278)          !------ Finding the new function -------------------------------------------------!
  3279)          fun  = theta_iceliq(exner,tempz,rliq,rice) - thil
  3280) 
  3281)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3282)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3283)          !write(unit=89,fmt='(a,1x,i5,1x,a,1x,l1,1x,6(1x,a,1x,f11.4))')         &
  3284)          !   'itn=',itb,'bisection=',.true.                                                 &
  3285)          !           ,'temp=',thil2temp-t00,'tempa=',tempa-t00,'tempz=',tempz-t00           &
  3286)          !           ,'fun=',fun,'funa=',funa,'funz=',funz
  3287)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3288)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3289) 
  3290)          !------ Defining my new interval based on the intermediate value theorem. --------!
  3291)          if (fun*funa < 0. ) then
  3292)             tempz = thil2temp
  3293)             funz  = fun
  3294)             !----- If we are updating zside again, modify aside (Illinois method) ---------!
  3295)             if (zside) funa=funa * 0.5
  3296)             !----- We just updated zside, setting zside to true. --------------------------!
  3297)             zside = .true.
  3298)          else
  3299)             tempa = thil2temp
  3300)             funa   = fun
  3301)             !----- If we are updating aside again, modify aside (Illinois method) ---------!
  3302)             if (.not. zside) funz=funz * 0.5
  3303)             !----- We just updated aside, setting aside to true. --------------------------!
  3304)             zside = .false.
  3305)          end if
  3306)       end do bisloop
  3307) 
  3308) 
  3309)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3310)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3311)       !write(unit=89,fmt='(60a1)') ('-',itn=1,60)
  3312)       !write(unit=89,fmt='(a)') ' '
  3313)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3314)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3315) 
  3316)       if (.not.converged) then
  3317)          write (unit=*,fmt='(a)') '-------------------------------------------------------'
  3318)          write (unit=*,fmt='(a)') ' Temperature finding didn''t converge!!!'
  3319)          write (unit=*,fmt='(a,1x,i5,1x,a)') ' I gave up, after',maxfpo,'iterations...'
  3320)          write (unit=*,fmt='(a)') ' '
  3321)          write (unit=*,fmt='(a)') ' Input values.'
  3322)          write (unit=*,fmt='(a)') ' '
  3323)          write (unit=*,fmt='(a,1x,f12.4)' ) 'theta_il        [     K] =',thil
  3324)          write (unit=*,fmt='(a,1x,f12.4)' ) 'Exner           [J/kg/K] =',exner
  3325)          write (unit=*,fmt='(a,1x,f12.4)' ) 'rliq            [  g/kg] =',1000.*rliq
  3326)          write (unit=*,fmt='(a,1x,f12.4)' ) 'rice            [  g/kg] =',1000.*rice
  3327)          write (unit=*,fmt='(a,1x,f12.4)' ) 't1stguess       [    C] =',t1stguess-t00
  3328)          write (unit=*,fmt='(a)') ' '
  3329)          write (unit=*,fmt='(a)') ' Last iteration outcome (downdraft values).'
  3330)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tempa           [    C] =',tempa-t00
  3331)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tempz           [    C] =',tempz-t00
  3332)          write (unit=*,fmt='(a,1x,f12.4)' ) 'fun             [     K] =',fun
  3333)          write (unit=*,fmt='(a,1x,f12.4)' ) 'funa            [     K] =',funa
  3334)          write (unit=*,fmt='(a,1x,f12.4)' ) 'funz            [     K] =',funz
  3335)          write (unit=*,fmt='(a,1x,f12.4)' ) 'deriv           [  ----] =',deriv
  3336)          write (unit=*,fmt='(a,1x,es12.4)') 'toler           [  ----] =',toler
  3337)          write (unit=*,fmt='(a,1x,es12.4)') 'error           [  ----] ='                   &
  3338)                                                             ,abs(thil2temp-tempa)/thil2temp
  3339)          write (unit=*,fmt='(a,1x,f12.4)' ) 'thil2temp       [     K] =',thil2temp
  3340) 
  3341)          call fatal_error('Temperature didn''t converge, giving up!!!'                     &
  3342)                        ,'thil2temp','therm_lib.f90')
  3343)       end if
  3344) 
  3345)       return
  3346)    end function thil2temp

  3357)    !     This function computes the partial derivative of temperature as a function of the !
  3358)    ! saturation mixing ratio [kg/kg], keeping pressure constant.  This is based on the     !
  3359)    ! ice-liquid potential temperature equation.                                            !
  3360)    !---------------------------------------------------------------------------------------!
  3361)    real(kind=4) function dtempdrs(exner,thil,temp,rliq,rice,rconmin)
  3362)       use consts_coms, only : alvl3      & ! intent(in)
  3363)                             , alvi3      & ! intent(in)
  3364)                             , dcpvl      & ! intent(in)
  3365)                             , dcpvi      & ! intent(in)
  3366)                             , cpdry      & ! intent(in)
  3367)                             , cpdryi     & ! intent(in)
  3368)                             , ttripoli   & ! intent(in)
  3369)                             , htripolii  ! ! intent(in)
  3370) 
  3371)       implicit none
  3372)       !----- Arguments --------------------------------------------------------------------!
  3373)       real(kind=4), intent(in) :: exner   ! Exner function                        [ J/kg/K]
  3374)       real(kind=4), intent(in) :: thil    ! Ice-liquid potential temperature (*)  [      K]
  3375)       real(kind=4), intent(in) :: temp    ! Temperature                           [      K]
  3376)       real(kind=4), intent(in) :: rliq    ! Liquid mixing ratio                   [  kg/kg]
  3377)       real(kind=4), intent(in) :: rice    ! Ice mixing ratio                      [  kg/kg]
  3378)       real(kind=4), intent(in) :: rconmin ! Min. non-zero condensate mix. ratio   [  kg/kg]
  3379)       !------------------------------------------------------------------------------------!
  3380)       ! (*) Thil is not used in this formulation but it may be used should you opt by      !
  3381)       !     other ways to compute theta_il, so don't remove this argument.                 !
  3382)       !------------------------------------------------------------------------------------!
  3383)       !----- Local variables --------------------------------------------------------------!
  3384)       real(kind=4)             :: qq      ! Enthalpy -- latent heat               [   J/kg]
  3385)       real(kind=4)             :: qpt     ! d(qq)/dT * T                          [   J/kg]
  3386)       real(kind=4)             :: hh      ! Enthalpy -- sensible heat             [   J/kg]
  3387)       real(kind=4)             :: rcon    ! Condensate mixing ratio               [  kg/kg]
  3388)       real(kind=4)             :: til     ! Ice-liquid temperature                [      K]
  3389)       !------------------------------------------------------------------------------------!
  3390) 
  3391) 
  3392)       !----- Find the total hydrometeor mixing ratio. -------------------------------------!
  3393)       rcon  = rliq+rice
  3394)       !------------------------------------------------------------------------------------!
  3395) 
  3396) 
  3397)       !------------------------------------------------------------------------------------!
  3398)       !      If the amount of condensate is negligible, temperature does not depend on     !
  3399)       ! saturation mixing ratio.                                                           !
  3400)       !------------------------------------------------------------------------------------!
  3401)       if (rcon < rconmin) then
  3402)          dtempdrs = 0.
  3403)       else
  3404) 
  3405)          !---------------------------------------------------------------------------------!
  3406)          !     Find the enthalpy associated with latent heat and its derivative            !
  3407)          ! correction.  This is dependent on the thermodynamics used.                      !
  3408)          !---------------------------------------------------------------------------------!
  3409)          if (newthermo) then
  3410)             qq  = alvl(temp) * rliq + alvi(temp) * rice
  3411)             qpt = dcpvl * rliq + dcpvi * rice
  3412)          else
  3413)             qq  = alvl3 * rliq + alvi3 * rice
  3414)             qpt = 0.0
  3415)          end if
  3416)          !---------------------------------------------------------------------------------!
  3417) 
  3418) 
  3419)          !----- Find the enthalpy associated with sensible heat. --------------------------!
  3420)          hh = cpdry * temp
  3421)          !---------------------------------------------------------------------------------!
  3422) 
  3423) 
  3424)          !---------------------------------------------------------------------------------!
  3425)          !    Decide how to compute, based on the thermodynamics method.                   !
  3426)          !---------------------------------------------------------------------------------!
  3427)          if (newthermo) then
  3428)             dtempdrs = - temp * qq / ( rcon * (hh + qq - qpt) )
  3429)          else
  3430)             til   = cpdryi * thil * exner
  3431)             !----- Decide how to compute, based on temperature. ---------------------------!
  3432)             if (temp > ttripoli) then
  3433)                dtempdrs = - til * qq / ( rcon * cpdry * (2.*temp-til))
  3434)             else
  3435)                dtempdrs = - til * qq * htripolii / rcon
  3436)             end if
  3437)             !------------------------------------------------------------------------------!
  3438)          end if
  3439)          !---------------------------------------------------------------------------------!
  3440)       end if
  3441)       !------------------------------------------------------------------------------------!
  3442) 
  3443)       return
  3444)    end function dtempdrs

  3510)    !    This function computes the derivative of ice-vapour equivalent potential tempera-  !
  3511)    ! ture, based on the expression used to compute the ice-vapour equivalent potential     !
  3512)    ! temperature (function thetaeiv).                                                      !
  3513)    !                                                                                       !
  3514)    !    IMPORTANT!!! This CANNOT BE USED to compute d(Thetae_ivs)/dT, because here         !
  3515)    !                 we assume that T(LCL) and saturation mixing ratio are known and       !
  3516)    !                 constants, and that the LCL pressure (actually the saturation  vapour !
  3517)    !                 pressure at the LCL) is a function of temperature.  In case you want  !
  3518)    !                 d(Thetae_ivs)/dT, use the dthetaeivs_dt function instead.             !
  3519)    !---------------------------------------------------------------------------------------!
  3520)    real(kind=4) function dthetaeiv_dtlcl(theiv,tlcl,rtot,eslcl,useice)
  3521)       use consts_coms, only : rocp       & ! intent(in)
  3522)                             , cpdry      & ! intent(in)
  3523)                             , dcpvl      ! ! intent(in)
  3524)       implicit none
  3525)       !----- Required arguments. ----------------------------------------------------------!
  3526)       real(kind=4), intent(in)           :: theiv    ! Ice-vap. equiv. pot. temp. [      K]
  3527)       real(kind=4), intent(in)           :: tlcl     ! LCL temperature            [      K]
  3528)       real(kind=4), intent(in)           :: rtot     ! Total mixing ratio         [  kg/kg]
  3529)       real(kind=4), intent(in)           :: eslcl    ! LCL sat. vapour pressure   [     Pa]
  3530)       !----- Optional arguments. ----------------------------------------------------------!
  3531)       logical     , intent(in), optional :: useice   ! Flag for considering ice   [    T|F]
  3532)       !----- Local variables --------------------------------------------------------------!
  3533)       real(kind=4)                       :: desdtlcl ! Sat. vapour pres. deriv.   [   Pa/K]
  3534)       real(kind=4)                       :: esterm   ! es(TLC) term               [   ----]
  3535)       real(kind=4)                       :: hhlcl    ! Enthalpy -- sensible       [   J/kg]
  3536)       real(kind=4)                       :: qqlcl    ! Enthalpy -- latent         [   J/kg]
  3537)       real(kind=4)                       :: qptlcl   ! Latent deriv. * T_LCL      [   J/kg]
  3538)       !------------------------------------------------------------------------------------!
  3539) 
  3540) 
  3541) 
  3542)       !----- Find the derivative of rs with temperature. ----------------------------------!
  3543)       if (present(useice)) then
  3544)          desdtlcl = eslifp(tlcl,useice)
  3545)       else
  3546)          desdtlcl = eslifp(tlcl)
  3547)       end if
  3548)       !------------------------------------------------------------------------------------!
  3549) 
  3550) 
  3551) 
  3552)       !------------------------------------------------------------------------------------!
  3553)       !     Saturation term.                                                               !
  3554)       !------------------------------------------------------------------------------------!
  3555)       esterm = rocp * tlcl * desdtlcl / eslcl
  3556)       !------------------------------------------------------------------------------------!
  3557) 
  3558) 
  3559)       !------------------------------------------------------------------------------------!
  3560)       !     Find the enthalpy terms.                                                       !
  3561)       !------------------------------------------------------------------------------------!
  3562)       hhlcl  = cpdry * tlcl
  3563)       qqlcl  = alvl(tlcl) * rtot
  3564)       qptlcl = dcpvl * rtot * tlcl
  3565)       !------------------------------------------------------------------------------------!
  3566) 
  3567) 
  3568) 
  3569)       !------------------------------------------------------------------------------------!
  3570)       !      Derivative.                                                                   !
  3571)       !------------------------------------------------------------------------------------!
  3572)       dthetaeiv_dtlcl = theiv / tlcl * (1. - esterm - (qqlcl - qptlcl) / hhlcl)
  3573)       !------------------------------------------------------------------------------------!
  3574) 
  3575)       return
  3576)    end function dthetaeiv_dtlcl

  3603)       use consts_coms, only : cpdry    ! ! intent(in)
  3604)       implicit none
  3605)       !----- Arguments. -------------------------------------------------------------------!
  3606)       real(kind=4), intent(in) :: thil  ! Theta_il, ice-liquid water pot. temp.    [     K]
  3607)       real(kind=4), intent(in) :: temp  ! Temperature                              [     K]
  3608)       real(kind=4), intent(in) :: rsat  ! Saturation water vapour mixing ratio     [ kg/kg]
  3609)       real(kind=4), intent(in) :: rliq  ! Liquid water mixing ratio                [ kg/kg]
  3610)       real(kind=4), intent(in) :: rice  ! Ice mixing ratio                         [ kg/kg]
  3611)       !----- Local variables --------------------------------------------------------------!
  3612)       real(kind=4)             :: rtots ! Saturated mixing ratio                   [     K]
  3613)       !------------------------------------------------------------------------------------!
  3614) 
  3615) 
  3616)       !------ Find the total saturation mixing ratio. -------------------------------------!
  3617)       rtots = rsat+rliq+rice
  3618)       !------------------------------------------------------------------------------------!
  3619) 
  3620) 
  3621)       !------ Find the saturation equivalent potential temperature. -----------------------!
  3622)       thetaeivs = thil * exp ( alvl(temp) * rtots / (cpdry * temp))
  3623)       !------------------------------------------------------------------------------------!
  3624) 
  3625)       return
  3626)    end function thetaeivs

  3637)    !    This function computes the derivative of saturation ice-vapour equivalent          !
  3638)    ! potential temperature, based on the expression used to compute the saturation         !
  3639)    ! ice-vapour equivalent potential temperature (function thetaeivs).                     !
  3640)    !                                                                                       !
  3641)    !    IMPORTANT!!! This CANNOT BE USED to compute d(Thetae_iv)/d(T_LCL), because here    !
  3642)    !                 we assume that temperature and pressure are known and constants, and  !
  3643)    !                 that the mixing ratio is a function of temperature. In case you want  !
  3644)    !                 d(Thetae_iv)/d(T_LCL), use the dthetaeiv_dtlcl function instead.      !
  3645)    !---------------------------------------------------------------------------------------!
  3646)    real(kind=4) function dthetaeivs_dt(theivs,temp,pres,rsat,useice)
  3647)       use consts_coms, only : cpdry     & ! intent(in)
  3648)                             , dcpvl     ! ! intent(in)
  3649)       implicit none
  3650)       !----- Required arguments. ----------------------------------------------------------!
  3651)       real(kind=4), intent(in)           :: theivs ! Sat. ice-vap. eq. pot. temp. [      K]
  3652)       real(kind=4), intent(in)           :: temp   ! Temperature                  [      K]
  3653)       real(kind=4), intent(in)           :: pres   ! Pressure                     [     Pa]
  3654)       real(kind=4), intent(in)           :: rsat   ! Saturation mixing ratio      [  kg/kg]
  3655)       !----- Optional arguments. ----------------------------------------------------------!
  3656)       logical     , intent(in), optional :: useice ! Flag for considering ice     [    T|F]
  3657)       !----- Local variables --------------------------------------------------------------!
  3658)       real(kind=4)                       :: drsdt  ! Sat. mixing ratio derivative [kg/kg/K]
  3659)       real(kind=4)                       :: hh     ! Enthalpy -- sensible         [   J/kg]
  3660)       real(kind=4)                       :: qqaux  ! Enthalpy -- sensible         [   J/kg]
  3661)       !------------------------------------------------------------------------------------!
  3662) 
  3663) 
  3664)       !----- Find the derivative of rs with temperature and associated term. --------------!
  3665)       if (present(useice)) then
  3666)          drsdt = rslifp(pres,temp,useice)
  3667)       else
  3668)          drsdt = rslifp(pres,temp)
  3669)       end if
  3670)       !------------------------------------------------------------------------------------!
  3671) 
  3672) 
  3673) 
  3674)       !------------------------------------------------------------------------------------!
  3675)       !      Find the enthalpy terms.                                                      !
  3676)       !------------------------------------------------------------------------------------!
  3677)       hh    = cpdry * temp
  3678)       qqaux = alvl(temp) * (drsdt * temp - rsat) + dcpvl * rsat * temp
  3679)       !------------------------------------------------------------------------------------!
  3680) 
  3681) 
  3682)       !----- Find the derivative.  Depending on the temperature, use different eqn. -------!
  3683)       dthetaeivs_dt = theivs / temp * ( 1. + qqaux / hh )
  3684)       !------------------------------------------------------------------------------------!
  3685) 
  3686)       return
  3687)    end function dthetaeivs_dt

  3698)    !    This function finds the ice-liquid potential temperature from the ice-vapour equi- !
  3699)    ! valent potential temperature.                                                         !
  3700)    ! Important remarks:                                                                    !
  3701)    ! 1. If you don't want to use ice thermodynamics, simply force useice to be .false.     !
  3702)    !    Otherwise, the model will decide based on the LEVEL given by the user from their   !
  3703)    !    RAMSIN.                                                                            !
  3704)    ! 2. If rtot < rsat, then this will convert theta_e into theta, which can be thought as !
  3705)    !    a particular case.                                                                 !
  3706)    !---------------------------------------------------------------------------------------!
  3707)    real(kind=4) function thetaeiv2thil(theiv,pres,rtot,useice)
  3708)       use consts_coms, only : ep       & ! intent(in)
  3709)                             , cpdry    & ! intent(in)
  3710)                             , p00      & ! intent(in)
  3711)                             , rocp     & ! intent(in)
  3712)                             , t3ple    & ! intent(in)
  3713)                             , t00      ! ! intent(in)
  3714)       implicit none
  3715)       !----- Required arguments. ----------------------------------------------------------!
  3716)       real(kind=4), intent(in)           :: theiv     ! Ice vap. equiv. pot. temp. [     K]
  3717)       real(kind=4), intent(in)           :: pres      ! Pressure                   [    Pa]
  3718)       real(kind=4), intent(in)           :: rtot      ! Total mixing ratio         [ kg/kg]
  3719)       !----- Optional arguments. ----------------------------------------------------------!
  3720)       logical     , intent(in), optional :: useice    ! May I use ice thermodyn.   [   T|F]
  3721)       !----- Local variables for iterative method -----------------------------------------!
  3722)       real(kind=4)                       :: pvap      ! Sat. vapour pressure
  3723)       real(kind=4)                       :: theta     ! Potential temperature
  3724)       real(kind=4)                       :: deriv     ! Function derivative
  3725)       real(kind=4)                       :: funnow    ! Function for which we seek a root.
  3726)       real(kind=4)                       :: funa      ! Smallest guess function
  3727)       real(kind=4)                       :: funz      ! Largest guess function
  3728)       real(kind=4)                       :: tlcla     ! Smallest guess (Newton: old guess)
  3729)       real(kind=4)                       :: tlclz     ! Largest guess (Newton: new guess)
  3730)       real(kind=4)                       :: tlcl      ! What will be the LCL temperature
  3731)       real(kind=4)                       :: es00      ! Defined as p00*rt/(epsilon + rt)
  3732)       real(kind=4)                       :: delta     ! Aux. variable (For 2nd guess).
  3733)       integer                            :: itn       ! Iteration counters
  3734)       integer                            :: itb       ! Iteration counters
  3735)       integer                            :: ii        ! Another counter
  3736)       logical                            :: converged ! Convergence handle
  3737)       logical                            :: zside     ! Side checker for Regula Falsi
  3738)       logical                            :: frozen    ! Will use ice thermodynamics
  3739)       !------------------------------------------------------------------------------------!
  3740) 
  3741) 
  3742) 
  3743)       !----- Fill the flag for ice thermodynamics so it will be present. ------------------!
  3744)       if (present(useice)) then
  3745)          frozen = useice
  3746)       else
  3747)          frozen = bulk_on
  3748)       end if
  3749)       !------------------------------------------------------------------------------------!
  3750) 
  3751) 
  3752) 
  3753)       !----- Find es00, which is a constant. ----------------------------------------------!
  3754)       es00 = p00 * rtot / (ep+rtot)
  3755)       !------------------------------------------------------------------------------------!
  3756) 
  3757) 
  3758)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3759)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3760)       !write (unit=36,fmt='(a)') '----------------------------------------------------------'
  3761)       !write (unit=36,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x))')                                  &
  3762)       !   'INPUT : it=',-1,'theiv=',theiv,'pres=',0.01*pres,'rtot=',rtot*1000.
  3763)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3764)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3765) 
  3766)       !------------------------------------------------------------------------------------!
  3767)       !     The 1st. guess, we assume we are lucky and right at the LCL.                   !
  3768)       !------------------------------------------------------------------------------------!
  3769)       pvap      = pres * rtot / (ep + rtot)
  3770)       tlclz     = tslif(pvap,frozen)
  3771)       theta     = tlclz * (es00 / pvap) ** rocp
  3772)       funnow    = thetaeivs(theta,tlclz,rtot,0.,0.)
  3773)       deriv     = dthetaeiv_dtlcl(funnow,tlclz,rtot,pvap,frozen)
  3774)       funnow    = funnow - theiv
  3775)       !------------------------------------------------------------------------------------!
  3776) 
  3777)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3778)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3779)       !write (unit=36,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')                &
  3780)       !   'NEWTON: it=',0,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funnow,'deriv=',deriv
  3781)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3782)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3783) 
  3784)       !----- Put something in tlcla in case we never loop through Newton's method. --------!
  3785)       tlcla     = tlclz
  3786)       funa      = funnow
  3787)       converged = .false.
  3788)       !------------------------------------------------------------------------------------!
  3789) 
  3790)       !----- Looping: Newton's iterative method -------------------------------------------!
  3791)       newloop: do itn=1,maxfpo/6
  3792)          if (abs(deriv) < toler) exit newloop !----- Too dangerous, skip to bisection -----!
  3793)          !----- Update guesses. -----------------------------------------------------------!
  3794)          tlcla   = tlclz
  3795)          funa    = funnow
  3796)          tlclz   = tlcla - funnow/deriv
  3797)          !---------------------------------------------------------------------------------!
  3798) 
  3799) 
  3800)          !----- Update the function evaluation and its derivative. ------------------------!
  3801)          pvap    = eslif(tlclz,frozen)
  3802)          theta   = tlclz * (es00/pvap)**rocp
  3803)          funnow  = thetaeivs(theta,tlclz,rtot,0.,0.)
  3804)          deriv   = dthetaeiv_dtlcl(funnow,tlclz,rtot,pvap,frozen)
  3805)          funnow  = funnow - theiv
  3806)          !---------------------------------------------------------------------------------!
  3807) 
  3808)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3809)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3810)          !write (unit=36,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')             &
  3811)          !   'NEWTON: it=',itn,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funnow           &
  3812)          !          ,'deriv=',deriv
  3813)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3814)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3815) 
  3816)          converged = abs(tlcla-tlclz) < toler * tlclz
  3817)          if (funnow == 0.) then
  3818)             tlcl = tlclz
  3819)             funz = funnow
  3820)             converged = .true.
  3821)             exit newloop
  3822)          elseif (converged) then
  3823)             tlcl = 0.5*(tlcla+tlclz)
  3824)             funz = funnow
  3825)             exit newloop
  3826)          end if
  3827)       end do newloop
  3828) 
  3829)       !------------------------------------------------------------------------------------!
  3830)       !     If I reached this point then it's because Newton's method failed. Using bisec- !
  3831)       ! tion instead.                                                                      !
  3832)       !------------------------------------------------------------------------------------!
  3833)       if (.not. converged) then
  3834)          !----- Set funz, and check whether funa and funz already have opposite sign. -----!
  3835)          funz = funnow
  3836)          zside=.true.
  3837)          if (funa*funnow > 0.) then
  3838)             !----- We fix funa, and try a funz that will work as 2nd guess ----------------!
  3839)             if (abs(funz-funa) < toler*tlcla) then
  3840)                delta = 100.*toler*tlcla
  3841)             else
  3842)                delta = max(abs(funa*(tlclz-tlcla)/(funz-funa)),100.*toler*tlcla)
  3843)             end if
  3844)             tlclz = tlcla + delta
  3845) 
  3846)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3847)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3848)             !write (unit=36,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x),3(a,1x,es11.4,1x))')          &
  3849)             !   '2NGGSS: tt=',0,'tlclz=',tlclz-t00,'tlcla=',tlcla-t00,'pvap=',0.01*pvap     &
  3850)             !           ,'funa=',funa,'funz=',funnow,'delta=',delta
  3851)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3852)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3853)             zside = .false.
  3854)             zgssloop: do itb=1,maxfpo
  3855)                !----- So the first time tlclz = tlcla - 2*delta ---------------------------!
  3856)                tlclz = tlcla + real((-1)**itb * (itb+3)/2) * delta
  3857)                pvap  = eslif(tlclz,frozen)
  3858)                theta = tlclz * (es00/pvap)**rocp
  3859)                funz  = thetaeivs(theta,tlclz,rtot,0.,0.) - theiv
  3860) 
  3861)                !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3862)                !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3863)                !write (unit=36,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')       &
  3864)                !   '2NGGSS: tt=',itb,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funz       &
  3865)                !           ,'delta=',delta
  3866)                !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3867)                !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3868) 
  3869)                zside = funa*funz < 0.
  3870)                if (zside) exit zgssloop
  3871)             end do zgssloop
  3872)             if (.not. zside) then
  3873)                write (unit=*,fmt='(a)') ' No second guess for you...'
  3874)                write (unit=*,fmt='(2(a,1x,i14,1x))')    'itn   =',itn   ,'itb   =',itb
  3875)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'theiv =',theiv ,'rtot  =',rtot
  3876)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'pres  =',pres  ,'pvap  =',pvap
  3877)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'theta =',theta ,'delta =',delta
  3878)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'tlcla =',tlcla ,'funa  =',funa
  3879)                write (unit=*,fmt='(2(a,1x,es14.7,1x))') 'tlclz =',tlclz ,'funz  =',funz
  3880)                call fatal_error('Failed finding the second guess for regula falsi'         &
  3881)                                ,'thetaeiv2thil','therm_lib.f90')
  3882)             end if
  3883)          end if
  3884)          !---- Continue iterative method. -------------------------------------------------!
  3885)          fpoloop: do itb=itn+1,maxfpo
  3886) 
  3887)             !----- Update the guess. ------------------------------------------------------!
  3888)             tlcl   = (funz*tlcla-funa*tlclz)/(funz-funa)
  3889) 
  3890)             !----- Updating function evaluation -------------------------------------------!
  3891)             pvap   = eslif(tlcl,frozen)
  3892)             theta  = tlcl * (es00/pvap)**rocp
  3893)             funnow = thetaeivs(theta,tlcl,rtot,0.,0.) - theiv
  3894) 
  3895)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3896)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3897)             !write (unit=36,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),1(a,1x,es11.4,1x))')          &
  3898)             !   'REGFAL: it=',itb,'tlcl =',tlcl -t00,'pvap=',0.01*pvap,'fun=',funnow
  3899)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3900)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3901) 
  3902)             !------------------------------------------------------------------------------!
  3903)             !    Check for convergence. If it did, return, we found the solution.          !
  3904)             ! Otherwise, constrain the guesses.                                            !
  3905)             !------------------------------------------------------------------------------!
  3906)             converged = abs(tlcl-tlcla) < toler * tlcl
  3907)             if (converged) then
  3908)                exit fpoloop
  3909)             elseif (funnow*funa < 0.) then
  3910)                tlclz = tlcl
  3911)                funz  = funnow
  3912)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  3913)                if (zside) funa=funa * 0.5
  3914)                !----- We just updated zside, setting zside to true. -----------------------!
  3915)                zside = .true.
  3916)             else
  3917)                tlcla = tlcl
  3918)                funa  = funnow
  3919)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  3920)                if (.not.zside) funz = funz * 0.5
  3921)                !----- We just updated aside, setting zside to false -----------------------!
  3922)                zside = .false.
  3923)             end if
  3924)          end do fpoloop
  3925)       end if
  3926) 
  3927)       if (converged) then
  3928)          thetaeiv2thil  = theiv * exp (- alvl(tlcl) * rtot / (cpdry * tlcl) )
  3929)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3930)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3931)          !write (unit=36,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')             &
  3932)          !   'ANSWER: itb=',itn,'tlcl=',tlcl-t00,'eslcl=',0.01*pvap                         &
  3933)          !          ,'thil=',thetaeiv2thil,'funa=',funa,'funz=',funz
  3934)          !write (unit=36,fmt='(a)') '-------------------------------------------------------'
  3935)          !write (unit=36,fmt='(a)') ' '
  3936)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  3937)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  3938)       else
  3939)          write (unit=*,fmt='(60a1)')        ('-',ii=1,60)
  3940)          write (unit=*,fmt='(a)')           ' THEIV2THIL failed!'
  3941)          write (unit=*,fmt='(a)')           ' '
  3942)          write (unit=*,fmt='(a)')           ' -> Input: '
  3943)          write (unit=*,fmt='(a,1x,f12.5)')  '    THEIV    [     K]:',theiv
  3944)          write (unit=*,fmt='(a,1x,f12.5)')  '    PRES     [    Pa]:',pres * 100.
  3945)          write (unit=*,fmt='(a,1x,f12.5)')  '    RTOT     [  g/kg]:',1000.*rtot

  3950)          write (unit=*,fmt='(a,1x,f12.5)')  '    THETA    [     K]:',theta

  3957)          write (unit=*,fmt='(a,1x,es12.5)') '    ERR_A    [   ---]:',abs(tlcl-tlcla)/tlcl
  3958)          write (unit=*,fmt='(a,1x,es12.5)') '    ERR_Z    [   ---]:',abs(tlcl-tlclz)/tlcl

  3960)          write (unit=*,fmt='(60a1)')        ('-',ii=1,60)

  3963)                          ,'thetaeiv2thil','therm_lib.f90')
  3964)       end if

  3978)    !    This subroutine converts saturated ice-vapour equivalent potential temperature     !
  3979)    ! into temperature, given pressure in Pa, and theta_es in Kelvin. It also returns the   !
  3980)    ! potential temperature in Kelvin, and saturation vapour mixing ratio in kg/kg. As      !
  3981)    ! usual,  we seek T using Newton's method as a starting point, and if it fails, we fall !
  3982)    ! back to the modified regula falsi (Illinois method).                                  !
  3983)    !                                                                                       !
  3984)    ! OBS: In case you want to ignore ice, send useice as false. The default is to consider !
  3985)    !      when level >= 3 and to ignore otherwise.                                         !
  3986)    !---------------------------------------------------------------------------------------!
  3987)    subroutine thetaeivs2temp(theivs,pres,theta,temp,rsat,useice)
  3988)       use consts_coms, only : cpdry     & ! intent(in)
  3989)                             , ep        & ! intent(in)
  3990)                             , p00       & ! intent(in)
  3991)                             , rocp      & ! intent(in)
  3992)                             , t00       ! ! intent(in)
  3993)       implicit none
  3994)       !----- Arguments --------------------------------------------------------------------!
  3995)       real(kind=4), intent(in)            :: theivs     ! Sat. thetae_iv          [      K]
  3996)       real(kind=4), intent(in)            :: pres       ! Pressure                [     Pa]
  3997)       real(kind=4), intent(out)           :: theta      ! Potential temperature   [      K]
  3998)       real(kind=4), intent(out)           :: temp       ! Temperature             [      K]
  3999)       real(kind=4), intent(out)           :: rsat       ! Saturation mixing ratio [  kg/kg]
  4000)       logical     , intent(in) , optional :: useice     ! May use ice thermodyn.  [    T|F]
  4001)       !----- Local variables, with other thermodynamic properties -------------------------!
  4002)       real(kind=4)                        :: exnernormi ! 1./ (Norm. Exner func.) [    ---]
  4003)       logical                             :: frozen     ! Will use ice thermodyn. [    T|F]
  4004)       !----- Local variables for iterative method -----------------------------------------!
  4005)       real(kind=4)                        :: deriv      ! Function derivative
  4006)       real(kind=4)                        :: funnow     ! Current function evaluation
  4007)       real(kind=4)                        :: funa       ! Smallest guess function
  4008)       real(kind=4)                        :: funz       ! Largest guess function
  4009)       real(kind=4)                        :: tempa      ! Smallest guess (Newton: previous)
  4010)       real(kind=4)                        :: tempz      ! Largest guess (Newton: new)
  4011)       real(kind=4)                        :: delta      ! Aux. variable for 2nd guess.
  4012)       integer                             :: itn        ! Iteration counter
  4013)       integer                             :: itb        ! Iteration counter
  4014)       logical                             :: converged  ! Convergence handle
  4015)       logical                             :: zside      ! Flag for side check.
  4016)       !------------------------------------------------------------------------------------!
  4017) 
  4018) 
  4019)       !----- Set up the ice check, in case useice is not present. -------------------------!
  4020)       if (present(useice)) then
  4021)          frozen = useice
  4022)       else
  4023)          frozen = bulk_on
  4024)       end if
  4025)       !------------------------------------------------------------------------------------!
  4026) 
  4027) 
  4028) 
  4029)       !----- Finding the inverse of normalised Exner, which is constant in this routine ---!
  4030)       exnernormi = (p00 /pres) ** rocp
  4031)       !------------------------------------------------------------------------------------!
  4032) 
  4033) 
  4034) 
  4035)       !------------------------------------------------------------------------------------!
  4036)       !     The 1st. guess, no idea, guess 0C.                                            !
  4037)       !------------------------------------------------------------------------------------!
  4038)       tempz     = t00
  4039)       theta     = tempz * exnernormi
  4040)       rsat      = rslif(pres,tempz,frozen)
  4041)       funnow    = thetaeivs(theta,tempz,rsat,0.,0.)
  4042)       deriv     = dthetaeivs_dt(funnow,tempz,pres,rsat,frozen)
  4043)       funnow    = funnow - theivs
  4044)       !------------------------------------------------------------------------------------!
  4045) 
  4046) 
  4047)       !----- Copy here just in case Newton is aborted at the 1st guess. -------------------!
  4048)       tempa     = tempz
  4049)       funa      = funnow
  4050)       !------------------------------------------------------------------------------------!
  4051) 
  4052)       converged = .false.
  4053)       !----- Newton's method loop. --------------------------------------------------------!
  4054)       newloop: do itn=1,maxfpo/6
  4055)          if (abs(deriv) < toler) exit newloop !----- Too dangerous, skip to bisection -----!
  4056)          !----- Updating guesses ----------------------------------------------------------!
  4057)          tempa   = tempz
  4058)          funa    = funnow
  4059) 
  4060)          tempz   = tempa - funnow/deriv
  4061)          theta   = tempz * exnernormi
  4062)          rsat    = rslif(pres,tempz,frozen)
  4063)          funnow  = thetaeivs(theta,tempz,rsat,0.,0.)
  4064)          deriv   = dthetaeivs_dt(funnow,tempz,pres,rsat,frozen)
  4065)          funnow  = funnow - theivs
  4066) 
  4067)          converged = abs(tempa-tempz) < toler*tempz
  4068)          if (funnow == 0.) then
  4069)             converged =.true.
  4070)             temp = tempz
  4071)             exit newloop
  4072)          elseif (converged) then
  4073)             temp = 0.5*(tempa+tempz)
  4074)             exit newloop
  4075)          end if
  4076)       end do newloop
  4077) 
  4078)       !------------------------------------------------------------------------------------!
  4079)       !     If we have reached this point then it's because Newton's method failed.  Use   !
  4080)       ! bisection instead.                                                                 !
  4081)       !------------------------------------------------------------------------------------!
  4082)       if (.not. converged) then
  4083)          !----- Set funz, and check whether funa and funz already have opposite sign. -----!
  4084)          funz  = funnow
  4085)          zside = .false.
  4086)          !---------------------------------------------------------------------------------!
  4087) 
  4088)          if (funa*funnow > 0.) then
  4089)             !----- We fix funa, and try a funz that will work as 2nd guess ----------------!
  4090)             if (abs(funz-funa) < toler*tempa) then
  4091)                delta = 100.*toler*tempa
  4092)             else
  4093)                delta = max(abs(funa*(tempz-tempa)/(funz-funa)),100.*toler*tempa)
  4094)             end if
  4095)             !------------------------------------------------------------------------------!
  4096) 
  4097)             tempz = tempa + delta
  4098)             zgssloop: do itb=1,maxfpo
  4099)                !----- So this will be +1 -1 +2 -2 etc. ------------------------------------!
  4100)                tempz = tempz + real((-1)**itb * (itb+3)/2) * delta
  4101)                theta = tempz * exnernormi
  4102)                rsat  = rslif(pres,tempz,frozen)
  4103)                funz  = thetaeivs(theta,tempz,rsat,0.,0.) - theivs
  4104)                zside = funa*funz < 0.
  4105)                if (zside) exit zgssloop
  4106)             end do zgssloop
  4107)             if (.not. zside) then
  4108)                call fatal_error('Failed finding the second guess for regula falsi'         &
  4109)                                ,'thetaes2temp','therm_lib.f90')
  4110)             end if
  4111)          end if
  4112)          !---- Continue iterative method --------------------------------------------------!
  4113)          fpoloop: do itb=itn+1,maxfpo
  4114)             if (abs(funz-funa) < toler*tempa) then
  4115)                temp   = 0.5*(tempa+tempz)
  4116)             else
  4117)                temp   = (funz*tempa-funa*tempz)/(funz-funa)
  4118)             end if
  4119)             theta  = temp * exnernormi
  4120)             rsat   = rslif(pres,temp,frozen)
  4121)             funnow = thetaeivs(theta,temp,rsat,0.,0.) - theivs
  4122) 
  4123)             !------------------------------------------------------------------------------!
  4124)             !    Checking for convergence. If it did, return, we found the solution.       !
  4125)             ! Otherwise, constrain the guesses.                                            !
  4126)             !------------------------------------------------------------------------------!
  4127)             converged = abs(temp-tempa) < toler*temp
  4128)             if (converged) then
  4129)                exit fpoloop
  4130)             elseif (funnow*funa < 0.) then
  4131)                tempz = temp
  4132)                funz  = funnow
  4133)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  4134)                if (zside) funa=funa * 0.5
  4135)                !----- We just updated zside, setting zside to true. -----------------------!
  4136)                zside = .true.
  4137)             else
  4138)                tempa = temp
  4139)                funa  = funnow
  4140)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  4141)                if (.not. zside) funz = funz * 0.5
  4142)                !----- We just updated aside, setting zside to false -----------------------!
  4143)                zside = .false.
  4144)             end if
  4145)          end do fpoloop
  4146)       end if
  4147) 
  4148)       if (converged) then
  4149)          !----- Compute theta and rsat with temp just for consistency ---------------------!
  4150)          theta = temp * exnernormi
  4151)          rsat  = rslif(pres,temp,frozen)
  4152)       else
  4153)          call fatal_error('Temperature didn''t converge, I gave up!'                       &
  4154)                          ,'thetaes2temp','therm_lib.f90')
  4155)       end if
  4156) 
  4157)       return
  4158)    end subroutine thetaeivs2temp

  4169)    !    This subroutine finds the lifting condensation level given the ice-liquid          !
  4170)    ! potential temperature in Kelvin, temperature in Kelvin, the pressure in Pascal, and   !
  4171)    ! the mixing ratio in kg/kg. The output will give the LCL temperature and pressure, and !
  4172)    ! the thickness of the layer between the initial point and the LCL.                     !
  4173)    !                                                                                       !
  4174)    !    References:                                                                        !
  4175)    !    Tripoli, J. T.; and Cotton, W.R., 1981: The use of ice-liquid water potential      !
  4176)    !        temperature as a thermodynamic variable in deep atmospheric models. Mon. Wea.  !
  4177)    !        Rev., v. 109, 1094-1102. (TC81)                                                !
  4178)    !    Bolton, D., 1980: The computation of the equivalent potential temperature. Mon.    !
  4179)    !        Wea. Rev., v. 108, 1046-1053. (BO80)                                           !
  4180)    !                                                                                       !
  4181)    !    Some algebra was needed to find this equation, essentially combining (TC81-26) and !
  4182)    ! (TC81-27), and the conservation of total water (TC81-16). It assumes that the divi-   !
  4183)    ! sion between the three phases is already taken care of.                               !
  4184)    !    Iterative procedure is needed, and here we iterate looking for T(LCL). Theta_il    !
  4185)    ! can be rewritten in terms of T(LCL) only, and once we know this thetae_iv becomes     !
  4186)    ! straightforward. T(LCL) will be found using Newton's method, and in the unlikely      !
  4187)    ! event it fails,we will fall back to the modified regula falsi (Illinois method).      !
  4188)    !                                                                                       !
  4189)    ! Important remarks:                                                                    !
  4190)    ! 1. TLCL and PLCL are the actual TLCL and PLCL, so in case condensation exists, they   !
  4191)    !    will be larger than the actual temperature and pressure (because one would go down !
  4192)    !    to reach the equilibrium);                                                         !
  4193)    ! 2. DZLCL WILL BE SET TO ZERO in case the LCL is beneath the starting level. So in     !
  4194)    !    case you want to force TLCL <= TEMP and PLCL <= PRES, you can use this variable    !
  4195)    !    to run the saturation check afterwards. DON'T CHANGE PLCL and TLCL here, they will !
  4196)    !    be used for conversions between theta_il and thetae_iv as they are defined here.   !
  4197)    ! 3. In case you don't want ice, simply pass useice=.false.. Otherwise let the model    !
  4198)    !    decide by itself based on the LEVEL variable.                                      !
  4199)    !---------------------------------------------------------------------------------------!
  4200)    subroutine lcl_il(thil,pres,temp,rtot,rvap,tlcl,plcl,dzlcl,useice)
  4201)       use consts_coms, only : cpog     & ! intent(in)
  4202)                             , ep       & ! intent(in)
  4203)                             , p00      & ! intent(in)
  4204)                             , rocp     & ! intent(in)
  4205)                             , t3ple    & ! intent(in)
  4206)                             , t00      ! ! intent(in)
  4207)       implicit none
  4208)       !----- Required arguments. ----------------------------------------------------------!
  4209)       real(kind=4), intent(in)            :: thil      ! Ice liquid pot. temp. (*)[      K]
  4210)       real(kind=4), intent(in)            :: pres      ! Pressure                 [     Pa]
  4211)       real(kind=4), intent(in)            :: temp      ! Temperature              [      K]
  4212)       real(kind=4), intent(in)            :: rtot      ! Total mixing ratio       [  kg/kg]
  4213)       real(kind=4), intent(in)            :: rvap      ! Vapour mixing ratio      [  kg/kg]
  4214)       real(kind=4), intent(out)           :: tlcl      ! LCL temperature          [      K]
  4215)       real(kind=4), intent(out)           :: plcl      ! LCL pressure             [     Pa]
  4216)       real(kind=4), intent(out)           :: dzlcl     ! Sub-LCL layer thickness  [      m]
  4217)       !------------------------------------------------------------------------------------!
  4218)       ! (*) This is the most general variable. Thil is exactly theta for no condensation   !
  4219)       !     condition, and it is the liquid potential temperature if no ice is present.    !
  4220)       !------------------------------------------------------------------------------------!
  4221)       !----- Optional arguments. ----------------------------------------------------------!

  4226)       real(kind=4)                        :: funnow    ! Current function evaluation
  4227)       real(kind=4)                        :: funa      ! Smallest guess function
  4228)       real(kind=4)                        :: funz      ! Largest  guess function
  4229)       real(kind=4)                        :: tlcla     ! Smallest guess (Newton: previous)
  4230)       real(kind=4)                        :: tlclz     ! Largest guess (Newton: new)
  4231)       real(kind=4)                        :: es00      ! Defined as p00*rt/(epsilon + rt)
  4232)       real(kind=4)                        :: delta     ! Aux. variable for bisection
  4233)       integer                             :: itn       ! Iteration counter
  4234)       integer                             :: itb       ! Iteration counter
  4235)       logical                             :: converged ! Convergence flag
  4236)       logical                             :: zside     ! Flag to check sides
  4237)       logical                             :: frozen    ! Will use ice thermodyn.  [    T|F]

  4239) 
  4240) 
  4241)       !------------------------------------------------------------------------------------!
  4242)       !     Check whether ice thermodynamics is the way to go.                             !
  4243)       !------------------------------------------------------------------------------------!
  4244)       if (present(useice)) then
  4245)          frozen = useice
  4246)       else
  4247)          frozen = bulk_on
  4248)       end if
  4249)       !------------------------------------------------------------------------------------!
  4250) 
  4251)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4252)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4253)       !write (unit=21,fmt='(a)') '----------------------------------------------------------'
  4254)       !write (unit=21,fmt='(a,1x,i5,1x,5(a,1x,f11.4,1x))')                                  &
  4255)       !   'INPUT : it=',-1,'thil=',thil,'pres=',0.01*pres,'temp=',temp-t00                  &
  4256)       !        ,'rvap=',rvap*1000.,'rtot=',rtot*1000.
  4257)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4258)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4259) 
  4260) 
  4261)       !----- Find es00, which is a constant. ----------------------------------------------!
  4262)       es00 = p00 * rtot / (ep+rtot)
  4263)       !------------------------------------------------------------------------------------!
  4264) 
  4265) 
  4266) 
  4267)       !------------------------------------------------------------------------------------!
  4268)       !     The 1st. guess, use equation 21 from Bolton (1980). For this we'll need the    !
  4269)       ! vapour pressure.                                                                   !
  4270)       !------------------------------------------------------------------------------------!
  4271)       pvap      = pres * rvap / (ep + rvap)
  4272)       tlclz     = 55. + 2840. / (3.5 * log(temp) - log(0.01*pvap) - 4.805)
  4273)       pvap      = eslif(tlclz,frozen)
  4274)       funnow    = tlclz * (es00/pvap)**rocp - thil

  4276)       !------------------------------------------------------------------------------------!

  4278)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4279)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4280)       !write (unit=21,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')                &
  4281)       !   'NEWTON: it=',0,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funnow,'deriv=',deriv
  4282)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4283)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!

  4285)       funa      = funnow

  4290)       !------------------------------------------------------------------------------------!
  4291)       newloop: do itn=1,maxfpo/6
  4292)          !----- If derivative is too small, skip Newton's and try bisection instead. ------!
  4293)          if (abs(deriv) < toler) exit newloop
  4294)          !---------------------------------------------------------------------------------!
  4295) 
  4296) 
  4297)          !----- Otherwise, update guesses. ------------------------------------------------!
  4298)          tlcla  = tlclz
  4299)          funa   = funnow
  4300)          tlclz  = tlcla - funnow/deriv
  4301)          pvap   = eslif(tlclz,frozen)
  4302)          funnow = tlclz * (es00/pvap)**rocp - thil

  4305)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4306)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!

  4309)          !          ,'deriv=',deriv
  4310)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4311)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4312) 
  4313)          !---------------------------------------------------------------------------------!
  4314)          !      Check for convergence.                                                     !
  4315)          !---------------------------------------------------------------------------------!
  4316)          converged = abs(tlcla-tlclz) < toler*tlclz
  4317)          if (converged) then
  4318)             !----- Guesses are almost identical, average them. ----------------------------!
  4319)             tlcl = 0.5*(tlcla+tlclz)
  4320)             funz = funnow
  4321)             exit newloop
  4322)             !------------------------------------------------------------------------------!
  4323)          elseif (funnow == 0.) then
  4324)             !----- We've hit the answer by luck, copy the answer. -------------------------!
  4325)             tlcl = tlclz
  4326)             funz = funnow
  4327)             converged = .true.
  4328)             exit newloop
  4329)             !------------------------------------------------------------------------------!
  4330)          end if
  4331)       end do newloop
  4332)       !------------------------------------------------------------------------------------!
  4333) 
  4334) 
  4335) 
  4336)       !------------------------------------------------------------------------------------!
  4337)       !      Check whether Newton's method has converged.                                  !
  4338)       !------------------------------------------------------------------------------------!
  4339)       if (.not. converged) then
  4340)          !---------------------------------------------------------------------------------!
  4341)          !     Newton's method has failed.  We use regula falsi instead.  First, we must   !
  4342)          ! find two guesses whose function evaluations have opposite signs.                !
  4343)          !---------------------------------------------------------------------------------!
  4344)          if (funa*funnow < 0. ) then
  4345)             !----- We already have two good guesses. --------------------------------------!
  4346)             funz  = funnow
  4347)             zside = .true.
  4348)             !------------------------------------------------------------------------------!
  4349)          else
  4350)             !------------------------------------------------------------------------------!
  4351)             !     We need to find another guess with opposite sign.                        !
  4352)             !------------------------------------------------------------------------------!
  4353) 
  4354)             !----- We fix funa, and try a funz that will work as 2nd guess ----------------!
  4355)             if (abs(funnow-funa) < toler*tlcla) then
  4356)                delta = 100.*toler*tlcla
  4357)             else
  4358)                delta = max(abs(funa*(tlclz-tlcla)/(funnow-funa)),100.*toler*tlcla)
  4359)             end if
  4360)             tlclz = tlcla + delta
  4361)             !------------------------------------------------------------------------------!
  4362) 
  4363)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4364)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4365)             !write (unit=21,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x),3(a,1x,es11.4,1x))')          &
  4366)             !   '2NGGSS: tt=',0,'tlclz=',tlclz-t00,'tlcla=',tlcla-t00,'pvap=',0.01*pvap     &
  4367)             !           ,'funa=',funa,'funz=',funnow,'delta=',delta
  4368)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4369)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4370)             zside = .false.
  4371)             zgssloop: do itb=1,maxfpo
  4372) 
  4373)                !----- So the first time tlclz = tlcla - 2*delta ---------------------------!
  4374)                tlclz = tlcla + real((-1)**itb * (itb+3)/2) * delta
  4375)                pvap  = eslif(tlclz,frozen)
  4376)                funz  = tlclz * (es00/pvap)**rocp - thil
  4377)                !---------------------------------------------------------------------------!
  4378) 
  4379) 
  4380)                !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4381)                !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4382)                !write (unit=21,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),2(a,1x,es11.4,1x))')       &
  4383)                !   '2NGGSS: tt=',itb,'tlclz=',tlclz-t00,'pvap=',0.01*pvap,'fun=',funz       &
  4384)                !           ,'delta=',delta
  4385)                !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4386)                !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4387)                zside = funa*funz < 0.0
  4388)                if (zside) exit zgssloop
  4389)             end do zgssloop
  4390)             if (.not. zside) then
  4391)                write (unit=*,fmt='(a)') ' ====== No second guess for you... ======'
  4392)                write (unit=*,fmt='(a)') ' + INPUT variables: '
  4393)                write (unit=*,fmt='(a,1x,es14.7)') 'THIL =',thil
  4394)                write (unit=*,fmt='(a,1x,es14.7)') 'TEMP =',temp
  4395)                write (unit=*,fmt='(a,1x,es14.7)') 'PRES =',pres
  4396)                write (unit=*,fmt='(a,1x,es14.7)') 'RTOT =',rtot
  4397)                write (unit=*,fmt='(a,1x,es14.7)') 'RVAP =',rvap
  4398)                write (unit=*,fmt='(a)') ' ============ Failed guess... ==========='
  4399)                write (unit=*,fmt='(2(a,1x,es14.7))') 'TLCLA =',tlcla,'FUNA =',funa
  4400)                write (unit=*,fmt='(2(a,1x,es14.7))') 'TLCLZ =',tlclz,'FUNC =',funz
  4401)                write (unit=*,fmt='(2(a,1x,es14.7))') 'DELTA =',delta,'FUNN =',funnow
  4402)                call fatal_error('Failed finding the second guess for regula falsi'         &
  4403)                              ,'lcl_il','therm_lib.f90')
  4404)             end if
  4405)          end if
  4406)          !---------------------------------------------------------------------------------!
  4407) 
  4408) 
  4409)          !---------------------------------------------------------------------------------!
  4410)          !      We have the guesses, solve the regula falsi method.                        !
  4411)          !---------------------------------------------------------------------------------!
  4412)          fpoloop: do itb=itn+1,maxfpo
  4413)             !----- Update guess and function evaluation. ----------------------------------!
  4414)             tlcl   = (funz*tlcla-funa*tlclz)/(funz-funa)
  4415)             pvap   = eslif(tlcl,frozen)
  4416)             funnow = tlcl * (es00/pvap)**rocp - thil
  4417)             !------------------------------------------------------------------------------!
  4418) 
  4419)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4420)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4421)             !write (unit=21,fmt='(a,1x,i5,1x,2(a,1x,f11.4,1x),1(a,1x,es11.4,1x))')          &
  4422)             !   'REGFAL: it=',itb,'tlcl =',tlcl -t00,'pvap=',0.01*pvap,'fun=',funnow
  4423)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4424)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4425) 
  4426)             !------------------------------------------------------------------------------!
  4427)             !    Check for convergence.  If it did, return, we found the solution.         !
  4428)             ! Otherwise, we update one of the guesses.                                     !
  4429)             !------------------------------------------------------------------------------!
  4430)             converged = abs(tlcl-tlcla) < toler*tlcl .and.  abs(tlcl-tlclz) < toler*tlcl
  4431)             if (funnow == 0. .or. converged) then
  4432)                converged = .true.
  4433)                exit fpoloop
  4434)             elseif (funnow*funa < 0.) then
  4435)                tlclz = tlcl
  4436)                funz  = funnow
  4437)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  4438)                if (zside) funa=funa * 0.5
  4439)                !---------------------------------------------------------------------------!
  4440) 
  4441) 
  4442)                !----- We have just updated zside, sett zside to true. ---------------------!
  4443)                zside = .true.
  4444)                !---------------------------------------------------------------------------!
  4445)             else
  4446)                tlcla = tlcl
  4447)                funa  = funnow
  4448)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  4449)                if (.not.zside) funz = funz * 0.5
  4450)                !---------------------------------------------------------------------------!
  4451) 
  4452) 
  4453)                !----- We have just updated aside, set zside to false. ---------------------!
  4454)                zside = .false.
  4455)                !---------------------------------------------------------------------------!
  4456)             end if
  4457)          end do fpoloop
  4458)       end if
  4459)       !------------------------------------------------------------------------------------!
  4460) 
  4461) 
  4462) 
  4463)       !------------------------------------------------------------------------------------!
  4464)       !     Check whether we have succeeded or not.                                        !
  4465)       !------------------------------------------------------------------------------------!
  4466)       if (converged) then
  4467)          !----- We have found a solution, find the remaining LCL properties. --------------!
  4468)          pvap  = eslif(tlcl,frozen)
  4469)          plcl  = (ep + rvap) * pvap / rvap
  4470)          dzlcl = max(cpog*(temp-tlcl),0.)
  4471)          !---------------------------------------------------------------------------------!
  4472) 
  4473) 
  4474)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4475)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4476)          !write (unit=21,fmt='(a,1x,i5,1x,3(a,1x,f11.4,1x),3(a,1x,es11.4,1x))')             &
  4477)          !   'ANSWER: itb=',itn,'tlcl=',tlcl-t00,'eslcl=',0.01*pvap                         &
  4478)          !        ,'dzlcl=',dzlcl,'plcl=',plcl*0.01,'funa=',funa,'funz=',funz
  4479)          !write (unit=21,fmt='(a)') '-------------------------------------------------------'
  4480)          !write (unit=21,fmt='(a)') ' '
  4481)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4482)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4483)       else
  4484)          write (unit=*,fmt='(a)') '-------------------------------------------------------'
  4485)          write (unit=*,fmt='(a)') ' LCL Temperature didn''t converge!!!'
  4486)          write (unit=*,fmt='(a,1x,i5,1x,a)') ' I gave up, after',maxfpo,'iterations...'
  4487)          write (unit=*,fmt='(a)') ' '
  4488)          write (unit=*,fmt='(a)') ' Input values.'
  4489)          write (unit=*,fmt='(a)') ' '
  4490)          write (unit=*,fmt='(a,1x,f12.4)' ) 'theta_il        [     K] =',thil
  4491)          write (unit=*,fmt='(a,1x,f12.4)' ) 'Pressure        [   hPa] =',0.01*pres
  4492)          write (unit=*,fmt='(a,1x,f12.4)' ) 'Temperature     [    C] =',temp-t00
  4493)          write (unit=*,fmt='(a,1x,f12.4)' ) 'rtot            [  g/kg] =',1000.*rtot
  4494)          write (unit=*,fmt='(a,1x,f12.4)' ) 'rvap            [  g/kg] =',1000.*rvap
  4495)          write (unit=*,fmt='(a)') ' '
  4496)          write (unit=*,fmt='(a)') ' Last iteration outcome.'
  4497)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tlcla           [    C] =',tlcla-t00
  4498)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tlclz           [    C] =',tlclz-t00
  4499)          write (unit=*,fmt='(a,1x,f12.4)' ) 'fun             [     K] =',funnow
  4500)          write (unit=*,fmt='(a,1x,f12.4)' ) 'funa            [     K] =',funa
  4501)          write (unit=*,fmt='(a,1x,f12.4)' ) 'funz            [     K] =',funz
  4502)          write (unit=*,fmt='(a,1x,f12.4)' ) 'deriv           [  ----] =',deriv
  4503)          write (unit=*,fmt='(a,1x,es12.4)') 'toler           [  ----] =',toler
  4504)          write (unit=*,fmt='(a,1x,es12.4)') 'error           [  ----] ='                   &
  4505)                                                             ,abs(tlclz-tlcla)/tlclz
  4506)          write (unit=*,fmt='(a,1x,f12.4)' ) 'tlcl            [    C] =',tlcl
  4507)          call fatal_error('TLCL didn''t converge, gave up!','lcl_il','therm_lib.f90')
  4508)       end if
  4509)       return

  4521)    !     This subroutine computes a consistent set of temperature and condensated phases   !
  4522)    ! mixing ratio for a given theta_il, Exner function, and total mixing ratio. This is    !
  4523)    ! very similar to the function thil2temp, except that now we don't know rliq and rice,  !
  4524)    ! and for this reason they also become functions of temperature, since they are defined !
  4525)    ! as rtot-rsat(T,p), remembering that rtot and p are known. If the air is not           !
  4526)    ! saturated, we rather use the fact that theta_il = theta and skip the hassle.          !
  4527)    ! Otherwise, we use iterative methods.  We will always try Newton's method, since it    !
  4528)    ! converges fast. The caveat is that Newton may fail, and it actually does fail very    !
  4529)    ! close to the triple point, because the saturation vapour pressure function has a      !
  4530)    ! "kink" at the triple point (continuous, but not differentiable).  If that's the case, !
  4531)    ! then we fall back to a modified regula falsi (Illinois) method, which is a mix of     !
  4532)    ! secant and bisection and will converge.                                               !
  4533)    !---------------------------------------------------------------------------------------!
  4534)    subroutine thil2tqall(thil,exner,pres,rtot,rliq,rice,temp,rvap,rsat)
  4535)       use consts_coms, only : cpdry    & ! intent(in)
  4536)                             , cpdryi   & ! intent(in)
  4537)                             , t00      & ! intent(in)
  4538)                             , toodry   & ! intent(in)
  4539)                             , t3ple    ! ! intent(in)
  4540) 
  4541)       implicit none
  4542)       !----- Arguments. -------------------------------------------------------------------!
  4543)       real(kind=4), intent(in)    :: thil      ! Ice-liquid water potential temp.  [     K]
  4544)       real(kind=4), intent(in)    :: exner     ! Exner function                    [J/kg/K]
  4545)       real(kind=4), intent(in)    :: pres      ! Pressure                          [    Pa]
  4546)       real(kind=4), intent(in)    :: rtot      ! Total mixing ratio                [ kg/kg]
  4547)       real(kind=4), intent(out)   :: rliq      ! Liquid water mixing ratio         [ kg/kg]
  4548)       real(kind=4), intent(out)   :: rice      ! Ice mixing ratio                  [ kg/kg]
  4549)       real(kind=4), intent(inout) :: temp      ! Temperature                       [     K]
  4550)       real(kind=4), intent(out)   :: rvap      ! Water vapour mixing ratio         [ kg/kg]
  4551)       real(kind=4), intent(out)   :: rsat      ! Sat. water vapour mixing ratio    [ kg/kg]
  4552)       !----- Local variables --------------------------------------------------------------!
  4553)       real(kind=4)                :: tempa     ! Lower bound for regula falsi iteration
  4554)       real(kind=4)                :: tempz     ! Upper bound for regula falsi iteration
  4555)       real(kind=4)                :: t1stguess ! Book keeping temperature 1st guess
  4556)       real(kind=4)                :: fun1st    ! Book keeping 1st guess function
  4557)       real(kind=4)                :: funa      ! Function evaluation at tempa
  4558)       real(kind=4)                :: funz      ! Function evaluation at tempz
  4559)       real(kind=4)                :: funnow    ! Function at this iteration.
  4560)       real(kind=4)                :: delta     ! Aux. var in case we need regula falsi.
  4561)       real(kind=4)                :: deriv     ! Derivative of this function.
  4562)       integer                     :: itn       ! Iteration counter
  4563)       integer                     :: itb       ! Iteration counter
  4564)       integer                     :: ii        ! Iteration counter
  4565)       logical                     :: converged ! Convergence handle
  4566)       logical                     :: zside     ! Aux. Flag, for two purposes:
  4567)                                                ! 1. Found a 2nd guess for regula falsi.
  4568)                                                ! 2. I retained the "zside" (T/F)
  4569)       !------------------------------------------------------------------------------------!
  4570) 
  4571)       t1stguess = temp
  4572) 
  4573)       !------------------------------------------------------------------------------------!
  4574)       !      First check: try to find temperature assuming sub-saturation and check if     !
  4575)       ! this is the case. If it is, then there is no need to go through the iterative      !
  4576)       ! loop.                                                                              !
  4577)       !------------------------------------------------------------------------------------!
  4578)       tempz  = cpdryi * thil * exner
  4579)       rsat   = max(toodry,rslif(pres,tempz))
  4580)       if (tempz >= t3ple) then
  4581)          rliq = max(0.,rtot-rsat)
  4582)          rice = 0.
  4583)       else
  4584)          rice = max(0.,rtot-rsat)
  4585)          rliq = 0.
  4586)       end if
  4587)       rvap = rtot-rliq-rice
  4588)       !------------------------------------------------------------------------------------!
  4589) 
  4590)       !------------------------------------------------------------------------------------!
  4591)       !    If rtot < rsat, this is not saturated, we can leave the subroutine and bypass   !
  4592)       ! the iterative part.                                                                !
  4593)       !------------------------------------------------------------------------------------!
  4594)       if (rtot < rsat) then
  4595)          temp = tempz
  4596)          return
  4597)       end if
  4598) 
  4599)       !------------------------------------------------------------------------------------!
  4600)       !   If not, then use the temperature the user gave as first guess and solve          !
  4601)       ! iteratively.  We use the user instead of what we just found because if the air is  !
  4602)       ! saturated, then this can be too far off which may be bad for Newton's method.      !
  4603)       !------------------------------------------------------------------------------------!
  4604)       tempz = temp
  4605)       rsat   = max(toodry,rslif(pres,tempz))
  4606)       if (tempz >= t3ple) then
  4607)          rliq = max(0.,rtot-rsat)
  4608)          rice = 0.
  4609)       else
  4610)          rice = max(0.,rtot-rsat)
  4611)          rliq = 0.
  4612)       end if
  4613)       rvap = rtot-rliq-rice
  4614) 
  4615) 
  4616)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4617)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4618)       !write (unit=46,fmt='(a)') '--------------------------------------------------------'
  4619)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4620)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4621) 
  4622) 
  4623)       !------------------------------------------------------------------------------------!
  4624)       !     Find the function. We are seeking a temperature which is associated with the   !
  4625)       ! theta_il we provided. Thus, the function is simply the difference between the      !
  4626)       ! theta_il associated with our guess and the actual theta_il.                        !
  4627)       !------------------------------------------------------------------------------------!
  4628)       funnow = theta_iceliq(exner,tempz,rliq,rice)
  4629)       !----- Updating the derivative. -----------------------------------------------------!
  4630)       deriv  = dthetail_dt(.false.,funnow,pres,tempz,rliq,rice)
  4631)       funnow = funnow - thil
  4632)       fun1st = funnow
  4633)       !------------------------------------------------------------------------------------!
  4634) 
  4635)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4636)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4637)       !write (unit=46,fmt='(a,1x,i5,1x,6(a,1x,f11.4,1x),a,1x,es11.4,1x)')                  &
  4638)       !   'NEWTON: it=',0,'temp=',tempz-t00,'rsat=',1000.*rsat,'rliq=',1000.*rliq          &
  4639)       !  ,'rice=',1000.*rice,'rvap=',1000.*rvap,'fun=',funnow,'deriv=',deriv
  4640)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4641)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4642) 
  4643)       !------------------------------------------------------------------------------------!
  4644)       !    Now we enter at the Newton's method iterative loop. We are always going to try  !
  4645)       ! this first, because it's fast, but if it turns out to be a dangerous choice or if  !
  4646)       ! it doesn't converge fast, we will fall back to regula falsi.                       !
  4647)       !    We start by initialising the flag and copying temp to tempz, the newest guess.  !
  4648)       !------------------------------------------------------------------------------------!
  4649)       converged=.false.
  4650)       newloop: do itn=1,maxfpo/6
  4651)          !---------------------------------------------------------------------------------!
  4652)          !     Saving previous guess. We also save the function is in case we give up on   !
  4653)          ! Newton's and switch to regula falsi.                                            !
  4654)          !---------------------------------------------------------------------------------!
  4655)          funa  = funnow
  4656)          tempa = tempz
  4657) 
  4658)          !----- Go to bisection if the derivative is too flat (too dangerous...) ----------!
  4659)          if (abs(deriv) < toler) exit newloop
  4660) 
  4661)          tempz = tempa - funnow / deriv
  4662) 
  4663)          !----- Finding the mixing ratios associated with this guess ----------------------!
  4664)          rsat  = max(toodry,rslif(pres,tempz))
  4665)          if (tempz >= t3ple) then
  4666)             rliq = max(0.,rtot-rsat)
  4667)             rice = 0.
  4668)          else
  4669)             rice = max(0.,rtot-rsat)
  4670)             rliq = 0.
  4671)          end if
  4672)          rvap = rtot-rliq-rice
  4673) 
  4674)          !----- Updating the function -----------------------------------------------------!
  4675)          funnow = theta_iceliq(exner,tempz,rliq,rice)
  4676)          !----- Updating the derivative. --------------------------------------------------!
  4677)          deriv  = dthetail_dt(.false.,funnow,pres,tempz,rliq,rice)
  4678)          funnow = funnow - thil
  4679) 
  4680)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4681)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4682)          !write (unit=46,fmt='(a,1x,i5,1x,6(a,1x,f11.4,1x),a,1x,es11.4,1x)')               &
  4683)          !   'NEWTON: it=',itn,'temp=',tempz-t00,'rsat=',1000.*rsat,'rliq=',1000.*rliq     &
  4684)          !  ,'rice=',1000.*rice,'rvap=',1000.*rvap,'fun=',funnow,'deriv=',deriv
  4685)          !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4686)          !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4687) 
  4688)          converged = abs(tempa-tempz) < toler*tempz
  4689)          !---------------------------------------------------------------------------------!
  4690)          !   Convergence. The temperature will be the mid-point between tempa and tempz.   !
  4691)          ! Fix the mixing ratios and return. But first check for converged due to luck. If !
  4692)          ! the guess gives a root, then that's it. It looks unlikely, but it actually      !
  4693)          ! happens sometimes and if not checked it becomes a singularity.                  !
  4694)          !---------------------------------------------------------------------------------!
  4695)          if (funnow == 0.) then
  4696)             temp = tempz
  4697)             converged = .true.
  4698)             exit newloop
  4699)          elseif (converged) then
  4700)             temp = 0.5 * (tempa+tempz)
  4701)             rsat  = max(toodry,rslif(pres,temp))
  4702)             if (temp >= t3ple) then
  4703)                rliq = max(0.,rtot-rsat)
  4704)                rice = 0.
  4705)             else
  4706)                rice = max(0.,rtot-rsat)
  4707)                rliq = 0.
  4708)             end if
  4709)             rvap = rtot-rliq-rice
  4710)             exit newloop
  4711)          end if
  4712) 
  4713)       end do newloop
  4714)       !------------------------------------------------------------------------------------!
  4715) 
  4716)       !----- For debugging only -----------------------------------------------------------!
  4717)       itb = itn+1
  4718) 
  4719)       if (.not. converged) then
  4720)          !---------------------------------------------------------------------------------!
  4721)          !    If I reach this point, then it means that Newton's method failed finding the !
  4722)          ! equilibrium, so we are going to use the regula falsi instead.  If Newton's      !
  4723)          ! method didn't converge, we use tempa as one guess and now we seek a tempz with  !
  4724)          ! opposite sign.                                                                  !
  4725)          !---------------------------------------------------------------------------------!
  4726)          !----- Check funa and funnow have opposite signs. If so, we are ready to go ------!
  4727)          if (funa*funnow < 0.0) then
  4728)             funz  = funnow
  4729)             zside = .true.
  4730)          !----- Otherwise, checking whether the 1st guess had opposite sign. --------------!
  4731)          elseif (funa*fun1st < 0.0) then
  4732)             funz  = fun1st
  4733)             zside = .true.
  4734)          !---------------------------------------------------------------------------------!
  4735)          !    Looking for a guess. Extrapolate funa linearly, trying to get the -funa.  We !
  4736)          ! don't need it to be funa, just with the opposite sign.  If that's not enough,   !
  4737)          ! we keep going further... Force the guesses to be at least 1K apart              !
  4738)          !---------------------------------------------------------------------------------!
  4739)          else
  4740)             if (abs(funnow-funa) < 100.*toler*tempa) then
  4741)                delta = 100.*toler*tempa
  4742)             else
  4743)                delta = max(abs(funa)*abs((tempz-tempa)/(funnow-funa)),100.*toler*tempa)
  4744)             end if
  4745)             tempz = tempa + delta
  4746)             funz  = funa
  4747)             !----- Just to enter at least once. The 1st time tempz=tempa-2*delta ----------!
  4748)             zside = .false.
  4749)             zgssloop: do itb=1,maxfpo
  4750)                 tempz = tempa + real((-1)**itb * (itb+3)/2) * delta
  4751)                 rsat   = max(toodry,rslif(pres,tempz))
  4752)                 if (tempz >= t3ple) then
  4753)                    rliq = max(0.,rtot-rsat)
  4754)                    rice = 0.
  4755)                 else
  4756)                    rice = max(0.,rtot-rsat)
  4757)                    rliq = 0.
  4758)                 end if
  4759)                 rvap = rtot-rliq-rice
  4760)                 funz = theta_iceliq(exner,tempz,rliq,rice) - thil
  4761)                 zside = funa*funz < 0.0
  4762)                 if (zside) exit zgssloop
  4763)             end do zgssloop
  4764)             if (.not. zside) then

  4766)                write (unit=*,fmt='(a)')           ' THIL2TQALL: NO SECOND GUESS FOR YOU!'
  4767)                write (unit=*,fmt='(a)')           ' '
  4768)                write (unit=*,fmt='(a)')           ' -> Input: '
  4769)                write (unit=*,fmt='(a,1x,f12.5)')  '    THETA_IL [     K]:',thil
  4770)                write (unit=*,fmt='(a,1x,f12.5)')  '    PRESS    [   hPa]:',0.01*pres
  4771)                write (unit=*,fmt='(a,1x,f12.5)')  '    EXNER    [J/kg/K]:',exner
  4772)                write (unit=*,fmt='(a,1x,f12.5)')  '    RTOT     [  g/kg]:',1000.*rtot
  4773)                write (unit=*,fmt='(a,1x,f12.5)')  '    T1ST     [  degC]:',t1stguess-t00
  4774)                write (unit=*,fmt='(a,1x,f12.5)')  '    TEMPA    [  degC]:',tempa-t00
  4775)                write (unit=*,fmt='(a,1x,f12.5)')  '    TEMPZ    [  degC]:',tempz-t00
  4776)                write (unit=*,fmt='(a,1x,f12.5)')  '    FUNNOW   [     K]:',funnow
  4777)                write (unit=*,fmt='(a,1x,f12.5)')  '    FUNA     [     K]:',funa
  4778)                write (unit=*,fmt='(a,1x,f12.5)')  '    FUNZ     [     K]:',funz
  4779)                write (unit=*,fmt='(a,1x,f12.5)')  '    DELTA    [     K]:',delta
  4780)                write (unit=*,fmt='(60a1)')        ('-',ii=1,60)
  4781) 
  4782)                call fatal_error('Failed finding the second guess for regula falsi'         &
  4783)                                ,'thil2tqall','therm_lib.f90')
  4784)             end if
  4785)          end if
  4786)          !---------------------------------------------------------------------------------!
  4787) 
  4788)          !---------------------------------------------------------------------------------!
  4789)          !     Now we loop until convergence is achieved.  One important thing to notice   !
  4790)          ! is that Newton's method fail only when T is almost T3ple, which means that ice  !
  4791)          ! and liquid should be present, and we are trying to find the saturation point    !
  4792)          ! with all ice or all liquid. This will converge but the final answer will        !
  4793)          ! contain significant error. To reduce it we redistribute the condensates between !
  4794)          ! ice and liquid conserving the total condensed mixing ratio.                     !
  4795)          !---------------------------------------------------------------------------------!
  4796)          fpoloop: do itb=itn,maxfpo
  4797)             temp = (funz*tempa-funa*tempz)/(funz-funa)
  4798)             !----- Checking whether this guess will fall outside the range ----------------!
  4799)             if (abs(temp-tempa) > abs(tempz-tempa) .or.                                    &
  4800)                 abs(temp-tempz) > abs(tempz-tempa)) then
  4801)                temp = 0.5*(tempa+tempz)
  4802)             end if
  4803)             !----- Distributing vapour into the three phases ------------------------------!
  4804)             rsat   = max(toodry,rslif(pres,temp))
  4805)             rvap   = min(rtot,rsat)
  4806)             if (temp >= t3ple) then
  4807)                rliq = max(0.,rtot-rsat)
  4808)                rice = 0.
  4809)             else
  4810)                rliq = 0.
  4811)                rice = max(0.,rtot-rsat)
  4812)             end if
  4813)             !----- Updating function ------------------------------------------------------!
  4814)             funnow = theta_iceliq(exner,temp,rliq,rice) - thil
  4815) 
  4816)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4817)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4818)             !write (unit=46,fmt='(a,1x,i5,1x,10(a,1x,f11.4,1x))')                          &
  4819)             !   'REGFAL: it=',itb,'temp=',temp-t00,'tempa=',tempa-t00,'tempz=',tempz-t00   &
  4820)             !  ,'rsat=',1000.*rsat,'rliq=',1000.*rliq,'rice=',1000.*rice                   &
  4821)             !  ,'rvap=',1000.*rvap,'fun=',funnow,'funa=',funa,'funz=',funz
  4822)             !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4823)             !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4824) 
  4825)             !------------------------------------------------------------------------------!
  4826)             !    Checking for convergence or lucky guess. If it did, return, we found the  !
  4827)             ! solution. Otherwise, constrain the guesses.                                  !
  4828)             !------------------------------------------------------------------------------!
  4829)             converged = abs(temp-tempa) < toler*temp .and. abs(temp-tempz) < toler*temp
  4830)             if (funnow == 0. .or. converged) then
  4831)                converged = .true.
  4832)                exit fpoloop
  4833)             elseif (funnow*funa < 0.) then
  4834)                tempz = temp
  4835)                funz  = funnow
  4836)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  4837)                if (zside) funa=funa * 0.5
  4838)                !----- We just updated zside, setting zside to true. -----------------------!
  4839)                zside = .true.
  4840)             elseif (funnow*funz < 0.) then
  4841)                tempa = temp
  4842)                funa  = funnow
  4843)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  4844)                if (.not.zside) funz = funz * 0.5
  4845)                !----- We just updated aside, setting zside to false -----------------------!
  4846)                zside = .false.
  4847)             end if
  4848)             !------------------------------------------------------------------------------!
  4849)          end do fpoloop
  4850)          !---------------------------------------------------------------------------------!
  4851) 
  4852) 
  4853)          !---------------------------------------------------------------------------------!
  4854)          !    Almost done... Usually when the method goes through regula falsi, it means   !
  4855)          ! that the temperature is too close to the triple point, and often all three      !
  4856)          ! phases will coexist.  The problem with the method is that it converges for      !
  4857)          ! temperature, but whenever regula falsi is called the function evaluation is     !
  4858)          ! usually far from zero.  This can be improved by finding a better partition      !
  4859)          ! between ice and liquid given the temperature and saturation mixing ratio we     !
  4860)          ! just found. So just to round these edges, we will invert the ice-liquid         !
  4861)          ! potential temperature using the set of temperature and rsat, and fiding the     !
  4862)          ! liquid mixing ratio.                                                            !
  4863)          !---------------------------------------------------------------------------------!
  4864)          if (abs(temp-t3ple) < toler*temp) then
  4865)             !----- Find rliq. -------------------------------------------------------------!
  4866)             rliq   = ( alvi(temp) * (rtot - rsat)                                          &
  4867)                      + cpdry * temp * log ( cpdryi * exner * thil / temp ) )               &
  4868)                    / ( alvi(temp) - alvl(temp) )
  4869)             !------------------------------------------------------------------------------!
  4870) 
  4871)             !----- Correct rliq so it is bounded, and then find rice as the remainder. ----!
  4872)             rliq   = max( 0., min( rtot - rsat,  rliq ) )
  4873)             rice   = max( 0., rtot-rsat-rliq)
  4874)             !------------------------------------------------------------------------------!
  4875) 
  4876) 
  4877)             !----- Function evaluation. ---------------------------------------------------!
  4878)             funnow = theta_iceliq(exner,temp,rliq,rice) - thil
  4879)             !------------------------------------------------------------------------------!
  4880)          end if
  4881)          !---------------------------------------------------------------------------------!
  4882)          itb=itb+1
  4883)       end if
  4884)       !------------------------------------------------------------------------------------!
  4885) 
  4886)       if (.not. converged) then
  4887)          write (unit=*,fmt='(60a1)')        ('-',ii=1,60)
  4888)          write (unit=*,fmt='(a)')           ' THIL2TQALL failed!'
  4889)          write (unit=*,fmt='(a)')           ' '
  4890)          write (unit=*,fmt='(a)')           ' -> Input: '
  4891)          write (unit=*,fmt='(a,1x,f12.5)')  '    THETA_IL [     K]:',thil
  4892)          write (unit=*,fmt='(a,1x,f12.5)')  '    EXNER    [J/kg/K]:',exner
  4893)          write (unit=*,fmt='(a,1x,f12.5)')  '    RTOT     [  g/kg]:',1000.*rtot
  4894)          write (unit=*,fmt='(a)')           ' '
  4895)          write (unit=*,fmt='(a)')           ' -> Output: '
  4896)          write (unit=*,fmt='(a,1x,i12)')    '    ITERATIONS       :',itb
  4897)          write (unit=*,fmt='(a,1x,f12.5)')  '    TEMP     [    C]:',temp-t00
  4898)          write (unit=*,fmt='(a,1x,f12.5)')  '    RVAP     [  g/kg]:',1000.*rvap
  4899)          write (unit=*,fmt='(a,1x,f12.5)')  '    RLIQ     [  g/kg]:',1000.*rliq
  4900)          write (unit=*,fmt='(a,1x,f12.5)')  '    RICE     [  g/kg]:',1000.*rice
  4901)          write (unit=*,fmt='(a,1x,f12.5)')  '    TEMPA    [    C]:',tempa-t00
  4902)          write (unit=*,fmt='(a,1x,f12.5)')  '    TEMPZ    [    C]:',tempz-t00
  4903)          write (unit=*,fmt='(a,1x,es12.5)') '    FUNA     [     K]:',funnow
  4904)          write (unit=*,fmt='(a,1x,es12.5)') '    FUNZ     [     K]:',funnow
  4905)          write (unit=*,fmt='(a,1x,es12.5)') '    DERIV    [   ---]:',deriv
  4906)          write (unit=*,fmt='(a,1x,es12.5)') '    ERR_A    [   ---]:',abs(temp-tempa)/temp
  4907)          write (unit=*,fmt='(a,1x,es12.5)') '    ERR_Z    [   ---]:',abs(temp-tempz)/temp
  4908)          write (unit=*,fmt='(a)')           ' '
  4909)          write (unit=*,fmt='(60a1)')        ('-',ii=1,60)
  4910)          call fatal_error('Failed finding equilibrium, I gave up!','thil2tqall'            &
  4911)                          ,'therm_lib.f90')
  4912)       end if

  4914)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4915)       !write (unit=46,fmt='(a,1x,i5,1x,6(a,1x,f11.4,1x))')                                 &
  4916)       !   'ANSWER: it=',itb,'funf=',funnow,'temp=',temp-t00                                &
  4917)       !  ,'rsat=',1000.*rsat,'rliq=',1000.*rliq,'rice=',1000.*rice,'rvap=',1000.*rvap
  4918)       !write (unit=46,fmt='(a)') '----------------------------------------------------------'
  4919)       !<><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>!
  4920)       !><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><!
  4921)       return

  4933)    !     This subroutine computes a consistent set of temperature and condensated phases   !
  4934)    ! mixing ratio for a given theta_il, Exner function, and total mixing ratio.  This is   !
  4935)    ! very similar to the function thil2temp, except that now we don't know rliq.  Rliq     !
  4936)    ! becomes a function of temperature, since it is defined as rtot-rsat(T,p), remembering !
  4937)    ! that rtot and p are known. If the air is not saturated, we use the fact that theta_il !
  4938)    ! is theta and skip the hassle.  Otherwise, we use iterative methods.  We will always   !
  4939)    ! try Newton's method, since it converges fast.  Not always will Newton converge, and   !
  4940)    ! if that's the case we use a modified regula falsi (Illinois) method.  This method is  !
  4941)    ! a mix of secant and bisection and will always converge.                               !
  4942)    !---------------------------------------------------------------------------------------!
  4943)    subroutine thil2tqliq(thil,exner,pres,rtot,rliq,temp,rvap,rsat)
  4944)       use consts_coms, only : cpdryi   & ! intent(in)
  4945)                             , toodry   ! ! intent(in)
  4946)       implicit none
  4947)       !----- Arguments. -------------------------------------------------------------------!
  4948)       real(kind=4), intent(in)    :: thil      ! Ice-liquid water potential temp.  [     K]
  4949)       real(kind=4), intent(in)    :: exner     ! Exner function                    [J/kg/K]
  4950)       real(kind=4), intent(in)    :: pres      ! Pressure                          [    Pa]
  4951)       real(kind=4), intent(in)    :: rtot      ! Total mixing ratio                [ kg/kg]
  4952)       real(kind=4), intent(out)   :: rliq      ! Liquid water mixing ratio         [ kg/kg]
  4953)       real(kind=4), intent(inout) :: temp      ! Temperature                       [     K]
  4954)       real(kind=4), intent(out)   :: rvap      ! Water vapour mixing ratio         [ kg/kg]
  4955)       real(kind=4), intent(out)   :: rsat      ! Sat. water vapour mixing ratio    [ kg/kg]
  4956)       !----- Local variables --------------------------------------------------------------!
  4957)       real(kind=4)                :: tempa     ! Lower bound for regula falsi iteration
  4958)       real(kind=4)                :: tempz     ! Upper bound for regula falsi iteration
  4959)       real(kind=4)                :: funa      ! Function evaluation at tempa
  4960)       real(kind=4)                :: funz      ! Function evaluation at tempz
  4961)       real(kind=4)                :: funnow    ! Function at this iteration.
  4962)       real(kind=4)                :: delta     ! Aux. var in case we need regula falsi.
  4963)       real(kind=4)                :: deriv     ! Derivative of this function.
  4964)       integer                     :: itn       ! Iteration counter
  4965)       integer                     :: itb       ! Iteration counter
  4966)       logical                     :: converged ! Convergence handle
  4967)       logical                     :: zside     ! Aux. Flag, for two purposes:
  4968)                                                ! 1. Found a 2nd guess for regula falsi.
  4969)                                                ! 2. I retained the "zside" (T/F)
  4970)       !------------------------------------------------------------------------------------!
  4971) 
  4972) 
  4973) 
  4974) 
  4975)       !------------------------------------------------------------------------------------!
  4976)       !      First check: try to find temperature assuming sub-saturation and check if     !
  4977)       ! this is the case. If it is, then there is no need to go through the iterative      !
  4978)       ! loop.                                                                              !
  4979)       !------------------------------------------------------------------------------------!
  4980)       tempz = cpdryi * thil * exner
  4981)       rsat  = max(toodry,rslf(pres,tempz))
  4982)       rliq  = max(0.,rtot-rsat)
  4983)       rvap  = rtot-rliq
  4984)       !------------------------------------------------------------------------------------!
  4985) 
  4986) 
  4987) 
  4988)       !------------------------------------------------------------------------------------!
  4989)       !    If rtot < rsat, this is not saturated, we can leave the subroutine and bypass   !
  4990)       ! the iterative part.                                                                !
  4991)       !------------------------------------------------------------------------------------!
  4992)       if (rtot < rsat) then
  4993)          temp = tempz
  4994)          return
  4995)       end if
  4996)       !------------------------------------------------------------------------------------!
  4997) 
  4998) 
  4999) 
  5000)       !------------------------------------------------------------------------------------!
  5001)       !   If not, then use the temperature the user gave as first guess and solve          !
  5002)       ! iteratively.  We use the user instead of what we just found because if the air is  !
  5003)       ! saturated, then this can be too far off which may be bad for Newton's method.      !
  5004)       !------------------------------------------------------------------------------------!
  5005)       tempz = temp
  5006)       rsat   = max(toodry,rslf(pres,tempz))
  5007)       rliq = max(0.,rtot-rsat)
  5008)       rvap = rtot-rliq
  5009)       !------------------------------------------------------------------------------------!
  5010) 
  5011) 
  5012)       !------------------------------------------------------------------------------------!
  5013)       !     Finding the function. We are seeking a temperature which is associated with    !
  5014)       ! the theta_il we provided. Thus, the function is simply the difference between the  !
  5015)       ! theta_il associated with our guess and the actual theta_il.                        !
  5016)       !------------------------------------------------------------------------------------!
  5017)       funnow = theta_iceliq(exner,tempz,rliq,0.) ! Finding thil from our guess
  5018)       deriv  = dthetail_dt(.false.,funnow,pres,tempz,rliq)
  5019)       funnow = funnow - thil ! Computing the function
  5020)       !------------------------------------------------------------------------------------!
  5021) 
  5022) 
  5023) 
  5024)       !------------------------------------------------------------------------------------!
  5025)       !    Now we enter at the Newton's method iterative loop. We are always going to try  !
  5026)       ! this first, because it's fast, but if it turns out to be a dangerous choice or if  !
  5027)       ! it doesn't converge fast, we will fall back to regula falsi.                       !
  5028)       !    We start by initialising the flag and copying temp to tempz, the newest guess.  !
  5029)       !------------------------------------------------------------------------------------!
  5030)       converged=.false.
  5031)       newloop: do itn=1,maxfpo/6
  5032)          !---------------------------------------------------------------------------------!
  5033)          !     Saving previous guess. We also save the function is in case we give up on   !
  5034)          ! Newton's and switch to regula falsi.                                            !
  5035)          !---------------------------------------------------------------------------------!
  5036)          funa  = funnow
  5037)          tempa = tempz
  5038) 
  5039)          !----- Go to bisection if the derivative is too flat (too dangerous...) ----------!
  5040)          if (abs(deriv) < toler) exit newloop
  5041) 
  5042)          tempz = tempa - funnow / deriv
  5043) 
  5044)          !----- Finding the mixing ratios associated with this guess ----------------------!
  5045)          rsat  = max(toodry,rslf(pres,tempz))
  5046)          rliq = max(0.,rtot-rsat)
  5047)          rvap = rtot-rliq
  5048) 
  5049)          !----- Updating the function -----------------------------------------------------!
  5050)          funnow = theta_iceliq(exner,tempz,rliq,0.)
  5051)          !----- Updating the derivative. --------------------------------------------------!
  5052)          deriv = dthetail_dt(.false.,funnow,pres,tempz,rliq)
  5053)          funnow = funnow - thil
  5054) 
  5055)          converged = abs(tempa-tempz) < toler*tempz
  5056)          !---------------------------------------------------------------------------------!
  5057)          !   Convergence. The temperature will be the mid-point between tempa and tempz.   !
  5058)          ! Fix the mixing ratios and return. But first check for converged due to luck. If !
  5059)          ! the guess gives a root, then that's it. It looks unlikely, but it actually      !
  5060)          ! happens sometimes and if not checked it becomes a singularity.                  !
  5061)          !---------------------------------------------------------------------------------!
  5062)          if (funnow == 0.) then
  5063)             temp = tempz
  5064)             converged = .true.
  5065)             exit newloop
  5066)          elseif (converged) then
  5067)             temp = 0.5 * (tempa+tempz)
  5068)             rsat  = max(toodry,rslf(pres,temp))
  5069)             rliq = max(0.,rtot-rsat)
  5070)             rvap = rtot-rliq
  5071)             exit newloop
  5072)          end if
  5073)          !---------------------------------------------------------------------------------!
  5074)       end do newloop
  5075)       !---------------------------------------------------------------------------------------!
  5076) 
  5077) 
  5078) 
  5079)       !---------------------------------------------------------------------------------------!
  5080)       if (.not. converged) then
  5081)          !---------------------------------------------------------------------------------!
  5082)          !    If I reach this point, then it means that Newton's method failed finding the !
  5083)          ! equilibrium, so we are going to use the regula falsi instead.  If Newton's      !
  5084)          ! method didn't converge, we use tempa as one guess and now we seek a tempz with  !
  5085)          ! opposite sign.                                                                  !
  5086)          !---------------------------------------------------------------------------------!
  5087)          !----- Check funa and funnow have opposite signs. If so, we are ready to go ------!
  5088)          if (funa*funnow < 0.0) then
  5089)             funz = funnow
  5090)             zside = .true.
  5091)          !---------------------------------------------------------------------------------!
  5092)          !    Looking for a guess. Extrapolate funa linearly, trying to get the -funa.  We !
  5093)          ! don't need it to be funa, just with the opposite sign.  If that's not enough,   !
  5094)          ! we keep going further... Force the guesses to be at least 1K apart              !
  5095)          !---------------------------------------------------------------------------------!
  5096)          else
  5097)             if (abs(funnow-funa) < toler*tempa) then
  5098)                delta = 100.*toler*tempa
  5099)             else
  5100)                delta = max(abs(funa*(tempz-tempa)/(funnow-funa)),100.*toler*tempa)
  5101)             end if
  5102)             tempz = tempa + delta
  5103)             funz  = funa
  5104)             !----- Just to enter at least once. The 1st time tempz=tempa-2*delta ----------!
  5105)             zside = .false.
  5106)             zgssloop: do itb=1,maxfpo
  5107)                tempz = tempz + real((-1)**itb * (itb+3)/2) * delta
  5108)                rsat  = max(toodry,rslf(pres,tempz))
  5109)                rliq  = max(0.,rtot-rsat)
  5110)                rvap  = rtot-rliq
  5111)                funz  = theta_iceliq(exner,tempz,rliq,0.) - thil
  5112)                zside = funa*funz < 0.0
  5113)                if (zside) exit zgssloop
  5114)             end do zgssloop
  5115)             if (.not. zside)                                                               &
  5116)                call fatal_error('Failed finding the second guess for regula falsi'         &
  5117)                                ,'thil2tqliq','rthrm.f90')
  5118)          end if
  5119)          !---------------------------------------------------------------------------------!
  5120) 
  5121)          !---------------------------------------------------------------------------------!
  5122)          !     Now we loop until convergence is achieved.                                  !
  5123)          !---------------------------------------------------------------------------------!
  5124)          fpoloop: do itb=itn,maxfpo
  5125)             temp = (funz*tempa-funa*tempz)/(funz-funa)
  5126)             !----- Distributing vapour into the three phases ------------------------------!
  5127)             rsat   = max(toodry,rslf(pres,temp))
  5128)             rvap   = min(rtot,rsat)
  5129)             rliq   = max(0.,rtot-rsat)
  5130)             !----- Updating function ------------------------------------------------------!
  5131)             funnow = theta_iceliq(exner,tempz,rliq,0.) - thil
  5132) 
  5133)             !------------------------------------------------------------------------------!
  5134)             !    Checking for convergence or lucky guess. If it did, return, we found the  !
  5135)             ! solution. Otherwise, constrain the guesses.                                  !
  5136)             !------------------------------------------------------------------------------!
  5137)             converged = abs(temp-tempa)< toler*temp  .and. abs(temp-tempz) < toler*temp
  5138)             if (funnow == 0. .or. converged) then
  5139)                converged = .true.
  5140)                exit fpoloop
  5141)             elseif (funnow*funa < 0.) then
  5142)                tempz = temp
  5143)                funz  = funnow
  5144)                !----- If we are updating zside again, modify aside (Illinois method) ------!
  5145)                if (zside) funa=funa * 0.5
  5146)                !----- We just updated zside, setting zside to true. -----------------------!
  5147)                zside = .true.
  5148)             else
  5149)                tempa = temp
  5150)                funa  = funnow
  5151)                !----- If we are updating aside again, modify zside (Illinois method) ------!
  5152)                if (.not.zside) funz = funz * 0.5
  5153)                !----- We just updated aside, setting zside to false -----------------------!
  5154)                zside = .false.
  5155)             end if
  5156) 
  5157)          end do fpoloop
  5158)
