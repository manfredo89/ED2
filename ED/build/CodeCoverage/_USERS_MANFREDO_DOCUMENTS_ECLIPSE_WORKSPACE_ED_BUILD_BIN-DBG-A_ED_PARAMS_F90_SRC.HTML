     3) !     This is the main loader of ecosystem parameters.  Since some compilers do not under- !
     4) ! stand the assignment in the modules when the variable is not a constant (parameter),     !
     5) ! this is the safest way to guarantee it will read something (not to mention that makes    !
     6) ! compilation much faster when you want to test the sensitivity of one number).            !
     7) !------------------------------------------------------------------------------------------!
     8) subroutine load_ed_ecosystem_params()
     9) 
    10)    use ed_max_dims , only : n_pft               ! ! intent(in)
    11)    use pft_coms    , only : include_these_pft   & ! intent(in)
    12)       , agri_stock          & ! intent(in)
    13)       , plantation_stock    & ! intent(in)
    14)       , pft_name16          & ! intent(out)
    15)       , is_tropical         & ! intent(out)
    16)       , is_grass            & ! intent(out)
    17)       , is_liana            & ! intent(out)
    18)       , include_pft         & ! intent(out)
    19)       , include_pft_ag      & ! intent(out)
    20)       , include_pft_fp      ! ! intent(out)
    21)    use disturb_coms, only : ianth_disturb       ! ! intent(in)
    22) 
    23)    implicit none
    24)    !----- Arguments -----------------------------------------------------------------------!
    25)    integer :: p
    26)    !---------------------------------------------------------------------------------------!
    27) 
    28) 
    29) 
    30)    !---------------------------------------------------------------------------------------!
    31)    !      Main table of Plant functional types.  If you add some PFT, please make sure     !
    32)    ! that you assign values for all PFT-dependent variables.  Below is a summary table of  !
    33)    ! the main characteristics of the currently available PFTs.                             !
    34)    !---------------------------------------------------------------------------------------!
    35)    !  PFT | Name                                       | Grass   | Tropical | agriculture? !
    36)    !------+--------------------------------------------+---------+----------+--------------!
    37)    !    1 | C4 grass                                   |     yes |      yes |          yes !
    38)    !    2 | Early tropical                             |      no |      yes |           no !
    39)    !    3 | Mid tropical                               |      no |      yes |           no !
    40)    !    4 | Late tropical                              |      no |      yes |           no !
    41)    !    5 | Temperate C3 grass                         |     yes |       no |          yes !
    42)    !    6 | Northern pines                             |      no |       no |           no !
    43)    !    7 | Southern pines                             |      no |       no |           no !
    44)    !    8 | Late conifers                              |      no |       no |           no !
    45)    !    9 | Early temperate deciduous                  |      no |       no |           no !
    46)    !   10 | Mid temperate deciduous                    |      no |       no |           no !
    47)    !   11 | Late temperate deciduous                   |      no |       no |           no !
    48)    !   12 | C3 pasture                                 |     yes |       no |          yes !
    49)    !   13 | C3 crop (e.g.,wheat, rice, soybean)        |     yes |       no |          yes !
    50)    !   14 | C4 pasture                                 |     yes |      yes |          yes !
    51)    !   15 | C4 crop (e.g.,corn/maize)                  |     yes |      yes |          yes !
    52)    !   16 | Tropical C3 grass                          |     yes |      yes |          yes !
    53)    !   17 | Liana                                      |      no |      yes |           no !
    54)    !------+--------------------------------------------+---------+----------+--------------!
    55) 
    56)    !----- Name the PFTs (no spaces, please). ----------------------------------------------!
    57)    pft_name16( 1) = 'C4_grass        '
    58)    pft_name16( 2) = 'Early_tropical  '
    59)    pft_name16( 3) = 'Mid_tropical    '
    60)    pft_name16( 4) = 'Late_tropical   '
    61)    pft_name16( 5) = 'C3_grass        '
    62)    pft_name16( 6) = 'North_pine      '
    63)    pft_name16( 7) = 'South_pine      '
    64)    pft_name16( 8) = 'Late_conifer    '
    65)    pft_name16( 9) = 'Early_hardwood  '
    66)    pft_name16(10) = 'Mid_hardwood    '
    67)    pft_name16(11) = 'Late_hardwood   '
    68)    pft_name16(12) = 'C3_pasture      '
    69)    pft_name16(13) = 'C3_crop         '
    70)    pft_name16(14) = 'C4_pasture      '
    71)    pft_name16(15) = 'C4_crop         '
    72)    pft_name16(16) = 'Subtrop_C3_grass'
    73)    pft_name16(17) = 'Liana           '
    74)    !---------------------------------------------------------------------------------------!
    75) 
    76) 
    77) 
    78)    !---------------------------------------------------------------------------------------!
    79)    !    This flag should be used to define whether the plant is tropical/subtropical or    !
    80)    ! not.                                                                                  !
    81)    !---------------------------------------------------------------------------------------!
    82)    is_tropical(1:4)   = .true.
    83)    is_tropical(5:13)  = .false.
    84)    is_tropical(14:17) = .true.
    85)    !---------------------------------------------------------------------------------------!
    86) 
    87)    !---------------------------------------------------------------------------------------!
    88)    !           This flag is used to define whether the plant is a liana or not             !
    89)    !---------------------------------------------------------------------------------------!
    90)    is_liana(1:16)  = .false.
    91)    is_liana(17)    = .true.
    92)    !---------------------------------------------------------------------------------------!
    93) 
    94) 
    95)    !---------------------------------------------------------------------------------------!
    96)    !    This flag should be used to define whether the plant is tree or grass              !
    97)    !---------------------------------------------------------------------------------------!
    98)    is_grass(1)     = .true.
    99)    is_grass(2:4)   = .false.
   100)    is_grass(5)     = .true.
   101)    is_grass(6:11)  = .false.
   102)    is_grass(12:16) = .true.
   103)    is_grass(17)    = .false.
   104)    !---------------------------------------------------------------------------------------!
   105) 
   106) 
   107) 
   108)    !---------------------------------------------------------------------------------------!
   109)    !    Include_pft: flag specifying to whether you want to include a plant functional     !
   110)    !                 type (T) or whether you want it excluded (F) from the simulation.     !
   111)    !---------------------------------------------------------------------------------------!
   112)    include_pft    = .false.
   113)    do p=1,n_pft
   114)       if (include_these_pft(p) >  0 .and. include_these_pft(p) <= n_pft) then
   115)          include_pft(include_these_pft(p)) = .true.
   116)       end if
   117)    end do
   118)    !---------------------------------------------------------------------------------------!
   119) 
   120) 
   121) 
   122) 
   123)    !---------------------------------------------------------------------------------------!
   124)    !     Only the PFTs listed in agri_stock are allowed in agriculture patches.  For the   !
   125)    ! time being this means a single PFT, but it could change in the future.                !
   126)    !---------------------------------------------------------------------------------------!
   127)    include_pft_ag             = .false.
   128)    include_pft_ag(agri_stock) = is_grass(agri_stock) .and. include_pft(agri_stock)
   129)    !---------------------------------------------------------------------------------------!
   130) 
   131) 
   132) 
   133) 
   134)    !---------------------------------------------------------------------------------------!
   135)    !     Only the PFTs listed in plantation_stock are allowed in agriculture patches.  For !
   136)    ! the time being this means a single PFT, but it could change in the future.            !
   137)    !---------------------------------------------------------------------------------------!
   138)    include_pft_fp                   = .false.
   139)    include_pft_fp(plantation_stock) = ( .not. is_grass(plantation_stock) ) .and.           &
   140)       include_pft(plantation_stock)
   141)    !---------------------------------------------------------------------------------------!
   142) 
   143) 
   144) 
   145)    !---------------------------------------------------------------------------------------!
   146)    !      Warn the user in case the PFT choice for agriculture or forest plantation was    !
   147)    ! inconsistent.                                                                         !
   148)    !---------------------------------------------------------------------------------------!
   149)    if (count(include_pft_ag) == 0 .and. ianth_disturb == 1) then
   150)       call warning ('PFT defined in agri_stock is not included in include_these_pft,'//    &
   151)          ' your croplands will be barren and not very profitable...'            &
   152)          ,'load_ecosystem_params','ed_params.f90')
   153)    end if
   154)    if (count(include_pft_fp) == 0 .and. ianth_disturb == 1) then
   155)       call warning ('PFT defined in plantation_stock is not listed in include_these_pft,'//&
   156)          ' your forest plantation will be barren and not very profitable ...'   &
   157)          ,'load_ecosystem_params','ed_params.f90')
   158)    end if
   159)    !---------------------------------------------------------------------------------------!
   160) 
   161) 
   162) 
   163) 
   164)    !----- Load several parameters ---------------------------------------------------------!
   165)    call init_decomp_params()
   166)    call init_ff_coms()
   167)    call init_disturb_params()
   168)    call init_physiology_params()
   169)    call init_met_params()
   170)    call init_lapse_params()
   171)    call init_hydro_coms()
   172)    call init_soil_coms()
   173)    call init_phen_coms()
   174)    call init_ed_misc_coms()
   175)    !---------------------------------------------------------------------------------------!
   176) 
   177) 
   178)    !---------------------------------------------------------------------------------------!
   179)    !     Assign many PFT-dependent parameters.  Here the order may matter, so think twice  !
   180)    ! before changing the order.                                                            !
   181)    !---------------------------------------------------------------------------------------!
   182)    !----- Photosynthesis and leaf respiration. --------------------------------------------!
   183)    call init_pft_photo_params()
   184)    !----- Root and heterotrophic respiration. ---------------------------------------------!
   185)    call init_pft_resp_params()
   186)    !----- Allometry and some plant traits. ------------------------------------------------!
   187)    call init_pft_alloc_params()
   188)    !----- Mortality. ----------------------------------------------------------------------!
   189)    call init_pft_mort_params()
   190)    !----- Nitrogen. -----------------------------------------------------------------------!
   191)    call init_pft_nitro_params()
   192)    !----- Miscellaneous leaf properties. --------------------------------------------------!
   193)    call init_pft_leaf_params()
   194)    !----- Reproduction. -------------------------------------------------------------------!
   195)    call init_pft_repro_params()
   196)    !----- Miscellaneous parameters that depend on the previous ones. ----------------------!
   197)    call init_pft_derived_params()
   198)    !---------------------------------------------------------------------------------------!
   199) 
   200) 
   201) 
   202)    !---------------------------------------------------------------------------------------!
   203)    !     Assign canopy properties.  This must be done after defining the PFT stuff.        !
   204)    ! Again, check the routines and think twice before changing the order.                  !
   205)    !---------------------------------------------------------------------------------------!
   206)    !----- Canopy turbulence and aerodynamic resistance. -----------------------------------!
   207)    call init_can_air_params()
   208)    !----- Canopy radiation properties. ----------------------------------------------------!
   209)    call init_can_rad_params()
   210)    !----- Canopy splitting into height layers. --------------------------------------------!
   211)    call init_can_lyr_params()
   212)    !---------------------------------------------------------------------------------------!
   213) 
   214)    !---------------------------------------------------------------------------------------!
   215)    !     This should be always the last one, since it depends on variables assigned in the !
   216)    ! previous init_????_params.                                                            !
   217)    !---------------------------------------------------------------------------------------!
   218)    call init_rk4_params()
   219)    !---------------------------------------------------------------------------------------!
   220) 
   221)    return
   222) end subroutine load_ed_ecosystem_params

   233) !     This subroutine assigns values for some variables that are in ed_misc_coms, which    !
   234) ! wouldn't fit in any of the other categories.                                             !
   235) !------------------------------------------------------------------------------------------!
   236) subroutine init_ed_misc_coms
   237)    use ed_max_dims  , only : n_pft                & ! intent(in)
   238)       , n_dbh                & ! intent(in)
   239)       , n_age                ! ! intent(in)
   240)    use consts_coms  , only : erad                 & ! intent(in)
   241)       , pio180               ! ! intent(in)
   242)    use ed_misc_coms , only : burnin               & ! intent(out)
   243)       , outputMonth          & ! intent(out)
   244)       , restart_target_year  & ! intent(out)
   245)       , use_target_year      & ! intent(out)
   246)       , maxage               & ! intent(out)
   247)       , dagei                & ! intent(out)
   248)       , maxdbh               & ! intent(out)
   249)       , ddbhi                & ! intent(out)
   250)       , vary_elev            & ! intent(out)
   251)       , vary_hyd             & ! intent(out)
   252)       , vary_rad             & ! intent(out)
   253)       , max_thsums_dist      & ! intent(out)
   254)       , max_poihist_dist     & ! intent(out)
   255)       , max_poi99_dist       & ! intent(out)
   256)       , suppress_h5_warnings ! ! intent(out)
   257)    implicit none
   258) 
   259) 
   260)    !----- Flags that allow components of subgrid heterogeneity to be turned on/off --------!
   261)    vary_elev = 1
   262)    vary_rad  = 1
   263)    vary_hyd  = 1
   264)    !---------------------------------------------------------------------------------------!
   265) 
   266) 
   267)    !----- Number of years to ignore demography when starting a run. -----------------------!
   268)    burnin = 0
   269)    !---------------------------------------------------------------------------------------!
   270) 
   271) 
   272)    !----- Month to output the yearly files. -----------------------------------------------!
   273)    outputMonth = 6
   274)    !---------------------------------------------------------------------------------------!
   275) 
   276) 
   277)    !----- Year to read when parsing pss/css with multiple years. --------------------------!
   278)    restart_target_year = 2000
   279)    !---------------------------------------------------------------------------------------!
   280) 
   281) 
   282)    !----- Flag specifying whether to search for a target year in pss/css. -----------------!
   283)    use_target_year = 0
   284)    !---------------------------------------------------------------------------------------!
   285) 
   286) 
   287) 
   288)    !----- Maximum age [yr] to split into classes. -----------------------------------------!
   289)    maxage = 200.
   290)    !---------------------------------------------------------------------------------------!
   291) 
   292) 
   293) 
   294)    !----- Maximum DBH [cm] to be split into classes. --------------------------------------!
   295)    maxdbh = 100.
   296)    !---------------------------------------------------------------------------------------!
   297) 
   298) 
   299) 
   300)    !---------------------------------------------------------------------------------------!
   301)    !     The inverse of bin classes will depend on max??? and n_???, leaving one class for !
   302)    ! when the number exceeds the maximum.                                                  !
   303)    !---------------------------------------------------------------------------------------!
   304)    dagei = real(n_age-1) / maxage
   305)    ddbhi = real(n_dbh-1) / maxdbh
   306)    !---------------------------------------------------------------------------------------!
   307) 
   308) 
   309) 
   310)    !---------------------------------------------------------------------------------------!
   311)    !    Maximum distance to the current polygon that we still consider the file grid point !
   312)    ! to be representative of the polygon.  The value below is 1.25 degree at the Equator.  !
   313)    !---------------------------------------------------------------------------------------!
   314)    max_thsums_dist    = 1.25 * erad * pio180
   315)    !---------------------------------------------------------------------------------------!
   316) 
   317) 
   318) 
   319)    !---------------------------------------------------------------------------------------!
   320)    !      Alternative method for mixing 1 grid and POI's.  Only use the grid if their is   !
   321)    ! NOT an POI  within a user specified resolution.  Remember, this assumes there is only !
   322)    ! 1 gridded file, and it is the first file when ied_init_mode is set to 99  (Developer  !
   323)    ! use only).                                                                            !
   324)    !---------------------------------------------------------------------------------------!
   325)    max_poi99_dist     = 5.0 * erad * pio180
   326)    !---------------------------------------------------------------------------------------!
   327) 
   328) 
   329) 
   330)    !---------------------------------------------------------------------------------------!
   331)    !      This variable is used for the history start initialisation.  This sets the       !
   332)    ! maximum acceptable distance between the expected polygon and the polygon found in the !
   333)    ! history file.  Units: m.                                                              !
   334)    !---------------------------------------------------------------------------------------!
   335)    max_poihist_dist   = 250.
   336)    !---------------------------------------------------------------------------------------!
   337) 
   338) 
   339) 
   340)    !---------------------------------------------------------------------------------------!
   341)    !      If you don't want to read a million warnings about certain initialization        !
   342)    ! variables not being available in the history input file, set this to .true. .  It's   !
   343)    ! better for new users to see what is missing though.                                   !
   344)    !---------------------------------------------------------------------------------------!
   345)    suppress_h5_warnings = .true.
   346)    !---------------------------------------------------------------------------------------!
   347) 
   348) 
   349)    return
   350) end subroutine init_ed_misc_coms

   361) !     This subroutine defines the minimum and maximum acceptable values in the meteoro-    !
   362) ! logical forcing.                                                                         !
   363) !------------------------------------------------------------------------------------------!
   364) subroutine init_met_params()
   365)    use ed_misc_coms   , only : dtlsm           ! ! intent(in)
   366)    use met_driver_coms, only : rshort_min      & ! intent(out)
   367)       , rshort_max      & ! intent(out)
   368)       , rlong_min       & ! intent(out)
   369)       , rlong_max       & ! intent(out)
   370)       , dt_radinterp    & ! intent(out)
   371)       , atm_tmp_min     & ! intent(out)
   372)       , atm_tmp_max     & ! intent(out)
   373)       , atm_shv_min     & ! intent(out)
   374)       , atm_shv_max     & ! intent(out)
   375)       , atm_rhv_min     & ! intent(out)
   376)       , atm_rhv_max     & ! intent(out)
   377)       , atm_co2_min     & ! intent(out)
   378)       , atm_co2_max     & ! intent(out)
   379)       , prss_min        & ! intent(out)
   380)       , prss_max        & ! intent(out)
   381)       , pcpg_min        & ! intent(out)
   382)       , pcpg_max        & ! intent(out)
   383)       , vels_min        & ! intent(out)
   384)       , vels_max        & ! intent(out)
   385)       , geoht_min       & ! intent(out)
   386)       , geoht_max       & ! intent(out)
   387)       , print_radinterp & ! intent(out)
   388)       , vbdsf_file      & ! intent(out)
   389)       , vddsf_file      & ! intent(out)
   390)       , nbdsf_file      & ! intent(out)
   391)       , nddsf_file      ! ! intent(out)
   392) 
   393) 
   394)    !----- Minimum and maximum acceptable shortwave radiation [W/m�]. ----------------------!
   395)    rshort_min  = 0.
   396)    rshort_max  = 1500.
   397)    !----- Minimum and maximum acceptable longwave radiation [W/m�]. -----------------------!
   398)    rlong_min   = 40.
   399)    rlong_max   = 600.
   400)    !----- Minimum and maximum acceptable air temperature    [   K]. -----------------------!
   401)    atm_tmp_min = 184.     ! Lowest temperature ever measured, in Vostok Basin, Antarctica
   402)    atm_tmp_max = 331.     ! Highest temperature ever measured, in El Azizia, Libya
   403)    !----- Minimum and maximum acceptable air specific humidity [kg_H2O/kg_air]. -----------!
   404)    atm_shv_min = 1.e-6    ! That corresponds to a relative humidity of 0.1% at 1000hPa
   405)    atm_shv_max = 3.2e-2   ! That corresponds to a dew point of 32�C at 1000hPa.
   406)    !----- Minimum and maximum acceptable CO2 mixing ratio [�mol/mol]. ---------------------!
   407)    atm_co2_min = 100.     !
   408)    atm_co2_max = 1100.    !
   409)    !----- Minimum and maximum acceptable pressure [Pa]. -----------------------------------!
   410)    prss_min =  45000. ! It may crash if you run a simulation in Mt. Everest.
   411)    prss_max = 110000. ! It may crash if you run a simulation under water.
   412)    !----- Minimum and maximum acceptable precipitation rates [kg/m�/s]. -------------------!
   413)    pcpg_min     = 0.0     ! No negative precipitation is allowed
   414)    pcpg_max     = 0.1111  ! This is a precipitation rate of 400mm/hr.
   415)    !----- Minimum and maximum acceptable wind speed [m/s]. --------------------------------!
   416)    vels_min     =  0.0    ! No negative wind is acceptable.
   417)    vels_max     = 85.0    ! Maximum sustained winds recorded during Typhoon Tip (1970).
   418)    !----- Minimum and maximum reference heights [m]. --------------------------------------!
   419)    geoht_min    =   1.0   ! This should be above-canopy measurement, but 1.0 is okay for
   420)                           !     grasslands...
   421)    geoht_max    = 350.0   ! This should be not that much above the canopy, but tall towers
   422)                           !     do exist...
   423)    !---------------------------------------------------------------------------------------!
   424) 
   425)    !---------------------------------------------------------------------------------------!
   426)    !     Minimum and maximum acceptable relative humidity (fraction).  This is not going   !
   427)    ! cause the simulation to crash, instead it will just impose these numbers to the       !
   428)    ! meteorological forcing.                                                               !
   429)    !---------------------------------------------------------------------------------------!
   430)    atm_rhv_min = 5.e-3 ! 0.5%
   431)    atm_rhv_max = 1.0   ! 100.0%.  Although canopy air space can experience super-
   432)                        !    saturation, we don't allow the air above to be super-saturated.
   433)    !---------------------------------------------------------------------------------------!
   434) 
   435) 
   436) 
   437) 
   438)    !---------------------------------------------------------------------------------------!
   439)    !     Time step used to perform the daytime average of the secant of the zenith angle.  !
   440)    !---------------------------------------------------------------------------------------!
   441)    dt_radinterp = dtlsm    ! Value in seconds.
   442)    !---------------------------------------------------------------------------------------!
   443) 
   444) 
   445)    !---------------------------------------------------------------------------------------!
   446)    !   These variables control the detailed interpolation output (for debugging only).     !
   447)    !---------------------------------------------------------------------------------------!
   448)    print_radinterp = .false.
   449)    vbdsf_file      = 'visible_beam.txt'
   450)    vddsf_file      = 'visible_diff.txt'
   451)    nbdsf_file      = 'near_infrared_beam.txt'
   452)    nddsf_file      = 'near_infrared_diff.txt'
   453)    !---------------------------------------------------------------------------------------!
   454) 
   455)    return
   456) end subroutine init_met_params

   515) !------------------------------------------------------------------------------------------!

   517) 
   518)    use canopy_radiation_coms , only : ltrans_vis                  & ! intent(in)

   520)       , lreflect_vis                & ! intent(in)
   521)       , lreflect_nir                & ! intent(in)
   522)       , orient_tree                 & ! intent(in)
   523)       , orient_grass                & ! intent(in)
   524)       , clump_tree                  & ! intent(in)

   526)       , leaf_reflect_nir            & ! intent(in)
   527)       , leaf_trans_nir              & ! intent(in)
   528)       , leaf_reflect_vis            & ! intent(in)
   529)       , leaf_trans_vis              & ! intent(in)
   530)       , leaf_backscatter_tir        & ! intent(out)

  3531)       , dbh2h                & ! function
  3532)       , size2bl          & ! function
  3533)       , dbh2bd               ! ! function
  3534)    use ed_therm_lib         , only : calc_veg_hcap        ! ! function
  3535)    implicit none
  3536)    !----- Local variables. ----------------------------------------------------------------!
  3537)    integer                           :: ipft
  3538)    real                              :: dbh
  3539)    real                              :: huge_dbh
  3540)    real                              :: huge_height
  3541)    real                              :: bleaf_min
  3542)    real                              :: broot_min
  3543)    real                              :: bsapwood_min
  3544)    real                              :: balive_min
  3545)    real                              :: bdead_min
  3546)    real                              :: bleaf_max
  3547)    real                              :: broot_max
  3548)    real                              :: bsapwood_max
  3549)    real                              :: balive_max
  3550)    real                              :: bdead_max
  3551)    real                              :: bleaf_bl
  3552)    real                              :: broot_bl
  3553)    real                              :: bsapwood_bl
  3554)    real                              :: balive_bl
  3555)    real                              :: bdead_bl
  3556)    real                              :: leaf_hcap_min
  3557)    real                              :: wood_hcap_min
  3558)    real                              :: lai_min
  3559)    real                              :: min_plant_dens
  3560)    logical                           :: print_zero_table
  3561)    character(len=str_len), parameter :: zero_table_fn    = 'pft_sizes.txt'
  3562)    !---------------------------------------------------------------------------------------!
  3563) 
  3564)    !---------------------------------------------------------------------------------------!
  3565)    !     Decide whether to write the table with the sizes.                                 !
  3566)    !---------------------------------------------------------------------------------------!
  3567)    print_zero_table = btest(idetailed,5)
  3568)    !---------------------------------------------------------------------------------------!
  3569) 
  3570)    !---------------------------------------------------------------------------------------!
  3571)    !     The minimum recruitment size and the recruit carbon to nitrogen ratio.  Both      !
  3572)    ! parameters actually depend on which PFT we are solving, since grasses always have     !
  3573)    ! significantly less biomass.                                                           !
  3574)    !---------------------------------------------------------------------------------------!
  3575)    if (print_zero_table) then
  3576)       open  (unit=61,file=trim(zero_table_fn),status='replace',action='write')
  3577)       write (unit=61,fmt='(29(a,1x))')                '  PFT',        'NAME            '   &
  3578)          ,'     HGT_MIN','         DBH'               &
  3579)          ,'   BLEAF_MIN','   BROOT_MIN'               &
  3580)          ,'BSAPWOOD_MIN','  BALIVE_MIN'               &
  3581)          ,'   BDEAD_MIN','    BLEAF_BL'               &
  3582)          ,'    BROOT_BL',' BSAPWOOD_BL'               &
  3583)          ,'   BALIVE_BL','    BDEAD_BL'               &
  3584)          ,'   BLEAF_MAX','   BROOT_MAX'               &
  3585)          ,'BSAPWOOD_MAX','  BALIVE_MAX'               &
  3586)          ,'   BDEAD_MAX','   INIT_DENS'               &
  3587)          ,'MIN_REC_SIZE','MIN_COH_SIZE'               &
  3588)          ,' NEGL_NPLANT','         SLA'               &
  3589)          ,'VEG_HCAP_MIN','     LAI_MIN'               &
  3590)          ,'     HGT_MAX','    DBH_CRIT'               &
  3591)          ,' ONE_PLANT_C'
  3592)    end if
  3593)    min_plant_dens = 0.1 * minval(init_density)
  3594)    do ipft = 1,n_pft
  3595) 
  3596)       !----- Find the DBH and carbon pools associated with a newly formed recruit. --------!
  3597)       dbh          = h2dbh(hgt_min(ipft),ipft)
  3598)       bleaf_min    = size2bl(dbh,hgt_min(ipft),ipft)
  3599)       broot_min    = bleaf_min * q(ipft)
  3600)       bsapwood_min = bleaf_min * qsw(ipft) * hgt_min(ipft)
  3601)       balive_min   = bleaf_min + broot_min + bsapwood_min
  3602)       bdead_min    = dbh2bd(dbh,ipft)
  3603)       !------------------------------------------------------------------------------------!
  3604) 
  3605) 
  3606)       !------------------------------------------------------------------------------------!
  3607)       !   Find the maximum bleaf and bdead supported.  This is to find the negligible      !
  3608)       ! nplant so we ensure that the cohort is always terminated if its mortality rate is  !
  3609)       ! very high.                                                                         !
  3610)       !------------------------------------------------------------------------------------!
  3611)       huge_dbh     = 3. * dbh_crit(ipft)
  3612)       huge_height  = dbh2h(ipft, dbh_crit(ipft))
  3613)       bleaf_max    = size2bl(huge_dbh,huge_height,ipft)
  3614)       broot_max    = bleaf_max * q(ipft)
  3615)       bsapwood_max = bleaf_max * qsw(ipft) * huge_height
  3616)       balive_max   = bleaf_max + broot_max + bsapwood_max
  3617)       bdead_max    = dbh2bd(huge_dbh,ipft)
  3618)       !------------------------------------------------------------------------------------!
  3619) 
  3620) 
  3621)       !------------------------------------------------------------------------------------!
  3622)       !    Biomass of one individual plant at recruitment.                                 !
  3623)       !------------------------------------------------------------------------------------!
  3624)       bleaf_bl          = size2bl(dbh_bigleaf(ipft),hgt_min(ipft),ipft)
  3625)       broot_bl          = bleaf_bl * q(ipft)
  3626)       bsapwood_bl       = bleaf_bl * qsw(ipft) * hgt_max(ipft)
  3627)       balive_bl         = bleaf_bl + broot_bl + bsapwood_bl
  3628)       bdead_bl          = dbh2bd(dbh_bigleaf(ipft),ipft)
  3629)       !------------------------------------------------------------------------------------!
  3630) 
  3631) 
  3632)       !------------------------------------------------------------------------------------!
  3633)       !     Find the carbon value for one plant.  If SAS approximation, we define it as    !
  3634)       ! the carbon of a seedling, and for the big leaf case we assume the typical big leaf !
  3635)       ! plant size.                                                                        !
  3636)       !------------------------------------------------------------------------------------!
  3637)       select case (ibigleaf)
  3638)          case (0)
  3639)             one_plant_c(ipft) = bdead_min + balive_min
  3640)          case (1)
  3641)             one_plant_c(ipft) = bdead_bl  + balive_bl
  3642)       end select
  3643)       !------------------------------------------------------------------------------------!
  3644) 
  3645) 
  3646)       !------------------------------------------------------------------------------------!
  3647)       !    The definition of the minimum recruitment size is the minimum amount of biomass !
  3648)       ! in kgC/m� is available for new recruits.  For the time being we use the near-bare  !
  3649)       ! ground state value as the minimum recruitment size, but this may change depending  !
  3650)       ! on how well it goes.                                                               !
  3651)       !------------------------------------------------------------------------------------!
  3652)       min_recruit_size(ipft) = min_plant_dens * one_plant_c(ipft)
  3653)       !------------------------------------------------------------------------------------!
  3654) 
  3655) 
  3656)       !------------------------------------------------------------------------------------!
  3657)       !    Minimum size (measured as biomass of living and structural tissues) allowed in  !
  3658)       ! a cohort.  Cohorts with less biomass than this are going to be terminated.         !
  3659)       !------------------------------------------------------------------------------------!
  3660)       min_cohort_size(ipft)  = 0.1 * min_recruit_size(ipft)
  3661)       !------------------------------------------------------------------------------------!
  3662) 
  3663) 
  3664)       !------------------------------------------------------------------------------------!
  3665)       !    Seed_rain is the density of seedling that will be added from somewhere else.    !
  3666)       !------------------------------------------------------------------------------------!
  3667)       seed_rain(ipft)  = 0.1 * init_density(ipft)
  3668)       !------------------------------------------------------------------------------------!
  3669) 
  3670) 
  3671) 
  3672)       !------------------------------------------------------------------------------------!
  3673)       !    The following variable is the absolute minimum cohort population that a cohort  !
  3674)       ! can have.  This should be used only to avoid nplant=0, but IMPORTANT: this will    !
  3675)       ! lead to a ridiculously small cohort almost guaranteed to be extinct and SHOULD BE  !
  3676)       ! USED ONLY IF THE AIM IS TO ELIMINATE THE COHORT.                                   !
  3677)       !------------------------------------------------------------------------------------!
  3678)       negligible_nplant(ipft) = onesixth * min_cohort_size(ipft) / (bdead_max + balive_max)
  3679)       !------------------------------------------------------------------------------------!
  3680) 
  3681) 
  3682)       !----- Find the recruit carbon to nitrogen ratio. -----------------------------------!
  3683)       c2n_recruit(ipft)      = (balive_min + bdead_min)                                    &
  3684)          / (balive_min * ( f_labile(ipft) / c2n_leaf(ipft)             &
  3685)          + (1.0 - f_labile(ipft)) / c2n_stem(ipft))                    &
  3686)          + bdead_min/c2n_stem(ipft))
  3687)       !------------------------------------------------------------------------------------!
  3688) 
  3689) 
  3690) 
  3691)       !------------------------------------------------------------------------------------!
  3692)       !     The following variable is the minimum heat capacity of either the leaf, or the !
  3693)       ! branches, or the combined pool that is solved by the biophysics.  Value is in      !
  3694)       ! J/m2/K.  Because leaves are the pools that can determine the fate of the tree, and !
  3695)       ! all PFTs have leaves (but not branches), we only consider the leaf heat capacity   !
  3696)       ! only for the minimum value.                                                        !
  3697)       !------------------------------------------------------------------------------------!
  3698)       call calc_veg_hcap(bleaf_min,bdead_min,bsapwood_min,init_density(ipft),ipft          &
  3699)          ,leaf_hcap_min,wood_hcap_min)
  3700)       veg_hcap_min(ipft) = onesixth * leaf_hcap_min
  3701)       lai_min            = onesixth * init_density(ipft) * bleaf_min * sla(ipft)
  3702)       !------------------------------------------------------------------------------------!
  3703) 
  3704) 
  3705)       if (print_zero_table) then
  3706)          write (unit=61,fmt='(i5,1x,a16,1x,27(es12.5,1x))')                                &
  3707)             ipft,pft_name16(ipft),hgt_min(ipft)   &
  3708)             ,dbh,bleaf_min,broot_min,bsapwood_min  &
  3709)             ,balive_min,bdead_min,bleaf_bl         &
  3710)             ,broot_bl,bsapwood_bl,balive_bl        &
  3711)             ,bdead_bl,bleaf_max,broot_max          &
  3712)             ,bsapwood_max,balive_max,bdead_max     &
  3713)             ,init_density(ipft)                    &
  3714)             ,min_recruit_size(ipft)                &
  3715)             ,min_cohort_size(ipft)                 &
  3716)             ,negligible_nplant(ipft)               &
  3717)             ,sla(ipft),veg_hcap_min(ipft)          &
  3718)             ,lai_min,hgt_max(ipft),dbh_crit(ipft)  &
  3719)             ,one_plant_c(ipft)
  3720)       end if
  3721)       !------------------------------------------------------------------------------------!
  3722)    end do
  3723) 
  3724)    if (print_zero_table) then
  3725)       close (unit=61,status='keep')
  3726)    end if
  3727) 
  3728)    return
  3729) end subroutine init_pft_derived_params
