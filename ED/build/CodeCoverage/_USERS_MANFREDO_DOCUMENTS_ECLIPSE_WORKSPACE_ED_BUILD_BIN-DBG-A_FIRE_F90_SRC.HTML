     3) !    This subroutine will evaluate whether fire conditions exist, and if that is the case, !
     4) ! it will calculate the disturbance rate due to fire.                                      !
     5) !------------------------------------------------------------------------------------------!
     6) subroutine fire_frequency(cgrid)
     7)    use ed_state_vars , only : edtype                 & ! structure
     8)                             , polygontype            & ! structure
     9)                             , sitetype               & ! structure
    10)                             , patchtype              ! ! structure
    11)    use ed_misc_coms  , only : simtime                & ! intent(in)
    12)                             , current_time           & ! intent(in)
    13)                             , dtlsm                  ! ! intent(in)
    14)    use grid_coms     , only : nzg                    ! ! intent(in)
    15)    use soil_coms     , only : soil                   & ! intent(in)
    16)                             , dslz                   ! ! intent(in)
    17)    use disturb_coms  , only : include_fire           & ! intent(in)
    18)                             , sm_fire                & ! intent(in)
    19)                             , fire_dryness_threshold & ! intent(in)
    20)                             , k_fire_first           & ! intent(in)
    21)                             , fire_parameter         ! ! intent(in)
    22)    use consts_coms   , only : wdns                   & ! intent(in)
    23)                             , wdnsi                  & ! intent(in)
    24)                             , day_sec                ! ! intent(in)
    25)    implicit none
    26)    !----- Arguments -----------------------------------------------------------------------!
    27)    type(edtype)      , target     :: cgrid
    28)    !----- Local variables -----------------------------------------------------------------!
    29)    type(polygontype) , pointer    :: cpoly
    30)    type(sitetype)    , pointer    :: csite
    31)    type(patchtype)   , pointer    :: cpatch
    32)    type(simtime)                  :: lastmonth
    33)    integer                        :: ipy
    34)    integer                        :: isi
    35)    integer                        :: ipa
    36)    integer                        :: ico
    37)    integer                        :: imon
    38)    integer                        :: k
    39)    integer                        :: nsoil
    40)    real                           :: ndaysi
    41)    real                           :: normfac
    42)    real                           :: fire_wmass_threshold
    43)    real                           :: fire_intensity
    44)    real                           :: fuel
    45)    real                           :: ignition_rate
    46)    real                           :: mean_fire_intensity
    47)    real                           :: sum_pcpg
    48)    !---------------------------------------------------------------------------------------!
    49) 
    50) 
    51) 
    52)    !---------------------------------------------------------------------------------------!
    53)    !     Find the number of days of last month so we can normalise the integrated ground   !
    54)    ! water.                                                                                !
    55)    !---------------------------------------------------------------------------------------!
    56)    call lastmonthdate(current_time,lastmonth,ndaysi)
    57)    normfac = dtlsm * ndaysi / (day_sec)
    58)    !---------------------------------------------------------------------------------------!
    59) 
    60) 
    61)    !----- Current month. ------------------------------------------------------------------!
    62)    imon = current_time%month
    63)    !---------------------------------------------------------------------------------------!
    64) 
    65) 
    66)    !----- Loop over polygons and sites. ---------------------------------------------------!
    67)    polyloop: do ipy = 1,cgrid%npolygons
    68)       cpoly => cgrid%polygon(ipy)
    69) 
    70) 
    71)       !------------------------------------------------------------------------------------!
    72)       !     Loop over all sites.                                                           !
    73)       !------------------------------------------------------------------------------------!
    74)       siteloop: do isi = 1,cpoly%nsites
    75)          csite => cpoly%site(isi)
    76) 
    77)          !---------------------------------------------------------------------------------!
    78)          !     Find the total rainfall of the past year and reset the counter for this     !
    79)          ! month.                                                                          !
    80)          !---------------------------------------------------------------------------------!
    81)          sum_pcpg                         = sum(cpoly%avg_monthly_pcpg(:,isi))
    82)          cpoly%avg_monthly_pcpg(imon,isi) = 0.
    83)          !---------------------------------------------------------------------------------!
    84) 
    85) 
    86) 
    87)          !----- Initialize ignition rate and the mean fire intensity (site variables). ----!
    88)          ignition_rate       = 0.0
    89)          mean_fire_intensity = 0.0
    90)          !---------------------------------------------------------------------------------!
    91) 
    92)          patchloop: do ipa=1,csite%npatches
    93)             cpatch => csite%patch(ipa)
    94) 
    95)             !----- Normalise the monthly mean ground water. -------------------------------!
    96)             csite%avg_monthly_gndwater(ipa) = csite%avg_monthly_gndwater(ipa) * normfac
    97)             !------------------------------------------------------------------------------!
    98) 
    99) 
   100)             !----- Normalise the monthly mean ground water. -------------------------------!
   101)             csite%avg_monthly_waterdef(ipa) = max(0.0,csite%avg_monthly_waterdef(ipa))
   102)             !------------------------------------------------------------------------------!
   103) 
   104) 
   105)             !----- Initialize patch fuel. -------------------------------------------------!
   106)             fuel = 0.0
   107)             !------------------------------------------------------------------------------!
   108) 
   109) 
   110) 
   111)             !------------------------------------------------------------------------------!
   112)             !    Loop through all cohorts in this patch, and compute the fuel.  Fuel will  !
   113)             ! be defined as the above-ground biomass per unit area.                        !
   114)             !------------------------------------------------------------------------------!
   115)             cohortloop: do ico = 1,cpatch%ncohorts
   116) 
   117)                fuel = fuel + cpatch%nplant(ico) * cpatch%agb(ico)
   118) 
   119)             end do cohortloop
   120)             !------------------------------------------------------------------------------!
   121) 
   122) 
   123) 
   124)             !------------------------------------------------------------------------------!
   125)             !     Determine the correct threshold to ignite fires according to the fire    !
   126)             ! method.                                                                      !
   127)             !------------------------------------------------------------------------------!
   128)             select case (include_fire)
   129)             case (0)
   130)                !------ Set water mass threshold to infinity, so fires will never happen. --!
   131)                fire_wmass_threshold = huge(1.)
   132)                fire_intensity       = 0.
   133)                !---------------------------------------------------------------------------!
   134) 
   135)             case (1)
   136)                !---------------------------------------------------------------------------!
   137)                !     The fire threshold is equivalent to the dryness factor, converted to  !
   138)                ! kg/m2.  This will be compared to the full column, so if the soil is too   !
   139)                ! deep fires will be nearly impossible.                                     !
   140)                !---------------------------------------------------------------------------!
   141)                fire_wmass_threshold = fire_dryness_threshold * wdns
   142)                if (csite%avg_monthly_gndwater(ipa) < fire_wmass_threshold) then
   143)                   fire_intensity      = fire_parameter
   144)                   mean_fire_intensity = mean_fire_intensity                                &
   145)                                       + fire_intensity * csite%area(ipa)
   146)                else
   147)                   fire_intensity      = 0.0
   148)                end if
   149)                !---------------------------------------------------------------------------!
   150) 
   151)             case (2)
   152)                !---------------------------------------------------------------------------!
   153)                !     We now compute the minimum amount of water in kg/m2 that the soil     !
   154)                ! must have to avoid fires, using the soil properties and the soil moisture !
   155)                ! fraction threshold.                                                       !
   156)                !---------------------------------------------------------------------------!
   157)                fire_wmass_threshold = 0.
   158)                do k = k_fire_first, nzg
   159)                   nsoil                = cpoly%ntext_soil(k,isi)
   160)                   fire_wmass_threshold = fire_wmass_threshold                              &
   161)                                        + soil(nsoil)%soilfr * dslz(k) * wdns
   162)                end do
   163)                if (csite%avg_monthly_gndwater(ipa) < fire_wmass_threshold) then
   164)                   fire_intensity      = fire_parameter
   165)                   mean_fire_intensity = mean_fire_intensity                                &
   166)                                       + fire_intensity * csite%area(ipa)
   167)                else
   168)                   fire_intensity      = 0.0
   169)                end if
   170)                !---------------------------------------------------------------------------!
   171) 
   172)             case (3)
   173)                !---------------------------------------------------------------------------!
   174)                !     The threshold is independent on soil moisture.  We use climatological !
   175)                ! water deficit instead.                                                    !
   176)                !---------------------------------------------------------------------------!
   177)                if (csite%avg_monthly_waterdef(ipa) >= sm_fire * sum_pcpg) then
   178)                   fire_intensity      = fire_parameter
   179)                   mean_fire_intensity = mean_fire_intensity                                &
   180)                                       + fire_intensity * csite%area(ipa)
   181)                else
   182)                   fire_intensity      = 0.0
   183)                end if
   184)                !---------------------------------------------------------------------------!
   185)             end select
   186)             !------------------------------------------------------------------------------!
   187) 
   188) 
   189) 
   190)             !------------------------------------------------------------------------------!
   191)             !    If the soil is dry, then calculate patch contribution to the ignition     !
   192)             ! rate.                                                                        !
   193)             !------------------------------------------------------------------------------!
   194)             ignition_rate = ignition_rate + fire_intensity * fuel * csite%area(ipa)
   195)             !------------------------------------------------------------------------------!
   196) 
   197) 
   198)             !----- Reset the ground water for next month. ---------------------------------!
   199)             csite%avg_monthly_gndwater(ipa) = 0.
   200)             !------------------------------------------------------------------------------!
   201) 
   202)          end do patchloop
   203)          !---------------------------------------------------------------------------------!
   204) 
   205) 
   206) 
   207)          !----- Calculate fire disturbance rate [1/month]. --------------------------------!
   208)          cpoly%lambda_fire  (imon,isi) = ignition_rate
   209)          if (mean_fire_intensity > 0.) then
   210)             cpoly%ignition_rate (isi) = ignition_rate / mean_fire_intensity
   211)          else
   212)             cpoly%ignition_rate (isi) = 0.0
   213)          end if
   214)          !---------------------------------------------------------------------------------!
   215)       end do siteloop
   216)       !------------------------------------------------------------------------------------!
   217)    end do polyloop
   218)    !---------------------------------------------------------------------------------------!
   219) 
   220)    return
   221) end subroutine fire_frequency
