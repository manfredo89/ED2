   268) !     This subroutine is the double precision version of the linear system solver above.   !
   269) ! It will solve the linear system AA . X = Y for given AA and Y, using the Gaussian        !
   270) ! elimination method with partial pivoting and back-substitution.  This subroutine is      !
   271) ! based on:                                                                                !
   272) !                                                                                          !
   273) ! Press, W. H., S. A. Teukolsky, W. T. Vetterling, B. P. Flannery: 1992. Numerical recipes !
   274) !    in Fortran 77.  Cambridge University Press.                                           !
   275) !------------------------------------------------------------------------------------------!
   276) subroutine lisys_solver8(nsiz,AA,Y,X,sing)
   277)    implicit none
   278)    !----- Arguments. ----------------------------------------------------------------------!
   279)    integer                           , intent(in)  :: nsiz  ! matrix and vector size
   280)    real(kind=8), dimension(nsiz,nsiz), intent(in)  :: AA    ! matrix
   281)    real(kind=8), dimension(nsiz)     , intent(in)  :: Y     ! right-hand side vector
   282)    real(kind=8), dimension(nsiz)     , intent(out) :: X     ! unknown vector
   283)    logical                           , intent(out) :: sing  ! The matrix was singular [T|F]
   284)    !----- Local variables. ----------------------------------------------------------------!
   285)    real(kind=8), dimension(nsiz,nsiz)              :: EE     ! Copy of AA, for elimination.
   286)    real(kind=8), dimension(nsiz)                   :: Z      ! Copy of Y, for scaling
   287)    real(kind=8), dimension(nsiz)                   :: dumvec ! Dummy vector (row swapping)
   288)    real(kind=8)                                    :: pivot  ! The pivot
   289)    real(kind=8)                                    :: multip ! Multiplier
   290)    integer                                         :: r      ! Row index
   291)    integer                                         :: b      ! Row below index
   292)    integer                                         :: p      ! Pivot index
   293)    real(kind=8)                                    :: dumsca ! Dummy scalar (row swapping)
   294)    !----- Local parameters. ---------------------------------------------------------------!
   295)    real(kind=8)                      , parameter   :: tinyoff=1.d-20
   296)    !---------------------------------------------------------------------------------------!
   297)    
   298)    !----- First thing, we copy AA to EE and Y to Z. ---------------------------------------!
   299)    EE(:,:) = AA(:,:)
   300)    Z (:)   = Y (:)
   301)    dumvec  = 0.d0
   302)    dumsca  = 0.d0
   303)    !---------------------------------------------------------------------------------------!
   304)    !     We initialise X with a huge, non-sense value, which will become the answer when   !
   305)    ! the matrix is singular.                                                               !
   306)    !---------------------------------------------------------------------------------------!
   307)    X (:)   = -huge(1.d0)
   308)    !----- We first assume that everything will be fine. -----------------------------------!
   309)    sing    = .false.
   310) 
   311)    !---------------------------------------------------------------------------------------!
   312)    ! 1. Main elimination loop, done row by row.                                            !
   313)    !---------------------------------------------------------------------------------------!
   314)    elimloop: do r = 1, nsiz-1
   315)       !------ 1a. Finding the largest element, which will become our pivot ----------------!
   316)       p = (r-1) + maxloc(abs(EE(r:nsiz,r)),dim=1)
   317)       
   318)       pivot = maxval(abs(EE(r:nsiz,r)))
   319)       !------------------------------------------------------------------------------------!
   320)       ! 1b. Check the pivot and make sure it is a good one.  If not, then this matrix is   !
   321)       !     singular or almost singular, and we cannot solve it, so we switch the flag and !
   322)       !     return.                                                                        !
   323)       !------------------------------------------------------------------------------------!
   324)       if (pivot < tinyoff) then
   325)          sing = .true.
   326)          return
   327)       end if
   328)       
   329)       !----- 1c. If the best pivot is not the current row, we must swap them. -------------!
   330)       if (p /= r) then
   331)          dumvec(r:nsiz) = EE(r,r:nsiz)
   332)          dumsca         = Z(r)
   333)          EE(r,r:nsiz)   = EE(p,r:nsiz)
   334)          Z(r)           = Z(p)
   335)          EE(p,r:nsiz)   = dumvec(r:nsiz)
   336)          Z(p)           = dumsca
   337)       end if
   338) 
   339)       !------------------------------------------------------------------------------------!
   340)       ! 1d.  Eliminate rows below, everything to the left of the (,r) column will become   !
   341)       !      zero (we won't compute that, but they will be.).                              !
   342)       !------------------------------------------------------------------------------------!
   343)       belowloop: do b=r+1,nsiz
   344)          multip = EE(b,r)/EE(r,r)
   345)          EE(b,r:nsiz) = EE(b,r:nsiz) - multip * EE(r,r:nsiz)
   346)          Z(b)         = Z(b)         - multip * Z(r)
   347)       end do belowloop
   348)    end do elimloop
   349) 
   350)    !---------------------------------------------------------------------------------------!
   351)    ! 2. We may be unlucky and discover that the matrix is singular at the last line, so we !
   352)    !    check the last pivot too.                                                          ! 
   353)    !---------------------------------------------------------------------------------------!
   354)    if (abs(EE(nsiz,nsiz)) < tinyoff) then
   355)       sing = .true.
   356)       return
   357)    end if
   358) 
   359)    !---------------------------------------------------------------------------------------!
   360)    ! 3. We now perform the back-substitution, to find the solution.                        !
   361)    !---------------------------------------------------------------------------------------!
   362)    X(nsiz) = Z(nsiz) / EE(nsiz,nsiz)
   363)    backsubloop: do r=nsiz-1,1,-1
   364)       b    = r+1
   365)       X(r) = (Z(r) - sum(EE(r,b:nsiz)*x(b:nsiz))) / EE(r,r)
   366)    end do backsubloop
   367) 
   368)    return
   369) end subroutine lisys_solver8

   377) !     EIFUN8 -- This function computes the exponential integral function, defined by       !
   378) !                                                                                          !
   379) !                        x_                                                                !
   380) !                        |    exp(t)                                                       !
   381) !              Ei(x) =   |   -------- dt                                                   !
   382) !                       _|      t                                                          !
   383) !                        0                                                                 !
   384) !                                                                                          !
   385) !     This function is based on:                                                           !
   386) !                                                                                          !
   387) ! Press, W. H., S. A. Teukolsky, W. T. Vetterling, B. P. Flannery: 1992. Numerical recipes !
   388) !    in Fortran 77.  Cambridge University Press, section 6.3 p. 215-219.                   !
   389) !                                                                                          !
   390) ! with the difference that we also solve for negative numbers.  Zero cannot be solved, so  !
   391) ! if this happens, or if the sought number would lead to infinity, we stop the model.      !
   392) !------------------------------------------------------------------------------------------!
   393) real(kind=8) function eifun8(x)
   394)    use consts_coms, only : euler_gam8 & ! intent(in)
   395)                          , lnexp_min8 & ! intent(in)
   396)                          , lnexp_max8 & ! intent(in)
   397)                          , tiny_num8  ! ! intent(in)
   398)    implicit none
   399)    !----- Arguments. ----------------------------------------------------------------------!
   400)    real(kind=8), intent(in) :: x
   401)    !----- Local variables. ----------------------------------------------------------------!
   402)    real(kind=8)             :: sum
   403)    real(kind=8)             :: term
   404)    real(kind=8)             :: fact
   405)    real(kind=8)             :: prev
   406)    real(kind=8)             :: diter
   407)    integer                  :: iter
   408)    !----- Local constants. ----------------------------------------------------------------!
   409)    real(kind=8), parameter  :: powerlim  = 1.5d+01
   410)    real(kind=8), parameter  :: converge  = 1.0d-7
   411)    integer     , parameter  :: maxiter   = 100
   412)    !---------------------------------------------------------------------------------------!
   413) 
   414) 
   415)    !---------------------------------------------------------------------------------------!
   416)    !     Check what to do depending on the value of x.                                     !
   417)    !---------------------------------------------------------------------------------------!
   418)    if (x == 0.d0) then
   419)       !------------------------------------------------------------------------------------!
   420)       !     Zero.  This is a singularity and the user should never call it in this case.   !
   421)       ! That's sad, but we ought to quit this run and tell the user why the run crashed.   !
   422)       !------------------------------------------------------------------------------------!
   423)       call fatal_error('Exponential integral cannot be solved for x = 0.'                  &
   424)                       ,'eifun8','numutils.f90')
   425)    elseif (x >= lnexp_max8) then
   426)       !----- Huge value, crash because this is iminent over-flow. -------------------------!
   427)       write(unit=*,fmt='(a,1x,es12.5)') 'Attempted X =         ',x
   428)       write(unit=*,fmt='(a,1x,es12.5)') 'Maximum acceptable X =',lnexp_max8
   429)       call fatal_error('Exponential integral cannot be solved for x = 0.'                  &
   430)                       ,'eifun8','numutils.f90')
   431)    elseif (abs(x) <= lnexp_min8) then
   432)       !----- Huge negative number, the result can be rounded to zero. ---------------------!
   433)       eifun8 = 0.d0
   434)    elseif (abs(x) <= tiny_num8) then
   435)       !----- The number is too close to zero, bypass iterative methods. -------------------!
   436)       eifun8 = euler_gam8 + log(abs(x))
   437)    elseif (abs(x) <= powerlim) then
   438)       !------------------------------------------------------------------------------------!
   439)       !    Input x is small, so we use the power method.                                   !
   440)       !------------------------------------------------------------------------------------!
   441)       fact      = 1.d0
   442)       sum       = 0.d0
   443)       powerloop: do iter=1,maxiter
   444)          diter = dble(iter)
   445)          fact  = fact * x / diter
   446)          term  = fact / diter
   447)          sum   = sum + term
   448)          !----- If the term is tiny, we have reached convergence, quit the loop. ----------!
   449)          if (abs(term) < converge * abs(sum)) exit powerloop
   450)       end do powerloop
   451)       eifun8   = euler_gam8 + log(abs(x)) + sum
   452)    else
   453)       !------------------------------------------------------------------------------------!
   454)       !    Input x is large, so we use the asymptotic approximation.                       !
   455)       !------------------------------------------------------------------------------------!
   456)       sum       = 0.d0
   457)       term      = 1.d0
   458)       asymploop: do iter=1,maxiter
   459)          diter = dble(iter)
   460)          prev  = term
   461)          term  = term * diter / x
   462)          if (abs(term) < converge) then
   463)             !----- The term is tiny, we have reached convergence, quit the loop. ----------!
   464)             exit asymploop
   465)          elseif (abs(term) >= abs(prev)) then
   466)             !------------------------------------------------------------------------------!
   467)             !   Series is diverging, we are probably reaching round-off errors, we better  !
   468)             ! stop now.                                                                    !
   469)             !------------------------------------------------------------------------------!
   470)             sum = sum - prev
   471)             exit asymploop
   472)          else
   473)             sum = sum + term
   474)          end if
   475)       end do asymploop
   476)       eifun8 = exp(x) * (1.d0 + sum) / x
   477)    end if
   478) 
   479)    return
   480) end function eifun8

   486) !   Function to compute the great circle distance between two points: the s suffix denotes !
   487) ! source point, and f denotes the destination - "forepoint"). The results are given in     !
   488) ! metres. The formula is intended to be accurate for both small and large distances and    !
   489) ! uses double precision to avoid ill-conditioned behaviour of sin and cos for numbers      !
   490) ! close to the n*pi/2.                                                                     !
   491) !------------------------------------------------------------------------------------------!
   492) real function dist_gc(slons,slonf,slats,slatf)
   493)    use consts_coms, only : erad    & ! intent(in)
   494)                          , pio1808 ! ! intent(in)
   495)    implicit none
   496)    !----- Local variables. ----------------------------------------------------------------!
   497)    real, intent(in) :: slons
   498)    real, intent(in) :: slonf
   499)    real, intent(in) :: slats
   500)    real, intent(in) :: slatf
   501)    !----- Local variables. ----------------------------------------------------------------!
   502)    real(kind=8)     :: lons
   503)    real(kind=8)     :: lonf
   504)    real(kind=8)     :: lats
   505)    real(kind=8)     :: latf
   506)    real(kind=8)     :: dlon
   507)    real(kind=8)     :: dlat
   508)    real(kind=8)     :: x
   509)    real(kind=8)     :: y
   510)    !---------------------------------------------------------------------------------------!
   511) 
   512)    !----- Convert the co-ordinates to double precision and to radians. --------------------!
   513)    lons = dble(slons) * pio1808
   514)    lonf = dble(slonf) * pio1808
   515)    lats = dble(slats) * pio1808
   516)    latf = dble(slatf) * pio1808
   517)    dlon = lonf - lons
   518)    dlat = latf - lats
   519) 
   520)    !----- Find the arcs. ------------------------------------------------------------------!
   521)    x    = dsin(lats) * dsin(latf) + dcos(lats) * dcos(latf) * dcos(dlon)
   522)    y    = dsqrt( (dcos(latf)*dsin(dlon)) * (dcos(latf)*dsin(dlon))                         &
   523)                + (dcos(lats)*dsin(latf)-dsin(lats)*dcos(latf)*dcos(dlon))                  &
   524)                * (dcos(lats)*dsin(latf)-dsin(lats)*dcos(latf)*dcos(dlon)) )
   525) 
   526)    !----- Convert the arcs to actual distance. --------------------------------------------!
   527)    dist_gc = erad*sngl(datan2(y,x))
   528) 
   529)    return
   530) end function dist_gc
