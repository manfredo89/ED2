    17) !   This subroutine checks the grid definition for ED, like grid size, dimensions and if   !
    18) ! the provided numbers are allowed.                                                        !
    19) !------------------------------------------------------------------------------------------!
    20) subroutine ed_opspec_grid
    21)    use ed_max_dims , only : maxgrds             & ! intent(in)
    22)                           , nxpmax              & ! intent(in)
    23)                           , nypmax              & ! intent(in)
    24)                           , nzpmax              & ! intent(in)
    25)                           , nzgmax              & ! intent(in)
    26)                           , nzsmax              & ! intent(in)
    27)                           , max_poi             & ! intent(in)
    28)                           , max_ed_regions      ! ! intent(in)
    29)    use grid_coms   , only : nnxp                & ! intent(in)
    30)                           , nnyp                & ! intent(in)
    31)                           , ngrids              & ! intent(in)
    32)                           , polelat             & ! intent(in)
    33)                           , polelon             & ! intent(in)
    34)                           , centlat             & ! intent(in)
    35)                           , centlon             & ! intent(in)
    36)                           , deltax              & ! intent(in)
    37)                           , deltay              & ! intent(in)
    38)                           , nstratx             & ! intent(in)
    39)                           , nstraty             & ! intent(in)
    40)                           , nzg                 & ! intent(in)
    41)                           , nzs                 ! ! intent(in)
    42)    use mem_polygons, only : n_ed_region         & ! intent(in)
    43)                           , n_poi               & ! intent(in)
    44)                           , grid_type           & ! intent(in)
    45)                           , grid_res            & ! intent(in)
    46)                           , poi_lat             & ! intent(in)
    47)                           , poi_lon             & ! intent(in)
    48)                           , poi_res             & ! intent(in)
    49)                           , ed_reg_latmin       & ! intent(in)
    50)                           , ed_reg_latmax       & ! intent(in)
    51)                           , ed_reg_lonmin       & ! intent(in)
    52)                           , ed_reg_lonmax       ! ! intent(in)
    53)    use soil_coms   , only : slz                 ! ! intent(in)
    54) 
    55)    implicit none
    56)    !----- Local variables. ----------------------------------------------------------------!
    57)    integer            :: ifaterr
    58)    integer            :: ifm
    59)    integer            :: k
    60)    character(len=222) :: reason
    61)    !---------------------------------------------------------------------------------------!
    62) 
    63) 
    64) 
    65)    !----- Start assuming everything will be fine. -----------------------------------------!
    66)    ifaterr = 0
    67)    !---------------------------------------------------------------------------------------!
    68) 
    69) 
    70)    !----- Check whether the number of grids is reasonable. --------------------------------!
    71)    if (ngrids < 1) then
    72)       write(reason,'(a,1x,i4,a)') &
    73)            'Number of grids needs to be positive. (Yours is ',ngrids,').'
    74)       call opspec_fatal(reason,'opspec_grid')
    75)       ifaterr=ifaterr+1
    76)    elseif (ngrids > maxgrds) then
    77)       write(reason,'(2(a,1x,i4,a))')  &
    78)            'Number of grids cannot be larger than ',maxgrds,'.',' (Yours is ',ngrids,').'
    79)       call opspec_fatal(reason,'opspec_grid')  
    80)       ifaterr=ifaterr+1
    81)    end if
    82)    !---------------------------------------------------------------------------------------!
    83) 
    84) 
    85) 
    86)    !---------------------------------------------------------------------------------------!
    87)    !      Check whether we are trying to run POI and regional simultaneously.  This is     !
    88)    ! currently forbidden.                                                                  !
    89)    !---------------------------------------------------------------------------------------!
    90)    if ((.not. (n_poi > 0 .and. n_ed_region == 0)) .and.                                    &
    91)        (.not. (n_poi == 0 .and. n_ed_region > 0))       ) then
    92)       write(reason,'(a,1x,i4,1x,a,1x,i4,1x,a)')                                            &
    93)             'One of n_poi or n_ed_region needs to be zero.( Yours: n_poi='                 &
    94)            ,n_poi,', n_ed_region=',n_ed_region,').'
    95)       call opspec_fatal(reason,'opspec_grid')  
    96)       ifaterr=ifaterr+1
    97) 
    98)    elseif (n_poi < 0) then
    99)       write(reason,'(a,1x,i4,a)')                                                          &
   100)          'N_POI needs to be non-negative. Yours is currently set to ',n_poi,'.'
   101)       call opspec_fatal(reason,'opspec_grid')  
   102)       ifaterr=ifaterr+1
   103) 
   104)    elseif (n_ed_region < 0) then
   105)       write(reason,'(a,1x,i4,a)')                                                          &
   106)          'N_ED_REGION needs to be non-negative. Yours is currently set to ',n_ed_region,'.'
   107)       call opspec_fatal(reason,'opspec_grid')  
   108)       ifaterr=ifaterr+1
   109) 
   110)    elseif (n_poi > max_poi) then
   111)       write(reason,'(a,1x,i4,a,1x,i4,a)')                                                  &
   112)          'N_POI cannot be greater than',max_poi,'. Yours is currently set to ',n_poi,'.'
   113)       call opspec_fatal(reason,'opspec_grid')  
   114)       ifaterr=ifaterr+1
   115) 
   116)    elseif (n_ed_region> max_ed_regions) then
   117)       write(reason,'(a,1x,i4,a,1x,i4,a)')                                                  &
   118)           'N_ED_REGION cannot be greater than',max_ed_regions                              &
   119)          ,'. Yours is currently set to ',n_ed_region,'.'
   120)       call opspec_fatal(reason,'opspec_grid')  
   121)       ifaterr=ifaterr+1
   122)    end if
   123)    !---------------------------------------------------------------------------------------!
   124) 
   125) 
   126) 
   127) 
   128)    !----- Check whether the number of grid points makes sense for ED regions. -------------!
   129)    do ifm=1,n_ed_region
   130)       if (nnxp(ifm) > nxpmax) then
   131)          write(reason,'(3(a,1x,i4,a))')                                                    &
   132)               'Number of x-points cannot be greater than ',nxpmax,'.',' (Yours is'         &
   133)              ,nnxp(ifm),' ','in grid',ifm,').'
   134)          call opspec_fatal(reason,'opspec_grid')  
   135)          ifaterr=ifaterr+1
   136)       end if
   137)       if (nnyp(ifm) > nypmax) then
   138)          write(reason,'(3(a,1x,i4,a))')                                                    &
   139)               'Number of y-points cannot be greater than ',nypmax,'.',' (Yours is'         &
   140)              ,nnyp(ifm),' ','in grid',ifm,').'
   141)          call opspec_fatal(reason,'opspec_grid')  
   142)          ifaterr=ifaterr+1
   143)       end if
   144)       if (nstratx(ifm) < 1) then
   145)          write (reason,'(a,1x,i4,1x,a,1x,i4,a)')                                           &
   146)                'Nest x ratio should be at least 1. Your nstratx for grid',ifm              &
   147)               ,'is currently set to',nstratx(ifm),'.'
   148)          call opspec_fatal(reason,'opspec_grid')  
   149)          ifaterr=ifaterr+1        
   150)       end if
   151)       if (nstraty(ifm) < 1) then
   152)          write (reason,'(a,1x,i4,1x,a,1x,i4,a)')                                           &
   153)                'Nest y ratio should be at least 1. Your nstraty for grid',ifm              &
   154)               ,'is currently set to',nstratx(ifm),'.'
   155)          call opspec_fatal(reason,'opspec_grid')  
   156)          ifaterr=ifaterr+1        
   157)       end if
   158)    end do
   159)    !---------------------------------------------------------------------------------------!
   160) 
   161) 
   162) 
   163)    !-----  Check whether ED has reasonable values to define the grids. --------------------!
   164)    if (n_ed_region > 0) then
   165)       if (grid_type < 0 .or. grid_type > 1) then
   166)          write (reason,'(a,2x,a,1x,i4,a)')                                                 &
   167)                'Grid_type should be either 0 or 1 for regional runs.'                      &
   168)               ,'Yours is currently set to',grid_type,'.'
   169)          call opspec_fatal(reason,'opspec_grid')  
   170)          ifaterr=ifaterr+1
   171)       end if
   172)       if (grid_type == 0) then
   173)          do ifm=1,n_ed_region
   174)             if (ed_reg_latmin(ifm) < -90. .or. ed_reg_latmin(ifm) > 90. ) then
   175)                write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                    &
   176)                      'ED_REG_LATMIN is outside [-90.;90.] for region ',ifm                 &
   177)                     ,'. Yours is currently set to',ed_reg_latmin(ifm),'...'
   178)                call opspec_fatal(reason,'opspec_grid')  
   179)                ifaterr=ifaterr+1
   180)             end if
   181)             if (ed_reg_latmax(ifm) < -90. .or. ed_reg_latmax(ifm) > 90. ) then
   182)                write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                    &
   183)                      'ED_REG_LATMAX is outside [-90.;90.] for region ',ifm                 &
   184)                     ,'. Yours is currently set to',ed_reg_latmax(ifm),'...'
   185)                call opspec_fatal(reason,'opspec_grid')  
   186)                ifaterr=ifaterr+1
   187)             end if
   188)             if (ed_reg_lonmin(ifm) < -180. .or. ed_reg_lonmin(ifm) > 180. ) then
   189)                write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                    &
   190)                      'ED_REG_LONMIN is outside [-180.;180.] for region ',ifm               &
   191)                     ,'. Yours is currently set to',ed_reg_latmin(ifm),'...'
   192)                call opspec_fatal(reason,'opspec_grid')  
   193)                ifaterr=ifaterr+1
   194)             end if
   195)             if (ed_reg_lonmax(ifm) < -180. .or. ed_reg_lonmax(ifm) > 180. ) then
   196)                write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                    &
   197)                      'ED_REG_LONMAX is outside [-180.;180.] for region ',ifm               &
   198)                     ,'. Yours is currently set to',ed_reg_latmax(ifm),'...'
   199)                call opspec_fatal(reason,'opspec_grid')  
   200)                ifaterr=ifaterr+1
   201)             end if
   202)          end do
   203)          if (grid_res <= 0.) then
   204)             write (reason,'(a,1x,es14.7,a)')                                               &
   205)                   'GRID_RES must be positive. Yours is currently set to',grid_res,'...'
   206)             call opspec_fatal(reason,'opspec_grid')  
   207)             ifaterr=ifaterr+1
   208)          end if
   209)       elseif (grid_type == 1) then
   210)          if (deltax <= 0.) then
   211)             write (reason,'(a,1x,es14.7,a)')                                               &
   212)                   'DELTAX must be positive. Yours is currently set to',deltax,'...'
   213)             call opspec_fatal(reason,'opspec_grid')  
   214)             ifaterr=ifaterr+1
   215)          end if
   216)          if (deltay <= 0.) then
   217)             write (reason,'(a,1x,es14.7,a)')                                               &
   218)                   'DELTAY must be positive. Yours is currently set to',deltax,'...'
   219)             call opspec_fatal(reason,'opspec_grid')  
   220)             ifaterr=ifaterr+1
   221)          end if
   222)          if (polelat < -90. .or. polelat > 90.) then
   223)             write (reason,'(a,1x,i4,a)')                                                   &
   224)                   'POLELAT is outside [-90.;90.].',' Yours is currently set to'            &
   225)                  ,polelat,'...'
   226)             call opspec_fatal(reason,'opspec_grid')  
   227)             ifaterr=ifaterr+1
   228)          end if
   229)          if (polelon < -180. .or. polelon > 180.) then
   230)             write (reason,'(a,1x,i4,a)')                                                   &
   231)                   'POLELON is outside [-180.;180.].',' Yours is currently set to'          &
   232)                  ,polelon,'...'
   233)             call opspec_fatal(reason,'opspec_grid')  
   234)             ifaterr=ifaterr+1
   235)          end if
   236)          do ifm=1,n_ed_region
   237)             if (centlat(ifm) < -90. .or. centlat(ifm) > 90. ) then
   238)                write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                    &
   239)                      'CENTLAT is outside [-90.;90.] for region ',ifm                       &
   240)                     ,'. Yours is currently set to',centlat(ifm),'...'
   241)                call opspec_fatal(reason,'opspec_grid')  
   242)                ifaterr=ifaterr+1
   243)             end if
   244)             if (centlon(ifm) < -180. .or. centlon(ifm) > 180. ) then
   245)                write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                    &
   246)                      'CENTLON is outside [-180.;180.] for region ',ifm                     &
   247)                     ,'. Yours is currently set to',centlon(ifm),'...'
   248)                call opspec_fatal(reason,'opspec_grid')  
   249)                ifaterr=ifaterr+1
   250)             end if
   251)          end do
   252)       end if
   253)    end if
   254)    !---------------------------------------------------------------------------------------!
   255) 
   256) 
   257) 
   258) 
   259)    !----- Check whether the grid specifications for POI runs are good. --------------------!
   260)    if (n_poi > 0)then
   261)       do ifm=1,n_poi
   262)          if (poi_lat(ifm) < -90. .or. poi_lat(ifm) > 90. ) then
   263)             write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                       &
   264)                   'POI_LAT is outside [-90.;90.] for POI #',ifm                            &
   265)                  ,'. Yours is currently set to',poi_lat(ifm),'...'
   266)             call opspec_fatal(reason,'opspec_grid')  
   267)             ifaterr=ifaterr+1
   268)          end if
   269)          if (poi_lon(ifm) < -180. .or. poi_lon(ifm) > 180. ) then
   270)             write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                       &
   271)                   'POI_LON is outside [-180.;180.] for POI #',ifm                          &
   272)                   ,'. Yours is currently set to',poi_lon(ifm),'...'
   273)             call opspec_fatal(reason,'opspec_grid')  
   274)             ifaterr=ifaterr+1
   275)          end if
   276)          if (poi_res(ifm) < 0.001 .or. poi_res(ifm) > 5.) then
   277)             write (reason,'(a,1x,i4,a,1x,es14.7,a)')                                       &
   278)                   'POI_RES is outside [0.001;5.] range for POI #'                          &
   279)                  ,ifm,'. Yours is currently set to',poi_res(ifm),'...'
   280)             call opspec_fatal(reason,'opspec_grid')  
   281)             ifaterr=ifaterr+1
   282)          end if
   283)       end do
   284) 
   285)    end if
   286) 
   287)    !---------------------------------------------------------------------------------------!
   288)    !      Check whether ED soil layers are reasonable, i.e, enough layers, sorted from the !
   289)    ! deepest to the shallowest.                                                            !
   290)    !---------------------------------------------------------------------------------------!
   291)    if (nzg < 2) then
   292)       write (reason,'(a,1x,i4,a)')                                                         &
   293)             'Too few soil layers.  Set it to at least 2. Your nzg is currently set to'     &
   294)            ,nzg,'...'
   295)       call opspec_fatal(reason,'opspec_grid')  
   296)       ifaterr=ifaterr+1        
   297)    elseif (nzg > nzgmax) then 
   298)       write (reason,'(2(a,1x,i5,a))')                                                      &
   299)             'The number of soil layers cannot be greater than ',nzgmax,'.'                 &
   300)            ,' Your nzg is currently set to',nzg,'.'
   301)       call opspec_fatal(reason,'opspec_grid') 
   302)       ifaterr=ifaterr+1 
   303)    end if
   304)    do k=1,nzg
   305)       if (slz(k) > -.001) then
   306)          write (reason,'(a,1x,i4,1x,a,1x,es14.7,a)')                                       &
   307)                'Your soil level #',k,'is not enough below ground. It is currently set to'  &
   308)                ,slz(k),', make it deeper than -0.001...'
   309)          call opspec_fatal(reason,'opspec_grid')  
   310)          ifaterr=ifaterr+1        
   311)       end if
   312)    end do
   313) 
   314)    do k=1,nzg-1
   315)       if (slz(k)-slz(k+1) > .001) then
   316)          write (reason,'(2(a,1x,i4,1x),a,2x,a,1x,es14.7,1x,a,1x,es14.7,a)')                &
   317)                'Soil layers #',k,'and',k+1,'are not enough apart (i.e. > 0.001).'          &
   318)               ,'They are currently set as ',slz(k),'and',slz(k+1),'...'
   319)          call opspec_fatal(reason,'opspec_grid')  
   320)          ifaterr=ifaterr+1        
   321)       end if
   322)    end do
   323) 
   324) 
   325)    !---------------------------------------------------------------------------------------!
   326)    !     Check whether ED snow layers are well set, i.e., the number of soil levels is     !
   327)    ! within the allowed range.                                                             !
   328)    !---------------------------------------------------------------------------------------!
   329)    if (nzs < 1) then
   330)       write (reason,'(a,2x,a,1x,i4,a)')                                                    &
   331)             'Too few maximum # of snow layers. Set it to at least 1.'                      &
   332)            ,'Your nzs is currently set to',nzs,'.'
   333)       call opspec_fatal(reason,'opspec_grid')  
   334)       ifaterr=ifaterr+1        
   335)    elseif (nzs > nzsmax) then 
   336)       write (reason,'(2(a,1x,i5,a))')                                                      &
   337)             'The number of snow layers cannot be greater than ',nzsmax,'.'                 &
   338)            ,' Your nzs is currently set to',nzs,'.'
   339)       call opspec_fatal(reason,'opspec_grid') 
   340)       ifaterr=ifaterr+1 
   341)    end if
   342)    !---------------------------------------------------------------------------------------!
   343) 
   344) 
   345)    !----- Stop the run in case there is any fatal error. ----------------------------------!
   346)    if (ifaterr > 0) then
   347)       write (unit=*,fmt='(a)')       ' -----------ED_OPSPEC_GRID -------------------------'
   348)       write (unit=*,fmt='(a,1x,i5)') ' Fatal errors:',ifaterr
   349)       write (unit=*,fmt='(a)')       ' ---------------------------------------------------'
   350)       call fatal_error('Fatal errors at namelist - Grid settings '                         &
   351)                       ,'ed_opspec_grid','ed_opspec.f90')
   352)    end if
   353)    !---------------------------------------------------------------------------------------!
   354)    return
   355) end subroutine ed_opspec_grid

   366) subroutine ed_opspec_par
   367) !------------------------------------------------------------------------------------------!
   368) !   This subroutine checks the grid definition for ED, like grid size, dimensions and if   !
   369) ! the provided numbers are allowed.                                                        !
   370) !------------------------------------------------------------------------------------------!
   371)    use ed_max_dims  , only : maxmach
   372)    use grid_coms    , only : nnxp,nnyp,ngrids
   373)    use mem_polygons , only : n_ed_region,n_poi
   374)    use ed_para_coms , only : nmachs
   375)    implicit none
   376)    integer :: ifaterr,ifm
   377)    character(len=222)           :: reason
   378)    ifaterr=0
   379) 
   380)    ! Check whether the number of slaves is more than what is allowed.
   381)    if (nmachs > maxmach-1) then
   382)       write(reason,'(2(a,1x,i4,a))')  &
   383)            'Number of nodes cannot be larger than ',maxmach,'.',' (Yours is ',nmachs+1,').'
   384)       call opspec_fatal(reason,'opspec_par')  
   385)       ifaterr=ifaterr+1
   386)    end if
   387)    do ifm=1,ngrids
   388)       if (n_ed_region > 0 .and. nnxp(ifm)*nnyp(ifm) < (nmachs +1)) then
   389)          write (reason,'(3(a,1x,i5,1x),a)')                                                 &
   390)                'Region number',ifm,'is too small. You have only',nnxp(ifm)*nnyp(ifm)        &
   391)               ,'points for',nmachs+1,'nodes.'
   392)          call opspec_fatal(reason,'opspec_par')  
   393)          ifaterr=ifaterr+1
   394)       end if
   395)    end do
   396)    ! Checking if the user is trying to run POI in parallel 
   397)    ! (this will be allowed at some point).
   398)    if (n_poi > 0 .and. nmachs > 0) then
   399)       write (reason,'(3(a,1x,i5,1x),a)')                                                    &
   400)         'You are attempting to run POI runs in parallel, this is not supported currently...'
   401)       call opspec_fatal(reason,'opspec_par')  
   402)       ifaterr=ifaterr+1
   403)    end if
   404) 
   405)    ! stop the run if there are any fatal errors.
   406)    if (ifaterr > 0) then
   407)       write (unit=*,fmt='(a)')       ' -----------ED_OPSPEC_GRID --------------------------'
   408)       write (unit=*,fmt='(a,1x,i5)') ' Fatal errors:',ifaterr
   409)       write (unit=*,fmt='(a)')       ' ----------------------------------------------------'
   410)       call fatal_error('Fatal errors at namelist - Grid settings '&
   411)                      & ,'ed_opspec_grid','ed_opspec.f90')
   412)    end if
   413) 
   414)    return
   415) end subroutine ed_opspec_par
