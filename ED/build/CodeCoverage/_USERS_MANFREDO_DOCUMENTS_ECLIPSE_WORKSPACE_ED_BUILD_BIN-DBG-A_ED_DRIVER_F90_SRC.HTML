   308) !     This sub-routine finds which frequency the model should use to normalise averaged    !
   309) ! variables.  FRQSUM should never exceed one day to avoid build up and overflows.          !
   310) !------------------------------------------------------------------------------------------!
   311) subroutine find_frqsum()
   312)    use ed_misc_coms, only : unitfast   & ! intent(in)
   313)                           , unitstate  & ! intent(in)
   314)                           , isoutput   & ! intent(in)
   315)                           , ifoutput   & ! intent(in)
   316)                           , itoutput   & ! intent(in)
   317)                           , imoutput   & ! intent(in)
   318)                           , idoutput   & ! intent(in)
   319)                           , iqoutput   & ! intent(in)
   320)                           , frqstate   & ! intent(in)
   321)                           , frqfast    & ! intent(in)
   322)                           , frqsum     ! ! intent(out)
   323)    use consts_coms, only: day_sec
   324) 
   325)    implicit none 
   326)    if (ifoutput == 0 .and. isoutput == 0 .and. idoutput == 0 .and. imoutput == 0 .and.     &
   327)        iqoutput == 0 .and. itoutput == 0 ) then
   328)       write(unit=*,fmt='(a)') '---------------------------------------------------------'
   329)       write(unit=*,fmt='(a)') '  WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! '
   330)       write(unit=*,fmt='(a)') '  WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! '
   331)       write(unit=*,fmt='(a)') '  WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! '
   332)       write(unit=*,fmt='(a)') '  WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! '
   333)       write(unit=*,fmt='(a)') '  WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! '
   334)       write(unit=*,fmt='(a)') '  WARNING! WARNING! WARNING! WARNING! WARNING! WARNING! '
   335)       write(unit=*,fmt='(a)') '---------------------------------------------------------'
   336)       write(unit=*,fmt='(a)') ' You are running a simulation that will have no output...'
   337)       frqsum=day_sec ! This avoids the number to get incredibly large.
   338) 
   339)    !---------------------------------------------------------------------------------------!
   340)    !    Mean diurnal cycle is on.  Frqfast will be in seconds, so it is likely to be the   !
   341)    ! smallest.  The only exception is if frqstate is more frequent thant frqfast, so we    !
   342)    ! just need to check that too.                                                          !
   343)    !---------------------------------------------------------------------------------------!
   344)    elseif (iqoutput > 0) then
   345)       if (unitstate == 0) then
   346)          frqsum = min(min(frqstate,frqfast),day_sec)
   347)       else
   348)          frqsum = min(frqfast,day_sec)
   349)       end if
   350) 
   351)    !---------------------------------------------------------------------------------------!
   352)    !     Either no instantaneous output was requested, or the user is outputting it at     !
   353)    ! monthly or yearly scale, force it to be one day.                                      !
   354)    !---------------------------------------------------------------------------------------!
   355)    elseif ((isoutput == 0  .and. (ifoutput == 0 .and. itoutput == 0)) .or.                 &
   356)            ((ifoutput == 0 .and. itoutput == 0) .and.                                      &
   357)              isoutput  > 0 .and. unitstate > 1) .or.                                       &
   358)            (isoutput == 0 .and.                                                            &
   359)             (ifoutput > 0 .or. itoutput > 0) .and. unitfast  > 1) .or.                     &
   360)            ((ifoutput  > 0 .or. itoutput > 0) .and.                                        &
   361)              isoutput  > 0 .and. unitstate > 1 .and. unitfast > 1)                         &
   362)           ) then
   363)       frqsum=day_sec
   364)    !---------------------------------------------------------------------------------------!
   365)    !    Only restarts, and the unit is in seconds, test which frqsum to use.               !
   366)    !---------------------------------------------------------------------------------------!
   367)    elseif (ifoutput == 0 .and. itoutput == 0 .and. isoutput > 0) then
   368)       frqsum=min(frqstate,day_sec)
   369)    !---------------------------------------------------------------------------------------!
   370)    !    Only fast analysis, and the unit is in seconds, test which frqsum to use.          !
   371)    !---------------------------------------------------------------------------------------!
   372)    elseif (isoutput == 0 .and. (ifoutput > 0 .or. itoutput > 0)) then
   373)       frqsum=min(frqfast,day_sec)
   374)    !---------------------------------------------------------------------------------------!
   375)    !    Both are on and both outputs are in seconds or day scales. Choose the minimum      !
   376)    ! between them and one day.                                                             !
   377)    !---------------------------------------------------------------------------------------!
   378)    elseif (unitfast < 2 .and. unitstate < 2) then 
   379)       frqsum=min(min(frqstate,frqfast),day_sec)
   380)    !---------------------------------------------------------------------------------------!
   381)    !    Both are on but unitstate is in month or years. Choose the minimum between frqfast !
   382)    ! and one day.                                                                          !
   383)    !---------------------------------------------------------------------------------------!
   384)    elseif (unitfast < 2) then 
   385)       frqsum=min(frqfast,day_sec)
   386)    !---------------------------------------------------------------------------------------!
   387)    !    Both are on but unitfast is in month or years. Choose the minimum between frqstate !
   388)    ! and one day.                                                                          !
   389)    !---------------------------------------------------------------------------------------!
   390)    else
   391)       frqsum=min(frqstate,day_sec)
   392)    end if
   393) 
   394)    return
   395) end subroutine find_frqsum

   406) !    This sub-routine eliminates all patches except the one you want to save...  This      !
   407) ! shouldn't be used unless you are debugging the code.                                     !
   408) !------------------------------------------------------------------------------------------!
   409) subroutine exterminate_patches_except(keeppa)
   410)    use ed_state_vars  , only : edgrid_g           & ! structure
   411)                              , edtype             & ! structure
   412)                              , polygontype        & ! structure
   413)                              , sitetype           & ! structure
   414)                              , patchtype          ! ! structure
   415)    use grid_coms      , only : ngrids             ! ! intent(in)
   416)    use fuse_fiss_utils, only : terminate_patches  ! ! sub-routine
   417)    !----- Arguments -----------------------------------------------------------------------!
   418)    integer                        , intent(in)  :: keeppa
   419)    !----- Local variables -----------------------------------------------------------------!
   420)    type(edtype)                   , pointer     :: cgrid
   421)    type(polygontype)              , pointer     :: cpoly
   422)    type(sitetype)                 , pointer     :: csite
   423)    type(patchtype)                , pointer     :: cpatch
   424)    integer                                      :: ifm
   425)    integer                                      :: ipy
   426)    integer                                      :: isi
   427)    integer                                      :: ipa
   428)    integer                                      :: keepact
   429)    real             , dimension(:), allocatable :: csite_lai
   430)    !---------------------------------------------------------------------------------------!
   431) 
   432) 
   433)    gridloop: do ifm=1,ngrids
   434)       cgrid => edgrid_g(ifm)
   435) 
   436)       polyloop: do ipy=1,cgrid%npolygons
   437)          cpoly => cgrid%polygon(ipy)
   438) 
   439)          siteloop: do isi=1,cpoly%nsites
   440)             csite => cpoly%site(isi)
   441) 
   442)             select case(keeppa)
   443)             case (0)
   444)                return
   445)             case (-2)
   446)                !----- Keep the one with the lowest LAI. -----------------------------------!
   447)                allocate(csite_lai(csite%npatches))
   448)                csite_lai(:) = 0.0
   449)                keepm2loop: do ipa=1,csite%npatches
   450)                   cpatch => csite%patch(ipa)
   451)                   if (cpatch%ncohorts > 0) csite_lai(ipa) = sum(cpatch%lai)
   452)                end do keepm2loop
   453)                keepact = minloc(csite_lai,dim=1)
   454)                deallocate(csite_lai)
   455)                !---------------------------------------------------------------------------!
   456)             case (-1)
   457)                !----- Keep the one with the lowest LAI. -----------------------------------!
   458)                allocate(csite_lai(csite%npatches))
   459)                csite_lai(:) = 0.0
   460)                keepm1loop: do ipa=1,csite%npatches
   461)                   cpatch => csite%patch(ipa)
   462)                   if (cpatch%ncohorts > 0) csite_lai(ipa) = sum(cpatch%lai)
   463)                end do keepm1loop
   464)                keepact = maxloc(csite_lai,dim=1)
   465)                deallocate(csite_lai)
   466)                !---------------------------------------------------------------------------!
   467)             case default
   468)                !----- Keep a fixed patch number. ------------------------------------------!
   469)                keepact = keeppa
   470)                
   471)                if (keepact > csite%npatches) then
   472)                   write(unit=*,fmt='(a)')       '-----------------------------------------'
   473)                   write(unit=*,fmt='(a,1x,i6)') ' - IPY      = ',ipy
   474)                   write(unit=*,fmt='(a,1x,i6)') ' - ISI      = ',isi
   475)                   write(unit=*,fmt='(a,1x,i6)') ' - NPATCHES = ',csite%npatches
   476)                   write(unit=*,fmt='(a,1x,i6)') ' - KEEPPA   = ',keeppa
   477)                   write(unit=*,fmt='(a)')       '-----------------------------------------'
   478)                   call fail_whale ()
   479)                   call fatal_error('KEEPPA can''t be greater than NPATCHES'                &
   480)                                   ,'exterminate_patches_except','ed_driver.f90')
   481)                end if
   482)             end select
   483) 
   484)             patchloop: do ipa=1,csite%npatches
   485)                if (ipa == keepact) then
   486)                   csite%area(ipa) = 1.0
   487)                else
   488)                   csite%area(ipa) = 0.0
   489)                end if
   490)             end do patchloop
   491) 
   492)             call terminate_patches(csite)
   493) 
   494)          end do siteloop
   495)       end do polyloop
   496)    end do gridloop
   497) 
   498)    return
   499) end subroutine exterminate_patches_except
